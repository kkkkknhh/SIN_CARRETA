41   def _is_int_type(x):
42       """ 
43       Check if input is of a scalar integer type (so ``5`` and ``array(5)`` will 
44       pass, while ``5.0`` and ``array([5])`` will fail. 
45       """
46       if np.ndim(x) != 0:
47           # Older versions of NumPy did not raise for np.array([1]).__index__()
48           # This is safe to remove when support for those versions is dropped
49           return False
50       try:
51           operator.index(x)
52       except TypeError:
53           return False
54       else:
55           return True

58   def findfreqs(num, den, N, kind='ba'):
59       """ 
60       Find array of frequencies for computing the response of an analog filter. 
61    
62       Parameters 
63       ---------- 
64       num, den : array_like, 1-D 
65           The polynomial coefficients of the numerator and denominator of the 
66           transfer function of the filter or LTI system, where the coefficients 
67           are ordered from highest to lowest degree. Or, the roots  of the 
68           transfer function numerator and denominator (i.e., zeroes and poles). 
69       N : int 
70           The length of the array to be computed. 
71       kind : str {'ba', 'zp'}, optional 
72           Specifies whether the numerator and denominator are specified by their 
73           polynomial coefficients ('ba'), or their roots ('zp'). 
74    
75       Returns 
76       ------- 
77       w : (N,) ndarray 
78           A 1-D array of frequencies, logarithmically spaced. 
79    
80       Examples 
81       -------- 
82       Find a set of nine frequencies that span the "interesting part" of the 
83       frequency response for the filter with the transfer function 
84    
85           H(s) = s / (s^2 + 8s + 25) 
86    
87       >>> from scipy import signal 
88       >>> signal.findfreqs([1, 0], [1, 8, 25], N=9) 
89       array([  1.00000000e-02,   3.16227766e-02,   1.00000000e-01, 
90                3.16227766e-01,   1.00000000e+00,   3.16227766e+00, 
91                1.00000000e+01,   3.16227766e+01,   1.00000000e+02]) 
92       """
93       if kind == 'ba':
94           ep = atleast_1d(roots(den)) + 0j
95           tz = atleast_1d(roots(num)) + 0j
96       elif kind == 'zp':
97           ep = atleast_1d(den) + 0j
98           tz = atleast_1d(num) + 0j
99       else:
100          raise ValueError("input must be one of {'ba', 'zp'}")
101  
102      if len(ep) == 0:
103          ep = atleast_1d(-1000) + 0j
104  
105      ez = np.r_[ep[ep.imag >= 0], tz[(np.abs(tz) < 1e5) & (tz.imag >= 0)]]
106  
107      integ = np.abs(ez) < 1e-10
108      hfreq = np.round(np.log10(np.max(3 * np.abs(ez.real + integ) +
109                                       1.5 * ez.imag)) + 0.5)
110      lfreq = np.round(np.log10(0.1 * np.min(np.abs((ez + integ).real) +
111                                             2 * ez.imag)) - 0.5)
112  
113      w = np.logspace(lfreq, hfreq, N)
114      return w

117  def freqs(b, a, worN=200, plot=None):
118      """ 
119      Compute frequency response of analog filter. 
120   
121      Given the M-order numerator `b` and N-order denominator `a` of an analog 
122      filter, compute its frequency response:: 
123   
124               b[0]*(jw)**M + b[1]*(jw)**(M-1) + ... + b[M] 
125       H(w) = ---------------------------------------------- 
126               a[0]*(jw)**N + a[1]*(jw)**(N-1) + ... + a[N] 
127   
128      Parameters 
129      ---------- 
130      b : array_like 
131          Numerator of a linear filter. 
132      a : array_like 
133          Denominator of a linear filter. 
134      worN : {None, int, array_like}, optional 
135          If None, then compute at 200 frequencies around the interesting parts 
136          of the response curve (determined by pole-zero locations). If a single 
137          integer, then compute at that many frequencies. Otherwise, compute the 
138          response at the angular frequencies (e.g., rad/s) given in `worN`. 
139      plot : callable, optional 
140          A callable that takes two arguments. If given, the return parameters 
141          `w` and `h` are passed to plot. Useful for plotting the frequency 
142          response inside `freqs`. 
143   
144      Returns 
145      ------- 
146      w : ndarray 
147          The angular frequencies at which `h` was computed. 
148      h : ndarray 
149          The frequency response. 
150   
151      See Also 
152      -------- 
153      freqz : Compute the frequency response of a digital filter. 
154   
155      Notes 
156      ----- 
157      Using Matplotlib's "plot" function as the callable for `plot` produces 
158      unexpected results, this plots the real part of the complex transfer 
159      function, not the magnitude. Try ``lambda w, h: plot(w, abs(h))``. 
160   
161      Examples 
162      -------- 
163      >>> from scipy.signal import freqs, iirfilter 
164      >>> import numpy as np 
165   
166      >>> b, a = iirfilter(4, [1, 10], 1, 60, analog=True, ftype='cheby1') 
167   
168      >>> w, h = freqs(b, a, worN=np.logspace(-1, 2, 1000)) 
169   
170      >>> import matplotlib.pyplot as plt 
171      >>> plt.semilogx(w, 20 * np.log10(abs(h))) 
172      >>> plt.xlabel('Frequency [rad/s]') 
173      >>> plt.ylabel('Amplitude response [dB]') 
174      >>> plt.grid(True) 
175      >>> plt.show() 
176   
177      """
178      if worN is None:
179          # For backwards compatibility
180          w = findfreqs(b, a, 200)
181      elif _is_int_type(worN):
182          w = findfreqs(b, a, worN)
183      else:
184          w = atleast_1d(worN)
185  
186      s = 1j * w
187      h = polyval(b, s) / polyval(a, s)
188      if plot is not None:
189          plot(w, h)
190  
191      return w, h

194  def freqs_zpk(z, p, k, worN=200):
195      """ 
196      Compute frequency response of analog filter. 
197   
198      Given the zeros `z`, poles `p`, and gain `k` of a filter, compute its 
199      frequency response:: 
200   
201                  (jw-z[0]) * (jw-z[1]) * ... * (jw-z[-1]) 
202       H(w) = k * ---------------------------------------- 
203                  (jw-p[0]) * (jw-p[1]) * ... * (jw-p[-1]) 
204   
205      Parameters 
206      ---------- 
207      z : array_like 
208          Zeroes of a linear filter 
209      p : array_like 
210          Poles of a linear filter 
211      k : scalar 
212          Gain of a linear filter 
213      worN : {None, int, array_like}, optional 
214          If None, then compute at 200 frequencies around the interesting parts 
215          of the response curve (determined by pole-zero locations). If a single 
216          integer, then compute at that many frequencies. Otherwise, compute the 
217          response at the angular frequencies (e.g., rad/s) given in `worN`. 
218   
219      Returns 
220      ------- 
221      w : ndarray 
222          The angular frequencies at which `h` was computed. 
223      h : ndarray 
224          The frequency response. 
225   
226      See Also 
227      -------- 
228      freqs : Compute the frequency response of an analog filter in TF form 
229      freqz : Compute the frequency response of a digital filter in TF form 
230      freqz_zpk : Compute the frequency response of a digital filter in ZPK form 
231   
232      Notes 
233      ----- 
234      .. versionadded:: 0.19.0 
235   
236      Examples 
237      -------- 
238      >>> import numpy as np 
239      >>> from scipy.signal import freqs_zpk, iirfilter 
240   
241      >>> z, p, k = iirfilter(4, [1, 10], 1, 60, analog=True, ftype='cheby1', 
242      ...                     output='zpk') 
243   
244      >>> w, h = freqs_zpk(z, p, k, worN=np.logspace(-1, 2, 1000)) 
245   
246      >>> import matplotlib.pyplot as plt 
247      >>> plt.semilogx(w, 20 * np.log10(abs(h))) 
248      >>> plt.xlabel('Frequency [rad/s]') 
249      >>> plt.ylabel('Amplitude response [dB]') 
250      >>> plt.grid(True) 
251      >>> plt.show() 
252   
253      """
254      k = np.asarray(k)
255      if k.size > 1:
256          raise ValueError('k must be a single scalar gain')
257  
258      if worN is None:
259          # For backwards compatibility
260          w = findfreqs(z, p, 200, kind='zp')
261      elif _is_int_type(worN):
262          w = findfreqs(z, p, worN, kind='zp')
263      else:
264          w = worN
265  
266      w = atleast_1d(w)
267      s = 1j * w
268      num = polyvalfromroots(s, z)
269      den = polyvalfromroots(s, p)
270      h = k * num/den
271      return w, h

274  def freqz(b, a=1, worN=512, whole=False, plot=None, fs=2*pi,
275            include_nyquist=False):
276      """ 
277      Compute the frequency response of a digital filter. 
278   
279      Given the M-order numerator `b` and N-order denominator `a` of a digital 
280      filter, compute its frequency response:: 
281   
282                   jw                 -jw              -jwM 
283          jw    B(e  )    b[0] + b[1]e    + ... + b[M]e 
284       H(e  ) = ------ = ----------------------------------- 
285                   jw                 -jw              -jwN 
286                A(e  )    a[0] + a[1]e    + ... + a[N]e 
287   
288      Parameters 
289      ---------- 
290      b : array_like 
291          Numerator of a linear filter. If `b` has dimension greater than 1, 
292          it is assumed that the coefficients are stored in the first dimension, 
293          and ``b.shape[1:]``, ``a.shape[1:]``, and the shape of the frequencies 
294          array must be compatible for broadcasting. 
295      a : array_like 
296          Denominator of a linear filter. If `b` has dimension greater than 1, 
297          it is assumed that the coefficients are stored in the first dimension, 
298          and ``b.shape[1:]``, ``a.shape[1:]``, and the shape of the frequencies 
299          array must be compatible for broadcasting. 
300      worN : {None, int, array_like}, optional 
301          If a single integer, then compute at that many frequencies (default is 
302          N=512). This is a convenient alternative to:: 
303   
304              np.linspace(0, fs if whole else fs/2, N, endpoint=include_nyquist) 
305   
306          Using a number that is fast for FFT computations can result in 
307          faster computations (see Notes). 
308   
309          If an array_like, compute the response at the frequencies given. 
310          These are in the same units as `fs`. 
311      whole : bool, optional 
312          Normally, frequencies are computed from 0 to the Nyquist frequency, 
313          fs/2 (upper-half of unit-circle). If `whole` is True, compute 
314          frequencies from 0 to fs. Ignored if worN is array_like. 
315      plot : callable 
316          A callable that takes two arguments. If given, the return parameters 
317          `w` and `h` are passed to plot. Useful for plotting the frequency 
318          response inside `freqz`. 
319      fs : float, optional 
320          The sampling frequency of the digital system. Defaults to 2*pi 
321          radians/sample (so w is from 0 to pi). 
322   
323          .. versionadded:: 1.2.0 
324      include_nyquist : bool, optional 
325          If `whole` is False and `worN` is an integer, setting `include_nyquist` 
326          to True will include the last frequency (Nyquist frequency) and is 
327          otherwise ignored. 
328   
329          .. versionadded:: 1.5.0 
330   
331      Returns 
332      ------- 
333      w : ndarray 
334          The frequencies at which `h` was computed, in the same units as `fs`. 
335          By default, `w` is normalized to the range [0, pi) (radians/sample). 
336      h : ndarray 
337          The frequency response, as complex numbers. 
338   
339      See Also 
340      -------- 
341      freqz_zpk 
342      freqz_sos 
343   
344      Notes 
345      ----- 
346      Using Matplotlib's :func:`matplotlib.pyplot.plot` function as the callable 
347      for `plot` produces unexpected results, as this plots the real part of the 
348      complex transfer function, not the magnitude. 
349      Try ``lambda w, h: plot(w, np.abs(h))``. 
350   
351      A direct computation via (R)FFT is used to compute the frequency response 
352      when the following conditions are met: 
353   
354      1. An integer value is given for `worN`. 
355      2. `worN` is fast to compute via FFT (i.e., 
356         `next_fast_len(worN) <scipy.fft.next_fast_len>` equals `worN`). 
357      3. The denominator coefficients are a single value (``a.shape[0] == 1``). 
358      4. `worN` is at least as long as the numerator coefficients 
359         (``worN >= b.shape[0]``). 
360      5. If ``b.ndim > 1``, then ``b.shape[-1] == 1``. 
361   
362      For long FIR filters, the FFT approach can have lower error and be much 
363      faster than the equivalent direct polynomial calculation. 
364   
365      Examples 
366      -------- 
367      >>> from scipy import signal 
368      >>> import numpy as np 
369      >>> taps, f_c = 80, 1.0  # number of taps and cut-off frequency 
370      >>> b = signal.firwin(taps, f_c, window=('kaiser', 8), fs=2*np.pi) 
371      >>> w, h = signal.freqz(b) 
372   
373      >>> import matplotlib.pyplot as plt 
374      >>> fig, ax1 = plt.subplots(tight_layout=True) 
375      >>> ax1.set_title(f"Frequency Response of {taps} tap FIR Filter" + 
376      ...               f"($f_c={f_c}$ rad/sample)") 
377      >>> ax1.axvline(f_c, color='black', linestyle=':', linewidth=0.8) 
378      >>> ax1.plot(w, 20 * np.log10(abs(h)), 'C0') 
379      >>> ax1.set_ylabel("Amplitude in dB", color='C0') 
380      >>> ax1.set(xlabel="Frequency in rad/sample", xlim=(0, np.pi)) 
381   
382      >>> ax2 = ax1.twinx() 
383      >>> phase = np.unwrap(np.angle(h)) 
384      >>> ax2.plot(w, phase, 'C1') 
385      >>> ax2.set_ylabel('Phase [rad]', color='C1') 
386      >>> ax2.grid(True) 
387      >>> ax2.axis('tight') 
388      >>> plt.show() 
389   
390      Broadcasting Examples 
391   
392      Suppose we have two FIR filters whose coefficients are stored in the 
393      rows of an array with shape (2, 25). For this demonstration, we'll 
394      use random data: 
395   
396      >>> rng = np.random.default_rng() 
397      >>> b = rng.random((2, 25)) 
398   
399      To compute the frequency response for these two filters with one call 
400      to `freqz`, we must pass in ``b.T``, because `freqz` expects the first 
401      axis to hold the coefficients. We must then extend the shape with a 
402      trivial dimension of length 1 to allow broadcasting with the array 
403      of frequencies.  That is, we pass in ``b.T[..., np.newaxis]``, which has 
404      shape (25, 2, 1): 
405   
406      >>> w, h = signal.freqz(b.T[..., np.newaxis], worN=1024) 
407      >>> w.shape 
408      (1024,) 
409      >>> h.shape 
410      (2, 1024) 
411   
412      Now, suppose we have two transfer functions, with the same numerator 
413      coefficients ``b = [0.5, 0.5]``. The coefficients for the two denominators 
414      are stored in the first dimension of the 2-D array  `a`:: 
415   
416          a = [   1      1  ] 
417              [ -0.25, -0.5 ] 
418   
419      >>> b = np.array([0.5, 0.5]) 
420      >>> a = np.array([[1, 1], [-0.25, -0.5]]) 
421   
422      Only `a` is more than 1-D. To make it compatible for 
423      broadcasting with the frequencies, we extend it with a trivial dimension 
424      in the call to `freqz`: 
425   
426      >>> w, h = signal.freqz(b, a[..., np.newaxis], worN=1024) 
427      >>> w.shape 
428      (1024,) 
429      >>> h.shape 
430      (2, 1024) 
431   
432      """
433      b = atleast_1d(b)
434      a = atleast_1d(a)
435  
436      fs = _validate_fs(fs, allow_none=False)
437  
438      if worN is None:
439          # For backwards compatibility
440          worN = 512
441  
442      h = None
443  
444      if _is_int_type(worN):
445          N = operator.index(worN)
446          del worN
447          if N < 0:
448              raise ValueError(f'worN must be nonnegative, got {N}')
449          lastpoint = 2 * pi if whole else pi
450          # if include_nyquist is true and whole is false, w should
451          # include end point
452          w = np.linspace(0, lastpoint, N,
453                          endpoint=include_nyquist and not whole)
454          n_fft = N if whole else 2 * (N - 1) if include_nyquist else 2 * N
455          if (a.size == 1 and (b.ndim == 1 or (b.shape[-1] == 1))
456                  and n_fft >= b.shape[0]
457                  and n_fft > 0):  # TODO: review threshold acc. to benchmark?
458              if np.isrealobj(b) and np.isrealobj(a):
459                  fft_func = sp_fft.rfft
460              else:
461                  fft_func = sp_fft.fft
462              h = fft_func(b, n=n_fft, axis=0)[:N]
463              h /= a
464              if fft_func is sp_fft.rfft and whole:
465                  # exclude DC and maybe Nyquist (no need to use axis_reverse
466                  # here because we can build reversal with the truncation)
467                  stop = -1 if n_fft % 2 == 1 else -2
468                  h_flip = slice(stop, 0, -1)
469                  h = np.concatenate((h, h[h_flip].conj()))
470              if b.ndim > 1:
471                  # Last axis of h has length 1, so drop it.
472                  h = h[..., 0]
473                  # Move the first axis of h to the end.
474                  h = np.moveaxis(h, 0, -1)
475      else:
476          w = atleast_1d(worN)
477          del worN
478          w = 2*pi*w/fs
479  
480      if h is None:  # still need to compute using freqs w
481          zm1 = exp(-1j * w)
482          h = (npp_polyval(zm1, b, tensor=False) /
483               npp_polyval(zm1, a, tensor=False))
484  
485      w = w*(fs/(2*pi))
486  
487      if plot is not None:
488          plot(w, h)
489  
490      return w, h

493  def freqz_zpk(z, p, k, worN=512, whole=False, fs=2*pi):
494      r""" 
495      Compute the frequency response of a digital filter in ZPK form. 
496   
497      Given the Zeros, Poles and Gain of a digital filter, compute its frequency 
498      response: 
499   
500      :math:`H(z)=k \prod_i (z - Z[i]) / \prod_j (z - P[j])` 
501   
502      where :math:`k` is the `gain`, :math:`Z` are the `zeros` and :math:`P` are 
503      the `poles`. 
504   
505      Parameters 
506      ---------- 
507      z : array_like 
508          Zeroes of a linear filter 
509      p : array_like 
510          Poles of a linear filter 
511      k : scalar 
512          Gain of a linear filter 
513      worN : {None, int, array_like}, optional 
514          If a single integer, then compute at that many frequencies (default is 
515          N=512). 
516   
517          If an array_like, compute the response at the frequencies given. 
518          These are in the same units as `fs`. 
519      whole : bool, optional 
520          Normally, frequencies are computed from 0 to the Nyquist frequency, 
521          fs/2 (upper-half of unit-circle). If `whole` is True, compute 
522          frequencies from 0 to fs. Ignored if w is array_like. 
523      fs : float, optional 
524          The sampling frequency of the digital system. Defaults to 2*pi 
525          radians/sample (so w is from 0 to pi). 
526   
527          .. versionadded:: 1.2.0 
528   
529      Returns 
530      ------- 
531      w : ndarray 
532          The frequencies at which `h` was computed, in the same units as `fs`. 
533          By default, `w` is normalized to the range [0, pi) (radians/sample). 
534      h : ndarray 
535          The frequency response, as complex numbers. 
536   
537      See Also 
538      -------- 
539      freqs : Compute the frequency response of an analog filter in TF form 
540      freqs_zpk : Compute the frequency response of an analog filter in ZPK form 
541      freqz : Compute the frequency response of a digital filter in TF form 
542   
543      Notes 
544      ----- 
545      .. versionadded:: 0.19.0 
546   
547      Examples 
548      -------- 
549      Design a 4th-order digital Butterworth filter with cut-off of 100 Hz in a 
550      system with sample rate of 1000 Hz, and plot the frequency response: 
551   
552      >>> import numpy as np 
553      >>> from scipy import signal 
554      >>> z, p, k = signal.butter(4, 100, output='zpk', fs=1000) 
555      >>> w, h = signal.freqz_zpk(z, p, k, fs=1000) 
556   
557      >>> import matplotlib.pyplot as plt 
558      >>> fig = plt.figure() 
559      >>> ax1 = fig.add_subplot(1, 1, 1) 
560      >>> ax1.set_title('Digital filter frequency response') 
561   
562      >>> ax1.plot(w, 20 * np.log10(abs(h)), 'b') 
563      >>> ax1.set_ylabel('Amplitude [dB]', color='b') 
564      >>> ax1.set_xlabel('Frequency [Hz]') 
565      >>> ax1.grid(True) 
566   
567      >>> ax2 = ax1.twinx() 
568      >>> phase = np.unwrap(np.angle(h)) 
569      >>> ax2.plot(w, phase, 'g') 
570      >>> ax2.set_ylabel('Phase [rad]', color='g') 
571   
572      >>> plt.axis('tight') 
573      >>> plt.show() 
574   
575      """
576      z, p = map(atleast_1d, (z, p))
577  
578      fs = _validate_fs(fs, allow_none=False)
579  
580      if whole:
581          lastpoint = 2 * pi
582      else:
583          lastpoint = pi
584  
585      if worN is None:
586          # For backwards compatibility
587          w = np.linspace(0, lastpoint, 512, endpoint=False)
588      elif _is_int_type(worN):
589          w = np.linspace(0, lastpoint, worN, endpoint=False)
590      else:
591          w = atleast_1d(worN)
592          w = 2*pi*w/fs
593  
594      zm1 = exp(1j * w)
595      h = k * polyvalfromroots(zm1, z) / polyvalfromroots(zm1, p)
596  
597      w = w*(fs/(2*pi))
598  
599      return w, h

602  def group_delay(system, w=512, whole=False, fs=2*pi):
603      r"""Compute the group delay of a digital filter. 
604   
605      The group delay measures by how many samples amplitude envelopes of 
606      various spectral components of a signal are delayed by a filter. 
607      It is formally defined as the derivative of continuous (unwrapped) phase:: 
608   
609                 d        jw 
610       D(w) = - -- arg H(e) 
611                dw 
612   
613      Parameters 
614      ---------- 
615      system : tuple of array_like (b, a) 
616          Numerator and denominator coefficients of a filter transfer function. 
617      w : {None, int, array_like}, optional 
618          If a single integer, then compute at that many frequencies (default is 
619          N=512). 
620   
621          If an array_like, compute the delay at the frequencies given. These 
622          are in the same units as `fs`. 
623      whole : bool, optional 
624          Normally, frequencies are computed from 0 to the Nyquist frequency, 
625          fs/2 (upper-half of unit-circle). If `whole` is True, compute 
626          frequencies from 0 to fs. Ignored if w is array_like. 
627      fs : float, optional 
628          The sampling frequency of the digital system. Defaults to 2*pi 
629          radians/sample (so w is from 0 to pi). 
630   
631          .. versionadded:: 1.2.0 
632   
633      Returns 
634      ------- 
635      w : ndarray 
636          The frequencies at which group delay was computed, in the same units 
637          as `fs`.  By default, `w` is normalized to the range [0, pi) 
638          (radians/sample). 
639      gd : ndarray 
640          The group delay. 
641   
642      See Also 
643      -------- 
644      freqz : Frequency response of a digital filter 
645   
646      Notes 
647      ----- 
648      The similar function in MATLAB is called `grpdelay`. 
649   
650      If the transfer function :math:`H(z)` has zeros or poles on the unit 
651      circle, the group delay at corresponding frequencies is undefined. 
652      When such a case arises the warning is raised and the group delay 
653      is set to 0 at those frequencies. 
654   
655      For the details of numerical computation of the group delay refer to [1]_ or [2]_. 
656   
657      .. versionadded:: 0.16.0 
658   
659      References 
660      ---------- 
661      .. [1] Richard G. Lyons, "Understanding Digital Signal Processing, 
662             3rd edition", p. 830. 
663      .. [2] Julius O. Smith III, "Numerical Computation of Group Delay", 
664             in "Introduction to Digital Filters with Audio Applications", 
665             online book, 2007, 
666             https://ccrma.stanford.edu/~jos/fp/Numerical_Computation_Group_Delay.html 
667   
668      Examples 
669      -------- 
670      >>> from scipy import signal 
671      >>> b, a = signal.iirdesign(0.1, 0.3, 5, 50, ftype='cheby1') 
672      >>> w, gd = signal.group_delay((b, a)) 
673   
674      >>> import matplotlib.pyplot as plt 
675      >>> plt.title('Digital filter group delay') 
676      >>> plt.plot(w, gd) 
677      >>> plt.ylabel('Group delay [samples]') 
678      >>> plt.xlabel('Frequency [rad/sample]') 
679      >>> plt.show() 
680   
681      """
682      if w is None:
683          # For backwards compatibility
684          w = 512
685  
686      fs = _validate_fs(fs, allow_none=False)
687  
688      if _is_int_type(w):
689          if whole:
690              w = np.linspace(0, 2 * pi, w, endpoint=False)
691          else:
692              w = np.linspace(0, pi, w, endpoint=False)
693      else:
694          w = np.atleast_1d(w)
695          w = 2*pi*w/fs
696  
697      b, a = map(np.atleast_1d, system)
698      c = np.convolve(b, conjugate(a[::-1]))
699      cr = c * np.arange(c.size)
700      z = np.exp(-1j * w)
701      num = np.polyval(cr[::-1], z)
702      den = np.polyval(c[::-1], z)
703      gd = np.real(num / den) - a.size + 1
704      singular = ~np.isfinite(gd)
705      near_singular = np.absolute(den) < 10 * EPSILON
706  
707      if np.any(singular):
708          gd[singular] = 0
709          warnings.warn(
710              "The group delay is singular at frequencies "
711              f"[{', '.join(f'{ws:.3f}' for ws in w[singular])}], setting to 0",
712              stacklevel=2
713          )
714  
715      elif np.any(near_singular):
716          warnings.warn(
717              "The filter's denominator is extremely small at frequencies "
718              f"[{', '.join(f'{ws:.3f}' for ws in w[near_singular])}], "
719              "around which a singularity may be present",
720              stacklevel=2
721          )
722  
723      w = w*(fs/(2*pi))
724  
725      return w, gd

728  def _validate_sos(sos):
729      """Helper to validate a SOS input"""
730      sos = np.atleast_2d(sos)
731      if sos.ndim != 2:
732          raise ValueError('sos array must be 2D')
733      n_sections, m = sos.shape
734      if m != 6:
735          raise ValueError('sos array must be shape (n_sections, 6)')

738      return sos, n_sections

741  def freqz_sos(sos, worN=512, whole=False, fs=2*pi):
742      r""" 
743      Compute the frequency response of a digital filter in SOS format. 
744   
745      Given `sos`, an array with shape (n, 6) of second order sections of 
746      a digital filter, compute the frequency response of the system function:: 
747   
748                 B0(z)   B1(z)         B{n-1}(z) 
749          H(z) = ----- * ----- * ... * --------- 
750                 A0(z)   A1(z)         A{n-1}(z) 
751   
752      for z = exp(omega*1j), where B{k}(z) and A{k}(z) are numerator and 
753      denominator of the transfer function of the k-th second order section. 
754   
755      Parameters 
756      ---------- 
757      sos : array_like 
758          Array of second-order filter coefficients, must have shape 
759          ``(n_sections, 6)``. Each row corresponds to a second-order 
760          section, with the first three columns providing the numerator 
761          coefficients and the last three providing the denominator 
762          coefficients. 
763      worN : {None, int, array_like}, optional 
764          If a single integer, then compute at that many frequencies (default is 
765          N=512).  Using a number that is fast for FFT computations can result 
766          in faster computations (see Notes of `freqz`). 
767   
768          If an array_like, compute the response at the frequencies given (must 
769          be 1-D). These are in the same units as `fs`. 
770      whole : bool, optional 
771          Normally, frequencies are computed from 0 to the Nyquist frequency, 
772          fs/2 (upper-half of unit-circle). If `whole` is True, compute 
773          frequencies from 0 to fs. 
774      fs : float, optional 
775          The sampling frequency of the digital system. Defaults to 2*pi 
776          radians/sample (so w is from 0 to pi). 
777   
778          .. versionadded:: 1.2.0 
779   
780      Returns 
781      ------- 
782      w : ndarray 
783          The frequencies at which `h` was computed, in the same units as `fs`. 
784          By default, `w` is normalized to the range [0, pi) (radians/sample). 
785      h : ndarray 
786          The frequency response, as complex numbers. 
787   
788      See Also 
789      -------- 
790      freqz, sosfilt 
791   
792      Notes 
793      ----- 
794      .. versionadded:: 0.19.0 
795   
796      Examples 
797      -------- 
798      Design a 15th-order bandpass filter in SOS format. 
799   
800      >>> from scipy import signal 
801      >>> import numpy as np 
802      >>> sos = signal.ellip(15, 0.5, 60, (0.2, 0.4), btype='bandpass', 
803      ...                    output='sos') 
804   
805      Compute the frequency response at 1500 points from DC to Nyquist. 
806   
807      >>> w, h = signal.freqz_sos(sos, worN=1500) 
808   
809      Plot the response. 
810   
811      >>> import matplotlib.pyplot as plt 
812      >>> plt.subplot(2, 1, 1) 
813      >>> db = 20*np.log10(np.maximum(np.abs(h), 1e-5)) 
814      >>> plt.plot(w/np.pi, db) 
815      >>> plt.ylim(-75, 5) 
816      >>> plt.grid(True) 
817      >>> plt.yticks([0, -20, -40, -60]) 
818      >>> plt.ylabel('Gain [dB]') 
819      >>> plt.title('Frequency Response') 
820      >>> plt.subplot(2, 1, 2) 
821      >>> plt.plot(w/np.pi, np.angle(h)) 
822      >>> plt.grid(True) 
823      >>> plt.yticks([-np.pi, -0.5*np.pi, 0, 0.5*np.pi, np.pi], 
824      ...            [r'$-\pi$', r'$-\pi/2$', '0', r'$\pi/2$', r'$\pi$']) 
825      >>> plt.ylabel('Phase [rad]') 
826      >>> plt.xlabel('Normalized frequency (1.0 = Nyquist)') 
827      >>> plt.show() 
828   
829      If the same filter is implemented as a single transfer function, 
830      numerical error corrupts the frequency response: 
831   
832      >>> b, a = signal.ellip(15, 0.5, 60, (0.2, 0.4), btype='bandpass', 
833      ...                    output='ba') 
834      >>> w, h = signal.freqz(b, a, worN=1500) 
835      >>> plt.subplot(2, 1, 1) 
836      >>> db = 20*np.log10(np.maximum(np.abs(h), 1e-5)) 
837      >>> plt.plot(w/np.pi, db) 
838      >>> plt.ylim(-75, 5) 
839      >>> plt.grid(True) 
840      >>> plt.yticks([0, -20, -40, -60]) 
841      >>> plt.ylabel('Gain [dB]') 
842      >>> plt.title('Frequency Response') 
843      >>> plt.subplot(2, 1, 2) 
844      >>> plt.plot(w/np.pi, np.angle(h)) 
845      >>> plt.grid(True) 
846      >>> plt.yticks([-np.pi, -0.5*np.pi, 0, 0.5*np.pi, np.pi], 
847      ...            [r'$-\pi$', r'$-\pi/2$', '0', r'$\pi/2$', r'$\pi$']) 
848      >>> plt.ylabel('Phase [rad]') 
849      >>> plt.xlabel('Normalized frequency (1.0 = Nyquist)') 
850      >>> plt.show() 
851   
852      """
853      fs = _validate_fs(fs, allow_none=False)
854  
855      sos, n_sections = _validate_sos(sos)
856      if n_sections == 0:
857          raise ValueError('Cannot compute frequencies with no sections')
858      h = 1.
859      for row in sos:
860          w, rowh = freqz(row[:3], row[3:], worN=worN, whole=whole, fs=fs)
861          h *= rowh
862      return w, h

876  def _cplxreal(z, tol=None):
877      """ 
878      Split into complex and real parts, combining conjugate pairs. 
879   
880      The 1-D input vector `z` is split up into its complex (`zc`) and real (`zr`) 
881      elements. Every complex element must be part of a complex-conjugate pair, 
882      which are combined into a single number (with positive imaginary part) in 
883      the output. Two complex numbers are considered a conjugate pair if their 
884      real and imaginary parts differ in magnitude by less than ``tol * abs(z)``. 
885   
886      Parameters 
887      ---------- 
888      z : array_like 
889          Vector of complex numbers to be sorted and split 
890      tol : float, optional 
891          Relative tolerance for testing realness and conjugate equality. 
892          Default is ``100 * spacing(1)`` of `z`'s data type (i.e., 2e-14 for 
893          float64) 
894   
895      Returns 
896      ------- 
897      zc : ndarray 
898          Complex elements of `z`, with each pair represented by a single value 
899          having positive imaginary part, sorted first by real part, and then 
900          by magnitude of imaginary part. The pairs are averaged when combined 
901          to reduce error. 
902      zr : ndarray 
903          Real elements of `z` (those having imaginary part less than 
904          `tol` times their magnitude), sorted by value. 
905   
906      Raises 
907      ------ 
908      ValueError 
909          If there are any complex numbers in `z` for which a conjugate 
910          cannot be found. 
911   
912      See Also 
913      -------- 
914      _cplxpair 
915   
916      Examples 
917      -------- 
918      >>> from scipy.signal._filter_design import _cplxreal 
919      >>> a = [4, 3, 1, 2-2j, 2+2j, 2-1j, 2+1j, 2-1j, 2+1j, 1+1j, 1-1j] 
920      >>> zc, zr = _cplxreal(a) 
921      >>> print(zc) 
922      [ 1.+1.j  2.+1.j  2.+1.j  2.+2.j] 
923      >>> print(zr) 
924      [ 1.  3.  4.] 
925      """
926  
927      z = atleast_1d(z)
928      if z.size == 0:
929          return z, z
930      elif z.ndim != 1:
931          raise ValueError('_cplxreal only accepts 1-D input')
932  
933      if tol is None:
934          # Get tolerance from dtype of input
935          tol = 100 * np.finfo((1.0 * z).dtype).eps
936  
937      # Sort by real part, magnitude of imaginary part (speed up further sorting)
938      z = z[np.lexsort((abs(z.imag), z.real))]
939  
940      # Split reals from conjugate pairs
941      real_indices = abs(z.imag) <= tol * abs(z)
942      zr = z[real_indices].real
943  
944      if len(zr) == len(z):
945          # Input is entirely real
946          return array([]), zr
947  
948      # Split positive and negative halves of conjugates
949      z = z[~real_indices]
950      zp = z[z.imag > 0]
951      zn = z[z.imag < 0]
952  
953      if len(zp) != len(zn):
954          raise ValueError('Array contains complex value with no matching '
955                           'conjugate.')
956  
957      # Find runs of (approximately) the same real part
958      same_real = np.diff(zp.real) <= tol * abs(zp[:-1])
959      diffs = np.diff(concatenate(([0], same_real, [0])))
960      run_starts = np.nonzero(diffs > 0)[0]
961      run_stops = np.nonzero(diffs < 0)[0]
962  
963      # Sort each run by their imaginary parts
964      for i in range(len(run_starts)):
965          start = run_starts[i]
966          stop = run_stops[i] + 1
967          for chunk in (zp[start:stop], zn[start:stop]):
968              chunk[...] = chunk[np.lexsort([abs(chunk.imag)])]
969  
970      # Check that negatives match positives
971      if any(abs(zp - zn.conj()) > tol * abs(zn)):
972          raise ValueError('Array contains complex value with no matching '
973                           'conjugate.')
974  
975      # Average out numerical inaccuracy in real vs imag parts of pairs
976      zc = (zp + zn.conj()) / 2
977  
978      return zc, zr

981  def _cplxpair(z, tol=None):
982      """ 
983      Sort into pairs of complex conjugates. 
984   
985      Complex conjugates in `z` are sorted by increasing real part. In each 
986      pair, the number with negative imaginary part appears first. 
987   
988      If pairs have identical real parts, they are sorted by increasing 
989      imaginary magnitude. 
990   
991      Two complex numbers are considered a conjugate pair if their real and 
992      imaginary parts differ in magnitude by less than ``tol * abs(z)``.  The 
993      pairs are forced to be exact complex conjugates by averaging the positive 
994      and negative values. 
995   
996      Purely real numbers are also sorted, but placed after the complex 
997      conjugate pairs. A number is considered real if its imaginary part is 
998      smaller than `tol` times the magnitude of the number. 
999   
1000     Parameters 
1001     ---------- 
1002     z : array_like 
1003         1-D input array to be sorted. 
1004     tol : float, optional 
1005         Relative tolerance for testing realness and conjugate equality. 
1006         Default is ``100 * spacing(1)`` of `z`'s data type (i.e., 2e-14 for 
1007         float64) 
1008  
1009     Returns 
1010     ------- 
1011     y : ndarray 
1012         Complex conjugate pairs followed by real numbers. 
1013  
1014     Raises 
1015     ------ 
1016     ValueError 
1017         If there are any complex numbers in `z` for which a conjugate 
1018         cannot be found. 
1019  
1020     See Also 
1021     -------- 
1022     _cplxreal 
1023  
1024     Examples 
1025     -------- 
1026     >>> from scipy.signal._filter_design import _cplxpair 
1027     >>> a = [4, 3, 1, 2-2j, 2+2j, 2-1j, 2+1j, 2-1j, 2+1j, 1+1j, 1-1j] 
1028     >>> z = _cplxpair(a) 
1029     >>> print(z) 
1030     [ 1.-1.j  1.+1.j  2.-1.j  2.+1.j  2.-1.j  2.+1.j  2.-2.j  2.+2.j  1.+0.j 
1031       3.+0.j  4.+0.j] 
1032     """
1033 
1034     z = atleast_1d(z)
1035     if z.size == 0 or np.isrealobj(z):
1036         return np.sort(z)
1037 
1038     if z.ndim != 1:
1039         raise ValueError('z must be 1-D')
1040 
1041     zc, zr = _cplxreal(z, tol)
1042 
1043     # Interleave complex values and their conjugates, with negative imaginary
1044     # parts first in each pair
1045     zc = np.dstack((zc.conj(), zc)).flatten()
1046     z = np.append(zc, zr)
1047     return z

1050 def tf2zpk(b, a):
1051     r"""Return zero, pole, gain (z, p, k) representation from a numerator, 
1052     denominator representation of a linear filter. 
1053  
1054     Parameters 
1055     ---------- 
1056     b : array_like 
1057         Numerator polynomial coefficients. 
1058     a : array_like 
1059         Denominator polynomial coefficients. 
1060  
1061     Returns 
1062     ------- 
1063     z : ndarray 
1064         Zeros of the transfer function. 
1065     p : ndarray 
1066         Poles of the transfer function. 
1067     k : float 
1068         System gain. 
1069  
1070     Notes 
1071     ----- 
1072     If some values of `b` are too close to 0, they are removed. In that case, 
1073     a BadCoefficients warning is emitted. 
1074  
1075     The `b` and `a` arrays are interpreted as coefficients for positive, 
1076     descending powers of the transfer function variable. So the inputs 
1077     :math:`b = [b_0, b_1, ..., b_M]` and :math:`a =[a_0, a_1, ..., a_N]` 
1078     can represent an analog filter of the form: 
1079  
1080     .. math:: 
1081  
1082         H(s) = \frac 
1083         {b_0 s^M + b_1 s^{(M-1)} + \cdots + b_M} 
1084         {a_0 s^N + a_1 s^{(N-1)} + \cdots + a_N} 
1085  
1086     or a discrete-time filter of the form: 
1087  
1088     .. math:: 
1089  
1090         H(z) = \frac 
1091         {b_0 z^M + b_1 z^{(M-1)} + \cdots + b_M} 
1092         {a_0 z^N + a_1 z^{(N-1)} + \cdots + a_N} 
1093  
1094     This "positive powers" form is found more commonly in controls 
1095     engineering. If `M` and `N` are equal (which is true for all filters 
1096     generated by the bilinear transform), then this happens to be equivalent 
1097     to the "negative powers" discrete-time form preferred in DSP: 
1098  
1099     .. math:: 
1100  
1101         H(z) = \frac 
1102         {b_0 + b_1 z^{-1} + \cdots + b_M z^{-M}} 
1103         {a_0 + a_1 z^{-1} + \cdots + a_N z^{-N}} 
1104  
1105     Although this is true for common filters, remember that this is not true 
1106     in the general case. If `M` and `N` are not equal, the discrete-time 
1107     transfer function coefficients must first be converted to the "positive 
1108     powers" form before finding the poles and zeros. 
1109  
1110     Examples 
1111     -------- 
1112     Find the zeroes, poles and gain of 
1113     a filter with the transfer function 
1114  
1115     .. math:: 
1116  
1117         H(s) = \frac{3s^2}{s^2 + 5s + 13} 
1118  
1119     >>> from scipy.signal import tf2zpk 
1120     >>> tf2zpk([3, 0, 0], [1, 5, 13]) 
1121     (   array([ 0.               ,  0.              ]), 
1122         array([ -2.5+2.59807621j ,  -2.5-2.59807621j]), 
1123         3.0) 
1124     """
1125     b, a = normalize(b, a)
1126     b = (b + 0.0) / a[0]
1127     a = (a + 0.0) / a[0]
1128     k = b[0]
1129     b /= b[0]
1130     z = roots(b)
1131     p = roots(a)
1132     return z, p, k

1135 def zpk2tf(z, p, k):
1136     r""" 
1137     Return polynomial transfer function representation from zeros and poles 
1138  
1139     Parameters 
1140     ---------- 
1141     z : array_like 
1142         Zeros of the transfer function. 
1143     p : array_like 
1144         Poles of the transfer function. 
1145     k : float 
1146         System gain. 
1147  
1148     Returns 
1149     ------- 
1150     b : ndarray 
1151         Numerator polynomial coefficients. 
1152     a : ndarray 
1153         Denominator polynomial coefficients. 
1154  
1155     Examples 
1156     -------- 
1157     Find the polynomial representation of a transfer function H(s) 
1158     using its 'zpk' (Zero-Pole-Gain) representation. 
1159  
1160     .. math:: 
1161  
1162         H(z) = 5 \frac 
1163         { (s - 2)(s - 6) } 
1164         { (s - 1)(s - 8) } 
1165  
1166     >>> from scipy.signal import zpk2tf 
1167     >>> z   = [2,   6] 
1168     >>> p   = [1,   8] 
1169     >>> k   = 5 
1170     >>> zpk2tf(z, p, k) 
1171     (   array([  5., -40.,  60.]), array([ 1., -9.,  8.])) 
1172     """
1173     z = atleast_1d(z)
1174     k = atleast_1d(k)
1175     if len(z.shape) > 1:
1176         temp = poly(z[0])
1177         b = np.empty((z.shape[0], z.shape[1] + 1), temp.dtype.char)
1178         if len(k) == 1:
1179             k = [k[0]] * z.shape[0]
1180         for i in range(z.shape[0]):
1181             b[i] = k[i] * poly(z[i])
1182     else:
1183         b = k * poly(z)
1184     a = atleast_1d(poly(p))
1185 
1186     # Use real output if possible. Copied from np.poly, since
1187     # we can't depend on a specific version of numpy.
1188     if issubclass(b.dtype.type, np.complexfloating):
1189         # if complex roots are all complex conjugates, the roots are real.
1190         roots = np.asarray(z, complex)
1191         pos_roots = np.compress(roots.imag > 0, roots)
1192         neg_roots = np.conjugate(np.compress(roots.imag < 0, roots))
1193         if len(pos_roots) == len(neg_roots):
1194             if np.all(np.sort_complex(neg_roots) == np.sort_complex(pos_roots)):
1195                 b = b.real.copy()
1196 
1197     if issubclass(a.dtype.type, np.complexfloating):
1198         # if complex roots are all complex conjugates, the roots are real.
1199         roots = np.asarray(p, complex)
1200         pos_roots = np.compress(roots.imag > 0, roots)
1201         neg_roots = np.conjugate(np.compress(roots.imag < 0, roots))
1202         if len(pos_roots) == len(neg_roots):
1203             if np.all(np.sort_complex(neg_roots) == np.sort_complex(pos_roots)):
1204                 a = a.real.copy()
1205 
1206     return b, a

1209 def tf2sos(b, a, pairing=None, *, analog=False):
1210     r""" 
1211     Return second-order sections from transfer function representation 
1212  
1213     Parameters 
1214     ---------- 
1215     b : array_like 
1216         Numerator polynomial coefficients. 
1217     a : array_like 
1218         Denominator polynomial coefficients. 
1219     pairing : {None, 'nearest', 'keep_odd', 'minimal'}, optional 
1220         The method to use to combine pairs of poles and zeros into sections. 
1221         See `zpk2sos` for information and restrictions on `pairing` and 
1222         `analog` arguments. 
1223     analog : bool, optional 
1224         If True, system is analog, otherwise discrete. 
1225  
1226         .. versionadded:: 1.8.0 
1227  
1228     Returns 
1229     ------- 
1230     sos : ndarray 
1231         Array of second-order filter coefficients, with shape 
1232         ``(n_sections, 6)``. See `sosfilt` for the SOS filter format 
1233         specification. 
1234  
1235     See Also 
1236     -------- 
1237     zpk2sos, sosfilt 
1238  
1239     Notes 
1240     ----- 
1241     It is generally discouraged to convert from TF to SOS format, since doing 
1242     so usually will not improve numerical precision errors. Instead, consider 
1243     designing filters in ZPK format and converting directly to SOS. TF is 
1244     converted to SOS by first converting to ZPK format, then converting 
1245     ZPK to SOS. 
1246  
1247     .. versionadded:: 0.16.0 
1248  
1249     Examples 
1250     -------- 
1251     Find the 'sos' (second-order sections) of the transfer function H(s) 
1252     using its polynomial representation. 
1253  
1254     .. math:: 
1255  
1256         H(s) = \frac{s^2 - 3.5s - 2}{s^4 + 3s^3 - 15s^2 - 19s + 30} 
1257  
1258     >>> from scipy.signal import tf2sos 
1259     >>> tf2sos([1, -3.5, -2], [1, 3, -15, -19, 30], analog=True) 
1260     array([[  0. ,   0. ,   1. ,   1. ,   2. , -15. ], 
1261            [  1. ,  -3.5,  -2. ,   1. ,   1. ,  -2. ]]) 
1262     """
1263     return zpk2sos(*tf2zpk(b, a), pairing=pairing, analog=analog)

1266 def sos2tf(sos):
1267     r""" 
1268     Return a single transfer function from a series of second-order sections 
1269  
1270     Parameters 
1271     ---------- 
1272     sos : array_like 
1273         Array of second-order filter coefficients, must have shape 
1274         ``(n_sections, 6)``. See `sosfilt` for the SOS filter format 
1275         specification. 
1276  
1277     Returns 
1278     ------- 
1279     b : ndarray 
1280         Numerator polynomial coefficients. 
1281     a : ndarray 
1282         Denominator polynomial coefficients. 
1283  
1284     Notes 
1285     ----- 
1286     .. versionadded:: 0.16.0 
1287  
1288     Examples 
1289     -------- 
1290     Find the polynomial representation of an elliptic filter 
1291     using its 'sos' (second-order sections) format. 
1292  
1293     >>> from scipy.signal import sos2tf 
1294     >>> from scipy import signal 
1295     >>> sos = signal.ellip(1, 0.001, 50, 0.1, output='sos') 
1296     >>> sos2tf(sos) 
1297     (   array([0.91256522, 0.91256522, 0.        ]), 
1298         array([1.        , 0.82513043, 0.        ])) 
1299     """
1300     sos = np.asarray(sos)
1301     result_type = sos.dtype
1302     if result_type.kind in 'bui':
1303         result_type = np.float64
1304 
1305     b = np.array([1], dtype=result_type)
1306     a = np.array([1], dtype=result_type)
1307     n_sections = sos.shape[0]
1308     for section in range(n_sections):
1309         b = np.polymul(b, sos[section, :3])
1310         a = np.polymul(a, sos[section, 3:])
1311     return b, a

1314 def sos2zpk(sos):
1315     """ 
1316     Return zeros, poles, and gain of a series of second-order sections 
1317  
1318     Parameters 
1319     ---------- 
1320     sos : array_like 
1321         Array of second-order filter coefficients, must have shape 
1322         ``(n_sections, 6)``. See `sosfilt` for the SOS filter format 
1323         specification. 
1324  
1325     Returns 
1326     ------- 
1327     z : ndarray 
1328         Zeros of the transfer function. 
1329     p : ndarray 
1330         Poles of the transfer function. 
1331     k : float 
1332         System gain. 
1333  
1334     Notes 
1335     ----- 
1336     The number of zeros and poles returned will be ``n_sections * 2`` 
1337     even if some of these are (effectively) zero. 
1338  
1339     .. versionadded:: 0.16.0 
1340     """
1341     sos = np.asarray(sos)
1342     n_sections = sos.shape[0]
1343     z = np.zeros(n_sections*2, np.complex128)
1344     p = np.zeros(n_sections*2, np.complex128)
1345     k = 1.
1346     for section in range(n_sections):
1347         zpk = tf2zpk(sos[section, :3], sos[section, 3:])
1348         z[2*section:2*section+len(zpk[0])] = zpk[0]
1349         p[2*section:2*section+len(zpk[1])] = zpk[1]
1350         k *= zpk[2]
1351     return z, p, k

1354 def _nearest_real_complex_idx(fro, to, which):
1355     """Get the next closest real or complex element based on distance"""
1356     assert which in ('real', 'complex', 'any')
1357     order = np.argsort(np.abs(fro - to))
1358     if which == 'any':
1359         return order[0]
1360     else:
1361         mask = np.isreal(fro[order])
1362         if which == 'complex':
1363             mask = ~mask
1364         return order[np.nonzero(mask)[0][0]]

1376 def zpk2sos(z, p, k, pairing=None, *, analog=False):
1377     """Return second-order sections from zeros, poles, and gain of a system 
1378  
1379     Parameters 
1380     ---------- 
1381     z : array_like 
1382         Zeros of the transfer function. 
1383     p : array_like 
1384         Poles of the transfer function. 
1385     k : float 
1386         System gain. 
1387     pairing : {None, 'nearest', 'keep_odd', 'minimal'}, optional 
1388         The method to use to combine pairs of poles and zeros into sections. 
1389         If analog is False and pairing is None, pairing is set to 'nearest'; 
1390         if analog is True, pairing must be 'minimal', and is set to that if 
1391         it is None. 
1392     analog : bool, optional 
1393         If True, system is analog, otherwise discrete. 
1394  
1395         .. versionadded:: 1.8.0 
1396  
1397     Returns 
1398     ------- 
1399     sos : ndarray 
1400         Array of second-order filter coefficients, with shape 
1401         ``(n_sections, 6)``. See `sosfilt` for the SOS filter format 
1402         specification. 
1403  
1404     See Also 
1405     -------- 
1406     sosfilt 
1407  
1408     Notes 
1409     ----- 
1410     The algorithm used to convert ZPK to SOS format is designed to 
1411     minimize errors due to numerical precision issues. The pairing 
1412     algorithm attempts to minimize the peak gain of each biquadratic 
1413     section. This is done by pairing poles with the nearest zeros, starting 
1414     with the poles closest to the unit circle for discrete-time systems, and 
1415     poles closest to the imaginary axis for continuous-time systems. 
1416  
1417     ``pairing='minimal'`` outputs may not be suitable for `sosfilt`, 
1418     and ``analog=True`` outputs will never be suitable for `sosfilt`. 
1419  
1420     *Algorithms* 
1421  
1422     The steps in the ``pairing='nearest'``, ``pairing='keep_odd'``, 
1423     and ``pairing='minimal'`` algorithms are mostly shared. The 
1424     ``'nearest'`` algorithm attempts to minimize the peak gain, while 
1425     ``'keep_odd'`` minimizes peak gain under the constraint that 
1426     odd-order systems should retain one section as first order. 
1427     ``'minimal'`` is similar to ``'keep_odd'``, but no additional 
1428     poles or zeros are introduced 
1429  
1430     The algorithm steps are as follows: 
1431  
1432     As a pre-processing step for ``pairing='nearest'``, 
1433     ``pairing='keep_odd'``, add poles or zeros to the origin as 
1434     necessary to obtain the same number of poles and zeros for 
1435     pairing.  If ``pairing == 'nearest'`` and there are an odd number 
1436     of poles, add an additional pole and a zero at the origin. 
1437  
1438     The following steps are then iterated over until no more poles or 
1439     zeros remain: 
1440  
1441     1. Take the (next remaining) pole (complex or real) closest to the 
1442        unit circle (or imaginary axis, for ``analog=True``) to 
1443        begin a new filter section. 
1444  
1445     2. If the pole is real and there are no other remaining real poles [#]_, 
1446        add the closest real zero to the section and leave it as a first 
1447        order section. Note that after this step we are guaranteed to be 
1448        left with an even number of real poles, complex poles, real zeros, 
1449        and complex zeros for subsequent pairing iterations. 
1450  
1451     3. Else: 
1452  
1453         1. If the pole is complex and the zero is the only remaining real 
1454            zero*, then pair the pole with the *next* closest zero 
1455            (guaranteed to be complex). This is necessary to ensure that 
1456            there will be a real zero remaining to eventually create a 
1457            first-order section (thus keeping the odd order). 
1458  
1459         2. Else pair the pole with the closest remaining zero (complex or 
1460            real). 
1461  
1462         3. Proceed to complete the second-order section by adding another 
1463            pole and zero to the current pole and zero in the section: 
1464  
1465             1. If the current pole and zero are both complex, add their 
1466                conjugates. 
1467  
1468             2. Else if the pole is complex and the zero is real, add the 
1469                conjugate pole and the next closest real zero. 
1470  
1471             3. Else if the pole is real and the zero is complex, add the 
1472                conjugate zero and the real pole closest to those zeros. 
1473  
1474             4. Else (we must have a real pole and real zero) add the next 
1475                real pole closest to the unit circle, and then add the real 
1476                zero closest to that pole. 
1477  
1478     .. [#] This conditional can only be met for specific odd-order inputs 
1479            with the ``pairing = 'keep_odd'`` or ``'minimal'`` methods. 
1480  
1481     .. versionadded:: 0.16.0 
1482  
1483     Examples 
1484     -------- 
1485  
1486     Design a 6th order low-pass elliptic digital filter for a system with a 
1487     sampling rate of 8000 Hz that has a pass-band corner frequency of 
1488     1000 Hz. The ripple in the pass-band should not exceed 0.087 dB, and 
1489     the attenuation in the stop-band should be at least 90 dB. 
1490  
1491     In the following call to `ellip`, we could use ``output='sos'``, 
1492     but for this example, we'll use ``output='zpk'``, and then convert 
1493     to SOS format with `zpk2sos`: 
1494  
1495     >>> from scipy import signal 
1496     >>> import numpy as np 
1497     >>> z, p, k = signal.ellip(6, 0.087, 90, 1000/(0.5*8000), output='zpk') 
1498  
1499     Now convert to SOS format. 
1500  
1501     >>> sos = signal.zpk2sos(z, p, k) 
1502  
1503     The coefficients of the numerators of the sections: 
1504  
1505     >>> sos[:, :3] 
1506     array([[0.0014152 , 0.00248677, 0.0014152 ], 
1507            [1.        , 0.72976874, 1.        ], 
1508            [1.        , 0.17607852, 1.        ]]) 
1509  
1510     The symmetry in the coefficients occurs because all the zeros are on the 
1511     unit circle. 
1512  
1513     The coefficients of the denominators of the sections: 
1514  
1515     >>> sos[:, 3:] 
1516     array([[ 1.        , -1.32544025,  0.46989976], 
1517            [ 1.        , -1.26118294,  0.62625924], 
1518            [ 1.        , -1.2570723 ,  0.8619958 ]]) 
1519  
1520     The next example shows the effect of the `pairing` option.  We have a 
1521     system with three poles and three zeros, so the SOS array will have 
1522     shape (2, 6). The means there is, in effect, an extra pole and an extra 
1523     zero at the origin in the SOS representation. 
1524  
1525     >>> z1 = np.array([-1, -0.5-0.5j, -0.5+0.5j]) 
1526     >>> p1 = np.array([0.75, 0.8+0.1j, 0.8-0.1j]) 
1527  
1528     With ``pairing='nearest'`` (the default), we obtain 
1529  
1530     >>> signal.zpk2sos(z1, p1, 1) 
1531     array([[ 1.  ,  1.  ,  0.5 ,  1.  , -0.75,  0.  ], 
1532            [ 1.  ,  1.  ,  0.  ,  1.  , -1.6 ,  0.65]]) 
1533  
1534     The first section has the zeros {-0.5-0.05j, -0.5+0.5j} and the poles 
1535     {0, 0.75}, and the second section has the zeros {-1, 0} and poles 
1536     {0.8+0.1j, 0.8-0.1j}. Note that the extra pole and zero at the origin 
1537     have been assigned to different sections. 
1538  
1539     With ``pairing='keep_odd'``, we obtain: 
1540  
1541     >>> signal.zpk2sos(z1, p1, 1, pairing='keep_odd') 
1542     array([[ 1.  ,  1.  ,  0.  ,  1.  , -0.75,  0.  ], 
1543            [ 1.  ,  1.  ,  0.5 ,  1.  , -1.6 ,  0.65]]) 
1544  
1545     The extra pole and zero at the origin are in the same section. 
1546     The first section is, in effect, a first-order section. 
1547  
1548     With ``pairing='minimal'``, the first-order section doesn't have 
1549     the extra pole and zero at the origin: 
1550  
1551     >>> signal.zpk2sos(z1, p1, 1, pairing='minimal') 
1552     array([[ 0.  ,  1.  ,  1.  ,  0.  ,  1.  , -0.75], 
1553            [ 1.  ,  1.  ,  0.5 ,  1.  , -1.6 ,  0.65]]) 
1554  
1555     """
1556     # TODO in the near future:
1557     # 1. Add SOS capability to `filtfilt`, `freqz`, etc. somehow (#3259).
1558     # 2. Make `decimate` use `sosfilt` instead of `lfilter`.
1559     # 3. Make sosfilt automatically simplify sections to first order
1560     #    when possible. Note this might make `sosfiltfilt` a bit harder (ICs).
1561     # 4. Further optimizations of the section ordering / pole-zero pairing.
1562     # See the wiki for other potential issues.
1563 
1564     if pairing is None:
1565         pairing = 'minimal' if analog else 'nearest'
1566 
1567     valid_pairings = ['nearest', 'keep_odd', 'minimal']
1568     if pairing not in valid_pairings:
1569         raise ValueError(f'pairing must be one of {valid_pairings}, not {pairing}')
1570 
1571     if analog and pairing != 'minimal':
1572         raise ValueError('for analog zpk2sos conversion, '
1573                          'pairing must be "minimal"')
1574 
1575     if len(z) == len(p) == 0:
1576         if not analog:
1577             return np.array([[k, 0., 0., 1., 0., 0.]])
1578         else:
1579             return np.array([[0., 0., k, 0., 0., 1.]])
1580 
1581     if pairing != 'minimal':
1582         # ensure we have the same number of poles and zeros, and make copies
1583         p = np.concatenate((p, np.zeros(max(len(z) - len(p), 0))))
1584         z = np.concatenate((z, np.zeros(max(len(p) - len(z), 0))))
1585         n_sections = (max(len(p), len(z)) + 1) // 2
1586 
1587         if len(p) % 2 == 1 and pairing == 'nearest':
1588             p = np.concatenate((p, [0.]))
1589             z = np.concatenate((z, [0.]))
1590         assert len(p) == len(z)
1591     else:
1592         if len(p) < len(z):
1593             raise ValueError('for analog zpk2sos conversion, '
1594                              'must have len(p)>=len(z)')
1595 
1596         n_sections = (len(p) + 1) // 2
1597 
1598     # Ensure we have complex conjugate pairs
1599     # (note that _cplxreal only gives us one element of each complex pair):
1600     z = np.concatenate(_cplxreal(z))
1601     p = np.concatenate(_cplxreal(p))
1602     if not np.isreal(k):
1603         raise ValueError('k must be real')
1604     k = k.real
1605 
1606     if not analog:
1607         # digital: "worst" is the closest to the unit circle
1608         def idx_worst(p):
1609             return np.argmin(np.abs(1 - np.abs(p)))
1610     else:
1611         # analog: "worst" is the closest to the imaginary axis
1612         def idx_worst(p):
1613             return np.argmin(np.abs(np.real(p)))
1614 
1615     sos = np.zeros((n_sections, 6))
1616 
1617     # Construct the system, reversing order so the "worst" are last
1618     for si in range(n_sections-1, -1, -1):
1619         # Select the next "worst" pole
1620         p1_idx = idx_worst(p)
1621         p1 = p[p1_idx]
1622         p = np.delete(p, p1_idx)
1623 
1624         # Pair that pole with a zero
1625 
1626         if np.isreal(p1) and np.isreal(p).sum() == 0:
1627             # Special case (1): last remaining real pole
1628             if pairing != 'minimal':
1629                 z1_idx = _nearest_real_complex_idx(z, p1, 'real')
1630                 z1 = z[z1_idx]
1631                 z = np.delete(z, z1_idx)
1632                 sos[si] = _single_zpksos([z1, 0], [p1, 0], 1)
1633             elif len(z) > 0:
1634                 z1_idx = _nearest_real_complex_idx(z, p1, 'real')
1635                 z1 = z[z1_idx]
1636                 z = np.delete(z, z1_idx)
1637                 sos[si] = _single_zpksos([z1], [p1], 1)
1638             else:
1639                 sos[si] = _single_zpksos([], [p1], 1)
1640 
1641         elif (len(p) + 1 == len(z)
1642               and not np.isreal(p1)
1643               and np.isreal(p).sum() == 1
1644               and np.isreal(z).sum() == 1):
1645 
1646             # Special case (2): there's one real pole and one real zero
1647             # left, and an equal number of poles and zeros to pair up.
1648             # We *must* pair with a complex zero
1649 
1650             z1_idx = _nearest_real_complex_idx(z, p1, 'complex')
1651             z1 = z[z1_idx]
1652             z = np.delete(z, z1_idx)
1653             sos[si] = _single_zpksos([z1, z1.conj()], [p1, p1.conj()], 1)
1654 
1655         else:
1656             if np.isreal(p1):
1657                 prealidx = np.flatnonzero(np.isreal(p))
1658                 p2_idx = prealidx[idx_worst(p[prealidx])]
1659                 p2 = p[p2_idx]
1660                 p = np.delete(p, p2_idx)
1661             else:
1662                 p2 = p1.conj()
1663 
1664             # find closest zero
1665             if len(z) > 0:
1666                 z1_idx = _nearest_real_complex_idx(z, p1, 'any')
1667                 z1 = z[z1_idx]
1668                 z = np.delete(z, z1_idx)
1669 
1670                 if not np.isreal(z1):
1671                     sos[si] = _single_zpksos([z1, z1.conj()], [p1, p2], 1)
1672                 else:
1673                     if len(z) > 0:
1674                         z2_idx = _nearest_real_complex_idx(z, p1, 'real')
1675                         z2 = z[z2_idx]
1676                         assert np.isreal(z2)
1677                         z = np.delete(z, z2_idx)
1678                         sos[si] = _single_zpksos([z1, z2], [p1, p2], 1)
1679                     else:
1680                         sos[si] = _single_zpksos([z1], [p1, p2], 1)
1681             else:
1682                 # no more zeros
1683                 sos[si] = _single_zpksos([], [p1, p2], 1)
1684 
1685     assert len(p) == len(z) == 0  # we've consumed all poles and zeros
1686     del p, z
1687 
1688     # put gain in first sos
1689     sos[0][:3] *= k
1690     return sos

1693 def _align_nums(nums):
1694     """Aligns the shapes of multiple numerators. 
1695  
1696     Given an array of numerator coefficient arrays [[a_1, a_2,..., 
1697     a_n],..., [b_1, b_2,..., b_m]], this function pads shorter numerator 
1698     arrays with zero's so that all numerators have the same length. Such 
1699     alignment is necessary for functions like 'tf2ss', which needs the 
1700     alignment when dealing with SIMO transfer functions. 
1701  
1702     Parameters 
1703     ---------- 
1704     nums: array_like 
1705         Numerator or list of numerators. Not necessarily with same length. 
1706  
1707     Returns 
1708     ------- 
1709     nums: array 
1710         The numerator. If `nums` input was a list of numerators then a 2-D 
1711         array with padded zeros for shorter numerators is returned. Otherwise 
1712         returns ``np.asarray(nums)``. 
1713     """
1714     try:
1715         # The statement can throw a ValueError if one
1716         # of the numerators is a single digit and another
1717         # is array-like e.g. if nums = [5, [1, 2, 3]]
1718         nums = asarray(nums)
1719 
1720         if not np.issubdtype(nums.dtype, np.number):
1721             raise ValueError("dtype of numerator is non-numeric")
1722 
1723         return nums
1724 
1725     except ValueError:
1726         nums = [np.atleast_1d(num) for num in nums]
1727         max_width = max(num.size for num in nums)
1728 
1729         # pre-allocate
1730         aligned_nums = np.zeros((len(nums), max_width))
1731 
1732         # Create numerators with padded zeros
1733         for index, num in enumerate(nums):
1734             aligned_nums[index, -num.size:] = num
1735 
1736         return aligned_nums

1739 def normalize(b, a):
1740     """Normalize numerator/denominator of a continuous-time transfer function. 
1741  
1742     If values of `b` are too close to 0, they are removed. In that case, a 
1743     BadCoefficients warning is emitted. 
1744  
1745     Parameters 
1746     ---------- 
1747     b: array_like 
1748         Numerator of the transfer function. Can be a 2-D array to normalize 
1749         multiple transfer functions. 
1750     a: array_like 
1751         Denominator of the transfer function. At most 1-D. 
1752  
1753     Returns 
1754     ------- 
1755     num: array 
1756         The numerator of the normalized transfer function. At least a 1-D 
1757         array. A 2-D array if the input `num` is a 2-D array. 
1758     den: 1-D array 
1759         The denominator of the normalized transfer function. 
1760  
1761     Notes 
1762     ----- 
1763     Coefficients for both the numerator and denominator should be specified in 
1764     descending exponent order (e.g., ``s^2 + 3s + 5`` would be represented as 
1765     ``[1, 3, 5]``). 
1766  
1767     Examples 
1768     -------- 
1769     >>> from scipy.signal import normalize 
1770  
1771     Normalize the coefficients of the transfer function 
1772     ``(3*s^2 - 2*s + 5) / (2*s^2 + 3*s + 1)``: 
1773  
1774     >>> b = [3, -2, 5] 
1775     >>> a = [2, 3, 1] 
1776     >>> normalize(b, a) 
1777     (array([ 1.5, -1. ,  2.5]), array([1. , 1.5, 0.5])) 
1778  
1779     A warning is generated if, for example, the first coefficient of 
1780     `b` is 0.  In the following example, the result is as expected: 
1781  
1782     >>> import warnings 
1783     >>> with warnings.catch_warnings(record=True) as w: 
1784     ...     num, den = normalize([0, 3, 6], [2, -5, 4]) 
1785  
1786     >>> num 
1787     array([1.5, 3. ]) 
1788     >>> den 
1789     array([ 1. , -2.5,  2. ]) 
1790  
1791     >>> print(w[0].message) 
1792     Badly conditioned filter coefficients (numerator): the results may be meaningless 
1793  
1794     """
1795     num, den = b, a
1796 
1797     den = np.atleast_1d(den)
1798     num = np.atleast_2d(_align_nums(num))
1799 
1800     if den.ndim != 1:
1801         raise ValueError("Denominator polynomial must be rank-1 array.")
1802     if num.ndim > 2:
1803         raise ValueError("Numerator polynomial must be rank-1 or"
1804                          " rank-2 array.")
1805     if np.all(den == 0):
1806         raise ValueError("Denominator must have at least on nonzero element.")
1807 
1808     # Trim leading zeros in denominator, leave at least one.
1809     den = np.trim_zeros(den, 'f')
1810 
1811     # Normalize transfer function
1812     num, den = num / den[0], den / den[0]
1813 
1814     # Count numerator columns that are all zero
1815     leading_zeros = 0
1816     for col in num.T:
1817         if np.allclose(col, 0, atol=1e-14):
1818             leading_zeros += 1
1819         else:
1820             break
1821 
1822     # Trim leading zeros of numerator

1995     r"""

1999     bandwidth `bw` from an analog low-pass filter prototype with unity

2003     ----------

2025     lp2bp_zpk

2027     Notes 
2028     ----- 
2029     This is derived from the s-plane substitution 
2030  
2031     .. math:: s \rightarrow \frac{s^2 + {\omega_0}^2}{s \cdot \mathrm{BW}} 
2032  
2033     This is the "wideband" transformation, producing a passband with 
2034     geometric (log frequency) symmetry about `wo`. 
2035  
2036     Examples 
2037     -------- 
2038     >>> from scipy import signal 
2039     >>> import matplotlib.pyplot as plt 
2040

2056     D = len(a) - 1
2057     N = len(b) - 1
2058     artype = mintypecode((a, b))
2059     ma = max([N, D])
2060     Np = N + ma
2061     Dp = D + ma
2062     bprime = np.empty(Np + 1, artype)
2063     aprime = np.empty(Dp + 1, artype)
2064     wosq = wo * wo
2065     for j in range(Np + 1):
2066         val = 0.0
2067         for i in range(0, N + 1):
2068             for k in range(0, i + 1):
2069                 if ma - i + 2 * k == j:

2073         val = 0.0
2074         for i in range(0, D + 1):
2075             for k in range(0, i + 1):
2076                 if ma - i + 2 * k == j:
2077                     val += comb(i, k) * a[D - i] * (wosq) ** (i - k) / bw ** i
2078         aprime[Dp - j] = val
2079 
2080     return normalize(bprime, aprime)

2267 def iirdesign(wp, ws, gpass, gstop, analog=False, ftype='ellip', output='ba',
2268               fs=None):

2270  
2271     Given passband and stopband frequencies and gains, construct an analog or 
2272     digital IIR filter of minimum order for a given basic type. Return the 
2273     output in numerator, denominator ('ba'), pole-zero ('zpk') or second order 
2274     sections ('sos') form. 
2275  
2276     Parameters 
2277     ---------- 
2278     wp, ws : float or array like, shape (2,) 
2279         Passband and stopband edge frequencies. Possible values are scalars 
2280         (for lowpass and highpass filters) or ranges (for bandpass and bandstop 
2281         filters). 
2282         For digital filters, these are in the same units as `fs`. By default, 
2283         `fs` is 2 half-cycles/sample, so these are normalized from 0 to 1, 
2284         where 1 is the Nyquist frequency. For example: 
2285  
2286             - Lowpass:   wp = 0.2,          ws = 0.3 
2287             - Highpass:  wp = 0.3,          ws = 0.2 
2288             - Bandpass:  wp = [0.2, 0.5],   ws = [0.1, 0.6]

2293         inside stopband or vice versa. Also note that the cutoff at the band edges  
2294         for IIR filters is defined as half-power, so -3dB, not half-amplitude (-6dB) 
2295         like for `scipy.signal.fiwin`.

2313

2316             - pole-zero                          : 'zpk' 
2317

2319         recommended because inferring the coefficients for the 
2320         numerator/denominator form ('ba') suffers from numerical 
2321         instabilities. For reasons of backward compatibility the default 
2322         form is the numerator/denominator form ('ba'), where the 'b' 
2323         and the 'a' in 'ba' refer to the commonly used names of the 
2324         coefficients used. 
2325

2327         associated with additional computational costs: for 
2328         data-intense use cases it is therefore recommended to also

2364     >>> import matplotlib.pyplot as plt

2400                          "Use iirfilter function.") from e

2426                (wp[0] < ws[0] and ws[1] < wp[1])):
2427             raise ValueError("Passband must lie strictly inside stopband "
2428                              "or vice versa")
2429 
2430     band_type = 2 * (len(wp) - 1)
2431     band_type += 1
2432     if wp[0] >= ws[0]:
2433         band_type += 1
2434 
2435     btype = {1: 'lowpass', 2: 'highpass',
2436              3: 'bandstop', 4: 'bandpass'}[band_type]
2437 
2438     N, Wn = ordfunc(wp, ws, gpass, gstop, analog=analog, fs=fs)
2439     return iirfilter(N, Wn, rp=gpass, rs=gstop, analog=analog, btype=btype,
2440                      ftype=ftype, output=output, fs=fs)

2443 def iirfilter(N, Wn, rp=None, rs=None, btype='band', analog=False,
2444               ftype='butter', output='ba', fs=None):
2445     """ 
2446     IIR digital and analog filter design given order and critical points. 
2447  
2448     Design an Nth-order digital or analog filter and return the filter 
2449     coefficients. 
2450  
2451     Parameters 
2452     ---------- 
2453     N : int 
2454         The order of the filter. 
2455     Wn : array_like 
2456         A scalar or length-2 sequence giving the critical frequencies. 
2457  
2458         For digital filters, `Wn` are in the same units as `fs`. By default, 
2459         `fs` is 2 half-cycles/sample, so these are normalized from 0 to 1, 
2460         where 1 is the Nyquist frequency. (`Wn` is thus in 
2461         half-cycles / sample.) 
2462  
2463         For analog filters, `Wn` is an angular frequency (e.g., rad/s). 
2464  
2465         When Wn is a length-2 sequence, ``Wn[0]`` must be less than ``Wn[1]``. 
2466     rp : float, optional 
2467         For Chebyshev and elliptic filters, provides the maximum ripple 
2468         in the passband. (dB) 
2469     rs : float, optional 
2470         For Chebyshev and elliptic filters, provides the minimum attenuation 
2471         in the stop band. (dB) 
2472     btype : {'bandpass', 'lowpass', 'highpass', 'bandstop'}, optional 
2473         The type of filter.  Default is 'bandpass'. 
2474     analog : bool, optional 
2475         When True, return an analog filter, otherwise a digital filter is 
2476         returned. 
2477     ftype : str, optional 
2478         The type of IIR filter to design: 
2479  
2480             - Butterworth   : 'butter' 
2481             - Chebyshev I   : 'cheby1' 
2482             - Chebyshev II  : 'cheby2' 
2483             - Cauer/elliptic: 'ellip' 
2484             - Bessel/Thomson: 'bessel' 
2485  
2486     output : {'ba', 'zpk', 'sos'}, optional 
2487         Filter form of the output: 
2488  
2489             - second-order sections (recommended): 'sos' 
2490             - numerator/denominator (default)    : 'ba' 
2491             - pole-zero                          : 'zpk' 
2492  
2493         In general the second-order sections ('sos') form  is 
2494         recommended because inferring the coefficients for the 
2495         numerator/denominator form ('ba') suffers from numerical 
2496         instabilities. For reasons of backward compatibility the default 
2497         form is the numerator/denominator form ('ba'), where the 'b' 
2498         and the 'a' in 'ba' refer to the commonly used names of the 
2499         coefficients used. 
2500  
2501         Note: Using the second-order sections form ('sos') is sometimes 
2502         associated with additional computational costs: for 
2503         data-intense use cases it is therefore recommended to also 
2504         investigate the numerator/denominator form ('ba'). 
2505  
2506     fs : float, optional 
2507         The sampling frequency of the digital system. 
2508  
2509         .. versionadded:: 1.2.0 
2510  
2511     Returns 
2512     ------- 
2513     b, a : ndarray, ndarray 
2514         Numerator (`b`) and denominator (`a`) polynomials of the IIR filter. 
2515         Only returned if ``output='ba'``. 
2516     z, p, k : ndarray, ndarray, float 
2517         Zeros, poles, and system gain of the IIR filter transfer 
2518         function.  Only returned if ``output='zpk'``. 
2519     sos : ndarray 
2520         Second-order sections representation of the IIR filter. 
2521         Only returned if ``output='sos'``. 
2522  
2523     See Also 
2524     -------- 
2525     butter : Filter design using order and critical points 
2526     cheby1, cheby2, ellip, bessel 
2527     buttord : Find order and critical points from passband and stopband spec 
2528     cheb1ord, cheb2ord, ellipord 
2529     iirdesign : General filter design using passband and stopband spec 
2530  
2531     Notes 
2532     ----- 
2533     The ``'sos'`` output parameter was added in 0.16.0. 
2534  
2535     Examples 
2536     -------- 
2537     Generate a 17th-order Chebyshev II analog bandpass filter from 50 Hz to 
2538     200 Hz and plot the frequency response: 
2539  
2540     >>> import numpy as np 
2541     >>> from scipy import signal 
2542     >>> import matplotlib.pyplot as plt 
2543  
2544     >>> b, a = signal.iirfilter(17, [2*np.pi*50, 2*np.pi*200], rs=60, 
2545     ...                         btype='band', analog=True, ftype='cheby2') 
2546     >>> w, h = signal.freqs(b, a, 1000) 
2547     >>> fig = plt.figure() 
2548     >>> ax = fig.add_subplot(1, 1, 1) 
2549     >>> ax.semilogx(w / (2*np.pi), 20 * np.log10(np.maximum(abs(h), 1e-5))) 
2550     >>> ax.set_title('Chebyshev Type II bandpass frequency response') 
2551     >>> ax.set_xlabel('Frequency [Hz]') 
2552     >>> ax.set_ylabel('Amplitude [dB]') 
2553     >>> ax.axis((10, 1000, -100, 10)) 
2554     >>> ax.grid(which='both', axis='both') 
2555     >>> plt.show() 
2556  
2557     Create a digital filter with the same properties, in a system with 
2558     sampling rate of 2000 Hz, and plot the frequency response. (Second-order 
2559     sections implementation is required to ensure stability of a filter of 
2560     this order): 
2561  
2562     >>> sos = signal.iirfilter(17, [50, 200], rs=60, btype='band', 
2563     ...                        analog=False, ftype='cheby2', fs=2000, 
2564     ...                        output='sos') 
2565     >>> w, h = signal.freqz_sos(sos, 2000, fs=2000) 
2566     >>> fig = plt.figure() 
2567     >>> ax = fig.add_subplot(1, 1, 1) 
2568     >>> ax.semilogx(w, 20 * np.log10(np.maximum(abs(h), 1e-5))) 
2569     >>> ax.set_title('Chebyshev Type II bandpass frequency response') 
2570     >>> ax.set_xlabel('Frequency [Hz]') 
2571     >>> ax.set_ylabel('Amplitude [dB]') 
2572     >>> ax.axis((10, 1000, -100, 10)) 
2573     >>> ax.grid(which='both', axis='both') 
2574     >>> plt.show() 
2575  
2576     """
2577     fs = _validate_fs(fs, allow_none=True)
2578     ftype, btype, output = (x.lower() for x in (ftype, btype, output))
2579     Wn = asarray(Wn)
2580     if fs is not None:
2581         if analog:
2582             raise ValueError("fs cannot be specified for an analog filter")
2583         Wn = Wn / (fs/2)
2584 
2585     if np.any(Wn <= 0):
2586         raise ValueError("filter critical frequencies must be greater than 0")
2587 
2588     if Wn.size > 1 and not Wn[0] < Wn[1]:
2589         raise ValueError("Wn[0] must be less than Wn[1]")
2590 
2591     try:
2592         btype = band_dict[btype]
2593     except KeyError as e:
2594         raise ValueError(f"'{btype}' is an invalid bandtype for filter.") from e
2595 
2596     try:
2597         typefunc = filter_dict[ftype][0]
2598     except KeyError as e:
2599         raise ValueError(f"'{ftype}' is not a valid basic IIR filter.") from e
2600 
2601     if output not in ['ba', 'zpk', 'sos']:
2602         raise ValueError(f"'{output}' is not a valid output form.")
2603 
2604     if rp is not None and rp < 0:
2605         raise ValueError("passband ripple (rp) must be positive")
2606 
2607     if rs is not None and rs < 0:
2608         raise ValueError("stopband attenuation (rs) must be positive")
2609 
2610     # Get analog lowpass prototype
2611     if typefunc == buttap:
2612         z, p, k = typefunc(N)
2613     elif typefunc == besselap:
2614         z, p, k = typefunc(N, norm=bessel_norms[ftype])
2615     elif typefunc == cheb1ap:
2616         if rp is None:
2617             raise ValueError("passband ripple (rp) must be provided to "
2618                              "design a Chebyshev I filter.")
2619         z, p, k = typefunc(N, rp)
2620     elif typefunc == cheb2ap:
2621         if rs is None:
2622             raise ValueError("stopband attenuation (rs) must be provided to "
2623                              "design an Chebyshev II filter.")
2624         z, p, k = typefunc(N, rs)
2625     elif typefunc == ellipap:
2626         if rs is None or rp is None:
2627             raise ValueError("Both rp and rs must be provided to design an "
2628                              "elliptic filter.")
2629         z, p, k = typefunc(N, rp, rs)
2630     else:
2631         raise NotImplementedError(f"'{ftype}' not implemented in iirfilter.")
2632 
2633     # Pre-warp frequencies for digital filter design
2634     if not analog:
2635         if np.any(Wn <= 0) or np.any(Wn >= 1):
2636             if fs is not None:
2637                 raise ValueError("Digital filter critical frequencies must "
2638                                  f"be 0 < Wn < fs/2 (fs={fs} -> fs/2={fs/2})")
2639             raise ValueError("Digital filter critical frequencies "
2640                              "must be 0 < Wn < 1")
2641         fs = 2.0
2642         warped = 2 * fs * tan(pi * Wn / fs)
2643     else:
2644         warped = Wn
2645 
2646     # transform to lowpass, bandpass, highpass, or bandstop
2647     if btype in ('lowpass', 'highpass'):
2648         if np.size(Wn) != 1:
2649             raise ValueError('Must specify a single critical frequency Wn '
2650                              'for lowpass or highpass filter')
2651 
2652         if btype == 'lowpass':
2653             z, p, k = lp2lp_zpk(z, p, k, wo=warped)
2654         elif btype == 'highpass':
2655             z, p, k = lp2hp_zpk(z, p, k, wo=warped)
2656     elif btype in ('bandpass', 'bandstop'):
2657         try:
2658             bw = warped[1] - warped[0]
2659             wo = sqrt(warped[0] * warped[1])
2660         except IndexError as e:
2661             raise ValueError('Wn must specify start and stop frequencies for '
2662                              'bandpass or bandstop filter') from e
2663 
2664         if btype == 'bandpass':
2665             z, p, k = lp2bp_zpk(z, p, k, wo=wo, bw=bw)
2666         elif btype == 'bandstop':
2667             z, p, k = lp2bs_zpk(z, p, k, wo=wo, bw=bw)
2668     else:
2669         raise NotImplementedError(f"'{btype}' not implemented in iirfilter.")
2670 
2671     # Find discrete equivalent if necessary
2672     if not analog:
2673         z, p, k = bilinear_zpk(z, p, k, fs=fs)
2674 
2675     # Transform to proper out type (pole-zero, state-space, numer-denom)
2676     if output == 'zpk':
2677         return z, p, k
2678     elif output == 'ba':
2679         return zpk2tf(z, p, k)
2680     elif output == 'sos':
2681         return zpk2sos(z, p, k, analog=analog)

2687     """
2688     degree = len(p) - len(z)
2689     if degree < 0:
2690         raise ValueError("Improper transfer function. "
2691                          "Must have at least as many poles as zeros.")
2692     else:
2693         return degree

2696 def bilinear_zpk(z, p, k, fs):
2697     r""" 
2698     Return a digital IIR filter from an analog one using a bilinear transform. 
2699  
2700     Transform a set of poles and zeros from the analog s-plane to the digital 
2701     z-plane using Tustin's method, which substitutes ``2*fs*(z-1) / (z+1)`` for 
2702     ``s``, maintaining the shape of the frequency response. 
2703  
2704     Parameters 
2705     ---------- 
2706     z : array_like 
2707         Zeros of the analog filter transfer function. 
2708     p : array_like 
2709         Poles of the analog filter transfer function. 
2710     k : float 
2711         System gain of the analog filter transfer function. 
2712     fs : float 
2713         Sample rate, as ordinary frequency (e.g., hertz). No prewarping is 
2714         done in this function. 
2715  
2716     Returns 
2717     ------- 
2718     z : ndarray 
2719         Zeros of the transformed digital filter transfer function. 
2720     p : ndarray 
2721         Poles of the transformed digital filter transfer function. 
2722     k : float 
2723         System gain of the transformed digital filter. 
2724  
2725     See Also 
2726     -------- 
2727     lp2lp_zpk, lp2hp_zpk, lp2bp_zpk, lp2bs_zpk 
2728     bilinear 
2729  
2730     Notes 
2731     ----- 
2732     .. versionadded:: 1.1.0 
2733  
2734     Examples 
2735     -------- 
2736     >>> import numpy as np 
2737     >>> from scipy import signal 
2738     >>> import matplotlib.pyplot as plt 
2739  
2740     >>> fs = 100 
2741     >>> bf = 2 * np.pi * np.array([7, 13]) 
2742     >>> filts = signal.lti(*signal.butter(4, bf, btype='bandpass', analog=True, 
2743     ...                                   output='zpk')) 
2744     >>> filtz = signal.lti(*signal.bilinear_zpk(filts.zeros, filts.poles, 
2745     ...                                         filts.gain, fs)) 
2746     >>> wz, hz = signal.freqz_zpk(filtz.zeros, filtz.poles, filtz.gain) 
2747     >>> ws, hs = signal.freqs_zpk(filts.zeros, filts.poles, filts.gain, 
2748     ...                           worN=fs*wz) 
2749     >>> plt.semilogx(wz*fs/(2*np.pi), 20*np.log10(np.abs(hz).clip(1e-15)), 
2750     ...              label=r'$|H_z(e^{j \omega})|$') 
2751     >>> plt.semilogx(wz*fs/(2*np.pi), 20*np.log10(np.abs(hs).clip(1e-15)), 
2752     ...              label=r'$|H(j \omega)|$') 
2753     >>> plt.legend() 
2754     >>> plt.xlabel('Frequency [Hz]') 
2755     >>> plt.ylabel('Amplitude [dB]') 
2756     >>> plt.grid(True) 
2757     """
2758     z = atleast_1d(z)
2759     p = atleast_1d(p)
2760 
2761     fs = _validate_fs(fs, allow_none=False)
2762 
2763     degree = _relative_degree(z, p)
2764 
2765     fs2 = 2.0*fs
2766 
2767     # Bilinear transform the poles and zeros
2768     z_z = (fs2 + z) / (fs2 - z)
2769     p_z = (fs2 + p) / (fs2 - p)
2770 
2771     # Any zeros that were at infinity get moved to the Nyquist frequency
2772     z_z = append(z_z, -ones(degree))
2773 
2774     # Compensate for gain change
2775     k_z = k * real(prod(fs2 - z) / prod(fs2 - p))
2776 
2777     return z_z, p_z, k_z

2780 def lp2lp_zpk(z, p, k, wo=1.0):
2781     r""" 
2782     Transform a lowpass filter prototype to a different frequency. 
2783  
2784     Return an analog low-pass filter with cutoff frequency `wo` 
2785     from an analog low-pass filter prototype with unity cutoff frequency, 
2786     using zeros, poles, and gain ('zpk') representation. 
2787  
2788     Parameters 
2789     ---------- 
2790     z : array_like 
2791         Zeros of the analog filter transfer function. 
2792     p : array_like 
2793         Poles of the analog filter transfer function. 
2794     k : float 
2795         System gain of the analog filter transfer function. 
2796     wo : float 
2797         Desired cutoff, as angular frequency (e.g., rad/s). 
2798         Defaults to no change. 
2799  
2800     Returns 
2801     ------- 
2802     z : ndarray 
2803         Zeros of the transformed low-pass filter transfer function. 
2804     p : ndarray 
2805         Poles of the transformed low-pass filter transfer function. 
2806     k : float 
2807         System gain of the transformed low-pass filter. 
2808  
2809     See Also 
2810     -------- 
2811     lp2hp_zpk, lp2bp_zpk, lp2bs_zpk, bilinear 
2812     lp2lp 
2813  
2814     Notes 
2815     ----- 
2816     This is derived from the s-plane substitution 
2817  
2818     .. math:: s \rightarrow \frac{s}{\omega_0} 
2819  
2820     .. versionadded:: 1.1.0 
2821  
2822     Examples 
2823     -------- 
2824     Use the 'zpk' (Zero-Pole-Gain) representation of a lowpass filter to 
2825     transform it to a new 'zpk' representation associated with a cutoff frequency wo. 
2826  
2827     >>> from scipy.signal import lp2lp_zpk 
2828     >>> z   = [7,   2] 
2829     >>> p   = [5,   13] 
2830     >>> k   = 0.8 
2831     >>> wo  = 0.4 
2832     >>> lp2lp_zpk(z, p, k, wo) 
2833     (   array([2.8, 0.8]), array([2. , 5.2]), 0.8) 
2834     """
2835     z = atleast_1d(z)
2836     p = atleast_1d(p)
2837     wo = float(wo)  # Avoid int wraparound
2838 
2839     degree = _relative_degree(z, p)
2840 
2841     # Scale all points radially from origin to shift cutoff frequency
2842     z_lp = wo * z
2843     p_lp = wo * p
2844 
2845     # Each shifted pole decreases gain by wo, each shifted zero increases it.
2846     # Cancel out the net change to keep overall gain the same
2847     k_lp = k * wo**degree
2848 
2849     return z_lp, p_lp, k_lp

2852 def lp2hp_zpk(z, p, k, wo=1.0):
2853     r""" 
2854     Transform a lowpass filter prototype to a highpass filter. 
2855  
2856     Return an analog high-pass filter with cutoff frequency `wo` 
2857     from an analog low-pass filter prototype with unity cutoff frequency, 
2858     using zeros, poles, and gain ('zpk') representation. 
2859  
2860     Parameters 
2861     ---------- 
2862     z : array_like 
2863         Zeros of the analog filter transfer function. 
2864     p : array_like 
2865         Poles of the analog filter transfer function. 
2866     k : float 
2867         System gain of the analog filter transfer function. 
2868     wo : float 
2869         Desired cutoff, as angular frequency (e.g., rad/s). 
2870         Defaults to no change. 
2871  
2872     Returns 
2873     ------- 
2874     z : ndarray 
2875         Zeros of the transformed high-pass filter transfer function. 
2876     p : ndarray 
2877         Poles of the transformed high-pass filter transfer function. 
2878     k : float 
2879         System gain of the transformed high-pass filter. 
2880  
2881     See Also 
2882     -------- 
2883     lp2lp_zpk, lp2bp_zpk, lp2bs_zpk, bilinear 
2884     lp2hp 
2885  
2886     Notes 
2887     ----- 
2888     This is derived from the s-plane substitution 
2889  
2890     .. math:: s \rightarrow \frac{\omega_0}{s} 
2891  
2892     This maintains symmetry of the lowpass and highpass responses on a 
2893     logarithmic scale. 
2894  
2895     .. versionadded:: 1.1.0 
2896  
2897     Examples 
2898     -------- 
2899     Use the 'zpk' (Zero-Pole-Gain) representation of a lowpass filter to 
2900     transform it to a highpass filter with a cutoff frequency wo. 
2901  
2902     >>> from scipy.signal import lp2hp_zpk 
2903     >>> z   = [ -2 + 3j ,  -0.5 - 0.8j ] 
2904     >>> p   = [ -1      ,  -4          ] 
2905     >>> k   = 10 
2906     >>> wo  = 0.6 
2907     >>> lp2hp_zpk(z, p, k, wo) 
2908     (   array([-0.09230769-0.13846154j, -0.33707865+0.53932584j]), 
2909         array([-0.6 , -0.15]), 
2910         8.5) 
2911     """
2912     z = atleast_1d(z)
2913     p = atleast_1d(p)
2914     wo = float(wo)
2915 
2916     degree = _relative_degree(z, p)
2917 
2918     # Invert positions radially about unit circle to convert LPF to HPF
2919     # Scale all points radially from origin to shift cutoff frequency
2920     z_hp = wo / z
2921     p_hp = wo / p
2922 
2923     # If lowpass had zeros at infinity, inverting moves them to origin.
2924     z_hp = append(z_hp, zeros(degree))
2925 
2926     # Cancel out gain change caused by inversion
2927     k_hp = k * real(prod(-z) / prod(-p))
2928 
2929     return z_hp, p_hp, k_hp

2932 def lp2bp_zpk(z, p, k, wo=1.0, bw=1.0):
2933     r""" 
2934     Transform a lowpass filter prototype to a bandpass filter. 
2935  
2936     Return an analog band-pass filter with center frequency `wo` and 
2937     bandwidth `bw` from an analog low-pass filter prototype with unity 
2938     cutoff frequency, using zeros, poles, and gain ('zpk') representation. 
2939  
2940     Parameters 
2941     ---------- 
2942     z : array_like 
2943         Zeros of the analog filter transfer function. 
2944     p : array_like 
2945         Poles of the analog filter transfer function. 
2946     k : float 
2947         System gain of the analog filter transfer function. 
2948     wo : float 
2949         Desired passband center, as angular frequency (e.g., rad/s). 
2950         Defaults to no change. 
2951     bw : float 
2952         Desired passband width, as angular frequency (e.g., rad/s). 
2953         Defaults to 1. 
2954  
2955     Returns 
2956     ------- 
2957     z : ndarray 
2958         Zeros of the transformed band-pass filter transfer function. 
2959     p : ndarray 
2960         Poles of the transformed band-pass filter transfer function. 
2961     k : float 
2962         System gain of the transformed band-pass filter. 
2963  
2964     See Also 
2965     -------- 
2966     lp2lp_zpk, lp2hp_zpk, lp2bs_zpk, bilinear 
2967     lp2bp 
2968  
2969     Notes 
2970     ----- 
2971     This is derived from the s-plane substitution 
2972  
2973     .. math:: s \rightarrow \frac{s^2 + {\omega_0}^2}{s \cdot \mathrm{BW}} 
2974  
2975     This is the "wideband" transformation, producing a passband with 
2976     geometric (log frequency) symmetry about `wo`. 
2977  
2978     .. versionadded:: 1.1.0 
2979  
2980     Examples 
2981     -------- 
2982     Use the 'zpk' (Zero-Pole-Gain) representation of a lowpass filter to 
2983     transform it to a bandpass filter with a center frequency wo and 
2984     bandwidth bw. 
2985  
2986     >>> from scipy.signal import lp2bp_zpk 
2987     >>> z   = [ 5 + 2j ,  5 - 2j ] 
2988     >>> p   = [ 7      ,  -16    ] 
2989     >>> k   = 0.8 
2990     >>> wo  = 0.62 
2991     >>> bw  = 15 
2992     >>> lp2bp_zpk(z, p, k, wo, bw) 
2993     (   array([7.49955815e+01+3.00017676e+01j, 7.49955815e+01-3.00017676e+01j, 
2994                4.41850748e-03-1.76761126e-03j, 4.41850748e-03+1.76761126e-03j]), 
2995         array([1.04996339e+02+0.j, -1.60167736e-03+0.j,  3.66108003e-03+0.j, 
2996                -2.39998398e+02+0.j]), 0.8) 
2997     """
2998     z = atleast_1d(z)
2999     p = atleast_1d(p)
3000     wo = float(wo)
3001     bw = float(bw)
3002 
3003     degree = _relative_degree(z, p)
3004 
3005     # Scale poles and zeros to desired bandwidth
3006     z_lp = z * bw/2
3007     p_lp = p * bw/2
3008 
3009     # Square root needs to produce complex result, not NaN
3010     z_lp = z_lp.astype(complex)
3011     p_lp = p_lp.astype(complex)
3012 
3013     # Duplicate poles and zeros and shift from baseband to +wo and -wo
3014     z_bp = concatenate((z_lp + sqrt(z_lp**2 - wo**2),
3015                         z_lp - sqrt(z_lp**2 - wo**2)))
3016     p_bp = concatenate((p_lp + sqrt(p_lp**2 - wo**2),
3017                         p_lp - sqrt(p_lp**2 - wo**2)))
3018 
3019     # Move degree zeros to origin, leaving degree zeros at infinity for BPF
3020     z_bp = append(z_bp, zeros(degree))

3028 def lp2bs_zpk(z, p, k, wo=1.0, bw=1.0):
3029     r""" 
3030     Transform a lowpass filter prototype to a bandstop filter. 
3031  
3032     Return an analog band-stop filter with center frequency `wo` and 
3033     stopband width `bw` from an analog low-pass filter prototype with unity 
3034     cutoff frequency, using zeros, poles, and gain ('zpk') representation. 
3035  
3036     Parameters 
3037     ---------- 
3038     z : array_like 
3039         Zeros of the analog filter transfer function. 
3040     p : array_like 
3041         Poles of the analog filter transfer function. 
3042     k : float 
3043         System gain of the analog filter transfer function. 
3044     wo : float 
3045         Desired stopband center, as angular frequency (e.g., rad/s). 
3046         Defaults to no change. 
3047     bw : float 
3048         Desired stopband width, as angular frequency (e.g., rad/s). 
3049         Defaults to 1. 
3050  
3051     Returns 
3052     ------- 
3053     z : ndarray 
3054         Zeros of the transformed band-stop filter transfer function. 
3055     p : ndarray 
3056         Poles of the transformed band-stop filter transfer function. 
3057     k : float 
3058         System gain of the transformed band-stop filter. 
3059  
3060     See Also 
3061     -------- 
3062     lp2lp_zpk, lp2hp_zpk, lp2bp_zpk, bilinear 
3063     lp2bs 
3064  
3065     Notes 
3066     ----- 
3067     This is derived from the s-plane substitution 
3068  
3069     .. math:: s \rightarrow \frac{s \cdot \mathrm{BW}}{s^2 + {\omega_0}^2} 
3070  
3071     This is the "wideband" transformation, producing a stopband with 
3072     geometric (log frequency) symmetry about `wo`. 
3073  
3074     .. versionadded:: 1.1.0 
3075  
3076     Examples 
3077     -------- 
3078     Transform a low-pass filter represented in 'zpk' (Zero-Pole-Gain) form 
3079     into a bandstop filter represented in 'zpk' form, with a center frequency wo and 
3080     bandwidth bw. 
3081  
3082     >>> from scipy.signal import lp2bs_zpk 
3083     >>> z   = [             ] 
3084     >>> p   = [ 0.7 ,    -1 ] 
3085     >>> k   = 9 
3086     >>> wo  = 0.5 
3087     >>> bw  = 10 
3088     >>> lp2bs_zpk(z, p, k, wo, bw) 
3089     (   array([0.+0.5j, 0.+0.5j, 0.-0.5j, 0.-0.5j]), 
3090         array([14.2681928 +0.j, -0.02506281+0.j,  0.01752149+0.j, -9.97493719+0.j]), 
3091         -12.857142857142858) 
3092     """
3093     z = atleast_1d(z)
3094     p = atleast_1d(p)
3095     wo = float(wo)
3096     bw = float(bw)
3097 
3098     degree = _relative_degree(z, p)
3099 
3100     # Invert to a highpass filter with desired bandwidth
3101     z_hp = (bw/2) / z
3102     p_hp = (bw/2) / p
3103 
3104     # Square root needs to produce complex result, not NaN
3105     z_hp = z_hp.astype(complex)
3106     p_hp = p_hp.astype(complex)
3107 
3108     # Duplicate poles and zeros and shift from baseband to +wo and -wo
3109     z_bs = concatenate((z_hp + sqrt(z_hp**2 - wo**2),
3110                         z_hp - sqrt(z_hp**2 - wo**2)))
3111     p_bs = concatenate((p_hp + sqrt(p_hp**2 - wo**2),
3112                         p_hp - sqrt(p_hp**2 - wo**2)))
3113 
3114     # Move any zeros that were at infinity to the center of the stopband
3115     z_bs = append(z_bs, full(degree, +1j*wo))
3116     z_bs = append(z_bs, full(degree, -1j*wo))
3117 
3118     # Cancel out gain change caused by inversion
3119     k_bs = k * real(prod(-z) / prod(-p))
3120 
3121     return z_bs, p_bs, k_bs

3604 def bessel(N, Wn, btype='low', analog=False, output='ba', norm='phase',
3605            fs=None):
3606     """ 
3607     Bessel/Thomson digital and analog filter design. 
3608  
3609     Design an Nth-order digital or analog Bessel filter and return the 
3610     filter coefficients. 
3611  
3612     Parameters 
3613     ---------- 
3614     N : int 
3615         The order of the filter. 
3616     Wn : array_like 
3617         A scalar or length-2 sequence giving the critical frequencies (defined 
3618         by the `norm` parameter). 
3619         For analog filters, `Wn` is an angular frequency (e.g., rad/s). 
3620  
3621         For digital filters, `Wn` are in the same units as `fs`.  By default, 
3622         `fs` is 2 half-cycles/sample, so these are normalized from 0 to 1, 
3623         where 1 is the Nyquist frequency. (`Wn` is thus in 
3624         half-cycles / sample.) 
3625     btype : {'lowpass', 'highpass', 'bandpass', 'bandstop'}, optional 
3626         The type of filter.  Default is 'lowpass'. 
3627     analog : bool, optional 
3628         When True, return an analog filter, otherwise a digital filter is 
3629         returned. (See Notes.) 
3630     output : {'ba', 'zpk', 'sos'}, optional 
3631         Type of output:  numerator/denominator ('ba'), pole-zero ('zpk'), or 
3632         second-order sections ('sos'). Default is 'ba'. 
3633     norm : {'phase', 'delay', 'mag'}, optional 
3634         Critical frequency normalization: 
3635  
3636         ``phase`` 
3637             The filter is normalized such that the phase response reaches its 
3638             midpoint at angular (e.g. rad/s) frequency `Wn`. This happens for 
3639             both low-pass and high-pass filters, so this is the 
3640             "phase-matched" case. 
3641  
3642             The magnitude response asymptotes are the same as a Butterworth 
3643             filter of the same order with a cutoff of `Wn`. 
3644  
3645             This is the default, and matches MATLAB's implementation. 
3646  
3647         ``delay`` 
3648             The filter is normalized such that the group delay in the passband 
3649             is 1/`Wn` (e.g., seconds). This is the "natural" type obtained by 
3650             solving Bessel polynomials. 
3651  
3652         ``mag`` 
3653             The filter is normalized such that the gain magnitude is -3 dB at 
3654             angular frequency `Wn`. 
3655  
3656         .. versionadded:: 0.18.0 
3657     fs : float, optional 
3658         The sampling frequency of the digital system. 
3659  
3660         .. versionadded:: 1.2.0 
3661  
3662     Returns 
3663     ------- 
3664     b, a : ndarray, ndarray 
3665         Numerator (`b`) and denominator (`a`) polynomials of the IIR filter. 
3666         Only returned if ``output='ba'``. 
3667     z, p, k : ndarray, ndarray, float 
3668         Zeros, poles, and system gain of the IIR filter transfer 
3669         function.  Only returned if ``output='zpk'``. 
3670     sos : ndarray 
3671         Second-order sections representation of the IIR filter. 
3672         Only returned if ``output='sos'``. 
3673  
3674     Notes 
3675     ----- 
3676     Also known as a Thomson filter, the analog Bessel filter has maximally 
3677     flat group delay and maximally linear phase response, with very little 
3678     ringing in the step response. [1]_ 
3679  
3680     The Bessel is inherently an analog filter. This function generates digital 
3681     Bessel filters using the bilinear transform, which does not preserve the 
3682     phase response of the analog filter. As such, it is only approximately 
3683     correct at frequencies below about fs/4. To get maximally-flat group 
3684     delay at higher frequencies, the analog Bessel filter must be transformed 
3685     using phase-preserving techniques. 
3686  
3687     See `besselap` for implementation details and references. 
3688  
3689     The ``'sos'`` output parameter was added in 0.16.0. 
3690  
3691     References 
3692     ---------- 
3693     .. [1] Thomson, W.E., "Delay Networks having Maximally Flat Frequency 
3694            Characteristics", Proceedings of the Institution of Electrical 
3695            Engineers, Part III, November 1949, Vol. 96, No. 44, pp. 487-490. 
3696  
3697     Examples 
3698     --------

3700     to the Butterworth's cutoff frequency (green):

3702     >>> from scipy import signal 
3703     >>> import matplotlib.pyplot as plt 
3704     >>> import numpy as np

3718     >>> plt.show()

3775 def band_stop_obj(wp, ind, passb, stopb, gpass, gstop, type):
3776     """ 
3777     Band Stop Objective Function for order minimization. 
3778  
3779     Returns the non-integer order for an analog band stop filter. 
3780  
3781     Parameters 
3782     ---------- 
3783     wp : scalar 
3784         Edge of passband `passb`. 
3785     ind : int, {0, 1} 
3786         Index specifying which `passb` edge to vary (0 or 1). 
3787     passb : ndarray 
3788         Two element sequence of fixed passband edges. 
3789     stopb : ndarray 
3790         Two element sequence of fixed stopband edges. 
3791     gstop : float 
3792         Amount of attenuation in stopband in dB. 
3793     gpass : float 
3794         Amount of ripple in the passband in dB. 
3795     type : {'butter', 'cheby', 'ellip'} 
3796         Type of filter. 
3797  
3798     Returns 
3799     ------- 
3800     n : scalar 
3801         Filter order (possibly non-integer). 
3802  
3803     Notes 
3804     ----- 
3805     Band-stop filters are used in applications where certain frequency 
3806     components need to be blocked while others are allowed; for instance, 
3807     removing noise at specific frequencies while allowing the desired signal 
3808     to pass through. The order of a filter often determines its complexity and 
3809     accuracy. Determining the right order can be a challenge. This function 
3810     aims to provide an appropriate order for an analog band stop filter. 
3811      
3812     Examples 
3813     -------- 
3814  
3815     >>> import numpy as np 
3816     >>> from scipy.signal import band_stop_obj 
3817     >>> wp = 2 
3818     >>> ind = 1 
3819     >>> passb = np.array([1, 3]) 
3820     >>> stopb = np.array([0.5, 4]) 
3821     >>> gstop = 30 
3822     >>> gpass = 3 
3823     >>> filter_type = 'butter' 
3824     >>> band_stop_obj(wp, ind, passb, stopb, gpass, gstop, filter_type) 
3825     np.float64(-2.758504160760643) 
3826  
3827     """
3828 
3829     _validate_gpass_gstop(gpass, gstop)
3830 
3831     passbC = passb.copy()
3832     passbC[ind] = wp
3833     nat = (stopb * (passbC[0] - passbC[1]) /
3834            (stopb ** 2 - passbC[0] * passbC[1]))
3835     nat = min(abs(nat))
3836 
3837     if type == 'butter':
3838         GSTOP = 10 ** (0.1 * abs(gstop))
3839         GPASS = 10 ** (0.1 * abs(gpass))
3840         n = (log10((GSTOP - 1.0) / (GPASS - 1.0)) / (2 * log10(nat)))
3841     elif type == 'cheby':
3842         GSTOP = 10 ** (0.1 * abs(gstop))
3843         GPASS = 10 ** (0.1 * abs(gpass))
3844         n = arccosh(sqrt((GSTOP - 1.0) / (GPASS - 1.0))) / arccosh(nat)
3845     elif type == 'ellip':
3846         GSTOP = 10 ** (0.1 * gstop)
3847         GPASS = 10 ** (0.1 * gpass)
3848         arg1 = sqrt((GPASS - 1.0) / (GSTOP - 1.0))
3849         arg0 = 1.0 / nat
3850         d0 = special.ellipk([arg0 ** 2, 1 - arg0 ** 2])
3851         d1 = special.ellipk([arg1 ** 2, 1 - arg1 ** 2])
3852         n = (d0[0] * d1[1] / (d0[1] * d1[0]))
3853     else:
3854         raise ValueError(f"Incorrect type: {type}")
3855     return n

3930     `gstop` dB attenuation in the stopband.

3932     Parameters 
3933     ---------- 
3934     wp, ws : float 
3935         Passband and stopband edge frequencies. 
3936  
3937         For digital filters, these are in the same units as `fs`. By default, 
3938         `fs` is 2 half-cycles/sample, so these are normalized from 0 to 1, 
3939         where 1 is the Nyquist frequency. (`wp` and `ws` are thus in 
3940         half-cycles / sample.) For example: 
3941  
3942             - Lowpass:   wp = 0.2,          ws = 0.3 
3943             - Highpass:  wp = 0.3,          ws = 0.2 
3944             - Bandpass:  wp = [0.2, 0.5],   ws = [0.1, 0.6] 
3945             - Bandstop:  wp = [0.1, 0.6],   ws = [0.2, 0.5] 
3946  
3947         For analog filters, `wp` and `ws` are angular frequencies (e.g., rad/s). 
3948     gpass : float 
3949         The maximum loss in the passband (dB). 
3950     gstop : float 
3951         The minimum attenuation in the stopband (dB). 
3952     analog : bool, optional 
3953         When True, return an analog filter, otherwise a digital filter is 
3954         returned. 
3955     fs : float, optional 
3956         The sampling frequency of the digital system. 
3957  
3958         .. versionadded:: 1.2.0 
3959  
3960     Returns 
3961     ------- 
3962     ord : int 
3963         The lowest order for a Butterworth filter which meets specs. 
3964     wn : ndarray or float 
3965         The Butterworth natural frequency (i.e. the "3dB frequency"). Should 
3966         be used with `butter` to give filter results. If `fs` is specified, 
3967         this is in the same units, and `fs` must also be passed to `butter`. 
3968  
3969     See Also 
3970     -------- 
3971     butter : Filter design using order and critical points 
3972     cheb1ord : Find order and critical points from passband and stopband spec 
3973     cheb2ord, ellipord 
3974     iirfilter : General filter design using order and critical frequencies 
3975     iirdesign : General filter design using passband and stopband spec 
3976  
3977     Examples 
3978     -------- 
3979     Design an analog bandpass filter with passband within 3 dB from 20 to 
3980     50 rad/s, while rejecting at least -40 dB below 14 and above 60 rad/s. 
3981     Plot its frequency response, showing the passband and stopband 
3982     constraints in gray. 
3983  
3984     >>> from scipy import signal 
3985     >>> import matplotlib.pyplot as plt 
3986     >>> import numpy as np 
3987  
3988     >>> N, Wn = signal.buttord([20, 50], [14, 60], 3, 40, True) 
3989     >>> b, a = signal.butter(N, Wn, 'band', True) 
3990     >>> w, h = signal.freqs(b, a, np.logspace(1, 2, 500)) 
3991     >>> plt.semilogx(w, 20 * np.log10(abs(h))) 
3992     >>> plt.title('Butterworth bandpass filter fit to constraints') 
3993     >>> plt.xlabel('Frequency [rad/s]') 
3994     >>> plt.ylabel('Amplitude [dB]') 
3995     >>> plt.grid(which='both', axis='both') 
3996     >>> plt.fill([1,  14,  14,   1], [-40, -40, 99, 99], '0.9', lw=0) # stop 
3997     >>> plt.fill([20, 20,  50,  50], [-99, -3, -3, -99], '0.9', lw=0) # pass 
3998     >>> plt.fill([60, 60, 1e9, 1e9], [99, -40, -40, 99], '0.9', lw=0) # stop 
3999     >>> plt.axis([10, 100, -60, 3]) 
4000     >>> plt.show() 
4001  
4002     """
4003     _validate_gpass_gstop(gpass, gstop)
4004     fs = _validate_fs(fs, allow_none=True)
4005     wp, ws, filter_type = _validate_wp_ws(wp, ws, fs, analog)
4006     passb, stopb = _pre_warp(wp, ws, analog)
4007     nat, passb = _find_nat_freq(stopb, passb, gpass, gstop, filter_type, 'butter')
4008 
4009     GSTOP = 10 ** (0.1 * abs(gstop))
4010     GPASS = 10 ** (0.1 * abs(gpass))
4011     ord = int(ceil(log10((GSTOP - 1.0) / (GPASS - 1.0)) / (2 * log10(nat))))
4012 
4013     # Find the Butterworth natural frequency WN (or the "3dB" frequency")
4014     # to give exactly gpass at passb.
4015     try:
4016         W0 = (GPASS - 1.0) ** (-1.0 / (2.0 * ord))
4017     except ZeroDivisionError:
4018         W0 = 1.0
4019         warnings.warn("Order is zero...check input parameters.",
4020                       RuntimeWarning, stacklevel=2)
4021 
4022     # now convert this frequency back from lowpass prototype
4023     # to the original analog filter
4024 
4025     if filter_type == 1:  # low
4026         WN = W0 * passb
4027     elif filter_type == 2:  # high
4028         WN = passb / W0
4029     elif filter_type == 3:  # stop
4030         WN = np.empty(2, float)
4031         discr = sqrt((passb[1] - passb[0]) ** 2 +
4032                      4 * W0 ** 2 * passb[0] * passb[1])
4033         WN[0] = ((passb[1] - passb[0]) + discr) / (2 * W0)
4034         WN[1] = ((passb[1] - passb[0]) - discr) / (2 * W0)
4035         WN = np.sort(abs(WN))
4036     elif filter_type == 4:  # pass
4037         W0 = np.array([-W0, W0], float)
4038         WN = (-W0 * (passb[1] - passb[0]) / 2.0 +
4039               sqrt(W0 ** 2 / 4.0 * (passb[1] - passb[0]) ** 2 +
4040                    passb[0] * passb[1]))
4041         WN = np.sort(abs(WN))
4042     else:
4043         raise ValueError(f"Bad type: {filter_type}")
4044 
4045     wn = _postprocess_wn(WN, analog, fs)
4046 
4047     return ord, wn

4437         # Avoid divide-by-zero warning
4438         return np.array([]), np.array([]), 1

4467 def _ellipdeg(n, m1):
4468     """Solve degree equation using nomes 
4469  
4470     Given n, m1, solve 
4471        n * K(m) / K'(m) = K1(m1) / K1'(m1) 
4472     for m

4474     See [1], Eq. (49)

4476     References 
4477     ---------- 
4478     .. [1] Orfanidis, "Lecture Notes on Elliptic Filter Design", 
4479            https://www.ece.rutgers.edu/~orfanidi/ece521/notes.pdf 
4480     """

4482     K1p = special.ellipkm1(m1)

4496 # Maximum number of iterations in Landen transformation recursion

4498 # (see _arc_jac_cn [1]) suggests 5.

4502 def _arc_jac_sn(w, m):
4503     """Inverse Jacobian elliptic sn 
4504  
4505     Solve for z in w = sn(z, m) 
4506  
4507     Parameters 
4508     ---------- 
4509     w : complex scalar 
4510         argument 
4511  
4512     m : scalar 
4513         modulus; in interval [0, 1] 
4514  
4515  
4516     See [1], Eq. (56) 
4517  
4518     References 
4519     ---------- 
4520     .. [1] Orfanidis, "Lecture Notes on Elliptic Filter Design", 
4521            https://www.ece.rutgers.edu/~orfanidi/ece521/notes.pdf 
4522  
4523     """
4524 
4525     def _complement(kx):
4526         # (1-k**2) ** 0.5; the expression below
4527         # works for small kx
4528         return ((1 - kx) * (1 + kx)) ** 0.5
4529 
4530     k = m ** 0.5
4531 
4532     if k > 1:
4533         return np.nan
4534     elif k == 1:
4535         return np.arctanh(w)
4536 
4537     ks = [k]
4538     niter = 0
4539     while ks[-1] != 0:
4540         k_ = ks[-1]
4541         k_p = _complement(k_)
4542         ks.append((1 - k_p) / (1 + k_p))
4543         niter += 1
4544         if niter > _ARC_JAC_SN_MAXITER:
4545             raise ValueError('Landen transformation not converging')
4546 
4547     K = np.prod(1 + np.array(ks[1:])) * np.pi/2
4548 
4549     wns = [w]
4550 
4551     for kn, knext in zip(ks[:-1], ks[1:]):
4552         wn = wns[-1]
4553         wnext = (2 * wn /
4554                  ((1 + knext) * (1 + _complement(kn * wn))))
4555         wns.append(wnext)
4556 
4557     u = 2 / np.pi * np.arcsin(wns[-1])
4558 
4559     z = K * u
4560     return z

4589 def ellipap(N, rp, rs):
4590     """Return (z,p,k) of Nth-order elliptic analog lowpass filter. 
4591  
4592     The filter is a normalized prototype that has `rp` decibels of ripple 
4593     in the passband and a stopband `rs` decibels down. 
4594  
4595     The filter's angular (e.g., rad/s) cutoff frequency is normalized to 1, 
4596     defined as the point at which the gain first drops below ``-rp``. 
4597  
4598     See Also 
4599     -------- 
4600     ellip : Filter design function using this prototype 
4601  
4602     References 
4603     ---------- 
4604     .. [1] Lutovac, Tosic, and Evans, "Filter Design for Signal Processing", 
4605            Chapters 5 and 12. 
4606  
4607     .. [2] Orfanidis, "Lecture Notes on Elliptic Filter Design", 
4608            https://www.ece.rutgers.edu/~orfanidi/ece521/notes.pdf 
4609  
4610     """
4611     if abs(int(N)) != N:
4612         raise ValueError("Filter order must be a nonnegative integer")
4613     elif N == 0:
4614         # Avoid divide-by-zero warning
4615         # Even order filters have DC gain of -rp dB
4616         return np.array([]), np.array([]), 10**(-rp/20)
4617     elif N == 1:
4618         p = -sqrt(1.0 / _pow10m1(0.1 * rp))
4619         k = -p
4620         z = []
4621         return asarray(z), asarray(p), k
4622 
4623     eps_sq = _pow10m1(0.1 * rp)
4624 
4625     eps = np.sqrt(eps_sq)
4626     ck1_sq = eps_sq / _pow10m1(0.1 * rs)
4627     if ck1_sq == 0:
4628         raise ValueError("Cannot design a filter with given rp and rs"
4629                          " specifications.")
4630 
4631     val = special.ellipk(ck1_sq), special.ellipkm1(ck1_sq)
4632 
4633     m = _ellipdeg(N, ck1_sq)
4634 
4635     capk = special.ellipk(m)
4636 
4637     j = np.arange(1 - N % 2, N, 2)
4638     jj = len(j)
4639 
4640     [s, c, d, phi] = special.ellipj(j * capk / N, m * np.ones(jj))
4641     snew = np.compress(abs(s) > EPSILON, s, axis=-1)
4642     z = 1.0 / (sqrt(m) * snew)
4643     z = 1j * z
4644     z = np.concatenate((z, conjugate(z)))
4645 
4646     r = _arc_jac_sc1(1. / eps, ck1_sq)
4647     v0 = capk * r / (N * val[0])
4648 
4649     [sv, cv, dv, phi] = special.ellipj(v0, 1 - m)
4650     p = -(c * d * sv * cv + 1j * s * dv) / (1 - (d * sv) ** 2.0)
4651 
4652     if N % 2:
4653         newp = np.compress(
4654             abs(p.imag) > EPSILON * np.sqrt(np.sum(p * np.conjugate(p), axis=0).real),
4655             p, axis=-1
4656         )
4657         p = np.concatenate((p, conjugate(newp)))
4658     else:
4659         p = np.concatenate((p, conjugate(p)))
4660 
4661     k = (np.prod(-p, axis=0) / np.prod(-z, axis=0)).real
4662     if N % 2 == 0:
4663         k = k / np.sqrt(1 + eps_sq)

4693 def _bessel_poly(n, reverse=False):
4694     """ 
4695     Return the coefficients of Bessel polynomial of degree `n` 
4696  
4697     If `reverse` is true, a reverse Bessel polynomial is output. 
4698  
4699     Output is a list of coefficients: 
4700     [1]                   = 1 
4701     [1,  1]               = 1*s   +  1 
4702     [1,  3,  3]           = 1*s^2 +  3*s   +  3 
4703     [1,  6, 15, 15]       = 1*s^3 +  6*s^2 + 15*s   +  15 
4704     [1, 10, 45, 105, 105] = 1*s^4 + 10*s^3 + 45*s^2 + 105*s + 105 
4705     etc. 
4706  
4707     Output is a Python list of arbitrary precision long ints, so n is only 
4708     limited by your hardware's memory. 
4709  
4710     Sequence is http://oeis.org/A001498, and output can be confirmed to 
4711     match http://oeis.org/A001498/b001498.txt : 
4712  
4713     >>> from scipy.signal._filter_design import _bessel_poly 
4714     >>> i = 0 
4715     >>> for n in range(51): 
4716     ...     for x in _bessel_poly(n, reverse=True): 
4717     ...         print(i, x) 
4718     ...         i += 1 
4719  
4720     """
4721     if abs(int(n)) != n:
4722         raise ValueError("Polynomial order must be a nonnegative integer")
4723     else:
4724         n = int(n)  # np.int32 doesn't work, for instance
4725 
4726     out = []
4727     for k in range(n + 1):
4728         num = _falling_factorial(2*n - k, n)
4729         den = 2**(n - k) * math.factorial(k)
4730         out.append(num // den)
4731 
4732     if reverse:
4733         return out[::-1]
4734     else:
4735         return out

4738 def _campos_zeros(n):
4739     """ 
4740     Return approximate zero locations of Bessel polynomials y_n(x) for order 
4741     `n` using polynomial fit (Campos-Calderon 2011) 
4742     """
4743     if n == 1:
4744         return asarray([-1+0j])
4745 
4746     s = npp_polyval(n, [0, 0, 2, 0, -3, 1])
4747     b3 = npp_polyval(n, [16, -8]) / s
4748     b2 = npp_polyval(n, [-24, -12, 12]) / s
4749     b1 = npp_polyval(n, [8, 24, -12, -2]) / s
4750     b0 = npp_polyval(n, [0, -6, 0, 5, -1]) / s
4751 
4752     r = npp_polyval(n, [0, 0, 2, 1])
4753     a1 = npp_polyval(n, [-6, -6]) / r
4754     a2 = 6 / r
4755 
4756     k = np.arange(1, n+1)
4757     x = npp_polyval(k, [0, a1, a2])
4758     y = npp_polyval(k, [b0, b1, b2, b3])
4759 
4760     return x + 1j*y

4763 def _aberth(f, fp, x0, tol=1e-15, maxiter=50):
4764     """ 
4765     Given a function `f`, its first derivative `fp`, and a set of initial 
4766     guesses `x0`, simultaneously find the roots of the polynomial using the 
4767     Aberth-Ehrlich method. 
4768  
4769     ``len(x0)`` should equal the number of roots of `f`. 
4770  
4771     (This is not a complete implementation of Bini's algorithm.) 
4772     """
4773 
4774     N = len(x0)
4775 
4776     x = array(x0, complex)
4777     beta = np.empty_like(x0)
4778 
4779     for iteration in range(maxiter):
4780         alpha = -f(x) / fp(x)  # Newton's method
4781 
4782         # Model "repulsion" between zeros
4783         for k in range(N):
4784             beta[k] = np.sum(1/(x[k] - x[k+1:]))
4785             beta[k] += np.sum(1/(x[k] - x[:k]))
4786 
4787         x += alpha / (1 + alpha * beta)
4788 
4789         if not all(np.isfinite(x)):
4790             raise RuntimeError('Root-finding calculation failed')
4791 
4792         # Mekwi: The iterative process can be stopped when |hn| has become
4793         # less than the largest error one is willing to permit in the root.
4794         if all(abs(alpha) <= tol):
4795             break
4796     else:
4797         raise Exception('Zeros failed to converge')
4798 
4799     return x

4802 def _bessel_zeros(N):
4803     """ 
4804     Find zeros of ordinary Bessel polynomial of order `N`, by root-finding of 
4805     modified Bessel function of the second kind 
4806     """
4807     if N == 0:
4808         return asarray([])
4809 
4810     # Generate starting points
4811     x0 = _campos_zeros(N)
4812 
4813     # Zeros are the same for exp(1/x)*K_{N+0.5}(1/x) and Nth-order ordinary
4814     # Bessel polynomial y_N(x)
4815     def f(x):
4816         return special.kve(N+0.5, 1/x)
4817 
4818     # First derivative of above
4819     def fp(x):
4820         return (special.kve(N-0.5, 1/x)/(2*x**2) -
4821                 special.kve(N+0.5, 1/x)/(x**2) +
4822                 special.kve(N+1.5, 1/x)/(2*x**2))
4823 
4824     # Starting points converge to true zeros
4825     x = _aberth(f, fp, x0)
4826 
4827     # Improve precision using Newton's method on each
4828     for i in range(len(x)):
4829         x[i] = optimize.newton(f, x[i], fp, tol=1e-15)
4830 
4831     # Average complex conjugates to make them exactly symmetrical
4832     x = np.mean((x, x[::-1].conj()), 0)
4833 
4834     # Zeros should sum to -1
4835     if abs(np.sum(x) + 1) > 1e-15:
4836         raise RuntimeError('Generated zeros are inaccurate')
4837 
4838     return x

4841 def _norm_factor(p, k):
4842     """ 
4843     Numerically find frequency shift to apply to delay-normalized filter such 
4844     that -3 dB point is at 1 rad/sec. 
4845  
4846     `p` is an array_like of polynomial poles 
4847     `k` is a float gain 
4848  
4849     First 10 values are listed in "Bessel Scale Factors" table, 
4850     "Bessel Filters Polynomials, Poles and Circuit Elements 2003, C. Bond." 
4851     """
4852     p = asarray(p, dtype=complex)
4853 
4854     def G(w):
4855         """ 
4856         Gain of filter 
4857         """
4858         return abs(k / prod(1j*w - p))
4859 
4860     def cutoff(w):
4861         """ 
4862         When gain = -3 dB, return 0 
4863         """
4864         return G(w) - 1/np.sqrt(2)
4865 
4866     return optimize.newton(cutoff, 1.5)

4869 def besselap(N, norm='phase'):
4870     """ 
4871     Return (z,p,k) for analog prototype of an Nth-order Bessel filter. 
4872  
4873     Parameters 
4874     ---------- 
4875     N : int 
4876         The order of the filter. 
4877     norm : {'phase', 'delay', 'mag'}, optional 
4878         Frequency normalization: 
4879  
4880         ``phase`` 
4881             The filter is normalized such that the phase response reaches its 
4882             midpoint at an angular (e.g., rad/s) cutoff frequency of 1. This 
4883             happens for both low-pass and high-pass filters, so this is the 
4884             "phase-matched" case. [6]_ 
4885  
4886             The magnitude response asymptotes are the same as a Butterworth 
4887             filter of the same order with a cutoff of `Wn`. 
4888  
4889             This is the default, and matches MATLAB's implementation. 
4890  
4891         ``delay`` 
4892             The filter is normalized such that the group delay in the passband 
4893             is 1 (e.g., 1 second). This is the "natural" type obtained by 
4894             solving Bessel polynomials 
4895  
4896         ``mag`` 
4897             The filter is normalized such that the gain magnitude is -3 dB at 
4898             angular frequency 1. This is called "frequency normalization" by 
4899             Bond. [1]_ 
4900  
4901         .. versionadded:: 0.18.0 
4902  
4903     Returns 
4904     ------- 
4905     z : ndarray 
4906         Zeros of the transfer function. Is always an empty array. 
4907     p : ndarray 
4908         Poles of the transfer function. 
4909     k : scalar 
4910         Gain of the transfer function. For phase-normalized, this is always 1. 
4911  
4912     See Also 
4913     -------- 
4914     bessel : Filter design function using this prototype 
4915  
4916     Notes 
4917     ----- 
4918     To find the pole locations, approximate starting points are generated [2]_ 
4919     for the zeros of the ordinary Bessel polynomial [3]_, then the 
4920     Aberth-Ehrlich method [4]_ [5]_ is used on the Kv(x) Bessel function to 
4921     calculate more accurate zeros, and these locations are then inverted about 
4922     the unit circle. 
4923  
4924     References 
4925     ---------- 
4926     .. [1] C.R. Bond, "Bessel Filter Constants", 
4927            http://www.crbond.com/papers/bsf.pdf 
4928     .. [2] Campos and Calderon, "Approximate closed-form formulas for the 
4929            zeros of the Bessel Polynomials", :arXiv:`1105.0957`. 
4930     .. [3] Thomson, W.E., "Delay Networks having Maximally Flat Frequency 
4931            Characteristics", Proceedings of the Institution of Electrical 
4932            Engineers, Part III, November 1949, Vol. 96, No. 44, pp. 487-490. 
4933     .. [4] Aberth, "Iteration Methods for Finding all Zeros of a Polynomial 
4934            Simultaneously", Mathematics of Computation, Vol. 27, No. 122, 
4935            April 1973 
4936     .. [5] Ehrlich, "A modified Newton method for polynomials", Communications 
4937            of the ACM, Vol. 10, Issue 2, pp. 107-108, Feb. 1967, 
4938            :DOI:`10.1145/363067.363115` 
4939     .. [6] Miller and Bohn, "A Bessel Filter Crossover, and Its Relation to 
4940            Others", RaneNote 147, 1998, 
4941            https://www.ranecommercial.com/legacy/note147.html 
4942  
4943     """
4944     if abs(int(N)) != N:
4945         raise ValueError("Filter order must be a nonnegative integer")
4946 
4947     N = int(N)  # calculation below doesn't always fit in np.int64
4948     if N == 0:
4949         p = []
4950         k = 1
4951     else:
4952         # Find roots of reverse Bessel polynomial
4953         p = 1/_bessel_zeros(N)
4954 
4955         a_last = _falling_factorial(2*N, N) // 2**N
4956 
4957         # Shift them to a different normalization if required
4958         if norm in ('delay', 'mag'):
4959             # Normalized for group delay of 1
4960             k = a_last
4961             if norm == 'mag':
4962                 # -3 dB magnitude point is at 1 rad/sec
4963                 norm_factor = _norm_factor(p, k)
4964                 p /= norm_factor
4965                 k = norm_factor**-N * a_last
4966         elif norm == 'phase':
4967             # Phase-matched (1/2 max phase shift at 1 rad/sec)
4968             # Asymptotes are same as Butterworth filter
4969             p *= 10**(-math.log10(a_last)/N)
4970             k = 1
4971         else:
4972             raise ValueError('normalization not understood')
4973 
4974     return asarray([]), asarray(p, dtype=complex), float(k)

5139 def _design_notch_peak_filter(w0, Q, ftype, fs=2.0):
5140     """ 
5141     Design notch or peak digital filter. 
5142  
5143     Parameters 
5144     ---------- 
5145     w0 : float 
5146         Normalized frequency to remove from a signal. If `fs` is specified, 
5147         this is in the same units as `fs`. By default, it is a normalized 
5148         scalar that must satisfy  ``0 < w0 < 1``, with ``w0 = 1`` 
5149         corresponding to half of the sampling frequency. 
5150     Q : float 
5151         Quality factor. Dimensionless parameter that characterizes 
5152         notch filter -3 dB bandwidth ``bw`` relative to its center 
5153         frequency, ``Q = w0/bw``. 
5154     ftype : str 
5155         The type of IIR filter to design: 
5156  
5157             - notch filter : ``notch`` 
5158             - peak filter  : ``peak`` 
5159     fs : float, optional 
5160         The sampling frequency of the digital system. 
5161  
5162         .. versionadded:: 1.2.0: 
5163  
5164     Returns 
5165     ------- 
5166     b, a : ndarray, ndarray 
5167         Numerator (``b``) and denominator (``a``) polynomials 
5168         of the IIR filter. 
5169     """
5170     fs = _validate_fs(fs, allow_none=False)
5171 
5172     # Guarantee that the inputs are floats
5173     w0 = float(w0)
5174     Q = float(Q)
5175     w0 = 2*w0/fs
5176 
5177     # Checks if w0 is within the range
5178     if w0 > 1.0 or w0 < 0.0:
5179         raise ValueError("w0 should be such that 0 < w0 < 1")
5180 
5181     # Get bandwidth
5182     bw = w0/Q
5183 
5184     # Normalize inputs
5185     bw = bw*np.pi
5186     w0 = w0*np.pi
5187 
5188     if ftype not in ("notch", "peak"):
5189         raise ValueError("Unknown ftype.")
5190 
5191     # Compute beta according to Eqs. 11.3.4 (p.575) and 11.3.19 (p.579) from
5192     # reference [1]. Due to assuming a -3 dB attenuation value, i.e, assuming
5193     # gb = 1 / np.sqrt(2), the following terms simplify to:
5194     #   (np.sqrt(1.0 - gb**2.0) / gb) = 1
5195     #   (gb / np.sqrt(1.0 - gb**2.0)) = 1
5196     beta = np.tan(bw/2.0)
5197 
5198     # Compute gain: formula 11.3.6 (p.575) from reference [1]
5199     gain = 1.0/(1.0+beta)
5200 
5201     # Compute numerator b and denominator a
5202     # formulas 11.3.7 (p.575) and 11.3.21 (p.579)
5203     # from reference [1]
5204     if ftype == "notch":
5205         b = gain*np.array([1.0, -2.0*np.cos(w0), 1.0])
5206     else:
5207         b = (1.0-gain)*np.array([1.0, 0.0, -1.0])
5208     a = np.array([1.0, -2.0*gain*np.cos(w0), (2.0*gain-1.0)])
5209 
5210     return b, a

5321     >>> f0 = 250.0  # Frequency to be retained (Hz)

5324     >>> b, a = signal.iircomb(f0, Q, ftype='peak', fs=fs, pass_zero=True) 
5325  
5326     >>> # Frequency response 
5327     >>> freq, h = signal.freqz(b, a, fs=fs) 
5328     >>> response = abs(h) 
5329     >>> # To avoid divide by zero when graphing 
5330     >>> response[response == 0] = 1e-20 
5331     >>> # Plot 
5332     >>> fig, ax = plt.subplots(2, 1, figsize=(8, 6), sharex=True) 
5333     >>> ax[0].plot(freq, 20*np.log10(np.maximum(abs(h), 1e-5)), color='blue') 
5334     >>> ax[0].set_title("Frequency Response") 
5335     >>> ax[0].set_ylabel("Amplitude [dB]", color='blue')

5343     >>> ax[1].set_yticks([-90, -60, -30, 0, 30, 60, 90]) 
5344     >>> ax[1].set_ylim([-90, 90]) 
5345     >>> ax[1].grid(True) 
5346     >>> plt.show() 
5347     """
5348 
5349     # Convert w0, Q, and fs to float
5350     w0 = float(w0)
5351     Q = float(Q)
5352     fs = _validate_fs(fs, allow_none=False)
5353 
5354     # Check for invalid cutoff frequency or filter type
5355     ftype = ftype.lower()
5356     if not 0 < w0 < fs / 2:
5357         raise ValueError(f"w0 must be between 0 and {fs / 2}"
5358                          f" (Nyquist), but given {w0}.")
5359     if ftype not in ('notch', 'peak'):
5360         raise ValueError('ftype must be either notch or peak.')
5361 
5362     # Compute the order of the filter
5363     N = round(fs / w0)
5364 
5365     # Check for cutoff frequency divisibility
5366     if abs(w0 - fs/N)/fs > 1e-14:
5367         raise ValueError('fs must be divisible by w0.')
5368 
5369     # Compute frequency in radians and filter bandwidth
5370     # Eq. 11.3.1 (p. 574) from reference [1]
5371     w0 = (2 * np.pi * w0) / fs
5372     w_delta = w0 / Q
5373 
5374     # Define base gain values depending on notch or peak filter
5375     # Compute -3dB attenuation
5376     # Eqs. 11.4.1 and 11.4.2 (p. 582) from reference [1]
5377     if ftype == 'notch':
5378         G0, G = 1, 0
5379     elif ftype == 'peak':
5380         G0, G = 0, 1
5381 
5382     # Compute beta according to Eq. 11.5.3 (p. 591) from reference [1]. Due to
5383     # assuming a -3 dB attenuation value, i.e, assuming GB = 1 / np.sqrt(2),
5384     # the following term simplifies to:
5385     #   np.sqrt((GB**2 - G0**2) / (G**2 - GB**2)) = 1
5386     beta = np.tan(N * w_delta / 4)
5387 
5388     # Compute filter coefficients
5389     # Eq 11.5.1 (p. 590) variables a, b, c from reference [1]
5390     ax = (1 - beta) / (1 + beta)
5391     bx = (G0 + G * beta) / (1 + beta)
5392     cx = (G0 - G * beta) / (1 + beta)
5393 
5394     # Last coefficients are negative to get peaking comb that passes zero or
5395     # notching comb that doesn't.
5396     negative_coef = ((ftype == 'peak' and pass_zero) or
5397                      (ftype == 'notch' and not pass_zero))
5398 
5399     # Compute numerator coefficients
5400     # Eq 11.5.1 (p. 590) or Eq 11.5.4 (p. 591) from reference [1]
5401     # b - cz^-N or b + cz^-N
5402     b = np.zeros(N + 1)
5403     b[0] = bx
5404     if negative_coef:
5405         b[-1] = -cx
5406     else:
5407         b[-1] = +cx
5408 
5409     # Compute denominator coefficients
5410     # Eq 11.5.1 (p. 590) or Eq 11.5.4 (p. 591) from reference [1]
5411     # 1 - az^-N or 1 + az^-N
5412     a = np.zeros(N + 1)
5413     a[0] = 1
5414     if negative_coef:
5415         a[-1] = -ax
5416     else:
5417         a[-1] = +ax
5418 
5419     return b, a

5433 def gammatone(freq, ftype, order=None, numtaps=None, fs=None):
5434     """ 
5435     Gammatone filter design. 
5436  
5437     This function computes the coefficients of an FIR or IIR gammatone 
5438     digital filter [1]_. 
5439  
5440     Parameters 
5441     ---------- 
5442     freq : float 
5443         Center frequency of the filter (expressed in the same units 
5444         as `fs`). 
5445     ftype : {'fir', 'iir'} 
5446         The type of filter the function generates. If 'fir', the function 
5447         will generate an Nth order FIR gammatone filter. If 'iir', the 
5448         function will generate an 8th order digital IIR filter, modeled as 
5449         as 4th order gammatone filter. 
5450     order : int, optional 
5451         The order of the filter. Only used when ``ftype='fir'``. 
5452         Default is 4 to model the human auditory system. Must be between 
5453         0 and 24. 
5454     numtaps : int, optional 
5455         Length of the filter. Only used when ``ftype='fir'``. 
5456         Default is ``fs*0.015`` if `fs` is greater than 1000, 
5457         15 if `fs` is less than or equal to 1000. 
5458     fs : float, optional 
5459         The sampling frequency of the signal. `freq` must be between 
5460         0 and ``fs/2``. Default is 2. 
5461  
5462     Returns 
5463     ------- 
5464     b, a : ndarray, ndarray 
5465         Numerator (``b``) and denominator (``a``) polynomials of the filter. 
5466  
5467     Raises 
5468     ------ 
5469     ValueError 
5470         If `freq` is less than or equal to 0 or greater than or equal to 
5471         ``fs/2``, if `ftype` is not 'fir' or 'iir', if `order` is less than 
5472         or equal to 0 or greater than 24 when ``ftype='fir'`` 
5473  
5474     See Also 
5475     -------- 
5476     firwin 
5477     iirfilter 
5478  
5479     References 
5480     ---------- 
5481     .. [1] Slaney, Malcolm, "An Efficient Implementation of the 
5482         Patterson-Holdsworth Auditory Filter Bank", Apple Computer 
5483         Technical Report 35, 1993, pp.3-8, 34-39. 
5484  
5485     Examples 
5486     -------- 
5487     16-sample 4th order FIR Gammatone filter centered at 440 Hz 
5488  
5489     >>> from scipy import signal 
5490     >>> signal.gammatone(440, 'fir', numtaps=16, fs=16000) 
5491     (array([ 0.00000000e+00,  2.22196719e-07,  1.64942101e-06,  4.99298227e-06, 
5492         1.01993969e-05,  1.63125770e-05,  2.14648940e-05,  2.29947263e-05, 
5493         1.76776931e-05,  2.04980537e-06, -2.72062858e-05, -7.28455299e-05, 
5494        -1.36651076e-04, -2.19066855e-04, -3.18905076e-04, -4.33156712e-04]), 
5495        [1.0]) 
5496  
5497     IIR Gammatone filter centered at 440 Hz 
5498  
5499     >>> import matplotlib.pyplot as plt 
5500     >>> import numpy as np 
5501  
5502     >>> fc, fs = 440, 16000 
5503     >>> b, a = signal.gammatone(fc, 'iir', fs=fs) 
5504     >>> w, h = signal.freqz(b, a) 
5505     >>> plt.plot(w * fs / (2 * np.pi), 20 * np.log10(abs(h))) 
5506     >>> plt.xscale('log') 
5507     >>> plt.title('Gammatone filter frequency response') 
5508     >>> plt.xlabel('Frequency [Hz]') 
5509     >>> plt.ylabel('Amplitude [dB]') 
5510     >>> plt.margins(0, 0.1) 
5511     >>> plt.grid(which='both', axis='both') 
5512     >>> plt.axvline(fc, color='green') # cutoff frequency 
5513     >>> plt.show() 
5514     """
5515     # Converts freq to float
5516     freq = float(freq)
5517 
5518     # Set sampling rate if not passed
5519     if fs is None:
5520         fs = 2
5521     fs = _validate_fs(fs, allow_none=False)
5522 
5523     # Check for invalid cutoff frequency or filter type
5524     ftype = ftype.lower()
5525     filter_types = ['fir', 'iir']
5526     if not 0 < freq < fs / 2:
5527         raise ValueError(f"The frequency must be between 0 and {fs / 2}"
5528                          f" (Nyquist), but given {freq}.")
5529     if ftype not in filter_types:
5530         raise ValueError('ftype must be either fir or iir.')
5531 
5532     # Calculate FIR gammatone filter
5533     if ftype == 'fir':
5534         # Set order and numtaps if not passed
5535         if order is None:
5536             order = 4
5537         order = operator.index(order)
5538 
5539         if numtaps is None:
5540             numtaps = max(int(fs * 0.015), 15)
5541         numtaps = operator.index(numtaps)
5542 
5543         # Check for invalid order
5544         if not 0 < order <= 24:
5545             raise ValueError("Invalid order: order must be > 0 and <= 24.")
5546 
5547         # Gammatone impulse response settings
5548         t = np.arange(numtaps) / fs
5549         bw = 1.019 * _hz_to_erb(freq)
5550 
5551         # Calculate the FIR gammatone filter
5552         b = (t ** (order - 1)) * np.exp(-2 * np.pi * bw * t)
5553         b *= np.cos(2 * np.pi * freq * t)
5554 
5555         # Scale the FIR filter so the frequency response is 1 at cutoff
5556         scale_factor = 2 * (2 * np.pi * bw) ** (order)
5557         scale_factor /= float_factorial(order - 1)
5558         scale_factor /= fs
5559         b *= scale_factor
5560         a = [1.0]
5561 
5562     # Calculate IIR gammatone filter
5563     elif ftype == 'iir':
5564         # Raise warning if order and/or numtaps is passed
5565         if order is not None:
5566             warnings.warn('order is not used for IIR gammatone filter.', stacklevel=2)
5567         if numtaps is not None:
5568             warnings.warn('numtaps is not used for IIR gammatone filter.', stacklevel=2)
5569 
5570         # Gammatone impulse response settings
5571         T = 1./fs
5572         bw = 2 * np.pi * 1.019 * _hz_to_erb(freq)
5573         fr = 2 * freq * np.pi * T
5574         bwT = bw * T
5575 
5576         # Calculate the gain to normalize the volume at the center frequency
5577         g1 = -2 * np.exp(2j * fr) * T
5578         g2 = 2 * np.exp(-(bwT) + 1j * fr) * T
5579         g3 = np.sqrt(3 + 2 ** (3 / 2)) * np.sin(fr)
5580         g4 = np.sqrt(3 - 2 ** (3 / 2)) * np.sin(fr)
5581         g5 = np.exp(2j * fr)
5582 
5583         g = g1 + g2 * (np.cos(fr) - g4)
5584         g *= (g1 + g2 * (np.cos(fr) + g4))
5585         g *= (g1 + g2 * (np.cos(fr) - g3))
5586         g *= (g1 + g2 * (np.cos(fr) + g3))
5587         g /= ((-2 / np.exp(2 * bwT) - 2 * g5 + 2 * (1 + g5) / np.exp(bwT)) ** 4)
5588         g = np.abs(g)
5589 
5590         # Create empty filter coefficient lists
5591         b = np.empty(5)
5592         a = np.empty(9)
5593 
5594         # Calculate the numerator coefficients
5595         b[0] = (T ** 4) / g
5596         b[1] = -4 * T ** 4 * np.cos(fr) / np.exp(bw * T) / g
5597         b[2] = 6 * T ** 4 * np.cos(2 * fr) / np.exp(2 * bw * T) / g
5598         b[3] = -4 * T ** 4 * np.cos(3 * fr) / np.exp(3 * bw * T) / g
5599         b[4] = T ** 4 * np.cos(4 * fr) / np.exp(4 * bw * T) / g
5600 
5601         # Calculate the denominator coefficients
5602         a[0] = 1
5603         a[1] = -8 * np.cos(fr) / np.exp(bw * T)
5604         a[2] = 4 * (4 + 3 * np.cos(2 * fr)) / np.exp(2 * bw * T)
5605         a[3] = -8 * (6 * np.cos(fr) + np.cos(3 * fr))
5606         a[3] /= np.exp(3 * bw * T)
5607         a[4] = 2 * (18 + 16 * np.cos(2 * fr) + np.cos(4 * fr))
5608         a[4] /= np.exp(4 * bw * T)
5609         a[5] = -8 * (6 * np.cos(fr) + np.cos(3 * fr))
5610         a[5] /= np.exp(5 * bw * T)
5611         a[6] = 4 * (4 + 3 * np.cos(2 * fr)) / np.exp(6 * bw * T)
5612         a[7] = -8 * np.cos(fr) / np.exp(7 * bw * T)
5613         a[8] = np.exp(-8 * bw * T)
5614 
5615     return b, a
