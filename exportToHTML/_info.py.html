1    """ 
2    Array API Inspection namespace 
3     
4    This is the namespace for inspection functions as defined by the array API 
5    standard. See 
6    https://data-apis.org/array-api/latest/API_specification/inspection.html for 
7    more details. 
8     
9    """
10   import torch
11   
12   from functools import cache
13   
14   class __array_namespace_info__:
15       """ 
16       Get the array API inspection namespace for PyTorch. 
17    
18       The array API inspection namespace defines the following functions: 
19    
20       - capabilities() 
21       - default_device() 
22       - default_dtypes() 
23       - dtypes() 
24       - devices() 
25    
26       See 
27       https://data-apis.org/array-api/latest/API_specification/inspection.html 
28       for more details. 
29    
30       Returns 
31       ------- 
32       info : ModuleType 
33           The array API inspection namespace for PyTorch. 
34    
35       Examples 
36       -------- 
37       >>> info = np.__array_namespace_info__() 
38       >>> info.default_dtypes() 
39       {'real floating': numpy.float64, 
40        'complex floating': numpy.complex128, 
41        'integral': numpy.int64, 
42        'indexing': numpy.int64} 
43    
44       """
45   
46       __module__ = 'torch'
47   
48       def capabilities(self):
49           """ 
50           Return a dictionary of array API library capabilities. 
51    
52           The resulting dictionary has the following keys: 
53    
54           - **"boolean indexing"**: boolean indicating whether an array library 
55             supports boolean indexing. Always ``True`` for PyTorch. 
56    
57           - **"data-dependent shapes"**: boolean indicating whether an array 
58             library supports data-dependent output shapes. Always ``True`` for 
59             PyTorch. 
60    
61           See 
62           https://data-apis.org/array-api/latest/API_specification/generated/array_api.info.capabilities.html 
63           for more details. 
64    
65           See Also 
66           -------- 
67           __array_namespace_info__.default_device, 
68           __array_namespace_info__.default_dtypes, 
69           __array_namespace_info__.dtypes, 
70           __array_namespace_info__.devices 
71    
72           Returns 
73           ------- 
74           capabilities : dict 
75               A dictionary of array API library capabilities. 
76    
77           Examples 
78           -------- 
79           >>> info = np.__array_namespace_info__() 
80           >>> info.capabilities() 
81           {'boolean indexing': True, 
82            'data-dependent shapes': True} 
83    
84           """
85           return {
86               "boolean indexing": True,
87               "data-dependent shapes": True,
88               # 'max rank' will be part of the 2024.12 standard
89               # "max rank": 64,
90           }
91   
92       def default_device(self):
93           """ 
94           The default device used for new PyTorch arrays. 
95    
96           See Also 
97           -------- 
98           __array_namespace_info__.capabilities, 
99           __array_namespace_info__.default_dtypes, 
100          __array_namespace_info__.dtypes, 
101          __array_namespace_info__.devices 
102   
103          Returns 
104          ------- 
105          device : str 
106              The default device used for new PyTorch arrays. 
107   
108          Examples 
109          -------- 
110          >>> info = np.__array_namespace_info__() 
111          >>> info.default_device() 
112          'cpu' 
113   
114          """
115          return torch.device("cpu")
116  
117      def default_dtypes(self, *, device=None):
118          """ 
119          The default data types used for new PyTorch arrays. 
120   
121          Parameters 
122          ---------- 
123          device : str, optional 
124              The device to get the default data types for. For PyTorch, only 
125              ``'cpu'`` is allowed. 
126   
127          Returns 
128          ------- 
129          dtypes : dict 
130              A dictionary describing the default data types used for new PyTorch 
131              arrays. 
132   
133          See Also 
134          -------- 
135          __array_namespace_info__.capabilities, 
136          __array_namespace_info__.default_device, 
137          __array_namespace_info__.dtypes, 
138          __array_namespace_info__.devices 
139   
140          Examples 
141          -------- 
142          >>> info = np.__array_namespace_info__() 
143          >>> info.default_dtypes() 
144          {'real floating': torch.float32, 
145           'complex floating': torch.complex64, 
146           'integral': torch.int64, 
147           'indexing': torch.int64} 
148   
149          """
150          # Note: if the default is set to float64, the devices like MPS that
151          # don't support float64 will error. We still return the default_dtype
152          # value here because this error doesn't represent a different default
153          # per-device.
154          default_floating = torch.get_default_dtype()
155          default_complex = torch.complex64 if default_floating == torch.float32 else torch.complex128
156          default_integral = torch.int64
157          return {
158              "real floating": default_floating,
159              "complex floating": default_complex,
160              "integral": default_integral,
161              "indexing": default_integral,
162          }
163  
164  
165      def _dtypes(self, kind):
166          bool = torch.bool
167          int8 = torch.int8
168          int16 = torch.int16
169          int32 = torch.int32
170          int64 = torch.int64
171          uint8 = torch.uint8
172          # uint16, uint32, and uint64 are present in newer versions of pytorch,
173          # but they aren't generally supported by the array API functions, so
174          # we omit them from this function.
175          float32 = torch.float32
176          float64 = torch.float64
177          complex64 = torch.complex64
178          complex128 = torch.complex128
179  
180          if kind is None:
181              return {
182                  "bool": bool,
183                  "int8": int8,
184                  "int16": int16,
185                  "int32": int32,
186                  "int64": int64,
187                  "uint8": uint8,
188                  "float32": float32,
189                  "float64": float64,
190                  "complex64": complex64,
191                  "complex128": complex128,
192              }
193          if kind == "bool":
194              return {"bool": bool}
195          if kind == "signed integer":
196              return {
197                  "int8": int8,
198                  "int16": int16,
199                  "int32": int32,
200                  "int64": int64,
201              }
202          if kind == "unsigned integer":
203              return {
204                  "uint8": uint8,
205              }
206          if kind == "integral":
207              return {
208                  "int8": int8,
209                  "int16": int16,
210                  "int32": int32,
211                  "int64": int64,
212                  "uint8": uint8,
213              }
214          if kind == "real floating":
215              return {
216                  "float32": float32,
217                  "float64": float64,
218              }
219          if kind == "complex floating":
220              return {
221                  "complex64": complex64,
222                  "complex128": complex128,
223              }
224          if kind == "numeric":
225              return {
226                  "int8": int8,
227                  "int16": int16,
228                  "int32": int32,
229                  "int64": int64,
230                  "uint8": uint8,
231                  "float32": float32,
232                  "float64": float64,
233                  "complex64": complex64,
234                  "complex128": complex128,
235              }
236          if isinstance(kind, tuple):
237              res = {}
238              for k in kind:
239                  res.update(self.dtypes(kind=k))
240              return res
241          raise ValueError(f"unsupported kind: {kind!r}")
242  
243      @cache
244      def dtypes(self, *, device=None, kind=None):
245          """ 
246          The array API data types supported by PyTorch. 
247   
248          Note that this function only returns data types that are defined by 
249          the array API. 
250   
251          Parameters 
252          ---------- 
253          device : str, optional 
254              The device to get the data types for. 
255          kind : str or tuple of str, optional 
256              The kind of data types to return. If ``None``, all data types are 
257              returned. If a string, only data types of that kind are returned. 
258              If a tuple, a dictionary containing the union of the given kinds 
259              is returned. The following kinds are supported: 
260   
261              - ``'bool'``: boolean data types (i.e., ``bool``). 
262              - ``'signed integer'``: signed integer data types (i.e., ``int8``, 
263                ``int16``, ``int32``, ``int64``). 
264              - ``'unsigned integer'``: unsigned integer data types (i.e., 
265                ``uint8``, ``uint16``, ``uint32``, ``uint64``). 
266              - ``'integral'``: integer data types. Shorthand for ``('signed 
267                integer', 'unsigned integer')``. 
268              - ``'real floating'``: real-valued floating-point data types 
269                (i.e., ``float32``, ``float64``). 
270              - ``'complex floating'``: complex floating-point data types (i.e., 
271                ``complex64``, ``complex128``). 
272              - ``'numeric'``: numeric data types. Shorthand for ``('integral', 
273                'real floating', 'complex floating')``. 
274   
275          Returns 
276          ------- 
277          dtypes : dict 
278              A dictionary mapping the names of data types to the corresponding 
279              PyTorch data types. 
280   
281          See Also 
282          -------- 
283          __array_namespace_info__.capabilities, 
284          __array_namespace_info__.default_device, 
285          __array_namespace_info__.default_dtypes, 
286          __array_namespace_info__.devices 
287   
288          Examples 
289          -------- 
290          >>> info = np.__array_namespace_info__() 
291          >>> info.dtypes(kind='signed integer') 
292          {'int8': numpy.int8, 
293           'int16': numpy.int16, 
294           'int32': numpy.int32, 
295           'int64': numpy.int64} 
296   
297          """
298          res = self._dtypes(kind)
299          for k, v in res.copy().items():
300              try:
301                  torch.empty((0,), dtype=v, device=device)
302              except:
303                  del res[k]
304          return res
305  
306      @cache
307      def devices(self):
308          """ 
309          The devices supported by PyTorch. 
310   
311          Returns 
312          ------- 
313          devices : list of str 
314              The devices supported by PyTorch. 
315   
316          See Also 
317          -------- 
318          __array_namespace_info__.capabilities, 
319          __array_namespace_info__.default_device, 
320          __array_namespace_info__.default_dtypes, 
321          __array_namespace_info__.dtypes 
322   
323          Examples 
324          -------- 
325          >>> info = np.__array_namespace_info__() 
326          >>> info.devices() 
327          [device(type='cpu'), device(type='mps', index=0), device(type='meta')] 
328   
329          """
330          # Torch doesn't have a straightforward way to get the list of all
331          # currently supported devices. To do this, we first parse the error
332          # message of torch.device to get the list of all possible types of
333          # device:
334          try:
335              torch.device('notadevice')
336          except RuntimeError as e:
337              # The error message is something like:
338              # "Expected one of cpu, cuda, ipu, xpu, mkldnn, opengl, opencl, ideep, hip, ve, fpga, ort, xla, lazy, vulkan, mps, meta, hpu, mtia, privateuseone device type at start of device string: notadevice"
339              devices_names = e.args[0].split('Expected one of ')[1].split(' device type')[0].split(', ')
340  
341          # Next we need to check for different indices for different devices.
342          # device(device_name, index=index) doesn't actually check if the
343          # device name or index is valid. We have to try to create a tensor
344          # with it (which is why this function is cached).
345          devices = []
346          for device_name in devices_names:
347              i = 0
348              while True:
349                  try:
350                      a = torch.empty((0,), device=torch.device(device_name, index=i))
351                      if a.device in devices:
352                          break
353                      devices.append(a.device)
354                  except:
355                      break
356                  i += 1
357  
358          return devices
