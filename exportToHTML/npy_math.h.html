32       const union { npy_uint32 __i; float __f;} __bint = {0x7f800000UL};

454  static inline npy_cdouble npy_cpack(double x, double y)
455  {
456      npy_cdouble z;
457      npy_csetreal(&z, x);
458      npy_csetimag(&z, y);
459      return z;
460  }
