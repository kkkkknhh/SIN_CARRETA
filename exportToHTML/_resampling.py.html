20   def _vectorize_statistic(statistic):
21       """Vectorize an n-sample statistic"""
22       # This is a little cleaner than np.nditer at the expense of some data
23       # copying: concatenate samples together, then use np.apply_along_axis
24       def stat_nd(*data, axis=0):
25           lengths = [sample.shape[axis] for sample in data]
26           split_indices = np.cumsum(lengths)[:-1]
27           z = _broadcast_concatenate(data, axis)
28   
29           # move working axis to position 0 so that new dimensions in the output
30           # of `statistic` are _prepended_. ("This axis is removed, and replaced
31           # with new dimensions...")
32           z = np.moveaxis(z, axis, 0)
33   
34           def stat_1d(z):
35               data = np.split(z, split_indices)
36               return statistic(*data)
37   
38           return np.apply_along_axis(stat_1d, 0, z)[()]
39       return stat_nd

42   def _jackknife_resample(sample, batch=None):
43       """Jackknife resample the sample. Only one-sample stats for now."""
44       n = sample.shape[-1]
45       batch_nominal = batch or n
46   
47       for k in range(0, n, batch_nominal):
48           # col_start:col_end are the observations to remove
49           batch_actual = min(batch_nominal, n-k)
50   
51           # jackknife - each row leaves out one observation
52           j = np.ones((batch_actual, n), dtype=bool)
53           np.fill_diagonal(j[:, k:k+batch_actual], False)
54           i = np.arange(n)
55           i = np.broadcast_to(i, (batch_actual, n))
56           i = i[j].reshape((batch_actual, n-1))
57   
58           resamples = sample[..., i]
59           yield resamples

62   def _bootstrap_resample(sample, n_resamples=None, rng=None):
63       """Bootstrap resample the sample."""
64       n = sample.shape[-1]
65   
66       # bootstrap - each row is a random resample of original observations
67       i = rng_integers(rng, 0, n, (n_resamples, n))
68   
69       resamples = sample[..., i]
70       return resamples

73   def _percentile_of_score(a, score, axis):
74       """Vectorized, simplified `scipy.stats.percentileofscore`. 
75       Uses logic of the 'mean' value of percentileofscore's kind parameter. 
76    
77       Unlike `stats.percentileofscore`, the percentile returned is a fraction 
78       in [0, 1]. 
79       """
80       B = a.shape[axis]
81       return ((a < score).sum(axis=axis) + (a <= score).sum(axis=axis)) / (2 * B)

84   def _percentile_along_axis(theta_hat_b, alpha):
85       """`np.percentile` with different percentile for each slice."""
86       # the difference between _percentile_along_axis and np.percentile is that
87       # np.percentile gets _all_ the qs for each axis slice, whereas
88       # _percentile_along_axis gets the q corresponding with each axis slice
89       shape = theta_hat_b.shape[:-1]
90       alpha = np.broadcast_to(alpha, shape)
91       percentiles = np.zeros_like(alpha, dtype=np.float64)
92       for indices, alpha_i in np.ndenumerate(alpha):
93           if np.isnan(alpha_i):
94               # e.g. when bootstrap distribution has only one unique element
95               msg = (
96                   "The BCa confidence interval cannot be calculated."
97                   " This problem is known to occur when the distribution"
98                   " is degenerate or the statistic is np.min."
99               )
100              warnings.warn(DegenerateDataWarning(msg), stacklevel=3)
101              percentiles[indices] = np.nan
102          else:
103              theta_hat_b_i = theta_hat_b[indices]
104              percentiles[indices] = np.percentile(theta_hat_b_i, alpha_i)
105      return percentiles[()]  # return scalar instead of 0d array

108  def _bca_interval(data, statistic, axis, alpha, theta_hat_b, batch):
109      """Bias-corrected and accelerated interval."""
110      # closely follows [1] 14.3 and 15.4 (Eq. 15.36)
111  
112      # calculate z0_hat
113      theta_hat = np.asarray(statistic(*data, axis=axis))[..., None]
114      percentile = _percentile_of_score(theta_hat_b, theta_hat, axis=-1)
115      z0_hat = ndtri(percentile)
116  
117      # calculate a_hat
118      theta_hat_ji = []  # j is for sample of data, i is for jackknife resample
119      for j, sample in enumerate(data):
120          # _jackknife_resample will add an axis prior to the last axis that
121          # corresponds with the different jackknife resamples. Do the same for
122          # each sample of the data to ensure broadcastability. We need to
123          # create a copy of the list containing the samples anyway, so do this
124          # in the loop to simplify the code. This is not the bottleneck...
125          samples = [np.expand_dims(sample, -2) for sample in data]
126          theta_hat_i = []
127          for jackknife_sample in _jackknife_resample(sample, batch):
128              samples[j] = jackknife_sample
129              broadcasted = _broadcast_arrays(samples, axis=-1)
130              theta_hat_i.append(statistic(*broadcasted, axis=-1))
131          theta_hat_ji.append(theta_hat_i)
132  
133      theta_hat_ji = [np.concatenate(theta_hat_i, axis=-1)
134                      for theta_hat_i in theta_hat_ji]
135  
136      n_j = [theta_hat_i.shape[-1] for theta_hat_i in theta_hat_ji]
137  
138      theta_hat_j_dot = [theta_hat_i.mean(axis=-1, keepdims=True)
139                         for theta_hat_i in theta_hat_ji]
140  
141      U_ji = [(n - 1) * (theta_hat_dot - theta_hat_i)
142              for theta_hat_dot, theta_hat_i, n
143              in zip(theta_hat_j_dot, theta_hat_ji, n_j)]
144  
145      nums = [(U_i**3).sum(axis=-1)/n**3 for U_i, n in zip(U_ji, n_j)]
146      dens = [(U_i**2).sum(axis=-1)/n**2 for U_i, n in zip(U_ji, n_j)]
147      a_hat = 1/6 * sum(nums) / sum(dens)**(3/2)
148  
149      # calculate alpha_1, alpha_2
150      z_alpha = ndtri(alpha)
151      z_1alpha = -z_alpha
152      num1 = z0_hat + z_alpha
153      alpha_1 = ndtr(z0_hat + num1/(1 - a_hat*num1))
154      num2 = z0_hat + z_1alpha
155      alpha_2 = ndtr(z0_hat + num2/(1 - a_hat*num2))
156      return alpha_1, alpha_2, a_hat  # return a_hat for testing

159  def _bootstrap_iv(data, statistic, vectorized, paired, axis, confidence_level,
160                    alternative, n_resamples, batch, method, bootstrap_result,
161                    rng):
162      """Input validation and standardization for `bootstrap`."""
163  
164      if vectorized not in {True, False, None}:
165          raise ValueError("`vectorized` must be `True`, `False`, or `None`.")
166  
167      if vectorized is None:
168          vectorized = 'axis' in inspect.signature(statistic).parameters
169  
170      if not vectorized:
171          statistic = _vectorize_statistic(statistic)
172  
173      axis_int = int(axis)
174      if axis != axis_int:
175          raise ValueError("`axis` must be an integer.")
176  
177      n_samples = 0
178      try:
179          n_samples = len(data)
180      except TypeError:
181          raise ValueError("`data` must be a sequence of samples.")
182  
183      if n_samples == 0:
184          raise ValueError("`data` must contain at least one sample.")
185  
186      message = ("Ignoring the dimension specified by `axis`, arrays in `data` do not "
187                 "have the same shape. Beginning in SciPy 1.16.0, `bootstrap` will "
188                 "explicitly broadcast elements of `data` to the same shape (ignoring "
189                 "`axis`) before performing the calculation. To avoid this warning in "
190                 "the meantime, ensure that all samples have the same shape (except "
191                 "potentially along `axis`).")
192      data = [np.atleast_1d(sample) for sample in data]
193      reduced_shapes = set()
194      for sample in data:
195          reduced_shape = list(sample.shape)
196          reduced_shape.pop(axis)
197          reduced_shapes.add(tuple(reduced_shape))
198      if len(reduced_shapes) != 1:
199          warnings.warn(message, FutureWarning, stacklevel=3)
200  
201      data_iv = []
202      for sample in data:
203          if sample.shape[axis_int] <= 1:
204              raise ValueError("each sample in `data` must contain two or more "
205                               "observations along `axis`.")
206          sample = np.moveaxis(sample, axis_int, -1)
207          data_iv.append(sample)
208  
209      if paired not in {True, False}:
210          raise ValueError("`paired` must be `True` or `False`.")
211  
212      if paired:
213          n = data_iv[0].shape[-1]
214          for sample in data_iv[1:]:
215              if sample.shape[-1] != n:
216                  message = ("When `paired is True`, all samples must have the "
217                             "same length along `axis`")
218                  raise ValueError(message)
219  
220          # to generate the bootstrap distribution for paired-sample statistics,
221          # resample the indices of the observations
222          def statistic(i, axis=-1, data=data_iv, unpaired_statistic=statistic):
223              data = [sample[..., i] for sample in data]
224              return unpaired_statistic(*data, axis=axis)
225  
226          data_iv = [np.arange(n)]
227  
228      confidence_level_float = float(confidence_level)
229  
230      alternative = alternative.lower()
231      alternatives = {'two-sided', 'less', 'greater'}
232      if alternative not in alternatives:
233          raise ValueError(f"`alternative` must be one of {alternatives}")
234  
235      n_resamples_int = int(n_resamples)
236      if n_resamples != n_resamples_int or n_resamples_int < 0:
237          raise ValueError("`n_resamples` must be a non-negative integer.")
238  
239      if batch is None:
240          batch_iv = batch
241      else:
242          batch_iv = int(batch)
243          if batch != batch_iv or batch_iv <= 0:
244              raise ValueError("`batch` must be a positive integer or None.")
245  
246      methods = {'percentile', 'basic', 'bca'}
247      method = method.lower()
248      if method not in methods:
249          raise ValueError(f"`method` must be in {methods}")
250  
251      message = "`bootstrap_result` must have attribute `bootstrap_distribution'"
252      if (bootstrap_result is not None
253              and not hasattr(bootstrap_result, "bootstrap_distribution")):
254          raise ValueError(message)
255  
256      message = ("Either `bootstrap_result.bootstrap_distribution.size` or "
257                 "`n_resamples` must be positive.")
258      if ((not bootstrap_result or
259           not bootstrap_result.bootstrap_distribution.size)
260              and n_resamples_int == 0):
261          raise ValueError(message)
262  
263      rng = check_random_state(rng)
264  
265      return (data_iv, statistic, vectorized, paired, axis_int,
266              confidence_level_float, alternative, n_resamples_int, batch_iv,
267              method, bootstrap_result, rng)

293  @_transition_to_rng('random_state')
294  def bootstrap(data, statistic, *, n_resamples=9999, batch=None,
295                vectorized=None, paired=False, axis=0, confidence_level=0.95,
296                alternative='two-sided', method='BCa', bootstrap_result=None,
297                rng=None):
298      r""" 
299      Compute a two-sided bootstrap confidence interval of a statistic. 
300   
301      When `method` is ``'percentile'`` and `alternative` is ``'two-sided'``, 
302      a bootstrap confidence interval is computed according to the following 
303      procedure. 
304   
305      1. Resample the data: for each sample in `data` and for each of 
306         `n_resamples`, take a random sample of the original sample 
307         (with replacement) of the same size as the original sample. 
308   
309      2. Compute the bootstrap distribution of the statistic: for each set of 
310         resamples, compute the test statistic. 
311   
312      3. Determine the confidence interval: find the interval of the bootstrap 
313         distribution that is 
314   
315         - symmetric about the median and 
316         - contains `confidence_level` of the resampled statistic values. 
317   
318      While the ``'percentile'`` method is the most intuitive, it is rarely 
319      used in practice. Two more common methods are available, ``'basic'`` 
320      ('reverse percentile') and ``'BCa'`` ('bias-corrected and accelerated'); 
321      they differ in how step 3 is performed. 
322   
323      If the samples in `data` are  taken at random from their respective 
324      distributions :math:`n` times, the confidence interval returned by 
325      `bootstrap` will contain the true value of the statistic for those 
326      distributions approximately `confidence_level`:math:`\, \times \, n` times. 
327   
328      Parameters 
329      ---------- 
330      data : sequence of array-like 
331           Each element of `data` is a sample containing scalar observations from an 
332           underlying distribution. Elements of `data` must be broadcastable to the 
333           same shape (with the possible exception of the dimension specified by `axis`). 
334   
335           .. versionchanged:: 1.14.0 
336               `bootstrap` will now emit a ``FutureWarning`` if the shapes of the 
337               elements of `data` are not the same (with the exception of the dimension 
338               specified by `axis`). 
339               Beginning in SciPy 1.16.0, `bootstrap` will explicitly broadcast the 
340               elements to the same shape (except along `axis`) before performing 
341               the calculation. 
342   
343      statistic : callable 
344          Statistic for which the confidence interval is to be calculated. 
345          `statistic` must be a callable that accepts ``len(data)`` samples 
346          as separate arguments and returns the resulting statistic. 
347          If `vectorized` is set ``True``, 
348          `statistic` must also accept a keyword argument `axis` and be 
349          vectorized to compute the statistic along the provided `axis`. 
350      n_resamples : int, default: ``9999`` 
351          The number of resamples performed to form the bootstrap distribution 
352          of the statistic. 
353      batch : int, optional 
354          The number of resamples to process in each vectorized call to 
355          `statistic`. Memory usage is O( `batch` * ``n`` ), where ``n`` is the 
356          sample size. Default is ``None``, in which case ``batch = n_resamples`` 
357          (or ``batch = max(n_resamples, n)`` for ``method='BCa'``). 
358      vectorized : bool, optional 
359          If `vectorized` is set ``False``, `statistic` will not be passed 
360          keyword argument `axis` and is expected to calculate the statistic 
361          only for 1D samples. If ``True``, `statistic` will be passed keyword 
362          argument `axis` and is expected to calculate the statistic along `axis` 
363          when passed an ND sample array. If ``None`` (default), `vectorized` 
364          will be set ``True`` if ``axis`` is a parameter of `statistic`. Use of 
365          a vectorized statistic typically reduces computation time. 
366      paired : bool, default: ``False`` 
367          Whether the statistic treats corresponding elements of the samples 
368          in `data` as paired. If True, `bootstrap` resamples an array of 
369          *indices* and uses the same indices for all arrays in `data`; otherwise, 
370          `bootstrap` independently resamples the elements of each array. 
371      axis : int, default: ``0`` 
372          The axis of the samples in `data` along which the `statistic` is 
373          calculated. 
374      confidence_level : float, default: ``0.95`` 
375          The confidence level of the confidence interval. 
376      alternative : {'two-sided', 'less', 'greater'}, default: ``'two-sided'`` 
377          Choose ``'two-sided'`` (default) for a two-sided confidence interval, 
378          ``'less'`` for a one-sided confidence interval with the lower bound 
379          at ``-np.inf``, and ``'greater'`` for a one-sided confidence interval 
380          with the upper bound at ``np.inf``. The other bound of the one-sided 
381          confidence intervals is the same as that of a two-sided confidence 
382          interval with `confidence_level` twice as far from 1.0; e.g. the upper 
383          bound of a 95% ``'less'``  confidence interval is the same as the upper 
384          bound of a 90% ``'two-sided'`` confidence interval. 
385      method : {'percentile', 'basic', 'bca'}, default: ``'BCa'`` 
386          Whether to return the 'percentile' bootstrap confidence interval 
387          (``'percentile'``), the 'basic' (AKA 'reverse') bootstrap confidence 
388          interval (``'basic'``), or the bias-corrected and accelerated bootstrap 
389          confidence interval (``'BCa'``). 
390      bootstrap_result : BootstrapResult, optional 
391          Provide the result object returned by a previous call to `bootstrap` 
392          to include the previous bootstrap distribution in the new bootstrap 
393          distribution. This can be used, for example, to change 
394          `confidence_level`, change `method`, or see the effect of performing 
395          additional resampling without repeating computations. 
396      rng : `numpy.random.Generator`, optional 
397          Pseudorandom number generator state. When `rng` is None, a new 
398          `numpy.random.Generator` is created using entropy from the 
399          operating system. Types other than `numpy.random.Generator` are 
400          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
401   
402      Returns 
403      ------- 
404      res : BootstrapResult 
405          An object with attributes: 
406   
407          confidence_interval : ConfidenceInterval 
408              The bootstrap confidence interval as an instance of 
409              `collections.namedtuple` with attributes `low` and `high`. 
410          bootstrap_distribution : ndarray 
411              The bootstrap distribution, that is, the value of `statistic` for 
412              each resample. The last dimension corresponds with the resamples 
413              (e.g. ``res.bootstrap_distribution.shape[-1] == n_resamples``). 
414          standard_error : float or ndarray 
415              The bootstrap standard error, that is, the sample standard 
416              deviation of the bootstrap distribution. 
417   
418      Warns 
419      ----- 
420      `~scipy.stats.DegenerateDataWarning` 
421          Generated when ``method='BCa'`` and the bootstrap distribution is 
422          degenerate (e.g. all elements are identical). 
423   
424      Notes 
425      ----- 
426      Elements of the confidence interval may be NaN for ``method='BCa'`` if 
427      the bootstrap distribution is degenerate (e.g. all elements are identical). 
428      In this case, consider using another `method` or inspecting `data` for 
429      indications that other analysis may be more appropriate (e.g. all 
430      observations are identical). 
431   
432      References 
433      ---------- 
434      .. [1] B. Efron and R. J. Tibshirani, An Introduction to the Bootstrap, 
435         Chapman & Hall/CRC, Boca Raton, FL, USA (1993) 
436      .. [2] Nathaniel E. Helwig, "Bootstrap Confidence Intervals", 
437         http://users.stat.umn.edu/~helwig/notes/bootci-Notes.pdf 
438      .. [3] Bootstrapping (statistics), Wikipedia, 
439         https://en.wikipedia.org/wiki/Bootstrapping_%28statistics%29 
440   
441      Examples 
442      -------- 
443      Suppose we have sampled data from an unknown distribution. 
444   
445      >>> import numpy as np 
446      >>> rng = np.random.default_rng() 
447      >>> from scipy.stats import norm 
448      >>> dist = norm(loc=2, scale=4)  # our "unknown" distribution 
449      >>> data = dist.rvs(size=100, random_state=rng) 
450   
451      We are interested in the standard deviation of the distribution. 
452   
453      >>> std_true = dist.std()      # the true value of the statistic 
454      >>> print(std_true) 
455      4.0 
456      >>> std_sample = np.std(data)  # the sample statistic 
457      >>> print(std_sample) 
458      3.9460644295563863 
459   
460      The bootstrap is used to approximate the variability we would expect if we 
461      were to repeatedly sample from the unknown distribution and calculate the 
462      statistic of the sample each time. It does this by repeatedly resampling 
463      values *from the original sample* with replacement and calculating the 
464      statistic of each resample. This results in a "bootstrap distribution" of 
465      the statistic. 
466   
467      >>> import matplotlib.pyplot as plt 
468      >>> from scipy.stats import bootstrap 
469      >>> data = (data,)  # samples must be in a sequence 
470      >>> res = bootstrap(data, np.std, confidence_level=0.9, rng=rng) 
471      >>> fig, ax = plt.subplots() 
472      >>> ax.hist(res.bootstrap_distribution, bins=25) 
473      >>> ax.set_title('Bootstrap Distribution') 
474      >>> ax.set_xlabel('statistic value') 
475      >>> ax.set_ylabel('frequency') 
476      >>> plt.show() 
477   
478      The standard error quantifies this variability. It is calculated as the 
479      standard deviation of the bootstrap distribution. 
480   
481      >>> res.standard_error 
482      0.24427002125829136 
483      >>> res.standard_error == np.std(res.bootstrap_distribution, ddof=1) 
484      True 
485   
486      The bootstrap distribution of the statistic is often approximately normal 
487      with scale equal to the standard error. 
488   
489      >>> x = np.linspace(3, 5) 
490      >>> pdf = norm.pdf(x, loc=std_sample, scale=res.standard_error) 
491      >>> fig, ax = plt.subplots() 
492      >>> ax.hist(res.bootstrap_distribution, bins=25, density=True) 
493      >>> ax.plot(x, pdf) 
494      >>> ax.set_title('Normal Approximation of the Bootstrap Distribution') 
495      >>> ax.set_xlabel('statistic value') 
496      >>> ax.set_ylabel('pdf') 
497      >>> plt.show() 
498   
499      This suggests that we could construct a 90% confidence interval on the 
500      statistic based on quantiles of this normal distribution. 
501   
502      >>> norm.interval(0.9, loc=std_sample, scale=res.standard_error) 
503      (3.5442759991341726, 4.3478528599786) 
504   
505      Due to central limit theorem, this normal approximation is accurate for a 
506      variety of statistics and distributions underlying the samples; however, 
507      the approximation is not reliable in all cases. Because `bootstrap` is 
508      designed to work with arbitrary underlying distributions and statistics, 
509      it uses more advanced techniques to generate an accurate confidence 
510      interval. 
511   
512      >>> print(res.confidence_interval) 
513      ConfidenceInterval(low=3.57655333533867, high=4.382043696342881) 
514   
515      If we sample from the original distribution 100 times and form a bootstrap 
516      confidence interval for each sample, the confidence interval 
517      contains the true value of the statistic approximately 90% of the time. 
518   
519      >>> n_trials = 100 
520      >>> ci_contains_true_std = 0 
521      >>> for i in range(n_trials): 
522      ...    data = (dist.rvs(size=100, random_state=rng),) 
523      ...    res = bootstrap(data, np.std, confidence_level=0.9, 
524      ...                    n_resamples=999, rng=rng) 
525      ...    ci = res.confidence_interval 
526      ...    if ci[0] < std_true < ci[1]: 
527      ...        ci_contains_true_std += 1 
528      >>> print(ci_contains_true_std) 
529      88 
530   
531      Rather than writing a loop, we can also determine the confidence intervals 
532      for all 100 samples at once. 
533   
534      >>> data = (dist.rvs(size=(n_trials, 100), random_state=rng),) 
535      >>> res = bootstrap(data, np.std, axis=-1, confidence_level=0.9, 
536      ...                 n_resamples=999, rng=rng) 
537      >>> ci_l, ci_u = res.confidence_interval 
538   
539      Here, `ci_l` and `ci_u` contain the confidence interval for each of the 
540      ``n_trials = 100`` samples. 
541   
542      >>> print(ci_l[:5]) 
543      [3.86401283 3.33304394 3.52474647 3.54160981 3.80569252] 
544      >>> print(ci_u[:5]) 
545      [4.80217409 4.18143252 4.39734707 4.37549713 4.72843584] 
546   
547      And again, approximately 90% contain the true value, ``std_true = 4``. 
548   
549      >>> print(np.sum((ci_l < std_true) & (std_true < ci_u))) 
550      93 
551   
552      `bootstrap` can also be used to estimate confidence intervals of 
553      multi-sample statistics. For example, to get a confidence interval 
554      for the difference between means, we write a function that accepts 
555      two sample arguments and returns only the statistic. The use of the 
556      ``axis`` argument ensures that all mean calculations are perform in 
557      a single vectorized call, which is faster than looping over pairs 
558      of resamples in Python. 
559   
560      >>> def my_statistic(sample1, sample2, axis=-1): 
561      ...     mean1 = np.mean(sample1, axis=axis) 
562      ...     mean2 = np.mean(sample2, axis=axis) 
563      ...     return mean1 - mean2 
564   
565      Here, we use the 'percentile' method with the default 95% confidence level. 
566   
567      >>> sample1 = norm.rvs(scale=1, size=100, random_state=rng)

785  def monte_carlo_test(data, rvs, statistic, *, vectorized=None,
786                       n_resamples=9999, batch=None, alternative="two-sided",
787                       axis=0):
788      r"""Perform a Monte Carlo hypothesis test. 
789   
790      `data` contains a sample or a sequence of one or more samples. `rvs` 
791      specifies the distribution(s) of the sample(s) in `data` under the null 
792      hypothesis. The value of `statistic` for the given `data` is compared 
793      against a Monte Carlo null distribution: the value of the statistic for 
794      each of `n_resamples` sets of samples generated using `rvs`. This gives 
795      the p-value, the probability of observing such an extreme value of the 
796      test statistic under the null hypothesis. 
797   
798      Parameters 
799      ---------- 
800      data : array-like or sequence of array-like 
801          An array or sequence of arrays of observations. 
802      rvs : callable or tuple of callables 
803          A callable or sequence of callables that generates random variates 
804          under the null hypothesis. Each element of `rvs` must be a callable 
805          that accepts keyword argument ``size`` (e.g. ``rvs(size=(m, n))``) and 
806          returns an N-d array sample of that shape. If `rvs` is a sequence, the 
807          number of callables in `rvs` must match the number of samples in 
808          `data`, i.e. ``len(rvs) == len(data)``. If `rvs` is a single callable, 
809          `data` is treated as a single sample. 
810      statistic : callable 
811          Statistic for which the p-value of the hypothesis test is to be 
812          calculated. `statistic` must be a callable that accepts a sample 
813          (e.g. ``statistic(sample)``) or ``len(rvs)`` separate samples (e.g. 
814          ``statistic(samples1, sample2)`` if `rvs` contains two callables and 
815          `data` contains two samples) and returns the resulting statistic. 
816          If `vectorized` is set ``True``, `statistic` must also accept a keyword 
817          argument `axis` and be vectorized to compute the statistic along the 
818          provided `axis` of the samples in `data`. 
819      vectorized : bool, optional 
820          If `vectorized` is set ``False``, `statistic` will not be passed 
821          keyword argument `axis` and is expected to calculate the statistic 
822          only for 1D samples. If ``True``, `statistic` will be passed keyword 
823          argument `axis` and is expected to calculate the statistic along `axis` 
824          when passed ND sample arrays. If ``None`` (default), `vectorized` 
825          will be set ``True`` if ``axis`` is a parameter of `statistic`. Use of 
826          a vectorized statistic typically reduces computation time. 
827      n_resamples : int, default: 9999 
828          Number of samples drawn from each of the callables of `rvs`. 
829          Equivalently, the number statistic values under the null hypothesis 
830          used as the Monte Carlo null distribution. 
831      batch : int, optional 
832          The number of Monte Carlo samples to process in each call to 
833          `statistic`. Memory usage is O( `batch` * ``sample.size[axis]`` ). Default 
834          is ``None``, in which case `batch` equals `n_resamples`. 
835      alternative : {'two-sided', 'less', 'greater'} 
836          The alternative hypothesis for which the p-value is calculated. 
837          For each alternative, the p-value is defined as follows. 
838   
839          - ``'greater'`` : the percentage of the null distribution that is 
840            greater than or equal to the observed value of the test statistic. 
841          - ``'less'`` : the percentage of the null distribution that is 
842            less than or equal to the observed value of the test statistic. 
843          - ``'two-sided'`` : twice the smaller of the p-values above. 
844   
845      axis : int, default: 0 
846          The axis of `data` (or each sample within `data`) over which to 
847          calculate the statistic. 
848   
849      Returns 
850      ------- 
851      res : MonteCarloTestResult 
852          An object with attributes: 
853   
854          statistic : float or ndarray 
855              The test statistic of the observed `data`. 
856          pvalue : float or ndarray 
857              The p-value for the given alternative. 
858          null_distribution : ndarray 
859              The values of the test statistic generated under the null 
860              hypothesis. 
861   
862      .. warning:: 
863          The p-value is calculated by counting the elements of the null 
864          distribution that are as extreme or more extreme than the observed 
865          value of the statistic. Due to the use of finite precision arithmetic, 
866          some statistic functions return numerically distinct values when the 
867          theoretical values would be exactly equal. In some cases, this could 
868          lead to a large error in the calculated p-value. `monte_carlo_test` 
869          guards against this by considering elements in the null distribution 
870          that are "close" (within a relative tolerance of 100 times the 
871          floating point epsilon of inexact dtypes) to the observed 
872          value of the test statistic as equal to the observed value of the 
873          test statistic. However, the user is advised to inspect the null 
874          distribution to assess whether this method of comparison is 
875          appropriate, and if not, calculate the p-value manually. 
876   
877      References 
878      ---------- 
879   
880      .. [1] B. Phipson and G. K. Smyth. "Permutation P-values Should Never Be 
881         Zero: Calculating Exact P-values When Permutations Are Randomly Drawn." 
882         Statistical Applications in Genetics and Molecular Biology 9.1 (2010). 
883   
884      Examples 
885      -------- 
886   
887      Suppose we wish to test whether a small sample has been drawn from a normal 
888      distribution. We decide that we will use the skew of the sample as a 
889      test statistic, and we will consider a p-value of 0.05 to be statistically 
890      significant. 
891   
892      >>> import numpy as np 
893      >>> from scipy import stats 
894      >>> def statistic(x, axis): 
895      ...     return stats.skew(x, axis) 
896   
897      After collecting our data, we calculate the observed value of the test 
898      statistic. 
899   
900      >>> rng = np.random.default_rng() 
901      >>> x = stats.skewnorm.rvs(a=1, size=50, random_state=rng) 
902      >>> statistic(x, axis=0) 
903      0.12457412450240658 
904   
905      To determine the probability of observing such an extreme value of the 
906      skewness by chance if the sample were drawn from the normal distribution, 
907      we can perform a Monte Carlo hypothesis test. The test will draw many 
908      samples at random from their normal distribution, calculate the skewness 
909      of each sample, and compare our original skewness against this 
910      distribution to determine an approximate p-value. 
911   
912      >>> from scipy.stats import monte_carlo_test 
913      >>> # because our statistic is vectorized, we pass `vectorized=True` 
914      >>> rvs = lambda size: stats.norm.rvs(size=size, random_state=rng) 
915      >>> res = monte_carlo_test(x, rvs, statistic, vectorized=True) 
916      >>> print(res.statistic) 
917      0.12457412450240658 
918      >>> print(res.pvalue) 
919      0.7012 
920   
921      The probability of obtaining a test statistic less than or equal to the 
922      observed value under the null hypothesis is ~70%. This is greater than 
923      our chosen threshold of 5%, so we cannot consider this to be significant 
924      evidence against the null hypothesis. 
925   
926      Note that this p-value essentially matches that of 
927      `scipy.stats.skewtest`, which relies on an asymptotic distribution of a 
928      test statistic based on the sample skewness. 
929   
930      >>> stats.skewtest(x).pvalue 
931      0.6892046027110614 
932   
933      This asymptotic approximation is not valid for small sample sizes, but 
934      `monte_carlo_test` can be used with samples of any size. 
935   
936      >>> x = stats.skewnorm.rvs(a=1, size=7, random_state=rng) 
937      >>> # stats.skewtest(x) would produce an error due to small sample 
938      >>> res = monte_carlo_test(x, rvs, statistic, vectorized=True) 
939   
940      The Monte Carlo distribution of the test statistic is provided for 
941      further investigation. 
942   
943      >>> import matplotlib.pyplot as plt 
944      >>> fig, ax = plt.subplots() 
945      >>> ax.hist(res.null_distribution, bins=50) 
946      >>> ax.set_title("Monte Carlo distribution of test statistic") 
947      >>> ax.set_xlabel("Value of Statistic") 
948      >>> ax.set_ylabel("Frequency") 
949      >>> plt.show() 
950   
951      """
952      args = _monte_carlo_test_iv(data, rvs, statistic, vectorized,
953                                  n_resamples, batch, alternative, axis)
954      (data, rvs, statistic, vectorized, n_resamples,
955       batch, alternative, axis, dtype, xp) = args
956  
957      # Some statistics return plain floats; ensure they're at least a NumPy float
958      observed = xp.asarray(statistic(*data, axis=-1))
959      observed = observed[()] if observed.ndim == 0 else observed
960  
961      n_observations = [sample.shape[-1] for sample in data]
962      batch_nominal = batch or n_resamples
963      null_distribution = []
964      for k in range(0, n_resamples, batch_nominal):
965          batch_actual = min(batch_nominal, n_resamples - k)
966          resamples = [rvs_i(size=(batch_actual, n_observations_i))
967                       for rvs_i, n_observations_i in zip(rvs, n_observations)]
968          null_distribution.append(statistic(*resamples, axis=-1))
969      null_distribution = xp.concat(null_distribution)
970      null_distribution = xp.reshape(null_distribution, [-1] + [1]*observed.ndim)
971  
972      # relative tolerance for detecting numerically distinct but
973      # theoretically equal values in the null distribution
974      eps =  (0 if not xp.isdtype(observed.dtype, ('real floating'))
975              else xp.finfo(observed.dtype).eps*100)
976      gamma = xp.abs(eps * observed)
977  
978      def less(null_distribution, observed):
979          cmps = null_distribution <= observed + gamma
980          cmps = xp.asarray(cmps, dtype=dtype)
981          pvalues = (xp.sum(cmps, axis=0, dtype=dtype) + 1.) / (n_resamples + 1.)
982          return pvalues
983  
984      def greater(null_distribution, observed):
985          cmps = null_distribution >= observed - gamma
986          cmps = xp.asarray(cmps, dtype=dtype)
987          pvalues = (xp.sum(cmps, axis=0, dtype=dtype) + 1.) / (n_resamples + 1.)
988          return pvalues

991          pvalues_less = less(null_distribution, observed)
992          pvalues_greater = greater(null_distribution, observed)
993          pvalues = xp.minimum(pvalues_less, pvalues_greater) * 2
994          return pvalues
995  
996      compare = {"less": less,
997                 "greater": greater,
998                 "two-sided": two_sided}
999  
1000     pvalues = compare[alternative](null_distribution, observed)
1001     pvalues = xp.clip(pvalues, 0., 1.)
1002

1358 def _all_partitions_concatenated(ns):
1359     """ 
1360     Generate all partitions of indices of groups of given sizes, concatenated 
1361  
1362     `ns` is an iterable of ints. 
1363     """
1364     def all_partitions(z, n):
1365         for c in combinations(z, n):
1366             x0 = set(c)
1367             x1 = z - x0
1368             yield [x0, x1]
1369 
1370     def all_partitions_n(z, ns):
1371         if len(ns) == 0:
1372             yield [z]
1373             return
1374         for c in all_partitions(z, ns[0]):
1375             for d in all_partitions_n(c[1], ns[1:]):
1376                 yield c[0:1] + d
1377 
1378     z = set(range(np.sum(ns)))
1379     for partitioning in all_partitions_n(z, ns[:]):
1380         x = np.concatenate([list(partition)
1381                             for partition in partitioning]).astype(int)
1382         yield x

1396 def _pairings_permutations_gen(n_permutations, n_samples, n_obs_sample, batch,
1397                                rng):
1398     # Returns a generator that yields arrays of size
1399     # `(batch, n_samples, n_obs_sample)`.
1400     # Each row is an independent permutation of indices 0 to `n_obs_sample`.
1401     batch = min(batch, n_permutations)
1402 
1403     if hasattr(rng, 'permuted'):
1404         def batched_perm_generator():
1405             indices = np.arange(n_obs_sample)
1406             indices = np.tile(indices, (batch, n_samples, 1))
1407             for k in range(0, n_permutations, batch):
1408                 batch_actual = min(batch, n_permutations-k)
1409                 # Don't permute in place, otherwise results depend on `batch`
1410                 permuted_indices = rng.permuted(indices, axis=-1)
1411                 yield permuted_indices[:batch_actual]
1412     else:  # RandomState and early Generators don't have `permuted`
1413         def batched_perm_generator():
1414             for k in range(0, n_permutations, batch):
1415                 batch_actual = min(batch, n_permutations-k)
1416                 size = (batch_actual, n_samples, n_obs_sample)
1417                 x = rng.random(size=size)
1418                 yield np.argsort(x, axis=-1)[:batch_actual]
1419

1423 def _calculate_null_both(data, statistic, n_permutations, batch,
1424                          rng=None):
1425     """ 
1426     Calculate null distribution for independent sample tests. 
1427     """
1428     n_samples = len(data)
1429 
1430     # compute number of permutations
1431     # (distinct partitions of data into samples of these sizes)
1432     n_obs_i = [sample.shape[-1] for sample in data]  # observations per sample
1433     n_obs_ic = np.cumsum(n_obs_i)
1434     n_obs = n_obs_ic[-1]  # total number of observations
1435     n_max = np.prod([comb(n_obs_ic[i], n_obs_ic[i-1])
1436                      for i in range(n_samples-1, 0, -1)])
1437 
1438     # perm_generator is an iterator that produces permutations of indices
1439     # from 0 to n_obs. We'll concatenate the samples, use these indices to
1440     # permute the data, then split the samples apart again.
1441     if n_permutations >= n_max:
1442         exact_test = True
1443         n_permutations = n_max
1444         perm_generator = _all_partitions_concatenated(n_obs_i)
1445     else:
1446         exact_test = False
1447         # Neither RandomState.permutation nor Generator.permutation
1448         # can permute axis-slices independently. If this feature is
1449         # added in the future, batches of the desired size should be
1450         # generated in a single call.
1451         perm_generator = (rng.permutation(n_obs)
1452                           for i in range(n_permutations))
1453 
1454     batch = batch or int(n_permutations)
1455     null_distribution = []
1456 
1457     # First, concatenate all the samples. In batches, permute samples with
1458     # indices produced by the `perm_generator`, split them into new samples of
1459     # the original sizes, compute the statistic for each batch, and add these
1460     # statistic values to the null distribution.
1461     data = np.concatenate(data, axis=-1)
1462     for indices in _batch_generator(perm_generator, batch=batch):
1463         indices = np.array(indices)
1464 
1465         # `indices` is 2D: each row is a permutation of the indices.
1466         # We use it to index `data` along its last axis, which corresponds
1467         # with observations.
1468         # After indexing, the second to last axis of `data_batch` corresponds
1469         # with permutations, and the last axis corresponds with observations.
1470         data_batch = data[..., indices]
1471 
1472         # Move the permutation axis to the front: we'll concatenate a list
1473         # of batched statistic values along this zeroth axis to form the
1474         # null distribution.
1475         data_batch = np.moveaxis(data_batch, -2, 0)
1476         data_batch = np.split(data_batch, n_obs_ic[:-1], axis=-1)
1477         null_distribution.append(statistic(*data_batch, axis=-1))
1478     null_distribution = np.concatenate(null_distribution, axis=0)
1479 
1480     return null_distribution, n_permutations, exact_test

1483 def _calculate_null_pairings(data, statistic, n_permutations, batch,
1484                              rng=None):
1485     """ 
1486     Calculate null distribution for association tests. 
1487     """
1488     n_samples = len(data)
1489 
1490     # compute number of permutations (factorial(n) permutations of each sample)
1491     n_obs_sample = data[0].shape[-1]  # observations per sample; same for each
1492     n_max = factorial(n_obs_sample)**n_samples
1493 
1494     # `perm_generator` is an iterator that produces a list of permutations of
1495     # indices from 0 to n_obs_sample, one for each sample.
1496     if n_permutations >= n_max:
1497         exact_test = True
1498         n_permutations = n_max
1499         batch = batch or int(n_permutations)
1500         # Cartesian product of the sets of all permutations of indices
1501         perm_generator = product(*(permutations(range(n_obs_sample))
1502                                    for i in range(n_samples)))
1503         batched_perm_generator = _batch_generator(perm_generator, batch=batch)
1504     else:
1505         exact_test = False
1506         batch = batch or int(n_permutations)
1507         # Separate random permutations of indices for each sample.
1508         # Again, it would be nice if RandomState/Generator.permutation
1509         # could permute each axis-slice separately.
1510         args = n_permutations, n_samples, n_obs_sample, batch, rng
1511         batched_perm_generator = _pairings_permutations_gen(*args)
1512 
1513     null_distribution = []
1514 
1515     for indices in batched_perm_generator:
1516         indices = np.array(indices)
1517 
1518         # `indices` is 3D: the zeroth axis is for permutations, the next is
1519         # for samples, and the last is for observations. Swap the first two
1520         # to make the zeroth axis correspond with samples, as it does for
1521         # `data`.
1522         indices = np.swapaxes(indices, 0, 1)
1523 
1524         # When we're done, `data_batch` will be a list of length `n_samples`.
1525         # Each element will be a batch of random permutations of one sample.
1526         # The zeroth axis of each batch will correspond with permutations,
1527         # and the last will correspond with observations. (This makes it
1528         # easy to pass into `statistic`.)
1529         data_batch = [None]*n_samples
1530         for i in range(n_samples):
1531             data_batch[i] = data[i][..., indices[i]]
1532             data_batch[i] = np.moveaxis(data_batch[i], -2, 0)
1533 
1534         null_distribution.append(statistic(*data_batch, axis=-1))
1535     null_distribution = np.concatenate(null_distribution, axis=0)
1536 
1537     return null_distribution, n_permutations, exact_test

1664         observations and observations with corresponding indices along `axis` 
1665         are considered to be paired; the third is for independent sample 
1666         statistics. 
1667  
1668         - ``'samples'`` : observations are assigned to different samples 
1669           but remain paired with the same observations from other samples. 
1670           This permutation type is appropriate for paired sample hypothesis 
1671           tests such as the Wilcoxon signed-rank test and the paired t-test. 
1672         - ``'pairings'`` : observations are paired with different observations, 
1673           but they remain within the same sample. This permutation type is 
1674           appropriate for association/correlation tests with statistics such 
1675           as Spearman's :math:`\rho`, Kendall's :math:`\tau`, and Pearson's 
1676           :math:`r`. 
1677         - ``'independent'`` (default) : observations are assigned to different 
1678           samples. Samples may contain different numbers of observations. This 
1679           permutation type is appropriate for independent sample hypothesis 
1680           tests such as the Mann-Whitney :math:`U` test and the independent 
1681           sample t-test. 
1682  
1683           Please see the Notes section below for more detailed descriptions 
1684           of the permutation types. 
1685  
1686     vectorized : bool, optional 
1687         If `vectorized` is set ``False``, `statistic` will not be passed 
1688         keyword argument `axis` and is expected to calculate the statistic 
1689         only for 1D samples. If ``True``, `statistic` will be passed keyword 
1690         argument `axis` and is expected to calculate the statistic along `axis` 
1691         when passed an ND sample array. If ``None`` (default), `vectorized` 
1692         will be set ``True`` if ``axis`` is a parameter of `statistic`. Use 
1693         of a vectorized statistic typically reduces computation time. 
1694     n_resamples : int or np.inf, default: 9999 
1695         Number of random permutations (resamples) used to approximate the null 
1696         distribution. If greater than or equal to the number of distinct 
1697         permutations, the exact null distribution will be computed. 
1698         Note that the number of distinct permutations grows very rapidly with 
1699         the sizes of samples, so exact tests are feasible only for very small 
1700         data sets. 
1701     batch : int, optional 
1702         The number of permutations to process in each call to `statistic`. 
1703         Memory usage is O( `batch` * ``n`` ), where ``n`` is the total size 
1704         of all samples, regardless of the value of `vectorized`. Default is 
1705         ``None``, in which case ``batch`` is the number of permutations. 
1706     alternative : {'two-sided', 'less', 'greater'}, optional 
1707         The alternative hypothesis for which the p-value is calculated. 
1708         For each alternative, the p-value is defined for exact tests as 
1709         follows. 
1710  
1711         - ``'greater'`` : the percentage of the null distribution that is 
1712           greater than or equal to the observed value of the test statistic. 
1713         - ``'less'`` : the percentage of the null distribution that is 
1714           less than or equal to the observed value of the test statistic. 
1715         - ``'two-sided'`` (default) : twice the smaller of the p-values above. 
1716  
1717         Note that p-values for randomized tests are calculated according to the 
1718         conservative (over-estimated) approximation suggested in [2]_ and [3]_ 
1719         rather than the unbiased estimator suggested in [4]_. That is, when 
1720         calculating the proportion of the randomized null distribution that is 
1721         as extreme as the observed value of the test statistic, the values in 
1722         the numerator and denominator are both increased by one. An 
1723         interpretation of this adjustment is that the observed value of the 
1724         test statistic is always included as an element of the randomized 
1725         null distribution. 
1726         The convention used for two-sided p-values is not universal; 
1727         the observed test statistic and null distribution are returned in 
1728         case a different definition is preferred. 
1729  
1730     axis : int, default: 0 
1731         The axis of the (broadcasted) samples over which to calculate the 
1732         statistic. If samples have a different number of dimensions, 
1733         singleton dimensions are prepended to samples with fewer dimensions 
1734         before `axis` is considered. 
1735     rng : `numpy.random.Generator`, optional 
1736         Pseudorandom number generator state. When `rng` is None, a new 
1737         `numpy.random.Generator` is created using entropy from the 
1738         operating system. Types other than `numpy.random.Generator` are 
1739         passed to `numpy.random.default_rng` to instantiate a ``Generator``.

1749             The p-value for the given alternative.

1754     Notes 
1755     ----- 
1756  
1757     The three types of permutation tests supported by this function are 
1758     described below. 
1759  
1760     **Unpaired statistics** (``permutation_type='independent'``): 
1761  
1762     The null hypothesis associated with this permutation type is that all 
1763     observations are sampled from the same underlying distribution and that 
1764     they have been assigned to one of the samples at random. 
1765  
1766     Suppose ``data`` contains two samples; e.g. ``a, b = data``. 
1767     When ``1 < n_resamples < binom(n, k)``, where 
1768  
1769     * ``k`` is the number of observations in ``a``, 
1770     * ``n`` is the total number of observations in ``a`` and ``b``, and 
1771     * ``binom(n, k)`` is the binomial coefficient (``n`` choose ``k``), 
1772  
1773     the data are pooled (concatenated), randomly assigned to either the first 
1774     or second sample, and the statistic is calculated. This process is 
1775     performed repeatedly, `permutation` times, generating a distribution of the 
1776     statistic under the null hypothesis. The statistic of the original 
1777     data is compared to this distribution to determine the p-value. 
1778  
1779     When ``n_resamples >= binom(n, k)``, an exact test is performed: the data 
1780     are *partitioned* between the samples in each distinct way exactly once, 
1781     and the exact null distribution is formed. 
1782     Note that for a given partitioning of the data between the samples, 
1783     only one ordering/permutation of the data *within* each sample is 
1784     considered. For statistics that do not depend on the order of the data 
1785     within samples, this dramatically reduces computational cost without 
1786     affecting the shape of the null distribution (because the frequency/count 
1787     of each value is affected by the same factor). 
1788  
1789     For ``a = [a1, a2, a3, a4]`` and ``b = [b1, b2, b3]``, an example of this 
1790     permutation type is ``x = [b3, a1, a2, b2]`` and ``y = [a4, b1, a3]``. 
1791     Because only one ordering/permutation of the data *within* each sample 
1792     is considered in an exact test, a resampling like ``x = [b3, a1, b2, a2]`` 
1793     and ``y = [a4, a3, b1]`` would *not* be considered distinct from the 
1794     example above. 
1795  
1796     ``permutation_type='independent'`` does not support one-sample statistics, 
1797     but it can be applied to statistics with more than two samples. In this 
1798     case, if ``n`` is an array of the number of observations within each 
1799     sample, the number of distinct partitions is:: 
1800  
1801         np.prod([binom(sum(n[i:]), sum(n[i+1:])) for i in range(len(n)-1)]) 
1802  
1803     **Paired statistics, permute pairings** (``permutation_type='pairings'``): 
1804  
1805     The null hypothesis associated with this permutation type is that 
1806     observations within each sample are drawn from the same underlying 
1807     distribution and that pairings with elements of other samples are 
1808     assigned at random. 
1809  
1810     Suppose ``data`` contains only one sample; e.g. ``a, = data``, and we 
1811     wish to consider all possible pairings of elements of ``a`` with elements 
1812     of a second sample, ``b``. Let ``n`` be the number of observations in 
1813     ``a``, which must also equal the number of observations in ``b``. 
1814  
1815     When ``1 < n_resamples < factorial(n)``, the elements of ``a`` are 
1816     randomly permuted. The user-supplied statistic accepts one data argument, 
1817     say ``a_perm``, and calculates the statistic considering ``a_perm`` and 
1818     ``b``. This process is performed repeatedly, `permutation` times, 
1819     generating a distribution of the statistic under the null hypothesis. 
1820     The statistic of the original data is compared to this distribution to 
1821     determine the p-value. 
1822  
1823     When ``n_resamples >= factorial(n)``, an exact test is performed: 
1824     ``a`` is permuted in each distinct way exactly once. Therefore, the 
1825     `statistic` is computed for each unique pairing of samples between ``a`` 
1826     and ``b`` exactly once. 
1827  
1828     For ``a = [a1, a2, a3]`` and ``b = [b1, b2, b3]``, an example of this 
1829     permutation type is ``a_perm = [a3, a1, a2]`` while ``b`` is left 
1830     in its original order. 
1831  
1832     ``permutation_type='pairings'`` supports ``data`` containing any number 
1833     of samples, each of which must contain the same number of observations. 
1834     All samples provided in ``data`` are permuted *independently*. Therefore, 
1835     if ``m`` is the number of samples and ``n`` is the number of observations 
1836     within each sample, then the number of permutations in an exact test is:: 
1837  
1838         factorial(n)**m 
1839  
1840     Note that if a two-sample statistic, for example, does not inherently 
1841     depend on the order in which observations are provided - only on the 
1842     *pairings* of observations - then only one of the two samples should be 
1843     provided in ``data``. This dramatically reduces computational cost without 
1844     affecting the shape of the null distribution (because the frequency/count 
1845     of each value is affected by the same factor). 
1846  
1847     **Paired statistics, permute samples** (``permutation_type='samples'``): 
1848  
1849     The null hypothesis associated with this permutation type is that 
1850     observations within each pair are drawn from the same underlying 
1851     distribution and that the sample to which they are assigned is random. 
1852  
1853     Suppose ``data`` contains two samples; e.g. ``a, b = data``. 
1854     Let ``n`` be the number of observations in ``a``, which must also equal 
1855     the number of observations in ``b``. 
1856  
1857     When ``1 < n_resamples < 2**n``, the elements of ``a`` are ``b`` are 
1858     randomly swapped between samples (maintaining their pairings) and the 
1859     statistic is calculated. This process is performed repeatedly, 
1860     `permutation` times,  generating a distribution of the statistic under the 
1861     null hypothesis. The statistic of the original data is compared to this 
1862     distribution to determine the p-value. 
1863  
1864     When ``n_resamples >= 2**n``, an exact test is performed: the observations 
1865     are assigned to the two samples in each distinct way (while maintaining 
1866     pairings) exactly once. 
1867  
1868     For ``a = [a1, a2, a3]`` and ``b = [b1, b2, b3]``, an example of this 
1869     permutation type is ``x = [b1, a2, b3]`` and ``y = [a1, b2, a3]``. 
1870  
1871     ``permutation_type='samples'`` supports ``data`` containing any number 
1872     of samples, each of which must contain the same number of observations. 
1873     If ``data`` contains more than one sample, paired observations within 
1874     ``data`` are exchanged between samples *independently*. Therefore, if ``m`` 
1875     is the number of samples and ``n`` is the number of observations within 
1876     each sample, then the number of permutations in an exact test is:: 
1877  
1878         factorial(m)**n 
1879  
1880     Several paired-sample statistical tests, such as the Wilcoxon signed rank 
1881     test and paired-sample t-test, can be performed considering only the 
1882     *difference* between two paired elements. Accordingly, if ``data`` contains 
1883     only one sample, then the null distribution is formed by independently 
1884     changing the *sign* of each observation. 
1885  
1886     .. warning:: 
1887         The p-value is calculated by counting the elements of the null 
1888         distribution that are as extreme or more extreme than the observed 
1889         value of the statistic. Due to the use of finite precision arithmetic, 
1890         some statistic functions return numerically distinct values when the 
1891         theoretical values would be exactly equal. In some cases, this could 
1892         lead to a large error in the calculated p-value. `permutation_test` 
1893         guards against this by considering elements in the null distribution 
1894         that are "close" (within a relative tolerance of 100 times the 
1895         floating point epsilon of inexact dtypes) to the observed 
1896         value of the test statistic as equal to the observed value of the 
1897         test statistic. However, the user is advised to inspect the null 
1898         distribution to assess whether this method of comparison is 
1899         appropriate, and if not, calculate the p-value manually. See example 
1900         below.

1914     Examples 
1915     -------- 
1916  
1917     Suppose we wish to test whether two samples are drawn from the same 
1918     distribution. Assume that the underlying distributions are unknown to us, 
1919     and that before observing the data, we hypothesized that the mean of the 
1920     first sample would be less than that of the second sample. We decide that 
1921     we will use the difference between the sample means as a test statistic, 
1922     and we will consider a p-value of 0.05 to be statistically significant. 
1923  
1924     For efficiency, we write the function defining the test statistic in a 
1925     vectorized fashion: the samples ``x`` and ``y`` can be ND arrays, and the 
1926     statistic will be calculated for each axis-slice along `axis`. 
1927  
1928     >>> import numpy as np 
1929     >>> def statistic(x, y, axis): 
1930     ...     return np.mean(x, axis=axis) - np.mean(y, axis=axis) 
1931  
1932     After collecting our data, we calculate the observed value of the test 
1933     statistic. 
1934  
1935     >>> from scipy.stats import norm 
1936     >>> rng = np.random.default_rng() 
1937     >>> x = norm.rvs(size=5, random_state=rng) 
1938     >>> y = norm.rvs(size=6, loc = 3, random_state=rng) 
1939     >>> statistic(x, y, 0) 
1940     -3.5411688580987266 
1941  
1942     Indeed, the test statistic is negative, suggesting that the true mean of 
1943     the distribution underlying ``x`` is less than that of the distribution 
1944     underlying ``y``. To determine the probability of this occurring by chance 
1945     if the two samples were drawn from the same distribution, we perform 
1946     a permutation test. 
1947  
1948     >>> from scipy.stats import permutation_test 
1949     >>> # because our statistic is vectorized, we pass `vectorized=True` 
1950     >>> # `n_resamples=np.inf` indicates that an exact test is to be performed 
1951     >>> res = permutation_test((x, y), statistic, vectorized=True, 
1952     ...                        n_resamples=np.inf, alternative='less') 
1953     >>> print(res.statistic) 
1954     -3.5411688580987266 
1955     >>> print(res.pvalue) 
1956     0.004329004329004329 
1957  
1958     The probability of obtaining a test statistic less than or equal to the 
1959     observed value under the null hypothesis is 0.4329%. This is less than our 
1960     chosen threshold of 5%, so we consider this to be significant evidence 
1961     against the null hypothesis in favor of the alternative. 
1962  
1963     Because the size of the samples above was small, `permutation_test` could 
1964     perform an exact test. For larger samples, we resort to a randomized 
1965     permutation test. 
1966  
1967     >>> x = norm.rvs(size=100, random_state=rng) 
1968     >>> y = norm.rvs(size=120, loc=0.2, random_state=rng) 
1969     >>> res = permutation_test((x, y), statistic, n_resamples=9999, 
1970     ...                        vectorized=True, alternative='less', 
1971     ...                        rng=rng) 
1972     >>> print(res.statistic) 
1973     -0.4230459671240913 
1974     >>> print(res.pvalue) 
1975     0.0015 
1976  
1977     The approximate probability of obtaining a test statistic less than or 
1978     equal to the observed value under the null hypothesis is 0.0225%. This is 
1979     again less than our chosen threshold of 5%, so again we have significant 
1980     evidence to reject the null hypothesis in favor of the alternative. 
1981  
1982     For large samples and number of permutations, the result is comparable to 
1983     that of the corresponding asymptotic test, the independent sample t-test. 
1984  
1985     >>> from scipy.stats import ttest_ind 
1986     >>> res_asymptotic = ttest_ind(x, y, alternative='less') 
1987     >>> print(res_asymptotic.pvalue) 
1988     0.0014669545224902675 
1989  
1990     The permutation distribution of the test statistic is provided for 
1991     further investigation. 
1992  
1993     >>> import matplotlib.pyplot as plt 
1994     >>> plt.hist(res.null_distribution, bins=50) 
1995     >>> plt.title("Permutation distribution of test statistic") 
1996     >>> plt.xlabel("Value of Statistic") 
1997     >>> plt.ylabel("Frequency") 
1998     >>> plt.show() 
1999  
2000     Inspection of the null distribution is essential if the statistic suffers 
2001     from inaccuracy due to limited machine precision. Consider the following 
2002     case: 
2003  
2004     >>> from scipy.stats import pearsonr 
2005     >>> x = [1, 2, 4, 3] 
2006     >>> y = [2, 4, 6, 8] 
2007     >>> def statistic(x, y, axis=-1): 
2008     ...     return pearsonr(x, y, axis=axis).statistic 
2009     >>> res = permutation_test((x, y), statistic, vectorized=True, 
2010     ...                        permutation_type='pairings', 
2011     ...                        alternative='greater') 
2012     >>> r, pvalue, null = res.statistic, res.pvalue, res.null_distribution 
2013  
2014     In this case, some elements of the null distribution differ from the 
2015     observed value of the correlation coefficient ``r`` due to numerical noise. 
2016     We manually inspect the elements of the null distribution that are nearly 
2017     the same as the observed value of the test statistic. 
2018  
2019     >>> r 
2020     0.7999999999999999 
2021     >>> unique = np.unique(null) 
2022     >>> unique 
2023     array([-1. , -1. , -0.8, -0.8, -0.8, -0.6, -0.4, -0.4, -0.2, -0.2, -0.2, 
2024         0. ,  0.2,  0.2,  0.2,  0.4,  0.4,  0.6,  0.8,  0.8,  0.8,  1. , 
2025         1. ])  # may vary 
2026     >>> unique[np.isclose(r, unique)].tolist() 
2027     [0.7999999999999998, 0.7999999999999999, 0.8]  # may vary 
2028  
2029     If `permutation_test` were to perform the comparison naively, the 
2030     elements of the null distribution with value ``0.7999999999999998`` would 
2031     not be considered as extreme or more extreme as the observed value of the 
2032     statistic, so the calculated p-value would be too small. 
2033  
2034     >>> incorrect_pvalue = np.count_nonzero(null >= r) / len(null) 
2035     >>> incorrect_pvalue 
2036     0.14583333333333334  # may vary 
2037  
2038     Instead, `permutation_test` treats elements of the null distribution that 
2039     are within ``max(1e-14, abs(r)*1e-14)`` of the observed value of the 
2040     statistic ``r`` to be equal to ``r``. 
2041  
2042     >>> correct_pvalue = np.count_nonzero(null >= r - 1e-14) / len(null) 
2043     >>> correct_pvalue 
2044     0.16666666666666666 
2045     >>> res.pvalue == correct_pvalue 
2046     True 
2047  
2048     This method of comparison is expected to be accurate in most practical 
2049     situations, but the user is advised to assess this by inspecting the 
2050     elements of the null distribution that are close to the observed value 
2051     of the statistic. Also, consider the use of statistics that can be 
2052     calculated using exact arithmetic (e.g. integer statistics). 
2053  
2054     """
2055     args = _permutation_test_iv(data, statistic, permutation_type, vectorized,
2056                                 n_resamples, batch, alternative, axis,
2057                                 rng)
2058     (data, statistic, permutation_type, vectorized, n_resamples, batch,
2059      alternative, axis, rng) = args
2060 
2061     observed = statistic(*data, axis=-1)
2062 
2063     null_calculators = {"pairings": _calculate_null_pairings,
2064                         "samples": _calculate_null_samples,
2065                         "independent": _calculate_null_both}
2066     null_calculator_args = (data, statistic, n_resamples,
2067                             batch, rng)
2068     calculate_null = null_calculators[permutation_type]
2069     null_distribution, n_resamples, exact_test = (
2070         calculate_null(*null_calculator_args))
2071 
2072     # See References [2] and [3]
2073     adjustment = 0 if exact_test else 1
2074 
2075     # relative tolerance for detecting numerically distinct but
2076     # theoretically equal values in the null distribution
2077     eps =  (0 if not np.issubdtype(observed.dtype, np.inexact)
2078             else np.finfo(observed.dtype).eps*100)
2079     gamma = np.abs(eps * observed)
2080 
2081     def less(null_distribution, observed):
2082         cmps = null_distribution <= observed + gamma
2083         pvalues = (cmps.sum(axis=0) + adjustment) / (n_resamples + adjustment)
2084         return pvalues
2085 
2086     def greater(null_distribution, observed):
2087         cmps = null_distribution >= observed - gamma
2088         pvalues = (cmps.sum(axis=0) + adjustment) / (n_resamples + adjustment)
2089         return pvalues
2090 
2091     def two_sided(null_distribution, observed):
2092         pvalues_less = less(null_distribution, observed)
2093         pvalues_greater = greater(null_distribution, observed)
2094         pvalues = np.minimum(pvalues_less, pvalues_greater) * 2
2095         return pvalues
2096 
2097     compare = {"less": less,
2098                "greater": greater,
2099                "two-sided": two_sided}
2100 
2101     pvalues = compare[alternative](null_distribution, observed)
2102     pvalues = np.clip(pvalues, 0, 1)
2103 
2104     return PermutationTestResult(observed, pvalues, null_distribution)
