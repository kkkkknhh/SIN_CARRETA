3    """ 
4    Provides functionality for compiling PyTorch's autograd (automatic differentiation) system. 
5     
6    This module implements compiled autograd, which traces and optimizes backward pass 
7    computations at runtime. The key components are: 
8     
9    - AutogradCompilerInstance: Traces and compiles autograd graphs using FX 
10   - Context managers (_enable/_disable): Control when compiled autograd is active 
11   - Utility functions: Support graph manipulation, tensor operations, and hooks 
12    
13   Compiled autograd can significantly improve backward pass performance by removing 
14   Python overhead and enabling additional optimizations. It works by capturing 
15   backward computations into an FX graph that can be compiled and optimized, 
16   while maintaining the same semantics as eager mode autograd. 
17   """

114  # Note: [Anomaly Mode Semantics in Compiled Autograd]
115  # In the eager autograd engine, anomaly mode is able to detect NaNs
116  # after each node. This is useful, because the executed code with
117  # and without anomaly mode are the same. So assuming determinism,
118  # a NaN in regular mode should also happen in anomaly mode.
119  #
120  # With torch.compile, following eager semantics would require inserting
121  # runtime asserts to check for NaNs, which could prevent some fusions.
122  # This results in different code being run with and without anomaly mode.
123  # So different semantics are needed, this implementation below will check
124  # for NaNs at the end of the autograd call, instead of after each node
125  class NaNChecker:
126      def __init__(self, accumulate_grad: bool):
127          self.accumulate_grad = accumulate_grad
128          self.params_indices: list[int] = []
129          self.params_to_check: dict[str, torch.Tensor] = {}
130          self.output_names: list[str] = []
131  
132      def prep_with_graph(self, graph: torch.fx.Graph):
133          inputs_node = next(iter(graph.nodes))
134          acc_grad_nodes = graph.find_nodes(
135              op="call_function", target=call_accumulate_grad
136          )
137          output_nodes = graph.find_nodes(op="output")[0].args[0]
138          assert self.accumulate_grad == bool(
139              acc_grad_nodes
140          ) and self.accumulate_grad == (not output_nodes)
141  
142          for node in acc_grad_nodes:
143              param_node = node.args[0]
144              # AccumulateGrad always saves a reference to the param
145              # so Compiled Autograd will always lift the param and
146              # this should always be true
147              assert (
148                  param_node.target == operator.getitem
149                  and param_node.args[0] is inputs_node  # type: ignore[possibly-undefined]
150                  and isinstance(param_node.args[1], int)
151              )
152              self.params_indices.append(param_node.args[1])
153  
154          self.output_names = [node.name for node in output_nodes]
155  
156      def prep_with_inputs(self, inputs: tuple[torch.Tensor]):
157          if not self.accumulate_grad:
158              # Using .grad, nothing to prep
159              return
160  
161          # Using .backward, we must check existing grads on params if any
162          for idx in self.params_indices:
163              grad = inputs[idx].grad
164              if grad is not None:
165                  assert not torch.isnan(grad).any(), (
166                      f"Compiled autograd running under anomaly mode with inputs[{idx}] already "
167                      "having NaN gradient. This is not supported. {TURN_OFF_MSG}"
168                  )
169  
170              self.params_to_check[f"inputs[{idx}]"] = inputs[idx]
171  
172      def check(self, out: tuple[torch.Tensor]):
173          if self.accumulate_grad:
174              # Using .backward, graph outputs are empty
175              assert not out
176              nan_params: list[str] = []
177              for inputs_str, param in self.params_to_check.items():
178                  assert param.grad is not None  # not true for autograd.grad
179                  if torch.isnan(param.grad).any():
180                      nan_params.append(inputs_str)
181  
182              if nan_params:
183                  raise RuntimeError(
184                      f"Compiled Autograd returned NaN gradients for parameters: {','.join(nan_params)}."
185                  )
186          else:
187              # Using .grad, graph outputs are grads
188              nan_grads: list[str] = []
189              for i, grad in enumerate(out):
190                  if torch.isnan(grad).any():
191                      nan_grads.append(self.output_names[i])
192  
193              if nan_grads:
194                  raise RuntimeError(
195                      f"Compiled Autograd returned NaN gradients for output nodes: {','.join(nan_grads)}."
196                  )

199  # We lazily bind "functional backward" variants for PyTorch built-in autograd
200  # nodes to this class. Example: torch._dynamo.compiled_autograd.ops.MulBackward0
201  # Each "functional backward" is bound the first time the node's apply_with_saved
202  # function is called. It's possible to avoid lazy binding and instead bind
203  # all of this upfront (perhaps at import time) via codegen changes.
204  class OpNamespace:
205      def __init__(self):
206          self.custom_function_name_counter: Counter[str] = Counter()
207  
208      def add(self, name, fn, is_custom_function, is_traceable):
209          if is_custom_function:
210              name = "CppNode" + name
211              count = self.custom_function_name_counter[name]
212              self.custom_function_name_counter[name] += 1
213              name = f"{name}{count}"
214  
215          assert not hasattr(self, name)
216          result = Op(name, fn, is_custom_function)
217          if is_traceable:
218              setattr(self, name, torch._dynamo.allow_in_graph(result))
219          else:
220              # C++ autograd function was not marked as traceable
221              # Dynamo can't dry run it at compile time, so must fallback to eager
222              @torch._dynamo.disable
223              def run_non_traceable_cpp_in_eager(*args, **kwargs):
224                  return result(*args, **kwargs)
225  
226              setattr(self, name, run_non_traceable_cpp_in_eager)
227          return name
228  
229      def get(self, name):
230          return getattr(self, name)
231

233  class Op:
234      def __init__(self, name, fn, is_custom_function):
235          self.fn = fn
236          self.is_custom_function = is_custom_function
237          self.__name__ = name
238          self.__module__ = "torch._dynamo.compiled_autograd.ops"

241          return self.fn(*args, **kwargs)

275  class AutogradCompilerInstance:
276      def __init__(self, compiler_fn) -> None:
277          self.compiler_fn = compiler_fn
278          self.stack = contextlib.ExitStack()
279          self.close = self.stack.close
280          self.shape_env = ShapeEnv()
281          self.fake_tensor_mode = FakeTensorMode(
282              allow_fallback_kernels=True,
283              allow_non_fake_inputs=True,
284              shape_env=self.shape_env,
285          )
286          self.fx_tracer = PythonKeyTracer()
287          self.proxy_mode = ProxyTorchDispatchMode(self.fx_tracer, "symbolic")
288          self.hooks_proxy: Optional[Proxy] = None
289  
290      def wrap_fake(self, x, source):
291          assert isinstance(x, torch.Tensor)
292          return self.fake_tensor_mode.from_tensor(x, source=source)
293  
294      @staticmethod
295      def source(name, idx) -> GetItemSource:

298      def begin_capture(
299          self,
300          inputs: list[torch.Tensor],
301          sizes: list[int],
302          scalars: list[Union[int, float]],
303          origins: list[list[tuple[int, str]]],
304          accumulate_grad: bool,
305          check_nans: bool,
306      ):
307          counters["compiled_autograd"]["captures"] += 1
308          self.id = next(COMPILE_COUNTER)
309          self.aot_id_counter: dict[int, int] = defaultdict(int)
310          self.compile_context = make_compile_context(self.id)
311          self.compile_context.__enter__()
312          self.nan_checker = NaNChecker(accumulate_grad) if check_nans else None
313          self.start_time_ns = time.time_ns()
314          get_chromium_event_logger().log_event_start(
315              "compiled_autograd",
316              self.start_time_ns,
317              {"graph_id": self.id},
318              log_pt2_compile_event=True,
319          )
320          self.fx_tracer.root = torch.nn.Module()
321          self.fx_tracer.graph = torch.fx.Graph(tracer_cls=PythonKeyTracer)
322          self.fx_tracer.tensor_attrs = {}
323          self.symnode_proxy_lookup = {}
324          (
325              args_proxy,
326              self.sizes_proxy,
327              self.scalars_proxy,
328              self.hooks_proxy,
329              self.packed_data_proxy,
330          ) = (
331              self.fx_tracer.create_proxy("placeholder", name, (), {})
332              for name in _graph_placeholders
333          )
334  
335          self.stack.enter_context(preserve_node_meta())
336          inputs_origins, sizes_origins, scalars_origins = origins
337  
338          # tensor inputs to fake tensors
339          x = inputs[0]  # mypy will complain about unbound x
340          try:
341              for idx, x in enumerate(inputs):
342                  inputs[idx] = self.wrap_fake(x, self.source("inputs", idx))
343          except Exception as e:
344              raise NotImplementedError(
345                  f"Found tensor of type {type(x)}, which is not supported by FakeTensorMode. {TURN_OFF_MSG}"
346              ) from e
347          self.bind_objects_to_proxies(inputs, args_proxy, inputs_origins)
348  
349          # size inputs to symints
350          sizes = [
351              self.shape_env.create_unspecified_symint_and_symbol(
352                  val,
353                  self.source("sizes", idx),
354                  DimDynamic.DYNAMIC,
355              )
356              for idx, val in enumerate(sizes)
357          ]
358  
359          # We want to mark every size as dynamic, but since there's no way to
360          # mark a primitive `int` as dynamic, we need to wrap it in a tensor.
361          # In the graph, we unwrap it with `unwrap_maybe_dynamic_int` back into a primitive.
362          proxies = [self.sizes_proxy[i] for i in range(len(sizes))]  # type: ignore[index]
363          for i, symint in enumerate(sizes):
364              proxies[i] = self.fx_tracer.create_proxy(
365                  "call_function",
366                  unwrap_maybe_dynamic_int,
367                  (proxies[i],),
368                  {},
369              )
370              self.symnode_proxy_lookup[symint.node] = proxies[i]
371          proxies = self.bind_objects_to_proxies(sizes, proxies, sizes_origins)
372  
373          for idx, val in enumerate(scalars):
374              source = self.source("scalars", idx)
375              if isinstance(val, int):
376                  scalars[idx] = self.shape_env.create_unspecified_symint_and_symbol(
377                      val,
378                      source,
379                      DimDynamic.DYNAMIC,
380                  )
381              elif isinstance(val, float):
382                  scalars[idx] = self.shape_env.create_symfloatnode(
383                      self.shape_env.create_unspecified_symbol(
384                          val,
385                          source=source,
386                          dynamic_dim=DimDynamic.DYNAMIC,
387                      ),
388                      hint=val,
389                      source=source,
390                  )
391              else:
392                  raise AssertionError("Unexpected scalar type: ", type(val))
393          self.bind_objects_to_proxies(scalars, self.scalars_proxy, scalars_origins)
394          for i, symval in enumerate(scalars):
395              self.symnode_proxy_lookup[symval.node] = self.scalars_proxy[i]  # type: ignore[union-attr]
396  
397          # TODO(jansel): are all these modes needed?
398          self.stack.enter_context(decompose({}))
399          self.stack.enter_context(self.fake_tensor_mode)
400          self.stack.enter_context(self.proxy_mode)
401          self.stack.enter_context(disable_autocast_cache())
402          # Needed to make sure we don't accidentally specialize any symbols
403          assert self.fake_tensor_mode.shape_env is not None
404          env = self.fake_tensor_mode.shape_env
405          self.stack.enter_context(
406              torch.fx.experimental.symbolic_shapes._suppress_guards(env)
407          )
408          return (
409              str(CompileContext.current_compile_id()),
410              inputs,
411              sizes,
412              scalars,
413          )

429      def proxy_call_aot_backward(
430          self,
431          pinputs,
432          psaved_tensors,
433          saved_tensors,
434          pctx,
435          ctx,
436          maybe_backward_state_idx,
437      ):
438          # The AOTBackward call consists of three things: the prologue, the
439          # backward graph, and the epilogue.
440          # Our strategy is:
441          # - allow_in_graph the prologue (in the CA graph and Dynamo graph),
442          # - copy-paste the backward graph into the CA graph so that CA passes and Dynamo can see it
443          # - trace directly through the epilogue. Anything that gets baked in is
444          #   constant metadata (for example, metadata about the number of outputs, or removing
445          #   RNG arguments or effect tokens).
446          # If Dynamo graph capture were better, then we could add a node for the prologue
447          # into the CA graph and have Dynamo trace into it.
448  
449          psymints = [self.to_proxy(e) for e in ctx._get_compiled_autograd_symints()]
450  
451          # NOTE: we should only close over constants
452          CompiledFunction = ctx._forward_cls
453          bw_module = extract_bw_module(CompiledFunction)
454          metadata = CompiledFunction.metadata
455          maybe_subclass_metadata = CompiledFunction.maybe_subclass_metadata
456          aot_id = CompiledFunction._aot_id
457          del CompiledFunction
458  
459          if torch.is_grad_enabled():
460              for output_alias_info in metadata.output_info:
461                  if output_alias_info.requires_grad:
462                      raise RuntimeError(
463                          "torch.compile does not currently support higher order gradients."
464                      )
465  
466          @torch._dynamo.allow_in_graph  # type: ignore[misc]
467          def call_aot_bwd_prologue(ctx_saved_tensors, ctx_symints, *flat_args):
468              out = torch._functorch._aot_autograd.runtime_wrappers._backward_prologue_functional(
469                  ctx_saved_tensors,
470                  ctx_symints,
471                  metadata,
472                  maybe_subclass_metadata,
473                  *flat_args,
474              )
475              return out
476  
477          pgrads = self.fx_tracer.create_proxy(
478              kind="call_function",
479              target=call_aot_bwd_prologue,
480              args=(
481                  psaved_tensors,
482                  psymints,
483                  *pinputs,
484              ),
485              kwargs={},
486          )
487  
488          pbackward_state = None
489          if maybe_backward_state_idx is not None:
490              pbackward_state = self.hooks_proxy[maybe_backward_state_idx]  # type: ignore[index]
491  
492          # Copy-paste the AOT backward graph into the compiled autograd graph
493          def copy_paste_aot_backward_graph():
494              def num_inputs(graph):
495                  num_args = 0
496                  for node in graph.nodes:
497                      if node.op == "placeholder":
498                          num_args += 1
499                          continue
500                      else:
501                          break
502                  return num_args
503  
504              # set up the proxy inputs to bw_module
505              # the calling convention is: [*symints, *args (primals and tangents), backward_state]
506              num_args = num_inputs(bw_module.graph)
507              pall_args = [
508                  pgrads[i] for i in range(num_args - int(pbackward_state is not None))
509              ]
510              # replace the symints with our symints
511              symints = ctx._get_compiled_autograd_symints()
512              assert len(symints) == len(ctx.symints)
513              psymints = [self.to_proxy(e) for e in symints]
514              pall_args[: len(symints)] = psymints
515              # Add backward_state
516              if pbackward_state is not None:
517                  pall_args.append(pbackward_state)
518  
519              # run over all nodes of the aot_backward graph.
520              # copy and paste them all into the compiled autograd graph.
521              args_idx = 0
522              value_remap = {}
523              poutputs: Optional[list[torch.fx.Proxy]] = None
524  
525              # names of nodes must appear only once in the fx.Graph
526              # dedup AOT backwards that appear multiple times
527              deduped_aot_id = str(aot_id)
528              if self.aot_id_counter[aot_id]:
529                  deduped_aot_id += f"_{self.aot_id_counter[aot_id]}"
530              self.aot_id_counter[aot_id] += 1
531  
532              def make_unique(node_name):
533                  # make it both informative and unique
534                  return f"aot{deduped_aot_id}_{node_name}"
535  
536              for node in bw_module.graph.nodes:
537                  if node.op == "placeholder":
538                      ph = pall_args[args_idx].node
539                      ph.name = make_unique(node.name)
540                      value_remap[node] = ph
541                      args_idx += 1
542                  elif node.op == "output":
543                      assert len(node.args) == 1
544                      poutputs = [
545                          torch.fx.Proxy(value_remap[n], self.fx_tracer)
546                          if isinstance(n, torch.fx.Node)
547                          else n
548                          for n in node.args[0]
549                      ]
550                  elif node.op == "get_attr":
551                      name = node.target
552                      qualname = self.fx_tracer.get_fresh_qualname(name)
553                      setattr(self.fx_tracer.root, qualname, getattr(bw_module, name))
554                      result = self.fx_tracer.create_node("get_attr", qualname, (), {})
555                      result.name = make_unique(node.name)
556                      value_remap[node] = result
557                  elif node.op == "call_function":
558                      if node.target == torch.ops.aten.view.default:
559                          # this aot bwd graph is being lazily compiled
560                          # we must manually apply the view_to_reshape post grad pass
561                          # since it was already applied to the aot fwd, and baked into the gradients
562                          node.target = torch.ops.aten.reshape.default
563                      result = self.fx_tracer.graph.node_copy(
564                          node, lambda n: value_remap[n]
565                      )
566                      result.name = make_unique(node.name)
567                      value_remap[node] = result
568                  elif node.op == "call_module":
569                      name = node.target
570                      qualname = self.fx_tracer.get_fresh_qualname(name)
571                      setattr(self.fx_tracer.root, qualname, getattr(bw_module, name))
572                      result = self.fx_tracer.graph.node_copy(
573                          node, lambda n: value_remap[n]
574                      )
575                      result.target = qualname
576                      value_remap[node] = result
577                  else:
578                      raise AssertionError("shouldn't get here")
579  
580              assert poutputs is not None
581  
582              # In general we don't know what the shapes of the outputs are, so allocate
583              # some dummy sizes for them.
584              def dummy():
585                  with disable_proxy_modes_tracing():
586                      return torch.zeros(0, 0, 0, 0, 123)
587  
588              outputs = [
589                  dummy() if isinstance(o, torch.fx.Proxy) else o for o in poutputs
590              ]
591              self.bind_objects_to_proxies(outputs, poutputs)
592              return outputs
593  
594          outputs = copy_paste_aot_backward_graph()
595  
596          def proxy_subclass_constructor(subclass_meta, is_runtime, unwrapped_args):
597              @torch._dynamo.allow_in_graph
598              def make_subclass(*unwrapped_args):
599                  return subclass_meta.creation_fn(unwrapped_args, is_runtime=is_runtime)
600  
601              punwrapped_args = pytree.tree_map(self.to_proxy, unwrapped_args)
602  
603              poutput = self.fx_tracer.create_proxy(
604                  kind="call_function",
605                  target=make_subclass,
606                  args=tuple(punwrapped_args),
607                  kwargs={},
608              )
609  
610              output = self.allocate_dummy()
611              self.bind_objects_to_proxies([output], [poutput])
612              return output
613  
614          results = torch._functorch._aot_autograd.runtime_wrappers._backward_epilogue_functional(
615              metadata,
616              maybe_subclass_metadata,
617              outputs,
618              make_subclass_override=proxy_subclass_constructor,
619          )
620          presults = pytree.tree_map(self.to_proxy, results)
621          return presults

623      def proxy_call_backward(
624          self,
625          inputs,

850      # Note: [Compiled autograd and cudagraphs]
851      # Eager autograd backward implements scalars as 0-dim tensors, see DivBackward0::other_.
852      # When compiled autograd traces those nodes, it lifts the scalar tensors, resulting in a graph
853      # with some cpu 0-dim tensor inputs. To prevent the entire graph from skipping cudagraph, we move the
854      # scalars tensors to cuda. This works because ATen/prims ops will accept cuda 0-dim tensors too.
855      def move_graph_nodes_to_cuda(self, graph) -> list[int]:
856          to_move: dict[int, torch.fx.Node] = {}
857          has_cuda_inputs = False
858          nodes = list(graph.nodes)
859          assert nodes[0].target == "inputs"
860          inputs = nodes[0]
861          inputs_users = list(inputs.users.keys())
862          # input access nodes should immediately follow placeholder nodes
863          first_getitem_idx = len(_graph_placeholders)
864          assert nodes[first_getitem_idx] == inputs_users[0]
865          last_getitem_idx = first_getitem_idx + len(inputs_users) - 1
866          assert nodes[last_getitem_idx] == inputs_users[-1]
867          # getitem nodes on inputs
868          for i, node in enumerate(inputs_users):
869              if not has_cuda_inputs and node.meta["val"].device.type == "cuda":
870                  has_cuda_inputs = True
871                  continue
872  
873              is_cpu = node.meta["val"].device.type == "cpu"
874              is_scalar = len(node.meta["val"].size()) == 0
875              if is_cpu and is_scalar:
876                  node_users = list(node.users.keys())
877                  # We can only move the cpu scalar if it is not exposed to user code.
878                  if all(
879                      (
880                          isinstance(user.target, torch._ops.OpOverload)
881                          and user.target.namespace in ("prims", "aten")
882                      )
883                      or (
884                          isinstance(user.target, Op)
885                          and not user.target.is_custom_function
886                      )
887                      for user in node_users
888                  ):
889                      # all users are prims/aten, can move safely
890                      to_move[i] = node
891  
892          # only move cpu scalars to cuda if there were cuda activations in this graph,
893          # this is to handle the case where cudagraphs is enabled on a cpu-only graph
894          if has_cuda_inputs:
895              for node in to_move.values():
896                  verbose_log.debug("Moving node %s from cpu to cuda", node)
897                  node.meta["val"] = node.meta["val"].cuda()
898  
899              # return runtime indices we need to move to cuda
900              return list(to_move.keys())
901  
902          return []

912      def dce(self):
913          # Most of these removed nodes would have been removed during Dynamo and AOTDispatch
914          # Remove some of these nodes earlier to improve compilation speed
915  
916          # Dynamo guards will error instead of creating aliasing guards unless we unpack them in the graph
917          unpack_nodes: OrderedSet[torch.fx.Node] = OrderedSet()
918          for i, node in enumerate(self.fx_tracer.graph.find_nodes(op="placeholder")):
919              unpack_nodes.update(node.users.keys())
920          assert i == len(_graph_placeholders) - 1
921  
922          def is_impure(node):
923              if node in unpack_nodes or (
924                  node.op == "call_function" and node.target in _impure_targets
925              ):
926                  return True
927              return node.is_impure()
928  
929          before = len(self.fx_tracer.graph.nodes)
930          self.fx_tracer.graph.eliminate_dead_code(is_impure)
931          after = len(self.fx_tracer.graph.nodes)
932          verbose_log.debug("DCE removed %d nodes", before - after)

934      def remove_unused_sizes(self):
935          used_sizes = []
936          unused_sizes = []
937  
938          # seek placeholder, should be at nodes[1]
939          it = iter(self.fx_tracer.graph.nodes)
940          next(it)
941          sizes_node = next(it)
942          assert sizes_node.name == "sizes"
943  
944          for getitem_node in sizes_node.users.keys():
945              assert getitem_node.target == operator.getitem
946              if getitem_node.users:
947                  used_sizes.append(getitem_node)
948              else:
949                  # remove from the graph
950                  unused_sizes.append(getitem_node)
951  
952          used_sizes_idx: set[int] = set()
953          for used in used_sizes:
954              assert isinstance(used.args, tuple)
955              assert used.args[0] == sizes_node
956              assert isinstance(used.args[1], int)
957              next_size_idx = len(used_sizes_idx)
958              # used later reindex the runtime sizes arg
959              used_sizes_idx.add(used.args[1])
960              # reindex the graph
961              used.args = (used.args[0], next_size_idx)
962  
963          for unused in unused_sizes:
964              self.fx_tracer.graph.erase_node(unused)
965  
966          return used_sizes_idx

1106     def reorder_accumulate_grad_nodes(self):
1107         """ 
1108         Usage of AOTAutograd causes all the accumulate_grad_ nodes to get pushed to the end of 
1109         the graph.  This differs from eager mode, which schedules them as soon as possible. This 
1110         pass attempts to reorder the graph to mimic eager behavior. 
1111         """
1112         for node in self.fx_tracer.graph.find_nodes(
1113             op="call_function", target=call_accumulate_grad
1114         ):
1115             param_node, grad_node = node.args[0], node.args[1]
1116             getitem_node = None
1117             if grad_node.target == operator.getitem:
1118                 getitem_node = grad_node
1119                 grad_node = getitem_node.args[0]
1120 
1121             arg = max([param_node, grad_node])  # last arg
1122             if arg is not node.prev and not self.is_placeholder(arg):
1123                 arg.append(node)
1124                 if getitem_node is not None:
1125                     arg.append(getitem_node)

1160     def reorder_pre_hook_nodes_to_schedule_asap(self):
1161         """ 
1162         In this function, we schedule the pre hooks as soon as possible. This 
1163         does not match eager behavior (schedule pre hook right before its 
1164         registered node), but it can make acc grad be scheduled properly when 
1165         the pre hooks are registered to them. After reordering acc grad node, we 
1166         will reorder the pre hooks again to mimic eager behavior. 
1167         """
1168         for node in self.fx_tracer.graph.find_nodes(
1169             op="call_function", target=call_hook
1170         ):
1171             if node.kwargs.get("hook_type", None) != "pre_hook":
1172                 continue
1173 
1174             getitem_node = node.args[0]
1175             # pre_hook handle a tuple of grad tensors
1176             input_nodes = self.get_all_nodes(node.args[1])
1177 
1178             to_remove = []
1179             to_append = []
1180             hook_block = [node]  # contain the hook and hook args getitem
1181             for n in input_nodes:
1182                 if n.op == "call_function" and n.target == operator.getitem:
1183                     to_append.append(n.args[0])
1184                     to_remove.append(n)
1185                     hook_block.append(n)
1186             for a, b in zip(to_remove, to_append):
1187                 input_nodes.remove(a)
1188                 input_nodes.append(b)
1189 
1190             arg = max(input_nodes)  # last input
1191             if arg is not node.prev and not self.is_placeholder(arg):
1192                 arg.append(getitem_node)
1193                 for n in hook_block:
1194                     getitem_node.append(n)

1196     def reorder_pre_hook_nodes_to_mimic_eager(self):
1197         """ 
1198         Usage of AOTAutograd causes all the pre_hook nodes to get pushed to the 
1199         end of the graph. This differs from eager mode, which schedules them 
1200         right before their registered node execution. This pass attempts to 
1201         reorder the graph to mimic eager behavior. 
1202         """
1203         pre_hooks = []
1204         for node in self.fx_tracer.graph.find_nodes(
1205             op="call_function", target=call_hook
1206         ):
1207             if node.kwargs.get("hook_type", None) != "pre_hook":
1208                 continue
1209             pre_hooks.append(node)
1210 
1211         for node in reversed(pre_hooks):
1212             hook_getitem_node = node.args[0]
1213 
1214             users = list(node.users.keys())
1215             if len(users) == 0:
1216                 continue
1217 
1218             # users are all getitem ops and they are used by same registered node
1219             assert all(
1220                 user.op == "call_function" and user.target == operator.getitem
1221                 for user in users
1222             )
1223             registered_node = next(iter(users[0].users.keys()))
1224 
1225             if registered_node is not node.next:
1226                 registered_node.prepend(hook_getitem_node)
1227                 registered_node.prepend(node)
1228                 for getitem in users:
1229                     registered_node.prepend(getitem)

1231     def reorder_post_acc_grad_hook_nodes(self):
1232         """ 
1233         Usage of AOTAutograd causes all the post_acc_grad_hook nodes to get 
1234         pushed to the end of the graph. This differs from eager mode, which 
1235         schedules them as soon as possible. This pass attempts to reorder the 
1236         graph to mimic eager behavior. 
1237         """
1238         post_acc_grad_hooks = []
1239         for node in self.fx_tracer.graph.find_nodes(
1240             op="call_function", target=call_hook
1241         ):
1242             if node.kwargs.get("hook_type", None) != "post_acc_grad_hook":
1243                 continue
1244             post_acc_grad_hooks.append(node)
1245 
1246         # nodes in post_acc_grad_hooks are in topo order. For hooks registered
1247         # to same node, we should keep their relative order
1248         for node in reversed(post_acc_grad_hooks):
1249             getitem_node = node.args[0]
1250             param_node = node.args[1]  # post_acc_grad_hook handle one param
1251 
1252             # find the corresponding acc_grad node
1253             acc_grad_node = None
1254             for n in list(param_node.users.keys()):
1255                 if n.op == "call_function" and n.target == call_accumulate_grad:
1256                     acc_grad_node = n
1257                     break
1258 
1259             assert acc_grad_node is not None, (
1260                 "post_acc_grad_hook must have corresponding acc grad node"
1261             )
1262 
1263             # append post_acc_grad_hook after acc_grad node
1264             acc_grad_node.append(getitem_node)
1265             getitem_node.append(node)
1266 
1267     def reorder_post_hook_nodes(self):
1268         """ 
1269         Usage of AOTAutograd causes all the post_hook nodes to get pushed to the 
1270         end of the graph. This differs from eager mode, which schedules them as 
1271         soon as possible. This pass attempts to reorder the graph to mimic eager 
1272         behavior. 
1273         """
1274         post_hooks = []
1275         for node in self.fx_tracer.graph.find_nodes(
1276             op="call_function", target=call_hook
1277         ):
1278             if node.kwargs.get("hook_type", None) != "post_hook":
1279                 continue
1280             post_hooks.append(node)
1281 
1282         for node in reversed(post_hooks):
1283             getitem_node = node.args[0]
1284             output_nodes = node.args[1]
1285             input_nodes = node.args[2]
1286 
1287             if len(output_nodes) > 0:
1288                 continue
1289 
1290             input_nodes_and_users = []
1291             input_nodes_and_users.extend(list(input_nodes))
1292             for input_node in input_nodes:
1293                 input_nodes_and_users.extend(
1294                     user
1295                     for user in list(input_node.users.keys())
1296                     if not (
1297                         user.op == "call_function"
1298                         and user.target == call_hook
1299                         and node.kwargs.get("hook_type", None) == "post_hook"
1300                     )
1301                 )
1302 
1303             arg = max(input_nodes_and_users)  # last input users
1304             if arg.op == "call_function" and arg.target == call_accumulate_grad:
1305                 param_node = arg.args[0]
1306                 post_acc_grad_hook_node = None
1307                 for n in list(param_node.users.keys()):
1308                     if (
1309                         n.op == "call_function"
1310                         and n.target == call_hook
1311                         and n.kwargs.get("hook_type", None) == "post_acc_grad_hook"
1312                     ):
1313                         post_acc_grad_hook_node = n
1314 
1315                 if post_acc_grad_hook_node is not None:
1316                     post_acc_grad_hook_node.append(getitem_node)
1317                     getitem_node.append(node)
1318                     continue
1319 
1320             if arg is not node.prev and not self.is_placeholder(arg):
1321                 arg.append(getitem_node)
1322                 getitem_node.append(node)

1324     def to_proxy(self, t):
1325         if t is None:
1326             return None
1327         if isinstance(t, list):
1328             return [self.to_proxy(x) for x in t]
1329         if isinstance(t, tuple):
1330             return tuple(self.to_proxy(x) for x in t)
1331         if isinstance(t, (torch.SymInt, torch.SymFloat)):
1332             return self.symnode_proxy_lookup[t.node]
1333         if not isinstance(t, torch.Tensor):
1334             # constant types like device, dtype, str
1335             return t
1336         proxy_tensor = fetch_object_proxy(self.fx_tracer, t)
1337         assert isinstance(proxy_tensor, torch.fx.experimental.proxy_tensor._ProxyTensor)
1338         return proxy_tensor.proxy

1340     def bind_objects_to_proxies(
1341         self, objects, proxies, origins: Optional[list[tuple[int, str]]] = None
1342     ):
1343         if isinstance(proxies, torch.fx.Proxy):
1344             if origins:
1345                 assert len(origins) == len(objects)
1346                 bound_proxies = []
1347                 for i in range(len(objects)):
1348                     nodecall_index, node_name = origins[i]
1349                     self.set_node_origin(node_name, nodecall_index, None)
1350                     bound_proxies.append(proxies[i])  # type: ignore[index]
1351                 proxies = bound_proxies
1352             else:
1353                 proxies = [proxies[i] for i in range(len(objects))]  # type: ignore[index]
1354 
1355         assert len(objects) == len(proxies)
1356         track_tensor_tree(objects, proxies, constant=None, tracer=self.fx_tracer)
1357         return proxies

1367         self,

1371     ):

1381                     )
1382                 maybe_aot_id = forward_cls._aot_id

1386             "raw_stack_trace = CapturedTraceback.extract().format()[-1]", new_code

1392 compiled_autograd_enabled = False

1394 # global flag to check if compiled autograd is enabled but Dynamo stance is "force_eager"
1395 compiled_autograd_enabled_force_eager = False
1396 
1397 # global flag to check if we are processing graphs produced from a compiled autograd graph
1398 in_compiled_autograd_region = False

1400 active_disable_ctx = False

1402 depth = 0

1405 @contextlib.contextmanager
1406 def _enable(compiler_fn, dynamic: bool = True, ignore_active_disable_ctx=True):
1407     # The entrypoint to enable CA.
1408     # It is recommended to enable via `torch._dynamo.config.compiled_autograd = True` rather
1409     # than using this context manager directly. If you are torch.compiling the corresponding
1410     # forward pass, make sure they are wrapped under this context as well.
1411     #
1412     # Example:
1413     #   def train(model, inputs, target):
1414     #     compiled_model = torch.compile(model)
1415     #     pred = compiled_model(data)
1416     #     loss = compute_loss(pred, target)
1417     #     loss.backward()
1418     #
1419     #   with _enable(compiler_fn):
1420     #      train(model, inputs, target)
1421     #
1422     # Inputs:
1423     # - compiler_fn: The wrapper that will consume the compiled autograd graph, e.g. `torch.compile`
1424     # - dynamic: Whether compiled autograd will treat tensors in the autograd graph (params, activations) as dynamic.
1425     #   This doesn't affect the dynamic configuration of the compilation wrapper.
1426 
1427     if not ignore_active_disable_ctx and active_disable_ctx:
1428         yield
1429     else:
1430         if dynamic:
1431             assert type(dynamic) is bool
1432 
1433         from torch._dynamo import eval_frame
1434 
1435         if eval_frame._stance.stance == "force_eager":
1436             # If user explicitly sets Dynamo stance to "force_eager", we want Compiled Autograd
1437             # to fall back to eager as well.
1438             global compiled_autograd_enabled_force_eager
1439             compiled_autograd_enabled_force_eager = True
1440             try:
1441                 yield
1442             finally:
1443                 compiled_autograd_enabled_force_eager = False
1444         else:
1445             # we need to import this, because user might not have imported it if they directly use this context manager
1446             # we need to lazily import it, because of circular dependencies
1447             import torch._inductor.cudagraph_trees
1448 
1449             (
1450                 prior_compiler,
1451                 prior_dynamic,
1452             ) = torch._C._dynamo.compiled_autograd.set_autograd_compiler(
1453                 functools.partial(AutogradCompilerInstance, compiler_fn), dynamic
1454             )
1455             if snapshot_verbose_logging_enabled():
1456                 torch._C._dynamo.compiled_autograd.set_verbose_logger(verbose_log)  # type:ignore[arg-type]
1457             global compiled_autograd_enabled
1458             compiled_autograd_enabled = True
1459             global depth
1460             prior_depth = depth
1461             depth += 1
1462             try:
1463                 with torch.autograd.set_multithreading_enabled(False):
1464                     yield
1465             finally:
1466                 if not prior_compiler:
1467                     compiled_autograd_enabled = False
1468                 torch._C._dynamo.compiled_autograd.set_autograd_compiler(
1469                     prior_compiler, prior_dynamic
1470                 )
1471                 depth -= 1
1472                 assert depth == prior_depth, (
1473                     "Nested Compiled Autograd Contexts must return before their parent context"
1474                 )

1513 def copy_slices_prologue(
1514     inputs,

1516     base_strides,

1518     view_sizes,

1524     assert grad is not None
