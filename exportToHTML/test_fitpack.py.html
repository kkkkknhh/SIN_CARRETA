27       return np.sqrt(np.dot(x.T, x))

30   def f1(x, d=0):
31       """Derivatives of sin->cos->-sin->-cos."""
32       if d % 4 == 0:
33           return np.sin(x)
34       if d % 4 == 1:
35           return np.cos(x)
36       if d % 4 == 2:
37           return -np.sin(x)
38       if d % 4 == 3:
39           return -np.cos(x)

42   def makepairs(x, y):
43       """Helper function to create an array of pairs of x and y."""
44       xy = np.array(list(itertools.product(np.asarray(x), np.asarray(y))))
45       return xy.T

53       def check_1(self, per=0, s=0, a=0, b=2*np.pi, at_nodes=False,
54                   xb=None, xe=None):
55           if xb is None:
56               xb = a
57           if xe is None:
58               xe = b
59   
60           N = 20
61           # nodes and middle points of the nodes
62           x = np.linspace(a, b, N + 1)
63           x1 = a + (b - a) * np.arange(1, N, dtype=float) / float(N - 1)
64           v = f1(x)
65   
66           def err_est(k, d):
67               # Assume f has all derivatives < 1
68               h = 1.0 / N
69               tol = 5 * h**(.75*(k-d))
70               if s > 0:
71                   tol += 1e5*s
72               return tol
73   
74           for k in range(1, 6):
75               tck = splrep(x, v, s=s, per=per, k=k, xe=xe)
76               tt = tck[0][k:-k] if at_nodes else x1
77   
78               for d in range(k+1):
79                   tol = err_est(k, d)
80                   err = norm2(f1(tt, d) - splev(tt, tck, d)) / norm2(f1(tt, d))
81                   assert err < tol
82   
83               # smoke test make_splrep
84               if not per:
85                   spl = make_splrep(x, v, k=k, s=s, xb=xb, xe=xe)
86                   if len(spl.t) == len(tck[0]):
87                       xp_assert_close(spl.t, tck[0], atol=1e-15)
88                       xp_assert_close(spl.c, tck[1][:spl.c.size], atol=1e-13)
89                   else:
90                       assert k == 5   # knot length differ in some k=5 cases

248          # Check that antiderivative + derivative is identity.
249          for n in range(5):

257          # Check derivative vs. FITPACK
258  
259          for n in range(3+1):
260              # Also extrapolation!
261              xx = np.linspace(-1, 2, 2000)
262              if n == 3:
263                  # ... except that FITPACK extrapolates strangely for
264                  # order 0, so let's not check that.
265                  xx = xx[(xx >= 0) & (xx <= 1)]

267              dy = splev(xx, self.spl, n)
268              spl2 = splder(self.spl, n)
269              dy2 = splev(xx, spl2)
270              if n == 1:
271                  xp_assert_close(dy, dy2, rtol=2e-6)
272              else:
273                  xp_assert_close(dy, dy2)

307          # c can have trailing dims

321  class TestSplint:
322      def test_len_c(self):
323          n, k = 7, 3
324          x = np.arange(n)
325          y = x**3
326          t, c, k = splrep(x, y, s=0)
327  
328          # note that len(c) == len(t) == 11 (== len(x) + 2*(k-1))
329          assert len(t) == len(c) == n + 2*(k-1)
330  
331          # integrate directly: $\int_0^6 x^3 dx = 6^4 / 4$
332          res = splint(0, 6, (t, c, k))
333          expected = 6**4 / 4
334          assert abs(res - expected) < 1e-13
335  
336          # check that the coefficients past len(t) - k - 1 are ignored
337          c0 = c.copy()
338          c0[len(t) - k - 1:] = np.nan
339          res0 = splint(0, 6, (t, c0, k))
340          assert abs(res0 - expected) < 1e-13
341  
342          # however, all other coefficients *are* used
343          c0[6] = np.nan
344          assert np.isnan(splint(0, 6, (t, c0, k)))
345  
346          # check that the coefficient array can have length `len(t) - k - 1`
347          c1 = c[:len(t) - k - 1]
348          res1 = splint(0, 6, (t, c1, k))
349          assert (res1 - expected) < 1e-13
350  
351  
352          # however shorter c arrays raise. The error from f2py is a
353          # `dftipack.error`, which is an Exception but not ValueError etc.
354          with assert_raises(Exception, match=r">=n-k-1"):
355              splint(0, 1, (np.ones(10), np.ones(5), 3))
