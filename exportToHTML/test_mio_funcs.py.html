13   def read_minimat_vars(rdr):
14       rdr.initialize_read()
15       mdict = {'__globals__': []}
16       i = 0
17       while not rdr.end_of_stream():
18           hdr, next_position = rdr.read_var_header()
19           name = 'None' if hdr.name is None else hdr.name.decode('latin1')
20           if name == '':
21               name = 'var_%d' % i
22               i += 1
23           res = rdr.read_var_array(hdr, process=False)
24           rdr.mat_stream.seek(next_position)
25           mdict[name] = res
26           if hdr.is_global:
27               mdict['__globals__'].append(name)
28       return mdict

31   def read_workspace_vars(fname):
32       fp = open(fname, 'rb')
33       rdr = MatFile5Reader(fp, struct_as_record=True)
34       vars = rdr.get_variables()
35       fws = vars['__function_workspace__']
36       ws_bs = io.BytesIO(fws.tobytes())
37       ws_bs.seek(2)
38       rdr.mat_stream = ws_bs
39       # Guess byte order.
40       mi = rdr.mat_stream.read(2)
41       rdr.byte_order = mi == b'IM' and '<' or '>'
42       rdr.mat_stream.read(4)  # presumably byte padding
43       mdict = read_minimat_vars(rdr)
44       fp.close()
45       return mdict
