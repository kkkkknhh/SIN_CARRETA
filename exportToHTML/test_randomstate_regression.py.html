91       def test_choice_sum_of_probs_tolerance(self):
92           # The sum of probs should be 1.0 with some tolerance.
93           # For low precision dtypes the tolerance was too tight.
94           # See numpy github issue 6123.
95           random.seed(1234)
96           a = [1, 2, 3]
97           counts = [4, 4, 2]
98           for dt in np.float16, np.float32, np.float64:
99               probs = np.array(counts, dtype=dt) / sum(counts)
100              c = random.choice(a, p=probs)
101              assert_(c in a)
102              assert_raises(ValueError, random.choice, a, p=probs*0.9)
