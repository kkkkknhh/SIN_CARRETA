42   def test_check_grad():
43       # Verify if check_grad is able to estimate the derivative of the
44       # expit (logistic sigmoid) function.
45   
46       def expit(x):
47           return 1 / (1 + np.exp(-x))
48   
49       def der_expit(x):
50           return np.exp(-x) / (1 + np.exp(-x))**2
51   
52       x0 = np.array([1.5])
53   
54       r = optimize.check_grad(expit, der_expit, x0)
55       assert_almost_equal(r, 0)
56       # SPEC-007 leave one call with seed to check it still works
57       r = optimize.check_grad(expit, der_expit, x0,
58                               direction='random', seed=1234)
59       assert_almost_equal(r, 0)
60   
61       r = optimize.check_grad(expit, der_expit, x0, epsilon=1e-6)
62       assert_almost_equal(r, 0)
63       r = optimize.check_grad(expit, der_expit, x0, epsilon=1e-6,
64                               direction='random', rng=1234)
65       assert_almost_equal(r, 0)
66   
67       # Check if the epsilon parameter is being considered.
68       r = abs(optimize.check_grad(expit, der_expit, x0, epsilon=1e-1) - 0)
69       assert r > 1e-7
70       r = abs(optimize.check_grad(expit, der_expit, x0, epsilon=1e-1,
71                                   direction='random', rng=1234) - 0)
72       assert r > 1e-7

102      def setup_method(self):
103          self.F = np.array([[1, 1, 1],
104                             [1, 1, 0],
105                             [1, 0, 1],
106                             [1, 0, 0],
107                             [1, 0, 0]])
108          self.K = np.array([1., 0.3, 0.5])
109          self.startparams = np.zeros(3, np.float64)
110          self.solution = np.array([0., -0.524869316, 0.487525860])
111          self.maxiter = 1000
112          self.funccalls = threading.local()
113          self.gradcalls = threading.local()
114          self.trace = threading.local()
115  
116      def func(self, x):
117          if not hasattr(self.funccalls, 'c'):
118              self.funccalls.c = 0
119  
120          if not hasattr(self.gradcalls, 'c'):
121              self.gradcalls.c = 0
122  
123          self.funccalls.c += 1
124          if self.funccalls.c > 6000:
125              raise RuntimeError("too many iterations in optimization routine")
126          log_pdot = np.dot(self.F, x)
127          logZ = np.log(sum(np.exp(log_pdot)))
128          f = logZ - np.dot(self.K, x)
129          if not hasattr(self.trace, 't'):
130              self.trace.t = []
131          self.trace.t.append(np.copy(x))
132          return f
133  
134      def grad(self, x):
135          if not hasattr(self.gradcalls, 'c'):
136              self.gradcalls.c = 0
137          self.gradcalls.c += 1
138          log_pdot = np.dot(self.F, x)
139          logZ = np.log(sum(np.exp(log_pdot)))
140          p = np.exp(log_pdot - logZ)
141          return np.dot(self.F.transpose(), p) - self.K
142  
143      def hess(self, x):
144          log_pdot = np.dot(self.F, x)
145          logZ = np.log(sum(np.exp(log_pdot)))
146          p = np.exp(log_pdot - logZ)
147          return np.dot(self.F.T,
148                        np.dot(np.diag(p), self.F - np.dot(self.F.T, p)))
149  
150      def hessp(self, x, p):
151          return np.dot(self.hess(x), p)
