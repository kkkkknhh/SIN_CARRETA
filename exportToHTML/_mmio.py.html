31   def mminfo(source):
32       """ 
33       Return size and storage parameters from Matrix Market file-like 'source'. 
34    
35       Parameters 
36       ---------- 
37       source : str or file-like 
38           Matrix Market filename (extension .mtx) or open file-like object 
39    
40       Returns 
41       ------- 
42       rows : int 
43           Number of matrix rows. 
44       cols : int 
45           Number of matrix columns. 
46       entries : int 
47           Number of non-zero entries of a sparse matrix 
48           or rows*cols for a dense matrix. 
49       format : str 
50           Either 'coordinate' or 'array'. 
51       field : str 
52           Either 'real', 'complex', 'pattern', or 'integer'. 
53       symmetry : str 
54           Either 'general', 'symmetric', 'skew-symmetric', or 'hermitian'. 
55    
56       Examples 
57       -------- 
58       >>> from io import StringIO 
59       >>> from scipy.io import mminfo 
60    
61       >>> text = '''%%MatrixMarket matrix coordinate real general 
62       ...  5 5 7 
63       ...  2 3 1.0 
64       ...  3 4 2.0 
65       ...  3 5 3.0 
66       ...  4 1 4.0 
67       ...  4 2 5.0 
68       ...  4 3 6.0 
69       ...  4 4 7.0 
70       ... ''' 
71    
72    
73       ``mminfo(source)`` returns the number of rows, number of columns, 
74       format, field type and symmetry attribute of the source file. 
75    
76       >>> mminfo(StringIO(text)) 
77       (5, 5, 7, 'coordinate', 'real', 'general') 
78       """
79       return MMFile.info(source)

84   def mmread(source, *, spmatrix=True):
85       """ 
86       Reads the contents of a Matrix Market file-like 'source' into a matrix. 
87    
88       Parameters 
89       ---------- 
90       source : str or file-like 
91           Matrix Market filename (extensions .mtx, .mtz.gz) 
92           or open file-like object. 
93       spmatrix : bool, optional (default: True) 
94           If ``True``, return sparse ``coo_matrix``. Otherwise return ``coo_array``. 
95    
96       Returns 
97       ------- 
98       a : ndarray or coo_array or coo_matrix 
99           Dense or sparse array depending on the matrix format in the 
100          Matrix Market file. 
101   
102      Examples 
103      -------- 
104      >>> from io import StringIO 
105      >>> from scipy.io import mmread 
106   
107      >>> text = '''%%MatrixMarket matrix coordinate real general 
108      ...  5 5 7 
109      ...  2 3 1.0 
110      ...  3 4 2.0 
111      ...  3 5 3.0 
112      ...  4 1 4.0 
113      ...  4 2 5.0 
114      ...  4 3 6.0 
115      ...  4 4 7.0 
116      ... ''' 
117   
118      ``mmread(source)`` returns the data as sparse matrix in COO format. 
119   
120      >>> m = mmread(StringIO(text), spmatrix=False) 
121      >>> m 
122      <COOrdinate sparse array of dtype 'float64' 
123           with 7 stored elements and shape (5, 5)> 
124      >>> m.toarray() 
125      array([[0., 0., 0., 0., 0.], 
126             [0., 0., 1., 0., 0.], 
127             [0., 0., 0., 2., 3.], 
128             [4., 5., 6., 7., 0.], 
129             [0., 0., 0., 0., 0.]]) 
130      """
131      return MMFile().read(source, spmatrix=spmatrix)

136  def mmwrite(target, a, comment='', field=None, precision=None, symmetry=None):
137      r""" 
138      Writes the sparse or dense array `a` to Matrix Market file-like `target`. 
139   
140      Parameters 
141      ---------- 
142      target : str or file-like 
143          Matrix Market filename (extension .mtx) or open file-like object. 
144      a : array like 
145          Sparse or dense 2-D array. 
146      comment : str, optional 
147          Comments to be prepended to the Matrix Market file. 
148      field : None or str, optional 
149          Either 'real', 'complex', 'pattern', or 'integer'. 
150      precision : None or int, optional 
151          Number of digits to display for real or complex values. 
152      symmetry : None or str, optional 
153          Either 'general', 'symmetric', 'skew-symmetric', or 'hermitian'. 
154          If symmetry is None the symmetry type of 'a' is determined by its 
155          values. 
156   
157      Returns 
158      ------- 
159      None 
160   
161      Examples 
162      -------- 
163      >>> from io import BytesIO 
164      >>> import numpy as np 
165      >>> from scipy.sparse import coo_array 
166      >>> from scipy.io import mmwrite 
167   
168      Write a small NumPy array to a matrix market file.  The file will be 
169      written in the ``'array'`` format. 
170   
171      >>> a = np.array([[1.0, 0, 0, 0], [0, 2.5, 0, 6.25]]) 
172      >>> target = BytesIO() 
173      >>> mmwrite(target, a) 
174      >>> print(target.getvalue().decode('latin1')) 
175      %%MatrixMarket matrix array real general 
176      % 
177      2 4 
178      1 
179      0 
180      0 
181      2.5 
182      0 
183      0 
184      0 
185      6.25 
186   
187      Add a comment to the output file, and set the precision to 3. 
188   
189      >>> target = BytesIO() 
190      >>> mmwrite(target, a, comment='\n Some test data.\n', precision=3) 
191      >>> print(target.getvalue().decode('latin1')) 
192      %%MatrixMarket matrix array real general 
193      % 
194      % Some test data. 
195      % 
196      2 4 
197      1.00e+00 
198      0.00e+00 
199      0.00e+00 
200      2.50e+00 
201      0.00e+00 
202      0.00e+00 
203      0.00e+00 
204      6.25e+00 
205   
206      Convert to a sparse matrix before calling ``mmwrite``.  This will 
207      result in the output format being ``'coordinate'`` rather than 
208      ``'array'``. 
209   
210      >>> target = BytesIO() 
211      >>> mmwrite(target, coo_array(a), precision=3) 
212      >>> print(target.getvalue().decode('latin1')) 
213      %%MatrixMarket matrix coordinate real general 
214      % 
215      2 4 3 
216      1 1 1.00e+00 
217      2 2 2.50e+00 
218      2 4 6.25e+00 
219   
220      Write a complex Hermitian array to a matrix market file.  Note that 
221      only six values are actually written to the file; the other values 
222      are implied by the symmetry. 
223   
224      >>> z = np.array([[3, 1+2j, 4-3j], [1-2j, 1, -5j], [4+3j, 5j, 2.5]]) 
225      >>> z 
226      array([[ 3. +0.j,  1. +2.j,  4. -3.j], 
227             [ 1. -2.j,  1. +0.j, -0. -5.j], 
228             [ 4. +3.j,  0. +5.j,  2.5+0.j]]) 
229   
230      >>> target = BytesIO() 
231      >>> mmwrite(target, z, precision=2) 
232      >>> print(target.getvalue().decode('latin1')) 
233      %%MatrixMarket matrix array complex hermitian 
234      % 
235      3 3 
236      3.0e+00 0.0e+00 
237      1.0e+00 -2.0e+00 
238      4.0e+00 3.0e+00 
239      1.0e+00 0.0e+00 
240      0.0e+00 5.0e+00 
241      2.5e+00 0.0e+00 
242   
243      """
244      MMFile().write(target, a, comment, field, precision, symmetry)

342      # -------------------------------------------------------------------------
343      @classmethod
344      def info(self, source):
345          """ 
346          Return size, storage parameters from Matrix Market file-like 'source'. 
347   
348          Parameters 
349          ---------- 
350          source : str or file-like 
351              Matrix Market filename (extension .mtx) or open file-like object 
352   
353          Returns 
354          ------- 
355          rows : int 
356              Number of matrix rows. 
357          cols : int 
358              Number of matrix columns. 
359          entries : int 
360              Number of non-zero entries of a sparse matrix 
361              or rows*cols for a dense matrix. 
362          format : str 
363              Either 'coordinate' or 'array'. 
364          field : str 
365              Either 'real', 'complex', 'pattern', or 'integer'. 
366          symmetry : str 
367              Either 'general', 'symmetric', 'skew-symmetric', or 'hermitian'. 
368          """
369  
370          stream, close_it = self._open(source)
371  
372          try:
373  
374              # read and validate header line
375              line = stream.readline()
376              mmid, matrix, format, field, symmetry = \
377                  (asstr(part.strip()) for part in line.split())
378              if not mmid.startswith('%%MatrixMarket'):
379                  raise ValueError('source is not in Matrix Market format')
380              if not matrix.lower() == 'matrix':
381                  raise ValueError("Problem reading file header: " + line)
382  
383              # http://math.nist.gov/MatrixMarket/formats.html
384              if format.lower() == 'array':
385                  format = self.FORMAT_ARRAY
386              elif format.lower() == 'coordinate':
387                  format = self.FORMAT_COORDINATE
388  
389              # skip comments
390              # line.startswith('%')
391              while line:
392                  if line.lstrip() and line.lstrip()[0] in ['%', 37]:
393                      line = stream.readline()
394                  else:
395                      break
396  
397              # skip empty lines
398              while not line.strip():
399                  line = stream.readline()
400  
401              split_line = line.split()
402              if format == self.FORMAT_ARRAY:
403                  if not len(split_line) == 2:
404                      raise ValueError("Header line not of length 2: " +
405                                       line.decode('ascii'))
406                  rows, cols = map(int, split_line)
407                  entries = rows * cols
408              else:
409                  if not len(split_line) == 3:
410                      raise ValueError("Header line not of length 3: " +
411                                       line.decode('ascii'))
412                  rows, cols, entries = map(int, split_line)
413  
414              return (rows, cols, entries, format, field.lower(),
415                      symmetry.lower())
416  
417          finally:
418              if close_it:
419                  stream.close()

421      # -------------------------------------------------------------------------
422      @staticmethod
423      def _open(filespec, mode='rb'):
424          """ Return an open file stream for reading based on source. 
425   
426          If source is a file name, open it (after trying to find it with mtx and 
427          gzipped mtx extensions). Otherwise, just return source. 
428   
429          Parameters 
430          ---------- 
431          filespec : str or file-like 
432              String giving file name or file-like object 
433          mode : str, optional 
434              Mode with which to open file, if `filespec` is a file name. 
435   
436          Returns 
437          ------- 
438          fobj : file-like 
439              Open file-like object. 
440          close_it : bool 
441              True if the calling function should close this file when done, 
442              false otherwise. 
443          """
444          # If 'filespec' is path-like (str, pathlib.Path, os.DirEntry, other class
445          # implementing a '__fspath__' method), try to convert it to str. If this
446          # fails by throwing a 'TypeError', assume it's an open file handle and
447          # return it as-is.
448          try:
449              filespec = os.fspath(filespec)
450          except TypeError:
451              return filespec, False
452  
453          # 'filespec' is definitely a str now
454  
455          # open for reading
456          if mode[0] == 'r':
457  
458              # determine filename plus extension
459              if not os.path.isfile(filespec):
460                  if os.path.isfile(filespec+'.mtx'):
461                      filespec = filespec + '.mtx'
462                  elif os.path.isfile(filespec+'.mtx.gz'):
463                      filespec = filespec + '.mtx.gz'
464                  elif os.path.isfile(filespec+'.mtx.bz2'):
465                      filespec = filespec + '.mtx.bz2'
466              # open filename
467              if filespec.endswith('.gz'):
468                  import gzip
469                  stream = gzip.open(filespec, mode)
470              elif filespec.endswith('.bz2'):
471                  import bz2
472                  stream = bz2.BZ2File(filespec, 'rb')
473              else:
474                  stream = open(filespec, mode)
475  
476          # open for writing
477          else:
478              if filespec[-4:] != '.mtx':
479                  filespec = filespec + '.mtx'
480              stream = open(filespec, mode)
481  
482          return stream, True
483  
484      # -------------------------------------------------------------------------
485      @staticmethod
486      def _get_symmetry(a):
487          m, n = a.shape
488          if m != n:
489              return MMFile.SYMMETRY_GENERAL
490          issymm = True
491          isskew = True
492          isherm = a.dtype.char in 'FD'
493  
494          # sparse input
495          if issparse(a):
496              # check if number of nonzero entries of lower and upper triangle
497              # matrix are equal
498              a = a.tocoo()
499              (row, col) = a.nonzero()
500              if (row < col).sum() != (row > col).sum():
501                  return MMFile.SYMMETRY_GENERAL
502  
503              # define iterator over symmetric pair entries
504              a = a.todok()
505  
506              def symm_iterator():
507                  for ((i, j), aij) in a.items():
508                      if i > j:
509                          aji = a[j, i]
510                          yield (aij, aji, False)
511                      elif i == j:
512                          yield (aij, aij, True)
513  
514          # non-sparse input
515          else:
516              # define iterator over symmetric pair entries
517              def symm_iterator():
518                  for j in range(n):
519                      for i in range(j, n):
520                          aij, aji = a[i][j], a[j][i]
521                          yield (aij, aji, i == j)
522  
523          # check for symmetry
524          # yields aij, aji, is_diagonal
525          for (aij, aji, is_diagonal) in symm_iterator():
526              if isskew and is_diagonal and aij != 0:
527                  isskew = False
528              else:
529                  if issymm and aij != aji:
530                      issymm = False
531                  with np.errstate(over="ignore"):
532                      # This can give a warning for uint dtypes, so silence that
533                      if isskew and aij != -aji:
534                          isskew = False
535                  if isherm and aij != conj(aji):
536                      isherm = False
537              if not (issymm or isskew or isherm):
538                  break
539  
540          # return symmetry value
541          if issymm:
542              return MMFile.SYMMETRY_SYMMETRIC
543          if isskew:
544              return MMFile.SYMMETRY_SKEW_SYMMETRIC
545          if isherm:
546              return MMFile.SYMMETRY_HERMITIAN
547          return MMFile.SYMMETRY_GENERAL

553                  MMFile.FIELD_INTEGER: '%i\n',

563      # -------------------------------------------------------------------------
564      def read(self, source, *, spmatrix=True):
565          """ 
566          Reads the contents of a Matrix Market file-like 'source' into a matrix. 
567   
568          Parameters 
569          ---------- 
570          source : str or file-like 
571              Matrix Market filename (extensions .mtx, .mtz.gz) 
572              or open file object. 
573          spmatrix : bool, optional (default: True) 
574              If ``True``, return sparse ``coo_matrix``. Otherwise return ``coo_array``. 
575   
576          Returns 
577          ------- 
578          a : ndarray or coo_array or coo_matrix 
579              Dense or sparse array depending on the matrix format in the 
580              Matrix Market file. 
581          """
582          stream, close_it = self._open(source)
583  
584          try:
585              self._parse_header(stream)
586              data = self._parse_body(stream)
587  
588          finally:
589              if close_it:
590                  stream.close()
591          if spmatrix and isinstance(data, coo_array):
592              data = coo_matrix(data)
593          return data

597      def write(self, target, a, comment='', field=None, precision=None,
598                symmetry=None):
599          """ 
600          Writes sparse or dense array `a` to Matrix Market file-like `target`. 
601   
602          Parameters 
603          ---------- 
604          target : str or file-like 
605              Matrix Market filename (extension .mtx) or open file-like object. 
606          a : array like 
607              Sparse or dense 2-D array. 
608          comment : str, optional 
609              Comments to be prepended to the Matrix Market file. 
610          field : None or str, optional 
611              Either 'real', 'complex', 'pattern', or 'integer'. 
612          precision : None or int, optional 
613              Number of digits to display for real or complex values. 
614          symmetry : None or str, optional 
615              Either 'general', 'symmetric', 'skew-symmetric', or 'hermitian'. 
616              If symmetry is None the symmetry type of 'a' is determined by its 
617              values. 
618          """
619  
620          stream, close_it = self._open(target, 'wb')
621  
622          try:
623              self._write(stream, a, comment, field, precision, symmetry)
624  
625          finally:
626              if close_it:
627                  stream.close()
628              else:
629                  stream.flush()

631      # -------------------------------------------------------------------------
632      def _init_attrs(self, **kwargs):
633          """ 
634          Initialize each attributes with the corresponding keyword arg value 
635          or a default of None 
636          """
637  
638          attrs = self.__class__.__slots__
639          public_attrs = [attr[1:] for attr in attrs]
640          invalid_keys = set(kwargs.keys()) - set(public_attrs)
641  
642          if invalid_keys:
643              raise ValueError(f"found {tuple(invalid_keys)} invalid keyword "
644                               f"arguments, please only use {public_attrs}")
645  
646          for attr in attrs:
647              setattr(self, attr, kwargs.get(attr[1:], None))
648  
649      # -------------------------------------------------------------------------
650      def _parse_header(self, stream):
651          rows, cols, entries, format, field, symmetry = \
652              self.__class__.info(stream)
653          self._init_attrs(rows=rows, cols=cols, entries=entries, format=format,
654                           field=field, symmetry=symmetry)
655  
656      # -------------------------------------------------------------------------
657      def _parse_body(self, stream):
658          rows, cols, entries, format, field, symm = (self.rows, self.cols,
659                                                      self.entries, self.format,
660                                                      self.field, self.symmetry)
661  
662          dtype = self.DTYPES_BY_FIELD.get(field, None)
663  
664          has_symmetry = self.has_symmetry
665          is_integer = field == self.FIELD_INTEGER
666          is_unsigned_integer = field == self.FIELD_UNSIGNED
667          is_complex = field == self.FIELD_COMPLEX
668          is_skew = symm == self.SYMMETRY_SKEW_SYMMETRIC
669          is_herm = symm == self.SYMMETRY_HERMITIAN
670          is_pattern = field == self.FIELD_PATTERN
671  
672          if format == self.FORMAT_ARRAY:
673              a = zeros((rows, cols), dtype=dtype)
674              line = 1
675              i, j = 0, 0
676              if is_skew:
677                  a[i, j] = 0
678                  if i < rows - 1:
679                      i += 1
680              while line:
681                  line = stream.readline()
682                  # line.startswith('%')
683                  if not line or line[0] in ['%', 37] or not line.strip():
684                      continue
685                  if is_integer:
686                      aij = int(line)
687                  elif is_unsigned_integer:
688                      aij = int(line)
689                  elif is_complex:
690                      aij = complex(*map(float, line.split()))
691                  else:
692                      aij = float(line)
693                  a[i, j] = aij
694                  if has_symmetry and i != j:
695                      if is_skew:
696                          a[j, i] = -aij
697                      elif is_herm:
698                          a[j, i] = conj(aij)
699                      else:
700                          a[j, i] = aij
701                  if i < rows-1:
702                      i = i + 1
703                  else:
704                      j = j + 1
705                      if not has_symmetry:
706                          i = 0
707                      else:
708                          i = j
709                          if is_skew:
710                              a[i, j] = 0
711                              if i < rows-1:
712                                  i += 1
713  
714              if is_skew:
715                  if not (i in [0, j] and j == cols - 1):
716                      raise ValueError("Parse error, did not read all lines.")
717              else:
718                  if not (i in [0, j] and j == cols):
719                      raise ValueError("Parse error, did not read all lines.")
720  
721          elif format == self.FORMAT_COORDINATE:
722              # Read sparse COOrdinate format
723  
724              if entries == 0:
725                  # empty matrix
726                  return coo_array((rows, cols), dtype=dtype)
727  
728              I = zeros(entries, dtype='intc')
729              J = zeros(entries, dtype='intc')
730              if is_pattern:
731                  V = ones(entries, dtype='int8')
732              elif is_integer:
733                  V = zeros(entries, dtype='intp')
734              elif is_unsigned_integer:
735                  V = zeros(entries, dtype='uint64')
736              elif is_complex:
737                  V = zeros(entries, dtype='complex')
738              else:
739                  V = zeros(entries, dtype='float')
740  
741              entry_number = 0
742              for line in stream:
743                  # line.startswith('%')
744                  if not line or line[0] in ['%', 37] or not line.strip():
745                      continue
746  
747                  if entry_number+1 > entries:
748                      raise ValueError("'entries' in header is smaller than "
749                                       "number of entries")
750                  l = line.split()
751                  I[entry_number], J[entry_number] = map(int, l[:2])
752  
753                  if not is_pattern:
754                      if is_integer:
755                          V[entry_number] = int(l[2])
756                      elif is_unsigned_integer:
757                          V[entry_number] = int(l[2])
758                      elif is_complex:
759                          V[entry_number] = complex(*map(float, l[2:]))
760                      else:
761                          V[entry_number] = float(l[2])
762                  entry_number += 1
763              if entry_number < entries:
764                  raise ValueError("'entries' in header is larger than "
765                                   "number of entries")
766  
767              I -= 1  # adjust indices (base 1 -> base 0)
768              J -= 1
769  
770              if has_symmetry:
771                  mask = (I != J)       # off diagonal mask
772                  od_I = I[mask]
773                  od_J = J[mask]
774                  od_V = V[mask]
775  
776                  I = concatenate((I, od_J))
777                  J = concatenate((J, od_I))
778  
779                  if is_skew:
780                      od_V *= -1
781                  elif is_herm:
782                      od_V = od_V.conjugate()
783  
784                  V = concatenate((V, od_V))
785  
786              a = coo_array((V, (I, J)), shape=(rows, cols), dtype=dtype)
787          else:
788              raise NotImplementedError(format)
789  
790          return a
791  
792      #  ------------------------------------------------------------------------
793      def _write(self, stream, a, comment='', field=None, precision=None,
794                 symmetry=None):
795          if isinstance(a, list) or isinstance(a, ndarray) or \
796             isinstance(a, tuple) or hasattr(a, '__array__'):
797              rep = self.FORMAT_ARRAY
798              a = asarray(a)
799              if len(a.shape) != 2:
800                  raise ValueError('Expected 2 dimensional array')
801              rows, cols = a.shape
802  
803              if field is not None:
804  
805                  if field == self.FIELD_INTEGER:
806                      if not can_cast(a.dtype, 'intp'):
807                          raise OverflowError("mmwrite does not support integer "
808                                              "dtypes larger than native 'intp'.")
809                      a = a.astype('intp')
810                  elif field == self.FIELD_REAL:
811                      if a.dtype.char not in 'fd':
812                          a = a.astype('d')
813                  elif field == self.FIELD_COMPLEX:
814                      if a.dtype.char not in 'FD':
815                          a = a.astype('D')
816  
817          else:
818              if not issparse(a):
819                  raise ValueError(f'unknown matrix type: {type(a)}')
820  
821              rep = 'coordinate'
822              rows, cols = a.shape
823  
824          typecode = a.dtype.char
825  
826          if precision is None:
827              if typecode in 'fF':
828                  precision = 8
829              else:
830                  precision = 16
831          if field is None:
832              kind = a.dtype.kind
833              if kind == 'i':
834                  if not can_cast(a.dtype, 'intp'):
835                      raise OverflowError("mmwrite does not support integer "
836                                          "dtypes larger than native 'intp'.")
837                  field = 'integer'
838              elif kind == 'f':
839                  field = 'real'
840              elif kind == 'c':
841                  field = 'complex'
842              elif kind == 'u':
843                  field = 'unsigned-integer'
844              else:
845                  raise TypeError('unexpected dtype kind ' + kind)
846  
847          if symmetry is None:
848              symmetry = self._get_symmetry(a)
849  
850          # validate rep, field, and symmetry
851          self.__class__._validate_format(rep)
852          self.__class__._validate_field(field)
853          self.__class__._validate_symmetry(symmetry)
854  
855          # write initial header line
856          data = f'%%MatrixMarket matrix {rep} {field} {symmetry}\n'
857          stream.write(data.encode('latin1'))
858  
859          # write comments
860          for line in comment.split('\n'):
861              data = f'%{line}\n'
862              stream.write(data.encode('latin1'))
863  
864          template = self._field_template(field, precision)
865          # write dense format
866          if rep == self.FORMAT_ARRAY:
867              # write shape spec
868              data = '%i %i\n' % (rows, cols)
869              stream.write(data.encode('latin1'))
870  
871              if field in (self.FIELD_INTEGER, self.FIELD_REAL,
872                           self.FIELD_UNSIGNED):
873                  if symmetry == self.SYMMETRY_GENERAL:
874                      for j in range(cols):
875                          for i in range(rows):
876                              data = template % a[i, j]
877                              stream.write(data.encode('latin1'))
878  
879                  elif symmetry == self.SYMMETRY_SKEW_SYMMETRIC:
880                      for j in range(cols):
881                          for i in range(j + 1, rows):
882                              data = template % a[i, j]
883                              stream.write(data.encode('latin1'))
884  
885                  else:
886                      for j in range(cols):
887                          for i in range(j, rows):
888                              data = template % a[i, j]
889                              stream.write(data.encode('latin1'))
890  
891              elif field == self.FIELD_COMPLEX:
892  
893                  if symmetry == self.SYMMETRY_GENERAL:
894                      for j in range(cols):
895                          for i in range(rows):
896                              aij = a[i, j]
897                              data = template % (real(aij), imag(aij))
898                              stream.write(data.encode('latin1'))
899                  else:
900                      for j in range(cols):
901                          for i in range(j, rows):
902                              aij = a[i, j]
903                              data = template % (real(aij), imag(aij))
904                              stream.write(data.encode('latin1'))
905  
906              elif field == self.FIELD_PATTERN:
907                  raise ValueError('pattern type inconsisted with dense format')
908  
909              else:
910                  raise TypeError(f'Unknown field type {field}')
911  
912          # write sparse format
913          else:
914              coo = a.tocoo()  # convert to COOrdinate format
915  
916              # if symmetry format used, remove values above main diagonal
917              if symmetry != self.SYMMETRY_GENERAL:
918                  lower_triangle_mask = coo.row >= coo.col
919                  coo = coo_array((coo.data[lower_triangle_mask],
920                                  (coo.row[lower_triangle_mask],
921                                   coo.col[lower_triangle_mask])),
922                                  shape=coo.shape)
923  
924              # write shape spec
925              data = '%i %i %i\n' % (rows, cols, coo.nnz)
926              stream.write(data.encode('latin1'))
927  
928              template = self._field_template(field, precision-1)
929  
930              if field == self.FIELD_PATTERN:
931                  for r, c in zip(coo.row+1, coo.col+1):
932                      data = "%i %i\n" % (r, c)
933                      stream.write(data.encode('latin1'))
934              elif field in (self.FIELD_INTEGER, self.FIELD_REAL,
935                             self.FIELD_UNSIGNED):
936                  for r, c, d in zip(coo.row+1, coo.col+1, coo.data):
937                      data = ("%i %i " % (r, c)) + (template % d)
938                      stream.write(data.encode('latin1'))
939              elif field == self.FIELD_COMPLEX:
940                  for r, c, d in zip(coo.row+1, coo.col+1, coo.data):
941                      data = ("%i %i " % (r, c)) + (template % (d.real, d.imag))
942                      stream.write(data.encode('latin1'))
943              else:
944                  raise TypeError(f'Unknown field type {field}')

947  def _is_fromfile_compatible(stream):
948      """ 
949      Check whether `stream` is compatible with numpy.fromfile. 
950   
951      Passing a gzipped file object to ``fromfile/fromstring`` doesn't work with 
952      Python 3. 
953      """
954  
955      bad_cls = []
956      try:
957          import gzip
958          bad_cls.append(gzip.GzipFile)
959      except ImportError:
960          pass
961      try:
962          import bz2
963          bad_cls.append(bz2.BZ2File)
964      except ImportError:
965          pass
966  
967      bad_cls = tuple(bad_cls)
968      return not isinstance(stream, bad_cls)
