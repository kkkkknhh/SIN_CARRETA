34   def _check_system(system):
35       """Function to check whether the dynamical system passed for plots is 
36       compatible or not."""
37       if not isinstance(system, SISOLinearTimeInvariant):
38           raise NotImplementedError("Only SISO LTI systems are currently supported.")
39       sys = system.to_expr()
40       len_free_symbols = len(sys.free_symbols)
41       if len_free_symbols > 1:
42           raise ValueError("Extra degree of freedom found. Make sure"
43               " that there are no free symbols in the dynamical system other"
44               " than the variable of Laplace transform.")
45       if sys.has(exp):
46           # Should test that exp is not part of a constant, in which case
47           # no exception is required, compare exp(s) with s*exp(1)
48           raise NotImplementedError("Time delay terms are not supported.")

51   def _poly_roots(poly):
52       """Function to get the roots of a polynomial."""
53       def _eval(l):
54           return [float(i) if i.is_real else complex(i) for i in l]
55       if poly.domain in (QQ, ZZ):
56           return _eval(poly.all_roots())
57       # XXX: Use all_roots() for irrational coefficients when possible
58       # See https://github.com/sympy/sympy/issues/22943
59       return _eval(poly.nroots())

62   def pole_zero_numerical_data(system):
63       """ 
64       Returns the numerical data of poles and zeros of the system. 
65       It is internally used by ``pole_zero_plot`` to get the data 
66       for plotting poles and zeros. Users can use this data to further 
67       analyse the dynamics of the system or plot using a different 
68       backend/plotting-module. 
69    
70       Parameters 
71       ========== 
72    
73       system : SISOLinearTimeInvariant 
74           The system for which the pole-zero data is to be computed. 
75    
76       Returns 
77       ======= 
78    
79       tuple : (zeros, poles) 
80           zeros = Zeros of the system as a list of Python float/complex. 
81           poles = Poles of the system as a list of Python float/complex. 
82    
83       Raises 
84       ====== 
85    
86       NotImplementedError 
87           When a SISO LTI system is not passed. 
88    
89           When time delay terms are present in the system. 
90    
91       ValueError 
92           When more than one free symbol is present in the system. 
93           The only variable in the transfer function should be 
94           the variable of the Laplace transform. 
95    
96       Examples 
97       ======== 
98    
99       >>> from sympy.abc import s 
100      >>> from sympy.physics.control.lti import TransferFunction 
101      >>> from sympy.physics.control.control_plots import pole_zero_numerical_data 
102      >>> tf1 = TransferFunction(s**2 + 1, s**4 + 4*s**3 + 6*s**2 + 5*s + 2, s) 
103      >>> pole_zero_numerical_data(tf1) 
104      ([-1j, 1j], [-2.0, -1.0, (-0.5-0.8660254037844386j), (-0.5+0.8660254037844386j)]) 
105   
106      See Also 
107      ======== 
108   
109      pole_zero_plot 
110   
111      """
112      _check_system(system)
113      system = system.doit()  # Get the equivalent TransferFunction object.
114  
115      num_poly = Poly(system.num, system.var)
116      den_poly = Poly(system.den, system.var)
117  
118      return _poly_roots(num_poly), _poly_roots(den_poly)

121  def pole_zero_plot(system, pole_color='blue', pole_markersize=10,
122      zero_color='orange', zero_markersize=7, grid=True, show_axes=True,
123      show=True, **kwargs):
124      r""" 
125      Returns the Pole-Zero plot (also known as PZ Plot or PZ Map) of a system. 
126   
127      A Pole-Zero plot is a graphical representation of a system's poles and 
128      zeros. It is plotted on a complex plane, with circular markers representing 
129      the system's zeros and 'x' shaped markers representing the system's poles. 
130   
131      Parameters 
132      ========== 
133   
134      system : SISOLinearTimeInvariant type systems 
135          The system for which the pole-zero plot is to be computed. 
136      pole_color : str, tuple, optional 
137          The color of the pole points on the plot. Default color 
138          is blue. The color can be provided as a matplotlib color string, 
139          or a 3-tuple of floats each in the 0-1 range. 
140      pole_markersize : Number, optional 
141          The size of the markers used to mark the poles in the plot. 
142          Default pole markersize is 10. 
143      zero_color : str, tuple, optional 
144          The color of the zero points on the plot. Default color 
145          is orange. The color can be provided as a matplotlib color string, 
146          or a 3-tuple of floats each in the 0-1 range. 
147      zero_markersize : Number, optional 
148          The size of the markers used to mark the zeros in the plot. 
149          Default zero markersize is 7. 
150      grid : boolean, optional 
151          If ``True``, the plot will have a grid. Defaults to True. 
152      show_axes : boolean, optional 
153          If ``True``, the coordinate axes will be shown. Defaults to False. 
154      show : boolean, optional 
155          If ``True``, the plot will be displayed otherwise 
156          the equivalent matplotlib ``plot`` object will be returned. 
157          Defaults to True. 
158   
159      Examples 
160      ======== 
161   
162      .. plot:: 
163          :context: close-figs 
164          :format: doctest 
165          :include-source: True 
166   
167          >>> from sympy.abc import s 
168          >>> from sympy.physics.control.lti import TransferFunction 
169          >>> from sympy.physics.control.control_plots import pole_zero_plot 
170          >>> tf1 = TransferFunction(s**2 + 1, s**4 + 4*s**3 + 6*s**2 + 5*s + 2, s) 
171          >>> pole_zero_plot(tf1)   # doctest: +SKIP 
172   
173      See Also 
174      ======== 
175   
176      pole_zero_numerical_data 
177   
178      References 
179      ========== 
180   
181      .. [1] https://en.wikipedia.org/wiki/Pole%E2%80%93zero_plot 
182   
183      """
184      zeros, poles = pole_zero_numerical_data(system)
185  
186      zero_real = [i.real for i in zeros]
187      zero_imag = [i.imag for i in zeros]
188  
189      pole_real = [i.real for i in poles]
190      pole_imag = [i.imag for i in poles]
191  
192      plt.plot(pole_real, pole_imag, 'x', mfc='none',
193          markersize=pole_markersize, color=pole_color)
194      plt.plot(zero_real, zero_imag, 'o', markersize=zero_markersize,
195          color=zero_color)
196      plt.xlabel('Real Axis')
197      plt.ylabel('Imaginary Axis')
198      plt.title(f'Poles and Zeros of ${latex(system)}$', pad=20)
199  
200      if grid:
201          plt.grid()
202      if show_axes:
203          plt.axhline(0, color='black')
204          plt.axvline(0, color='black')
205      if show:
206          plt.show()
207          return
208  
209      return plt

212  def step_response_numerical_data(system, prec=8, lower_limit=0,
213      upper_limit=10, **kwargs):
214      """ 
215      Returns the numerical values of the points in the step response plot 
216      of a SISO continuous-time system. By default, adaptive sampling 
217      is used. If the user wants to instead get an uniformly 
218      sampled response, then ``adaptive`` kwarg should be passed ``False`` 
219      and ``n`` must be passed as additional kwargs. 
220      Refer to the parameters of class :class:`sympy.plotting.series.LineOver1DRangeSeries` 
221      for more details. 
222   
223      Parameters 
224      ========== 
225   
226      system : SISOLinearTimeInvariant 
227          The system for which the unit step response data is to be computed. 
228      prec : int, optional 
229          The decimal point precision for the point coordinate values. 
230          Defaults to 8. 
231      lower_limit : Number, optional 
232          The lower limit of the plot range. Defaults to 0. 
233      upper_limit : Number, optional 
234          The upper limit of the plot range. Defaults to 10. 
235      kwargs : 
236          Additional keyword arguments are passed to the underlying 
237          :class:`sympy.plotting.series.LineOver1DRangeSeries` class. 
238   
239      Returns 
240      ======= 
241   
242      tuple : (x, y) 
243          x = Time-axis values of the points in the step response. NumPy array. 
244          y = Amplitude-axis values of the points in the step response. NumPy array. 
245   
246      Raises 
247      ====== 
248   
249      NotImplementedError 
250          When a SISO LTI system is not passed. 
251   
252          When time delay terms are present in the system. 
253   
254      ValueError 
255          When more than one free symbol is present in the system. 
256          The only variable in the transfer function should be 
257          the variable of the Laplace transform. 
258   
259          When ``lower_limit`` parameter is less than 0. 
260   
261      Examples 
262      ======== 
263   
264      >>> from sympy.abc import s 
265      >>> from sympy.physics.control.lti import TransferFunction 
266      >>> from sympy.physics.control.control_plots import step_response_numerical_data 
267      >>> tf1 = TransferFunction(s, s**2 + 5*s + 8, s) 
268      >>> step_response_numerical_data(tf1)   # doctest: +SKIP 
269      ([0.0, 0.025413462339411542, 0.0484508722725343, ... , 9.670250533855183, 9.844291913708725, 10.0], 
270      [0.0, 0.023844582399907256, 0.042894276802320226, ..., 6.828770759094287e-12, 6.456457160755703e-12]) 
271   
272      See Also 
273      ======== 
274   
275      step_response_plot 
276   
277      """
278      if lower_limit < 0:
279          raise ValueError("Lower limit of time must be greater "
280              "than or equal to zero.")
281      _check_system(system)
282      _x = Dummy("x")
283      expr = system.to_expr()/(system.var)
284      expr = apart(expr, system.var, full=True)
285      _y = _fast_inverse_laplace(expr, system.var, _x).evalf(prec)
286      return LineOver1DRangeSeries(_y, (_x, lower_limit, upper_limit),
287          **kwargs).get_points()

518      of a SISO continuous-time system. By default, adaptive sampling

527

530      slope : Number, optional

532      prec : int, optional 
533          The decimal point precision for the point coordinate values. 
534          Defaults to 8. 
535      lower_limit : Number, optional 
536          The lower limit of the plot range. Defaults to 0. 
537      upper_limit : Number, optional 
538          The upper limit of the plot range. Defaults to 10.

541          :class:`sympy.plotting.series.LineOver1DRangeSeries` class.

545   
546      tuple : (x, y) 
547          x = Time-axis values of the points in the ramp response plot. NumPy array. 
548          y = Amplitude-axis values of the points in the ramp response plot. NumPy array. 
549   
550      Raises 
551      ====== 
552   
553      NotImplementedError 
554          When a SISO LTI system is not passed. 
555   
556          When time delay terms are present in the system. 
557   
558      ValueError 
559          When more than one free symbol is present in the system. 
560          The only variable in the transfer function should be 
561          the variable of the Laplace transform.

566

568      ======== 
569

571      >>> from sympy.physics.control.lti import TransferFunction 
572      >>> from sympy.physics.control.control_plots import ramp_response_numerical_data

595      return LineOver1DRangeSeries(_y, (_x, lower_limit, upper_limit),
596          **kwargs).get_points()

627          The decimal point precision for the point coordinate values. 
628          Defaults to 8. 
629      show_axes : boolean, optional

678  def bode_magnitude_numerical_data(system, initial_exp=-5, final_exp=5, freq_unit='rad/sec', **kwargs):
679      """ 
680      Returns the numerical data of the Bode magnitude plot of the system. 
681      It is internally used by ``bode_magnitude_plot`` to get the data 
682      for plotting Bode magnitude plot. Users can use this data to further 
683      analyse the dynamics of the system or plot using a different 
684      backend/plotting-module. 
685   
686      Parameters 
687      ========== 
688   
689      system : SISOLinearTimeInvariant 
690          The system for which the data is to be computed. 
691      initial_exp : Number, optional 
692          The initial exponent of 10 of the semilog plot. Defaults to -5. 
693      final_exp : Number, optional 
694          The final exponent of 10 of the semilog plot. Defaults to 5. 
695      freq_unit : string, optional 
696          User can choose between ``'rad/sec'`` (radians/second) and ``'Hz'`` (Hertz) as frequency units. 
697   
698      Returns 
699      ======= 
700   
701      tuple : (x, y) 
702          x = x-axis values of the Bode magnitude plot. 
703          y = y-axis values of the Bode magnitude plot. 
704   
705      Raises 
706      ====== 
707   
708      NotImplementedError 
709          When a SISO LTI system is not passed. 
710   
711          When time delay terms are present in the system. 
712   
713      ValueError 
714          When more than one free symbol is present in the system. 
715          The only variable in the transfer function should be 
716          the variable of the Laplace transform. 
717   
718          When incorrect frequency units are given as input. 
719   
720      Examples 
721      ======== 
722   
723      >>> from sympy.abc import s 
724      >>> from sympy.physics.control.lti import TransferFunction 
725      >>> from sympy.physics.control.control_plots import bode_magnitude_numerical_data 
726      >>> tf1 = TransferFunction(s**2 + 1, s**4 + 4*s**3 + 6*s**2 + 5*s + 2, s) 
727      >>> bode_magnitude_numerical_data(tf1)   # doctest: +SKIP 
728      ([1e-05, 1.5148378120533502e-05,..., 68437.36188804005, 100000.0], 
729      [-6.020599914256786, -6.0205999155219505,..., -193.4117304087953, -200.00000000260573]) 
730   
731      See Also 
732      ======== 
733   
734      bode_magnitude_plot, bode_phase_numerical_data 
735   
736      """
737      _check_system(system)
738      expr = system.to_expr()
739      freq_units = ('rad/sec', 'Hz')
740      if freq_unit not in freq_units:
741          raise ValueError('Only "rad/sec" and "Hz" are accepted frequency units.')
742  
743      _w = Dummy("w", real=True)
744      if freq_unit == 'Hz':
745          repl = I*_w*2*pi
746      else:
747          repl = I*_w
748      w_expr = expr.subs({system.var: repl})
749  
750      mag = 20*log(Abs(w_expr), 10)
751  
752      x, y = LineOver1DRangeSeries(mag,
753          (_w, 10**initial_exp, 10**final_exp), xscale='log', **kwargs).get_points()
754  
755      return x, y

787  def bode_phase_numerical_data(system, initial_exp=-5, final_exp=5, freq_unit='rad/sec', phase_unit='rad', phase_unwrap = True, **kwargs):
788      """ 
789      Returns the numerical data of the Bode phase plot of the system. 
790      It is internally used by ``bode_phase_plot`` to get the data 
791      for plotting Bode phase plot. Users can use this data to further 
792      analyse the dynamics of the system or plot using a different 
793      backend/plotting-module. 
794   
795      Parameters 
796      ========== 
797   
798      system : SISOLinearTimeInvariant 
799          The system for which the Bode phase plot data is to be computed. 
800      initial_exp : Number, optional 
801          The initial exponent of 10 of the semilog plot. Defaults to -5. 
802      final_exp : Number, optional 
803          The final exponent of 10 of the semilog plot. Defaults to 5. 
804      freq_unit : string, optional 
805          User can choose between ``'rad/sec'`` (radians/second) and '``'Hz'`` (Hertz) as frequency units. 
806      phase_unit : string, optional 
807          User can choose between ``'rad'`` (radians) and ``'deg'`` (degree) as phase units. 
808      phase_unwrap : bool, optional 
809          Set to ``True`` by default. 
810   
811      Returns 
812      ======= 
813   
814      tuple : (x, y) 
815          x = x-axis values of the Bode phase plot. 
816          y = y-axis values of the Bode phase plot. 
817   
818      Raises 
819      ====== 
820   
821      NotImplementedError 
822          When a SISO LTI system is not passed. 
823   
824          When time delay terms are present in the system. 
825   
826      ValueError 
827          When more than one free symbol is present in the system. 
828          The only variable in the transfer function should be 
829          the variable of the Laplace transform. 
830   
831          When incorrect frequency or phase units are given as input. 
832   
833      Examples 
834      ======== 
835   
836      >>> from sympy.abc import s 
837      >>> from sympy.physics.control.lti import TransferFunction 
838      >>> from sympy.physics.control.control_plots import bode_phase_numerical_data 
839      >>> tf1 = TransferFunction(s**2 + 1, s**4 + 4*s**3 + 6*s**2 + 5*s + 2, s) 
840      >>> bode_phase_numerical_data(tf1)   # doctest: +SKIP 
841      ([1e-05, 1.4472354033813751e-05, 2.035581932165858e-05,..., 47577.3248186011, 67884.09326036123, 100000.0], 
842      [-2.5000000000291665e-05, -3.6180885085e-05, -5.08895483066e-05,...,-3.1415085799262523, -3.14155265358979]) 
843   
844      See Also 
845      ======== 
846   
847      bode_magnitude_plot, bode_phase_numerical_data 
848   
849      """
850      _check_system(system)
851      expr = system.to_expr()
852      freq_units = ('rad/sec', 'Hz')
853      phase_units = ('rad', 'deg')
854      if freq_unit not in freq_units:
855          raise ValueError('Only "rad/sec" and "Hz" are accepted frequency units.')
856      if phase_unit not in phase_units:
857          raise ValueError('Only "rad" and "deg" are accepted phase units.')
858  
859      _w = Dummy("w", real=True)
860      if freq_unit == 'Hz':
861          repl = I*_w*2*pi
862      else:
863          repl = I*_w
864      w_expr = expr.subs({system.var: repl})
865  
866      if phase_unit == 'deg':
867          phase = arg(w_expr)*180/pi
868      else:
869          phase = arg(w_expr)
870  
871      x, y = LineOver1DRangeSeries(phase,
872          (_w, 10**initial_exp, 10**final_exp), xscale='log', **kwargs).get_points()
873  
874      half = None
875      if phase_unwrap:
876          if(phase_unit == 'rad'):
877              half = pi
878          elif(phase_unit == 'deg'):
879              half = 180
880      if half:
881          unit = 2*half
882          for i in range(1, len(y)):
883              diff = y[i] - y[i - 1]
884              if diff > half:      # Jump from -half to half
885                  y[i] = (y[i] - unit)
886              elif diff < -half:   # Jump from half to -half
887                  y[i] = (y[i] + unit)
888  
889      return x, y
890

987  def nyquist_plot_expr(system):
988      """Function to get the expression for Nyquist plot."""
989      s = system.var
990      w = Dummy('w', real=True)
991      repl = I * w
992      expr = system.to_expr()
993      w_expr = expr.subs({s: repl})
994      w_expr = w_expr.as_real_imag()
995      real_expr = w_expr[0]
996      imag_expr = w_expr[1]
997      return real_expr, imag_expr, w

1000 def nichols_plot_expr(system):
1001     """Function to get the expression for Nichols plot."""
1002     s = system.var
1003     w = Dummy('w', real=True)
1004     sys_expr = system.to_expr()
1005     H_jw = sys_expr.subs(s, I*w)
1006     mag_expr = Abs(H_jw)
1007     mag_dB_expr = 20*log(mag_expr, 10)
1008     phase_expr = arg(H_jw)
1009     phase_deg_expr = deg(phase_expr)
1010     return mag_dB_expr, phase_deg_expr, w
