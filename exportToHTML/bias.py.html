34   class CausalVariant(IntEnum):
35       r""" 
36       Enum for causal variants used in attention mechanisms. 
37    
38       Defines two types of causal biases: 
39    
40       ``UPPER_LEFT``: Represents upper-left triangular bias for standard causal attention. 
41       The equivalent pytorch code for constructing this bias is: 
42    
43       .. code-block:: python 
44    
45           torch.tril(torch.ones(size, dtype=torch.bool)) 
46    
47       For instance, with ``shape=(3,4)``, the materialized bias tensor will be: 
48    
49       .. code-block:: text 
50    
51           [[1, 0, 0, 0], 
52            [1, 1, 0, 0], 
53            [1, 1, 1, 0]] 
54    
55    
56       ``LOWER_RIGHT``: Represents lower-right triangular bias, the include values are aligned to the lower 
57       right corner of the matrix. 
58    
59       The equivalent pytorch code for constructing this bias is: 
60    
61       .. code-block:: python 
62    
63           diagonal_offset = size[1] - size[0] 
64           torch.tril( 
65               torch.ones(size, dtype=torch.bool), 
66               diagonal=diagonal_offset, 
67           ) 
68    
69       For instance, with ``shape=(3,4)``, the materialized bias tensor will be: 
70    
71       .. code-block:: text 
72    
73           [[1, 1, 0, 0], 
74            [1, 1, 1, 0], 
75            [1, 1, 1, 1]] 
76    
77       Note that these variants are equivalent to each other when the sequence lengths of the query and key/value 
78       tensors are equal since the triangular matrix is square. 
79    
80       .. warning:: This enum is a prototype and subject to change. 
81       """
82   
83       UPPER_LEFT = auto()
84       LOWER_RIGHT = auto()

87   class CausalBias(torch.Tensor):
88       """ 
89       A bias representing causal attention patterns. For an overview of the bias structure, see the :class:`CausalVariant` enum. 
90    
91       This class is used for defining causal (triangular) attention biases. For construing the bias, there exist 
92       two factory functions: :func:`causal_upper_left` and :func:`causal_lower_right`. 
93    
94       Example: 
95    
96       .. code-block:: python 
97    
98           from torch.nn.attention.bias import causal_lower_right 
99    
100          bsz, num_heads, seqlen_q, seqlen_kv, head_dim = 32, 8, 4, 12, 8 
101   
102          # Create a lower-right causal bias 
103          attn_bias = causal_lower_right(seqlen_q, seqlen_kv) 
104   
105          q = torch.randn( 
106              bsz, num_heads, seqlen_q, head_dim, device="cuda", dtype=torch.float16 
107          ) 
108          k = torch.randn( 
109              bsz, num_heads, seqlen_kv, head_dim, device="cuda", dtype=torch.float16 
110          ) 
111          v = torch.randn( 
112              bsz, num_heads, seqlen_kv, head_dim, device="cuda", dtype=torch.float16 
113          ) 
114   
115          out = F.scaled_dot_product_attention(q, k, v, attn_bias) 
116   
117      .. warning:: This class is a prototype and subject to change. 
118      """
119  
120      def __init__(self, variant: CausalVariant, seq_len_q: int, seq_len_kv: int):
121          """ 
122          Initializes the CausalBias instance with a specified variant and sequence lengths. 
123   
124          Args: 
125              variant (CausalVariant): The type of causal bias to use (either UPPER_LEFT or LOWER_RIGHT). 
126              seq_len_q (int): The sequence length of the query tensor. 
127              seq_len_kv (int): The sequence length of the key/value tensor. 
128   
129          Raises a warning if the LOWER_RIGHT variant is used with seq_len_q > seq_len_kv, as it may produce NaNs. 
130          """
131          assert isinstance(variant, CausalVariant)
132          self.variant = variant
133          self.seq_len_q = seq_len_q
134          self.seq_len_kv = seq_len_kv
135          if seq_len_q > seq_len_kv and variant == CausalVariant.LOWER_RIGHT:
136              warn(
137                  "Lower right causal bias will produce NaNs in the output when seq_len_q > seq_len_kv!"
138              )
139  
140      def _upper_left(self, device: torch.device) -> torch.Tensor:
141          """Upper left causal bias"""
142          return torch.tril(
143              torch.ones(self.seq_len_q, self.seq_len_kv, device=device, dtype=torch.bool)
144          )
145  
146      def _lower_right(self, device: torch.device) -> torch.Tensor:
147          """Lower right causal bias"""
148          diagonal_offset = self.seq_len_kv - self.seq_len_q
149          return torch.tril(
150              torch.ones(
151                  self.seq_len_q, self.seq_len_kv, device=device, dtype=torch.bool
152              ),
153              diagonal=diagonal_offset,
154          )
155  
156      def _materialize(self, device: Optional[torch.device] = None) -> torch.Tensor:
157          """ 
158          Materializes the causal bias into a tensor form. 
159   
160          Depending on the variant, this method generates either an upper-left or lower-right 
161          triangular matrix to represent the causal bias. 
162   
163          Args: 
164              device (Optional[torch.device]): The device on which to create the tensor. Defaults to CPU. 
165   
166          Returns: 
167              torch.Tensor: The materialized bias tensor. 
168          """
169          if device is None:
170              device = torch.device("cpu")
171          if self.variant == CausalVariant.UPPER_LEFT:
172              return self._upper_left(device)
173          elif self.variant == CausalVariant.LOWER_RIGHT:
174              return self._lower_right(device)
175  
176      @staticmethod
177      def _dispatch(
178          query: torch.Tensor,
179          key: torch.Tensor,
180          value: torch.Tensor,
181          attn_mask: "CausalBias",
182          dropout_p: float = 0.0,
183          is_causal: bool = False,
184          scale: Optional[float] = None,
185          enable_gqa: bool = False,
186      ) -> torch.Tensor:
187          r""" 
188          Handles the logic for computing attention with the specified causal bias. 
189   
190          Args: 
191              query (Tensor): Query tensor; shape :math:`(N, ..., L, E)`. 
192              key (Tensor): Key tensor; shape :math:`(N, ..., S, E)`. 
193              value (Tensor): Value tensor; shape :math:`(N, ..., S, Ev)`. 
194              attn_mask (CausalBias): The type of causal attention to apply. 
195                  A boolean mask where a value of True indicates that the element *should* take part in attention. 
196                  A float mask of the same type as query, key, value that is added to the attention score. 
197              dropout_p (float): Dropout probability; if greater than 0.0, dropout is applied 
198              is_causal (bool): If true, assumes upper left causal attention masking and errors if both attn_mask and is_causal 
199                  are set. 
200              scale (optional float): Scaling factor applied prior to softmax. If None, the default value is set 
201                  to :math:`\frac{1}{\sqrt{E}}`. 
202              enable_gqa (optional bool): If set to True, Grouped Query Attention (GQA) is enabled, by default it is set to False. 
203   
204          Returns: 
205              output (Tensor): Attention output; shape :math:`(N, ..., L, Ev)`. 
206   
207          Raises: 
208              ValueError: If the causal bias variant is not a CausalVariant type. 
209   
210          """
211          if is_causal:
212              raise ValueError("CausalBias should not be used with causal=True")
213  
214          if (
215              attn_mask.seq_len_q == attn_mask.seq_len_kv
216              or attn_mask.variant == CausalVariant.UPPER_LEFT
217          ):
218              return F.scaled_dot_product_attention(
219                  query,
220                  key,
221                  value,
222                  attn_mask=None,
223                  dropout_p=dropout_p,
224                  is_causal=True,
225                  scale=scale,
226                  enable_gqa=enable_gqa,
227              )
228          elif attn_mask.variant == CausalVariant.LOWER_RIGHT:
229              _validate_sdpa_input(query, key, value, None, dropout_p, is_causal, scale)
230              sdpa_params = SDPAParams(
231                  query, key, value, None, dropout_p, is_causal, enable_gqa
232              )
233              if can_use_flash_attention(sdpa_params):
234                  needs_padding = query.size(-1) % 8 != 0
235                  og_head_size = query.size(-1)
236                  og_scale = _calculate_scale(og_head_size, scale)
237                  if needs_padding:
238                      query = torch.nn.functional.pad(query, (0, 8 - query.size(-1) % 8))
239                      key = torch.nn.functional.pad(key, (0, 8 - key.size(-1) % 8))
240                      value = torch.nn.functional.pad(value, (0, 8 - value.size(-1) % 8))
241                  out = torch.ops.aten._scaled_dot_product_flash_attention(
242                      query,
243                      key,
244                      value,
245                      dropout_p,
246                      is_causal=True,  # TODO: Flash accepts causal = True and for this particular op it means lower right
247                      return_debug_mask=False,
248                      scale=og_scale,
249                  )[0]
250                  return _postprocess_flash_output(out, og_head_size)
251              if can_use_efficient_attention(sdpa_params):
252                  compute_log_sumexp = False
253                  if _input_requires_grad(query, key, value):
254                      compute_log_sumexp = True
255                  return torch.ops.aten._efficient_attention_forward(
256                      query.transpose(1, 2),
257                      key.transpose(1, 2),
258                      value.transpose(1, 2),
259                      bias=None,
260                      cu_seqlens_q=None,
261                      cu_seqlens_k=None,
262                      max_seqlen_q=None,
263                      max_seqlen_k=None,
264                      dropout_p=dropout_p,
265                      custom_mask_type=int(attn_mask.variant),
266                      compute_log_sumexp=compute_log_sumexp,
267                      scale=scale,
268                      seqlen_k=None,
269                  )[0].transpose(1, 2)
270              else:
271                  _raise_kernel_warnings(sdpa_params)
272                  # We cant use efficient attention the only support for lower right is via materialization
273                  return F.scaled_dot_product_attention(
274                      query,
275                      key,
276                      value,
277                      attn_mask=attn_mask._materialize(query.device),
278                      dropout_p=dropout_p,
279                      is_causal=False,
280                      scale=scale,
281                      enable_gqa=enable_gqa,
282                  )
283          else:
284              raise ValueError(
285                  f"CausalBias.variant must be a CausalVariant type, but found: {attn_mask.variant}"
286              )

288      @classmethod
289      def __torch_function__(cls, func, types, args=(), kwargs=None):
290          """Defines the behavior of torch.nn.functional.scaled_dot_product_attention when the attn_bias is an AttnBias"""
291          if kwargs is None:
292              kwargs = {}
293          if func is torch.nn.functional.scaled_dot_product_attention:
294              return cls._dispatch(*args, **kwargs)
295          return super().__torch_function__(func, types, args, kwargs)

297      def __repr__(self):  # type:ignore[override]
298          return self._materialize().__repr__()

301  def causal_upper_left(*size) -> CausalBias:
302      """ 
303      Creates an upper-left triangular causal bias. 
304   
305      This function generates a upper-left triangular matrix to represent causal attention bias with a 
306      diagonal offset set so that the inclusive values are aligned to the upper left corner of the matrix. 
307      This equivalent to the `is_causal=True` argument in `scaled_dot_product_attention`. 
308   
309      The equivalent pytorch code for constructing this bias is: 
310   
311      .. code-block:: python 
312   
313          torch.tril(torch.ones(size, dtype=torch.bool)) 
314   
315      For instance, with `shape=(3,4)`, the materialized bias tensor will be: 
316   
317      .. code-block:: text 
318   
319          [[1, 0, 0, 0], 
320           [1, 1, 0, 0], 
321           [1, 1, 1, 0]] 
322   
323      Args: 
324          size: The size of the bias matrix. 
325   
326      Returns: 
327          CausalBias: The UPPER_LEFT triangular causal bias variant. 
328      """
329      assert len(size) == 2, "causal_upper_left only supports 2D tensors"
330      seq_len_q, seq_len_kv = size
331      return CausalBias(CausalVariant.UPPER_LEFT, seq_len_q, seq_len_kv)
