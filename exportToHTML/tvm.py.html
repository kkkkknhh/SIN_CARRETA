3    """ 
4    This module provides TVM backend integration for TorchDynamo. 
5     
6    Apache TVM is a deep learning compiler framework that can optimize and execute 
7    models on various hardware backends. This module enables: 
8     
9    - Compilation of PyTorch models to TVM's computation graphs 
10   - Multiple scheduling options: 
11     - Default scheduler 
12     - Auto-scheduler for automatic optimization 
13     - Meta-schedule for evolutionary search-based tuning 
14   - Hardware-specific optimizations: 
15     - CUDA GPU support 
16     - CPU support with LLVM targeting and architecture-specific tuning 
17     - Automatic detection of CPU capabilities (AVX2, AVX512) 
18   - Tensor conversion utilities between PyTorch and TVM formats 
19   - Configurable optimization levels and tuning trials 
20    
21   The backend can be used with torch.compile(): 
22       model = torch.compile(model, backend="tvm") 
23   """
24   
25   import functools
26   import importlib
27   import logging
28   import os
29   import sys
30   import tempfile
31   from types import MappingProxyType
32   from typing import Optional
33   
34   import torch
35   
36   from .common import device_from_inputs, fake_tensor_unsupported
37   from .registry import register_backend
38   
39   
40   log = logging.getLogger(__name__)
41

43   @register_backend
44   @fake_tensor_unsupported
45   def tvm(
46       gm,
47       example_inputs,
48       *,
49       options: Optional[MappingProxyType] = MappingProxyType(
50           {"scheduler": None, "trials": 20000, "opt_level": 3}
51       ),
52   ):
53       import tvm  # type: ignore[import]
54       from tvm import relay  # type: ignore[import]
55       from tvm.contrib import graph_executor  # type: ignore[import]
56   
57       jit_mod = torch.jit.trace(gm, example_inputs)
58       device = device_from_inputs(example_inputs)
59       shape_list = [(f"inp_{idx}", i.shape) for idx, i in enumerate(example_inputs)]
60       example_outputs = gm(*example_inputs)
61       if len(example_outputs) == 0:
62           log.warning("Explicitly fall back to eager due to zero output")
63           return gm.forward
64       mod, params = relay.frontend.from_pytorch(jit_mod, shape_list)
65       if device.type == "cuda":
66           dev = tvm.cuda(device.index)
67           target = tvm.target.cuda()
68       else:
69           dev = tvm.cpu(0)
70           target = tvm.target.Target(llvm_target())
71   
72       scheduler = options.get("scheduler", None)
73       if scheduler is None:
74           scheduler = os.environ.get("TVM_SCHEDULER", None)
75   
76       trials = options.get("trials", 20000)
77       opt_level = options.get("opt_level", 3)
78   
79       if scheduler == "auto_scheduler":
80           from tvm import auto_scheduler
81   
82           log_file = tempfile.NamedTemporaryFile()
83   
84           if not os.path.exists(log_file):
85               tasks, task_weights = auto_scheduler.extract_tasks(
86                   mod["main"], params, target
87               )
88               if len(tasks) != 0:
89                   tuner = auto_scheduler.TaskScheduler(tasks, task_weights)
90                   if not os.path.exists(log_file):
91                       assert trials > 0
92                       tune_option = auto_scheduler.TuningOptions(
93                           num_measure_trials=trials,
94                           measure_callbacks=[auto_scheduler.RecordToFile(log_file)],
95                           early_stopping=2000,
96                       )
97                       try:
98                           tuner.tune(tune_option)
99                       except Exception:
100                          if os.path.exists(log_file):
101                              os.unlink(log_file)
102                          raise
103  
104          with auto_scheduler.ApplyHistoryBest(log_file):
105              with tvm.transform.PassContext(
106                  opt_level=opt_level, config={"relay.backend.use_auto_scheduler": True}
107              ):
108                  lib = relay.build(mod, target=target, params=params)
109      elif scheduler == "meta_schedule":
110          from tvm import meta_schedule as ms
111  
112          with tempfile.TemporaryDirectory() as work_dir:
113              if device.type != "cuda":
114                  # meta_schedule needs num-cores to be specified
115                  # here we use the maximum core count
116                  target = tvm.target.Target(
117                      f"{llvm_target()} --num-cores {ms.utils.cpu_count(logical=False)}"
118                  )
119              # TODO(shingjan): This could be replaced by tvm.contrib.torch.optimize_torch
120              # once USE_PT_TVMDSOOP is updated and turned on by default in TVM.
121              assert trials > 0
122              database = ms.relay_integration.tune_relay(
123                  mod=mod,
124                  target=target,
125                  work_dir=work_dir,
126                  max_trials_global=trials,
127                  num_trials_per_iter=64,
128                  params=params,
129                  strategy="evolutionary",
130                  opt_level=opt_level,
131              )
132              lib = ms.relay_integration.compile_relay(
133                  database=database,
134                  mod=mod,
135                  target=target,
136                  params=params,
137                  opt_level=opt_level,
138              )
139      elif scheduler == "default" or not scheduler:
140          # no autotuning
141          with tvm.transform.PassContext(opt_level=opt_level):
142              lib = relay.build(mod, target=target, params=params)
143      else:
144          raise NotImplementedError(
145              "This tuning option is invalid/not implemented for torchdynamo's TVM-related backend. "
146              "There are three available options: default, auto_scheduler and meta_schedule."
147          )
148      m = graph_executor.GraphModule(lib["default"](dev))
149  
150      def to_torch_tensor(nd_tensor):
151          """A helper function to transfer a NDArray to torch.tensor."""
152          if nd_tensor.dtype == "bool":
153              # DLPack does not support boolean so it can't be handled by
154              # torch.utils.dlpack.from_pack. Workaround by going through
155              # numpy, although this brings additional data copy overhead.
156              return torch.from_numpy(nd_tensor.numpy())
157          return torch.utils.dlpack.from_dlpack(nd_tensor.to_dlpack())
158  
159      def to_tvm_tensor(torch_tensor):
160          """A helper function to transfer a torch.tensor to NDArray."""
161          if torch_tensor.dtype == torch.bool:
162              # same reason as above, fallback to numpy conversion which
163              # could introduce data copy overhead
164              return tvm.nd.array(torch_tensor.cpu().numpy())
165          return tvm.nd.from_dlpack(torch_tensor)
166  
167      def exec_tvm(*i_args):
168          args = [a.contiguous() for a in i_args]
169          shape_info, _ = m.get_input_info()
170          active_inputs = {name for name, _ in shape_info.items()}
171          for idx, arg in enumerate(args, 0):
172              if arg.dim() != 0:
173                  if arg.requires_grad:
174                      arg = arg.detach()
175                  inp_name = f"inp_{idx}"
176                  if inp_name not in active_inputs:
177                      log.warning(
178                          "input %s skipped as not found in tvm's runtime library",
179                          inp_name,
180                      )
181                      continue
182                  m.set_input(
183                      inp_name,
184                      to_tvm_tensor(arg),
185                  )
186          m.run()
187          return [to_torch_tensor(m.get_output(i)) for i in range(m.get_num_outputs())]
188  
189      return exec_tvm

196  def has_tvm():
197      try:
198          importlib.import_module("tvm")
199          return True
200      except ImportError:
201          return False

204  @functools.cache
205  def llvm_target():
206      if sys.platform == "linux":
207          cpuinfo = open("/proc/cpuinfo").read()
208          if "avx512" in cpuinfo:
209              return "llvm -mcpu=skylake-avx512"
210          elif "avx2" in cpuinfo:
211              return "llvm -mcpu=core-avx2"
212      return "llvm"
