88   Attribute.__doc__ = """ 
89       This method is a pass-through function that returns `value`, mostly 
90       used to indicate to the TorchScript compiler that the left-hand side 
91       expression is a class instance attribute with type of `type`. Note that 
92       `torch.jit.Attribute` should only be used in `__init__` method of `jit.ScriptModule` 
93       subclasses. 
94    
95       Though TorchScript can infer correct type for most Python expressions, there are some cases where 
96       type inference can be wrong, including: 
97    
98       - Empty containers like `[]` and `{}`, which TorchScript assumes to be container of `Tensor` 
99       - Optional types like `Optional[T]` but assigned a valid value of type `T`, TorchScript would assume 
100        it is type `T` rather than `Optional[T]` 
101   
102      In eager mode, it is simply a pass-through function that returns `value` 
103      without other implications. 
104   
105      Example: 
106   
107      .. testcode:: 
108   
109          import torch 
110          from typing import Dict 
111   
112          class AttributeModule(torch.jit.ScriptModule): 
113              def __init__(self) -> None: 
114                  super().__init__() 
115                  self.foo = torch.jit.Attribute(0.1, float) 
116   
117                  # we should be able to use self.foo as a float here 
118                  assert 0.0 < self.foo 
119   
120                  self.names_ages = torch.jit.Attribute({}, Dict[str, int]) 
121                  self.names_ages["someone"] = 20 
122                  assert isinstance(self.names_ages["someone"], int) 
123   
124          m = AttributeModule() 
125          # m will contain two attributes 
126          # 1. foo of type float 
127          # 2. names_ages of type Dict[str, int] 
128   
129      .. testcleanup:: 
130   
131          del AttributeModule 
132          del m 
133   
134      Note: it's now preferred to instead use type annotations instead of `torch.jit.Attribute`: 
135   
136      .. testcode:: 
137   
138          import torch 
139          from typing import Dict 
140   
141          class AttributeModule(torch.nn.Module): 
142              names: Dict[str, int] 
143   
144              def __init__(self) -> None: 
145                  super().__init__() 
146                  self.names = {} 
147   
148          m = AttributeModule() 
149   
150      .. testcleanup:: 
151   
152          del AttributeModule 
153          del m 
154   
155      Args: 
156          value: An initial value to be assigned to attribute. 
157          type: A Python type 
158   
159      Returns: 
160          Returns `value` 
161  """

181  # These OrderedDictWrapper classes replace the actual OrderedDicts in
182  # module with versions that get/set properties inside of Module.
183  # This allows us to reuse most of nn.Module while still storing the
184  # data in C++.
185  # Each OrderedDict needs to support:
186  #  x not in view
187  #  x in view
188  #  view[name] = ...
189  #  view.values()
190  #  del view[name]
191  #  view.items()
192  #  view.keys()
193  #  len(view)

196  class OrderedDictWrapper:
197      def __init__(self, _c):
198          self._c = _c
199  
200      def keys(self):
201          return [k for k, v in self.items()]
202  
203      def values(self):
204          return [v for k, v in self.items()]
205  
206      def __len__(self):
207          return len(self.values())
208  
209      def __delitem__(self, k):
210          raise RuntimeError("cannot delete methods or parameters of a script module")
211  
212      def items(self):
213          return self._c.items()
214  
215      def __setitem__(self, k, v):
216          if k not in self:
217              raise RuntimeError(
218                  f"Can't add a new parameter after ScriptModule construction. Tried to add '{k}"
219              )
220          self._c.setattr(k, v)
221  
222      def __contains__(self, k):
223          return self._c.contains(k)
224  
225      def __getitem__(self, k):
226          if k not in self:
227              raise KeyError(k)
228          return self._c.getattr(k)

231  class OrderedModuleDict(OrderedDictWrapper):
232      def __init__(self, module, python_dict):
233          super().__init__(torch._C.ModuleDict(module))
234          # contains _both_ script modules and non-script python-only modules
235  
236          # because script modules are subclassed in python and the
237          # C++ Module class will not hold references to them,
238          # to ensure that you always get the same python value here
239          # we store it in the python dict as well
240          self._python_modules = python_dict
241  
242      def items(self):
243          r = self._python_modules.items()
244          return r
245  
246      def __contains__(self, k):
247          return k in self._python_modules
248  
249      def __setitem__(self, k, v):
250          # Cases where sub-module can be re-assigned after ScriptModule construction
251          # 1. If the attr is an module interface type, it's guaranteed that the module is
252          #    not inlined in the graph, so it's safe to swap a new ScriptModule in.
253          # 2. if the new value if a ScriptModule with the same JIT type, IR won't change
254          #    and it's legit to swap a new module in.
255          # In these two cases we allow swapping a new scripted module and update the
256          # corresponding python module dict to keep sync.
257          # Note: the value to be swapped in has to be ScriptModule instead of nn.Module,
258          # otherwise it's illegal and we throw error.
259          if isinstance(v, ScriptModule):
260              self._c.setattr(k, v)
261              self._python_modules[k] = v
262          else:
263              raise RuntimeError(
264                  "Cannot re-assign modules in a ScriptModule with non-scripted "
265                  f"module, tried to replace existing module '{k}': {v}"
266              )
267  
268      def __getitem__(self, k):
269          return self._python_modules[k]

272  # For each user-defined class that subclasses ScriptModule, this meta-class:
273  # (1) finds all the methods annotated with @script_method in a ScriptModule and
274  #     removes them from the class attributes
275  # (2) puts a wrapper around the class's __init__ method to recursively compile
276  #     all of the script_methods with the module after the original __init__ has
277  #     run. This has to occur after the user-defined __init__ so that submodules and
278  #     parameters are initialized _before_ the script compiler resolve references to
279  #     `self.param` or `self.module`.
280  class ScriptMeta(type):
281      def __init__(cls, name, bases, attrs):  # noqa: B902
282          # Aggregate all the ScriptMethods and constants from superclasses
283          cls._methods: dict[str, Any] = {}
284          cls._constants_set = set(getattr(cls, "__constants__", ()))
285          for base in reversed(bases):
286              for k, v in getattr(base, "_methods", {}).items():
287                  cls._methods[k] = v
288              base_constants: set = getattr(base, "_constants_set", set())
289              cls._constants_set = cls._constants_set.union(base_constants)
290  
291          # find all the script methods of the current class
292          for k, v in sorted(attrs.items()):
293              if isinstance(v, ScriptMethodStub):
294                  delattr(cls, k)
295                  cls._methods[v.original_method.__name__] = v
296  
297          if getattr(cls, "_disable_script_meta", False):
298              # We leave built-in ScriptModule types alone, since this metaclass
299              # is only for compiling user classes that inherit from
300              # ScriptModule.
301              super().__init__(name, bases, attrs)
302              return
303  
304          original_init = getattr(cls, "__init__", lambda self: None)
305  
306          @functools.wraps(original_init)
307          def init_then_script(self, *args, **kwargs):
308              num_methods = len(cls._methods)
309              original_init(self, *args, **kwargs)
310              added_methods_in_init = len(cls._methods) > num_methods
311  
312              if type(self) == cls:
313  
314                  def make_stubs(module):
315                      cls = type(module)
316                      if hasattr(cls, "_methods"):
317                          return [v for k, v in sorted(cls._methods.items())]
318                      else:
319                          return infer_methods_to_compile(module)
320  
321                  self.__dict__["_actual_script_module"] = (
322                      torch.jit._recursive.create_script_module(
323                          self, make_stubs, share_types=not added_methods_in_init
324                      )
325                  )
326  
327                  # Delete the Python attributes that now shadow the ScriptModule
328                  # ones, so that __getattr__ and __setattr__ will properly find
329                  # the scripted versions.
330                  concrete_type = self._actual_script_module._concrete_type
331                  for name in concrete_type.get_attributes():
332                      delattr(self, name)
333                  for name, _ in concrete_type.get_modules():
334                      delattr(self, name)
335                  for name in ("_parameters", "_buffers", "_modules"):
336                      delattr(self, name)
337  
338          cls.__init__ = init_then_script  # type: ignore[misc]
339          super().__init__(name, bases, attrs)

351  def script_method(fn):
352      if not _enabled:
353          return fn
354      # NOTE: we need to traverse two frames here because the meta-class frame
355      # for ScriptModule will be present, as opposed to invoking @script on a
356      # a function or invoking define() on a CompilationUnit.
357      # The stack will look like:
358      #
359      # 0. createResolutionCallback()
360      # 1. script_method()
361      # 2. ScriptModule metaclass frame
362      # 3. Surrounding scope
363      #
364      # createResolutionCallback internally adds 1 to get us to the scope of this
365      # function (the calling function). Adding 2 gets us to the proper surrounding scope.
366      _rcb = _jit_internal.createResolutionCallbackFromFrame(frames_up=2)
367      ast = get_jit_def(fn, fn.__name__, self_name="ScriptModule")
368      return ScriptMethodStub(_rcb, ast, fn)

435      class RecursiveScriptClass:
436          """Wrapper for a TorchScript class instance for use in Python. 
437   
438          An analogue of RecursiveScriptModule for regular objects that are not modules. 
439          This class is a wrapper around a torch._C.ScriptObject that represents an instance 
440          of a TorchScript class and allows it to be used in Python. 
441   
442          Attributes: 
443              _c [torch._C.ScriptObject]: The C++ object to which attribute lookups and method 
444                  calls are forwarded. 
445              _props [Dict[str, property]]: A dictionary of properties fetched from self._c and 
446                  exposed on this wrppaer. 
447          """
448  
449          def __init__(self, cpp_class):
450              super().__init__()
451              self.__dict__["_initializing"] = True
452              self._c = cpp_class
453  
454              # Add wrapped object's properties to this class instance.
455              self._props = {
456                  prop.name: property(prop.getter, prop.setter)
457                  for prop in self._c._properties()
458              }
459  
460              self.__dict__["_initializing"] = False
461  
462          def __getattr__(self, attr):
463              if self.__dict__.get("_initializing"):
464                  return super().__getattr__(attr)  # type: ignore[misc]
465  
466              if attr in self._props:
467                  return self._props[attr].fget()  # type: ignore[call-arg, misc]
468  
469              return getattr(self._c, attr)
470  
471          def __setattr__(self, attr, value):
472              if self.__dict__.get("_initializing"):
473                  return super().__setattr__(attr, value)
474  
475              if attr in self._props:
476                  return self._props[attr].fset(value)  # type: ignore[call-arg, misc]
477  
478              setattr(self._c, attr, value)
479  
480          # Delegate calls to magic methods like __len__ to the C++ module backing the
481          # RecursiveScriptClass.
482          def forward_magic_method(self, method_name, *args, **kwargs):
483              if not self._c._has_method(method_name):
484                  raise TypeError
485  
486              self_method = self.__getattr__(method_name)
487              return self_method(*args, **kwargs)
488  
489          def __getstate__(self):
490              raise pickle.PickleError("ScriptClasses cannot be pickled")
491  
492          def __iadd__(self, other):
493              if self._c._has_method("__iadd__"):
494                  return self.forward_magic_method("__iadd__", other)
495              else:
496                  return self.forward_magic_method("__add__", other)
497  
498      for method_name in _magic_methods:
499  
500          def method_template(self, *args, **kwargs):
501              return self.forward_magic_method(method_name, *args, **kwargs)
502  
503          setattr(RecursiveScriptClass, method_name, method_template)
504  
505      # this is a Python 'non-data descriptor' that causes the first access
506      # to ScriptModule's forward to look up the forward method and stash
507      # it in the objects dict. Due to the standard rules for attribute lookup,
508      # subsequent lookups will just directly return the previously looked up method.
509      # This is necessary because nn.Module defines forward as a method. If we
510      # did nothing, __getattr__ would not be called. Instead we'd get nn.Module.forward
511      # which always throws an exception.

513      class ScriptModule(Module, metaclass=ScriptMeta):
514          r"""Wrapper for C++ torch::jit::Module with methods, attributes, and parameters. 
515   
516          A wrapper around C++ ``torch::jit::Module``. ``ScriptModule``\s 
517          contain methods, attributes, parameters, and 
518          constants. These can be accessed the same way as on a normal ``nn.Module``. 
519          """
520  
521          __jit_unused_properties__ = [
522              "code",
523              "code_with_constants",
524              "graph",
525              "inlined_graph",
526              "original_name",
527          ]
528  
529          def __init__(self) -> None:
530              super().__init__()
531  
532          forward: Callable[..., Any] = _CachedForward()  # type: ignore[assignment]
533  
534          def __getattr__(self, attr):
535              if "_actual_script_module" not in self.__dict__:
536                  return super().__getattr__(attr)
537              return getattr(self._actual_script_module, attr)
538  
539          def __setattr__(self, attr, value):
540              if "_actual_script_module" not in self.__dict__:
541                  # Unwrap torch.jit.Attribute into a regular setattr + record
542                  # the provided type in __annotations__.
543                  #
544                  # This ensures that if we use the attr again in `__init__`, it
545                  # will look like the actual value, not an instance of Attribute.
546                  if isinstance(value, Attribute):
547                      # NB: Ensure that we set __annotations__ on the specific
548                      # class in question, and not on a superclass (which would
549                      # be wrong wrong wrong!).
550                      # See also https://github.com/pytorch/pytorch/issues/39463
551                      if "__annotations__" not in self.__class__.__dict__:
552                          self.__class__.__annotations__ = {}
553                      self.__annotations__[attr] = value.type
554                      value = value.value
555                  return super().__setattr__(attr, value)
556  
557              setattr(self._actual_script_module, attr, value)
558  
559          def define(self, src):
560              if "_actual_script_module" in self.__dict__:
561                  # If we have completed initialization, just defer to the
562                  # backing RecursiveScriptModule to eagerly compile the provided
563                  # source.
564                  return self._actual_script_module.define(src)
565  
566              # Otherwise, we are still in the object's __init__.
567              # In that case, add `src` as a stub to be compiled.
568              #
569              # We use frames_up=1 to get to the proper surrounding scope. The stack
570              # will look like:
571              # 0. createResolutionCallback
572              # 1. define()
573              # 2. surrounding scope.
574              #
575              # createResolutionCallback internally adds 1 to get us to our frame, then
576              # we add 1 to get to the proper surrounding scope.
577              rcb = _jit_internal.createResolutionCallbackFromFrame(frames_up=1)
578              ast = torch._C._parse_source_def(src)
579              self._methods[ast.name().name] = ScriptMethodStub(rcb, ast, None)
580  
581          def _replicate_for_data_parallel(self):
582              return self._actual_script_module._replicate_for_data_parallel()

587              Called by ``torch.package.PackageExporter``'s Pickler's ``persistent_id`` when 
588              saving TorchScript objects. Performs act of saving a ScriptModule inside of 
589              a ``torch.package`` archive.

593              """

595              exporter.script_module_serializer.serialize(self._c, int(script_module_id))
596              return (unpackage_script_module, (script_module_id,))

598      class RecursiveScriptModule(ScriptModule):
599          # XXX: RecursiveScriptModule inherits from ScriptModule for the sole
600          # reason that it retains the existing isinstance(ScriptModule)
601          # behavior.
602          r"""Retain the existing isinstance(ScriptModule) behavior. 
603   
604          The core data structure in TorchScript is the ``ScriptModule``. It is an 
605          analogue of torch's ``nn.Module`` and represents an entire model as a tree of 
606          submodules. Like normal modules, each individual module in a ``ScriptModule`` can 
607          have submodules, parameters, and methods. In ``nn.Module``\s methods are implemented 
608          as Python functions, but in ``ScriptModule``\s methods are implemented as 
609          TorchScript functions, a statically-typed subset of Python that contains all 
610          of PyTorch's built-in Tensor operations. This difference allows your 
611          ``ScriptModule``\s code to run without the need for a Python interpreter. 
612   
613          ``ScriptModule``\s should not be created manually, instead use 
614          either :func:`tracing <torch.jit.trace>` or :func:`scripting <torch.jit.script>`. 
615          Tracing and scripting can be applied incrementally and :ref:`composed as necessary <Types>`. 
616   
617          * Tracing records the tensor operations as executed with a set of example inputs and uses these 
618            operations to construct a computation graph. You can use the full dynamic behavior of Python with tracing, 
619            but values other than Tensors and control flow aren't captured in the graph. 
620   
621          * Scripting inspects the Python code of the model 
622            and compiles it to TorchScript. Scripting allows the use of many `types`_ of values and supports dynamic control flow. 
623            Many, but not all features of Python are supported by the compiler, so changes to the source code may be necessary. 
624          """
625  
626          _disable_script_meta = True
627  
628          def __init__(self, cpp_module):
629              self.__dict__["_initializing"] = True
630              self._c = cpp_module
631              super().__init__()
632              # Delete the 'training' attribute set up by `Module.__init__`. It
633              # will get set on the underlying cpp module, so we delete it here
634              # to avoid this version shadowing the cpp module version.
635              delattr(self, "training")
636  
637          @staticmethod
638          def _construct(cpp_module, init_fn):
639              """ 
640              Construct a RecursiveScriptModule that's ready for use. 
641   
642              PyTorch code should use this to construct a RecursiveScriptModule instead 
643              of instead of calling `__init__` directly, as it makes sure the 
644              object is properly finalized (and in the future, we may take 
645              control of how the RecursiveScriptModule instance is created). 
646   
647              Args: 
648                  cpp_module:  The C++ Module that will hold the actual state of 
649                               this RecursiveScriptModule instance. 
650                  init_fn:  Lambda that initializes the RecursiveScriptModule passed to it. 
651              """
652              script_module = RecursiveScriptModule(cpp_module)
653              init_fn(script_module)

1328         features supported in TorchScript, no changes to the original module code should be necessary. ``script``

1534 def interface(obj):
1535     r"""Decorate to annotate classes or modules of different types. 
1536  
1537     This decorator can be used to define an interface that can be used to annotate 
1538     classes or modules of different types. This can be used for to annotate a submodule 
1539     or attribute class that could have different types that implement the same 
1540     interface, or which could be swapped at runtime; or to store a list of modules or 
1541     classes of varying types. 
1542  
1543     It is sometimes used to implement "Callables" - functions or modules that implement 
1544     an interface but whose implementations differ and which can be swapped out. 
1545  
1546     Example: 
1547     .. testcode:: 
1548  
1549         import torch 
1550         from typing import List 
1551  
1552         @torch.jit.interface 
1553         class InterfaceType: 
1554             def run(self, x: torch.Tensor) -> torch.Tensor: 
1555                 pass 
1556  
1557         # implements InterfaceType 
1558         @torch.jit.script 
1559         class Impl1: 
1560             def run(self, x: torch.Tensor) -> torch.Tensor: 
1561                 return x.relu() 
1562  
1563         class Impl2(torch.nn.Module): 
1564             def __init__(self) -> None: 
1565                 super().__init__() 
1566                 self.val = torch.rand(()) 
1567  
1568             @torch.jit.export 
1569             def run(self, x: torch.Tensor) -> torch.Tensor: 
1570                 return x + self.val 
1571  
1572         def user_fn(impls: List[InterfaceType], idx: int, val: torch.Tensor) -> torch.Tensor: 
1573             return impls[idx].run(val) 
1574  
1575         user_fn_jit = torch.jit.script(user_fn) 
1576  
1577         impls = [Impl1(), torch.jit.script(Impl2())] 
1578         val = torch.rand(4, 4) 
1579         user_fn_jit(impls, 0, val) 
1580         user_fn_jit(impls, 1, val) 
1581     """
1582     if not inspect.isclass(obj):
1583         raise RuntimeError("interface must be applied to a class")
1584     if not _is_new_style_class(obj):
1585         raise RuntimeError("TorchScript interfaces must inherit from 'object'")
1586 
1587     # Expected MRO is:
1588     #   User module
1589     #   torch.nn.modules.module.Module
1590     #   object
1591     is_module_interface = issubclass(obj, torch.nn.Module) and len(obj.mro()) == 3
1592 
1593     if not is_module_interface and len(obj.mro()) > 2:
1594         raise RuntimeError(
1595             "TorchScript interface does not support inheritance yet. "
1596             "Please directly inherit from 'object' or 'nn.Module'."
1597         )
1598 
1599     qualified_name = _qualified_name(obj)
1600     rcb = _jit_internal.createResolutionCallbackFromFrame(1)
1601     # if this type is a `nn.Module` subclass, generate a module interface type
1602     # instead of a class interface type; a module interface type only compiles
1603     # the user provided methods as part of the interface
1604     ast = get_jit_class_def(obj, obj.__name__)
1605     mangled_classname = torch._C._jit_script_interface_compile(
1606         qualified_name, ast, rcb, is_module_interface
1607     )
1608     obj.__torch_script_interface__ = mangled_classname
1609     return obj

1631 class _ScriptProfileColumn:
1632     def __init__(self, header: str, alignment: int = 4, offset: int = 0):
1633         self.header = header
1634         self.alignment = alignment
1635         self.offset = offset
1636         self.rows: dict[int, Any] = {}
1637 
1638     def add_row(self, lineno: int, value: Any):
1639         self.rows[lineno] = value
1640 
1641     def materialize(self):
1642         max_length = len(self.header)
1643         rows: list[tuple[int, str]] = []
1644         for key, value in self.rows.items():
1645             cell = str(value)
1646             rows.append((key, cell))
1647             max_length = max(len(cell), max_length)
1648 
1649         if self.alignment > 0:
1650             padding = max_length + self.alignment
1651             padding -= padding % self.alignment
1652         else:
1653             padding = 0
1654 
1655         rows = [(key, pad(cell, padding, self.offset)) for key, cell in rows]
1656         return pad(self.header, padding, self.offset), rows

1659 class _ScriptProfileTable:
1660     def __init__(self, cols: list[_ScriptProfileColumn], source_range: list[int]):
1661         self.cols = cols
1662         self.source_range = source_range
1663 
1664     def dump_string(self):
1665         outputs: list[str] = []
1666         cells: list[tuple[str, dict[int, str]]] = []
1667         header_buffer = ""
1668         for col in self.cols:
1669             header, rows = col.materialize()
1670             header_buffer += header
1671             cells.append((header, dict(rows)))
1672 
1673         outputs.append(header_buffer)
1674         outputs.append(pad("", len(header_buffer), 0, "="))
1675         for line in self.source_range:
1676             row_buffer = ""
1677             for header, rows in cells:
1678                 cell = rows.get(line)
1679                 if cell is None:
1680                     row_buffer += pad("", len(header))
1681                 else:
1682                     row_buffer += cell
1683             outputs.append(row_buffer)
1684         return "\n".join(outputs)
