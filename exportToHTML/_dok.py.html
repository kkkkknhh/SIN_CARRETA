22       def __init__(self, arg1, shape=None, dtype=None, copy=False, *, maxprint=None):
23           _spbase.__init__(self, arg1, maxprint=maxprint)
24   
25           if isinstance(arg1, tuple) and isshape(arg1, allow_nd=self._allow_nd):
26               self._shape = check_shape(arg1, allow_nd=self._allow_nd)
27               self._dict = {}
28               self.dtype = getdtype(dtype, default=float)
29           elif issparse(arg1):  # Sparse ctor
30               if arg1.format == self.format:
31                   arg1 = arg1.copy() if copy else arg1
32               else:
33                   arg1 = arg1.todok()
34   
35               if dtype is not None:
36                   arg1 = arg1.astype(dtype, copy=False)

38               self._dict = arg1._dict
39               self._shape = check_shape(arg1.shape, allow_nd=self._allow_nd)

42               try:

127      def get(self, key, default=0.0):
128          """This provides dict.get method functionality with type checking"""
129          if key in self._dict:
130              return self._dict[key]
131          if isintlike(key) and self.ndim == 1:
132              key = (key,)
133          if self.ndim != len(key):
134              raise IndexError(f'Index {key} length needs to match self.shape')
135          try:
136              for i in key:
137                  assert isintlike(i)
138          except (AssertionError, TypeError, ValueError) as e:
139              raise IndexError('Index must be or consist of integers.') from e
140          key = tuple(i + M if i < 0 else i for i, M in zip(key, self.shape))
141          if any(i < 0 or i >= M for i, M in zip(key, self.shape)):
142              raise IndexError('Index out of bounds.')
143          if self.ndim == 1:
144              key = key[0]
145          return self._dict.get(key, default)

155      def _get_array(self, idx):
156          idx = np.asarray(idx)
157          if idx.ndim == 0:
158              val = self._dict.get(int(idx), self.dtype.type(0))
159              return np.array(val, stype=self.dtype)
160          new_dok = self._dok_container(idx.shape, dtype=self.dtype)
161          dok_vals = [self._dict.get(i, 0) for i in idx.ravel()]
162          if dok_vals:
163              if len(idx.shape) == 1:
164                  for i, v in enumerate(dok_vals):
165                      if v:
166                          new_dok._dict[i] = v
167              else:
168                  new_idx = np.unravel_index(np.arange(len(dok_vals)), idx.shape)
169                  new_idx = new_idx[0] if len(new_idx) == 1 else zip(*new_idx)
170                  for i, v in zip(new_idx, dok_vals, strict=True):
171                      if v:
172                          new_dok._dict[i] = v
173          return new_dok

185      def _get_sliceXslice(self, row, col):
186          row_start, row_stop, row_step = row.indices(self.shape[0])
187          col_start, col_stop, col_step = col.indices(self.shape[1])
188          row_range = range(row_start, row_stop, row_step)
189          col_range = range(col_start, col_stop, col_step)
190          shape = (len(row_range), len(col_range))
191          # Switch paths only when advantageous
192          # (count the iterations in the loops, adjust for complexity)
193          if len(self) >= 2 * shape[0] * shape[1]:
194              # O(nr*nc) path: loop over <row x col>
195              return self._get_columnXarray(row_range, col_range)
196          # O(nnz) path: loop over entries of self
197          newdok = self._dok_container(shape, dtype=self.dtype)
198          for key in self.keys():
199              i, ri = divmod(int(key[0]) - row_start, row_step)
200              if ri != 0 or i < 0 or i >= shape[0]:
201                  continue
202              j, rj = divmod(int(key[1]) - col_start, col_step)
203              if rj != 0 or j < 0 or j >= shape[1]:
204                  continue
205              newdok._dict[i, j] = self._dict[key]
206          return newdok

225      def _get_columnXarray(self, row, col):
226          # outer indexing
227          newdok = self._dok_container((len(row), len(col)), dtype=self.dtype)
228  
229          for i, r in enumerate(row):
230              for j, c in enumerate(col):
231                  v = self._dict.get((r, c), 0)
232                  if v:
233                      newdok._dict[i, j] = v
234          return newdok

236      def _get_arrayXarray(self, row, col):
237          # inner indexing
238          i, j = map(np.atleast_2d, np.broadcast_arrays(row, col))
239          newdok = self._dok_container(i.shape, dtype=self.dtype)
240  
241          for key in itertools.product(range(i.shape[0]), range(i.shape[1])):
242              v = self._dict.get((i[key], j[key]), 0)
243              if v:
244                  newdok._dict[key] = v
245          return newdok

277          row = list(map(int, row.ravel()))
278          col = list(map(int, col.ravel()))
279          x = x.ravel()
280          self._dict.update(zip(zip(row, col), x))

285                  # may have been superseded by later update
286                  del self._dict[key]

288      def __add__(self, other):
289          if isscalarlike(other):
290              res_dtype = upcast_scalar(self.dtype, other)
291              new = self._dok_container(self.shape, dtype=res_dtype)
292              # Add this scalar to each element.
293              for key in itertools.product(*[range(d) for d in self.shape]):
294                  aij = self._dict.get(key, 0) + other
295                  if aij:
296                      new[key] = aij
297          elif issparse(other):
298              if other.shape != self.shape:
299                  raise ValueError("Matrix dimensions are not equal.")
300              res_dtype = upcast(self.dtype, other.dtype)
301              new = self._dok_container(self.shape, dtype=res_dtype)
302              new._dict = self._dict.copy()
303              if other.format == "dok":
304                  o_items = other.items()
305              else:
306                  other = other.tocoo()
307                  if self.ndim == 1:
308                      o_items = zip(other.coords[0], other.data)
309                  else:
310                      o_items = zip(zip(*other.coords), other.data)
311              with np.errstate(over='ignore'):
312                  new._dict.update((k, new[k] + v) for k, v in o_items)
313          elif isdense(other):
314              new = self.todense() + other
315          else:
316              return NotImplemented
317          return new

331      def _mul_scalar(self, other):
332          res_dtype = upcast_scalar(self.dtype, other)
333          # Multiply this scalar by every element.
334          new = self._dok_container(self.shape, dtype=res_dtype)
335          new._dict.update(((k, v * other) for k, v in self.items()))

338      def _matmul_vector(self, other):
339          res_dtype = upcast(self.dtype, other.dtype)
340  
341          # vector @ vector
342          if self.ndim == 1:
343              if issparse(other):
344                  if other.format == "dok":
345                      keys = self.keys() & other.keys()
346                  else:
347                      keys = self.keys() & other.tocoo().coords[0]
348                  return res_dtype(sum(self._dict[k] * other._dict[k] for k in keys))
349              elif isdense(other):
350                  return res_dtype(sum(other[k] * v for k, v in self.items()))
351              else:
352                  return NotImplemented
353  
354          # matrix @ vector
355          result = np.zeros(self.shape[0], dtype=res_dtype)
356          for (i, j), v in self.items():
357              result[i] += v * other[j]
358          return result
359  
360      def _matmul_multivector(self, other):
361          result_dtype = upcast(self.dtype, other.dtype)
362          # vector @ multivector
363          if self.ndim == 1:
364              # works for other 1d or 2d
365              return sum(v * other[j] for j, v in self._dict.items())
366  
367          # matrix @ multivector
368          M = self.shape[0]
369          new_shape = (M,) if other.ndim == 1 else (M, other.shape[1])
370          result = np.zeros(new_shape, dtype=result_dtype)
371          for (i, j), v in self.items():
372              result[i] += v * other[j]
373          return result

453      @classmethod
454      def fromkeys(cls, iterable, value=1, /):
455          tmp = dict.fromkeys(iterable, value)
456          if isinstance(next(iter(tmp)), tuple):
457              shape = tuple(max(idx) + 1 for idx in zip(*tmp))
458          else:
459              shape = (max(tmp) + 1,)
460          result = cls(shape, dtype=type(value))
461          result._dict = tmp
462          return result

464      def tocoo(self, copy=False):
465          nnz = self.nnz
466          if nnz == 0:
467              return self._coo_container(self.shape, dtype=self.dtype)
468  
469          idx_dtype = self._get_index_dtype(maxval=max(self.shape))
470          data = np.fromiter(self.values(), dtype=self.dtype, count=nnz)
471          # handle 1d keys specially b/c not a tuple
472          inds = zip(*self.keys()) if self.ndim > 1 else (self.keys(),)
473          coords = tuple(np.fromiter(ix, dtype=idx_dtype, count=nnz) for ix in inds)
474          A = self._coo_container((data, coords), shape=self.shape, dtype=self.dtype)
475          A.has_canonical_format = True
476          return A

494      def resize(self, *shape):
495          shape = check_shape(shape, allow_nd=self._allow_nd)
496          if len(shape) != len(self.shape):
497              # TODO implement resize across dimensions
498              raise NotImplementedError
499  
500          if self.ndim == 1:
501              newN = shape[-1]
502              for i in list(self._dict):
503                  if i >= newN:
504                      del self._dict[i]
505              self._shape = shape
506              return
507  
508          newM, newN = shape
509          M, N = self.shape
510          if newM < M or newN < N:
511              # Remove all elements outside new dimensions
512              for i, j in list(self.keys()):
513                  if i >= newM or j >= newN:
514                      del self._dict[i, j]
515          self._shape = shape

522          if self.dtype != dtype:

558  # This namespace class separates array from matrix with isinstance
559  class dok_array(_dok_base, sparray):
560      """ 
561      Dictionary Of Keys based sparse array. 
562   
563      This is an efficient structure for constructing sparse 
564      arrays incrementally. 
565   
566      This can be instantiated in several ways: 
567          dok_array(D) 
568              where D is a 2-D ndarray 
569   
570          dok_array(S) 
571              with another sparse array or matrix S (equivalent to S.todok()) 
572   
573          dok_array((M,N), [dtype]) 
574              create the array with initial shape (M,N) 
575              dtype is optional, defaulting to dtype='d' 
576   
577      Attributes 
578      ---------- 
579      dtype : dtype 
580          Data type of the array 
581      shape : 2-tuple 
582          Shape of the array 
583      ndim : int 
584          Number of dimensions (this is always 2) 
585      nnz 
586          Number of nonzero elements 
587      size 
588      T 
589   
590      Notes 
591      ----- 
592   
593      Sparse arrays can be used in arithmetic operations: they support 
594      addition, subtraction, multiplication, division, and matrix power. 
595   
596      - Allows for efficient O(1) access of individual elements. 
597      - Duplicates are not allowed. 
598      - Can be efficiently converted to a coo_array once constructed. 
599   
600      Examples 
601      -------- 
602      >>> import numpy as np 
603      >>> from scipy.sparse import dok_array 
604      >>> S = dok_array((5, 5), dtype=np.float32) 
605      >>> for i in range(5): 
606      ...     for j in range(5): 
607      ...         S[i, j] = i + j    # Update element 
608   
609      """
610

612  class dok_matrix(spmatrix, _dok_base):
613      """ 
614      Dictionary Of Keys based sparse matrix. 
615   
616      This is an efficient structure for constructing sparse 
617      matrices incrementally. 
618   
619      This can be instantiated in several ways: 
620          dok_matrix(D) 
621              where D is a 2-D ndarray 
622   
623          dok_matrix(S) 
624              with another sparse array or matrix S (equivalent to S.todok()) 
625   
626          dok_matrix((M,N), [dtype]) 
627              create the matrix with initial shape (M,N) 
628              dtype is optional, defaulting to dtype='d' 
629   
630      Attributes 
631      ---------- 
632      dtype : dtype 
633          Data type of the matrix 
634      shape : 2-tuple 
635          Shape of the matrix 
636      ndim : int 
637          Number of dimensions (this is always 2) 
638      nnz 
639          Number of nonzero elements 
640      size 
641      T 
642   
643      Notes

646      Sparse matrices can be used in arithmetic operations: they support

651      - Can be efficiently converted to a coo_matrix once constructed.

655      >>> import numpy as np 
656      >>> from scipy.sparse import dok_matrix 
657      >>> S = dok_matrix((5, 5), dtype=np.float32) 
658      >>> for i in range(5): 
659      ...     for j in range(5):
