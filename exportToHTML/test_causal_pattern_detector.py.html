189          # Check that new patterns are present with expected relative weights

260      def test_overlapping_pattern_resolution(self):
261          """Test that overlapping patterns are resolved correctly with confidence priority."""
262          # Create text with potential overlapping matches
263          overlap_text = (
264              "El proceso implica y conduce a mejores resultados mediante la innovación."
265          )
266          matches = self.detector.detect_causal_patterns(overlap_text)
267  
268          # Should not have overlapping matches
269          for i, match1 in enumerate(matches):
270              for j, match2 in enumerate(matches[i + 1:], i + 1):
271                  self.assertFalse(
272                      self.detector._matches_overlap(match1, match2),
273                      f"Matches {i} and {j} overlap: {match1.text} and {match2.text}",
274                  )
