14   class TestPaths:
15       data = np.arange(5).astype(np.int64)

17       def test_savemat(self):
18           with tempdir() as temp_dir:
19               path = Path(temp_dir) / 'data.mat'
20               scipy.io.savemat(path, {'data': self.data})
21               assert path.is_file()
22   
23       def test_loadmat(self):
24           # Save data with string path, load with pathlib.Path
25           with tempdir() as temp_dir:
26               path = Path(temp_dir) / 'data.mat'
27               scipy.io.savemat(str(path), {'data': self.data})
28   
29               mat_contents = scipy.io.loadmat(path)
30               assert (mat_contents['data'] == self.data).all()
31   
32       def test_whosmat(self):
33           # Save data with string path, load with pathlib.Path
34           with tempdir() as temp_dir:
35               path = Path(temp_dir) / 'data.mat'
36               scipy.io.savemat(str(path), {'data': self.data})
37   
38               contents = scipy.io.whosmat(path)
39               assert contents[0] == ('data', (1, 5), 'int64')
