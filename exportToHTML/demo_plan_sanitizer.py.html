209  def extract_all_keys(obj, keys=None):
210      """Helper function to extract all keys from nested JSON object."""
211      if keys is None:
212          keys = set()
213      
214      if isinstance(obj, dict):
215          for key, value in obj.items():
216              keys.add(key)
217              extract_all_keys(value, keys)
218      elif isinstance(obj, list):
219          for item in obj:
220              extract_all_keys(item, keys)
221      
222      return keys
