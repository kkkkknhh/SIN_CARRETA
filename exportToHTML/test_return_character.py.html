11   class TestReturnCharacter(util.F2PyTest):
12       def check_function(self, t, tname):
13           if tname in ["t0", "t1", "s0", "s1"]:
14               assert t("23") == b"2"
15               r = t("ab")
16               assert r == b"a"
17               r = t(array("ab"))
18               assert r == b"a"
19               r = t(array(77, "u1"))
20               assert r == b"M"
21           elif tname in ["ts", "ss"]:
22               assert t(23) == b"23"
23               assert t("123456789abcdef") == b"123456789a"
24           elif tname in ["t5", "s5"]:
25               assert t(23) == b"23"
26               assert t("ab") == b"ab"
27               assert t("123456789abcdef") == b"12345"
28           else:
29               raise NotImplementedError
