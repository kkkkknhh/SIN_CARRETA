11   class HessianUpdateStrategy:
12       """Interface for implementing Hessian update strategies. 
13    
14       Many optimization methods make use of Hessian (or inverse Hessian) 
15       approximations, such as the quasi-Newton methods BFGS, SR1, L-BFGS. 
16       Some of these  approximations, however, do not actually need to store 
17       the entire matrix or can compute the internal matrix product with a 
18       given vector in a very efficiently manner. This class serves as an 
19       abstract interface between the optimization algorithm and the 
20       quasi-Newton update strategies, giving freedom of implementation 
21       to store and update the internal matrix as efficiently as possible. 
22       Different choices of initialization and update procedure will result 
23       in different quasi-Newton strategies. 
24    
25       Four methods should be implemented in derived classes: ``initialize``, 
26       ``update``, ``dot`` and ``get_matrix``. The matrix multiplication 
27       operator ``@`` is also defined to call the ``dot`` method. 
28    
29       Notes 
30       ----- 
31       Any instance of a class that implements this interface, 
32       can be accepted by the method ``minimize`` and used by 
33       the compatible solvers to approximate the Hessian (or 
34       inverse Hessian) used by the optimization algorithms. 
35       """
36   
37       def initialize(self, n, approx_type):
38           """Initialize internal matrix. 
39    
40           Allocate internal memory for storing and updating 
41           the Hessian or its inverse. 
42    
43           Parameters 
44           ---------- 
45           n : int 
46               Problem dimension. 
47           approx_type : {'hess', 'inv_hess'} 
48               Selects either the Hessian or the inverse Hessian. 
49               When set to 'hess' the Hessian will be stored and updated. 
50               When set to 'inv_hess' its inverse will be used instead. 
51           """
52           raise NotImplementedError("The method ``initialize(n, approx_type)``"
53                                     " is not implemented.")
54   
55       def update(self, delta_x, delta_grad):
56           """Update internal matrix. 
57    
58           Update Hessian matrix or its inverse (depending on how 'approx_type' 
59           is defined) using information about the last evaluated points. 
60    
61           Parameters 
62           ---------- 
63           delta_x : ndarray 
64               The difference between two points the gradient 
65               function have been evaluated at: ``delta_x = x2 - x1``. 
66           delta_grad : ndarray 
67               The difference between the gradients: 
68               ``delta_grad = grad(x2) - grad(x1)``. 
69           """
70           raise NotImplementedError("The method ``update(delta_x, delta_grad)``"
71                                     " is not implemented.")
72   
73       def dot(self, p):
74           """Compute the product of the internal matrix with the given vector. 
75    
76           Parameters 
77           ---------- 
78           p : array_like 
79               1-D array representing a vector. 
80    
81           Returns 
82           ------- 
83           Hp : array 
84               1-D represents the result of multiplying the approximation matrix 
85               by vector p. 
86           """
87           raise NotImplementedError("The method ``dot(p)``"
88                                     " is not implemented.")
89   
90       def get_matrix(self):
91           """Return current internal matrix. 
92    
93           Returns 
94           ------- 
95           H : ndarray, shape (n, n) 
96               Dense matrix containing either the Hessian 
97               or its inverse (depending on how 'approx_type' 
98               is defined). 
99           """
100          raise NotImplementedError("The method ``get_matrix(p)``"
101                                    " is not implemented.")
102  
103      def __matmul__(self, p):
104          return self.dot(p)

107  class FullHessianUpdateStrategy(HessianUpdateStrategy):
108      """Hessian update strategy with full dimensional internal representation. 
109      """
110      _syr = get_blas_funcs('syr', dtype='d')  # Symmetric rank 1 update
111      _syr2 = get_blas_funcs('syr2', dtype='d')  # Symmetric rank 2 update
112      # Symmetric matrix-vector product
113      _symv = get_blas_funcs('symv', dtype='d')
114  
115      def __init__(self, init_scale='auto'):
116          self.init_scale = init_scale
117          # Until initialize is called we can't really use the class,
118          # so it makes sense to set everything to None.
119          self.first_iteration = None
120          self.approx_type = None
121          self.B = None
122          self.H = None
123  
124      def initialize(self, n, approx_type):
125          """Initialize internal matrix. 
126   
127          Allocate internal memory for storing and updating 
128          the Hessian or its inverse. 
129   
130          Parameters 
131          ---------- 
132          n : int 
133              Problem dimension. 
134          approx_type : {'hess', 'inv_hess'} 
135              Selects either the Hessian or the inverse Hessian. 
136              When set to 'hess' the Hessian will be stored and updated. 
137              When set to 'inv_hess' its inverse will be used instead. 
138          """
139          self.first_iteration = True
140          self.n = n
141          self.approx_type = approx_type
142          if approx_type not in ('hess', 'inv_hess'):
143              raise ValueError("`approx_type` must be 'hess' or 'inv_hess'.")
144          # Create matrix
145          if self.approx_type == 'hess':
146              self.B = np.eye(n, dtype=float)
147          else:
148              self.H = np.eye(n, dtype=float)
149  
150      def _auto_scale(self, delta_x, delta_grad):
151          # Heuristic to scale matrix at first iteration.
152          # Described in Nocedal and Wright "Numerical Optimization"
153          # p.143 formula (6.20).
154          s_norm2 = np.dot(delta_x, delta_x)
155          y_norm2 = np.dot(delta_grad, delta_grad)
156          ys = np.abs(np.dot(delta_grad, delta_x))
157          if ys == 0.0 or y_norm2 == 0 or s_norm2 == 0:
158              return 1
159          if self.approx_type == 'hess':
160              return y_norm2 / ys
161          else:
162              return ys / y_norm2
163  
164      def _update_implementation(self, delta_x, delta_grad):
165          raise NotImplementedError("The method ``_update_implementation``"
166                                    " is not implemented.")
167  
168      def update(self, delta_x, delta_grad):
169          """Update internal matrix. 
170   
171          Update Hessian matrix or its inverse (depending on how 'approx_type' 
172          is defined) using information about the last evaluated points. 
173   
174          Parameters 
175          ---------- 
176          delta_x : ndarray 
177              The difference between two points the gradient 
178              function have been evaluated at: ``delta_x = x2 - x1``. 
179          delta_grad : ndarray 
180              The difference between the gradients: 
181              ``delta_grad = grad(x2) - grad(x1)``. 
182          """
183          if np.all(delta_x == 0.0):
184              return
185          if np.all(delta_grad == 0.0):
186              warn('delta_grad == 0.0. Check if the approximated '
187                   'function is linear. If the function is linear '
188                   'better results can be obtained by defining the '
189                   'Hessian as zero instead of using quasi-Newton '
190                   'approximations.',
191                   UserWarning, stacklevel=2)
192              return
193          if self.first_iteration:
194              # Get user specific scale
195              if isinstance(self.init_scale, str) and self.init_scale == "auto":
196                  scale = self._auto_scale(delta_x, delta_grad)
197              else:
198                  scale = self.init_scale
199  
200              # Check for complex: numpy will silently cast a complex array to
201              # a real one but not so for scalar as it raises a TypeError.
202              # Checking here brings a consistent behavior.
203              replace = False
204              if np.size(scale) == 1:
205                  # to account for the legacy behavior having the exact same cast
206                  scale = float(scale)
207              elif np.iscomplexobj(scale):
208                  raise TypeError("init_scale contains complex elements, "
209                                  "must be real.")
210              else:  # test explicitly for allowed shapes and values
211                  replace = True
212                  if self.approx_type == 'hess':
213                      shape = np.shape(self.B)
214                      dtype = self.B.dtype
215                  else:
216                      shape = np.shape(self.H)
217                      dtype = self.H.dtype
218                  # copy, will replace the original
219                  scale = np.array(scale, dtype=dtype, copy=True)
220  
221                  # it has to match the shape of the matrix for the multiplication,
222                  # no implicit broadcasting is allowed
223                  if shape != (init_shape := np.shape(scale)):
224                      raise ValueError("If init_scale is an array, it must have the "
225                                       f"dimensions of the hess/inv_hess: {shape}."
226                                       f" Got {init_shape}.")
227                  if not issymmetric(scale):
228                      raise ValueError("If init_scale is an array, it must be"
229                                       " symmetric (passing scipy.linalg.issymmetric)"
230                                       " to be an approximation of a hess/inv_hess.")
231  
232              # Scale initial matrix with ``scale * np.eye(n)`` or replace
233              # This is not ideal, we could assign the scale directly in
234              # initialize, but we would need to
235              if self.approx_type == 'hess':
236                  if replace:
237                      self.B = scale
238                  else:
239                      self.B *= scale
240              else:
241                  if replace:
242                      self.H = scale
243                  else:
244                      self.H *= scale
245              self.first_iteration = False
246          self._update_implementation(delta_x, delta_grad)
247  
248      def dot(self, p):
249          """Compute the product of the internal matrix with the given vector. 
250   
251          Parameters 
252          ---------- 
253          p : array_like 
254              1-D array representing a vector. 
255   
256          Returns 
257          ------- 
258          Hp : array 
259              1-D represents the result of multiplying the approximation matrix 
260              by vector p. 
261          """
262          if self.approx_type == 'hess':
263              return self._symv(1, self.B, p)
264          else:
265              return self._symv(1, self.H, p)
266  
267      def get_matrix(self):
268          """Return the current internal matrix. 
269   
270          Returns 
271          ------- 
272          M : ndarray, shape (n, n) 
273              Dense matrix containing either the Hessian or its inverse 
274              (depending on how `approx_type` was defined). 
275          """
276          if self.approx_type == 'hess':
277              M = np.copy(self.B)
278          else:
279              M = np.copy(self.H)
280          li = np.tril_indices_from(M, k=-1)
281          M[li] = M.T[li]
282          return M

285  class BFGS(FullHessianUpdateStrategy):
286      """Broyden-Fletcher-Goldfarb-Shanno (BFGS) Hessian update strategy. 
287   
288      Parameters 
289      ---------- 
290      exception_strategy : {'skip_update', 'damp_update'}, optional 
291          Define how to proceed when the curvature condition is violated. 
292          Set it to 'skip_update' to just skip the update. Or, alternatively, 
293          set it to 'damp_update' to interpolate between the actual BFGS 
294          result and the unmodified matrix. Both exceptions strategies 
295          are explained  in [1]_, p.536-537. 
296      min_curvature : float 
297          This number, scaled by a normalization factor, defines the 
298          minimum curvature ``dot(delta_grad, delta_x)`` allowed to go 
299          unaffected by the exception strategy. By default is equal to 
300          1e-8 when ``exception_strategy = 'skip_update'`` and equal 
301          to 0.2 when ``exception_strategy = 'damp_update'``. 
302      init_scale : {float, np.array, 'auto'} 
303          This parameter can be used to initialize the Hessian or its 
304          inverse. When a float is given, the relevant array is initialized 
305          to ``np.eye(n) * init_scale``, where ``n`` is the problem dimension. 
306          Alternatively, if a precisely ``(n, n)`` shaped, symmetric array is given, 
307          this array will be used. Otherwise an error is generated. 
308          Set it to 'auto' in order to use an automatic heuristic for choosing 
309          the initial scale. The heuristic is described in [1]_, p.143. 
310          The default is 'auto'. 
311   
312      Notes 
313      ----- 
314      The update is based on the description in [1]_, p.140. 
315   
316      References 
317      ---------- 
318      .. [1] Nocedal, Jorge, and Stephen J. Wright. "Numerical optimization" 
319             Second Edition (2006). 
320      """
321  
322      def __init__(self, exception_strategy='skip_update', min_curvature=None,
323                   init_scale='auto'):
324          if exception_strategy == 'skip_update':
325              if min_curvature is not None:
326                  self.min_curvature = min_curvature
327              else:
328                  self.min_curvature = 1e-8
329          elif exception_strategy == 'damp_update':
330              if min_curvature is not None:
331                  self.min_curvature = min_curvature
332              else:
333                  self.min_curvature = 0.2
334          else:
335              raise ValueError("`exception_strategy` must be 'skip_update' "
336                               "or 'damp_update'.")
337  
338          super().__init__(init_scale)
339          self.exception_strategy = exception_strategy
340  
341      def _update_inverse_hessian(self, ys, Hy, yHy, s):
342          """Update the inverse Hessian matrix. 
343   
344          BFGS update using the formula: 
345   
346              ``H <- H + ((H*y).T*y + s.T*y)/(s.T*y)^2 * (s*s.T) 
347                       - 1/(s.T*y) * ((H*y)*s.T + s*(H*y).T)`` 
348   
349          where ``s = delta_x`` and ``y = delta_grad``. This formula is 
350          equivalent to (6.17) in [1]_ written in a more efficient way 
351          for implementation. 
352   
353          References 
354          ---------- 
355          .. [1] Nocedal, Jorge, and Stephen J. Wright. "Numerical optimization" 
356                 Second Edition (2006). 
357          """
358          self.H = self._syr2(-1.0 / ys, s, Hy, a=self.H)
359          self.H = self._syr((ys + yHy) / ys ** 2, s, a=self.H)
360  
361      def _update_hessian(self, ys, Bs, sBs, y):
362          """Update the Hessian matrix. 
363   
364          BFGS update using the formula: 
365   
366              ``B <- B - (B*s)*(B*s).T/s.T*(B*s) + y*y^T/s.T*y`` 
367   
368          where ``s`` is short for ``delta_x`` and ``y`` is short 
369          for ``delta_grad``. Formula (6.19) in [1]_. 
370   
371          References 
372          ---------- 
373          .. [1] Nocedal, Jorge, and Stephen J. Wright. "Numerical optimization" 
374                 Second Edition (2006). 
375          """
376          self.B = self._syr(1.0 / ys, y, a=self.B)
377          self.B = self._syr(-1.0 / sBs, Bs, a=self.B)
378  
379      def _update_implementation(self, delta_x, delta_grad):
380          # Auxiliary variables w and z
381          if self.approx_type == 'hess':
382              w = delta_x
383              z = delta_grad
384          else:
385              w = delta_grad
386              z = delta_x
387          # Do some common operations
388          wz = np.dot(w, z)
389          Mw = self @ w
390          wMw = Mw.dot(w)
391          # Guarantee that wMw > 0 by reinitializing matrix.
392          # While this is always true in exact arithmetic,
393          # indefinite matrix may appear due to roundoff errors.
394          if wMw <= 0.0:
395              scale = self._auto_scale(delta_x, delta_grad)
396              # Reinitialize matrix
397              if self.approx_type == 'hess':
398                  self.B = scale * np.eye(self.n, dtype=float)
399              else:
400                  self.H = scale * np.eye(self.n, dtype=float)
401              # Do common operations for new matrix
402              Mw = self @ w
403              wMw = Mw.dot(w)
404          # Check if curvature condition is violated
405          if wz <= self.min_curvature * wMw:
406              # If the option 'skip_update' is set
407              # we just skip the update when the condition
408              # is violated.
409              if self.exception_strategy == 'skip_update':
410                  return
411              # If the option 'damp_update' is set we
412              # interpolate between the actual BFGS
413              # result and the unmodified matrix.
414              elif self.exception_strategy == 'damp_update':
415                  update_factor = (1-self.min_curvature) / (1 - wz/wMw)
416                  z = update_factor*z + (1-update_factor)*Mw
417                  wz = np.dot(w, z)
418          # Update matrix
419          if self.approx_type == 'hess':
420              self._update_hessian(wz, Mw, wMw, z)
421          else:
422              self._update_inverse_hessian(wz, Mw, wMw, z)
423

425  class SR1(FullHessianUpdateStrategy):
426      """Symmetric-rank-1 Hessian update strategy. 
427   
428      Parameters 
429      ---------- 
430      min_denominator : float 
431          This number, scaled by a normalization factor, 
432          defines the minimum denominator magnitude allowed 
433          in the update. When the condition is violated we skip 
434          the update. By default uses ``1e-8``. 
435      init_scale : {float, np.array, 'auto'}, optional 
436          This parameter can be used to initialize the Hessian or its 
437          inverse. When a float is given, the relevant array is initialized 
438          to ``np.eye(n) * init_scale``, where ``n`` is the problem dimension. 
439          Alternatively, if a precisely ``(n, n)`` shaped, symmetric array is given, 
440          this array will be used. Otherwise an error is generated. 
441          Set it to 'auto' in order to use an automatic heuristic for choosing 
442          the initial scale. The heuristic is described in [1]_, p.143. 
443          The default is 'auto'. 
444   
445      Notes 
446      ----- 
447      The update is based on the description in [1]_, p.144-146. 
448   
449      References 
450      ---------- 
451      .. [1] Nocedal, Jorge, and Stephen J. Wright. "Numerical optimization" 
452             Second Edition (2006). 
453      """
454  
455      def __init__(self, min_denominator=1e-8, init_scale='auto'):
456          self.min_denominator = min_denominator
457          super().__init__(init_scale)
458  
459      def _update_implementation(self, delta_x, delta_grad):
460          # Auxiliary variables w and z
461          if self.approx_type == 'hess':
462              w = delta_x
463              z = delta_grad
464          else:
465              w = delta_grad
466              z = delta_x
467          # Do some common operations
468          Mw = self @ w
469          z_minus_Mw = z - Mw
470          denominator = np.dot(w, z_minus_Mw)
471          # If the denominator is too small
472          # we just skip the update.
473          if np.abs(denominator) <= self.min_denominator*norm(w)*norm(z_minus_Mw):
474              return
475          # Update matrix
476          if self.approx_type == 'hess':
477              self.B = self._syr(1/denominator, z_minus_Mw, a=self.B)
478          else:
479              self.H = self._syr(1/denominator, z_minus_Mw, a=self.H)
