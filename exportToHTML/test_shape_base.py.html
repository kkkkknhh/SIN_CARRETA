237          # test axis must be in bounds
238          for ndim in [1, 2, 3]:
239              a = np.ones((1,)*ndim)
240              np.concatenate((a, a), axis=0)  # OK
241              assert_raises(AxisError, np.concatenate, (a, a), axis=ndim)
242              assert_raises(AxisError, np.concatenate, (a, a), axis=-(ndim + 1))
243  
244          # Scalars cannot be concatenated
245          assert_raises(ValueError, concatenate, (0,))
246          assert_raises(ValueError, concatenate, (np.array(0),))
247  
248          # dimensionality must match
249          assert_raises_regex(
250              ValueError,
251              r"all the input arrays must have same number of dimensions, but "
252              r"the array at index 0 has 1 dimension\(s\) and the array at "
253              r"index 1 has 2 dimension\(s\)",
254              np.concatenate, (np.zeros(1), np.zeros((1, 1))))
255  
256          # test shapes must match except for concatenation axis
257          a = np.ones((1, 2, 3))
258          b = np.ones((2, 2, 3))
259          axis = list(range(3))
260          for i in range(3):
261              np.concatenate((a, b), axis=axis[0])  # OK
262              assert_raises_regex(
263                  ValueError,
264                  "all the input array dimensions except for the concatenation axis "
265                  "must match exactly, but along dimension {}, the array at "
266                  "index 0 has size 1 and the array at index 1 has size 2"
267                  .format(i),
268                  np.concatenate, (a, b), axis=axis[1])
269              assert_raises(ValueError, np.concatenate, (a, b), axis=axis[2])
270              a = np.moveaxis(a, -1, 0)
271              b = np.moveaxis(b, -1, 0)
272              axis.append(axis.pop(0))
273  
274          # No arrays to concatenate raises ValueError
275          assert_raises(ValueError, concatenate, ())

310      def test_concatenate(self):
311          # Test concatenate function
312          # One sequence returns unmodified (but as array)
313          r4 = list(range(4))
314          assert_array_equal(concatenate((r4,)), r4)
315          # Any sequence
316          assert_array_equal(concatenate((tuple(r4),)), r4)
317          assert_array_equal(concatenate((array(r4),)), r4)
318          # 1D default concatenation
319          r3 = list(range(3))
320          assert_array_equal(concatenate((r4, r3)), r4 + r3)
321          # Mixed sequence types
322          assert_array_equal(concatenate((tuple(r4), r3)), r4 + r3)
323          assert_array_equal(concatenate((array(r4), r3)), r4 + r3)
324          # Explicit axis specification
325          assert_array_equal(concatenate((r4, r3), 0), r4 + r3)
326          # Including negative
327          assert_array_equal(concatenate((r4, r3), -1), r4 + r3)
328          # 2D
329          a23 = array([[10, 11, 12], [13, 14, 15]])
330          a13 = array([[0, 1, 2]])
331          res = array([[10, 11, 12], [13, 14, 15], [0, 1, 2]])
332          assert_array_equal(concatenate((a23, a13)), res)
333          assert_array_equal(concatenate((a23, a13), 0), res)
334          assert_array_equal(concatenate((a23.T, a13.T), 1), res.T)
335          assert_array_equal(concatenate((a23.T, a13.T), -1), res.T)
336          # Arrays much match shape
337          assert_raises(ValueError, concatenate, (a23.T, a13.T), 0)
338          # 3D
339          res = arange(2 * 3 * 7).reshape((2, 3, 7))
340          a0 = res[..., :4]
341          a1 = res[..., 4:6]
342          a2 = res[..., 6:]
343          assert_array_equal(concatenate((a0, a1, a2), 2), res)
344          assert_array_equal(concatenate((a0, a1, a2), -1), res)
345          assert_array_equal(concatenate((a0.T, a1.T, a2.T), 0), res.T)
346  
347          out = res.copy()
348          rout = concatenate((a0, a1, a2), 2, out=out)
349          assert_(out is rout)
350          assert_equal(res, rout)

380          out = np.empty(4, dtype=out_dtype)
381          to_concat = (array([1.1, 2.2]), array([3.3, 4.4]))

425  def test_stack():
426      # non-iterable input
427      assert_raises(TypeError, stack, 1)
428  
429      # 0d input
430      for input_ in [(1, 2, 3),
431                     [np.int32(1), np.int32(2), np.int32(3)],
432                     [np.array(1), np.array(2), np.array(3)]]:
433          assert_array_equal(stack(input_), [1, 2, 3])
434      # 1d input examples
435      a = np.array([1, 2, 3])
436      b = np.array([4, 5, 6])
437      r1 = array([[1, 2, 3], [4, 5, 6]])
438      assert_array_equal(np.stack((a, b)), r1)
439      assert_array_equal(np.stack((a, b), axis=1), r1.T)
440      # all input types
441      assert_array_equal(np.stack([a, b]), r1)
442      assert_array_equal(np.stack(array([a, b])), r1)
443      # all shapes for 1d input
444      arrays = [np.random.randn(3) for _ in range(10)]
445      axes = [0, 1, -1, -2]
446      expected_shapes = [(10, 3), (3, 10), (3, 10), (10, 3)]
447      for axis, expected_shape in zip(axes, expected_shapes):
448          assert_equal(np.stack(arrays, axis).shape, expected_shape)
449      assert_raises_regex(AxisError, 'out of bounds', stack, arrays, axis=2)
450      assert_raises_regex(AxisError, 'out of bounds', stack, arrays, axis=-3)
451      # all shapes for 2d input
452      arrays = [np.random.randn(3, 4) for _ in range(10)]
453      axes = [0, 1, 2, -1, -2, -3]
454      expected_shapes = [(10, 3, 4), (3, 10, 4), (3, 4, 10),
455                         (3, 4, 10), (3, 10, 4), (10, 3, 4)]
456      for axis, expected_shape in zip(axes, expected_shapes):
457          assert_equal(np.stack(arrays, axis).shape, expected_shape)
458      # empty arrays
459      assert_(stack([[], [], []]).shape == (3, 0))
460      assert_(stack([[], [], []], axis=1).shape == (0, 3))
461      # out
462      out = np.zeros_like(r1)
463      np.stack((a, b), out=out)
464      assert_array_equal(out, r1)
465      # edge cases
466      assert_raises_regex(ValueError, 'need at least one array', stack, [])
467      assert_raises_regex(ValueError, 'must have the same shape',
468                          stack, [1, np.arange(3)])
469      assert_raises_regex(ValueError, 'must have the same shape',
470                          stack, [np.arange(3), 1])
471      assert_raises_regex(ValueError, 'must have the same shape',
472                          stack, [np.arange(3), 1], axis=1)
473      assert_raises_regex(ValueError, 'must have the same shape',
474                          stack, [np.zeros((3, 3)), np.zeros(3)], axis=1)
475      assert_raises_regex(ValueError, 'must have the same shape',
476                          stack, [np.arange(2), np.arange(3)])
477  
478      # do not accept generators
479      with pytest.raises(TypeError, match="arrays to stack must be"):
480          stack(x for x in range(3))
481  
482      #casting and dtype test
483      a = np.array([1, 2, 3])
484      b = np.array([2.5, 3.5, 4.5])
485      res = np.stack((a, b), axis=1, casting="unsafe", dtype=np.int64)
486      expected_res = np.array([[1, 2], [2, 3], [3, 4]])
487      assert_array_equal(res, expected_res)
488      #casting and dtype with TypeError
489      with assert_raises(TypeError):
490          stack((a, b), dtype=np.int64, axis=1, casting="safe")
