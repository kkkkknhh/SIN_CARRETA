18   CHECKSUM_DIGEST_SIZE = 4
19   
20   
21   class BytesWriter:
22       def __init__(self) -> None:
23           # Reserve CHECKSUM_DIGEST_SIZE bytes for checksum
24           self._data = bytearray(CHECKSUM_DIGEST_SIZE)
25   
26       def write_uint64(self, i: int) -> None:
27           self._data.extend(i.to_bytes(8, byteorder="big", signed=False))
28   
29       def write_str(self, s: str) -> None:
30           payload = base64.b64encode(s.encode("utf-8"))
31           self.write_bytes(payload)
32   
33       def write_bytes(self, b: bytes) -> None:
34           self.write_uint64(len(b))
35           self._data.extend(b)
36   
37       def to_bytes(self) -> bytes:
38           digest = zlib.crc32(self._data[CHECKSUM_DIGEST_SIZE:]).to_bytes(
39               4, byteorder="big", signed=False
40           )
41           assert len(digest) == CHECKSUM_DIGEST_SIZE
42           self._data[0:CHECKSUM_DIGEST_SIZE] = digest
43           return bytes(self._data)
44   
45   
46   class BytesReader:
47       def __init__(self, data: bytes) -> None:
48           # Check for data corruption
49           assert len(data) >= CHECKSUM_DIGEST_SIZE
50           digest = zlib.crc32(data[CHECKSUM_DIGEST_SIZE:]).to_bytes(
51               4, byteorder="big", signed=False
52           )
53           assert len(digest) == CHECKSUM_DIGEST_SIZE
54           if data[0:CHECKSUM_DIGEST_SIZE] != digest:
55               raise RuntimeError(
56                   "Bytes object is corrupted, checksum does not match. "
57                   f"Expected: {data[0:CHECKSUM_DIGEST_SIZE]!r}, Got: {digest!r}"
58               )
59   
60           self._data = data
61           self._i = CHECKSUM_DIGEST_SIZE
62   
63       def is_finished(self) -> bool:
64           return len(self._data) == self._i
65   
66       def read_uint64(self) -> int:
67           result = int.from_bytes(
68               self._data[self._i : self._i + 8], byteorder="big", signed=False
69           )
70           self._i += 8
71           return result
72   
73       def read_str(self) -> str:
74           return base64.b64decode(self.read_bytes()).decode("utf-8")
75   
76       def read_bytes(self) -> bytes:
77           size = self.read_uint64()
78           result = self._data[self._i : self._i + size]
79           self._i += size
80           return result

88   class AppendingByteSerializer(Generic[T]):
89       """ 
90       Provides efficient serialization and deserialization of list of bytes 
91       Note that this does not provide any guarantees around byte order 
92       """
93   
94       _serialize_fn: Callable[[BytesWriter, T], None]
95       _writer: BytesWriter
96   
97       def __init__(
98           self,
99           *,
100          serialize_fn: Callable[[BytesWriter, T], None],
101      ) -> None:
102          self._serialize_fn = serialize_fn
103          self.clear()
104  
105      def clear(self) -> None:
106          self._writer = BytesWriter()
107          # First 8-bytes are for version
108          self._writer.write_uint64(_ENCODING_VERSION)
109  
110      def append(self, data: T) -> None:
111          self._serialize_fn(self._writer, data)
112  
113      def extend(self, elems: Iterable[T]) -> None:
114          for elem in elems:
115              self.append(elem)
116  
117      def to_bytes(self) -> bytes:
118          return self._writer.to_bytes()
119  
120      @staticmethod
121      def to_list(data: bytes, *, deserialize_fn: Callable[[BytesReader], T]) -> list[T]:
122          reader = BytesReader(data)
123          assert reader.read_uint64() == _ENCODING_VERSION
124  
125          result: list[T] = []
126          while not reader.is_finished():
127              result.append(deserialize_fn(reader))
128          return result
