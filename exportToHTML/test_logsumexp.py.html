38       def test_logsumexp(self, xp):
39           # Test with zero-size array
40           a = xp.asarray([])
41           desired = xp.asarray(-xp.inf)
42           xp_assert_equal(logsumexp(a), desired)
43   
44           # Test whether logsumexp() function correctly handles large inputs.
45           a = xp.arange(200., dtype=xp.float64)
46           desired = xp.log(xp.sum(xp.exp(a)))
47           xp_assert_close(logsumexp(a), desired)
48   
49           # Now test with large numbers
50           b = xp.asarray([1000., 1000.])
51           desired = xp.asarray(1000.0 + math.log(2.0))
52           xp_assert_close(logsumexp(b), desired)
53   
54           n = 1000
55           b = xp.full((n,), 10000)
56           desired = xp.asarray(10000.0 + math.log(n))
57           xp_assert_close(logsumexp(b), desired)
58   
59           x = xp.asarray([1e-40] * 1000000)
60           logx = xp.log(x)
61           X = xp.stack([x, x])
62           logX = xp.stack([logx, logx])
63           xp_assert_close(xp.exp(logsumexp(logX)), xp.sum(X))
64           xp_assert_close(xp.exp(logsumexp(logX, axis=0)), xp.sum(X, axis=0))
65           xp_assert_close(xp.exp(logsumexp(logX, axis=1)), xp.sum(X, axis=1))
66   
67           # Handling special values properly
68           inf = xp.asarray([xp.inf])
69           nan = xp.asarray([xp.nan])
70           xp_assert_equal(logsumexp(inf), inf[0])
71           xp_assert_equal(logsumexp(-inf), -inf[0])
72           xp_assert_equal(logsumexp(nan), nan[0])
73           xp_assert_equal(logsumexp(xp.asarray([-xp.inf, -xp.inf])), -inf[0])
74   
75           # Handling an array with different magnitudes on the axes
76           a = xp.asarray([[1e10, 1e-10],
77                           [-1e10, -np.inf]])
78           ref = xp.asarray([1e10, -1e10])
79           xp_assert_close(logsumexp(a, axis=-1), ref)
80   
81           # Test keeping dimensions
82           xp_test = array_namespace(a) # `torch` needs `expand_dims`
83           ref = xp_test.expand_dims(ref, axis=-1)
84           xp_assert_close(logsumexp(a, axis=-1, keepdims=True), ref)
85   
86           # Test multiple axes
87           xp_assert_close(logsumexp(a, axis=(-1, -2)), xp.asarray(1e10))

222      @pytest.mark.parametrize('dtype', ['complex64', 'complex128'])
223      def test_gh21610(self, xp, dtype):
224          # gh-21610 noted that `logsumexp` could return imaginary components
225          # outside the range (-pi, pi]. Check that this is resolved.
226          # While working on this, I noticed that all other tests passed even
227          # when the imaginary component of the result was zero. This suggested
228          # the need of a stronger test with imaginary dtype.
229          rng = np.random.default_rng(324984329582349862)
230          dtype = getattr(xp, dtype)
231          shape = (10, 100)
232          x = rng.uniform(1, 40, shape) + 1.j * rng.uniform(1, 40, shape)
233          x = xp.asarray(x, dtype=dtype)
234  
235          res = logsumexp(x, axis=1)
236          ref = xp.log(xp.sum(xp.exp(x), axis=1))
237          max = xp.full_like(xp.imag(res), xp.asarray(xp.pi))
238          xp_assert_less(xp.abs(xp.imag(res)), max)
239          xp_assert_close(res, ref)
240  
241          out, sgn = logsumexp(x, return_sign=True, axis=1)
242          ref = xp.sum(xp.exp(x), axis=1)
243          xp_assert_less(xp.abs(xp.imag(sgn)), max)
244          xp_assert_close(out, xp.real(xp.log(ref)))
245          xp_assert_close(sgn, ref/xp.abs(ref))

275  class TestSoftmax:
276      def test_softmax_fixtures(self):
277          assert_allclose(softmax([1000, 0, 0, 0]), np.array([1, 0, 0, 0]),
278                          rtol=1e-13)
279          assert_allclose(softmax([1, 1]), np.array([.5, .5]), rtol=1e-13)
280          assert_allclose(softmax([0, 1]), np.array([1, np.e])/(1 + np.e),
281                          rtol=1e-13)
282  
283          # Expected value computed using mpmath (with mpmath.mp.dps = 200) and then
284          # converted to float.
285          x = np.arange(4)
286          expected = np.array([0.03205860328008499,
287                              0.08714431874203256,
288                              0.23688281808991013,
289                              0.6439142598879722])
290  
291          assert_allclose(softmax(x), expected, rtol=1e-13)
292  
293          # Translation property.  If all the values are changed by the same amount,
294          # the softmax result does not change.
295          assert_allclose(softmax(x + 100), expected, rtol=1e-13)
296  
297          # When axis=None, softmax operates on the entire array, and preserves
298          # the shape.
299          assert_allclose(softmax(x.reshape(2, 2)), expected.reshape(2, 2),
300                          rtol=1e-13)

303      def test_softmax_multi_axes(self):
304          assert_allclose(softmax([[1000, 0], [1000, 0]], axis=0),
305                          np.array([[.5, .5], [.5, .5]]), rtol=1e-13)
306          assert_allclose(softmax([[1000, 0], [1000, 0]], axis=1),
307                          np.array([[1, 0], [1, 0]]), rtol=1e-13)
308  
309          # Expected value computed using mpmath (with mpmath.mp.dps = 200) and then
310          # converted to float.
311          x = np.array([[-25, 0, 25, 50],
312                      [1, 325, 749, 750]])
313          expected = np.array([[2.678636961770877e-33,
314                              1.9287498479371314e-22,
315                              1.3887943864771144e-11,
316                              0.999999999986112],
317                              [0.0,
318                              1.9444526359919372e-185,
319                              0.2689414213699951,
320                              0.7310585786300048]])
321          assert_allclose(softmax(x, axis=1), expected, rtol=1e-13)
322          assert_allclose(softmax(x.T, axis=0), expected.T, rtol=1e-13)
323  
324          # 3-d input, with a tuple for the axis.
325          x3d = x.reshape(2, 2, 2)
326          assert_allclose(softmax(x3d, axis=(1, 2)), expected.reshape(2, 2, 2),
327                          rtol=1e-13)
