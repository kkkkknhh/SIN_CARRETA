10   class PythonPrinter(ReprPrinter, StrPrinter):
11       """A printer which converts an expression into its Python interpretation."""
12   
13       def __init__(self, settings=None):
14           super().__init__(settings)
15           self.symbols = []
16           self.functions = []
17   
18           # Create print methods for classes that should use StrPrinter instead
19           # of ReprPrinter.
20           for name in STRPRINT:
21               f_name = "_print_%s" % name
22               f = getattr(StrPrinter, f_name)
23               setattr(PythonPrinter, f_name, f)
24   
25       def _print_Function(self, expr):
26           func = expr.func.__name__
27           if not hasattr(sympy, func) and func not in self.functions:
28               self.functions.append(func)
29           return StrPrinter._print_Function(self, expr)
30   
31       # procedure (!) for defining symbols which have be defined in print_python()
32       def _print_Symbol(self, expr):
33           symbol = self._str(expr)
34           if symbol not in self.symbols:
35               self.symbols.append(symbol)
36           return StrPrinter._print_Symbol(self, expr)
37   
38       def _print_module(self, expr):
39           raise ValueError('Modules in the expression are unacceptable')

42   def python(expr, **settings):
43       """Return Python interpretation of passed expression 
44       (can be passed to the exec() function without any modifications)"""
45   
46       printer = PythonPrinter(settings)
47       exprp = printer.doprint(expr)
48   
49       result = ''
50       # Returning found symbols and functions
51       renamings = {}
52       for symbolname in printer.symbols:
53           # Remove curly braces from subscripted variables
54           if '{' in symbolname:
55               newsymbolname = symbolname.replace('{', '').replace('}', '')
56               renamings[sympy.Symbol(symbolname)] = newsymbolname
57           else:
58               newsymbolname = symbolname
59   
60           # Escape symbol names that are reserved Python keywords
61           if kw.iskeyword(newsymbolname):
62               while True:
63                   newsymbolname += "_"
64                   if (newsymbolname not in printer.symbols and
65                           newsymbolname not in printer.functions):
66                       renamings[sympy.Symbol(
67                           symbolname)] = sympy.Symbol(newsymbolname)
68                       break
69           result += newsymbolname + ' = Symbol(\'' + symbolname + '\')\n'
70   
71       for functionname in printer.functions:
72           newfunctionname = functionname
73           # Escape function names that are reserved Python keywords
74           if kw.iskeyword(newfunctionname):
75               while True:
76                   newfunctionname += "_"
77                   if (newfunctionname not in printer.symbols and
78                           newfunctionname not in printer.functions):
79                       renamings[sympy.Function(
80                           functionname)] = sympy.Function(newfunctionname)
81                       break
82           result += newfunctionname + ' = Function(\'' + functionname + '\')\n'
83   
84       if renamings:
85           exprp = expr.subs(renamings)
86       result += 'e = ' + printer._str(exprp)
87       return result

90   def print_python(expr, **settings):
91       """Print output of python() function"""
92       print(python(expr, **settings))
