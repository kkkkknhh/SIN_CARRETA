1    import numpy as np

7    def test_consistency():
8        # Make sure the implementation of spence for real arguments
9        # agrees with the implementation of spence for imaginary arguments.
10   
11       x = np.logspace(-30, 300, 200)
12       dataset = np.vstack((x + 0j, spence(x))).T
13       FuncData(spence, dataset, 0, 1, rtol=1e-14).check()

16   def test_special_points():
17       # Check against known values of Spence's function.
18   
19       phi = (1 + sqrt(5))/2
20       dataset = [(1, 0),
21                  (2, -pi**2/12),
22                  (0.5, pi**2/12 - log(2)**2/2),
23                  (0, pi**2/6),
24                  (-1, pi**2/4 - 1j*pi*log(2)),
25                  ((-1 + sqrt(5))/2, pi**2/15 - log(phi)**2),
26                  ((3 - sqrt(5))/2, pi**2/10 - log(phi)**2),
27                  (phi, -pi**2/15 + log(phi)**2/2),
28                  # Corrected from Zagier, "The Dilogarithm Function"
29                  ((3 + sqrt(5))/2, -pi**2/10 - log(phi)**2)]
30   
31       dataset = np.asarray(dataset)
32       FuncData(spence, dataset, 0, 1, rtol=1e-14).check()
