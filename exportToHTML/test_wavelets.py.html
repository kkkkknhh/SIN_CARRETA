16               assert len(w) == length

18               assert max_loc == (length // 2)

20           points = 100
21           w = wavelets._ricker(points, 2.0)
22           half_vec = np.arange(0, points // 2)
23           # Wavelet should be symmetric
24           assert_array_almost_equal(w[half_vec], w[-(half_vec + 1)])

26           # Check zeros
27           aas = [5, 10, 15, 20, 30]
28           points = 99
29           for a in aas:
30               w = wavelets._ricker(points, a)
31               vec = np.arange(0, points) - (points - 1.0) / 2
32               exp_zero1 = np.argmin(np.abs(vec - a))
33               exp_zero2 = np.argmin(np.abs(vec + a))
34               assert_array_almost_equal(w[exp_zero1], 0)
35               assert_array_almost_equal(w[exp_zero2], 0)
