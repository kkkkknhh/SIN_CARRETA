19   def toeplitz(c, r=None):
20       r""" 
21       Construct a Toeplitz matrix. 
22    
23       The Toeplitz matrix has constant diagonals, with c as its first column 
24       and r as its first row. If r is not given, ``r == conjugate(c)`` is 
25       assumed. 
26    
27       Parameters 
28       ---------- 
29       c : array_like 
30           First column of the matrix. 
31       r : array_like, optional 
32           First row of the matrix. If None, ``r = conjugate(c)`` is assumed; 
33           in this case, if c[0] is real, the result is a Hermitian matrix. 
34           r[0] is ignored; the first row of the returned matrix is 
35           ``[c[0], r[1:]]``. 
36    
37           .. warning:: 
38    
39               Beginning in SciPy 1.17, multidimensional input will be treated as a batch, 
40               not ``ravel``\ ed. To preserve the existing behavior, ``ravel`` arguments 
41               before passing them to `toeplitz`. 
42    
43       Returns 
44       ------- 
45       A : (len(c), len(r)) ndarray 
46           The Toeplitz matrix. Dtype is the same as ``(c[0] + r[0]).dtype``. 
47    
48       See Also 
49       -------- 
50       circulant : circulant matrix 
51       hankel : Hankel matrix 
52       solve_toeplitz : Solve a Toeplitz system. 
53    
54       Notes 
55       ----- 
56       The behavior when `c` or `r` is a scalar, or when `c` is complex and 
57       `r` is None, was changed in version 0.8.0. The behavior in previous 
58       versions was undocumented and is no longer supported. 
59    
60       Examples 
61       -------- 
62       >>> from scipy.linalg import toeplitz 
63       >>> toeplitz([1,2,3], [1,4,5,6]) 
64       array([[1, 4, 5, 6], 
65              [2, 1, 4, 5], 
66              [3, 2, 1, 4]]) 
67       >>> toeplitz([1.0, 2+3j, 4-1j]) 
68       array([[ 1.+0.j,  2.-3.j,  4.+1.j], 
69              [ 2.+3.j,  1.+0.j,  2.-3.j], 
70              [ 4.-1.j,  2.+3.j,  1.+0.j]]) 
71    
72       """
73       c = np.asarray(c)
74       if r is None:
75           r = c.conjugate()
76       else:
77           r = np.asarray(r)
78   
79       if c.ndim > 1 or r.ndim > 1:
80           msg = ("Beginning in SciPy 1.17, multidimensional input will be treated as a "
81                  "batch, not `ravel`ed. To preserve the existing behavior and silence "
82                  "this warning, `ravel` arguments before passing them to `toeplitz`.")
83           warnings.warn(msg, FutureWarning, stacklevel=2)
84   
85       c, r = c.ravel(), r.ravel()
86       # Form a 1-D array containing a reversed c followed by r[1:] that could be
87       # strided to give us toeplitz matrix.
88       vals = np.concatenate((c[::-1], r[1:]))
89       out_shp = len(c), len(r)
90       n = vals.strides[0]
91       return as_strided(vals[len(c)-1:], shape=out_shp, strides=(-n, n)).copy()

94   def circulant(c):
95       """ 
96       Construct a circulant matrix. 
97    
98       Parameters 
99       ---------- 
100      c : (..., N,)  array_like 
101          The first column(s) of the matrix. Multidimensional arrays are treated as a 
102          batch: each slice along the last axis is the first column of an output matrix. 
103   
104      Returns 
105      ------- 
106      A : (..., N, N) ndarray 
107          A circulant matrix whose first column is given by `c`.  For batch input, each 
108          slice of shape ``(N, N)`` along the last two dimensions of the output 
109          corresponds with a slice of shape ``(N,)`` along the last dimension of the 
110          input. 
111   
112   
113      See Also 
114      -------- 
115      toeplitz : Toeplitz matrix 
116      hankel : Hankel matrix 
117      solve_circulant : Solve a circulant system. 
118   
119      Notes 
120      ----- 
121      .. versionadded:: 0.8.0 
122   
123      Examples 
124      -------- 
125      >>> from scipy.linalg import circulant 
126      >>> circulant([1, 2, 3]) 
127      array([[1, 3, 2], 
128             [2, 1, 3], 
129             [3, 2, 1]]) 
130   
131      >>> circulant([[1, 2, 3], [4, 5, 6]]) 
132      array([[[1, 3, 2], 
133              [2, 1, 3], 
134              [3, 2, 1]], 
135             [[4, 6, 5], 
136              [5, 4, 6], 
137              [6, 5, 4]]]) 
138      """
139      c = np.atleast_1d(c)
140      batch_shape, N = c.shape[:-1], c.shape[-1]
141      # Need to use `prod(batch_shape)` instead of `-1` in case array has zero size
142      c = c.reshape(math.prod(batch_shape), N) if batch_shape else c
143      # Form an extended array that could be strided to give circulant version
144      c_ext = np.concatenate((c[..., ::-1], c[..., :0:-1]), axis=-1).ravel()
145      L = c.shape[-1]
146      n = c_ext.strides[-1]
147      if c.ndim == 1:
148          A = as_strided(c_ext[L-1:], shape=(L, L), strides=(-n, n))
149      else:
150          m = c.shape[0]
151          A = as_strided(c_ext[L-1:], shape=(m, L, L), strides=(n*(2*L-1), -n, n))
152      return A.reshape(batch_shape + (N, N)).copy()

155  def hankel(c, r=None):
156      """ 
157      Construct a Hankel matrix. 
158   
159      The Hankel matrix has constant anti-diagonals, with `c` as its 
160      first column and `r` as its last row. If the first element of `r` 
161      differs from the last element of `c`, the first element of `r` is 
162      replaced by the last element of `c` to ensure that anti-diagonals 
163      remain constant. If `r` is not given, then `r = zeros_like(c)` is 
164      assumed. 
165   
166      Parameters 
167      ---------- 
168      c : array_like 
169          First column of the matrix. Whatever the actual shape of `c`, it 
170          will be converted to a 1-D array. 
171      r : array_like, optional 
172          Last row of the matrix. If None, ``r = zeros_like(c)`` is assumed. 
173          r[0] is ignored; the last row of the returned matrix is 
174          ``[c[-1], r[1:]]``. Whatever the actual shape of `r`, it will be 
175          converted to a 1-D array. 
176   
177      Returns 
178      ------- 
179      A : (len(c), len(r)) ndarray 
180          The Hankel matrix. Dtype is the same as ``(c[0] + r[0]).dtype``. 
181   
182      See Also 
183      -------- 
184      toeplitz : Toeplitz matrix 
185      circulant : circulant matrix 
186   
187      Examples 
188      -------- 
189      >>> from scipy.linalg import hankel 
190      >>> hankel([1, 17, 99]) 
191      array([[ 1, 17, 99], 
192             [17, 99,  0], 
193             [99,  0,  0]]) 
194      >>> hankel([1,2,3,4], [4,7,7,8,9]) 
195      array([[1, 2, 3, 4, 7], 
196             [2, 3, 4, 7, 7], 
197             [3, 4, 7, 7, 8], 
198             [4, 7, 7, 8, 9]]) 
199   
200      """
201      c = np.asarray(c).ravel()
202      if r is None:
203          r = np.zeros_like(c)
204      else:
205          r = np.asarray(r).ravel()
206      # Form a 1-D array of values to be used in the matrix, containing `c`
207      # followed by r[1:].
208      vals = np.concatenate((c, r[1:]))
209      # Stride on concatenated array to get hankel matrix
210      out_shp = len(c), len(r)
211      n = vals.strides[0]
212      return as_strided(vals, shape=out_shp, strides=(n, n)).copy()

215  def hadamard(n, dtype=int):
216      """ 
217      Construct an Hadamard matrix. 
218   
219      Constructs an n-by-n Hadamard matrix, using Sylvester's 
220      construction. `n` must be a power of 2. 
221   
222      Parameters 
223      ---------- 
224      n : int 
225          The order of the matrix. `n` must be a power of 2. 
226      dtype : dtype, optional 
227          The data type of the array to be constructed. 
228   
229      Returns 
230      ------- 
231      H : (n, n) ndarray 
232          The Hadamard matrix. 
233   
234      Notes 
235      ----- 
236      .. versionadded:: 0.8.0 
237   
238      Examples 
239      -------- 
240      >>> from scipy.linalg import hadamard 
241      >>> hadamard(2, dtype=complex) 
242      array([[ 1.+0.j,  1.+0.j], 
243             [ 1.+0.j, -1.-0.j]]) 
244      >>> hadamard(4) 
245      array([[ 1,  1,  1,  1], 
246             [ 1, -1,  1, -1], 
247             [ 1,  1, -1, -1], 
248             [ 1, -1, -1,  1]]) 
249   
250      """
251  
252      # This function is a slightly modified version of the
253      # function contributed by Ivo in ticket #675.
254  
255      if n < 1:
256          lg2 = 0
257      else:
258          lg2 = int(math.log(n, 2))
259      if 2 ** lg2 != n:
260          raise ValueError("n must be an positive integer, and n must be "
261                           "a power of 2")
262  
263      H = np.array([[1]], dtype=dtype)
264  
265      # Sylvester's construction
266      for i in range(0, lg2):
267          H = np.vstack((np.hstack((H, H)), np.hstack((H, -H))))
268  
269      return H

272  def leslie(f, s):
273      """ 
274      Create a Leslie matrix. 
275   
276      Given the length n array of fecundity coefficients `f` and the length 
277      n-1 array of survival coefficients `s`, return the associated Leslie 
278      matrix. 
279   
280      Parameters 
281      ---------- 
282      f : (..., N,) array_like 
283          The "fecundity" coefficients. 
284      s : (..., N-1,) array_like 
285          The "survival" coefficients. The length of each slice of `s` (along the last 
286          axis) must be one less than the length of `f`, and it must be at least 1. 
287   
288      Returns 
289      ------- 
290      L : (..., N, N) ndarray 
291          The array is zero except for the first row, 
292          which is `f`, and the first sub-diagonal, which is `s`. 
293          For 1-D input, the data-type of the array will be the data-type of 
294          ``f[0]+s[0]``. 
295   
296      Notes 
297      ----- 
298      .. versionadded:: 0.8.0 
299   
300      The Leslie matrix is used to model discrete-time, age-structured 
301      population growth [1]_ [2]_. In a population with `n` age classes, two sets 
302      of parameters define a Leslie matrix: the `n` "fecundity coefficients", 
303      which give the number of offspring per-capita produced by each age 
304      class, and the `n` - 1 "survival coefficients", which give the 
305      per-capita survival rate of each age class. 
306   
307      N-dimensional input are treated as a batches of coefficient arrays: each 
308      slice along the last axis of the input arrays is a 1-D coefficient array, 
309      and each slice along the last two dimensions of the output is the 
310      corresponding Leslie matrix. 
311   
312      References 
313      ---------- 
314      .. [1] P. H. Leslie, On the use of matrices in certain population 
315             mathematics, Biometrika, Vol. 33, No. 3, 183--212 (Nov. 1945) 
316      .. [2] P. H. Leslie, Some further notes on the use of matrices in 
317             population mathematics, Biometrika, Vol. 35, No. 3/4, 213--245 
318             (Dec. 1948) 
319   
320      Examples 
321      -------- 
322      >>> from scipy.linalg import leslie 
323      >>> leslie([0.1, 2.0, 1.0, 0.1], [0.2, 0.8, 0.7]) 
324      array([[ 0.1,  2. ,  1. ,  0.1], 
325             [ 0.2,  0. ,  0. ,  0. ], 
326             [ 0. ,  0.8,  0. ,  0. ], 
327             [ 0. ,  0. ,  0.7,  0. ]]) 
328   
329      """
330      f = np.atleast_1d(f)
331      s = np.atleast_1d(s)
332  
333      if f.shape[-1] != s.shape[-1] + 1:
334          raise ValueError("Incorrect lengths for f and s. The length of s along "
335                           "the last axis must be one less than the length of f.")
336      if s.shape[-1] == 0:
337          raise ValueError("The length of s must be at least 1.")
338  
339      n = f.shape[-1]
340  
341      if f.ndim > 1 or s.ndim > 1:
342          from scipy.stats._resampling import _vectorize_statistic
343          _leslie_nd = _vectorize_statistic(leslie)
344          return np.moveaxis(_leslie_nd(f, s, axis=-1), [0, 1], [-2, -1])
345  
346      tmp = f[0] + s[0]
347      a = np.zeros((n, n), dtype=tmp.dtype)
348      a[0] = f
349      a[list(range(1, n)), list(range(0, n - 1))] = s
350      return a

353  def kron(a, b):
354      """ 
355      Kronecker product. 
356   
357      .. deprecated:: 1.15.0 
358          `kron` has been deprecated in favour of `numpy.kron` and will be 
359          removed in SciPy 1.17.0. 
360   
361      The result is the block matrix:: 
362   
363          a[0,0]*b    a[0,1]*b  ... a[0,-1]*b 
364          a[1,0]*b    a[1,1]*b  ... a[1,-1]*b 
365          ... 
366          a[-1,0]*b   a[-1,1]*b ... a[-1,-1]*b 
367   
368      Parameters 
369      ---------- 
370      a : (M, N) ndarray 
371          Input array 
372      b : (P, Q) ndarray 
373          Input array 
374   
375      Returns 
376      ------- 
377      A : (M*P, N*Q) ndarray 
378          Kronecker product of `a` and `b`. 
379   
380      Examples 
381      -------- 
382      >>> from numpy import array 
383      >>> from scipy.linalg import kron 
384      >>> kron(array([[1,2],[3,4]]), array([[1,1,1]])) 
385      array([[1, 1, 1, 2, 2, 2], 
386             [3, 3, 3, 4, 4, 4]]) 
387   
388      """
389      msg = ("`kron` has been deprecated in favour of `numpy.kron` in SciPy"
390             " 1.15.0 and will be removed in SciPy 1.17.0.")
391      warnings.warn(msg, DeprecationWarning, stacklevel=2)
392      # accommodate empty arrays
393      if a.size == 0 or b.size == 0:
394          m = a.shape[0] * b.shape[0]
395          n = a.shape[1] * b.shape[1]
396          return np.empty_like(a, shape=(m, n))
397  
398      if not a.flags['CONTIGUOUS']:
399          a = np.reshape(a, a.shape)
400      if not b.flags['CONTIGUOUS']:
401          b = np.reshape(b, b.shape)
402      o = np.outer(a, b)
403      o = o.reshape(a.shape + b.shape)
404      return np.concatenate(np.concatenate(o, axis=1), axis=1)

407  def block_diag(*arrs):
408      """ 
409      Create a block diagonal matrix from provided arrays. 
410   
411      Given the inputs `A`, `B` and `C`, the output will have these 
412      arrays arranged on the diagonal:: 
413   
414          [[A, 0, 0], 
415           [0, B, 0], 
416           [0, 0, C]] 
417   
418      Parameters 
419      ---------- 
420      A, B, C, ... : array_like, up to 2-D 
421          Input arrays.  A 1-D array or array_like sequence of length `n` is 
422          treated as a 2-D array with shape ``(1,n)``. 
423   
424      Returns 
425      ------- 
426      D : ndarray 
427          Array with `A`, `B`, `C`, ... on the diagonal. `D` has the 
428          same dtype as `A`. 
429   
430      Notes 
431      ----- 
432      If all the input arrays are square, the output is known as a 
433      block diagonal matrix. 
434   
435      Empty sequences (i.e., array-likes of zero size) will not be ignored. 
436      Noteworthy, both [] and [[]] are treated as matrices with shape ``(1,0)``. 
437   
438      Examples 
439      -------- 
440      >>> import numpy as np 
441      >>> from scipy.linalg import block_diag 
442      >>> A = [[1, 0], 
443      ...      [0, 1]] 
444      >>> B = [[3, 4, 5], 
445      ...      [6, 7, 8]] 
446      >>> C = [[7]] 
447      >>> P = np.zeros((2, 0), dtype='int32') 
448      >>> block_diag(A, B, C) 
449      array([[1, 0, 0, 0, 0, 0], 
450             [0, 1, 0, 0, 0, 0], 
451             [0, 0, 3, 4, 5, 0], 
452             [0, 0, 6, 7, 8, 0], 
453             [0, 0, 0, 0, 0, 7]]) 
454      >>> block_diag(A, P, B, C) 
455      array([[1, 0, 0, 0, 0, 0], 
456             [0, 1, 0, 0, 0, 0], 
457             [0, 0, 0, 0, 0, 0], 
458             [0, 0, 0, 0, 0, 0], 
459             [0, 0, 3, 4, 5, 0], 
460             [0, 0, 6, 7, 8, 0], 
461             [0, 0, 0, 0, 0, 7]]) 
462      >>> block_diag(1.0, [2, 3], [[4, 5], [6, 7]]) 
463      array([[ 1.,  0.,  0.,  0.,  0.], 
464             [ 0.,  2.,  3.,  0.,  0.], 
465             [ 0.,  0.,  0.,  4.,  5.], 
466             [ 0.,  0.,  0.,  6.,  7.]]) 
467   
468      """
469      if arrs == ():
470          arrs = ([],)
471      arrs = [np.atleast_2d(a) for a in arrs]
472  
473      bad_args = [k for k in range(len(arrs)) if arrs[k].ndim > 2]
474      if bad_args:
475          raise ValueError("arguments in the following positions "
476                           f"have dimension greater than 2: {bad_args}")
477  
478      shapes = np.array([a.shape for a in arrs])
479      out_dtype = np.result_type(*[arr.dtype for arr in arrs])
480      out = np.zeros(np.sum(shapes, axis=0), dtype=out_dtype)
481  
482      r, c = 0, 0
483      for i, (rr, cc) in enumerate(shapes):
484          out[r:r + rr, c:c + cc] = arrs[i]
485          r += rr
486          c += cc
487      return out

490  def companion(a):
491      """ 
492      Create a companion matrix. 
493   
494      Create the companion matrix [1]_ associated with the polynomial whose 
495      coefficients are given in `a`. 
496   
497      Parameters 
498      ---------- 
499      a : (..., N) array_like 
500          1-D array of polynomial coefficients. The length of `a` must be 
501          at least two, and ``a[0]`` must not be zero. 
502          M-dimensional arrays are treated as a batch: each slice along the last 
503          axis is a 1-D array of polynomial coefficients. 
504   
505      Returns 
506      ------- 
507      c : (..., N-1, N-1) ndarray 
508          For 1-D input, the first row of `c` is ``-a[1:]/a[0]``, and the first 
509          sub-diagonal is all ones.  The data-type of the array is the same 
510          as the data-type of ``1.0*a[0]``. 
511          For batch input, each slice of shape ``(N-1, N-1)`` along the last two 
512          dimensions of the output corresponds with a slice of shape ``(N,)`` 
513          along the last dimension of the input. 
514   
515      Raises 
516      ------ 
517      ValueError 
518          If any of the following are true: a) ``a.shape[-1] < 2``; b) ``a[..., 0] == 0``. 
519   
520      Notes 
521      ----- 
522      .. versionadded:: 0.8.0 
523   
524      References 
525      ---------- 
526      .. [1] R. A. Horn & C. R. Johnson, *Matrix Analysis*.  Cambridge, UK: 
527          Cambridge University Press, 1999, pp. 146-7. 
528   
529      Examples 
530      -------- 
531      >>> from scipy.linalg import companion 
532      >>> companion([1, -10, 31, -30]) 
533      array([[ 10., -31.,  30.], 
534             [  1.,   0.,   0.], 
535             [  0.,   1.,   0.]]) 
536   
537      """
538      a = np.atleast_1d(a)
539      n = a.shape[-1]
540  
541      if n < 2:
542          raise ValueError("The length of `a` along the last axis must be at least 2.")
543  
544      if np.any(a[..., 0] == 0):
545          raise ValueError("The first coefficient(s) of `a` (i.e. elements "
546                           "of `a[..., 0]`) must not be zero.")
547  
548      first_row = -a[..., 1:] / (1.0 * a[..., 0:1])
549      c = np.zeros(a.shape[:-1] + (n - 1, n - 1), dtype=first_row.dtype)
550      c[..., 0, :] = first_row
551      c[..., np.arange(1, n - 1), np.arange(0, n - 2)] = 1
552      return c

555  def helmert(n, full=False):
556      """ 
557      Create an Helmert matrix of order `n`. 
558   
559      This has applications in statistics, compositional or simplicial analysis, 
560      and in Aitchison geometry. 
561   
562      Parameters 
563      ---------- 
564      n : int 
565          The size of the array to create. 
566      full : bool, optional 
567          If True the (n, n) ndarray will be returned. 
568          Otherwise the submatrix that does not include the first 
569          row will be returned. 
570          Default: False. 
571   
572      Returns 
573      ------- 
574      M : ndarray 
575          The Helmert matrix. 
576          The shape is (n, n) or (n-1, n) depending on the `full` argument. 
577   
578      Examples 
579      -------- 
580      >>> from scipy.linalg import helmert 
581      >>> helmert(5, full=True) 
582      array([[ 0.4472136 ,  0.4472136 ,  0.4472136 ,  0.4472136 ,  0.4472136 ], 
583             [ 0.70710678, -0.70710678,  0.        ,  0.        ,  0.        ], 
584             [ 0.40824829,  0.40824829, -0.81649658,  0.        ,  0.        ], 
585             [ 0.28867513,  0.28867513,  0.28867513, -0.8660254 ,  0.        ], 
586             [ 0.2236068 ,  0.2236068 ,  0.2236068 ,  0.2236068 , -0.89442719]]) 
587   
588      """
589      H = np.tril(np.ones((n, n)), -1) - np.diag(np.arange(n))
590      d = np.arange(n) * np.arange(1, n+1)
591      H[0] = 1
592      d[0] = n
593      H_full = H / np.sqrt(d)[:, np.newaxis]
594      if full:
595          return H_full
596      else:
597          return H_full[1:]

600  def hilbert(n):
601      """ 
602      Create a Hilbert matrix of order `n`. 
603   
604      Returns the `n` by `n` array with entries `h[i,j] = 1 / (i + j + 1)`. 
605   
606      Parameters 
607      ---------- 
608      n : int 
609          The size of the array to create. 
610   
611      Returns 
612      ------- 
613      h : (n, n) ndarray 
614          The Hilbert matrix. 
615   
616      See Also 
617      -------- 
618      invhilbert : Compute the inverse of a Hilbert matrix. 
619   
620      Notes 
621      ----- 
622      .. versionadded:: 0.10.0 
623   
624      Examples 
625      -------- 
626      >>> from scipy.linalg import hilbert 
627      >>> hilbert(3) 
628      array([[ 1.        ,  0.5       ,  0.33333333], 
629             [ 0.5       ,  0.33333333,  0.25      ], 
630             [ 0.33333333,  0.25      ,  0.2       ]]) 
631   
632      """
633      values = 1.0 / (1.0 + np.arange(2 * n - 1))
634      h = hankel(values[:n], r=values[n - 1:])
635      return h

638  def invhilbert(n, exact=False):
639      """ 
640      Compute the inverse of the Hilbert matrix of order `n`. 
641   
642      The entries in the inverse of a Hilbert matrix are integers. When `n` 
643      is greater than 14, some entries in the inverse exceed the upper limit 
644      of 64 bit integers. The `exact` argument provides two options for 
645      dealing with these large integers. 
646   
647      Parameters 
648      ---------- 
649      n : int 
650          The order of the Hilbert matrix. 
651      exact : bool, optional 
652          If False, the data type of the array that is returned is np.float64, 
653          and the array is an approximation of the inverse. 
654          If True, the array is the exact integer inverse array. To represent 
655          the exact inverse when n > 14, the returned array is an object array 
656          of long integers. For n <= 14, the exact inverse is returned as an 
657          array with data type np.int64. 
658   
659      Returns 
660      ------- 
661      invh : (n, n) ndarray 
662          The data type of the array is np.float64 if `exact` is False. 
663          If `exact` is True, the data type is either np.int64 (for n <= 14) 
664          or object (for n > 14). In the latter case, the objects in the 
665          array will be long integers. 
666   
667      See Also 
668      -------- 
669      hilbert : Create a Hilbert matrix. 
670   
671      Notes 
672      ----- 
673      .. versionadded:: 0.10.0 
674   
675      Examples 
676      -------- 
677      >>> from scipy.linalg import invhilbert 
678      >>> invhilbert(4) 
679      array([[   16.,  -120.,   240.,  -140.], 
680             [ -120.,  1200., -2700.,  1680.], 
681             [  240., -2700.,  6480., -4200.], 
682             [ -140.,  1680., -4200.,  2800.]]) 
683      >>> invhilbert(4, exact=True) 
684      array([[   16,  -120,   240,  -140], 
685             [ -120,  1200, -2700,  1680], 
686             [  240, -2700,  6480, -4200], 
687             [ -140,  1680, -4200,  2800]], dtype=int64) 
688      >>> invhilbert(16)[7,7] 
689      4.2475099528537506e+19 
690      >>> invhilbert(16, exact=True)[7,7] 
691      42475099528537378560 
692   
693      """
694      from scipy.special import comb
695      if exact:
696          if n > 14:
697              dtype = object
698          else:
699              dtype = np.int64
700      else:
701          dtype = np.float64
702      invh = np.empty((n, n), dtype=dtype)
703      for i in range(n):
704          for j in range(0, i + 1):
705              s = i + j
706              invh[i, j] = ((-1) ** s * (s + 1) *
707                            comb(n + i, n - j - 1, exact=exact) *
708                            comb(n + j, n - i - 1, exact=exact) *
709                            comb(s, i, exact=exact) ** 2)
710              if i != j:
711                  invh[j, i] = invh[i, j]
712      return invh

715  def pascal(n, kind='symmetric', exact=True):
716      """ 
717      Returns the n x n Pascal matrix. 
718   
719      The Pascal matrix is a matrix containing the binomial coefficients as 
720      its elements. 
721   
722      Parameters 
723      ---------- 
724      n : int 
725          The size of the matrix to create; that is, the result is an n x n 
726          matrix. 
727      kind : str, optional 
728          Must be one of 'symmetric', 'lower', or 'upper'. 
729          Default is 'symmetric'. 
730      exact : bool, optional 
731          If `exact` is True, the result is either an array of type 
732          numpy.uint64 (if n < 35) or an object array of Python long integers. 
733          If `exact` is False, the coefficients in the matrix are computed using 
734          `scipy.special.comb` with ``exact=False``. The result will be a floating 
735          point array, and the values in the array will not be the exact 
736          coefficients, but this version is much faster than ``exact=True``. 
737   
738      Returns 
739      ------- 
740      p : (n, n) ndarray 
741          The Pascal matrix. 
742   
743      See Also 
744      -------- 
745      invpascal 
746   
747      Notes 
748      ----- 
749      See https://en.wikipedia.org/wiki/Pascal_matrix for more information 
750      about Pascal matrices. 
751   
752      .. versionadded:: 0.11.0 
753   
754      Examples 
755      -------- 
756      >>> from scipy.linalg import pascal 
757      >>> pascal(4) 
758      array([[ 1,  1,  1,  1], 
759             [ 1,  2,  3,  4], 
760             [ 1,  3,  6, 10], 
761             [ 1,  4, 10, 20]], dtype=uint64) 
762      >>> pascal(4, kind='lower') 
763      array([[1, 0, 0, 0], 
764             [1, 1, 0, 0], 
765             [1, 2, 1, 0], 
766             [1, 3, 3, 1]], dtype=uint64) 
767      >>> pascal(50)[-1, -1] 
768      25477612258980856902730428600 
769      >>> from scipy.special import comb 
770      >>> comb(98, 49, exact=True) 
771      25477612258980856902730428600 
772   
773      """
774  
775      from scipy.special import comb
776      if kind not in ['symmetric', 'lower', 'upper']:
777          raise ValueError("kind must be 'symmetric', 'lower', or 'upper'")
778  
779      if exact:
780          if n >= 35:
781              L_n = np.empty((n, n), dtype=object)
782              L_n.fill(0)
783          else:
784              L_n = np.zeros((n, n), dtype=np.uint64)
785          for i in range(n):
786              for j in range(i + 1):
787                  L_n[i, j] = comb(i, j, exact=True)
788      else:
789          L_n = comb(*np.ogrid[:n, :n])
790  
791      if kind == 'lower':
792          p = L_n
793      elif kind == 'upper':
794          p = L_n.T
795      else:
796          p = np.dot(L_n, L_n.T)
797  
798      return p

801  def invpascal(n, kind='symmetric', exact=True):
802      """ 
803      Returns the inverse of the n x n Pascal matrix. 
804   
805      The Pascal matrix is a matrix containing the binomial coefficients as 
806      its elements. 
807   
808      Parameters 
809      ---------- 
810      n : int 
811          The size of the matrix to create; that is, the result is an n x n 
812          matrix. 
813      kind : str, optional 
814          Must be one of 'symmetric', 'lower', or 'upper'. 
815          Default is 'symmetric'. 
816      exact : bool, optional 
817          If `exact` is True, the result is either an array of type 
818          ``numpy.int64`` (if `n` <= 35) or an object array of Python integers. 
819          If `exact` is False, the coefficients in the matrix are computed using 
820          `scipy.special.comb` with `exact=False`. The result will be a floating 
821          point array, and for large `n`, the values in the array will not be the 
822          exact coefficients. 
823   
824      Returns 
825      ------- 
826      invp : (n, n) ndarray 
827          The inverse of the Pascal matrix. 
828   
829      See Also 
830      -------- 
831      pascal 
832   
833      Notes 
834      ----- 
835   
836      .. versionadded:: 0.16.0 
837   
838      References 
839      ---------- 
840      .. [1] "Pascal matrix", https://en.wikipedia.org/wiki/Pascal_matrix 
841      .. [2] Cohen, A. M., "The inverse of a Pascal matrix", Mathematical 
842             Gazette, 59(408), pp. 111-112, 1975. 
843   
844      Examples 
845      -------- 
846      >>> from scipy.linalg import invpascal, pascal 
847      >>> invp = invpascal(5) 
848      >>> invp 
849      array([[  5, -10,  10,  -5,   1], 
850             [-10,  30, -35,  19,  -4], 
851             [ 10, -35,  46, -27,   6], 
852             [ -5,  19, -27,  17,  -4], 
853             [  1,  -4,   6,  -4,   1]]) 
854   
855      >>> p = pascal(5) 
856      >>> p.dot(invp) 
857      array([[ 1.,  0.,  0.,  0.,  0.], 
858             [ 0.,  1.,  0.,  0.,  0.], 
859             [ 0.,  0.,  1.,  0.,  0.], 
860             [ 0.,  0.,  0.,  1.,  0.], 
861             [ 0.,  0.,  0.,  0.,  1.]]) 
862   
863      An example of the use of `kind` and `exact`: 
864   
865      >>> invpascal(5, kind='lower', exact=False) 
866      array([[ 1., -0.,  0., -0.,  0.], 
867             [-1.,  1., -0.,  0., -0.], 
868             [ 1., -2.,  1., -0.,  0.], 
869             [-1.,  3., -3.,  1., -0.], 
870             [ 1., -4.,  6., -4.,  1.]]) 
871   
872      """
873      from scipy.special import comb
874  
875      if kind not in ['symmetric', 'lower', 'upper']:
876          raise ValueError("'kind' must be 'symmetric', 'lower' or 'upper'.")
877  
878      if kind == 'symmetric':
879          if exact:
880              if n > 34:
881                  dt = object
882              else:
883                  dt = np.int64
884          else:
885              dt = np.float64
886          invp = np.empty((n, n), dtype=dt)
887          for i in range(n):
888              for j in range(0, i + 1):
889                  v = 0
890                  for k in range(n - i):
891                      v += comb(i + k, k, exact=exact) * comb(i + k, i + k - j,
892                                                              exact=exact)
893                  invp[i, j] = (-1)**(i - j) * v
894                  if i != j:
895                      invp[j, i] = invp[i, j]
896      else:
897          # For the 'lower' and 'upper' cases, we computer the inverse by
898          # changing the sign of every other diagonal of the pascal matrix.
899          invp = pascal(n, kind=kind, exact=exact)
900          if invp.dtype == np.uint64:
901              # This cast from np.uint64 to int64 OK, because if `kind` is not
902              # "symmetric", the values in invp are all much less than 2**63.
903              invp = invp.view(np.int64)
904  
905          # The toeplitz matrix has alternating bands of 1 and -1.
906          invp *= toeplitz((-1)**np.arange(n)).astype(invp.dtype)
907  
908      return invp

911  def dft(n, scale=None):
912      """ 
913      Discrete Fourier transform matrix. 
914   
915      Create the matrix that computes the discrete Fourier transform of a 
916      sequence [1]_. The nth primitive root of unity used to generate the 
917      matrix is exp(-2*pi*i/n), where i = sqrt(-1). 
918   
919      Parameters 
920      ---------- 
921      n : int 
922          Size the matrix to create. 
923      scale : str, optional 
924          Must be None, 'sqrtn', or 'n'. 
925          If `scale` is 'sqrtn', the matrix is divided by `sqrt(n)`. 
926          If `scale` is 'n', the matrix is divided by `n`. 
927          If `scale` is None (the default), the matrix is not normalized, and the 
928          return value is simply the Vandermonde matrix of the roots of unity. 
929   
930      Returns 
931      ------- 
932      m : (n, n) ndarray 
933          The DFT matrix. 
934   
935      Notes 
936      ----- 
937      When `scale` is None, multiplying a vector by the matrix returned by 
938      `dft` is mathematically equivalent to (but much less efficient than) 
939      the calculation performed by `scipy.fft.fft`. 
940   
941      .. versionadded:: 0.14.0 
942   
943      References 
944      ---------- 
945      .. [1] "DFT matrix", https://en.wikipedia.org/wiki/DFT_matrix 
946   
947      Examples 
948      -------- 
949      >>> import numpy as np 
950      >>> from scipy.linalg import dft 
951      >>> np.set_printoptions(precision=2, suppress=True)  # for compact output 
952      >>> m = dft(5) 
953      >>> m 
954      array([[ 1.  +0.j  ,  1.  +0.j  ,  1.  +0.j  ,  1.  +0.j  ,  1.  +0.j  ], 
955             [ 1.  +0.j  ,  0.31-0.95j, -0.81-0.59j, -0.81+0.59j,  0.31+0.95j], 
956             [ 1.  +0.j  , -0.81-0.59j,  0.31+0.95j,  0.31-0.95j, -0.81+0.59j], 
957             [ 1.  +0.j  , -0.81+0.59j,  0.31-0.95j,  0.31+0.95j, -0.81-0.59j], 
958             [ 1.  +0.j  ,  0.31+0.95j, -0.81+0.59j, -0.81-0.59j,  0.31-0.95j]]) 
959      >>> x = np.array([1, 2, 3, 0, 3]) 
960      >>> m @ x  # Compute the DFT of x 
961      array([ 9.  +0.j  ,  0.12-0.81j, -2.12+3.44j, -2.12-3.44j,  0.12+0.81j]) 
962   
963      Verify that ``m @ x`` is the same as ``fft(x)``. 
964   
965      >>> from scipy.fft import fft 
966      >>> fft(x)     # Same result as m @ x 
967      array([ 9.  +0.j  ,  0.12-0.81j, -2.12+3.44j, -2.12-3.44j,  0.12+0.81j]) 
968      """
969      if scale not in [None, 'sqrtn', 'n']:
970          raise ValueError("scale must be None, 'sqrtn', or 'n'; "
971                           f"{scale!r} is not valid.")
972  
973      omegas = np.exp(-2j * np.pi * np.arange(n) / n).reshape(-1, 1)
974      m = omegas ** np.arange(n)
975      if scale == 'sqrtn':
976          m /= math.sqrt(n)
977      elif scale == 'n':
978          m /= n
979      return m

982  def fiedler(a):
983      """Returns a symmetric Fiedler matrix 
984   
985      Given an sequence of numbers `a`, Fiedler matrices have the structure 
986      ``F[i, j] = np.abs(a[i] - a[j])``, and hence zero diagonals and nonnegative 
987      entries. A Fiedler matrix has a dominant positive eigenvalue and other 
988      eigenvalues are negative. Although not valid generally, for certain inputs, 
989      the inverse and the determinant can be derived explicitly as given in [1]_. 
990   
991      Parameters 
992      ---------- 
993      a : (..., n,) array_like 
994          Coefficient array. N-dimensional arrays are treated as a batch: 
995          each slice along the last axis is a 1-D coefficient array. 
996   
997      Returns 
998      ------- 
999      F : (..., n, n) ndarray 
1000         Fiedler matrix. For batch input, each slice of shape ``(n, n)`` 
1001         along the last two dimensions of the output corresponds with a 
1002         slice of shape ``(n,)`` along the last dimension of the input. 
1003  
1004     See Also 
1005     -------- 
1006     circulant, toeplitz 
1007  
1008     Notes 
1009     ----- 
1010  
1011     .. versionadded:: 1.3.0 
1012  
1013     References 
1014     ---------- 
1015     .. [1] J. Todd, "Basic Numerical Mathematics: Vol.2 : Numerical Algebra", 
1016         1977, Birkhauser, :doi:`10.1007/978-3-0348-7286-7` 
1017  
1018     Examples 
1019     -------- 
1020     >>> import numpy as np 
1021     >>> from scipy.linalg import det, inv, fiedler 
1022     >>> a = [1, 4, 12, 45, 77] 
1023     >>> n = len(a) 
1024     >>> A = fiedler(a) 
1025     >>> A 
1026     array([[ 0,  3, 11, 44, 76], 
1027            [ 3,  0,  8, 41, 73], 
1028            [11,  8,  0, 33, 65], 
1029            [44, 41, 33,  0, 32], 
1030            [76, 73, 65, 32,  0]]) 
1031  
1032     The explicit formulas for determinant and inverse seem to hold only for 
1033     monotonically increasing/decreasing arrays. Note the tridiagonal structure 
1034     and the corners. 
1035  
1036     >>> Ai = inv(A) 
1037     >>> Ai[np.abs(Ai) < 1e-12] = 0.  # cleanup the numerical noise for display 
1038     >>> Ai 
1039     array([[-0.16008772,  0.16666667,  0.        ,  0.        ,  0.00657895], 
1040            [ 0.16666667, -0.22916667,  0.0625    ,  0.        ,  0.        ], 
1041            [ 0.        ,  0.0625    , -0.07765152,  0.01515152,  0.        ], 
1042            [ 0.        ,  0.        ,  0.01515152, -0.03077652,  0.015625  ], 
1043            [ 0.00657895,  0.        ,  0.        ,  0.015625  , -0.00904605]]) 
1044     >>> det(A) 
1045     15409151.999999998 
1046     >>> (-1)**(n-1) * 2**(n-2) * np.diff(a).prod() * (a[-1] - a[0]) 
1047     15409152 
1048  
1049     """
1050     a = np.atleast_1d(a)
1051 
1052     if a.ndim > 1:
1053         return np.apply_along_axis(fiedler, -1, a)
1054 
1055     if a.size == 0:
1056         return np.array([], dtype=float)
1057     elif a.size == 1:
1058         return np.array([[0.]])
1059     else:
1060         return np.abs(a[:, None] - a)

1063 def fiedler_companion(a):
1064     """ Returns a Fiedler companion matrix 
1065  
1066     Given a polynomial coefficient array ``a``, this function forms a 
1067     pentadiagonal matrix with a special structure whose eigenvalues coincides 
1068     with the roots of ``a``. 
1069  
1070     Parameters 
1071     ---------- 
1072     a : (..., N) array_like 
1073         1-D array of polynomial coefficients in descending order with a nonzero 
1074         leading coefficient. For ``N < 2``, an empty array is returned. 
1075         N-dimensional arrays are treated as a batch: each slice along the last 
1076         axis is a 1-D array of polynomial coefficients. 
1077  
1078     Returns 
1079     ------- 
1080     c : (..., N-1, N-1) ndarray 
1081         Resulting companion matrix. For batch input, each slice of shape 
1082         ``(N-1, N-1)`` along the last two dimensions of the output corresponds 
1083         with a slice of shape ``(N,)`` along the last dimension of the input. 
1084  
1085     See Also 
1086     -------- 
1087     companion 
1088  
1089     Notes 
1090     ----- 
1091     Similar to `companion`, each leading coefficient along the last axis of the 
1092     input should be nonzero. 
1093     If the leading coefficient is not 1, other coefficients are rescaled before 
1094     the array generation. To avoid numerical issues, it is best to provide a 
1095     monic polynomial. 
1096  
1097     .. versionadded:: 1.3.0 
1098  
1099     References 
1100     ---------- 
1101     .. [1] M. Fiedler, " A note on companion matrices", Linear Algebra and its 
1102         Applications, 2003, :doi:`10.1016/S0024-3795(03)00548-2` 
1103  
1104     Examples 
1105     -------- 
1106     >>> import numpy as np 
1107     >>> from scipy.linalg import fiedler_companion, eigvals 
1108     >>> p = np.poly(np.arange(1, 9, 2))  # [1., -16., 86., -176., 105.] 
1109     >>> fc = fiedler_companion(p) 
1110     >>> fc 
1111     array([[  16.,  -86.,    1.,    0.], 
1112            [   1.,    0.,    0.,    0.], 
1113            [   0.,  176.,    0., -105.], 
1114            [   0.,    1.,    0.,    0.]]) 
1115     >>> eigvals(fc) 
1116     array([7.+0.j, 5.+0.j, 3.+0.j, 1.+0.j]) 
1117  
1118     """
1119     a = np.atleast_1d(a)
1120 
1121     if a.ndim > 1:
1122         return np.apply_along_axis(fiedler_companion, -1, a)
1123 
1124     if a.size <= 2:
1125         if a.size == 2:
1126             return np.array([[-(a/a[0])[-1]]])
1127         return np.array([], dtype=a.dtype)
1128 
1129     if a[0] == 0.:
1130         raise ValueError('Leading coefficient is zero.')
1131 
1132     a = a/a[0]
1133     n = a.size - 1
1134     c = np.zeros((n, n), dtype=a.dtype)
1135     # subdiagonals
1136     c[range(3, n, 2), range(1, n-2, 2)] = 1.
1137     c[range(2, n, 2), range(1, n-1, 2)] = -a[3::2]
1138     # superdiagonals
1139     c[range(0, n-2, 2), range(2, n, 2)] = 1.
1140     c[range(0, n-1, 2), range(1, n, 2)] = -a[2::2]
1141     c[[0, 1], 0] = [-a[1], 1]
1142 
1143     return c

1146 def convolution_matrix(a, n, mode='full'):
1147     """ 
1148     Construct a convolution matrix. 
1149  
1150     Constructs the Toeplitz matrix representing one-dimensional 
1151     convolution [1]_.  See the notes below for details. 
1152  
1153     Parameters 
1154     ---------- 
1155     a : (..., m) array_like 
1156         The 1-D array to convolve. N-dimensional arrays are treated as a 
1157         batch: each slice along the last axis is a 1-D array to convolve. 
1158     n : int 
1159         The number of columns in the resulting matrix.  It gives the length 
1160         of the input to be convolved with `a`.  This is analogous to the 
1161         length of `v` in ``numpy.convolve(a, v)``. 
1162     mode : str 
1163         This is analogous to `mode` in ``numpy.convolve(v, a, mode)``. 
1164         It must be one of ('full', 'valid', 'same'). 
1165         See below for how `mode` determines the shape of the result. 
1166  
1167     Returns 
1168     ------- 
1169     A : (..., k, n) ndarray 
1170         The convolution matrix whose row count `k` depends on `mode`:: 
1171  
1172             =======  ========================= 
1173              mode    k 
1174             =======  ========================= 
1175             'full'   m + n -1 
1176             'same'   max(m, n) 
1177             'valid'  max(m, n) - min(m, n) + 1 
1178             =======  ========================= 
1179  
1180         For batch input, each slice of shape ``(k, n)`` along the last two 
1181         dimensions of the output corresponds with a slice of shape ``(m,)`` 
1182         along the last dimension of the input. 
1183  
1184     See Also 
1185     -------- 
1186     toeplitz : Toeplitz matrix 
1187  
1188     Notes 
1189     ----- 
1190     The code:: 
1191  
1192         A = convolution_matrix(a, n, mode) 
1193  
1194     creates a Toeplitz matrix `A` such that ``A @ v`` is equivalent to 
1195     using ``convolve(a, v, mode)``.  The returned array always has `n` 
1196     columns.  The number of rows depends on the specified `mode`, as 
1197     explained above. 
1198  
1199     In the default 'full' mode, the entries of `A` are given by:: 
1200  
1201         A[i, j] == (a[i-j] if (0 <= (i-j) < m) else 0) 
1202  
1203     where ``m = len(a)``.  Suppose, for example, the input array is 
1204     ``[x, y, z]``.  The convolution matrix has the form:: 
1205  
1206         [x, 0, 0, ..., 0, 0] 
1207         [y, x, 0, ..., 0, 0] 
1208         [z, y, x, ..., 0, 0] 
1209         ... 
1210         [0, 0, 0, ..., x, 0] 
1211         [0, 0, 0, ..., y, x] 
1212         [0, 0, 0, ..., z, y] 
1213         [0, 0, 0, ..., 0, z] 
1214  
1215     In 'valid' mode, the entries of `A` are given by:: 
1216  
1217         A[i, j] == (a[i-j+m-1] if (0 <= (i-j+m-1) < m) else 0) 
1218  
1219     This corresponds to a matrix whose rows are the subset of those from 
1220     the 'full' case where all the coefficients in `a` are contained in the 
1221     row.  For input ``[x, y, z]``, this array looks like:: 
1222  
1223         [z, y, x, 0, 0, ..., 0, 0, 0] 
1224         [0, z, y, x, 0, ..., 0, 0, 0] 
1225         [0, 0, z, y, x, ..., 0, 0, 0] 
1226         ... 
1227         [0, 0, 0, 0, 0, ..., x, 0, 0] 
1228         [0, 0, 0, 0, 0, ..., y, x, 0] 
1229         [0, 0, 0, 0, 0, ..., z, y, x] 
1230  
1231     In the 'same' mode, the entries of `A` are given by:: 
1232  
1233         d = (m - 1) // 2 
1234         A[i, j] == (a[i-j+d] if (0 <= (i-j+d) < m) else 0) 
1235  
1236     The typical application of the 'same' mode is when one has a signal of 
1237     length `n` (with `n` greater than ``len(a)``), and the desired output 
1238     is a filtered signal that is still of length `n`. 
1239

1306

1320         tb = trim//2
