28   class Adadelta(Optimizer):
29       def __init__(
30           self,
31           params: ParamsT,
32           lr: Union[float, Tensor] = 1.0,
33           rho: float = 0.9,
34           eps: float = 1e-6,
35           weight_decay: float = 0,
36           foreach: Optional[bool] = None,
37           *,
38           capturable: bool = False,
39           maximize: bool = False,
40           differentiable: bool = False,
41       ):
42           if isinstance(lr, Tensor) and lr.numel() != 1:
43               raise ValueError("Tensor lr must be 1-element")
44           if not 0.0 <= lr:
45               raise ValueError(f"Invalid learning rate: {lr}")
46           if not 0.0 <= rho <= 1.0:
47               raise ValueError(f"Invalid rho value: {rho}")
48           if not 0.0 <= eps:
49               raise ValueError(f"Invalid epsilon value: {eps}")
50           if not 0.0 <= weight_decay:
51               raise ValueError(f"Invalid weight_decay value: {weight_decay}")
52   
53           defaults = dict(
54               lr=lr,
55               rho=rho,
56               eps=eps,
57               weight_decay=weight_decay,
58               maximize=maximize,
59               capturable=capturable,
60               foreach=foreach,
61               differentiable=differentiable,
62           )
63           super().__init__(params, defaults)
64   
65       def __setstate__(self, state):
66           super().__setstate__(state)
67           for group in self.param_groups:
68               group.setdefault("foreach", None)
69               group.setdefault("maximize", False)
70               group.setdefault("differentiable", False)
71               group.setdefault("capturable", False)
72               for p in group["params"]:
73                   p_state = self.state.get(p, [])
74                   if len(p_state) != 0 and not torch.is_tensor(p_state["step"]):
75                       step_val = float(p_state["step"])
76                       p_state["step"] = (
77                           torch.tensor(
78                               step_val, dtype=_get_scalar_dtype(), device=p.device
79                           )
80                           if group["capturable"]
81                           else torch.tensor(step_val, dtype=_get_scalar_dtype())
82                       )
83   
84       def _init_group(
85           self,
86           group: dict[str, Any],
87           params_with_grad: list[Tensor],
88           grads: list[Tensor],
89           square_avgs: list[Tensor],
90           acc_deltas: list[Tensor],
91           state_steps: list[Tensor],
92       ):
93           has_complex = False
94           p: Tensor
95           for p in group["params"]:
96               if p.grad is None:
97                   continue
98               has_complex |= torch.is_complex(p)
99               params_with_grad.append(p)
100              if p.grad.is_sparse:
101                  raise RuntimeError("Adadelta does not support sparse gradients")
102              grads.append(p.grad)
103  
104              state = self.state[p]
105  
106              # Lazy state initialization
107              if len(state) == 0:
108                  state["step"] = (
109                      torch.zeros((), dtype=_get_scalar_dtype(), device=p.device)
110                      if group["capturable"]
111                      else torch.zeros((), dtype=_get_scalar_dtype())
112                  )
113  
114                  state["square_avg"] = torch.zeros_like(
115                      p, memory_format=torch.preserve_format
116                  )
117                  state["acc_delta"] = torch.zeros_like(
118                      p, memory_format=torch.preserve_format
119                  )
120  
121              square_avgs.append(state["square_avg"])
122              acc_deltas.append(state["acc_delta"])
123              state_steps.append(state["step"])
124  
125          return has_complex

127      @_use_grad_for_differentiable
128      def step(self, closure=None):
129          """Perform a single optimization step. 
130   
131          Args: 
132              closure (Callable, optional): A closure that reevaluates the model 
133                  and returns the loss. 
134          """
135          self._cuda_graph_capture_health_check()
136  
137          loss = None
138          if closure is not None:
139              with torch.enable_grad():
140                  loss = closure()
141  
142          for group in self.param_groups:
143              params_with_grad: list[Tensor] = []
144              grads: list[Tensor] = []
145              square_avgs: list[Tensor] = []
146              acc_deltas: list[Tensor] = []
147              state_steps: list[Tensor] = []
148              (
149                  lr,
150                  rho,
151                  eps,
152                  weight_decay,
153                  foreach,
154                  maximize,
155                  differentiable,
156                  capturable,
157              ) = (
158                  group["lr"],
159                  group["rho"],
160                  group["eps"],
161                  group["weight_decay"],
162                  group["foreach"],
163                  group["maximize"],
164                  group["differentiable"],
165                  group["capturable"],
166              )
167  
168              has_complex = self._init_group(
169                  group, params_with_grad, grads, square_avgs, acc_deltas, state_steps
170              )
171  
172              adadelta(
173                  params_with_grad,
174                  grads,
175                  square_avgs,
176                  acc_deltas,
177                  state_steps,
178                  lr=lr,
179                  rho=rho,
180                  eps=eps,
181                  weight_decay=weight_decay,
182                  foreach=foreach,
183                  maximize=maximize,
184                  differentiable=differentiable,
185                  capturable=capturable,
186                  has_complex=has_complex,
187              )
188  
189          return loss

192  Adadelta.__doc__ = (
193      r"""Implements Adadelta algorithm. 
194   
195      .. math:: 
196         \begin{aligned} 
197              &\rule{110mm}{0.4pt}                                                                 \\ 
198              &\textbf{input}      : \gamma \text{ (lr)}, \: \theta_0 \text{ (params)}, 
199                  \: f(\theta) \text{ (objective)}, \: \rho \text{ (decay)}, 
200                  \: \lambda \text{ (weight decay)}                                                \\ 
201              &\textbf{initialize} :  v_0  \leftarrow 0 \: \text{ (square avg)}, 
202                  \: u_0 \leftarrow 0 \: \text{ (accumulate variables)}                     \\[-1.ex] 
203              &\rule{110mm}{0.4pt}                                                                 \\ 
204              &\textbf{for} \: t=1 \: \textbf{to} \: \ldots \: \textbf{do}                         \\ 
205              &\hspace{5mm}g_t           \leftarrow   \nabla_{\theta} f_t (\theta_{t-1})           \\ 
206              &\hspace{5mm}if \: \lambda \neq 0                                                    \\ 
207              &\hspace{10mm} g_t \leftarrow g_t + \lambda  \theta_{t-1}                            \\ 
208              &\hspace{5mm} v_t      \leftarrow v_{t-1} \rho + g^2_t (1 - \rho)                    \\ 
209              &\hspace{5mm}\Delta x_t    \leftarrow   \frac{\sqrt{u_{t-1} + 
210                  \epsilon }}{ \sqrt{v_t + \epsilon}  }g_t \hspace{21mm}                           \\ 
211              &\hspace{5mm} u_t  \leftarrow   u_{t-1}  \rho + 
212                   \Delta x^2_t  (1 - \rho)                                                        \\ 
213              &\hspace{5mm}\theta_t      \leftarrow   \theta_{t-1} - \gamma  \Delta x_t            \\ 
214              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
215              &\bf{return} \:  \theta_t                                                     \\[-1.ex] 
216              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
217         \end{aligned} 
218   
219      For further details regarding the algorithm we refer to `ADADELTA: An Adaptive Learning Rate Method`_. 
220      """
221      + rf"""
222      Args:
223          {_params_doc}
224          lr (float, Tensor, optional): coefficient that scale delta before it is applied
225              to the parameters (default: 1.0)
226          rho (float, optional): coefficient used for computing a running average
227              of squared gradients (default: 0.9). A higher value of `rho` will
228              result in a slower average, which can be helpful for preventing
229              oscillations in the learning process.
230          eps (float, optional): term added to the denominator to improve
231              numerical stability (default: 1e-6).
232          weight_decay (float, optional): weight decay (L2 penalty) (default: 0)
233          {_foreach_doc}
234          {_capturable_doc}
235          {_maximize_doc}
236          {_differentiable_doc}
237  
238      .. _ADADELTA\: An Adaptive Learning Rate Method:
239          https://arxiv.org/abs/1212.5701
240  
241      """

245  def _single_tensor_adadelta(
246      params: list[Tensor],
247      grads: list[Tensor],
248      square_avgs: list[Tensor],
249      acc_deltas: list[Tensor],
250      state_steps: list[Tensor],
251      *,
252      lr: float,
253      rho: float,
254      eps: float,
255      weight_decay: float,
256      maximize: bool,
257      differentiable: bool,
258      capturable: bool,
259      has_complex: bool,
260  ):
261      # If compiling, the compiler will handle cudagraph checks, see note [torch.compile x capturable]
262      if not torch.compiler.is_compiling() and capturable:
263          capturable_supported_devices = _get_capturable_supported_devices(
264              supports_xla=False
265          )
266          assert all(
267              p.device.type == step.device.type
268              and p.device.type in capturable_supported_devices
269              for p, step in zip(params, state_steps)
270          ), (
271              f"If capturable=True, params and state_steps must be on supported devices: {capturable_supported_devices}."
272          )
273  
274      if not torch.jit.is_scripting():
275          lr = _to_scalar(lr)
276  
277      for param, grad, square_avg, acc_delta, step in zip(
278          params, grads, square_avgs, acc_deltas, state_steps
279      ):
280          step += 1
281          grad = grad if not maximize else -grad
282  
283          if weight_decay != 0:
284              grad = grad.add(param, alpha=weight_decay)
285  
286          if torch.is_complex(param):
287              square_avg = torch.view_as_real(square_avg)
288              acc_delta = torch.view_as_real(acc_delta)
289              grad = torch.view_as_real(grad)
290  
291          square_avg.mul_(rho).addcmul_(grad, grad, value=1 - rho)
292          std = square_avg.add(eps).sqrt_()
293          delta = acc_delta.add(eps).sqrt_()
294          if differentiable:
295              delta = delta.clone()
296          delta.div_(std).mul_(grad)
297          acc_delta.mul_(rho).addcmul_(delta, delta, value=1 - rho)
298  
299          if torch.is_complex(param):
300              delta = torch.view_as_complex(delta)
301          param.add_(delta, alpha=-lr)

304  def _multi_tensor_adadelta(
305      params: list[Tensor],
306      grads: list[Tensor],
307      square_avgs: list[Tensor],
308      acc_deltas: list[Tensor],
309      state_steps: list[Tensor],
310      *,
311      lr: float,
312      rho: float,
313      eps: float,
314      weight_decay: float,
315      maximize: bool,
316      differentiable: bool,
317      capturable: bool,
318      has_complex: bool,
319  ):
320      assert not differentiable, "_foreach ops don't support autograd"
321  
322      # If compiling, the compiler will handle cudagraph checks, see note [torch.compile x capturable]
323      if not torch.compiler.is_compiling() and capturable:
324          capturable_supported_devices = _get_capturable_supported_devices(
325              supports_xla=False
326          )
327          assert all(
328              p.device.type == step.device.type
329              and p.device.type in capturable_supported_devices
330              for p, step in zip(params, state_steps)
331          ), (
332              f"If capturable=True, params and state_steps must be on supported devices: {capturable_supported_devices}."
333          )
334  
335      if len(params) == 0:
336          return
337  
338      lr = _to_scalar(lr)
339  
340      grouped_tensors = Optimizer._group_tensors_by_device_and_dtype(
341          [params, grads, square_avgs, acc_deltas, state_steps]  # type: ignore[list-item]
342      )
343      for (
344          device_params_,
345          device_grads_,
346          device_square_avgs_,
347          device_acc_deltas_,
348          device_state_steps_,
349      ), _ in grouped_tensors.values():
350          device_params = cast(list[Tensor], device_params_)
351          device_grads = cast(list[Tensor], device_grads_)
352          device_square_avgs = cast(list[Tensor], device_square_avgs_)
353          device_acc_deltas = cast(list[Tensor], device_acc_deltas_)
354          device_state_steps = cast(list[Tensor], device_state_steps_)
355          if has_complex:
356              _view_as_real(
357                  device_params, device_grads, device_square_avgs, device_acc_deltas
358              )
359  
360          # Update steps
361          # If steps are on CPU, foreach will fall back to the slow path, which is a for-loop calling t.add(1) over
362          # and over. 1 will then be wrapped into a Tensor over and over again, which is slower than if we just
363          # wrapped it once now. The alpha is required to assure we go to the right overload.
364          if not torch.compiler.is_compiling() and device_state_steps[0].is_cpu:
365              torch._foreach_add_(
366                  device_state_steps, torch.tensor(1.0, device="cpu"), alpha=1.0
367              )
368          else:
369              torch._foreach_add_(device_state_steps, 1)
370  
371          if maximize:
372              device_grads = torch._foreach_neg(device_grads)  # type: ignore[assignment]
373  
374          if weight_decay != 0:
375              # Re-use the intermediate memory (device_grads) already allocated for maximize
376              if maximize:
377                  torch._foreach_add_(device_grads, device_params, alpha=weight_decay)
378              else:
379                  device_grads = torch._foreach_add(  # type: ignore[assignment]
380                      device_grads, device_params, alpha=weight_decay
381                  )
382  
383          torch._foreach_mul_(device_square_avgs, rho)
384          torch._foreach_addcmul_(
385              device_square_avgs, device_grads, device_grads, value=1 - rho
386          )
387  
388          std = torch._foreach_add(device_square_avgs, eps)
389          torch._foreach_sqrt_(std)
390  
391          deltas = torch._foreach_add(device_acc_deltas, eps)
392          torch._foreach_sqrt_(deltas)
393          torch._foreach_div_(deltas, std)
394          torch._foreach_mul_(deltas, device_grads)
395  
396          torch._foreach_mul_(device_acc_deltas, rho)
397          torch._foreach_addcmul_(device_acc_deltas, deltas, deltas, value=1 - rho)
398  
399          # If LR is a tensor, the else branch will internally call item()
400          # which will cause silent incorrectness if we are capturing
401          if capturable and isinstance(lr, torch.Tensor):
402              torch._foreach_mul_(deltas, -lr)
403              torch._foreach_add_(device_params, deltas)
404          else:
405              torch._foreach_add_(device_params, deltas, alpha=-lr)

408  @_disable_dynamo_if_unsupported(single_tensor_fn=_single_tensor_adadelta)
409  def adadelta(
410      params: list[Tensor],
411      grads: list[Tensor],
412      square_avgs: list[Tensor],
413      acc_deltas: list[Tensor],
414      state_steps: list[Tensor],
415      # kwonly args with defaults are not supported by functions compiled with torchscript issue #70627
416      # setting this as kwarg for now as functional API is compiled by torch/distributed/optim
417      capturable: bool = False,
418      foreach: Optional[bool] = None,
419      differentiable: bool = False,
420      has_complex: bool = False,
421      *,
422      lr: float,
423      rho: float,
424      eps: float,
425      weight_decay: float,
426      maximize: bool,
427  ):
428      r"""Functional API that performs Adadelta algorithm computation. 
429   
430      See :class:`~torch.optim.Adadelta` for details. 
431      """
432  
433      # this check is slow during compilation, so we skip it
434      # if it's strictly needed we can add this check back in dynamo
435      if not torch.compiler.is_compiling() and not all(
436          isinstance(t, torch.Tensor) for t in state_steps
437      ):
438          raise RuntimeError(
439              "API has changed, `state_steps` argument must contain a list of singleton tensors"
440          )
441  
442      # We still respect when the user inputs False for foreach.
443      if foreach is None:
444          _, foreach = _default_to_fused_or_foreach(
445              params, differentiable, use_fused=False
446          )
447  
448      if foreach and torch.jit.is_scripting():
449          raise RuntimeError("torch.jit.script not supported with foreach optimizers")
450  
451      if foreach and not torch.jit.is_scripting():
452          func = _multi_tensor_adadelta
453      else:
454          func = _single_tensor_adadelta
455  
456      func(
457          params,
458          grads,
459          square_avgs,
460          acc_deltas,
461          state_steps,
462          lr=lr,
463          rho=rho,
464          eps=eps,
465          weight_decay=weight_decay,
466          maximize=maximize,
467          differentiable=differentiable,
468          capturable=capturable,
469          has_complex=has_complex,
470      )
