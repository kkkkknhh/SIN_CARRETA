42   def _get_residuals(x, y, t, k, w):
43       # FITPACK has (w*(spl(x)-y))**2; make_lsq_spline has w*(spl(x)-y)**2
44       w2 = w**2
45   
46       # inline the relevant part of
47       # >>> spl = make_lsq_spline(x, y, w=w2, t=t, k=k)
48       # NB:
49       #     1. y is assumed to be 2D here. For 1D case (parametric=False),
50       #        the call must have been preceded by y = y[:, None] (cf _validate_inputs)
51       #     2. We always sum the squares across axis=1:
52       #         * For 1D (parametric=False), the last dimension has size one,
53       #           so the summation is a no-op.
54       #         * For 2D (parametric=True), the summation is actually how the
55       #           'residuals' are defined, see Eq. (42) in Dierckx1982
56       #           (the reference is in the docstring of `class F`) below.
57       _, _, c = _lsq_solve_qr(x, y, t, k, w)
58       c = np.ascontiguousarray(c)
59       spl = BSpline(t, c, k)
60       return _compute_residuals(w2, spl(x), y)

63   def _compute_residuals(w2, splx, y):
64       delta = ((splx - y)**2).sum(axis=1)
65       return w2 * delta

68   def add_knot(x, t, k, residuals):
69       """Add a new knot. 
70    
71       (Approximately) replicate FITPACK's logic: 
72         1. split the `x` array into knot intervals, ``t(j+k) <= x(i) <= t(j+k+1)`` 
73         2. find the interval with the maximum sum of residuals 
74         3. insert a new knot into the middle of that interval. 
75    
76       NB: a new knot is in fact an `x` value at the middle of the interval. 
77       So *the knots are a subset of `x`*. 
78    
79       This routine is an analog of 
80       https://github.com/scipy/scipy/blob/v1.11.4/scipy/interpolate/fitpack/fpcurf.f#L190-L215 
81       (cf _split function) 
82    
83       and https://github.com/scipy/scipy/blob/v1.11.4/scipy/interpolate/fitpack/fpknot.f 
84       """
85       new_knot = _dierckx.fpknot(x, t, k, residuals)
86   
87       idx_t = np.searchsorted(t, new_knot)
88       t_new = np.r_[t[:idx_t], new_knot, t[idx_t:]]
89       return t_new

92   def _validate_inputs(x, y, w, k, s, xb, xe, parametric):
93       """Common input validations for generate_knots and make_splrep. 
94       """
95       x = np.asarray(x, dtype=float)
96       y = np.asarray(y, dtype=float)
97   
98       if w is None:
99           w = np.ones_like(x, dtype=float)
100      else:
101          w = np.asarray(w, dtype=float)
102          if w.ndim != 1:
103              raise ValueError(f"{w.ndim = } not implemented yet.")
104          if (w < 0).any():
105              raise ValueError("Weights must be non-negative")
106  
107      if y.ndim == 0 or y.ndim > 2:
108          raise ValueError(f"{y.ndim = } not supported (must be 1 or 2.)")
109  
110      parametric = bool(parametric)
111      if parametric:
112          if y.ndim != 2:
113              raise ValueError(f"{y.ndim = } != 2 not supported with {parametric =}.")
114      else:
115          if y.ndim != 1:
116              raise ValueError(f"{y.ndim = } != 1 not supported with {parametric =}.")
117          # all _impl functions expect y.ndim = 2
118          y = y[:, None]
119  
120      if w.shape[0] != x.shape[0]:
121          raise ValueError(f"Weights is incompatible: {w.shape =} != {x.shape}.")
122  
123      if x.shape[0] != y.shape[0]:
124          raise ValueError(f"Data is incompatible: {x.shape = } and {y.shape = }.")
125      if x.ndim != 1 or (x[1:] < x[:-1]).any():
126          raise ValueError("Expect `x` to be an ordered 1D sequence.")
127  
128      k = operator.index(k)
129  
130      if s < 0:
131          raise ValueError(f"`s` must be non-negative. Got {s = }")
132  
133      if xb is None:
134          xb = min(x)
135      if xe is None:
136          xe = max(x)
137  
138      return x, y, w, k, s, xb, xe

141  def generate_knots(x, y, *, w=None, xb=None, xe=None, k=3, s=0, nest=None):
142      """Replicate FITPACK's constructing the knot vector. 
143   
144      Parameters 
145      ---------- 
146      x, y : array_like 
147          The data points defining the curve ``y = f(x)``. 
148      w : array_like, optional 
149          Weights. 
150      xb : float, optional 
151          The boundary of the approximation interval. If None (default), 
152          is set to ``x[0]``. 
153      xe : float, optional 
154          The boundary of the approximation interval. If None (default), 
155          is set to ``x[-1]``. 
156      k : int, optional 
157          The spline degree. Default is cubic, ``k = 3``. 
158      s : float, optional 
159          The smoothing factor. Default is ``s = 0``. 
160      nest : int, optional 
161          Stop when at least this many knots are placed. 
162   
163      Yields 
164      ------ 
165      t : ndarray 
166          Knot vectors with an increasing number of knots. 
167          The generator is finite: it stops when the smoothing critetion is 
168          satisfied, or when then number of knots exceeds the maximum value: 
169          the user-provided `nest` or `x.size + k + 1` --- which is the knot vector 
170          for the interpolating spline. 
171   
172      Examples 
173      -------- 
174      Generate some noisy data and fit a sequence of LSQ splines: 
175   
176      >>> import numpy as np 
177      >>> import matplotlib.pyplot as plt 
178      >>> from scipy.interpolate import make_lsq_spline, generate_knots 
179      >>> rng = np.random.default_rng(12345) 
180      >>> x = np.linspace(-3, 3, 50) 
181      >>> y = np.exp(-x**2) + 0.1 * rng.standard_normal(size=50) 
182   
183      >>> knots = list(generate_knots(x, y, s=1e-10)) 
184      >>> for t in knots[::3]: 
185      ...     spl = make_lsq_spline(x, y, t) 
186      ...     xs = xs = np.linspace(-3, 3, 201) 
187      ...     plt.plot(xs, spl(xs), '-', label=f'n = {len(t)}', lw=3, alpha=0.7) 
188      >>> plt.plot(x, y, 'o', label='data') 
189      >>> plt.plot(xs, np.exp(-xs**2), '--') 
190      >>> plt.legend() 
191   
192      Note that increasing the number of knots make the result follow the data 
193      more and more closely. 
194   
195      Also note that a step of the generator may add multiple knots: 
196   
197      >>> [len(t) for t in knots] 
198      [8, 9, 10, 12, 16, 24, 40, 48, 52, 54] 
199   
200      Notes 
201      ----- 
202      The routine generates successive knots vectors of increasing length, starting 
203      from ``2*(k+1)`` to ``len(x) + k + 1``, trying to make knots more dense 
204      in the regions where the deviation of the LSQ spline from data is large. 
205   
206      When the maximum number of knots, ``len(x) + k + 1`` is reached 
207      (this happens when ``s`` is small and ``nest`` is large), the generator 
208      stops, and the last output is the knots for the interpolation with the 
209      not-a-knot boundary condition. 
210   
211      Knots are located at data sites, unless ``k`` is even and the number of knots 
212      is ``len(x) + k + 1``. In that case, the last output of the generator 
213      has internal knots at Greville sites, ``(x[1:] + x[:-1]) / 2``. 
214   
215      .. versionadded:: 1.15.0 
216   
217      """
218      if s == 0:
219          if nest is not None or w is not None:
220              raise ValueError("s == 0 is interpolation only")
221          t = _not_a_knot(x, k)
222          yield t
223          return
224  
225      x, y, w, k, s, xb, xe = _validate_inputs(
226          x, y, w, k, s, xb, xe, parametric=np.ndim(y) == 2
227      )
228  
229      yield from _generate_knots_impl(x, y, w=w, xb=xb, xe=xe, k=k, s=s, nest=nest)

232  def _generate_knots_impl(x, y, *, w=None, xb=None, xe=None, k=3, s=0, nest=None):
233  
234      acc = s * TOL
235      m = x.size    # the number of data points
236  
237      if nest is None:
238          # the max number of knots. This is set in _fitpack_impl.py line 274
239          # and fitpack.pyf line 198
240          nest = max(m + k + 1, 2*k + 3)
241      else:
242          if nest < 2*(k + 1):
243              raise ValueError(f"`nest` too small: {nest = } < 2*(k+1) = {2*(k+1)}.")
244  
245      nmin = 2*(k + 1)    # the number of knots for an LSQ polynomial approximation
246      nmax = m + k + 1  # the number of knots for the spline interpolation
247  
248      # start from no internal knots
249      t = np.asarray([xb]*(k+1) + [xe]*(k+1), dtype=float)
250      n = t.shape[0]
251      fp = 0.0
252      fpold = 0.0
253  
254      # c  main loop for the different sets of knots. m is a safe upper bound
255      # c  for the number of trials.
256      for _ in range(m):
257          yield t
258  
259          # construct the LSQ spline with this set of knots
260          fpold = fp
261          residuals = _get_residuals(x, y, t, k, w=w)
262          fp = residuals.sum()
263          fpms = fp - s
264  
265          # c  test whether the approximation sinf(x) is an acceptable solution.
266          # c  if f(p=inf) < s accept the choice of knots.
267          if (abs(fpms) < acc) or (fpms < 0):
268              return
269  
270          # ### c  increase the number of knots. ###
271  
272          # c  determine the number of knots nplus we are going to add.
273          if n == nmin:
274              # the first iteration
275              nplus = 1
276          else:
277              delta = fpold - fp
278              npl1 = int(nplus * fpms / delta) if delta > acc else nplus*2
279              nplus = min(nplus*2, max(npl1, nplus//2, 1))
280  
281          # actually add knots
282          for j in range(nplus):
283              t = add_knot(x, t, k, residuals)
284  
285              # check if we have enough knots already
286  
287              n = t.shape[0]
288              # c  if n = nmax, sinf(x) is an interpolating spline.
289              # c  if n=nmax we locate the knots as for interpolation.
290              if n >= nmax:
291                  t = _not_a_knot(x, k)
292                  yield t
293                  return
294  
295              # c  if n=nest we cannot increase the number of knots because of
296              # c  the storage capacity limitation.
297              if n >= nest:
298                  yield t
299                  return
300  
301              # recompute if needed
302              if j < nplus - 1:
303                  residuals = _get_residuals(x, y, t, k, w=w)
304  
305      # this should never be reached
306      return
307

309  #   cccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
310  #   c  part 2: determination of the smoothing spline sp(x).                c
311  #   c  ***************************************************                 c
312  #   c  we have determined the number of knots and their position.          c
313  #   c  we now compute the b-spline coefficients of the smoothing spline    c
314  #   c  sp(x). the observation matrix a is extended by the rows of matrix   c
315  #   c  b expressing that the kth derivative discontinuities of sp(x) at    c
316  #   c  the interior knots t(k+2),...t(n-k-1) must be zero. the corres-     c
317  #   c  ponding weights of these additional rows are set to 1/p.            c
318  #   c  iteratively we then have to determine the value of p such that      c
319  #   c  f(p)=sum((w(i)*(y(i)-sp(x(i))))**2) be = s. we already know that    c
320  #   c  the least-squares kth degree polynomial corresponds to p=0, and     c
321  #   c  that the least-squares spline corresponds to p=infinity. the        c
322  #   c  iteration process which is proposed here, makes use of rational     c
323  #   c  interpolation. since f(p) is a convex and strictly decreasing       c
324  #   c  function of p, it can be approximated by a rational function        c
325  #   c  r(p) = (u*p+v)/(p+w). three values of p(p1,p2,p3) with correspond-  c
326  #   c  ing values of f(p) (f1=f(p1)-s,f2=f(p2)-s,f3=f(p3)-s) are used      c
327  #   c  to calculate the new value of p such that r(p)=s. convergence is    c
328  #   c  guaranteed by taking f1>0 and f3<0.                                 c
329  #   cccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
330  
331  
332  def prodd(t, i, j, k):
333      res = 1.0
334      for s in range(k+2):
335          if i + s != j:
336              res *= (t[j] - t[i+s])
337      return res

340  def disc(t, k):
341      """Discontinuity matrix: jumps of k-th derivatives of b-splines at internal knots. 
342   
343      See Eqs. (9)-(10) of Ref. [1], or, equivalently, Eq. (3.43) of Ref. [2]. 
344   
345      This routine assumes internal knots are all simple (have multiplicity =1). 
346   
347      Parameters 
348      ---------- 
349      t : ndarray, 1D, shape(n,) 
350          Knots. 
351      k : int 
352          The spline degree 
353   
354      Returns 
355      ------- 
356      disc : ndarray, shape(n-2*k-1, k+2) 
357          The jumps of the k-th derivatives of b-splines at internal knots, 
358          ``t[k+1], ...., t[n-k-1]``. 
359      offset : ndarray, shape(2-2*k-1,) 
360          Offsets 
361      nc : int 
362   
363      Notes 
364      ----- 
365   
366      The normalization here follows FITPACK: 
367      (https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpdisc.f#L36) 
368   
369      The k-th derivative jumps are multiplied by a factor:: 
370   
371          (delta / nrint)**k / k! 
372   
373      where ``delta`` is the length of the interval spanned by internal knots, and 
374      ``nrint`` is one less the number of internal knots (i.e., the number of 
375      subintervals between them). 
376   
377      References 
378      ---------- 
379      .. [1] Paul Dierckx, Algorithms for smoothing data with periodic and parametric 
380             splines, Computer Graphics and Image Processing, vol. 20, p. 171 (1982). 
381             :doi:`10.1016/0146-664X(82)90043-0` 
382   
383      .. [2] Tom Lyche and Knut Morken, Spline methods, 
384          http://www.uio.no/studier/emner/matnat/ifi/INF-MAT5340/v05/undervisningsmateriale/ 
385   
386      """
387      n = t.shape[0]
388  
389      # the length of the base interval spanned by internal knots & the number
390      # of subintervas between these internal knots
391      delta = t[n - k - 1] - t[k]
392      nrint = n - 2*k - 1
393  
394      matr = np.empty((nrint - 1, k + 2), dtype=float)
395      for jj in range(nrint - 1):
396          j = jj + k + 1
397          for ii in range(k + 2):
398              i = jj + ii
399              matr[jj, ii] = (t[i + k + 1] - t[i]) / prodd(t, i, j, k)
400          # NB: equivalent to
401          # row = [(t[i + k + 1] - t[i]) / prodd(t, i, j, k) for i in range(j-k-1, j+1)]
402          # assert (matr[j-k-1, :] == row).all()
403  
404      # follow FITPACK
405      matr *= (delta/ nrint)**k
406  
407      # make it packed
408      offset = np.array([i for i in range(nrint-1)], dtype=np.int64)
409      nc = n - k - 1
410      return matr, offset, nc

413  class F:
414      """ The r.h.s. of ``f(p) = s``. 
415   
416      Given scalar `p`, we solve the system of equations in the LSQ sense: 
417   
418          | A     |  @ | c | = | y | 
419          | B / p |    | 0 |   | 0 | 
420   
421      where `A` is the matrix of b-splines and `b` is the discontinuity matrix 
422      (the jumps of the k-th derivatives of b-spline basis elements at knots). 
423   
424      Since we do that repeatedly while minimizing over `p`, we QR-factorize 
425      `A` only once and update the QR factorization only of the `B` rows of the 
426      augmented matrix |A, B/p|. 
427   
428      The system of equations is Eq. (15) Ref. [1]_, the strategy and implementation 
429      follows that of FITPACK, see specific links below. 
430   
431      References 
432      ---------- 
433      [1] P. Dierckx, Algorithms for Smoothing Data with Periodic and Parametric Splines, 
434          COMPUTER GRAPHICS AND IMAGE PROCESSING vol. 20, pp 171-184 (1982.) 
435          https://doi.org/10.1016/0146-664X(82)90043-0 
436   
437      """
438      def __init__(self, x, y, t, k, s, w=None, *, R=None, Y=None):
439          self.x = x
440          self.y = y
441          self.t = t
442          self.k = k
443          w = np.ones_like(x, dtype=float) if w is None else w
444          if w.ndim != 1:
445              raise ValueError(f"{w.ndim = } != 1.")
446          self.w = w
447          self.s = s
448  
449          if y.ndim != 2:
450              raise ValueError(f"F: expected y.ndim == 2, got {y.ndim = } instead.")
451  
452          # ### precompute what we can ###
453  
454          # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L250
455          # c  evaluate the discontinuity jump of the kth derivative of the
456          # c  b-splines at the knots t(l),l=k+2,...n-k-1 and store in b.
457          b, b_offset, b_nc = disc(t, k)
458  
459          # the QR factorization of the data matrix, if not provided
460          # NB: otherwise, must be consistent with x,y & s, but this is not checked
461          if R is None and Y is None:
462              R, Y, _ = _lsq_solve_qr(x, y, t, k, w)
463  
464          # prepare to combine R and the discontinuity matrix (AB); also r.h.s. (YY)
465          # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L269
466          # c  the rows of matrix b with weight 1/p are rotated into the
467          # c  triangularised observation matrix a which is stored in g.
468          nc = t.shape[0] - k - 1
469          nz = k + 1
470          if R.shape[1] != nz:
471              raise ValueError(f"Internal error: {R.shape[1] =} != {k+1 =}.")
472  
473          # r.h.s. of the augmented system
474          z = np.zeros((b.shape[0], Y.shape[1]), dtype=float)
475          self.YY = np.r_[Y[:nc], z]
476  
477          # l.h.s. of the augmented system
478          AA = np.zeros((nc + b.shape[0], self.k+2), dtype=float)
479          AA[:nc, :nz] = R[:nc, :]
480          # AA[nc:, :] = b.a / p  # done in __call__(self, p)
481          self.AA  = AA
482          self.offset = np.r_[np.arange(nc, dtype=np.int64), b_offset]
483  
484          self.nc = nc
485          self.b = b
486  
487      def __call__(self, p):
488          # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L279
489          # c  the row of matrix b is rotated into triangle by givens transformation
490  
491          # copy the precomputed matrices over for in-place work
492          # R = PackedMatrix(self.AB.a.copy(), self.AB.offset.copy(), nc)
493          AB = self.AA.copy()
494          offset = self.offset.copy()
495          nc = self.nc
496  
497          AB[nc:, :] = self.b / p
498          QY = self.YY.copy()
499  
500          # heavy lifting happens here, in-place
501          _dierckx.qr_reduce(AB, offset, nc, QY, startrow=nc)
502  
503          # solve for the coefficients
504          c = _dierckx.fpback(AB, nc, QY)
505  
506          spl = BSpline(self.t, c, self.k)
507          residuals = _compute_residuals(self.w**2, spl(self.x), self.y)
508          fp = residuals.sum()
509  
510          self.spl = spl   # store it
511  
512          return fp - self.s

515  def fprati(p1, f1, p2, f2, p3, f3):
516      """The root of r(p) = (u*p + v) / (p + w) given three points and values, 
517      (p1, f2), (p2, f2) and (p3, f3). 
518   
519      The FITPACK analog adjusts the bounds, and we do not 
520      https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fprati.f 
521   
522      NB: FITPACK uses p < 0 to encode p=infinity. We just use the infinity itself. 
523      Since the bracket is ``p1 <= p2 <= p3``, ``p3`` can be infinite (in fact, 
524      this is what the minimizer starts with, ``p3=inf``). 
525      """
526      h1 = f1 * (f2 - f3)
527      h2 = f2 * (f3 - f1)
528      h3 = f3 * (f1 - f2)
529      if p3 == np.inf:
530          return -(p2*h1 + p1*h2) / h3
531      return -(p1*p2*h3 + p2*p3*h1 + p1*p3*h2) / (p1*h1 + p2*h2 + p3*h3)

557  def root_rati(f, p0, bracket, acc):
558      """Solve `f(p) = 0` using a rational function approximation. 
559   
560      In a nutshell, since the function f(p) is known to be monotonically decreasing, we 
561         - maintain the bracket (p1, f1), (p2, f2) and (p3, f3) 
562         - at each iteration step, approximate f(p) by a rational function 
563           r(p) = (u*p + v) / (p + w) 
564           and make a step to p_new to the root of f(p): r(p_new) = 0. 
565           The coefficients u, v and w are found from the bracket values p1..3 and f1...3 
566   
567      The algorithm and implementation follows 
568      https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L229 
569      and 
570      https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fppara.f#L290 
571   
572      Note that the latter is for parametric splines and the former is for 1D spline 
573      functions. The minimization is indentical though [modulo a summation over the 
574      dimensions in the computation of f(p)], so we reuse the minimizer for both 
575      d=1 and d>1. 
576      """
577      # Magic values from
578      # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L27
579      con1 = 0.1
580      con9 = 0.9
581      con4 = 0.04
582  
583      # bracketing flags (follow FITPACK)
584      # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fppara.f#L365
585      ich1, ich3 = 0, 0
586  
587      (p1, f1), (p3, f3)  = bracket
588      p = p0
589  
590      for it in range(MAXIT):
591          p2, f2 = p, f(p)
592  
593          # c  test whether the approximation sp(x) is an acceptable solution.
594          if abs(f2) < acc:
595              ier, converged = 0, True
596              break
597  
598          # c  carry out one more step of the iteration process.
599          if ich3 == 0:
600              if f2 - f3 <= acc:
601                  # c  our initial choice of p is too large.
602                  p3 = p2
603                  f3 = f2
604                  p = p*con4
605                  if p <= p1:
606                       p = p1*con9 + p2*con1
607                  continue
608              else:
609                  if f2 < 0:
610                      ich3 = 1
611  
612          if ich1 == 0:
613              if f1 - f2 <= acc:
614                  # c  our initial choice of p is too small
615                  p1 = p2
616                  f1 = f2
617                  p = p/con4
618                  if p3 != np.inf and p <= p3:
619                       p = p2*con1 + p3*con9
620                  continue
621              else:
622                  if f2 > 0:
623                      ich1 = 1
624  
625          # c  test whether the iteration process proceeds as theoretically expected.
626          # [f(p) should be monotonically decreasing]
627          if f1 <= f2 or f2 <= f3:
628              ier, converged = 2, False
629              break
630  
631          # actually make the iteration step
632          p = fprati(p1, f1, p2, f2, p3, f3)
633  
634          # c  adjust the value of p1,f1,p3 and f3 such that f1 > 0 and f3 < 0.
635          if f2 < 0:
636              p3, f3 = p2, f2
637          else:
638              p1, f1 = p2, f2
639  
640      else:
641          # not converged in MAXIT iterations
642          ier, converged = 3, False
643  
644      if ier != 0:
645          warnings.warn(RuntimeWarning(_iermesg[ier]), stacklevel=2)
646  
647      return Bunch(converged=converged, root=p, iterations=it, ier=ier)

650  def _make_splrep_impl(x, y, *, w=None, xb=None, xe=None, k=3, s=0, t=None, nest=None):
651      """Shared infra for make_splrep and make_splprep. 
652      """
653      acc = s * TOL
654      m = x.size    # the number of data points
655  
656      if nest is None:
657          # the max number of knots. This is set in _fitpack_impl.py line 274
658          # and fitpack.pyf line 198
659          nest = max(m + k + 1, 2*k + 3)
660      else:
661          if nest < 2*(k + 1):
662              raise ValueError(f"`nest` too small: {nest = } < 2*(k+1) = {2*(k+1)}.")    
663          if t is not None:
664              raise ValueError("Either supply `t` or `nest`.")
665  
666      if t is None:
667          gen = _generate_knots_impl(x, y, w=w, k=k, s=s, xb=xb, xe=xe, nest=nest)
668          t = list(gen)[-1]
669      else:
670          fpcheck(x, t, k)
671  
672      if t.shape[0] == 2 * (k + 1):
673          # nothing to optimize
674          _, _, c = _lsq_solve_qr(x, y, t, k, w)
675          return BSpline(t, c, k)
676  
677      ### solve ###
678  
679      # c  initial value for p.
680      # https://github.com/scipy/scipy/blob/maintenance/1.11.x/scipy/interpolate/fitpack/fpcurf.f#L253
681      R, Y, _ = _lsq_solve_qr(x, y, t, k, w)
682      nc = t.shape[0] -k -1
683      p = nc / R[:, 0].sum()
684  
685      # ### bespoke solver ####
686      # initial conditions
687      # f(p=inf) : LSQ spline with knots t   (XXX: reuse R, c)
688      residuals = _get_residuals(x, y, t, k, w=w)
689      fp = residuals.sum()
690      fpinf = fp - s
691  
692      # f(p=0): LSQ spline without internal knots
693      residuals = _get_residuals(x, y, np.array([xb]*(k+1) + [xe]*(k+1)), k, w)
694      fp0 = residuals.sum()
695      fp0 = fp0 - s
696  
697      # solve
698      bracket = (0, fp0), (np.inf, fpinf)
699      f = F(x, y, t, k=k, s=s, w=w, R=R, Y=Y)
700      _ = root_rati(f, p, bracket, acc)
701  
702      # solve ALTERNATIVE: is roughly equivalent, gives slightly different results
703      # starting from scratch, that would have probably been tolerable;
704      # backwards compatibility dictates that we replicate the FITPACK minimizer though.
705   #   f = F(x, y, t, k=k, s=s, w=w, R=R, Y=Y)
706   #   from scipy.optimize import root_scalar
707   #   res_ = root_scalar(f, x0=p, rtol=acc)
708   #   assert res_.converged
709  
710      # f.spl is the spline corresponding to the found `p` value
711      return f.spl

714  def make_splrep(x, y, *, w=None, xb=None, xe=None, k=3, s=0, t=None, nest=None):
715      r"""Find the B-spline representation of a 1D function. 
716   
717      Given the set of data points ``(x[i], y[i])``, determine a smooth spline 
718      approximation of degree ``k`` on the interval ``xb <= x <= xe``. 
719   
720      Parameters 
721      ---------- 
722      x, y : array_like, shape (m,) 
723          The data points defining a curve ``y = f(x)``. 
724      w : array_like, shape (m,), optional 
725          Strictly positive 1D array of weights, of the same length as `x` and `y`. 
726          The weights are used in computing the weighted least-squares spline 
727          fit. If the errors in the y values have standard-deviation given by the 
728          vector ``d``, then `w` should be ``1/d``. 
729          Default is ``np.ones(m)``. 
730      xb, xe : float, optional 
731          The interval to fit.  If None, these default to ``x[0]`` and ``x[-1]``, 
732          respectively. 
733      k : int, optional 
734          The degree of the spline fit. It is recommended to use cubic splines, 
735          ``k=3``, which is the default. Even values of `k` should be avoided, 
736          especially with small `s` values. 
737      s : float, optional 
738          The smoothing condition. The amount of smoothness is determined by 
739          satisfying the conditions:: 
740   
741              sum((w * (g(x)  - y))**2 ) <= s 
742   
743          where ``g(x)`` is the smoothed fit to ``(x, y)``. The user can use `s` 
744          to control the tradeoff between closeness to data and smoothness of fit. 
745          Larger `s` means more smoothing while smaller values of `s` indicate less 
746          smoothing. 
747          Recommended values of `s` depend on the weights, `w`. If the weights 
748          represent the inverse of the standard deviation of `y`, then a good `s` 
749          value should be found in the range ``(m-sqrt(2*m), m+sqrt(2*m))`` where 
750          ``m`` is the number of datapoints in `x`, `y`, and `w`. 
751          Default is ``s = 0.0``, i.e. interpolation. 
752      t : array_like, optional 
753          The spline knots. If None (default), the knots will be constructed 
754          automatically. 
755          There must be at least ``2*k + 2`` and at most ``m + k + 1`` knots. 
756      nest : int, optional 
757          The target length of the knot vector. Should be between ``2*(k + 1)`` 
758          (the minimum number of knots for a degree-``k`` spline), and 
759          ``m + k + 1`` (the number of knots of the interpolating spline). 
760          The actual number of knots returned by this routine may be slightly 
761          larger than `nest`. 
762          Default is None (no limit, add up to ``m + k + 1`` knots). 
763   
764      Returns 
765      ------- 
766      spl : a `BSpline` instance 
767          For `s=0`,  ``spl(x) == y``. 
768          For non-zero values of `s` the `spl` represents the smoothed approximation 
769          to `(x, y)`, generally with fewer knots. 
770   
771      See Also 
772      -------- 
773      generate_knots : is used under the hood for generating the knots 
774      make_splprep : the analog of this routine for parametric curves 
775      make_interp_spline : construct an interpolating spline (``s = 0``) 
776      make_lsq_spline : construct the least-squares spline given the knot vector 
777      splrep : a FITPACK analog of this routine 
778   
779      References 
780      ---------- 
781      .. [1] P. Dierckx, "Algorithms for smoothing data with periodic and 
782          parametric splines, Computer Graphics and Image Processing", 
783          20 (1982) 171-184. 
784      .. [2] P. Dierckx, "Curve and surface fitting with splines", Monographs on 
785          Numerical Analysis, Oxford University Press, 1993. 
786   
787      Notes 
788      ----- 
789      This routine constructs the smoothing spline function, :math:`g(x)`, to 
790      minimize the sum of jumps, :math:`D_j`, of the ``k``-th derivative at the 
791      internal knots (:math:`x_b < t_i < x_e`), where 
792   
793      .. math:: 
794   
795          D_i = g^{(k)}(t_i + 0) - g^{(k)}(t_i - 0) 
796   
797      Specifically, the routine constructs the spline function :math:`g(x)` which 
798      minimizes 
799   
800      .. math:: 
801   
802              \sum_i | D_i |^2 \to \mathrm{min} 
803   
804      provided that 
805   
806      .. math:: 
807   
808             \sum_{j=1}^m (w_j \times (g(x_j) - y_j))^2 \leqslant s , 
809   
810      where :math:`s > 0` is the input parameter. 
811   
812      In other words, we balance maximizing the smoothness (measured as the jumps 
813      of the derivative, the first criterion), and the deviation of :math:`g(x_j)` 
814      from the data :math:`y_j` (the second criterion). 
815   
816      Note that the summation in the second criterion is over all data points, 
817      and in the first criterion it is over the internal spline knots (i.e. 
818      those with ``xb < t[i] < xe``). The spline knots are in general a subset 
819      of data, see `generate_knots` for details. 
820   
821      Also note the difference of this routine to `make_lsq_spline`: the latter 
822      routine does not consider smoothness and simply solves a least-squares 
823      problem 
824   
825      .. math:: 
826   
827          \sum w_j \times (g(x_j) - y_j)^2 \to \mathrm{min} 
828   
829      for a spline function :math:`g(x)` with a _fixed_ knot vector ``t``. 
830   
831      .. versionadded:: 1.15.0 
832      """
833      if s == 0:
834          if t is not None or w is not None or nest is not None:
835              raise ValueError("s==0 is for interpolation only")
836          return make_interp_spline(x, y, k=k)
837  
838      x, y, w, k, s, xb, xe = _validate_inputs(x, y, w, k, s, xb, xe, parametric=False)
839  
840      spl = _make_splrep_impl(x, y, w=w, xb=xb, xe=xe, k=k, s=s, t=t, nest=nest)
841  
842      # postprocess: squeeze out the last dimension: was added to simplify the internals.
843      spl.c = spl.c[:, 0]
844      return spl

847  def make_splprep(x, *, w=None, u=None, ub=None, ue=None, k=3, s=0, t=None, nest=None):
848      r""" 
849      Find a smoothed B-spline representation of a parametric N-D curve. 
850   
851      Given a list of N 1D arrays, `x`, which represent a curve in 
852      N-dimensional space parametrized by `u`, find a smooth approximating 
853      spline curve ``g(u)``. 
854   
855      Parameters 
856      ---------- 
857      x : array_like, shape (m, ndim) 
858          Sampled data points representing the curve in ``ndim`` dimensions. 
859          The typical use is a list of 1D arrays, each of length ``m``. 
860      w : array_like, shape(m,), optional 
861          Strictly positive 1D array of weights. 
862          The weights are used in computing the weighted least-squares spline 
863          fit. If the errors in the `x` values have standard deviation given by 
864          the vector d, then `w` should be 1/d. Default is ``np.ones(m)``. 
865      u : array_like, optional 
866          An array of parameter values for the curve in the parametric form. 
867          If not given, these values are calculated automatically, according to:: 
868   
869              v[0] = 0 
870              v[i] = v[i-1] + distance(x[i], x[i-1]) 
871              u[i] = v[i] / v[-1] 
872   
873      ub, ue : float, optional 
874          The end-points of the parameters interval. Default to ``u[0]`` and ``u[-1]``. 
875      k : int, optional 
876          Degree of the spline. Cubic splines, ``k=3``, are recommended. 
877          Even values of `k` should be avoided especially with a small ``s`` value. 
878          Default is ``k=3`` 
879      s : float, optional 
880          A smoothing condition.  The amount of smoothness is determined by 
881          satisfying the conditions:: 
882   
883              sum((w * (g(u) - x))**2) <= s, 
884   
885          where ``g(u)`` is the smoothed approximation to ``x``.  The user can 
886          use `s` to control the trade-off between closeness and smoothness 
887          of fit.  Larger ``s`` means more smoothing while smaller values of ``s`` 
888          indicate less smoothing. 
889          Recommended values of ``s`` depend on the weights, ``w``.  If the weights 
890          represent the inverse of the standard deviation of ``x``, then a good 
891          ``s`` value should be found in the range ``(m - sqrt(2*m), m + sqrt(2*m))``, 
892          where ``m`` is the number of data points in ``x`` and ``w``. 
893      t : array_like, optional 
894          The spline knots. If None (default), the knots will be constructed 
895          automatically. 
896          There must be at least ``2*k + 2`` and at most ``m + k + 1`` knots. 
897      nest : int, optional 
898          The target length of the knot vector. Should be between ``2*(k + 1)`` 
899          (the minimum number of knots for a degree-``k`` spline), and 
900          ``m + k + 1`` (the number of knots of the interpolating spline). 
901          The actual number of knots returned by this routine may be slightly 
902          larger than `nest`. 
903          Default is None (no limit, add up to ``m + k + 1`` knots). 
904   
905      Returns 
906      ------- 
907      spl : a `BSpline` instance 
908          For `s=0`,  ``spl(u) == x``. 
909          For non-zero values of ``s``, `spl` represents the smoothed approximation 
910          to ``x``, generally with fewer knots. 
911      u : ndarray 
912          The values of the parameters 
913   
914      See Also 
915      -------- 
916      generate_knots : is used under the hood for generating the knots 
917      make_splrep : the analog of this routine 1D functions 
918      make_interp_spline : construct an interpolating spline (``s = 0``) 
919      make_lsq_spline : construct the least-squares spline given the knot vector 
920      splprep : a FITPACK analog of this routine 
921   
922      Notes 
923      ----- 
924      Given a set of :math:`m` data points in :math:`D` dimensions, :math:`\vec{x}_j`, 
925      with :math:`j=1, ..., m` and :math:`\vec{x}_j = (x_{j; 1}, ..., x_{j; D})`, 
926      this routine constructs the parametric spline curve :math:`g_a(u)` with 
927      :math:`a=1, ..., D`, to minimize the sum of jumps, :math:`D_{i; a}`, of the 
928      ``k``-th derivative at the internal knots (:math:`u_b < t_i < u_e`), where 
929   
930      .. math:: 
931   
932          D_{i; a} = g_a^{(k)}(t_i + 0) - g_a^{(k)}(t_i - 0) 
933   
934      Specifically, the routine constructs the spline function :math:`g(u)` which 
935      minimizes 
936   
937      .. math:: 
938   
939              \sum_i \sum_{a=1}^D | D_{i; a} |^2 \to \mathrm{min} 
940   
941      provided that 
942   
943      .. math:: 
944   
945          \sum_{j=1}^m \sum_{a=1}^D (w_j \times (g_a(u_j) - x_{j; a}))^2 \leqslant s 
946   
947      where :math:`u_j` is the value of the parameter corresponding to the data point 
948      :math:`(x_{j; 1}, ..., x_{j; D})`, and :math:`s > 0` is the input parameter. 
949   
950      In other words, we balance maximizing the smoothness (measured as the jumps 
951      of the derivative, the first criterion), and the deviation of :math:`g(u_j)` 
952      from the data :math:`x_j` (the second criterion). 
953   
954      Note that the summation in the second criterion is over all data points, 
955      and in the first criterion it is over the internal spline knots (i.e. 
956      those with ``ub < t[i] < ue``). The spline knots are in general a subset 
957      of data, see `generate_knots` for details. 
958   
959      .. versionadded:: 1.15.0 
960   
961      References 
962      ---------- 
963      .. [1] P. Dierckx, "Algorithms for smoothing data with periodic and 
964          parametric splines, Computer Graphics and Image Processing", 
965          20 (1982) 171-184. 
966      .. [2] P. Dierckx, "Curve and surface fitting with splines", Monographs on 
967          Numerical Analysis, Oxford University Press, 1993. 
968      """
969      x = np.stack(x, axis=1)
970  
971      # construct the default parametrization of the curve
972      if u is None:
973          dp = (x[1:, :] - x[:-1, :])**2
974          u = np.sqrt((dp).sum(axis=1)).cumsum()
975          u = np.r_[0, u / u[-1]]
976  
977      if s == 0:
978          if t is not None or w is not None or nest is not None:
979              raise ValueError("s==0 is for interpolation only")
980          return make_interp_spline(u, x.T, k=k, axis=1), u
981  
982      u, x, w, k, s, ub, ue = _validate_inputs(u, x, w, k, s, ub, ue, parametric=True)
983  
984      spl = _make_splrep_impl(u, x, w=w, xb=ub, xe=ue, k=k, s=s, t=t, nest=nest)
985  
986      # posprocess: `axis=1` so that spl(u).shape == np.shape(x)
987      # when `x` is a list of 1D arrays (cf original splPrep)
988      cc = spl.c.T
989      spl1 = BSpline(spl.t, cc, spl.k, axis=1)
990  
991      return spl1, u
