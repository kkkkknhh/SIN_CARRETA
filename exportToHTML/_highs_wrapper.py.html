9    def _highs_wrapper(c, indptr, indices, data, lhs, rhs, lb, ub, integrality, options):
10       '''Solve linear programs using HiGHS [1]_. 
11    
12       Assume problems of the form: 
13    
14           MIN c.T @ x 
15           s.t. lhs <= A @ x <= rhs 
16                lb <= x <= ub 
17    
18       Parameters 
19       ---------- 
20       c : 1-D array, (n,) 
21           Array of objective value coefficients. 
22       astart : 1-D array 
23           CSC format index array. 
24       aindex : 1-D array 
25           CSC format index array. 
26       avalue : 1-D array 
27           Data array of the matrix. 
28       lhs : 1-D array (or None), (m,) 
29           Array of left hand side values of the inequality constraints. 
30           If ``lhs=None``, then an array of ``-inf`` is assumed. 
31       rhs : 1-D array, (m,) 
32           Array of right hand side values of the inequality constraints. 
33       lb : 1-D array (or None), (n,) 
34           Lower bounds on solution variables x.  If ``lb=None``, then an 
35           array of all `0` is assumed. 
36       ub : 1-D array (or None), (n,) 
37           Upper bounds on solution variables x.  If ``ub=None``, then an 
38           array of ``inf`` is assumed. 
39       options : dict 
40           A dictionary of solver options 
41    
42       Returns 
43       ------- 
44       res : dict 
45    
46           If model_status is one of kOptimal, 
47           kObjectiveBound, kTimeLimit, 
48           kIterationLimit: 
49    
50               - ``status`` : HighsModelStatus 
51                   Model status code. 
52    
53               - ``message`` : str 
54                   Message corresponding to model status code. 
55    
56               - ``x`` : list 
57                   Solution variables. 
58    
59               - ``slack`` : list 
60                   Slack variables. 
61    
62               - ``lambda`` : list 
63                   Lagrange multipliers associated with the constraints 
64                   Ax = b. 
65    
66               - ``s`` : list 
67                   Lagrange multipliers associated with the constraints 
68                   x >= 0. 
69    
70               - ``fun`` 
71                   Final objective value. 
72    
73               - ``simplex_nit`` : int 
74                   Number of iterations accomplished by the simplex 
75                   solver. 
76    
77               - ``ipm_nit`` : int 
78                   Number of iterations accomplished by the interior- 
79                   point solver. 
80    
81           If model_status is not one of the above: 
82    
83               - ``status`` : HighsModelStatus 
84                   Model status code. 
85    
86               - ``message`` : str 
87                   Message corresponding to model status code. 
88    
89       Notes 
90       ----- 
91       If ``options['write_solution_to_file']`` is ``True`` but 
92       ``options['solution_file']`` is unset or ``''``, then the solution 
93       will be printed to ``stdout``. 
94    
95       If any iteration limit is reached, no solution will be 
96       available. 
97    
98       ``OptimizeWarning`` will be raised if any option value set by 
99       the user is found to be incorrect. 
100   
101      References 
102      ---------- 
103      .. [1] https://highs.dev/ 
104      .. [2] https://www.maths.ed.ac.uk/hall/HiGHS/HighsOptions.html 
105      '''
106      numcol = c.size
107      numrow = rhs.size
108      isMip = integrality is not None and np.sum(integrality) > 0
109  
110      # default "null" return values
111      res = {
112          "x": None,
113          "fun": None,
114      }
115  
116      # Fill up a HighsLp object
117      lp = _h.HighsLp()
118      lp.num_col_ = numcol
119      lp.num_row_ = numrow
120      lp.a_matrix_.num_col_ = numcol
121      lp.a_matrix_.num_row_ = numrow
122      lp.a_matrix_.format_ = _h.MatrixFormat.kColwise
123      lp.col_cost_ = c
124      lp.col_lower_ = lb
125      lp.col_upper_ = ub
126      lp.row_lower_ = lhs
127      lp.row_upper_ = rhs
128      lp.a_matrix_.start_ = indptr
129      lp.a_matrix_.index_ = indices
130      lp.a_matrix_.value_ = data
131      if integrality.size > 0:
132          lp.integrality_ = [_h.HighsVarType(i) for i in integrality]
133  
134      # Make a Highs object and pass it everything
135      highs = _h._Highs()
136      highs_options = _h.HighsOptions()
137      hoptmanager = hopt.HighsOptionsManager()
138      for key, val in options.items():
139          # handle filtering of unsupported and default options
140          if val is None or key in ("sense",):
141              continue
142  
143          # ask for the option type
144          opt_type = hoptmanager.get_option_type(key)
145          if -1 == opt_type:
146              warn(
147                  f"Unrecognized options detected: {dict({key: val})}",
148                  OptimizeWarning,
149                  stacklevel=2,
150              )
151              continue
152          else:
153              if key in ("presolve", "parallel"):
154                  # handle fake bools (require bool -> str conversions)
155                  if isinstance(val, bool):
156                      val = "on" if val else "off"
157                  else:
158                      warn(
159                          f'Option f"{key}" is "{val}", but only True or False is '
160                          f"allowed. Using default.",
161                          OptimizeWarning,
162                          stacklevel=2,
163                      )
164                      continue
165              opt_type = _h.HighsOptionType(opt_type)
166              status, msg = check_option(highs, key, val)
167              if opt_type == _h.HighsOptionType.kBool:
168                  if not isinstance(val, bool):
169                      warn(
170                          f'Option f"{key}" is "{val}", but only True or False is '
171                          f"allowed. Using default.",
172                          OptimizeWarning,
173                          stacklevel=2,
174                      )
175                      continue
176  
177              # warn or set option
178              if status != 0:
179                  warn(msg, OptimizeWarning, stacklevel=2)
180              else:
181                  setattr(highs_options, key, val)
182  
183      opt_status = highs.passOptions(highs_options)
184      if opt_status == _h.HighsStatus.kError:
185          res.update(
186              {
187                  "status": highs.getModelStatus(),
188                  "message": highs.modelStatusToString(highs.getModelStatus()),
189              }
190          )
191          return res
192  
193      init_status = highs.passModel(lp)
194      if init_status == _h.HighsStatus.kError:
195          # if model fails to load, highs.getModelStatus() will be NOT_SET
196          err_model_status = _h.HighsModelStatus.kModelError
197          res.update(
198              {
199                  "status": err_model_status,
200                  "message": highs.modelStatusToString(err_model_status),
201              }
202          )
203          return res
204  
205      # Solve the LP
206      run_status = highs.run()
207      if run_status == _h.HighsStatus.kError:
208          res.update(
209              {
210                  "status": highs.getModelStatus(),
211                  "message": highs.modelStatusToString(highs.getModelStatus()),
212              }
213          )
214          return res
215  
216      # Extract what we need from the solution
217      model_status = highs.getModelStatus()
218  
219      # it should always be safe to get the info object
220      info = highs.getInfo()
221  
222      # Failure modes:
223      #     LP: if we have anything other than an Optimal status, it
224      #         is unsafe (and unhelpful) to read any results
225      #    MIP: has a non-Optimal status or has timed out/reached max iterations
226      #             1) If not Optimal/TimedOut/MaxIter status, there is no solution
227      #             2) If TimedOut/MaxIter status, there may be a feasible solution.
228      #                if the objective function value is not Infinity, then the
229      #                current solution is feasible and can be returned.  Else, there
230      #                is no solution.
231      mipFailCondition = model_status not in (
232          _h.HighsModelStatus.kOptimal,
233          _h.HighsModelStatus.kTimeLimit,
234          _h.HighsModelStatus.kIterationLimit,
235          _h.HighsModelStatus.kSolutionLimit,
236      ) or (
237          model_status
238          in {
239              _h.HighsModelStatus.kTimeLimit,
240              _h.HighsModelStatus.kIterationLimit,
241              _h.HighsModelStatus.kSolutionLimit,
242          }
243          and (info.objective_function_value == _h.kHighsInf)
244      )
245      lpFailCondition = model_status != _h.HighsModelStatus.kOptimal
246      if (isMip and mipFailCondition) or (not isMip and lpFailCondition):
247          res.update(
248              {
249                  "status": model_status,
250                  "message": "model_status is "
251                  f"{highs.modelStatusToString(model_status)}; "
252                  "primal_status is "
253                  f"{highs.solutionStatusToString(info.primal_solution_status)}",
254                  "simplex_nit": info.simplex_iteration_count,
255                  "ipm_nit": info.ipm_iteration_count,
256                  "crossover_nit": info.crossover_iteration_count,
257              }
258          )
259          return res
260  
261      # Should be safe to read the solution:
262      solution = highs.getSolution()
263      basis = highs.getBasis()
264  
265      # Lagrangians for bounds based on column statuses
266      marg_bnds = np.zeros((2, numcol))
267      basis_col_status = basis.col_status
268      solution_col_dual = solution.col_dual
269      for ii in range(numcol):
270          if basis_col_status[ii] == _h.HighsBasisStatus.kLower:
271              marg_bnds[0, ii] = solution_col_dual[ii]
272          elif basis_col_status[ii] == _h.HighsBasisStatus.kUpper:
273              marg_bnds[1, ii] = solution_col_dual[ii]
274  
275      res.update(
276          {
277              "status": model_status,
278              "message": highs.modelStatusToString(model_status),
279              # Primal solution
280              "x": np.array(solution.col_value),
281              # Ax + s = b => Ax = b - s
282              # Note: this is for all constraints (A_ub and A_eq)
283              "slack": rhs - solution.row_value,
284              # lambda are the lagrange multipliers associated with Ax=b
285              "lambda": np.array(solution.row_dual),
286              "marg_bnds": marg_bnds,
287              "fun": info.objective_function_value,
288              "simplex_nit": info.simplex_iteration_count,
289              "ipm_nit": info.ipm_iteration_count,
290              "crossover_nit": info.crossover_iteration_count,
291          }
292      )
293  
294      if isMip:
295          res.update(
296              {
297                  "mip_node_count": info.mip_node_count,
298                  "mip_dual_bound": info.mip_dual_bound,
299                  "mip_gap": info.mip_gap,
300              }
301          )
302  
303      return res

306  def check_option(highs_inst, option, value):
307      status, option_type = highs_inst.getOptionType(option)
308      hoptmanager = hopt.HighsOptionsManager()
309  
310      if status != _h.HighsStatus.kOk:
311          return -1, "Invalid option name."
312  
313      valid_types = {
314          _h.HighsOptionType.kBool: bool,
315          _h.HighsOptionType.kInt: int,
316          _h.HighsOptionType.kDouble: float,
317          _h.HighsOptionType.kString: str,
318      }
319  
320      expected_type = valid_types.get(option_type, None)
321  
322      if expected_type is str:
323          if not hoptmanager.check_string_option(option, value):
324              return -1, "Invalid option value."
325      if expected_type is float:
326          if not hoptmanager.check_double_option(option, value):
327              return -1, "Invalid option value."
328      if expected_type is int:
329          if not hoptmanager.check_int_option(option, value):
330              return -1, "Invalid option value."
331  
332      if expected_type is None:
333          return 3, "Unknown option type."
334  
335      status, current_value = highs_inst.getOptionValue(option)
336      if status != _h.HighsStatus.kOk:
337          return 4, "Failed to validate option value."
338      return 0, "Check option succeeded."
