1    """ Test reading of files not conforming to matlab specification 
2     
3    We try and read any file that matlab reads, these files included

5    from os.path import dirname, join as pjoin
6    
7    from numpy.testing import assert_
8    from pytest import raises as assert_raises
9    
10   from scipy.io.matlab._mio import loadmat
11   
12   TEST_DATA_PATH = pjoin(dirname(__file__), 'data')
13   
14   
15   def test_multiple_fieldnames():
16       # Example provided by Dharhas Pothina
17       # Extracted using mio5.varmats_from_mat
18       multi_fname = pjoin(TEST_DATA_PATH, 'nasty_duplicate_fieldnames.mat')
19       vars = loadmat(multi_fname)
20       funny_names = vars['Summary'].dtype.names
21       assert_({'_1_Station_Q', '_2_Station_Q',
22                        '_3_Station_Q'}.issubset(funny_names))

25   def test_malformed1():
26       # Example from gh-6072
27       # Contains malformed header data, which previously resulted into a
28       # buffer overflow.
29       #
30       # Should raise an exception, not segfault
31       fname = pjoin(TEST_DATA_PATH, 'malformed1.mat')
32       with open(fname, 'rb') as f:
33           assert_raises(ValueError, loadmat, f)
