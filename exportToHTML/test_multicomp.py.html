233      def test_ttest_ind(self, alternative):
234          # check that `dunnett` agrees with `ttest_ind`
235          # when there are only two groups
236          rng = np.random.default_rng(114184017807316971636137493526995620351)
237  
238          for _ in range(10):
239              sample = rng.integers(-100, 100, size=(10,))
240              control = rng.integers(-100, 100, size=(10,))
241  
242              # preserve use of old random_state during SPEC 7 transition
243              res = stats.dunnett(
244                  sample, control=control,

247              ref = stats.ttest_ind(
248                  sample, control,
249                  alternative=alternative
250              )
251  
252              assert_allclose(res.statistic, ref.statistic, rtol=1e-3, atol=1e-5)

255      @pytest.mark.parametrize(
256          'alternative, pvalue',
257          [
258              ('less', [0, 1]),
259              ('greater', [1, 0]),
260              ('two-sided', [0, 0]),
261          ]
262      )
263      def test_alternatives(self, alternative, pvalue):
264          rng = np.random.default_rng(114184017807316971636137493526995620351)
265  
266          # width of 20 and min diff between samples/control is 60
267          # and maximal diff would be 100
268          sample_less = rng.integers(0, 20, size=(10,))
269          control = rng.integers(80, 100, size=(10,))
270          sample_greater = rng.integers(160, 180, size=(10,))
271  
272          res = stats.dunnett(
273              sample_less, sample_greater, control=control,
274              alternative=alternative, rng=rng
275          )
276          assert_allclose(res.pvalue, pvalue, atol=1e-7)
277  
278          ci = res.confidence_interval()
279          # two-sided is comparable for high/low
280          if alternative == 'less':
281              assert np.isneginf(ci.low).all()
282              assert -100 < ci.high[0] < -60
283              assert 60 < ci.high[1] < 100
284          elif alternative == 'greater':
285              assert -100 < ci.low[0] < -60
286              assert 60 < ci.low[1] < 100
287              assert np.isposinf(ci.high).all()
288          elif alternative == 'two-sided':
289              assert -100 < ci.low[0] < -60
290              assert 60 < ci.low[1] < 100
291              assert -100 < ci.high[0] < -60
292              assert 60 < ci.high[1] < 100

294      @pytest.mark.parametrize("case", [case_1, case_2, case_3, case_4])
295      @pytest.mark.parametrize("alternative", ['less', 'greater', 'two-sided'])
296      def test_against_R_multicomp_glht(self, case, alternative):
297          rng = np.random.default_rng(189117774084579816190295271136455278291)
298          samples = case['samples']
299          control = case['control']
300          alternatives = {'less': 'less', 'greater': 'greater',
301                          'two-sided': 'twosided'}
302          p_ref = case['pvalues'][alternative.replace('-', '')]
303  
304          res = stats.dunnett(*samples, control=control, alternative=alternative,
305                              rng=rng)
306          # atol can't be tighter because R reports some pvalues as "< 1e-4"
307          assert_allclose(res.pvalue, p_ref, rtol=5e-3, atol=1e-4)
308  
309          ci_ref = case['cis'][alternatives[alternative]]
310          if alternative == "greater":
311              ci_ref = [ci_ref, np.inf]
312          elif alternative == "less":
313              ci_ref = [-np.inf, ci_ref]
314          assert res._ci is None
315          assert res._ci_cl is None
316          ci = res.confidence_interval(confidence_level=0.95)
317          assert_allclose(ci.low, ci_ref[0], rtol=5e-3, atol=1e-5)
318          assert_allclose(ci.high, ci_ref[1], rtol=5e-3, atol=1e-5)
319  
320          # re-run to use the cached value "is" to check id as same object
321          assert res._ci is ci
322          assert res._ci_cl == 0.95
323          ci_ = res.confidence_interval(confidence_level=0.95)
324          assert ci_ is ci

371              stats.dunnett(*samples_, control=control)
