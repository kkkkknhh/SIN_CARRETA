44       """ 
45       Context manager that sets whether checkpoint should print additional debug 
46       information when running. See the ``debug`` flag for 
47       :func:`~torch.utils.checkpoint.checkpoint` for more information. Note that 
48       when set, this context manager overrides the value of ``debug`` passed to 
49       checkpoint. To defer to the local setting, pass ``None`` to this context.

51       Args:

58           _checkpoint_debug_enabled = enabled
59           yield
60       finally:
61           _checkpoint_debug_enabled = prev

64   def detach_variable(inputs: Tuple[Any, ...]) -> Tuple[torch.Tensor, ...]:
65       if isinstance(inputs, tuple):
66           out = []
67           for inp in inputs:
68               if not isinstance(inp, torch.Tensor):
69                   out.append(inp)
70                   continue
71   
72               x = inp.detach()
73               x.requires_grad = inp.requires_grad
74               out.append(x)
75           return tuple(out)
76       else:
77           raise RuntimeError(
78               "Only tuple of tensors is supported. Got Unsupported input type: ",
79               type(inputs).__name__,
80           )

130  def _infer_device_type(*args):
131      device_types = []
132  
133      def add_device_types(arg):
134          nonlocal device_types
135          if isinstance(arg, torch.Tensor) and not arg.device.type == "cpu":
136              device_types.append(arg.device.type)
137      tree_map(add_device_types, args)
138  
139      device_types_set = set(device_types)
140      if len(device_types_set) > 1:
141          warnings.warn(
142              "Tensor arguments, excluding CPU tensors, are detected on at least two types of devices. "
143              "Device state will only be saved for devices of a single device type, and the remaining "
144              "devices will be ignored. Consequently, if any checkpointed functions involve randomness, "
145              "this may result in incorrect gradients. (Note that if CUDA devices are among the devices "
146              "detected, it will be prioritized; otherwise, the first device encountered will be selected.)"
147              f"\nDevice types: {sorted(device_types_set)} first device type: {device_types[0]}"
148          )
149      if len(device_types) == 0:
150          return DefaultDeviceType.get_device_type()
151      elif "cuda" in device_types_set:
152          return "cuda"
153      else:
154          return device_types[0]

157  # We can't know if the run_fn will internally move some args to different devices,
158  # which would require logic to preserve rng states for those devices as well.
159  # We could paranoically stash and restore ALL the rng states for all visible devices,
160  # but that seems very wasteful for most cases.  Compromise:  Stash the RNG state for
161  # the device of all Tensor args.
162  #
163  # To consider:  maybe get_device_states and set_device_states should reside in torch/random.py?
164  def get_device_states(*args) -> Tuple[List[int], List[torch.Tensor]]:
165      # This will not error out if "arg" is a CPU tensor or a non-tensor type because
166      # the conditionals short-circuit.
167      fwd_device_ids = []
168  
169      def add_device_ids(arg):
170          nonlocal fwd_device_ids
171          if isinstance(arg, torch.Tensor) and arg.device.type not in {"cpu", "meta"}:
172              fwd_device_ids.append(arg.get_device())
173      tree_map(add_device_ids, args)
174  
175      fwd_device_states = []
176      device_module = _get_device_module(_infer_device_type(*args))
177      for device_id in fwd_device_ids:
178          with device_module.device(device_id):
179              fwd_device_states.append(device_module.get_rng_state())
180  
181      return fwd_device_ids, fwd_device_states

184  def set_device_states(devices, states, *, device_type=None) -> None:
185      """Sets random number generator states for the specified devices. 
186   
187      Args: 
188          devices: Device ids to set states for. 
189          states: States to set. 
190          device_type: ``device_type`` of the devices to set states for. Default 
191              is the device returned by a call to ``DefaultDeviceType.get_device_type()``, 
192              which is ``cuda`` if not changed by calling ``DefaultDeviceType::set_device_type()``. 
193      """
194      if device_type is None:
195          device_type = DefaultDeviceType.get_device_type()
196      if device_type == "meta":
197          return
198      device_module = _get_device_module(device_type)
199      for device, state in zip(devices, states):
200          with device_module.device(device):
201              device_module.set_rng_state(state)

223  class CheckpointFunction(torch.autograd.Function):
224      @staticmethod
225      def forward(ctx, run_function, preserve_rng_state, *args):
226          check_backward_validity(args)
227          ctx.run_function = run_function
228          ctx.preserve_rng_state = preserve_rng_state
229          # Accommodates the (remote) possibility that autocast is enabled for cpu AND gpu.
230          ctx.device_type = _infer_device_type(*args)
231          ctx.device_autocast_kwargs, ctx.cpu_autocast_kwargs = _get_autocast_kwargs(
232              ctx.device_type
233          )
234          if preserve_rng_state:
235              ctx.fwd_cpu_state = torch.get_rng_state()
236              # Don't eagerly initialize the cuda context by accident.
237              # (If the user intends that the context is initialized later, within their
238              # run_function, we SHOULD actually stash the cuda state here.  Unfortunately,
239              # we have no way to anticipate this will happen before we run the function.)
240              ctx.had_device_in_fwd = False
241              device_module = _get_device_module(ctx.device_type)
242              if getattr(device_module, "_initialized", False):
243                  ctx.had_device_in_fwd = True
244                  ctx.fwd_devices, ctx.fwd_device_states = get_device_states(*args)
245  
246          # Save non-tensor inputs in ctx, keep a placeholder None for tensors
247          # to be filled out during the backward.
248          ctx.inputs = []
249          ctx.tensor_indices = []
250          tensor_inputs = []
251          for i, arg in enumerate(args):
252              if torch.is_tensor(arg):
253                  tensor_inputs.append(arg)
254                  ctx.tensor_indices.append(i)
255                  ctx.inputs.append(None)
256              else:
257                  ctx.inputs.append(arg)
258  
259          ctx.save_for_backward(*tensor_inputs)
260  
261          with torch.no_grad():
262              outputs = run_function(*args)
263          return outputs
264  
265      @staticmethod
266      def backward(ctx, *args):
267          if not torch.autograd._is_checkpoint_valid():
268              raise RuntimeError(
269                  "When use_reentrant=True, torch.utils.checkpoint is incompatible"
270                  " with .grad() or passing an `inputs` parameter to .backward()."
271                  " To resolve this error, you can either set use_reentrant=False,"
272                  " or call .backward() without passing the `inputs` argument."
273              )
274          # Copy the list to avoid modifying original list.
275          inputs = list(ctx.inputs)
276          tensor_indices = ctx.tensor_indices
277          tensors = ctx.saved_tensors
278  
279          # Fill in inputs with appropriate saved tensors.
280          for i, idx in enumerate(tensor_indices):
281              inputs[idx] = tensors[i]
282  
283          # Stash the surrounding rng state, and mimic the state that was
284          # present at this time during forward.  Restore the surrounding state
285          # when we're done.
286          rng_devices = []
287          if ctx.preserve_rng_state and ctx.had_device_in_fwd:
288              rng_devices = ctx.fwd_devices
289          with torch.random.fork_rng(
290              devices=rng_devices, enabled=ctx.preserve_rng_state, device_type=ctx.device_type
291          ):
292              if ctx.preserve_rng_state:
293                  torch.set_rng_state(ctx.fwd_cpu_state)
294                  if ctx.had_device_in_fwd:
295                      set_device_states(ctx.fwd_devices, ctx.fwd_device_states, device_type=ctx.device_type)
296              detached_inputs = detach_variable(tuple(inputs))
297  
298              device_autocast_ctx = torch.amp.autocast(
299                  device_type=ctx.device_type, **ctx.device_autocast_kwargs
300              ) if torch.amp.is_autocast_available(ctx.device_type) else contextlib.nullcontext()
301              with torch.enable_grad(), device_autocast_ctx, torch.amp.autocast("cpu", **ctx.cpu_autocast_kwargs):  # type: ignore[attr-defined]
302                  outputs = ctx.run_function(*detached_inputs)
303  
304          if isinstance(outputs, torch.Tensor):
305              outputs = (outputs,)
306  
307          # run backward() with only tensor that requires grad
308          outputs_with_grad = []
309          args_with_grad = []
310          for i in range(len(outputs)):
311              if torch.is_tensor(outputs[i]) and outputs[i].requires_grad:
312                  outputs_with_grad.append(outputs[i])
313                  args_with_grad.append(args[i])
314          if len(outputs_with_grad) == 0:
315              raise RuntimeError(
316                  "none of output has requires_grad=True,"
317                  " this checkpoint() is not necessary"
318              )
319          torch.autograd.backward(outputs_with_grad, args_with_grad)
320          grads = tuple(
321              inp.grad if isinstance(inp, torch.Tensor) else None
322              for inp in detached_inputs
323          )
324  
325          return (None, None) + grads

332  # TorchDynamo does not step inside utils.checkpoint function.  The flow
333  # looks likes this
334  #  1) TorchDynamo tries to wrap utils.checkpoint in a HigherOrderOp by
335  #     speculatively checking if the forward function is safe to trace.
336  #  2) If yes, then Dynamo-generated Fx graph has the wrapped higher
337  #     order op. As a result, TorchDynamo does not look inside utils.checkpoint.
338  #  3) If not, then TorchDynamo falls back to eager by performing a graph
339  #     break. And here, the following disable wrapper ensures that
340  #     TorchDynamo does not trigger again on the frames created by
341  #     utils.checkpoint innards.
342  @torch._disable_dynamo
343  def checkpoint(
344      function,
345      *args,
346      use_reentrant: Optional[bool] = None,
347      context_fn: Callable[[], Tuple[ContextManager, ContextManager]] = noop_context_fn,
348      determinism_check: str = _DEFAULT_DETERMINISM_MODE,
349      debug: bool = False,
350      **kwargs
351  ):
352      r"""Checkpoint a model or part of the model. 
353   
354      Activation checkpointing is a technique that trades compute for memory. 
355      Instead of keeping tensors needed for backward alive until they are used in 
356      gradient computation during backward, forward computation in checkpointed 
357      regions omits saving tensors for backward and recomputes them during the 
358      backward pass. Activation checkpointing can be applied to any part of a 
359      model. 
360   
361      There are currently two checkpointing implementations available, determined 
362      by the :attr:`use_reentrant` parameter. It is recommended that you use 
363      ``use_reentrant=False``. Please refer the note below for a discussion of 
364      their differences. 
365   
366      .. warning:: 
367   
368          If the :attr:`function` invocation during the backward pass differs 
369          from the forward pass, e.g., due to a global variable, the checkpointed 
370          version may not be equivalent, potentially causing an 
371          error being raised or leading to silently incorrect gradients. 
372   
373      .. warning:: 
374   
375          The ``use_reentrant`` parameter should be passed explicitly. In version 
376          2.4 we will raise an exception if ``use_reentrant`` is not passed. 
377          If you are using the ``use_reentrant=True`` variant, please refer to the 
378          note below for important considerations and potential limitations. 
379   
380      .. note:: 
381   
382          The reentrant variant of checkpoint (``use_reentrant=True``) and 
383          the non-reentrant variant of checkpoint (``use_reentrant=False``) 
384          differ in the following ways: 
385   
386          * Non-reentrant checkpoint stops recomputation as soon as all needed 
387            intermediate activations have been recomputed. This feature is enabled 
388            by default, but can be disabled with :func:`set_checkpoint_early_stop`. 
389            Reentrant checkpoint always recomputes :attr:`function` in its 
390            entirety during the backward pass. 
391   
392          * The reentrant variant does not record the autograd graph during the 
393            forward pass, as it runs with the forward pass under 
394            :func:`torch.no_grad`. The non-reentrant version does record the 
395            autograd graph, allowing one to perform backward on the graph within 
396            checkpointed regions. 
397   
398          * The reentrant checkpoint only supports the 
399            :func:`torch.autograd.backward` API for the backward pass without its 
400            `inputs` argument, while the non-reentrant version supports all ways 
401            of performing the backward pass. 
402   
403          * At least one input and output must have ``requires_grad=True`` for the 
404            reentrant variant. If this condition is unmet, the checkpointed part 
405            of the model will not have gradients. The non-reentrant version does 
406            not have this requirement. 
407   
408          * The reentrant version does not consider tensors in nested structures 
409            (e.g., custom objects, lists, dicts, etc) as participating in 
410            autograd, while the non-reentrant version does. 
411   
412          * The reentrant checkpoint does not support checkpointed regions with 
413            detached tensors from the computational graph, whereas the 
414            non-reentrant version does. For the reentrant variant, if the 
415            checkpointed segment contains tensors detached using ``detach()`` or 
416            with :func:`torch.no_grad`, the backward pass will raise an error. 
417            This is because ``checkpoint`` makes all the outputs require gradients 
418            and this causes issues when a tensor is defined to have no gradient in 
419            the model. To avoid this, detach the tensors outside of the 
420            ``checkpoint`` function. 
421   
422      Args: 
423          function: describes what to run in the forward pass of the model or 
424              part of the model. It should also know how to handle the inputs 
425              passed as the tuple. For example, in LSTM, if user passes 
426              ``(activation, hidden)``, :attr:`function` should correctly use the 
427              first input as ``activation`` and the second input as ``hidden`` 
428          preserve_rng_state(bool, optional):  Omit stashing and restoring 
429              the RNG state during each checkpoint. Note that under torch.compile, 
430              this flag doesn't take effect and we always preserve RNG state. 
431              Default: ``True`` 
432          use_reentrant(bool): 
433              specify whether to use the activation checkpoint variant that 
434              requires reentrant autograd. This parameter should be passed 
435              explicitly. In version 2.5 we will raise an exception if 
436              ``use_reentrant`` is not passed. If ``use_reentrant=False``, 
437              ``checkpoint`` will use an implementation that does not require 
438              reentrant autograd. This allows ``checkpoint`` to support additional 
439              functionality, such as working as expected with 
440              ``torch.autograd.grad`` and support for keyword arguments input into 
441              the checkpointed function. 
442          context_fn(Callable, optional): A callable returning a tuple of two 
443              context managers. The function and its recomputation will be run 
444              under the first and second context managers respectively. 
445              This argument is only supported if ``use_reentrant=False``. 
446          determinism_check(str, optional): A string specifying the determinism 
447              check to perform. By default it is set to ``"default"`` which 
448              compares the shapes, dtypes, and devices of the recomputed tensors 
449              against those the saved tensors. To turn off this check, specify 
450              ``"none"``. Currently these are the only two supported values. 
451              Please open an issue if you would like to see more determinism 
452              checks. This argument is only supported if ``use_reentrant=False``, 
453              if ``use_reentrant=True``, the determinism check is always disabled. 
454          debug(bool, optional): If ``True``, error messages will also include 
455              a trace of the operators ran during the original forward computation 
456              as well as the recomputation. This argument is only supported if 
457              ``use_reentrant=False``. 
458          args: tuple containing inputs to the :attr:`function` 
459   
460      Returns: 
461          Output of running :attr:`function` on :attr:`*args` 
462      """
463      if use_reentrant is None:
464          warnings.warn(
465              "torch.utils.checkpoint: the use_reentrant parameter should be "
466              "passed explicitly. In version 2.5 we will raise an exception "
467              "if use_reentrant is not passed. use_reentrant=False is "
468              "recommended, but if you need to preserve the current default "
469              "behavior, you can pass use_reentrant=True. Refer to docs for more "
470              "details on the differences between the two variants.",
471              stacklevel=2
472          )
473          use_reentrant = True
474  
475      # Hack to mix *args with **kwargs in a python 2.7-compliant way
476      preserve = kwargs.pop("preserve_rng_state", True)
477      if kwargs and use_reentrant:
478          raise ValueError(
479              "Unexpected keyword arguments: " + ",".join(arg for arg in kwargs)
480          )
481  
482      if use_reentrant:
483          if context_fn is not noop_context_fn or debug is not False:
484              raise ValueError(
485                  "Passing `context_fn` or `debug` is only supported when "
486                  "use_reentrant=False."
487              )
488          return CheckpointFunction.apply(function, preserve, *args)
489      else:
490          gen = _checkpoint_without_reentrant_generator(
491              function, preserve, context_fn, determinism_check, debug, *args, **kwargs
492          )
493          # Runs pre-forward logic
494          next(gen)
495          ret = function(*args, **kwargs)
496          # Runs post-forward logic
497          try:
498              next(gen)
499          except StopIteration:
500              return ret

503  def checkpoint_sequential(functions, segments, input, use_reentrant=None, **kwargs):
504      r"""Checkpoint a sequential model to save memory. 
505   
506      Sequential models execute a list of modules/functions in order 
507      (sequentially). Therefore, we can divide such a model in various segments 
508      and checkpoint each segment. All segments except the last will not store 
509      the intermediate activations. The inputs of each checkpointed segment will 
510      be saved for re-running the segment in the backward pass. 
511   
512      .. warning:: 
513          The ``use_reentrant`` parameter should be passed explicitly. In version 
514          2.4 we will raise an exception if ``use_reentrant`` is not passed. 
515          If you are using the ``use_reentrant=True` variant, please see 
516          :func:`~torch.utils.checkpoint.checkpoint` for 
517          the important considerations and limitations of this variant. It is 
518          recommended that you use ``use_reentrant=False``. 
519   
520      .. warning: 
521          Since PyTorch 1.4, it allows only one Tensor as the input and 
522          intermediate outputs, just like :class:`torch.nn.Sequential`. 
523   
524      Args: 
525          functions: A :class:`torch.nn.Sequential` or the list of modules or 
526              functions (comprising the model) to run sequentially. 
527          segments: Number of chunks to create in the model 
528          input: A Tensor that is input to :attr:`functions` 
529          preserve_rng_state(bool, optional):  Omit stashing and restoring 
530              the RNG state during each checkpoint. 
531              Default: ``True`` 
532          use_reentrant(bool): 
533              specify whether to use the activation checkpoint variant that 
534              requires reentrant autograd. This parameter should be passed 
535              explicitly. In version 2.5 we will raise an exception if 
536              ``use_reentrant`` is not passed. If ``use_reentrant=False``, 
537              ``checkpoint`` will use an implementation that does not require 
538              reentrant autograd. This allows ``checkpoint`` to support additional 
539              functionality, such as working as expected with 
540              ``torch.autograd.grad`` and support for keyword arguments input into 
541              the checkpointed function. 
542   
543      Returns: 
544          Output of running :attr:`functions` sequentially on :attr:`*inputs` 
545   
546      Example: 
547          >>> # xdoctest: +SKIP("stub") 
548          >>> model = nn.Sequential(...) 
549          >>> input_var = checkpoint_sequential(model, chunks, input_var) 
550      """
551      if use_reentrant is None:
552          warnings.warn(
553              "torch.utils.checkpoint.checkpoint_sequential: the use_reentrant "
554              "parameter should be passed explicitly. "
555              "In version 2.5 we will raise an exception if use_reentrant "
556              "is not passed. use_reentrant=False is "
557              "recommended, but if you need to preserve the current default "
558              "behavior, you can pass use_reentrant=True. Refer to docs for more "
559              "details on the differences between the two variants."
560          )
561          use_reentrant = True
562  
563      # Hack for keyword-only parameter in a python 2.7-compliant way
564      preserve = kwargs.pop("preserve_rng_state", True)
565      if kwargs:
566          raise ValueError(
567              "Unexpected keyword arguments: " + ",".join(arg for arg in kwargs)
568          )
569  
570      def run_function(start, end, functions):
571          def forward(input):
572              for j in range(start, end + 1):
573                  input = functions[j](input)
574              return input
575  
576          return forward
577  
578      if isinstance(functions, torch.nn.Sequential):
579          functions = list(functions.children())
580  
581      segment_size = len(functions) // segments
582      # the last chunk has to be non-volatile
583      end = -1
584      for start in range(0, segment_size * (segments - 1), segment_size):
585          end = start + segment_size - 1
586          input = checkpoint(
587              run_function(start, end, functions),
588              input,
589              use_reentrant=use_reentrant,
590              preserve_rng_state=preserve,
591          )
592      return run_function(end + 1, len(functions) - 1, functions)(input)

603  # NOTE [ Nestable Checkpoint ]
604  #
605  # The semantics of nested checkpoint can be defined by two basic rules.
606  # Following the two rules leads to an important implication that is central
607  # to motivating the design.
608  #
609  # Rule 1. Saved tensors are managed by inner-most checkpoint only and hidden
610  #         from any outer layers of checkpoint.
611  #
612  # Rule 2. The inputs of inner checkpoints are treated as tensors saved to its
613  #         parent checkpoint.
614  #
615  # Implication: To recompute any given saved tensor, we need to recompute all of
616  #              the checkpoints wrapping it.
617  #
618  # Why is this implied? To unpack a saved tensor X during backward we need to
619  # recompute the inner-most checkpoint (#1), and in order to recompute that
620  # checkpoint I need to have its inputs, which are managed by that checkpoint's
621  # parent (#2), which thus also needs to be recomputed first. Continue this line
622  # of reasoning and we realize that in order to unpack X, all checkpoints that
623  # were active at the time X was saved need to be recomputed. (unless we have
624  # already done so in that backward for some other saved tensor).
625  #
626  # In practice, we use a noop autograd Function to save inputs as saved tensors.
627  # During unpack calling ctx.saved_tensor triggers the parent checkpoint to
628  # recompute.
629  #
630  # Rule 3. We should start recomputation as if there are no checkpoints currently
631  #         active. Checkpoints encountered during recomputation are still
632  #         respected.
633  #
634  # When we start recomputation, we push the saved variable hook meant for
635  # recomputation on the stack. See examples in Rule 6 for more context.
636  #
637  #                                  * * * *
638  #
639  # Beyond the basic semantics specific to nested checkpoint, we impose several
640  # more constraints that may apply to checkpointing in general.
641  #
642  # Rule 4. Lifetime of recomputed tensors
643  #
644  #         Recomputed tensors are considered specific to particular invocations
645  #         of backward and are always cleared immediately as they are unpacked
646  #         Particularly, we require this to happen even if retain_graph=True.
647  #
648  # [ Implementation details of Rule 4 ]
649  #
650  # If we were okay with recomputed tensors staying alive after backward is run
651  # with retain_graph=True, we would store recomputed variables as the values of a
652  # WeakKeyDictionary and pack strong references to the keys, so that as we
653  # backward, those packed keys would be cleared as long as retain_graph=False.
654  # Clearing the packed key clears the corresponding entry in the WKD.
655  #
656  # If we wish recomputed variables to be immediately cleared as we unpack them in
657  # the retain_graph=True case, we cannot rely on the packed keys to be cleared by
658  # backward automatically. Instead of packing the strong reference to the key
659  # directly, we pack a container object, which we manually clear as we unpack.
660  #
661  # An important detail is that if a second backward happens, the second
662  # recomputation needs to reset the container with a newly created key.
663  #
664  # Rule 5. Stop recomputation as soon as we've recomputed the saved tensors we
665  #         know we need.
666  #
667  # [ Implementation details of Rule 5 ]
668  #
669  # During recomputation, raise an exception if the number of recomputed tensors
670  # matches the number of tensors that we expected to recompute. We wrap the
671  # recomputation call with a try-catch to catch this specific exception. See
672  # Rule #6 below for some examples.
673  #
674  # Rule 6. We support doing backward inside checkpoint context
675  #
676  # [ retain_graph is True]
677  #
678  # def fn(x):
679  #   y = x.sin()
680  #   z = y.cos()
681  #   gx, = torch.autograd.grad(z, x, retains_grad=True)
682  #   return gx, z
683  #
684  # out = checkpoint(fn)(inp)
685  # out.backward()
686  #
687  # Because z is saved by cos while checkpoint is enabled, it would not be
688  # actually saved, and so the .grad() call inside must trigger a recomputation.
689  #
690  # During recomputation the "inner pack hook" has two responsibilities:
691  #
692  # 1) As usual, populating the WeakKeyDictionary storing recomputed tensors
693  # 2) Pack the actual tensor (detached) so that one may perform backward on the
694  #    recomputed graph. The tensors saved to this graph will live until the end
695  #    of recomputation, or die earlier if someone performs backward with
696  #    retain_graph=False.
697  #
698  # More generally performing backward on the recomputed graph occurs in the
699  # following cases:
700  # - If backward is performed inside forward,
701  #   - During the original forward IF early-stop is disabled
702  #   - During the original backward
703  # - If there are multiple .grad()/.backward() calls, we would perform backward
704  #   on the recomputed graph even if early-stop is enabled (see the example below)
705  #
706  # [ retain_graph is False ]
707  #
708  # The example below shows what happens if during recomputation we find that some
709  # of the tensors we are trying to recompute have already been cleared.
710  #
711  # Spoiler: we don't do anything special, we just skip over them!
712  #
713  # def fn(x):
714  #   y = x.sin()                           # (1)
715  #   z = y.cos()                           # (2)
716  #   gx, = torch.autograd.grad(z, x)       # (3)
717  #   return x.cos() * gx                   # (4)
718  #
719  # out = checkpoint(fn)(inp)
720  # out.backward()                          # (5)
721  #
722  # 1, 2. Don't save x and y since we are inside a checkpoint.
723  # 3. Trigger a recompute of fn since x and y weren't saved.
724  #    And depending on whether early stop is enabled, either stop at (2) or
725  #    continue running the function.
726  #    Because we are running backward with retain_graph=False, we clear x and y's
727  #    holders.
728  # 4. Don't save x since we are inside a checkpoint.
729  # 5. Calling backward triggers another recompute of fn. During recompute, we see
730  #    that x and y have already been cleared in the original graph as indicated
731  #    by holder=None. We skip over them. We still save x at (4) (since its holder
732  #    is still alive.)

737  @contextlib.contextmanager
738  def set_checkpoint_early_stop(enable: bool):
739      """Context manager that sets whether checkpoint should stop recomputation early. 
740   
741      By default, non-reentrant checkpoint stops recomputation as soon as it 
742      has computed all needed Tensors. This context manager can be used to disable 
743      that feature if it is problematic for your specific application. 
744   
745      This context manager only needs to be active when forward is run. It does 
746      not need to be active during backward. 
747   
748      Example:: 
749   
750      >>> # xdoctest: +SKIP(failing) 
751      >>> message = "saved tensors default hooks are disabled" 
752      >>> with set_checkpoint_early_stop(False): 
753      ...     # Any checkpoint under this context manager will respect this 
754      ...     # context manager, even if its backward is performed outside. 
755      ...     out = checkpoint(fn, inputs) 
756      ... 
757      >>> out.backward() 
758      """
759      global _enable_checkpoint_early_stop
760      try:
761          prev = _enable_checkpoint_early_stop
762          _enable_checkpoint_early_stop = enable
763          yield
764      finally:
765          _enable_checkpoint_early_stop = prev

782      @staticmethod
783      def setup_context(ctx: Any, inputs: Tuple[Any, ...], output: Any) -> None:
784          # Only tensors can be saved with ctx.save_for_backward, everything else
785          # is captured by get_args, which is saved directly on ctx
786          tensor_indices, tensors = zip(
787              *[(i, o) for i, o in enumerate(inputs) if isinstance(o, torch.Tensor)]
788          )
789          idx2saved_idx = {b: a for a, b in enumerate(tensor_indices)}
790          # args but with tensors replaced with None as placeholders
791          args = [None if isinstance(o, torch.Tensor) else o for o in inputs]
792  
793          def get_args(saved_tensors):
794              # restore the placeholders with the original tensors grabbed from
795              # ctx.saved_tensors (which may be saved on a parent checkpoint if
796              # this checkpoint is nested, and that would trigger a recursive
797              # unpack!)
798              ret = [
799                  saved_tensors[idx2saved_idx[i]] if i in tensor_indices else o
800                  for i, o in enumerate(args)
801              ]
802              # grab the tail since we also saved the dummy to avoid having to explicitly
803              # handle the case where there are no tensor inputs
804              return ret[1:]
805  
806          ctx.get_args = get_args
807          ctx.save_for_backward(*tensors)
808

810      def backward(ctx, *grad_outputs):
811          raise AssertionError("Did not expect to backward on this graph")

814  class _CheckpointFrame:
815      def __init__(self, recompute_fn, early_stop, unpack_error_cb, metadata_fn):
816          self.recompute_fn = recompute_fn
817          self.input_saver = None
818          self.weak_holders: List[ReferenceType] = []
819          # We store this as a weakkeydictionary so that in the case of a partial
820          # backward, the entries in the dict are cleared alongside the Holder
821          # which will be removed when the SavedVariable is cleared.
822          self.recomputed: DefaultDict[
823              int, weakref.WeakKeyDictionary[_Handle, torch.Tensor]
824          ] = defaultdict(weakref.WeakKeyDictionary)
825          # We need both recomp_counter and recomputed since they can diverge
826          # https://github.com/pytorch/pytorch/pull/90105#discussion_r1135889885
827          self.recomp_counter: DefaultDict[int, int] = defaultdict(int)
828          self.is_recomputed: DefaultDict[int, bool] = defaultdict(bool)
829  
830          # See Rule 5
831          self.early_stop = early_stop
832  
833          # Debugging
834          self.metadata_fn = metadata_fn
835          self.unpack_error_cb = unpack_error_cb
836          self.x_metadatas = []
837          self.forward_completed = False
838          self.ignore_saved_mismatch = False
839  
840      def check_recomputed_tensors_match(self, gid):
841          if self.ignore_saved_mismatch:
842              # TODO: we can probably make this check stricter by checking that
843              #       the metadata of the first tensors still match.
844              return
845          # NOTE [ Error handling for checkpoint ]
846          #
847          # At a high level, we need to check that the tensors saved
848          # during original forward matches tensors saved during recompute
849          # This means handling 3 cases:
850          #
851          # 1. During recompute, more tensors were saved.
852          #
853          #    Usually this is hidden due to the StopRecomputationError
854          #    but if early stop is not enabled, or we would have errored
855          #    anyway because there aren't enough weak_holders. But we
856          #    do want to have a nice error. See the _recomputation_hook
857          #    for details.
858          if not len(self.weak_holders) == self.recomp_counter[gid]:
859              # 2. During recompute, fewer tensors were saved
860              #
861              # We know that everytime we save something do original forward
862              # we append to weak_holder, and every time we save a tensor
863              # during recompute we increment recompute_counter.
864              raise CheckpointError(
865                  "torch.utils.checkpoint: A different number of tensors was saved "
866                  "during the original forward and recomputation.\n"
867                  f"Number of tensors saved during forward: {len(self.weak_holders)}\n"
868                  f"Number of tensors saved during recomputation: {self.recomp_counter[gid]}.\n"
869                  f"{_debug_tip_msg}"
870              )
871  
872          # 3. During recompute, the same tensors were saved, but they
873          #    have different metadata
874          nb_meta_different = []
875          for idx, weak_holder in enumerate(self.weak_holders):
876              holder = weak_holder()
877              if holder is None:
878                  continue
879              # We've seen all holders since we iterate over them in order
880              # For every holder that is still alive now, it must've been
881              # alive when we saw it during recompute, therefore, the
882              # gid must be set.
883              _internal_assert(gid in holder.handles)
884              # We know this is the first unpack, so it couldn't have been set
885              # to None yet.
886              _internal_assert(holder.handles[gid] is not None)
887              # We always set these together in the recomputation hook
888              _internal_assert(holder.handles[gid] in self.recomputed[gid])
889              # see pack hook, x_metadata is 1:1 with weak_holders.
890              x_meta = self.x_metadatas[idx]
891              recomputed_x = self.recomputed[gid][holder.handles[gid]]
892              if x_meta != self.metadata_fn(recomputed_x):
893                  nb_meta_different.append((idx, x_meta, self.metadata_fn(recomputed_x)))
894  
895          if len(nb_meta_different) > 0:
896              mismatched_tensors = ""
897              for idx, x_meta, recomputed_meta in nb_meta_different:
898                  mismatched_tensors += (
899                      f"tensor at position {idx}:\n"
900                      f"saved metadata: {x_meta}\n"
901                      f"recomputed metadata: {recomputed_meta}\n"
902                  )
903              raise CheckpointError(
904                  "torch.utils.checkpoint: Recomputed values for the following tensors "
905                  "have different metadata than during the forward pass.\n"
906                  f"{mismatched_tensors}.\n"
907                  f"{_debug_tip_msg}"
908              )

1060 class _recomputation_hook(torch.autograd.graph.saved_tensors_hooks):
1061     def __init__(self, target_frame_ref: ReferenceType, gid: int):
1062         def pack_hook(x):
1063             x = x.detach() if x.requires_grad else x
1064             target_frame = target_frame_ref()
1065             assert target_frame is not None  # appease mypy
1066             recomp_idx = target_frame.recomp_counter[gid]
1067             target_frame.recomp_counter[gid] += 1
1068 
1069             if recomp_idx >= len(target_frame.weak_holders):
1070                 assert not target_frame.early_stop
1071                 if not target_frame.forward_completed:
1072                     # We run into this case when early stop is not enabled and do
1073                     # grad within checkpoint.
1074                     # We need to set this flag, so we don't error out later when
1075                     # we check if the number of tensors saved during forward and
1076                     # recomputation match.
1077                     target_frame.ignore_saved_mismatch = True
1078                     return x
1079                 raise CheckpointError(
1080                     "torch.utils.checkpoint: trying to save more tensors during "
1081                     "recomputation than during the original forward pass.\n"
1082                     f"{_debug_tip_msg}"
1083                 )
1084 
1085             holder = target_frame.weak_holders[recomp_idx]()
1086 
1087             # This holder may have been cleared because someone may have called
1088             # backward within forward. If so, we don't need to save.
1089             if holder is not None:
1090                 _internal_assert(holder.handles.get(gid, None) is None)
1091                 holder.handles[gid] = _Handle()
1092                 target_frame.recomputed[gid][holder.handles[gid]] = x
1093 
1094             if target_frame.early_stop and target_frame.recomp_counter[gid] == len(
1095                 target_frame.weak_holders
1096             ):
1097                 raise _StopRecomputationError
1098             # See Rule 6: [ retain_graph is True ] above
1099             return x
1100 
1101         def unpack_hook(x):
1102             # See Rule 6: [ retain_graph is True ] above for an example of when
1103             # the graph created during recomputation could be backwarded.
1104             return x
1105 
1106         super().__init__(pack_hook, unpack_hook)

1119 class _checkpoint_hook(torch.autograd.graph.saved_tensors_hooks):
1120     def __init__(self, frame):
1121         def pack_hook(x):
1122             # See Rule 4 above
1123             holder = _Holder()
1124             frame.weak_holders.append(weakref.ref(holder))
1125             # Save metadata to detect non-determinism
1126             if frame.metadata_fn is not None:
1127                 with torch.no_grad():
1128                     frame.x_metadatas.append(frame.metadata_fn(x))
1129             return holder
1130 
1131         def unpack_hook(holder):
1132             gid = torch._C._current_graph_task_id()
1133             if gid == -1:
1134                 # generate a temporary id if we trigger unpack outside of a backward call
1135                 gid = int(uuid.uuid4())
1136 
1137             if not frame.is_recomputed[gid]:
1138                 ctx = frame.input_saver.grad_fn
1139                 args = ctx.get_args(ctx.saved_tensors)
1140 
1141                 try:
1142                     with _recomputation_hook(
1143                         weakref.ref(frame), gid
1144                     ), torch.autograd.enable_grad():
1145                         # See Note: [compiled autograd and checkpoint unpack hook]
1146                         _run_fn_with_dynamo_disabled(frame.recompute_fn, *args)
1147                 except _StopRecomputationError:
1148                     pass
1149                 frame.is_recomputed[gid] = True
1150                 frame.check_recomputed_tensors_match(gid)
1151 
1152             _internal_assert(gid in holder.handles)
1153 
1154             if holder.handles[gid] is None:
1155                 raise CheckpointError(
1156                     "torch.utils.checkpoint: Unpack is being triggered for a tensor that was already "
1157                     "unpacked once. If you are calling ctx.saved_tensors in backward, make sure to do "
1158                     "so only once. Otherwise please open an issue with details on your use case."
1159                 )
1160             _internal_assert(holder.handles[gid] in frame.recomputed[gid])
1161             ret = frame.recomputed[gid][holder.handles[gid]]
1162             holder.handles[gid] = None
1163             return ret
1164 
1165         if frame.unpack_error_cb is not None:
1166             def unpack_hook_with_error_cb(holder):
1167                 try:
1168                     return unpack_hook(holder)
1169                 except CheckpointError as e:
1170                     frame.unpack_error_cb(e)
1171             super().__init__(pack_hook, unpack_hook_with_error_cb)
1172         else:
1173             super().__init__(pack_hook, unpack_hook)

1198 def _maybe_detach(x, any_ret_has_alias_info):
1199     # We detach for two separate reasons:
1200     # - For view ops, we need to ensure that when the tensor is returned from
1201     #   CachedDispatchMode, as_view sees that the AutogradMeta is nullptr
1202     # - Avoid reference cycles
1203     # For case 1, it is not enough to check whether x has differentiable dtype
1204     # because non-differentiable dtype can have non-nullptr AutogradMeta, e.g.
1205     # when the tensor is a view.
1206     if isinstance(x, torch.Tensor) and (x.is_floating_point() or x.is_complex() or any_ret_has_alias_info):
1207         with torch._C._SetExcludeDispatchKeyGuard(torch._C.DispatchKey.ADInplaceOrView, False):
1208             # Ensure that view performed beneath autograd properly propagates
1209             # version counter. TODO: Use reentrant_dispatch instead of
1210             # manually manipulating dispatch keys. Using reentrant_dispatch
1211             # would respect inference_mode, though that is not relevant for
1212             # this case.
1213             x = x.detach()
1214     return x

1243 class CheckpointPolicy(enum.Enum):
1244     """ 
1245     Enum for specifying the policy for checkpointing during backpropagation. 
1246  
1247     The following policies are supported: 
1248  
1249     - ``{MUST,PREFER}_SAVE``: The operation's output will be saved during the forward 
1250       pass and will not be recomputed during the backward pass 
1251     - ``{MUST,PREFER}_RECOMPUTE``: The operation's output will not be saved during the 
1252       forward pass and will be recomputed during the backward pass 
1253  
1254     Use ``MUST_*`` over ``PREFER_*`` to indicate that the policy should not be overridden 
1255     by other subsystems like `torch.compile`. 
1256  
1257     .. note:: 
1258         A policy function that always returns ``PREFER_RECOMPUTE`` is 
1259         equivalent to vanilla checkpointing. 
1260  
1261         A policy function that returns ``PREFER_SAVE`` every op is 
1262         NOT equivalent to not using checkpointing. Using such a policy would 
1263         save additional tensors not limited to ones that are actually needed for 
1264         gradient computation. 
1265     """
1266     MUST_SAVE = 0
1267     PREFER_SAVE = 1
1268     MUST_RECOMPUTE = 2
1269     PREFER_RECOMPUTE = 3

1272 def _policy_from_bool(b):
1273     # For backward compatability
1274     return CheckpointPolicy.MUST_SAVE if b else CheckpointPolicy.PREFER_RECOMPUTE

1323 class _CachedTorchDispatchMode(TorchDispatchMode):
1324     # Used together with _CachedTorchDispatchMode to implement SAC.
1325     def __init__(self, policy_fn, storage, allow_cache_entry_mutation):
1326         self.policy_fn = policy_fn
1327         self.storage = storage
1328         self.allow_cache_entry_mutation = allow_cache_entry_mutation
1329 
1330     def __torch_dispatch__(self, func, types, args=(), kwargs=None):
1331         if func in SAC_IGNORED_OPS:
1332             return func(*args, **kwargs)
1333 
1334         kwargs = {} if kwargs is None else kwargs
1335         policy = self.policy_fn(SelectiveCheckpointContext(is_recompute=True),
1336                                 func, *args, **kwargs)
1337         if isinstance(policy, bool):
1338             policy = _policy_from_bool(policy)
1339 
1340         is_compiling = _is_compiling(func, args, kwargs)
1341 
1342         if policy in (CheckpointPolicy.MUST_SAVE, CheckpointPolicy.PREFER_SAVE) or is_compiling:

1345                 raise RuntimeError(f"{func} encountered during backward, but not found in storage")

1348                     "Trying to backward an extra time. You are only allowed to backward once "
1349                     "on any region computed under selective activation checkpoint."
1350                 )
1351             out = tree_map(lambda x: x.get_val(self.allow_cache_entry_mutation), storage.pop(0))
1352         else:
1353             out = func(*args, **kwargs)
1354         return out

1357 def create_selective_checkpoint_contexts(policy_fn_or_list, allow_cache_entry_mutation=False):
1358     """ 
1359     Helper to avoid recomputing certain ops during activation checkpointing. 
1360  
1361     Use this with `torch.utils.checkpoint.checkpoint` to control which 
1362     operations are recomputed during the backward pass. 
1363  
1364     Args: 
1365         policy_fn_or_list (Callable or List): 
1366           - If a policy function is provided, it should accept a 
1367             :class:`SelectiveCheckpointContext`, the :class:`OpOverload`, args and 
1368             kwargs to the op, and return a :class:`CheckpointPolicy` enum value 
1369             indicating whether the execution of the op should be recomputed or not. 
1370           - If a list of operations is provided, it is equivalent to a policy 
1371             returning `CheckpointPolicy.MUST_SAVE` for the specified 
1372             operations and `CheckpointPolicy.PREFER_RECOMPUTE` for all other 
1373             operations. 
1374         allow_cache_entry_mutation (bool, optional): By default, an error is 
1375             raised if any tensors cached by selective activation checkpoint are 
1376             mutated in order to ensure correctness. If set to `True`, this check 
1377             is disabled. 
1378     Returns: 
1379         A tuple of two context managers. 
1380  
1381     Example: 
1382         >>> # xdoctest: +REQUIRES(LINUX) 
1383         >>> import functools 
1384         >>> 
1385         >>> x = torch.rand(10, 10, requires_grad=True) 
1386         >>> y = torch.rand(10, 10, requires_grad=True) 
1387         >>> 
1388         >>> ops_to_save = [ 
1389         >>>    torch.ops.aten.mm.default, 
1390         >>> ] 
1391         >>> 
1392         >>> def policy_fn(ctx, op, *args, **kwargs): 
1393         >>>    if op in ops_to_save: 
1394         >>>        return CheckpointPolicy.MUST_SAVE 
1395         >>>    else: 
1396         >>>        return CheckpointPolicy.PREFER_RECOMPUTE 
1397         >>> 
1398         >>> context_fn = functools.partial(create_selective_checkpoint_contexts, policy_fn) 
1399         >>> 
1400         >>> # or equivalently 
1401         >>> context_fn = functools.partial(create_selective_checkpoint_contexts, ops_to_save) 
1402         >>> 
1403         >>> def fn(x, y): 
1404         >>>     return torch.sigmoid(torch.matmul(torch.matmul(x, y), y)) * y 
1405         >>> 
1406         >>> out = torch.utils.checkpoint.checkpoint( 
1407         >>>     fn, x, y, 
1408         >>>     use_reentrant=False, 
1409         >>>     context_fn=context_fn, 
1410         >>> ) 
1411     """
1412     # NB: If grad_mode is disabled, checkpoint would not run forward under
1413     #     context_fn anyway, so proceed as usual.
1414     if isinstance(policy_fn_or_list, list):
1415         for op in policy_fn_or_list:
1416             if not isinstance(op, (torch._ops.OpOverload, torch._ops.HigherOrderOperator)):
1417                 _extra_msg = (
1418                     "Please update the OpOverloadPacket to a specific OpOverload."
1419                     "For example, if you have `torch.ops.aten.mm`, change it to `torch.ops.aten.mm.default`."
1420                 ) if isinstance(op, torch._ops.OpOverloadPacket) else ""
1421                 raise ValueError(
1422                     f"Expected op in `op_list` to be an OpOverload but got: {op} "
1423                     f"of type {type(op)}. {_extra_msg}"
1424                 )
1425 
1426         def policy_fn(ctx, op, *args, **kwargs):
1427             if op in policy_fn_or_list:
1428                 return CheckpointPolicy.MUST_SAVE
1429             else:
1430                 return CheckpointPolicy.PREFER_RECOMPUTE
1431     elif callable(policy_fn_or_list):
1432         policy_fn = policy_fn_or_list
1433     else:
1434         raise TypeError("policy_fn_or_list must be either a function or a list of ops.")
1435 
1436     storage: Dict[Any, List[Any]] = defaultdict(list)
1437     return (
1438         _CachingTorchDispatchMode(policy_fn, storage),
1439         _CachedTorchDispatchMode(policy_fn, storage, allow_cache_entry_mutation),
1440     )

1445 def _checkpoint_without_reentrant_generator(
1446     fn,
1447     preserve_rng_state=True,
1448     context_fn: Callable[[], Tuple[ContextManager, ContextManager]] = noop_context_fn,
1449     determinism_check: str = _DEFAULT_DETERMINISM_MODE,
1450     debug: bool = False,
1451     *args,
1452     **kwargs
1453 ):
1454     """Checkpointing without reentrant autograd. 
1455  
1456     Args: 
1457         fn: describes what to run in the forward pass of the model or 
1458             part of the model. It should also know how to handle the inputs 
1459             passed as the tuple. For example, in LSTM, if user passes 
1460             ``(activation, hidden)``, :attr:`function` should correctly use the 
1461             first input as ``activation`` and the second input as ``hidden`` 
1462         preserve_rng_state(bool, optional):  Omit stashing and restoring 
1463             the RNG state during each checkpoint. 
1464             Default: ``True`` 
1465         context_fn(Callable, optional): A callable returning a tuple of two 
1466             context managers. The function and its recomputation will be run 
1467             under the first and second context managers respectively. 
1468         determinism_check(str, optional): A string specifying the determinism 
1469             check to perform. By default it is set to ``"default"`` which 
1470             compares the shapes, dtypes, and devices of the recomputed tensors 
1471             against those the saved tensors. To turn off this check, specify 
1472             ``"none"``. Currently these are the only two supported values. 
1473             Please open an issue if you would like to see more determinism 
1474             checks. 
1475         debug(bool, optional): If ``True``, error messages will also include 
1476             a trace of the operators ran during the original forward computation 
1477             as well as the recomputation. 
1478         *args: Arguments to pass in to the given ``function``. 
1479         **kwargs: Keyword arguments to pass into the given ``function``. 
1480     """
1481     unpack_error_cb = None
1482 
1483     if _checkpoint_debug_enabled if _checkpoint_debug_enabled is not None else debug:
1484         if context_fn != noop_context_fn:
1485             raise ValueError(
1486                 "debug=True is incompatible with non-default context_fn"
1487             )
1488         context_fn, unpack_error_cb = _get_debug_context_and_cb()
1489 
1490     if determinism_check in _allowed_determinism_checks_to_fns:
1491         metadata_fn = _allowed_determinism_checks_to_fns[determinism_check]
1492     else:
1493         raise ValueError(
1494             f"determinism_check should be one of {list(_allowed_determinism_checks_to_fns.keys())}, "
1495             f"but got {determinism_check}"
1496         )
1497 
1498     device_type = _infer_device_type(*args)
1499     device_module = _get_device_module(device_type)
1500     forward_context, recompute_context = context_fn()
1501     if _is_compiling(fn, args, kwargs) and context_fn != noop_context_fn:
1502         assert (
1503             isinstance(forward_context, TorchDispatchMode) and
1504             isinstance(recompute_context, TorchDispatchMode)
1505         ), \
1506             "In torch.compile mode, `context_fn` arg passed to `torch.utils.checkpoint` " + \
1507             "must generate a tuple of two `TorchDispatchMode`s."
1508     # Accommodates the (remote) possibility that autocast is enabled for cpu AND gpu.
1509     device_autocast_kwargs, cpu_autocast_kwargs = _get_autocast_kwargs(device_type=device_type)
1510 
1511     if preserve_rng_state:
1512         fwd_cpu_state = torch.get_rng_state()
1513         # Don't eagerly initialize the cuda context by accident.
1514         # (If the user intends that the context is initialized later, within their
1515         # run_function, we SHOULD actually stash the cuda state here.  Unfortunately,
1516         # we have no way to anticipate this will happen before we run the function.
1517         # If they do so, we raise an error.)
1518         had_device_in_fwd = False
1519         if getattr(device_module, "_initialized", False):
1520             had_device_in_fwd = True
1521             fwd_devices, fwd_device_states = get_device_states(*args)
1522 
1523     def recompute_fn(*inputs):
1524         kwargs, *args = inputs
1525         # This will be called later during recomputation. This wrapping enables
1526         # the necessary global state to be captured.
1527         rng_devices = []
1528         if preserve_rng_state and had_device_in_fwd:
1529             rng_devices = fwd_devices
1530         with torch.random.fork_rng(
1531             devices=rng_devices, enabled=preserve_rng_state, device_type=device_type
1532         ):
1533             if preserve_rng_state:
1534                 torch.set_rng_state(fwd_cpu_state)
1535                 if had_device_in_fwd:
1536                     set_device_states(fwd_devices, fwd_device_states, device_type=device_type)
1537 
1538             device_autocast_ctx = torch.amp.autocast(
1539                 device_type=device_type, **device_autocast_kwargs
1540             ) if torch.amp.is_autocast_available(device_type) else contextlib.nullcontext()
1541             with device_autocast_ctx, torch.amp.autocast("cpu", **cpu_autocast_kwargs), recompute_context:  # type: ignore[attr-defined]
1542                 fn(*args, **kwargs)
1543 
1544     new_frame = _CheckpointFrame(
1545         recompute_fn,
1546         _enable_checkpoint_early_stop,
1547         unpack_error_cb,
1548         metadata_fn
1549     )
1550     dummy = torch.empty((0,), requires_grad=True)
1551     new_frame.input_saver = _NoopSaveInputs.apply(dummy, kwargs, *args)
1552 
1553     # When ambient grad_mode is False
1554     if new_frame.input_saver.grad_fn is None:
1555         yield
1556         return
1557 
1558     with _checkpoint_hook(new_frame), forward_context:
1559         yield
1560     new_frame.forward_completed = True
1561 
1562     if getattr(device_module, "_initialized", False) and \
1563        preserve_rng_state and not had_device_in_fwd:  # type: ignore[possibly-undefined]
1564         # Device was not initialized before running the forward, so we didn't
1565         # stash the device state.
1566         raise RuntimeError(
1567             "PyTorch's device state was initialized in the forward pass "
1568             "of a Checkpoint, which is not allowed. Please open an issue "
1569             "if you need this feature."
1570         )
1571 
1572     return
