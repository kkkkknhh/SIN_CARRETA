9    @pytest.mark.slow
10   class TestReturnReal(util.F2PyTest):
11       def check_function(self, t, tname):
12           if tname in ["t0", "t4", "s0", "s4"]:
13               err = 1e-5
14           else:
15               err = 0.0
16           assert abs(t(234) - 234.0) <= err
17           assert abs(t(234.6) - 234.6) <= err
18           assert abs(t("234") - 234) <= err
19           assert abs(t("234.6") - 234.6) <= err
20           assert abs(t(-234) + 234) <= err
21           assert abs(t([234]) - 234) <= err
22           assert abs(t((234, )) - 234.0) <= err
23           assert abs(t(array(234)) - 234.0) <= err
24           assert abs(t(array(234).astype("b")) + 22) <= err
25           assert abs(t(array(234, "h")) - 234.0) <= err
26           assert abs(t(array(234, "i")) - 234.0) <= err
27           assert abs(t(array(234, "l")) - 234.0) <= err
28           assert abs(t(array(234, "B")) - 234.0) <= err
29           assert abs(t(array(234, "f")) - 234.0) <= err
30           assert abs(t(array(234, "d")) - 234.0) <= err
31           if tname in ["t0", "t4", "s0", "s4"]:
32               assert t(1e200) == t(1e300)  # inf
33   
34           # pytest.raises(ValueError, t, array([234], 'S1'))
35           pytest.raises(ValueError, t, "abc")
36   
37           pytest.raises(IndexError, t, [])
38           pytest.raises(IndexError, t, ())
39   
40           pytest.raises(Exception, t, t)
41           pytest.raises(Exception, t, {})
42   
43           try:
44               r = t(10**400)
45               assert repr(r) in ["inf", "Infinity"]
46           except OverflowError:
47               pass
