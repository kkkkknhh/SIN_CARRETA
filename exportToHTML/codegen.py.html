72       def __init__(
73           self,
74           tx: "InstructionTranslatorBase",
75           root: Optional[torch.nn.Module] = None,
76           graph_output_var: Optional[str] = None,
77           tempvars=None,
78           overridden_sources=None,
79       ) -> None:
80           self.root = root
81           self.top_of_stack: Optional[Union[VariableTracker, Source]] = None
82           self.uses: Counter[Union[VariableTracker, Source]] = collections.Counter()
83           self.graph_outputs: dict[int, GraphOutputEntry] = {}
84           self._output: list[Instruction] = []
85           # This determines which VariableTracker/Source should be stored as
86           # locals, and maps the VariableTracker/Source to the local variable
87           # name. Note that it could map to None initially, in which case we'll
88           # overwrite it to map to real temporary names via `add_cache`.
89           self.tempvars = tempvars or {}
90           self.tx = tx
91           self.graph_output_var = graph_output_var
92           self.code_options = self.tx.output.code_options
93           self.cell_and_freevars = self.tx.cell_and_freevars

96           # This serves as a way for codegen to use a different source; we need
97           # this because sometimes we can't easily modify the original source
98           # without affecting other components, e.g., guards.
99           self.overridden_sources: dict[Source, Source] = overridden_sources or {}

101      def restore_stack(self, stack_values, *, value_from_source=True):
102          prev = self.value_from_source
103          self.value_from_source &= value_from_source
104          try:
105              self.foreach(stack_values)
106          finally:
107              self.value_from_source = prev

116      def add_push_null(self, gen_fn, call_function_ex=False):
117          """ 
118          `gen_fn` generates instructions via PyCodegen methods 
119          that push a single callable to the stack. 
120   
121          `add_push_null` pushes a NULL to the stack before or after the 
122          instructions generated by `gen_fn`, depending on Python version. 
123   
124          Will attempt to use the NULL push bit for instructions 
125          with such bits (LOAD_GLOBAL 3.11+, LOAD_ATTR 3.12+, LOAD_SUPER_ATTR). 
126          """
127          old_len = len(self._output)
128          if sys.version_info < (3, 13):
129              # gen_fn may DUP_TOP instead if TOS is not cleared.
130              # Will cause problems since NULL will be pushed right
131              # before the generated instructions in <= 3.12
132              self.clear_tos()
133          gen_fn()
134          # inplace modify self._output
135          added_insts = self._output[old_len:]
136          del self._output[old_len:]
137          if call_function_ex:
138              self._output.extend(add_push_null_call_function_ex(added_insts))
139          else:
140              self._output.extend(add_push_null(added_insts))
141          if sys.version_info >= (3, 13):
142              # NULL will be at top of stack
143              self.clear_tos()

145      def __call__(self, value, allow_cache=True):
146          """ 
147          Generate code such that top-of-stack (TOS) is set to value. 
148   
149          `allow_cache` controls the behavior in the following manner. `value` can 
150          either be a VariableTracker or a Source. 
151   
152          If `value` is a `Source`, `allow_cache` must be True (invariant asserted 
153          below). If the source was reconstructed earlier, we will reuse the 
154          generated code by loading from top of stack or tempvars. 
155   
156          If `value` is a `VariableTracker`, we have the following cases: 
157   
158          1) `allow_cache=True` 
159              a) If the value.source is not None, we will emit the code based on 
160              `value.source` to handle aliasing. 
161              b) If value.source is None (example reconstructing a local list 
162              returned by the compiled function), we will reconstruct the variable 
163              tracker (w/o any source) to emit bytecode that generates a new 
164              python object. 
165   
166              In both cases of value.source being None or not, if the value was 
167              reconstructed earlier, we will reuse the generated code by loading from 
168              top of stack or tempvars. 
169   
170          2) `allow_cache=False` - This is a special case (allow_cache defaults to 
171          True). 
172              a) If the value.source is not None, we reconstruct the variable 
173              tracker and emit a new python object. You might wonder what about 
174              aliasing? The place where we use this config also has the followup 
175              code where the original python object is assigned to this new python 
176              value to handle aliasing (check side_effects.py and search for 
177              allow_cache=False). 
178   
179              b) If value.source is None, this is not allowed. TODO - assert this. 
180   
181          Notable effects: 
182          1. `self.top_of_stack` will be set to `value`, if we don't codegen 
183             `value` based on source. 
184          2. `self.uses[value]` will increment, unless (a). we codegen via 
185              `top_of_stack` or cached `tempvars`, or (b). `value` has special VT 
186              types like `NNModuleVariable`, etc. 
187          """
188          if isinstance(value, Source):
189              # If the source needs to be overridden, use the new one.
190              source = self.overridden_sources.get(value, value)
191              assert allow_cache is True, "allow_cache must be True for Source"
192              if self.top_of_stack is value:
193                  self._output.append(create_dup_top())
194                  return
195  
196              if self.tempvars.get(source) is not None:
197                  self._output.append(self.create_load(self.tempvars[source]))
198                  self.top_of_stack = source
199                  return
200  
201              self.uses[source] += 1
202              try:
203                  self.call_reconstruct(source)
204              except NotImplementedError:
205                  unimplemented_v2(
206                      gb_type="Reconstruction failure: source.reconstruct not implemented",
207                      context=str(source),
208                      explanation=f"Dynamo has no bytecode reconstruction implemented for {type(source)} variable {source}.",
209                      hints=[*graph_break_hints.DYNAMO_BUG],
210                  )
211              if source in self.tempvars:
212                  self._output.append(create_dup_top())
213                  self.add_cache(source)
214              self.top_of_stack = source
215  
216              return
217  
218          assert isinstance(value, VariableTracker)
219          output = self._output
220          graph_outputs = self.graph_outputs
221  
222          if allow_cache:
223              if self.top_of_stack is value:
224                  output.append(create_dup_top())
225                  return
226  
227              if self.tempvars.get(value) is not None:
228                  output.append(self.create_load(self.tempvars[value]))
229                  self.top_of_stack = value
230                  return
231  
232          if value.is_realized() and isinstance(
233              value, ContextlibContextManagerLocalGeneratorObjectVariable
234          ):
235              raise IncorrectUsage(
236                  "NYI: Returning a @contextmanager object from a torch.compile function"
237              )
238  
239          # Dynamo normally prefers codegen from source to account for aliasing.
240          if (
241              value.source is not None
242              and allow_cache
243              and not (
244                  value.is_realized() and isinstance(value, LocalGeneratorObjectVariable)
245              )
246          ):
247              # There's a corner case for export: for instance, if the computation
248              # graph is just identity on an input tensor, Dynamo would just emit
249              # a `LOAD_FAST` from the input source, rather than generating an
250              # identity FX graph.
251              #
252              # However, export wants to maximize graph capture; in the case
253              # above, export _wants to_ obtain an identity FX graph (despite it
254              # appears unnecessarily expensive for `torch.compile`), so we have
255              # the following option to override Dynamo's preference for codegen
256              # from source. Moreover, this option applies recursively, for cases
257              # like input tensor being returned in a new dictionary.
258              #
259              # And why the `ValueMutationExisting` check? Not sure, so leaving it
260              # to keep the old behavior, as when `value_from_source` was
261              # introduced. TODO sort out the invariants among side effect,
262              # codegen and export.
263              if (
264                  isinstance(value.mutation_type, ValueMutationExisting)
265                  or self.value_from_source
266              ):
267                  return self(value.source)
268  
269          if value.is_python_constant() and is_safe_constant(value.as_python_constant()):
270              output.append(self.create_load_const(value.as_python_constant()))
271          elif isinstance(value, TensorWithTFOverrideVariable):
272              graph_outputs_key = self.add_graph_output(value)
273  
274              self.add_push_null(
275                  lambda: self.load_import_from(utils.__name__, "to_subclass")
276              )
277              self.load_graph_output(graph_outputs[graph_outputs_key].index)
278              output.append(
279                  self.create_load_global(
280                      value.global_mangled_class_name(self.tx), add=True
281                  )
282              )
283              output.extend(create_call_function(2, False))
284          elif (
285              isinstance(value, SymNodeVariable)
286              and value.python_type() == float
287              and not self.tx.export
288          ):
289              # This is a little unusual; force the output convention to be a
290              # Tensor here.  Don't do this for export because this is
291              # apparently load bearing for export tests (but I am a bit
292              # doubtful it actually works in the real world)
293              # NB: It works to add_graph_output on a computed expression
294              # as_tensor here, because we memoize as_tensor calls on
295              # SymNodeVariable!
296              graph_outputs_key = self.add_graph_output(
297                  value.as_tensor(self.tx, torch.float64)
298              )
299  
300              def gen_fn():
301                  self.load_graph_output(graph_outputs[graph_outputs_key].index)
302                  output.append(self.create_load_attr("item"))
303  
304              self.add_push_null(gen_fn)
305              output.extend(create_call_function(0, False))
306          elif isinstance(
307              value,
308              (
309                  TensorVariable,
310                  SymNodeVariable,
311                  UnspecializedPythonVariable,
312                  NumpyNdarrayVariable,
313              ),
314          ):
315              graph_outputs_key = self.add_graph_output(value)
316  
317              if isinstance(value, NumpyNdarrayVariable):
318                  self.add_push_null(
319                      lambda: self.load_import_from(utils.__name__, "to_numpy_helper")
320                  )
321                  self.load_graph_output(graph_outputs[graph_outputs_key].index)
322                  output.extend(create_call_function(1, False))
323              elif isinstance(value, UnspecializedPythonVariable) and value.need_unwrap:
324  
325                  def gen_fn():
326                      self.load_graph_output(graph_outputs[graph_outputs_key].index)
327                      output.append(self.create_load_attr("item"))
328  
329                  self.add_push_null(gen_fn)
330                  output.extend(create_call_function(0, False))
331              else:
332                  self.load_graph_output(graph_outputs[graph_outputs_key].index)
333          elif isinstance(value, NNModuleVariable):
334              parts = value.module_key.split(".")
335              if parts[0] in self.code_options["co_varnames"]:
336                  output.append(self.create_load(parts[0]))
337                  parts = parts[1:]
338              else:
339                  assert self.root is not None
340                  output.append(self.create_load_const_unchecked(self.root))
341              for part in parts:
342                  output.append(self.create_load_attr(part))
343          else:
344              self.uses[value] += 1
345              try:
346                  self.call_reconstruct(value)
347              except NotImplementedError:
348                  unimplemented_v2(
349                      gb_type="Reconstruction failure",
350                      context=str(value),
351                      explanation=f"Dynamo has no bytecode reconstruction implemented for sourceless variable {value}.",
352                      hints=[
353                          "If Dynamo is attempting to trace a return statement and your code is attempting to return a variable "
354                          "that Dynamo cannot reconstruct, then remove it from the return statement.",
355                          *graph_break_hints.CAUSED_BY_EARLIER_GRAPH_BREAK,
356                          "Report an issue to PyTorch if you need reconstrtuction support. Note that objects that don't have "
357                          "reconstruction rules may be fundamentally unreconstructable.",
358                      ],
359                  )
360              if allow_cache and value in self.tempvars:
361                  self._output.append(create_dup_top())
362                  self.add_cache(value)
363  
364          self.top_of_stack = value

366      def add_graph_output(self, value):
367          graph_outputs_key = id(value.as_proxy())
368          if graph_outputs_key not in self.graph_outputs:
369              self.graph_outputs[graph_outputs_key] = GraphOutputEntry(
370                  len(self.graph_outputs), value
371              )
372          return graph_outputs_key

374      def load_graph_output(self, index):
375          output = self._output
376          output.append(self.create_load(self.graph_output_var))
377          output.append(self.create_load_const(index))
378          output.append(self.create_binary_subscr())

380      def add_cache(self, value):
381          var = self.new_var()
382          self.tempvars[value] = var
383          self._output.append(self.create_store(var))

392      def setup_globally_cached(self, name, value):
393          """Store value in a new global"""
394          name = re.sub(r"[^a-zA-Z0-9_]+", "_", name)
395          f_globals = self.tx.f_globals
396          if name in f_globals:
397              assert id(f_globals[name]) == id(value)
398          else:
399              f_globals[name] = value
400          return [self.create_load_global(name, add=True)]

477      def load_function_name(self, fn_name, push_null, num_on_stack=0):
478          """Load the global fn_name on the stack num_on_stack down"""
479          output = []
480          if push_null and sys.version_info >= (3, 11):
481              output.extend(add_push_null(self.create_load_global(fn_name, add=True)))
482              if num_on_stack > 0:
483                  output.extend(
484                      [
485                          *self.rot_n(num_on_stack + 2),
486                          *self.rot_n(num_on_stack + 2),
487                      ]
488                  )
489          else:
490              output.extend(
491                  [
492                      self.create_load_global(fn_name, add=True),
493                      *self.rot_n(num_on_stack + 1),
494                  ]
495              )
496          return output

498      def rot_n(self, n):
499          try:
500              return create_rot_n(n)
501          except AttributeError:
502              # desired rotate bytecode doesn't exist, generate equivalent bytecode
503              return [
504                  create_instruction("BUILD_TUPLE", arg=n),
505                  self.create_load_const_unchecked(rot_n_helper(n)),
506                  *create_rot_n(2),
507                  create_instruction("CALL_FUNCTION_EX", arg=0),
508                  create_instruction("UNPACK_SEQUENCE", arg=n),
509              ]

526      def make_function_with_closure(
527          self, fn_name: str, code: types.CodeType, push_null: bool, num_on_stack=0
528      ):
529          freevars = code.co_freevars
530          assert freevars
531          output = self._output
532  
533          def gen_fn():
534              # Emitting `LOAD_FAST/LOAD_CLOSURE` with names in `co_freevars`
535              # requires that in the generated bytecode, these cells would keep
536              # their original local names, which we ensure via
537              # `CellVariable.local_name`.
538              for var in freevars:
539                  assert var in self.cell_and_freevars()
540                  output.append(self.create_load_closure(var))
541              output.append(create_instruction("BUILD_TUPLE", arg=len(freevars)))
542              output.append(self.create_load_const(code))
543              if sys.version_info < (3, 11):
544                  output.append(self.create_load_const(fn_name))
545              if sys.version_info >= (3, 13):
546                  output.extend(
547                      [
548                          create_instruction("MAKE_FUNCTION"),
549                          create_instruction("SET_FUNCTION_ATTRIBUTE", arg=0x08),
550                      ]
551                  )
552              else:
553                  output.append(create_instruction("MAKE_FUNCTION", arg=0x08))
554  
555          if push_null and sys.version_info >= (3, 11):
556              self.add_push_null(gen_fn)
557              output.extend(self.rot_n(num_on_stack + 2))
558              output.extend(self.rot_n(num_on_stack + 2))
559          else:
560              gen_fn()
561              output.extend(self.rot_n(num_on_stack + 1))
562          self.clear_tos()

564      def create_load_python_module(self, mod) -> Instruction:
565          """ 
566          Generate a LOAD_GLOBAL instruction to fetch a given python module. 
567          """
568          output = self.tx.output
569          global_scope = output.global_scope
570          name = re.sub(r"^.*[.]", "", mod.__name__)
571          if global_scope.get(name, None) is mod:
572              return self.create_load_global(name, add=True)
573          prefix = f"___module_{name}"
574          global_name = self.tx.output.install_global_by_id(prefix, mod)
575          return self.create_load_global(global_name, add=True)

577      def mark_source_temp(self, source: Source) -> None:
578          """ 
579          Mark a source as a temp variable, so that it can be reused. 
580          """
581          if source not in self.tempvars:
582              self.tempvars[source] = None

584      def make_call_generated_code(self, fn_name: str) -> None:
585          """Call the generated code function stored in fn_name"""
586          self.extend_output(self.load_function_name(fn_name, True))
587  
588          graphargs = self.tx.output.graphargs
589  
590          seen_sources: OrderedSet[Source] = OrderedSet()
591  
592          def collect_temp_source(source):
593              if source in seen_sources:
594                  # This source is used at least twice, so it can be reused
595                  self.mark_source_temp(source)
596                  # Dont trace source further. This prevents us from marking too
597                  # many nodes as temp sources.
598                  return
599  
600              seen_sources.add(source)
601  
602              if isinstance(source, ChainedSource):
603                  collect_temp_source(source.base)
604  
605              if isinstance(source, DictGetItemSource) and isinstance(
606                  source.index, Source
607              ):
608                  collect_temp_source(source.index)
609  
610          # Collect all the sources that are used more than once, so that we can
611          # generate tmp variables in the generated pre-graph bytecode. This
612          # essentially implements CSE.
613          for arg in graphargs:
614              if arg.source is not None:
615                  collect_temp_source(arg.source)
616  
617          cm_var = None
618          if config.record_runtime_overhead:
619              # Record the pregraph bytecode start
620              self.add_push_null(
621                  lambda: self.load_import_from(
622                      utils.__name__, "record_pregraph_bytecode_enter"
623                  )
624              )
625              self.extend_output(create_call_function(0, False))
626              cm_var = self.new_var()
627              self.store(cm_var)
628  
629          for arg in graphargs:
630              if arg.pass_arg_as_tensor:
631                  self.add_push_null(
632                      lambda: self.extend_output(
633                          [
634                              self.create_load_python_module(torch),
635                              self.create_load_attr("_as_tensor_fullprec"),
636                          ]
637                      )
638                  )
639                  self.call_reconstruct(arg)
640                  self.extend_output(create_call_function(1, False))
641              else:
642                  self.call_reconstruct(arg)
643  
644          if config.record_runtime_overhead:
645              # Record the pregraph bytecode end
646              self.add_push_null(
647                  lambda: self.load_import_from(
648                      utils.__name__, "record_pregraph_bytecode_exit"
649                  )
650              )
651              assert cm_var is not None
652              self.extend_output([self.create_load(cm_var)])
653              self.extend_output(create_call_function(1, False))
654              self.pop_top()
655  
656          self.extend_output(create_call_function(len(graphargs), False))
657  
658      def load_import_from(self, module_name, object_name) -> None:
659          source = AttrSource(self.tx.import_source(module_name), object_name)
660          # Note: This approach is somewhat aggressive because typically, a source is marked
661          # as a tempvar only when it is used more than once. In this case, we're marking it
662          # as a tempvar without performing that analysis. However, this is a simple solution,
663          # and in many cases, load imports are reused multiple times.
664          self.mark_source_temp(source)
665          self(source)

667      def create_call_function_kw(self, nargs, kw_names, push_null) -> list[Instruction]:
668          if sys.version_info >= (3, 13):
669              output = create_call_function(nargs, push_null)
670              assert output[-1].opname == "CALL"
671              output.insert(-1, self.create_load_const(kw_names))
672              output[-1] = create_instruction("CALL_KW", arg=nargs)
673              return output
674          elif sys.version_info >= (3, 11):
675              output = create_call_function(nargs, push_null)
676              if sys.version_info >= (3, 12):
677                  idx = -1
678                  expected_inst = "CALL"
679              else:
680                  idx = -2
681                  expected_inst = "PRECALL"
682              assert output[idx].opname == expected_inst
683              kw_names_inst = create_instruction("KW_NAMES", argval=kw_names)
684              output.insert(idx, kw_names_inst)
685              return output
686          return [
687              self.create_load_const(kw_names),
688              create_instruction("CALL_FUNCTION_KW", arg=nargs),
689          ]

692          return create_instruction("DELETE_FAST", argval=value)
