107  class VarReader4:
108      ''' Class to read matlab 4 variables '''
109  
110      def __init__(self, file_reader):
111          self.file_reader = file_reader
112          self.mat_stream = file_reader.mat_stream
113          self.dtypes = file_reader.dtypes
114          self.chars_as_strings = file_reader.chars_as_strings
115          self.squeeze_me = file_reader.squeeze_me
116  
117      def read_header(self):
118          ''' Read and return header for variable '''
119          data = read_dtype(self.mat_stream, self.dtypes['header'])
120          name = self.mat_stream.read(int(data['namlen'])).strip(b'\x00')
121          if data['mopt'] < 0 or data['mopt'] > 5000:
122              raise ValueError('Mat 4 mopt wrong format, byteswapping problem?')
123          M, rest = divmod(data['mopt'], 1000)  # order code
124          if M not in (0, 1):
125              warnings.warn(f"We do not support byte ordering '{order_codes[M]}';"
126                            " returned data may be corrupt",
127                            UserWarning, stacklevel=3)
128          O, rest = divmod(rest, 100)  # unused, should be 0
129          if O != 0:
130              raise ValueError('O in MOPT integer should be 0, wrong format?')
131          P, rest = divmod(rest, 10)  # data type code e.g miDOUBLE (see above)
132          T = rest  # matrix type code e.g., mxFULL_CLASS (see above)
133          dims = (data['mrows'], data['ncols'])
134          is_complex = data['imagf'] == 1
135          dtype = self.dtypes[P]
136          return VarHeader4(
137              name,
138              dtype,
139              T,
140              dims,
141              is_complex)

143      def array_from_header(self, hdr, process=True):
144          mclass = hdr.mclass
145          if mclass == mxFULL_CLASS:
146              arr = self.read_full_array(hdr)
147          elif mclass == mxCHAR_CLASS:
148              arr = self.read_char_array(hdr)
149              if process and self.chars_as_strings:
150                  arr = chars_to_strings(arr)
151          elif mclass == mxSPARSE_CLASS:
152              # no current processing (below) makes sense for sparse
153              return self.read_sparse_array(hdr)
154          else:
155              raise TypeError(f'No reader for class code {mclass}')
156          if process and self.squeeze_me:
157              return squeeze_element(arr)
158          return arr

160      def read_sub_array(self, hdr, copy=True):
161          ''' Mat4 read using header `hdr` dtype and dims 
162   
163          Parameters 
164          ---------- 
165          hdr : object 
166             object with attributes ``dtype``, ``dims``. dtype is assumed to be 
167             the correct endianness 
168          copy : bool, optional 
169             copies array before return if True (default True) 
170             (buffer is usually read only) 
171   
172          Returns 
173          ------- 
174          arr : ndarray 
175              of dtype given by `hdr` ``dtype`` and shape given by `hdr` ``dims`` 
176          '''
177          dt = hdr.dtype
178          # Fast product for large (>2GB) arrays.
179          num_bytes = reduce(mul, hdr.dims, np.int64(dt.itemsize))
180          if num_bytes > _MAX_INTP:
181              raise ValueError(
182                  f"Variable '{hdr.name.decode('latin1')}' has byte length "
183                  f"longer than largest possible NumPy array on this platform.")
184          buffer = self.mat_stream.read(num_bytes)
185          if len(buffer) != num_bytes:
186              raise ValueError(
187                  f"Not enough bytes to read matrix "
188                  f"'{hdr.name.decode('latin1')}'; is this a badly-formed file? "
189                  f"Consider listing matrices with `whosmat` and loading named "
190                  f"matrices with `variable_names` kwarg to `loadmat`")
191          arr = np.ndarray(shape=hdr.dims,
192                           dtype=dt,
193                           buffer=buffer,
194                           order='F')
195          if copy:
196              arr = arr.copy()
197          return arr

199      def read_full_array(self, hdr):
200          ''' Full (rather than sparse) matrix getter 
201   
202          Read matrix (array) can be real or complex 
203   
204          Parameters 
205          ---------- 
206          hdr : ``VarHeader4`` instance 
207   
208          Returns 
209          ------- 
210          arr : ndarray 
211              complex array if ``hdr.is_complex`` is True, otherwise a real 
212              numeric array 
213          '''
214          if hdr.is_complex:
215              # avoid array copy to save memory
216              res = self.read_sub_array(hdr, copy=False)
217              res_j = self.read_sub_array(hdr, copy=False)
218              return res + (res_j * 1j)
219          return self.read_sub_array(hdr)
220  
221      def read_char_array(self, hdr):
222          ''' latin-1 text matrix (char matrix) reader 
223   
224          Parameters 
225          ---------- 
226          hdr : ``VarHeader4`` instance 
227   
228          Returns 
229          ------- 
230          arr : ndarray 
231              with dtype 'U1', shape given by `hdr` ``dims`` 
232          '''
233          arr = self.read_sub_array(hdr).astype(np.uint8)
234          S = arr.tobytes().decode('latin-1')
235          return np.ndarray(shape=hdr.dims,
236                            dtype=np.dtype('U1'),
237                            buffer=np.array(S)).copy()
238  
239      def read_sparse_array(self, hdr):
240          ''' Read and return sparse matrix type 
241   
242          Parameters 
243          ---------- 
244          hdr : ``VarHeader4`` instance 
245   
246          Returns 
247          ------- 
248          arr : coo_array 
249              with dtype ``float`` and shape read from the sparse array data 
250   
251          Notes 
252          ----- 
253          MATLAB 4 real sparse arrays are saved in a N+1 by 3 array format, where 
254          N is the number of non-zero values. Column 1 values [0:N] are the 
255          (1-based) row indices of the each non-zero value, column 2 [0:N] are the 
256          column indices, column 3 [0:N] are the (real) values. The last values 
257          [-1,0:2] of the rows, column indices are shape[0] and shape[1] 
258          respectively of the output matrix. The last value for the values column 
259          is a padding 0. mrows and ncols values from the header give the shape of 
260          the stored matrix, here [N+1, 3]. Complex data are saved as a 4 column 
261          matrix, where the fourth column contains the imaginary component; the 
262          last value is again 0. Complex sparse data do *not* have the header 
263          ``imagf`` field set to True; the fact that the data are complex is only 
264          detectable because there are 4 storage columns. 
265          '''
266          res = self.read_sub_array(hdr)
267          tmp = res[:-1,:]
268          # All numbers are float64 in Matlab, but SciPy sparse expects int shape
269          dims = (int(res[-1,0]), int(res[-1,1]))
270          I = np.ascontiguousarray(tmp[:,0],dtype='intc')  # fixes byte order also
271          J = np.ascontiguousarray(tmp[:,1],dtype='intc')
272          I -= 1  # for 1-based indexing
273          J -= 1
274          if res.shape[1] == 3:
275              V = np.ascontiguousarray(tmp[:,2],dtype='float')
276          else:
277              V = np.ascontiguousarray(tmp[:,2],dtype='complex')
278              V.imag = tmp[:,3]
279          return scipy.sparse.coo_array((V,(I,J)), dims)
280  
281      def shape_from_header(self, hdr):
282          '''Read the shape of the array described by the header. 
283          The file position after this call is unspecified. 
284          '''
285          mclass = hdr.mclass
286          if mclass == mxFULL_CLASS:
287              shape = tuple(map(int, hdr.dims))
288          elif mclass == mxCHAR_CLASS:
289              shape = tuple(map(int, hdr.dims))
290              if self.chars_as_strings:
291                  shape = shape[:-1]
292          elif mclass == mxSPARSE_CLASS:
293              dt = hdr.dtype
294              dims = hdr.dims
295  
296              if not (len(dims) == 2 and dims[0] >= 1 and dims[1] >= 1):
297                  return ()
298  
299              # Read only the row and column counts
300              self.mat_stream.seek(dt.itemsize * (dims[0] - 1), 1)
301              rows = np.ndarray(shape=(), dtype=dt,
302                                buffer=self.mat_stream.read(dt.itemsize))
303              self.mat_stream.seek(dt.itemsize * (dims[0] - 1), 1)
304              cols = np.ndarray(shape=(), dtype=dt,
305                                buffer=self.mat_stream.read(dt.itemsize))
306  
307              shape = (int(rows), int(cols))
308          else:
309              raise TypeError(f'No reader for class code {mclass}')
310  
311          if self.squeeze_me:
312              shape = tuple([x for x in shape if x != 1])
313          return shape

316  class MatFile4Reader(MatFileReader):
317      ''' Reader for Mat4 files '''
318      @docfiller
319      def __init__(self, mat_stream, *args, **kwargs):
320          ''' Initialize matlab 4 file reader 
321   
322      %(matstream_arg)s 
323      %(load_args)s 
324          '''
325          super().__init__(mat_stream, *args, **kwargs)
326          self._matrix_reader = None
327  
328      def guess_byte_order(self):
329          self.mat_stream.seek(0)
330          mopt = read_dtype(self.mat_stream, np.dtype('i4'))
331          self.mat_stream.seek(0)
332          if mopt == 0:
333              return '<'
334          if mopt < 0 or mopt > 5000:
335              # Number must have been byteswapped
336              return SYS_LITTLE_ENDIAN and '>' or '<'
337          # Not byteswapped
338          return SYS_LITTLE_ENDIAN and '<' or '>'
339  
340      def initialize_read(self):
341          ''' Run when beginning read of variables 
342   
343          Sets up readers from parameters in `self` 
344          '''
345          self.dtypes = convert_dtypes(mdtypes_template, self.byte_order)
346          self._matrix_reader = VarReader4(self)
347  
348      def read_var_header(self):
349          ''' Read and return header, next position 
350   
351          Parameters 
352          ---------- 
353          None 
354   
355          Returns 
356          ------- 
357          header : object 
358             object that can be passed to self.read_var_array, and that 
359             has attributes ``name`` and ``is_global`` 
360          next_position : int 
361             position in stream of next variable 
362          '''
363          hdr = self._matrix_reader.read_header()
364          # Fast product for large (>2GB) arrays.
365          remaining_bytes = reduce(mul, hdr.dims, np.int64(hdr.dtype.itemsize))
366          if hdr.is_complex and not hdr.mclass == mxSPARSE_CLASS:
367              remaining_bytes *= 2
368          next_position = self.mat_stream.tell() + remaining_bytes
369          return hdr, next_position
370  
371      def read_var_array(self, header, process=True):
372          ''' Read array, given `header` 
373   
374          Parameters 
375          ---------- 
376          header : header object 
377             object with fields defining variable header 
378          process : {True, False}, optional 
379             If True, apply recursive post-processing during loading of array. 
380   
381          Returns 
382          ------- 
383          arr : array 
384             array with post-processing applied or not according to 
385             `process`. 
386          '''
387          return self._matrix_reader.array_from_header(header, process)
388  
389      def get_variables(self, variable_names=None):
390          ''' get variables from stream as dictionary 
391   
392          Parameters 
393          ---------- 
394          variable_names : None or str or sequence of str, optional 
395              variable name, or sequence of variable names to get from Mat file / 
396              file stream. If None, then get all variables in file. 
397          '''
398          if isinstance(variable_names, str):
399              variable_names = [variable_names]
400          elif variable_names is not None:
401              variable_names = list(variable_names)
402          self.mat_stream.seek(0)
403          # set up variable reader
404          self.initialize_read()
405          mdict = {}
406          while not self.end_of_stream():
407              hdr, next_position = self.read_var_header()
408              name = 'None' if hdr.name is None else hdr.name.decode('latin1')
409              if variable_names is not None and name not in variable_names:
410                  self.mat_stream.seek(next_position)
411                  continue
412              mdict[name] = self.read_var_array(hdr)
413              self.mat_stream.seek(next_position)
414              if variable_names is not None:
415                  variable_names.remove(name)
416                  if len(variable_names) == 0:
417                      break
418          return mdict
419  
420      def list_variables(self):
421          ''' list variables from stream '''
422          self.mat_stream.seek(0)
423          # set up variable reader
424          self.initialize_read()
425          vars = []
426          while not self.end_of_stream():
427              hdr, next_position = self.read_var_header()
428              name = 'None' if hdr.name is None else hdr.name.decode('latin1')
429              shape = self._matrix_reader.shape_from_header(hdr)
430              info = mclass_info.get(hdr.mclass, 'unknown')
431              vars.append((name, shape, info))
432  
433              self.mat_stream.seek(next_position)
434          return vars

437  def arr_to_2d(arr, oned_as='row'):
438      ''' Make ``arr`` exactly two dimensional 
439   
440      If `arr` has more than 2 dimensions, raise a ValueError 
441   
442      Parameters 
443      ---------- 
444      arr : array 
445      oned_as : {'row', 'column'}, optional 
446         Whether to reshape 1-D vectors as row vectors or column vectors. 
447         See documentation for ``matdims`` for more detail 
448   
449      Returns 
450      ------- 
451      arr2d : array 
452         2-D version of the array 
453      '''
454      dims = matdims(arr, oned_as)
455      if len(dims) > 2:
456          raise ValueError('Matlab 4 files cannot save arrays with more than '
457                           '2 dimensions')
458      return arr.reshape(dims)

461  class VarWriter4:
462      def __init__(self, file_writer):
463          self.file_stream = file_writer.file_stream
464          self.oned_as = file_writer.oned_as
465  
466      def write_bytes(self, arr):
467          self.file_stream.write(arr.tobytes(order='F'))
468  
469      def write_string(self, s):
470          self.file_stream.write(s)
471  
472      def write_header(self, name, shape, P=miDOUBLE, T=mxFULL_CLASS, imagf=0):
473          ''' Write header for given data options 
474   
475          Parameters 
476          ---------- 
477          name : str 
478              name of variable 
479          shape : sequence 
480              Shape of array as it will be read in matlab 
481          P : int, optional 
482              code for mat4 data type, one of ``miDOUBLE, miSINGLE, miINT32, 
483              miINT16, miUINT16, miUINT8`` 
484          T : int, optional 
485              code for mat4 matrix class, one of ``mxFULL_CLASS, mxCHAR_CLASS, 
486              mxSPARSE_CLASS`` 
487          imagf : int, optional 
488              flag indicating complex 
489          '''
490          header = np.empty((), mdtypes_template['header'])
491          M = not SYS_LITTLE_ENDIAN
492          O = 0
493          header['mopt'] = (M * 1000 +
494                            O * 100 +
495                            P * 10 +
496                            T)
497          header['mrows'] = shape[0]
498          header['ncols'] = shape[1]
499          header['imagf'] = imagf
500          header['namlen'] = len(name) + 1
501          self.write_bytes(header)
502          data = name + '\0'
503          self.write_string(data.encode('latin1'))
504  
505      def write(self, arr, name):
506          ''' Write matrix `arr`, with name `name` 
507   
508          Parameters 
509          ---------- 
510          arr : array_like 
511             array to write 
512          name : str 
513             name in matlab workspace 
514          '''
515          # we need to catch sparse first, because np.asarray returns an
516          # an object array for scipy.sparse
517          if scipy.sparse.issparse(arr):
518              self.write_sparse(arr, name)
519              return
520          arr = np.asarray(arr)
521          dt = arr.dtype
522          if not dt.isnative:
523              arr = arr.astype(dt.newbyteorder('='))
524          dtt = dt.type
525          if dtt is np.object_:
526              raise TypeError('Cannot save object arrays in Mat4')
527          elif dtt is np.void:
528              raise TypeError('Cannot save void type arrays')
529          elif dtt in (np.str_, np.bytes_):
530              self.write_char(arr, name)
531              return
532          self.write_numeric(arr, name)
533  
534      def write_numeric(self, arr, name):
535          arr = arr_to_2d(arr, self.oned_as)
536          imagf = arr.dtype.kind == 'c'
537          try:
538              P = np_to_mtypes[arr.dtype.str[1:]]
539          except KeyError:
540              if imagf:
541                  arr = arr.astype('c128')
542              else:
543                  arr = arr.astype('f8')
544              P = miDOUBLE
545          self.write_header(name,
546                            arr.shape,
547                            P=P,
548                            T=mxFULL_CLASS,
549                            imagf=imagf)
550          if imagf:
551              self.write_bytes(arr.real)
552              self.write_bytes(arr.imag)
553          else:
554              self.write_bytes(arr)
555  
556      def write_char(self, arr, name):
557          if arr.dtype.type == np.str_ and arr.dtype.itemsize != np.dtype('U1').itemsize:
558              arr = arr_to_chars(arr)
559          arr = arr_to_2d(arr, self.oned_as)
560          dims = arr.shape
561          self.write_header(
562              name,
563              dims,
564              P=miUINT8,
565              T=mxCHAR_CLASS)
566          if arr.dtype.kind == 'U':
567              # Recode unicode to latin1
568              n_chars = math.prod(dims)
569              st_arr = np.ndarray(shape=(),
570                                  dtype=arr_dtype_number(arr, n_chars),
571                                  buffer=arr)
572              st = st_arr.item().encode('latin-1')
573              arr = np.ndarray(shape=dims, dtype='S1', buffer=st)
574          self.write_bytes(arr)
575  
576      def write_sparse(self, arr, name):
577          ''' Sparse matrices are 2-D 
578   
579          See docstring for VarReader4.read_sparse_array 
580          '''
581          A = arr.tocoo()  # convert to sparse COO format (ijv)
582          imagf = A.dtype.kind == 'c'
583          ijv = np.zeros((A.nnz + 1, 3+imagf), dtype='f8')
584          ijv[:-1,0] = A.row
585          ijv[:-1,1] = A.col
586          ijv[:-1,0:2] += 1  # 1 based indexing
587          if imagf:
588              ijv[:-1,2] = A.data.real
589              ijv[:-1,3] = A.data.imag
590          else:
591              ijv[:-1,2] = A.data
592          ijv[-1,0:2] = A.shape
593          self.write_header(
594              name,
595              ijv.shape,
596              P=miDOUBLE,
597              T=mxSPARSE_CLASS)
598          self.write_bytes(ijv)

601  class MatFile4Writer:
602      ''' Class for writing matlab 4 format files '''
603      def __init__(self, file_stream, oned_as=None):
604          self.file_stream = file_stream
605          if oned_as is None:
606              oned_as = 'row'
607          self.oned_as = oned_as
608          self._matrix_writer = None
609  
610      def put_variables(self, mdict, write_header=None):
611          ''' Write variables in `mdict` to stream 
612   
613          Parameters 
614          ---------- 
615          mdict : mapping 
616             mapping with method ``items`` return name, contents pairs 
617             where ``name`` which will appeak in the matlab workspace in 
618             file load, and ``contents`` is something writeable to a 
619             matlab file, such as a NumPy array. 
620          write_header : {None, True, False} 
621             If True, then write the matlab file header before writing the 
622             variables. If None (the default) then write the file header 
623             if we are at position 0 in the stream. By setting False 
624             here, and setting the stream position to the end of the file, 
625             you can append variables to a matlab file 
626          '''
627          # there is no header for a matlab 4 mat file, so we ignore the
628          # ``write_header`` input argument. It's there for compatibility
629          # with the matlab 5 version of this method
630          self._matrix_writer = VarWriter4(self)
631          for name, var in mdict.items():
632              self._matrix_writer.write(var, name)
633
