237  def serialize_sym_int(s: Union[int, torch.SymInt]) -> SymInt:
238      if isinstance(s, (torch.SymInt, sympy.Symbol, int)):
239          if symbolic_shapes.is_concrete_int(s):
240              return SymInt.create(as_int=int(s))
241          else:
242              assert isinstance(s, (torch.SymInt, sympy.Symbol))
243              if s.node.hint is None:
244                  return SymInt.create(as_expr=SymExpr(_print_sympy(s)))
245              else:
246                  return SymInt.create(
247                      as_expr=SymExpr(
248                          _print_sympy(s),
249                          hint=SymExprHint.create(as_int=s.node.hint),
250                      )
251                  )
252      else:
253          raise SerializeError(
254              f"SymInt should be either symbol or int, got `{s}` of type `{type(s)}`"
255          )

310      is_parameter = isinstance(fake_tensor, torch.nn.Parameter)

318  def _reconstruct_fake_tensor(
319      serialized_tensor_meta: bytes, is_parameter: bool
320  ) -> FakeTensor:
321      # Deserialize the bytes into a TensorMeta
322      json_tensor_meta = json.loads(serialized_tensor_meta.decode("utf-8"))
323      tensor_meta = _dict_to_dataclass(TensorMeta, json_tensor_meta)
324      # Find the current fake mode
325      assert (
326          _CURRENT_DESERIALIZER is not None
327      ), "Need access to current deserializer state"
328      fake_tensor = _CURRENT_DESERIALIZER.deserialize_tensor_meta(tensor_meta)
329      if is_parameter:
330          fake_tensor = torch.nn.Parameter(fake_tensor)  # type: ignore[assignment]
331      return fake_tensor

334  def serialize_torch_artifact(
335      artifact: Optional[Any], pickle_protocol: int = DEFAULT_PICKLE_PROTOCOL
336  ) -> bytes:
337      if artifact is None:
338          return b""
339  
340      assert (
341          FakeTensor not in copyreg.dispatch_table
342      ), "Refusing to stomp on existing FakeTensor reducer"
343      try:
344          copyreg.pickle(FakeTensor, _reduce_fake_tensor)
345          buffer = io.BytesIO()
346          # This is a workaround for backend's tensor deserialization problem:
347          # unpickleTensor() always create a tensor on the device where it was originally saved
348          # This behavior is bad for multi-gpu training, as we wish to directly load the tensor
349          # on the designated device.
350          # For now, we simply move the tensor to cpu before saving.
351          # TODO: this should be fixed by deserialization instead.
352          torch.save(artifact, buffer, pickle_protocol=pickle_protocol)
353          return buffer.getvalue()
354      finally:
355          del copyreg.dispatch_table[FakeTensor]

358  def deserialize_torch_artifact(
359      serialized: Union[dict[str, Any], tuple[Any, ...], bytes]
360  ):
361      if isinstance(serialized, (dict, tuple)):
362          return serialized
363      if len(serialized) == 0:
364          return {}
365      buffer = io.BytesIO(serialized)
366      buffer.seek(0)
367      # weights_only=False as we want to load custom objects here (e.g. ScriptObject)
368      artifact = torch.load(buffer, weights_only=False)
369      assert isinstance(artifact, (tuple, dict))
370      return artifact

373  def _sympy_int_to_int(val: sympy.Expr, adjust: str) -> Optional[int]:
374      # Convert simple sympy Integers into concrete int
375      if val in (sympy.oo, int_oo):
376          return None
377      if val in (-sympy.oo, -int_oo):
378          return None
379      if isinstance(val, sympy.Integer):
380          return int(val)
381  
382      # TODO: Remove this adjustment when Ed gets rid of fractional ranges
383      log.warning(
384          "Export constraints cannot be non-integer expressions. Found "
385          "type %s, and value %s. We will attempt to %s "
386          "this value.",
387          type(val),
388          val,
389          adjust,
390      )
391  
392      if adjust == "floor":
393          return math.floor(val)
394      elif adjust == "ceil":
395          return math.ceil(val)
396      else:
397          raise RuntimeError(f"Got invalid adjustment {adjust}")

400  def _int_to_sympy_int(val: Optional[int], default) -> sympy.Expr:
401      # Convert concrete int into simple sympy Integers
402      if val is None:
403          return default
404      if val in [-int_oo, int_oo]:
405          return val
406      if val == math.inf:
407          return int_oo
408      if val == -math.inf:
409          return -int_oo
410      return sympy.Integer(val)

460      def __init__(
461          self,
462          graph_signature: ep.ExportGraphSignature,
463          module_call_graph: list[ep.ModuleCallEntry],
464      ):
465          self.graph_state = GraphState()
466          self.graph_signature = graph_signature
467          self.module_call_graph = module_call_graph
468          self.custom_objs: dict[str, torch._C.ScriptObject] = {}
469          self.duplicate_getitem_nodes: dict[str, str] = {}
470          self.treespec_namedtuple_fields: dict[str, NamedTupleDef] = {}

472      @contextmanager
473      def save_graph_state(self):
474          saved = self.graph_state
475          self.graph_state = GraphState()
476          try:
477              yield
478          finally:
479              self.graph_state = saved

481      def handle_placeholder(self, node: torch.fx.Node):
482          assert node.op == "placeholder"
483          val = node.meta["val"]
484          log.debug("[handle_placeholder] %s: %s", node.name, val)
485          if isinstance(val, torch.Tensor):
486              graph_input = Argument.create(
487                  as_tensor=self.serialize_tensor_output(node.name, val)
488              )
489          elif isinstance(val, torch.SymInt):
490              graph_input = Argument.create(
491                  as_sym_int=self.serialize_sym_int_output(node.name, val)
492              )
493          elif isinstance(val, torch.SymFloat):
494              raise AssertionError("SymFloat graph input is not implemented yet.")
495          elif isinstance(val, (int, bool, str, float, type(None))):
496              graph_input = self.serialize_input(val)
497          elif isinstance(val, ep.CustomObjArgument):
498              class_fqn = val.class_fqn
499              graph_input = Argument.create(
500                  as_custom_obj=CustomObjArgument(name=node.name, class_fqn=class_fqn)
501              )
502              self.graph_state.custom_obj_values[
503                  node.name
504              ] = self.serialize_script_obj_meta(val)
505          else:
506              raise AssertionError(f"Unimplemented graph input type: {node.meta['val']}")
507          self.graph_state.inputs.append(graph_input)
508  
509      def handle_output(self, node: torch.fx.Node):
510          assert node.op == "output"
511          assert len(node.args) == 1, "FX.Node's args should have one arg"
512          node_args = node.args[0]
513          log.debug("[handle_output] %s: %s", node.name, node_args)
514          if isinstance(node_args, torch.fx.Node):
515              # For singleton tensor returns
516              self.graph_state.is_single_tensor_return = True
517              self.graph_state.outputs = [self.serialize_input(node_args)]
518          else:
519              assert isinstance(node_args, (tuple, list))
520              self.graph_state.outputs = [self.serialize_input(arg) for arg in node_args]
521  
522      def serialize_operator(self, target) -> str:
523          if isinstance(target, str):
524              return target
525          elif target.__module__.startswith("torch._ops"):
526              # TODO(zhxchen17) Maybe provide a function name helper in FX.
527              # From torch.fx.node._get_qualified_name
528              module = target.__module__.replace("torch._ops", "torch.ops")
529              return f"{module}.{target.__name__}"
530          else:  # TODO(zhxchen17) Don't catch all here.
531              return f"{target.__module__}.{target.__name__}"
532  
533      def handle_call_function(self, node: torch.fx.Node):
534          assert node.op == "call_function"
535          meta_val = node.meta.get("val")
536          log.debug(
537              "[handle_call_function] %s: %s(%s, {%s}) -> %s",
538              node.name,
539              node.target,
540              node.args,
541              node.kwargs,
542              meta_val,
543          )
544  
545          # getitem has been handled in the producer node, skip it here
546          if node.target is operator.getitem:
547              return
548  
549          if node.target in _SYM_OPS or (
550              meta_val is not None
551              and isinstance(meta_val, (torch.SymInt, torch.SymBool, torch.SymFloat))
552          ):
553              assert len(node.kwargs) == 0
554              ex_node = Node(
555                  target=self.serialize_operator(node.target),
556                  inputs=self.serialize_sym_op_inputs(node.target, node.args),
557                  outputs=[self.serialize_output(node.name, meta_val)],
558                  metadata=self.serialize_metadata(node),
559              )
560          elif isinstance(node.target, torch._ops.OpOverload):
561              ex_node = Node(
562                  target=self.serialize_operator(node.target),
563                  inputs=self.serialize_inputs(node.target, node.args, node.kwargs),
564                  outputs=self.serialize_outputs(node),
565                  # TODO: create a new tensor_values here, meta might have faketensor info
566                  metadata=self.serialize_metadata(node),
567              )
568          elif isinstance(node.target, torch._ops.HigherOrderOperator):
569  
570              def _is_hop_single_tensor_return(node) -> bool:
571                  assert isinstance(node.target, torch._ops.HigherOrderOperator)
572                  # HOP schema is not always available, so we look at node.meta["val"]
573                  meta_val = node.meta.get("val", None)
574                  return meta_val is not None and isinstance(meta_val, torch.Tensor)
575  
576              # Special handle serialization for aoti_call_delegate
577              if node.target is torch._higher_order_ops.aoti_call_delegate:
578                  serializable_args = list(node.args)
579  
580                  # AOTI lowered module is not serializable, serialize the aoti_path instead
581                  lowered_module_name: str = node.args[0].name  # type: ignore[assignment, no-untyped-def, union-attr]
582                  assert hasattr(node.graph.owning_module, lowered_module_name)
583                  lowered_module = getattr(node.graph.owning_module, lowered_module_name)  # type: ignore[no-untyped-def]
584                  serializable_args[0] = lowered_module.aoti_path
585  
586                  # AOTI compiled graph module in node.args[0] is stateful, and will fail the verifier check
587                  # Skip serializing original_gm as a workaround
588                  serializable_args[1] = None
589  
590                  serializable_weight_nodes = []
591                  if serializable_args[2] is not None and isinstance(
592                      serializable_args[2], Iterable
593                  ):
594                      for weight_node in serializable_args[2]:
595                          # skip passing custom obj into the weight arg as an hack
596                          # The schema of weight input is a list of Tensors.
597                          # Downstream runtime is not actively consuming the weighs arg for anything meaningful.
598                          if isinstance(weight_node, torch.fx.Node) and isinstance(
599                              weight_node.meta.get("val", None), ep.CustomObjArgument
600                          ):
601                              continue
602                          serializable_weight_nodes.append(weight_node)
603                      serializable_args[2] = serializable_weight_nodes
604  
605                  def serialize_tensor_list_output(node):
606                      meta_val = node.meta.get("val", None)
607                      tensor_args = []
608                      for idx, meta in enumerate(meta_val):
609                          name = self._output_node_name_at_index(node, idx)
610                          tensor_args.append(self.serialize_tensor_output(name, meta))
611                      return [Argument.create(as_tensors=tensor_args)]
612  
613                  ex_node = Node(
614                      target=self.serialize_operator(node.target),
615                      inputs=self.serialize_hoo_inputs(serializable_args, node.kwargs),
616                      outputs=serialize_tensor_list_output(node),
617                      metadata=self.serialize_metadata(node),
618                      is_hop_single_tensor_return=False,
619                  )
620              else:
621                  ex_node = Node(
622                      target=self.serialize_operator(node.target),
623                      inputs=self.serialize_hoo_inputs(node.args, node.kwargs),
624                      outputs=self.serialize_hoo_outputs(node),
625                      metadata=self.serialize_metadata(node),
626                      is_hop_single_tensor_return=_is_hop_single_tensor_return(node),
627                  )
628          elif type(node.target) in _serialization_registry:
629              # Sanity check for unhandled serialization.
630              assert (
631                  type(node.target) in _serialization_registry
632              ), f"{type(node.target)} is not supported in export serialization."
633  
634              handler = _serialization_registry[type(node.target)]
635              namespace = handler.namespace()
636              op_name = handler.to_op_name(node.target)
637              assert isinstance(namespace, str) and isinstance(op_name, str)
638              assert ":" not in namespace and ":" not in op_name
639              ex_node = Node(

673      def serialize_metadata(self, node: torch.fx.Node) -> dict[str, str]:
674          ret = {}
675  
676          if stack_trace := node.meta.get("stack_trace"):
677              ret["stack_trace"] = stack_trace
678  
679          if nn_module_stack := node.meta.get("nn_module_stack"):
680  
681              def export_nn_module_stack(val):
682                  assert isinstance(val, tuple) and len(val) == 2
683                  path, ty = val
684  
685                  assert isinstance(path, str)
686                  assert isinstance(ty, str)
687  
688                  return path + "," + ty
689  
690              # Serialize to "key,orig_path,type_str"
691              nn_module_list = [
692                  f"{k},{export_nn_module_stack(v)}" for k, v in nn_module_stack.items()
693              ]
694              ret["nn_module_stack"] = ST_DELIMITER.join(nn_module_list)
695  
696          if source_fn_st := node.meta.get("source_fn_stack"):
697              source_fn_list = [
698                  f"{source_fn[0]},{self.serialize_operator(source_fn[1])}"
699                  for source_fn in source_fn_st
700              ]
701              ret["source_fn_stack"] = ST_DELIMITER.join(source_fn_list)
702  
703          if torch_fn := node.meta.get("torch_fn"):
704              ret["torch_fn"] = ST_DELIMITER.join(list(torch_fn))
705  
706          if custom := node.meta.get("custom"):
707              try:
708                  ret["custom"] = json.dumps(custom)
709              except Exception as e:
710                  raise SerializeError(
711                      f"Failed to serialize custom metadata for node {node.name} with error {e}"
712                  ) from e
713  
714          return ret

742      def serialize_inputs(
743          self,
744          target: Any,  # torch._ops.OpOverload and other custom operator types.
745          args,
746          kwargs=None,
747      ) -> list[NamedArgument]:
748          schema = None
749          serialized_args = []
750  
751          if isinstance(target, torch._higher_order_ops.torchbind.CallTorchBind):
752              obj = args[0]
753              method = args[1]
754              schema = target.schema(obj, method)
755          else:
756              assert isinstance(
757                  target, (torch._ops.OpOverload, *_registered_extension_types())
758              )
759              schema = _get_schema_from_target(target)
760          assert schema is not None
761          kwargs = kwargs or {}
762  
763          for i, schema_arg in enumerate(schema.arguments):
764              if schema_arg.name in kwargs:
765                  serialized_args.append(
766                      NamedArgument(
767                          name=schema_arg.name,
768                          arg=self.serialize_input(
769                              kwargs[schema_arg.name], schema_arg.type
770                          ),
771                          kind=ArgumentKind.KEYWORD,
772                      )
773                  )
774              elif not schema_arg.kwarg_only and i < len(args):
775                  serialized_args.append(
776                      NamedArgument(
777                          name=schema_arg.name,
778                          arg=self.serialize_input(args[i], schema_arg.type),
779                          kind=ArgumentKind.POSITIONAL,
780                      )
781                  )
782              else:
783                  # We intentionally don't serialize the missing arguments
784                  # with default values
785                  pass
786  
787          return serialized_args

836      # should be torch._C.JitType but that annotation is busted
837      def serialize_input(self, arg, arg_type: Optional[Any] = None) -> Argument:
838          import torch._inductor.ir as inductor_ir
839  
840          inductor_tensor_buffers = (
841              inductor_ir.Buffer,
842              inductor_ir.ReinterpretView,
843          )
844  
845          if isinstance(arg, torch.fx.Node):
846              if arg.op == "get_attr":
847                  assert isinstance(arg.target, str)
848                  attr = getattr(arg.graph.owning_module, arg.target)
849  
850                  if isinstance(attr, torch.Tensor):
851                      raise SerializeError(
852                          "getattr nodes containing tensors should not appear in the graph"
853                      )
854                  elif isinstance(attr, torch.fx.GraphModule):
855                      with self.save_graph_state():
856                          graph = self.serialize_graph(attr)
857                      return Argument.create(
858                          as_graph=GraphArgument(name=arg.target, graph=graph)
859                      )
860                  else:
861                      raise SerializeError(
862                          f"Unsupported getattr attribute {arg.target} with type: {type(attr)}"
863                      )
864              elif self.is_sym_int_arg(arg):
865                  return Argument.create(
866                      as_sym_int=SymIntArgument.create(as_name=arg.name)
867                  )
868              elif self.is_sym_float_arg(arg):
869                  return Argument.create(
870                      as_sym_float=SymFloatArgument.create(as_name=arg.name)
871                  )
872              elif self.is_sym_bool_arg(arg):
873                  return Argument.create(
874                      as_sym_bool=SymBoolArgument.create(as_name=arg.name)
875                  )
876              elif isinstance(arg.meta["val"], ep.CustomObjArgument):
877                  return Argument.create(
878                      as_custom_obj=CustomObjArgument(
879                          name=arg.name, class_fqn=arg.meta["val"].class_fqn
880                      )
881                  )
882              elif arg.name in self.duplicate_getitem_nodes:
883                  dedup_name = self.duplicate_getitem_nodes[arg.name]
884                  return Argument.create(as_tensor=TensorArgument(name=dedup_name))
885              else:
886                  return Argument.create(as_tensor=TensorArgument(name=arg.name))
887          elif isinstance(arg, inductor_tensor_buffers):
888              # Other branches are for arguments in fx node.
889              # This is a special branch for handling buffers (representing tensor arguments)
890              # for inductor's ExternalFallbackNode
891              # export_extern_kernel_node() is using this function to serialize arguments
892              arg_name = arg.get_name()
893              assert arg_name is not None, "Buffer must have valid name"
894              return Argument.create(as_tensor=TensorArgument(name=arg_name))
895          elif isinstance(arg, inductor_ir.TorchBindObject):
896              # This is a special branch for handling TorchBindObject
897              # for inductor's ExternalFallbackNode
898              # export_extern_kernel_node() is using this function to serialize arguments
899              arg_name = arg.get_name()
900              assert arg_name is not None, "Buffer must have valid name"
901              arg_val = arg.get_real_obj()
902              class_fqn = arg_val._type().qualified_name()
903              self.custom_objs[arg_name] = arg_val
904              return Argument.create(as_custom_obj=CustomObjArgument(arg_name, class_fqn))
905          elif isinstance(arg, torch.SymInt):
906              # This is a special branch for handling SymInt args in inductor's
907              # ExternalFallbackNode.
908              # For regular FX graph, SymInt arg should be a fx.Node with
909              # self.is_sym_int_arg(arg) being true
910              return Argument.create(as_sym_int=SymIntArgument.create(as_name=str(arg)))
911          elif isinstance(arg, torch.SymFloat):
912              # This is a special branch for handling SymFloat args in inductor's
913              # ExternalFallbackNode.
914              # For regular FX graph, SymInt arg should be a fx.Node with
915              # self.is_sym_float_arg(arg) being true
916              return Argument.create(
917                  as_sym_float=SymFloatArgument.create(as_name=str(arg))
918              )
919          elif type(arg) is bool:
920              return Argument.create(as_bool=arg)
921          elif type(arg) is str:
922              return Argument.create(as_string=arg)
923          elif type(arg) is int:
924              return Argument.create(as_int=arg)
925          elif type(arg) is float:
926              return Argument.create(as_float=arg)
927          elif arg is None:
928              return Argument.create(as_none=True)
929          elif isinstance(arg, (list, tuple)):
930              if len(arg) == 0:
931                  if arg_type is not None:
932                      if isinstance(arg_type, torch.OptionalType):
933                          arg_type = arg_type.getElementType()  # type: ignore[assignment]
934                      assert isinstance(arg_type, torch.ListType)
935                      elem_type = arg_type.getElementType()
936                      if isinstance(elem_type, torch.OptionalType):
937                          elem_type = elem_type.getElementType()
938  
939                      if isinstance(elem_type, torch.BoolType):
940                          return Argument.create(as_bools=[])
941                      elif isinstance(elem_type, torch.IntType):
942                          return Argument.create(as_ints=[])
943                      elif isinstance(elem_type, torch.FloatType):
944                          return Argument.create(as_floats=[])
945                      elif isinstance(elem_type, torch.StringType):
946                          return Argument.create(as_strings=[])
947                      elif isinstance(elem_type, torch.TensorType):
948                          return Argument.create(as_tensors=[])
949                      else:
950                          # I believe empty symint lists default to ints, but
951                          # please file an issue if this is not the case
952                          raise SerializeError(f"Empty list with type {elem_type} nyi.")
953                  else:
954                      # We could serialize this by default to a tensor list. This
955                      # is needed in the HOO case
956                      log.warning(
957                          "Unsure how to serialize the given empty list, "
958                          "as we don't know what is the type of this argument. "
959                          "Serializing it as a tensor list by default."
960                      )
961                      return Argument.create(as_tensors=[])
962  
963              if all(type(a) is bool for a in arg):
964                  return Argument.create(as_bools=list(arg))
965              elif all(type(a) is int for a in arg):
966                  return Argument.create(as_ints=list(arg))
967              elif all(type(a) is float for a in arg):
968                  return Argument.create(as_floats=list(arg))
969              elif all(type(a) is str for a in arg):
970                  return Argument.create(as_strings=list(arg))
971              elif all(self.is_inductor_sym_int_arg(a) for a in arg):
972                  # This is a special branch for handling SymInt args in inductor's
973                  # ExternalFallbackNode.
974                  # For regular FX graph, SymInt arg should be a fx.Node
975                  values = []
976                  for a in arg:
977                      if isinstance(a, torch.SymInt):
978                          values.append(SymIntArgument.create(as_name=str(a)))
979                      elif type(a) is int:
980                          values.append(SymIntArgument.create(as_int=a))
981                  return Argument.create(as_sym_ints=values)
982              elif all(isinstance(a, torch.SymFloat) for a in arg):
983                  return Argument.create(
984                      as_sym_floats=[SymFloatArgument.create(as_name=str(a)) for a in arg]
985                  )
986              elif all(self.is_sym_int_arg(a) for a in arg):
987                  # list of sym_ints
988                  values = []
989                  for a in arg:
990                      if isinstance(a, torch.fx.Node):
991                          values.append(SymIntArgument.create(as_name=a.name))
992                      elif type(a) is int:
993                          values.append(SymIntArgument.create(as_int=a))
994                  return Argument.create(as_sym_ints=values)
995              elif all(self.is_sym_float_arg(a) for a in arg):
996                  # list of sym_float
997                  values = []
998                  for a in arg:
999                      if isinstance(a, torch.fx.Node):
1000                         values.append(SymFloatArgument.create(as_name=a.name))
1001                     elif isinstance(a, float):
1002                         values.append(SymFloatArgument.create(as_float=a))
1003                 return Argument.create(as_sym_floats=values)
1004             elif all(self.is_sym_bool_arg(a) for a in arg):
1005                 # list of sym_bools
1006                 values = []
1007                 for a in arg:
1008                     if isinstance(a, torch.fx.Node):
1009                         values.append(SymBoolArgument.create(as_name=a.name))
1010                     elif isinstance(a, bool):
1011                         values.append(SymBoolArgument.create(as_bool=a))
1012                 return Argument.create(as_sym_bools=values)
1013             elif all(isinstance(a, torch.fx.Node) for a in arg):
1014                 # list of tensors
1015                 arguments = []
1016                 for a in arg:
1017                     if a.op == "get_attr":
1018                         raise SerializeError(
1019                             "getattr nodes containing tensors should not appear in the graph"
1020                         )
1021                     arguments.append(TensorArgument(name=a.name))
1022                 return Argument.create(as_tensors=arguments)
1023             elif all(isinstance(a, (torch.fx.Node, type(None))) for a in arg):
1024                 # list of optional tensors
1025                 def serialize_optional_tensor_args(a):
1026                     if a is None:
1027                         return OptionalTensorArgument.create(as_none=True)
1028                     elif isinstance(a, torch.fx.Node):
1029                         return OptionalTensorArgument.create(
1030                             as_tensor=TensorArgument(name=a.name)
1031                         )
1032                     else:
1033                         raise SerializeError(f"Unsupported list/tuple argument: {a}")
1034 
1035                 return Argument.create(
1036                     as_optional_tensors=list(map(serialize_optional_tensor_args, arg))
1037                 )
1038             elif all(isinstance(a, inductor_tensor_buffers) for a in arg):
1039                 # list of inductor buffers
1040                 return Argument.create(
1041                     as_tensors=[TensorArgument(name=a.get_name()) for a in arg],
1042                 )
1043             elif all(
1044                 isinstance(a, (*inductor_tensor_buffers, type(None))) for a in arg
1045             ):
1046                 # list of inductor buffers as optional tensors
1047                 def serialize_optional_tensor_args(a):
1048                     if a is None:
1049                         return OptionalTensorArgument.create(as_none=True)
1050                     elif isinstance(a, inductor_tensor_buffers):
1051                         return OptionalTensorArgument.create(
1052                             as_tensor=TensorArgument(name=a.get_name())
1053                         )
1054                     else:
1055                         raise SerializeError(f"Unsupported list/tuple argument: {a}")
1056 
1057                 return Argument.create(
1058                     as_optional_tensors=list(map(serialize_optional_tensor_args, arg))
1059                 )
1060             else:
1061                 raise SerializeError(
1062                     f"Unsupported list/tuple argument type: {[type(a) for a in arg]}"
1063                 )
1064         elif isinstance(arg, torch.dtype):
1065             return Argument.create(as_scalar_type=_TORCH_TO_SERIALIZE_DTYPE[arg])
1066         elif isinstance(arg, torch.device):
1067             return Argument.create(as_device=Device(type=arg.type, index=arg.index))
1068         elif isinstance(arg, torch.memory_format):
1069             return Argument.create(
1070                 as_memory_format=_TORCH_TO_SERIALIZE_MEMORY_FORMAT[arg]
1071             )
1072         elif isinstance(arg, torch.layout):
1073             return Argument.create(as_layout=_TORCH_TO_SERIALIZE_LAYOUT[arg])
1074         elif isinstance(arg, torch._C.ScriptObject):
1075             if not (
1076                 arg._has_method("__getstate__")  # type: ignore[attr-defined]
1077                 and arg._has_method("__setstate__")  # type: ignore[attr-defined]
1078             ):
1079                 raise SerializeError(
1080                     f"Unable to serialize custom class {arg}. Please define "
1081                     "serialization methods via def_pickle()."
1082                 )
1083             # Custom objects through torchind are serializable with pickle,
1084             # through implementing the .def_pickle function.  This should result
1085             # in the object containing a __getstate__ and __setstate__
1086             # serialize/deserialize function.
1087             custom_obj_name = f"_custom_obj_{len(self.custom_objs)}"
1088             self.custom_objs[custom_obj_name] = arg
1089             class_fqn = arg._type().qualified_name()  # type: ignore[attr-defined]
1090             return Argument.create(
1091                 as_custom_obj=CustomObjArgument(custom_obj_name, class_fqn)
1092             )
1093         elif isinstance(arg, (torch._ops.OpOverload, torch._ops.HigherOrderOperator)):
1094             return Argument.create(as_operator=self.serialize_operator(arg))
1095         else:
1096             raise SerializeError(
1097                 f"Unsupported argument type: {type(arg)} with schema arg_type {arg_type}"
1098             )

1343     def serialize_outputs(self, node: torch.fx.Node) -> list[Argument]:
1344         """For a given node, return the dataclass representing its output values. 
1345  
1346         [NOTE: Multiple outputs] We handle aggregates differently than FX. For 
1347         FX, it looks like: 
1348  
1349             x = call_function("multiple_return", ...) 
1350             element0 = call_function(getitem, x, 0) 
1351             foo = call_function("use_output", element0) 
1352  
1353         We do not want the intermediate `getitem` call, so our serialized thing looks like: 
1354  
1355             element0, element1, element2 = call_function("multiple_return", ...) 
1356             foo = call_function("use_output", element0) 
1357  
1358         We want names to be consistent across these two schemes, so that we can 
1359         mostly reuse the names coming from FX. This function computes a mapping from 
1360         the FX representation to our representation, preserving the names. 
1361         """
1362 
1363         def _is_single_tensor_list_return(target: Any) -> bool:
1364             schema = _get_schema_from_target(target)
1365             returns = schema.returns
1366 
1367             if len(returns) != 1:
1368                 return False
1369             return_type = returns[0].real_type
1370             return isinstance(return_type, torch.ListType) and isinstance(
1371                 return_type.getElementType(), torch.TensorType
1372             )
1373 
1374         assert node.op == "call_function" and isinstance(
1375             node.target, (torch._ops.OpOverload, *_registered_extension_types())
1376         )
1377 
1378         schema = _get_schema_from_target(node.target)
1379         returns = schema.returns
1380 
1381         if len(returns) == 0:
1382             return []
1383 
1384         meta_val = node.meta["val"]
1385 
1386         # Check single value return
1387         if _is_single_tensor_list_return(node.target):
1388             # e.g "-> Tensor[]"
1389             tensor_args = []
1390             for idx, meta in enumerate(meta_val):
1391                 name = self._output_node_name_at_index(node, idx)
1392                 tensor_args.append(self.serialize_tensor_output(name, meta))
1393             return [Argument.create(as_tensors=tensor_args)]
1394         elif len(returns) == 1:
1395             return [self.serialize_output(node.name, meta_val)]
1396 
1397         # There are a two possibilities at this point:
1398         # - This operator returns a tuple of Tensors, e.g. "-> (Tensor, Tensor)"
1399         # - This operator returns a tuple of mixed of Tensor and Tensors, e.g. "-> (Tensor, Tensor[])"
1400         #
1401         # Either way, start by gathering a list of TensorArguments with the correct names.
1402         # For consistent naming with FX, consult the downstream `getitem` node and
1403         # make sure our outputs have the same name.
1404 
1405         output_arguments = []
1406         for idx, (meta, return_schema) in enumerate(zip(meta_val, returns)):
1407             if meta is None:
1408                 assert isinstance(
1409                     return_schema.real_type, (torch.OptionalType, torch.TensorType)
1410                 )
1411                 # When the return type is annoated as Tensor type, the op can also return an
1412                 # undefined Tensor which will be implicitly converted to None in Python.
1413                 output_arguments.append(Argument.create(as_none=True))
1414             elif isinstance(meta, FakeTensor):
1415                 assert isinstance(
1416                     return_schema.real_type, (torch.OptionalType, torch.TensorType)
1417                 )
1418                 name = self._output_node_name_at_index(node, idx)
1419                 output_arguments.append(self.serialize_output(name, meta))
1420             elif isinstance(meta, list):
1421                 # for List[Tensor] return type
1422                 assert isinstance(
1423                     return_schema.real_type, torch.ListType
1424                 ) and isinstance(
1425                     return_schema.real_type.getElementType(), torch.TensorType
1426                 )
1427                 user_node = self._output_node_at_index(node, idx)
1428                 assert user_node is not None
1429 
1430                 args = []
1431                 for i, m in enumerate(meta):
1432                     if m is None:
1433                         continue
1434                     sub_user_node_name = self._output_node_name_at_index(user_node, i)
1435                     args.append(self.serialize_tensor_output(sub_user_node_name, m))
1436                 output_arguments.append(Argument.create(as_tensors=args))
1437             elif isinstance(meta, (int, SymInt, float, SymFloat)):
1438                 user_node_name = self._output_node_name_at_index(node, idx)
1439                 output_arguments.append(self.serialize_output(user_node_name, meta))
1440             else:
1441                 raise ValueError(
1442                     f"Unhandled output type {type(meta)} from node {node.format_node()}"

1444

1990     def deserialize_node(self, serialized_node: Node, target: Callable) -> None:
1991         def _is_single_tensor_return(target) -> bool:
1992             schema = _get_schema_from_target(target)
1993             returns = schema.returns
1994             return len(returns) == 1 and isinstance(
1995                 returns[0].real_type, torch.TensorType

1997 
1998         if (
1999             target in _SYM_OPS
2000             or target
2001             == torch.ops.aten.item.default  # this can produce either SymInt or SymBool

2009         elif isinstance(target, torch._ops.HigherOrderOperator):

2047                 serialized_node.outputs[0].as_tensor.name
2048                 if _is_single_tensor_return(target)
2049                 else None  # FX will generate a name for us.
2050             )
2051             args, kwargs = self.deserialize_inputs(target, serialized_node)

2053                 "call_function", target, args, kwargs, name

2060                     + "If it's a custom op/custom triton op, this is usally because the custom op is not registered"
2061                     + " when deserializing. Please import the custom op to register it before deserializing."
2062                     + " Otherwise, please file an issue on github."
2063                 )
2064                 if isinstance(target, str)
2065                 else ""

2069                 + f" Unsupported target type for node {serialized_node}: {type(target)}."

2072         fx_node.meta.update(self.deserialize_metadata(serialized_node.metadata))

2074             "[deserialize_node] %s: %s(%s, {%s}) -> %s",
2075             fx_node.name,
2076             fx_node.target,
2077             fx_node.args,
2078             fx_node.kwargs,
2079             fx_node.meta.get("val"),
2080         )
2081         if (
2082             fx_node.op not in ["placeholder", "output"]
2083             and "nn_module_stack" not in fx_node.meta
