42   def linprog_verbose_callback(res):
43       """ 
44       A sample callback function demonstrating the linprog callback interface. 
45       This callback produces detailed output to sys.stdout before each iteration 
46       and after the final iteration of the simplex algorithm. 
47    
48       Parameters 
49       ---------- 
50       res : A `scipy.optimize.OptimizeResult` consisting of the following fields: 
51    
52           x : 1-D array 
53               The independent variable vector which optimizes the linear 
54               programming problem. 
55           fun : float 
56               Value of the objective function. 
57           success : bool 
58               True if the algorithm succeeded in finding an optimal solution. 
59           slack : 1-D array 
60               The values of the slack variables. Each slack variable corresponds 
61               to an inequality constraint. If the slack is zero, then the 
62               corresponding constraint is active. 
63           con : 1-D array 
64               The (nominally zero) residuals of the equality constraints, that is, 
65               ``b - A_eq @ x`` 
66           phase : int 
67               The phase of the optimization being executed. In phase 1 a basic 
68               feasible solution is sought and the T has an additional row 
69               representing an alternate objective function. 
70           status : int 
71               An integer representing the exit status of the optimization: 
72    
73               ``0`` : Optimization terminated successfully 
74    
75               ``1`` : Iteration limit reached 
76    
77               ``2`` : Problem appears to be infeasible 
78    
79               ``3`` : Problem appears to be unbounded 
80    
81               ``4`` : Serious numerical difficulties encountered 
82    
83           nit : int 
84               The number of iterations performed. 
85           message : str 
86               A string descriptor of the exit status of the optimization. 
87       """
88       x = res['x']
89       fun = res['fun']
90       phase = res['phase']
91       status = res['status']
92       nit = res['nit']
93       message = res['message']
94       complete = res['complete']
95   
96       saved_printoptions = np.get_printoptions()
97       np.set_printoptions(linewidth=500,
98                           formatter={'float': lambda x: f"{x: 12.4f}"})
99       if status:
100          print('--------- Simplex Early Exit -------\n')
101          print(f'The simplex method exited early with status {status:d}')
102          print(message)
103      elif complete:
104          print('--------- Simplex Complete --------\n')
105          print(f'Iterations required: {nit}')
106      else:
107          print(f'--------- Iteration {nit:d}  ---------\n')
108  
109      if nit > 0:
110          if phase == 1:
111              print('Current Pseudo-Objective Value:')
112          else:
113              print('Current Objective Value:')
114          print('f = ', fun)
115          print()
116          print('Current Solution Vector:')
117          print('x = ', x)
118          print()
119  
120      np.set_printoptions(**saved_printoptions)

178  def linprog(c, A_ub=None, b_ub=None, A_eq=None, b_eq=None,
179              bounds=(0, None), method='highs', callback=None,
180              options=None, x0=None, integrality=None):
181      r""" 
182      Linear programming: minimize a linear objective function subject to linear 
183      equality and inequality constraints. 
184   
185      Linear programming solves problems of the following form: 
186   
187      .. math:: 
188   
189          \min_x \ & c^T x \\ 
190          \mbox{such that} \ & A_{ub} x \leq b_{ub},\\ 
191          & A_{eq} x = b_{eq},\\ 
192          & l \leq x \leq u , 
193   
194      where :math:`x` is a vector of decision variables; :math:`c`, 
195      :math:`b_{ub}`, :math:`b_{eq}`, :math:`l`, and :math:`u` are vectors; and 
196      :math:`A_{ub}` and :math:`A_{eq}` are matrices. 
197   
198      Alternatively, that's: 
199   
200      - minimize :: 
201   
202          c @ x 
203   
204      - such that :: 
205   
206          A_ub @ x <= b_ub 
207          A_eq @ x == b_eq 
208          lb <= x <= ub 
209   
210      Note that by default ``lb = 0`` and ``ub = None``. Other bounds can be 
211      specified with ``bounds``. 
212   
213      Parameters 
214      ---------- 
215      c : 1-D array 
216          The coefficients of the linear objective function to be minimized. 
217      A_ub : 2-D array, optional 
218          The inequality constraint matrix. Each row of ``A_ub`` specifies the 
219          coefficients of a linear inequality constraint on ``x``. 
220      b_ub : 1-D array, optional 
221          The inequality constraint vector. Each element represents an 
222          upper bound on the corresponding value of ``A_ub @ x``. 
223      A_eq : 2-D array, optional 
224          The equality constraint matrix. Each row of ``A_eq`` specifies the 
225          coefficients of a linear equality constraint on ``x``. 
226      b_eq : 1-D array, optional 
227          The equality constraint vector. Each element of ``A_eq @ x`` must equal 
228          the corresponding element of ``b_eq``. 
229      bounds : sequence, optional 
230          A sequence of ``(min, max)`` pairs for each element in ``x``, defining 
231          the minimum and maximum values of that decision variable. 
232          If a single tuple ``(min, max)`` is provided, then ``min`` and ``max`` 
233          will serve as bounds for all decision variables. 
234          Use ``None`` to indicate that there is no bound. For instance, the 
235          default bound ``(0, None)`` means that all decision variables are 
236          non-negative, and the pair ``(None, None)`` means no bounds at all, 
237          i.e. all variables are allowed to be any real. 
238      method : str, optional 
239          The algorithm used to solve the standard form problem. 
240          The following are supported. 
241   
242          - :ref:`'highs' <optimize.linprog-highs>` (default) 
243          - :ref:`'highs-ds' <optimize.linprog-highs-ds>` 
244          - :ref:`'highs-ipm' <optimize.linprog-highs-ipm>` 
245          - :ref:`'interior-point' <optimize.linprog-interior-point>` (legacy) 
246          - :ref:`'revised simplex' <optimize.linprog-revised_simplex>` (legacy) 
247          - :ref:`'simplex' <optimize.linprog-simplex>` (legacy) 
248   
249          The legacy methods are deprecated and will be removed in SciPy 1.11.0. 
250      callback : callable, optional 
251          If a callback function is provided, it will be called at least once per 
252          iteration of the algorithm. The callback function must accept a single 
253          `scipy.optimize.OptimizeResult` consisting of the following fields: 
254   
255          x : 1-D array 
256              The current solution vector. 
257          fun : float 
258              The current value of the objective function ``c @ x``. 
259          success : bool 
260              ``True`` when the algorithm has completed successfully. 
261          slack : 1-D array 
262              The (nominally positive) values of the slack, 
263              ``b_ub - A_ub @ x``. 
264          con : 1-D array 
265              The (nominally zero) residuals of the equality constraints, 
266              ``b_eq - A_eq @ x``. 
267          phase : int 
268              The phase of the algorithm being executed. 
269          status : int 
270              An integer representing the status of the algorithm. 
271   
272              ``0`` : Optimization proceeding nominally. 
273   
274              ``1`` : Iteration limit reached. 
275   
276              ``2`` : Problem appears to be infeasible. 
277   
278              ``3`` : Problem appears to be unbounded. 
279   
280              ``4`` : Numerical difficulties encountered. 
281   
282          nit : int 
283              The current iteration number. 
284          message : str 
285              A string descriptor of the algorithm status. 
286   
287          Callback functions are not currently supported by the HiGHS methods. 
288   
289      options : dict, optional 
290          A dictionary of solver options. All methods accept the following 
291          options: 
292   
293          maxiter : int 
294              Maximum number of iterations to perform. 
295              Default: see method-specific documentation. 
296          disp : bool 
297              Set to ``True`` to print convergence messages. 
298              Default: ``False``. 
299          presolve : bool 
300              Set to ``False`` to disable automatic presolve. 
301              Default: ``True``. 
302   
303          All methods except the HiGHS solvers also accept: 
304   
305          tol : float 
306              A tolerance which determines when a residual is "close enough" to 
307              zero to be considered exactly zero. 
308          autoscale : bool 
309              Set to ``True`` to automatically perform equilibration. 
310              Consider using this option if the numerical values in the 
311              constraints are separated by several orders of magnitude. 
312              Default: ``False``. 
313          rr : bool 
314              Set to ``False`` to disable automatic redundancy removal. 
315              Default: ``True``. 
316          rr_method : string 
317              Method used to identify and remove redundant rows from the 
318              equality constraint matrix after presolve. For problems with 
319              dense input, the available methods for redundancy removal are: 
320   
321              ``SVD``: 
322                  Repeatedly performs singular value decomposition on 
323                  the matrix, detecting redundant rows based on nonzeros 
324                  in the left singular vectors that correspond with 
325                  zero singular values. May be fast when the matrix is 
326                  nearly full rank. 
327              ``pivot``: 
328                  Uses the algorithm presented in [5]_ to identify 
329                  redundant rows. 
330              ``ID``: 
331                  Uses a randomized interpolative decomposition. 
332                  Identifies columns of the matrix transpose not used in 
333                  a full-rank interpolative decomposition of the matrix. 
334              ``None``: 
335                  Uses ``svd`` if the matrix is nearly full rank, that is, 
336                  the difference between the matrix rank and the number 
337                  of rows is less than five. If not, uses ``pivot``. The 
338                  behavior of this default is subject to change without 
339                  prior notice. 
340   
341              Default: None. 
342              For problems with sparse input, this option is ignored, and the 
343              pivot-based algorithm presented in [5]_ is used. 
344   
345          For method-specific options, see 
346          :func:`show_options('linprog') <show_options>`. 
347   
348      x0 : 1-D array, optional 
349          Guess values of the decision variables, which will be refined by 
350          the optimization algorithm. This argument is currently used only by the 
351          :ref:`'revised simplex' <optimize.linprog-revised_simplex>` method, 
352          and can only be used if `x0` represents a basic feasible solution. 
353   
354      integrality : 1-D array or int, optional 
355          Indicates the type of integrality constraint on each decision variable. 
356   
357          ``0`` : Continuous variable; no integrality constraint. 
358   
359          ``1`` : Integer variable; decision variable must be an integer 
360          within `bounds`. 
361   
362          ``2`` : Semi-continuous variable; decision variable must be within 
363          `bounds` or take value ``0``. 
364   
365          ``3`` : Semi-integer variable; decision variable must be an integer 
366          within `bounds` or take value ``0``. 
367   
368          By default, all variables are continuous. 
369   
370          For mixed integrality constraints, supply an array of shape ``c.shape``. 
371          To infer a constraint on each decision variable from shorter inputs, 
372          the argument will be broadcast to ``c.shape`` using `numpy.broadcast_to`. 
373   
374          This argument is currently used only by the 
375          :ref:`'highs' <optimize.linprog-highs>` method and is ignored otherwise. 
376   
377      Returns 
378      ------- 
379      res : OptimizeResult 
380          A :class:`scipy.optimize.OptimizeResult` consisting of the fields 
381          below. Note that the return types of the fields may depend on whether 
382          the optimization was successful, therefore it is recommended to check 
383          `OptimizeResult.status` before relying on the other fields: 
384   
385          x : 1-D array 
386              The values of the decision variables that minimizes the 
387              objective function while satisfying the constraints. 
388          fun : float 
389              The optimal value of the objective function ``c @ x``. 
390          slack : 1-D array 
391              The (nominally positive) values of the slack variables, 
392              ``b_ub - A_ub @ x``. 
393          con : 1-D array 
394              The (nominally zero) residuals of the equality constraints, 
395              ``b_eq - A_eq @ x``. 
396          success : bool 
397              ``True`` when the algorithm succeeds in finding an optimal 
398              solution. 
399          status : int 
400              An integer representing the exit status of the algorithm. 
401   
402              ``0`` : Optimization terminated successfully. 
403   
404              ``1`` : Iteration limit reached. 
405   
406              ``2`` : Problem appears to be infeasible. 
407   
408              ``3`` : Problem appears to be unbounded. 
409   
410              ``4`` : Numerical difficulties encountered. 
411   
412          nit : int 
413              The total number of iterations performed in all phases. 
414          message : str 
415              A string descriptor of the exit status of the algorithm. 
416

418      -------- 
419      show_options : Additional options accepted by the solvers.

421      Notes 
422      ----- 
423      This section describes the available solvers that can be selected by the 
424      'method' parameter. 
425   
426      :ref:`'highs-ds' <optimize.linprog-highs-ds>`, and 
427      :ref:`'highs-ipm' <optimize.linprog-highs-ipm>` are interfaces to the 
428      HiGHS simplex and interior-point method solvers [13]_, respectively. 
429      :ref:`'highs' <optimize.linprog-highs>` (default) chooses between 
430      the two automatically. These are the fastest linear 
431      programming solvers in SciPy, especially for large, sparse problems; 
432      which of these two is faster is problem-dependent. 
433      The other solvers are legacy methods and will be removed when `callback` is 
434      supported by the HiGHS methods. 
435   
436      Method :ref:`'highs-ds' <optimize.linprog-highs-ds>`, is a wrapper of the C++ high 
437      performance dual revised simplex implementation (HSOL) [13]_, [14]_. 
438      Method :ref:`'highs-ipm' <optimize.linprog-highs-ipm>` is a wrapper of a C++ 
439      implementation of an **i**\ nterior-\ **p**\ oint **m**\ ethod [13]_; it 
440      features a crossover routine, so it is as accurate as a simplex solver. 
441      Method :ref:`'highs' <optimize.linprog-highs>` chooses between the two 
442      automatically. 
443      For new code involving `linprog`, we recommend explicitly choosing one of 
444      these three method values. 
445   
446      .. versionadded:: 1.6.0 
447   
448      Method :ref:`'interior-point' <optimize.linprog-interior-point>` 
449      uses the primal-dual path following algorithm 
450      as outlined in [4]_. This algorithm supports sparse constraint matrices and 
451      is typically faster than the simplex methods, especially for large, sparse 
452      problems. Note, however, that the solution returned may be slightly less 
453      accurate than those of the simplex methods and will not, in general, 
454      correspond with a vertex of the polytope defined by the constraints. 
455   
456      .. versionadded:: 1.0.0 
457   
458      Method :ref:`'revised simplex' <optimize.linprog-revised_simplex>` 
459      uses the revised simplex method as described in 
460      [9]_, except that a factorization [11]_ of the basis matrix, rather than 
461      its inverse, is efficiently maintained and used to solve the linear systems 
462      at each iteration of the algorithm. 
463   
464      .. versionadded:: 1.3.0 
465   
466      Method :ref:`'simplex' <optimize.linprog-simplex>` uses a traditional, 
467      full-tableau implementation of 
468      Dantzig's simplex algorithm [1]_, [2]_ (*not* the 
469      Nelder-Mead simplex). This algorithm is included for backwards 
470      compatibility and educational purposes. 
471   
472      .. versionadded:: 0.15.0 
473   
474      Before applying :ref:`'interior-point' <optimize.linprog-interior-point>`, 
475      :ref:`'revised simplex' <optimize.linprog-revised_simplex>`, or 
476      :ref:`'simplex' <optimize.linprog-simplex>`, 
477      a presolve procedure based on [8]_ attempts 
478      to identify trivial infeasibilities, trivial unboundedness, and potential 
479      problem simplifications. Specifically, it checks for: 
480   
481      - rows of zeros in ``A_eq`` or ``A_ub``, representing trivial constraints; 
482      - columns of zeros in ``A_eq`` `and` ``A_ub``, representing unconstrained 
483        variables; 
484      - column singletons in ``A_eq``, representing fixed variables; and 
485      - column singletons in ``A_ub``, representing simple bounds. 
486   
487      If presolve reveals that the problem is unbounded (e.g. an unconstrained 
488      and unbounded variable has negative cost) or infeasible (e.g., a row of 
489      zeros in ``A_eq`` corresponds with a nonzero in ``b_eq``), the solver 
490      terminates with the appropriate status code. Note that presolve terminates 
491      as soon as any sign of unboundedness is detected; consequently, a problem 
492      may be reported as unbounded when in reality the problem is infeasible 
493      (but infeasibility has not been detected yet). Therefore, if it is 
494      important to know whether the problem is actually infeasible, solve the 
495      problem again with option ``presolve=False``. 
496   
497      If neither infeasibility nor unboundedness are detected in a single pass 
498      of the presolve, bounds are tightened where possible and fixed 
499      variables are removed from the problem. Then, linearly dependent rows 
500      of the ``A_eq`` matrix are removed, (unless they represent an 
501      infeasibility) to avoid numerical difficulties in the primary solve 
502      routine. Note that rows that are nearly linearly dependent (within a 
503      prescribed tolerance) may also be removed, which can change the optimal 
504      solution in rare cases. If this is a concern, eliminate redundancy from 
505      your problem formulation and run with option ``rr=False`` or 
506      ``presolve=False``. 
507   
508      Several potential improvements can be made here: additional presolve 
509      checks outlined in [8]_ should be implemented, the presolve routine should 
510      be run multiple times (until no further simplifications can be made), and 
511      more of the efficiency improvements from [5]_ should be implemented in the 
512      redundancy removal routines. 
513   
514      After presolve, the problem is transformed to standard form by converting 
515      the (tightened) simple bounds to upper bound constraints, introducing 
516      non-negative slack variables for inequality constraints, and expressing 
517      unbounded variables as the difference between two non-negative variables. 
518      Optionally, the problem is automatically scaled via equilibration [12]_. 
519      The selected algorithm solves the standard form problem, and a 
520      postprocessing routine converts the result to a solution to the original 
521      problem. 
522

564      Examples 
565      -------- 
566      Consider the following problem: 
567   
568      .. math:: 
569   
570          \min_{x_0, x_1} \ -x_0 + 4x_1 & \\ 
571          \mbox{such that} \ -3x_0 + x_1 & \leq 6,\\ 
572          -x_0 - 2x_1 & \geq -4,\\ 
573          x_1 & \geq -3. 
574   
575      The problem is not presented in the form accepted by `linprog`. This is 
576      easily remedied by converting the "greater than" inequality 
577      constraint to a "less than" inequality constraint by 
578      multiplying both sides by a factor of :math:`-1`. Note also that the last 
579      constraint is really the simple bound :math:`-3 \leq x_1 \leq \infty`. 
580      Finally, since there are no bounds on :math:`x_0`, we must explicitly 
581      specify the bounds :math:`-\infty \leq x_0 \leq \infty`, as the 
582      default is for variables to be non-negative. After collecting coeffecients 
583      into arrays and tuples, the input for this problem is: 
584   
585      >>> from scipy.optimize import linprog 
586      >>> c = [-1, 4] 
587      >>> A = [[-3, 1], [1, 2]] 
588      >>> b = [6, 4] 
589      >>> x0_bounds = (None, None) 
590      >>> x1_bounds = (-3, None) 
591      >>> res = linprog(c, A_ub=A, b_ub=b, bounds=[x0_bounds, x1_bounds]) 
592      >>> res.fun 
593      -22.0 
594      >>> res.x 
595      array([10., -3.]) 
596      >>> res.message 
597      'Optimization terminated successfully. (HiGHS Status 7: Optimal)' 
598   
599      The marginals (AKA dual values / shadow prices / Lagrange multipliers) 
600      and residuals (slacks) are also available. 
601   
602      >>> res.ineqlin 
603        residual: [ 3.900e+01  0.000e+00] 
604       marginals: [-0.000e+00 -1.000e+00] 
605   
606      For example, because the marginal associated with the second inequality 
607      constraint is -1, we expect the optimal value of the objective function 
608      to decrease by ``eps`` if we add a small amount ``eps`` to the right hand 
609      side of the second inequality constraint: 
610   
611      >>> eps = 0.05 
612      >>> b[1] += eps 
613      >>> linprog(c, A_ub=A, b_ub=b, bounds=[x0_bounds, x1_bounds]).fun 
614      -22.05 
615   
616      Also, because the residual on the first inequality constraint is 39, we 
617      can decrease the right hand side of the first constraint by 39 without 
618      affecting the optimal solution. 
619   
620      >>> b = [6, 4]  # reset to original values 
621      >>> b[0] -= 39 
622      >>> linprog(c, A_ub=A, b_ub=b, bounds=[x0_bounds, x1_bounds]).fun 
623      -22.0 
624

627      meth = method.lower()
628      methods = {"highs", "highs-ds", "highs-ipm",
629                 "simplex", "revised simplex", "interior-point"}
630  
631      if meth not in methods:
632          raise ValueError(f"Unknown solver '{method}'")
633  
634      if x0 is not None and meth != "revised simplex":
635          warning_message = "x0 is used only when method is 'revised simplex'. "
636          warn(warning_message, OptimizeWarning, stacklevel=2)
637  
638      if np.any(integrality) and not meth == "highs":
639          integrality = None
640          warning_message = ("Only `method='highs'` supports integer "
641                             "constraints. Ignoring `integrality`.")
642          warn(warning_message, OptimizeWarning, stacklevel=2)
643      elif np.any(integrality):
644          integrality = np.broadcast_to(integrality, np.shape(c))
645      else:
646          integrality = None
647  
648      lp = _LPProblem(c, A_ub, b_ub, A_eq, b_eq, bounds, x0, integrality)
649      lp, solver_options = _parse_linprog(lp, options, meth)
650      tol = solver_options.get('tol', 1e-9)
651  
652      # Give unmodified problem to HiGHS
653      if meth.startswith('highs'):
654          if callback is not None:
655              raise NotImplementedError("HiGHS solvers do not support the "
656                                        "callback interface.")
657          highs_solvers = {'highs-ipm': 'ipm', 'highs-ds': 'simplex',
658                           'highs': None}
659  
660          sol = _linprog_highs(lp, solver=highs_solvers[meth],
661                               **solver_options)
662          sol['status'], sol['message'] = (
663              _check_result(sol['x'], sol['fun'], sol['status'], sol['slack'],
664                            sol['con'], lp.bounds, tol, sol['message'],
665                            integrality))
666          sol['success'] = sol['status'] == 0
667          return OptimizeResult(sol)
668  
669      warn(f"`method='{meth}'` is deprecated and will be removed in SciPy "
670           "1.11.0. Please use one of the HiGHS solvers (e.g. "
671           "`method='highs'`) in new code.", DeprecationWarning, stacklevel=2)
672  
673      iteration = 0
674      complete = False  # will become True if solved in presolve
675      undo = []
676  
677      # Keep the original arrays to calculate slack/residuals for original
678      # problem.
679      lp_o = deepcopy(lp)
680  
681      # Solve trivial problem, eliminate variables, tighten bounds, etc.
682      rr_method = solver_options.pop('rr_method', None)  # need to pop these;
683      rr = solver_options.pop('rr', True)  # they're not passed to methods
684      c0 = 0  # we might get a constant term in the objective
685      if solver_options.pop('presolve', True):
686          (lp, c0, x, undo, complete, status, message) = _presolve(lp, rr,
687                                                                   rr_method,
688                                                                   tol)
689  
690      C, b_scale = 1, 1  # for trivial unscaling if autoscale is not used
691      postsolve_args = (lp_o._replace(bounds=lp.bounds), undo, C, b_scale)
692  
693      if not complete:
694          A, b, c, c0, x0 = _get_Abc(lp, c0)
695          if solver_options.pop('autoscale', False):
696              A, b, c, x0, C, b_scale = _autoscale(A, b, c, x0)
697              postsolve_args = postsolve_args[:-2] + (C, b_scale)
698  
699          if meth == 'simplex':
700              x, status, message, iteration = _linprog_simplex(
701                  c, c0=c0, A=A, b=b, callback=callback,
702                  postsolve_args=postsolve_args, **solver_options)
703          elif meth == 'interior-point':
704              x, status, message, iteration = _linprog_ip(
705                  c, c0=c0, A=A, b=b, callback=callback,
706                  postsolve_args=postsolve_args, **solver_options)
707          elif meth == 'revised simplex':
708              x, status, message, iteration = _linprog_rs(
709                  c, c0=c0, A=A, b=b, x0=x0, callback=callback,
710                  postsolve_args=postsolve_args, **solver_options)
711  
712      # Eliminate artificial variables, re-introduce presolved variables, etc.
713      disp = solver_options.get('disp', False)
714  
715      x, fun, slack, con = _postsolve(x, postsolve_args, complete)
716  
717      status, message = _check_result(x, fun, status, slack, con, lp_o.bounds,
718                                      tol, message, integrality)
719  
720      if disp:
721          _display_summary(message, status, fun, iteration)
722  
723      sol = {
724          'x': x,
725          'fun': fun,
726          'slack': slack,
727          'con': con,
728          'status': status,
729          'message': message,
730          'nit': iteration,
731          'success': status == 0}
732  
733      return OptimizeResult(sol)
