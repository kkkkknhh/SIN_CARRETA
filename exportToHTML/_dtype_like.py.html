115  # Anything that can be coerced into numpy.dtype.
116  # Reference: https://docs.scipy.org/doc/numpy/reference/arrays.dtypes.html
117  DTypeLike: TypeAlias = (
118      np.dtype[Any]
119      # default data type (float64)
120      | None
121      # array-scalar types and generic types
122      | type[Any]  # NOTE: We're stuck with `type[Any]` due to object dtypes
123      # anything with a dtype attribute
124      | _SupportsDType[np.dtype[Any]]
125      # character codes, type strings or comma-separated fields, e.g., 'float64'
126      | str
127      | _VoidDTypeLike
128  )
