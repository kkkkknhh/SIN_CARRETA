12   class MatrixProductOperator(scipy.sparse.linalg.LinearOperator):
13       """ 
14       This is purely for onenormest testing. 
15       """
16   
17       def __init__(self, A, B):
18           if A.ndim != 2 or B.ndim != 2:
19               raise ValueError('expected ndarrays representing matrices')
20           if A.shape[1] != B.shape[0]:
21               raise ValueError('incompatible shapes')
22           self.A = A
23           self.B = B
24           self.ndim = 2
25           self.shape = (A.shape[0], B.shape[1])
26   
27       def _matvec(self, x):
28           return np.dot(self.A, np.dot(self.B, x))
29   
30       def _rmatvec(self, x):
31           return np.dot(np.dot(x, self.A), self.B)
32   
33       def _matmat(self, X):
34           return np.dot(self.A, np.dot(self.B, X))
35   
36       @property
37       def T(self):
38           return MatrixProductOperator(self.B.T, self.A.T)

45           # This will take multiple seconds if your computer is slow like mine.
46           # It is stochastic, so the tolerance could be too strict.
47           np.random.seed(1234)
48           t = 2
49           n = 100
50           itmax = 5
51           nsamples = 5000
52           observed = []
53           expected = []
54           nmult_list = []
55           nresample_list = []
56           for i in range(nsamples):
57               A = scipy.linalg.inv(np.random.randn(n, n))
58               est, v, w, nmults, nresamples = _onenormest_core(A, A.T, t, itmax)
59               observed.append(est)
60               expected.append(scipy.linalg.norm(A, 1))
61               nmult_list.append(nmults)
62               nresample_list.append(nresamples)
63           observed = np.array(observed, dtype=float)
64           expected = np.array(expected, dtype=float)
65           relative_errors = np.abs(observed - expected) / expected
66   
67           # check the mean underestimation ratio
68           underestimation_ratio = observed / expected
69           assert_(0.99 < np.mean(underestimation_ratio) < 1.0)
70   
71           # check the max and mean required column resamples
72           assert_equal(np.max(nresample_list), 2)
73           assert_(0.05 < np.mean(nresample_list) < 0.2)
74   
75           # check the proportion of norms computed exactly correctly
76           nexact = np.count_nonzero(relative_errors < 1e-14)
77           proportion_exact = nexact / float(nsamples)
78           assert_(0.9 < proportion_exact < 0.95)
79   
80           # check the average number of matrix*vector multiplications
81           assert_(3.5 < np.mean(nmult_list) < 4.5)

122      def test_onenormest_table_5_t_1(self):
123          # "note that there is no randomness and hence only one estimate for t=1"
124          t = 1
125          n = 100
126          itmax = 5
127          alpha = 1 - 1e-6
128          A = -scipy.linalg.inv(np.identity(n) + alpha*np.eye(n, k=1))
129          first_col = np.array([1] + [0]*(n-1))
130          first_row = np.array([(-alpha)**i for i in range(n)])
131          B = -scipy.linalg.toeplitz(first_col, first_row)
132          assert_allclose(A, B)
133          est, v, w, nmults, nresamples = _onenormest_core(B, B.T, t, itmax)
134          exact_value = scipy.linalg.norm(B, 1)
135          underest_ratio = est / exact_value
136          assert_allclose(underest_ratio, 0.05, rtol=1e-4)
137          assert_equal(nmults, 11)
138          assert_equal(nresamples, 0)
139          # check the non-underscored version of onenormest
140          est_plain = scipy.sparse.linalg.onenormest(B, t=t, itmax=itmax)
141          assert_allclose(est, est_plain)

202      @pytest.mark.slow
203      def test_onenormest_linear_operator(self):
204          # Define a matrix through its product A B.
205          # Depending on the shapes of A and B,
206          # it could be easy to multiply this product by a small matrix,
207          # but it could be annoying to look at all of
208          # the entries of the product explicitly.
209          np.random.seed(1234)
210          n = 6000
211          k = 3
212          A = np.random.randn(n, k)
213          B = np.random.randn(k, n)
214          fast_estimate = self._help_product_norm_fast(A, B)
215          exact_value = self._help_product_norm_slow(A, B)
216          assert_(fast_estimate <= exact_value <= 3*fast_estimate,
217                  f'fast: {fast_estimate:g}\nexact:{exact_value:g}')

219      def test_returns(self):
220          np.random.seed(1234)
221          A = scipy.sparse.rand(50, 50, 0.1)
222  
223          s0 = scipy.linalg.norm(A.toarray(), 1)
224          s1, v = scipy.sparse.linalg.onenormest(A, compute_v=True)
225          s2, w = scipy.sparse.linalg.onenormest(A, compute_w=True)
226          s3, v2, w2 = scipy.sparse.linalg.onenormest(A, compute_w=True, compute_v=True)
227  
228          assert_allclose(s1, s0, rtol=1e-9)
229          assert_allclose(np.linalg.norm(A.dot(v), 1), s0*np.linalg.norm(v, 1), rtol=1e-9)
230          assert_allclose(A.dot(v), w, rtol=1e-9)
