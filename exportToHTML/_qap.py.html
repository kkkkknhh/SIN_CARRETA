14   def quadratic_assignment(A, B, method="faq", options=None):
15       r""" 
16       Approximates solution to the quadratic assignment problem and 
17       the graph matching problem. 
18    
19       Quadratic assignment solves problems of the following form: 
20    
21       .. math:: 
22    
23           \min_P & \ {\ \text{trace}(A^T P B P^T)}\\ 
24           \mbox{s.t. } & {P \ \epsilon \ \mathcal{P}}\\ 
25    
26       where :math:`\mathcal{P}` is the set of all permutation matrices, 
27       and :math:`A` and :math:`B` are square matrices. 
28    
29       Graph matching tries to *maximize* the same objective function. 
30       This algorithm can be thought of as finding the alignment of the 
31       nodes of two graphs that minimizes the number of induced edge 
32       disagreements, or, in the case of weighted graphs, the sum of squared 
33       edge weight differences. 
34    
35       Note that the quadratic assignment problem is NP-hard. The results given 
36       here are approximations and are not guaranteed to be optimal. 
37    
38    
39       Parameters 
40       ---------- 
41       A : 2-D array, square 
42           The square matrix :math:`A` in the objective function above. 
43    
44       B : 2-D array, square 
45           The square matrix :math:`B` in the objective function above. 
46    
47       method :  str in {'faq', '2opt'} (default: 'faq') 
48           The algorithm used to solve the problem. 
49           :ref:`'faq' <optimize.qap-faq>` (default) and 
50           :ref:`'2opt' <optimize.qap-2opt>` are available. 
51    
52       options : dict, optional 
53           A dictionary of solver options. All solvers support the following: 
54    
55           maximize : bool (default: False) 
56               Maximizes the objective function if ``True``. 
57    
58           partial_match : 2-D array of integers, optional (default: None) 
59               Fixes part of the matching. Also known as a "seed" [2]_. 
60    
61               Each row of `partial_match` specifies a pair of matched nodes: 
62               node ``partial_match[i, 0]`` of `A` is matched to node 
63               ``partial_match[i, 1]`` of `B`. The array has shape ``(m, 2)``, 
64               where ``m`` is not greater than the number of nodes, :math:`n`. 
65    
66           rng : `numpy.random.Generator`, optional 
67               Pseudorandom number generator state. When `rng` is None, a new 
68               `numpy.random.Generator` is created using entropy from the 
69               operating system. Types other than `numpy.random.Generator` are 
70               passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
71    
72               .. versionchanged:: 1.15.0 
73                   As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
74                   transition from use of `numpy.random.RandomState` to 
75                   `numpy.random.Generator` is occurring. Supplying 
76                   `np.random.RandomState` to this function will now emit a 
77                   `DeprecationWarning`. In SciPy 1.17 its use will raise an exception. 
78                   In addition relying on global state using `np.random.seed` 
79                   will emit a `FutureWarning`. In SciPy 1.17 the global random number 
80                   generator will no longer be used. 
81                   Use of an int-like seed will raise a `FutureWarning`, in SciPy 1.17 it 
82                   will be normalized via `np.random.default_rng` rather than 
83                   `np.random.RandomState`. 
84    
85           For method-specific options, see 
86           :func:`show_options('quadratic_assignment') <show_options>`. 
87    
88       Returns 
89       ------- 
90       res : OptimizeResult 
91           `OptimizeResult` containing the following fields. 
92    
93           col_ind : 1-D array 
94               Column indices corresponding to the best permutation found of the 
95               nodes of `B`. 
96           fun : float 
97               The objective value of the solution. 
98           nit : int 
99               The number of iterations performed during optimization. 
100   
101      Notes 
102      ----- 
103      The default method :ref:`'faq' <optimize.qap-faq>` uses the Fast 
104      Approximate QAP algorithm [1]_; it typically offers the best combination of 
105      speed and accuracy. 
106      Method :ref:`'2opt' <optimize.qap-2opt>` can be computationally expensive, 
107      but may be a useful alternative, or it can be used to refine the solution 
108      returned by another method. 
109   
110      References 
111      ---------- 
112      .. [1] J.T. Vogelstein, J.M. Conroy, V. Lyzinski, L.J. Podrazik, 
113             S.G. Kratzer, E.T. Harley, D.E. Fishkind, R.J. Vogelstein, and 
114             C.E. Priebe, "Fast approximate quadratic programming for graph 
115             matching," PLOS one, vol. 10, no. 4, p. e0121002, 2015, 
116             :doi:`10.1371/journal.pone.0121002` 
117   
118      .. [2] D. Fishkind, S. Adali, H. Patsolic, L. Meng, D. Singh, V. Lyzinski, 
119             C. Priebe, "Seeded graph matching", Pattern Recognit. 87 (2019): 
120             203-215, :doi:`10.1016/j.patcog.2018.09.014` 
121   
122      .. [3] "2-opt," Wikipedia. 
123             https://en.wikipedia.org/wiki/2-opt 
124   
125      Examples 
126      -------- 
127      >>> import numpy as np 
128      >>> from scipy.optimize import quadratic_assignment 
129      >>> rng = np.random.default_rng() 
130      >>> A = np.array([[0, 80, 150, 170], [80, 0, 130, 100], 
131      ...               [150, 130, 0, 120], [170, 100, 120, 0]]) 
132      >>> B = np.array([[0, 5, 2, 7], [0, 0, 3, 8], 
133      ...               [0, 0, 0, 3], [0, 0, 0, 0]]) 
134      >>> res = quadratic_assignment(A, B, options={'rng': rng}) 
135      >>> print(res) 
136           fun: 3260 
137       col_ind: [0 3 2 1] 
138           nit: 9 
139   
140      The see the relationship between the returned ``col_ind`` and ``fun``, 
141      use ``col_ind`` to form the best permutation matrix found, then evaluate 
142      the objective function :math:`f(P) = trace(A^T P B P^T )`. 
143   
144      >>> perm = res['col_ind'] 
145      >>> P = np.eye(len(A), dtype=int)[perm] 
146      >>> fun = np.trace(A.T @ P @ B @ P.T) 
147      >>> print(fun) 
148      3260 
149   
150      Alternatively, to avoid constructing the permutation matrix explicitly, 
151      directly permute the rows and columns of the distance matrix. 
152   
153      >>> fun = np.trace(A.T @ B[perm][:, perm]) 
154      >>> print(fun) 
155      3260 
156   
157      Although not guaranteed in general, ``quadratic_assignment`` happens to 
158      have found the globally optimal solution. 
159   
160      >>> from itertools import permutations 
161      >>> perm_opt, fun_opt = None, np.inf 
162      >>> for perm in permutations([0, 1, 2, 3]): 
163      ...     perm = np.array(perm) 
164      ...     fun = np.trace(A.T @ B[perm][:, perm]) 
165      ...     if fun < fun_opt: 
166      ...         fun_opt, perm_opt = fun, perm 
167      >>> print(np.array_equal(perm_opt, res['col_ind'])) 
168      True 
169   
170      Here is an example for which the default method, 
171      :ref:`'faq' <optimize.qap-faq>`, does not find the global optimum. 
172   
173      >>> A = np.array([[0, 5, 8, 6], [5, 0, 5, 1], 
174      ...               [8, 5, 0, 2], [6, 1, 2, 0]]) 
175      >>> B = np.array([[0, 1, 8, 4], [1, 0, 5, 2], 
176      ...               [8, 5, 0, 5], [4, 2, 5, 0]]) 
177      >>> res = quadratic_assignment(A, B, options={'rng': rng}) 
178      >>> print(res) 
179           fun: 178 
180       col_ind: [1 0 3 2] 
181           nit: 13 
182   
183      If accuracy is important, consider using  :ref:`'2opt' <optimize.qap-2opt>` 
184      to refine the solution. 
185   
186      >>> guess = np.array([np.arange(len(A)), res.col_ind]).T 
187      >>> res = quadratic_assignment(A, B, method="2opt", 
188      ...     options = {'rng': rng, 'partial_guess': guess}) 
189      >>> print(res) 
190           fun: 176 
191       col_ind: [1 2 3 0] 
192           nit: 17 
193   
194      """
195  
196      if options is None:
197          options = {}
198  
199      method = method.lower()
200      methods = {"faq": _quadratic_assignment_faq,
201                 "2opt": _quadratic_assignment_2opt}
202      if method not in methods:
203          raise ValueError(f"method {method} must be in {methods}.")
204  
205      _spec007_transition(options.get("rng", None))
206      res = methods[method](A, B, **options)
207      return res

210  def _spec007_transition(rng):
211      if isinstance(rng, np.random.RandomState):
212          warnings.warn(
213              "Use of `RandomState` with `quadratic_assignment` is deprecated"
214              " and will result in an exception in SciPy 1.17",
215              DeprecationWarning,
216              stacklevel=2
217          )
218      if ((rng is None or rng is np.random) and
219              np.random.mtrand._rand._bit_generator._seed_seq is None):
220          warnings.warn(
221              "The NumPy global RNG was seeded by calling `np.random.seed`."
222              " From SciPy 1.17, this function will no longer use the global RNG.",
223              FutureWarning,
224              stacklevel=2
225          )
226      if isinstance(rng, numbers.Integral | np.integer):
227          warnings.warn(
228              "The behavior when the rng option is an integer is changing: the value"
229              " will be normalized using np.random.default_rng beginning in SciPy 1.17,"
230              " and the resulting Generator will be used to generate random numbers.",
231              FutureWarning,
232              stacklevel=2
233          )

236  def _calc_score(A, B, perm):
237      # equivalent to objective function but avoids matmul
238      return np.sum(A * B[perm][:, perm])

241  def _common_input_validation(A, B, partial_match):
242      A = np.atleast_2d(A)
243      B = np.atleast_2d(B)
244  
245      if partial_match is None:
246          partial_match = np.array([[], []]).T
247      partial_match = np.atleast_2d(partial_match).astype(int)
248  
249      msg = None
250      if A.shape[0] != A.shape[1]:
251          msg = "`A` must be square"
252      elif B.shape[0] != B.shape[1]:
253          msg = "`B` must be square"
254      elif A.ndim != 2 or B.ndim != 2:
255          msg = "`A` and `B` must have exactly two dimensions"
256      elif A.shape != B.shape:
257          msg = "`A` and `B` matrices must be of equal size"
258      elif partial_match.shape[0] > A.shape[0]:
259          msg = "`partial_match` can have only as many seeds as there are nodes"
260      elif partial_match.shape[1] != 2:
261          msg = "`partial_match` must have two columns"
262      elif partial_match.ndim != 2:
263          msg = "`partial_match` must have exactly two dimensions"
264      elif (partial_match < 0).any():
265          msg = "`partial_match` must contain only positive indices"
266      elif (partial_match >= len(A)).any():
267          msg = "`partial_match` entries must be less than number of nodes"
268      elif (not len(set(partial_match[:, 0])) == len(partial_match[:, 0]) or
269            not len(set(partial_match[:, 1])) == len(partial_match[:, 1])):
270          msg = "`partial_match` column entries must be unique"
271  
272      if msg is not None:
273          raise ValueError(msg)
274  
275      return A, B, partial_match

278  def _quadratic_assignment_faq(A, B,
279                                maximize=False, partial_match=None, rng=None,
280                                P0="barycenter", shuffle_input=False, maxiter=30,
281                                tol=0.03, **unknown_options):
282      r"""Solve the quadratic assignment problem (approximately). 
283   
284      This function solves the Quadratic Assignment Problem (QAP) and the 
285      Graph Matching Problem (GMP) using the Fast Approximate QAP Algorithm 
286      (FAQ) [1]_. 
287   
288      Quadratic assignment solves problems of the following form: 
289   
290      .. math:: 
291   
292          \min_P & \ {\ \text{trace}(A^T P B P^T)}\\ 
293          \mbox{s.t. } & {P \ \epsilon \ \mathcal{P}}\\ 
294   
295      where :math:`\mathcal{P}` is the set of all permutation matrices, 
296      and :math:`A` and :math:`B` are square matrices. 
297   
298      Graph matching tries to *maximize* the same objective function. 
299      This algorithm can be thought of as finding the alignment of the 
300      nodes of two graphs that minimizes the number of induced edge 
301      disagreements, or, in the case of weighted graphs, the sum of squared 
302      edge weight differences. 
303   
304      Note that the quadratic assignment problem is NP-hard. The results given 
305      here are approximations and are not guaranteed to be optimal. 
306   
307      Parameters 
308      ---------- 
309      A : 2-D array, square 
310          The square matrix :math:`A` in the objective function above. 
311      B : 2-D array, square 
312          The square matrix :math:`B` in the objective function above. 
313      method :  str in {'faq', '2opt'} (default: 'faq') 
314          The algorithm used to solve the problem. This is the method-specific 
315          documentation for 'faq'. 
316          :ref:`'2opt' <optimize.qap-2opt>` is also available. 
317   
318      Options 
319      ------- 
320      maximize : bool (default: False) 
321          Maximizes the objective function if ``True``. 
322      partial_match : 2-D array of integers, optional (default: None) 
323          Fixes part of the matching. Also known as a "seed" [2]_. 
324   
325          Each row of `partial_match` specifies a pair of matched nodes: 
326          node ``partial_match[i, 0]`` of `A` is matched to node 
327          ``partial_match[i, 1]`` of `B`. The array has shape ``(m, 2)``, where 
328          ``m`` is not greater than the number of nodes, :math:`n`. 
329   
330      rng : {None, int, `numpy.random.Generator`}, optional 
331          Pseudorandom number generator state. See `quadratic_assignment` for details. 
332      P0 : 2-D array, "barycenter", or "randomized" (default: "barycenter") 
333          Initial position. Must be a doubly-stochastic matrix [3]_. 
334   
335          If the initial position is an array, it must be a doubly stochastic 
336          matrix of size :math:`m' \times m'` where :math:`m' = n - m`. 
337   
338          If ``"barycenter"`` (default), the initial position is the barycenter 
339          of the Birkhoff polytope (the space of doubly stochastic matrices). 
340          This is a :math:`m' \times m'` matrix with all entries equal to 
341          :math:`1 / m'`. 
342   
343          If ``"randomized"`` the initial search position is 
344          :math:`P_0 = (J + K) / 2`, where :math:`J` is the barycenter and 
345          :math:`K` is a random doubly stochastic matrix. 
346      shuffle_input : bool (default: False) 
347          Set to `True` to resolve degenerate gradients randomly. For 
348          non-degenerate gradients this option has no effect. 
349      maxiter : int, positive (default: 30) 
350          Integer specifying the max number of Frank-Wolfe iterations performed. 
351      tol : float (default: 0.03) 
352          Tolerance for termination. Frank-Wolfe iteration terminates when 
353          :math:`\frac{||P_{i}-P_{i+1}||_F}{\sqrt{m'}} \leq tol`, 
354          where :math:`i` is the iteration number. 
355   
356      Returns 
357      ------- 
358      res : OptimizeResult 
359          `OptimizeResult` containing the following fields. 
360   
361          col_ind : 1-D array 
362              Column indices corresponding to the best permutation found of the 
363              nodes of `B`. 
364          fun : float 
365              The objective value of the solution. 
366          nit : int 
367              The number of Frank-Wolfe iterations performed. 
368   
369      Notes 
370      ----- 
371      The algorithm may be sensitive to the initial permutation matrix (or 
372      search "position") due to the possibility of several local minima 
373      within the feasible region. A barycenter initialization is more likely to 
374      result in a better solution than a single random initialization. However, 
375      calling ``quadratic_assignment`` several times with different random 
376      initializations may result in a better optimum at the cost of longer 
377      total execution time. 
378   
379      Examples 
380      -------- 
381      As mentioned above, a barycenter initialization often results in a better 
382      solution than a single random initialization. 
383   
384      >>> from scipy.optimize import quadratic_assignment 
385      >>> import numpy as np 
386      >>> rng = np.random.default_rng() 
387      >>> n = 15 
388      >>> A = rng.random((n, n)) 
389      >>> B = rng.random((n, n)) 
390      >>> options = {"rng": rng} 
391      >>> res = quadratic_assignment(A, B, options=options)  # FAQ is default method 
392      >>> print(res.fun) 
393      47.797048706380636  # may vary 
394   
395      >>> options = {"rng": rng, "P0": "randomized"}  # use randomized initialization 
396      >>> res = quadratic_assignment(A, B, options=options) 
397      >>> print(res.fun) 
398      47.37287069769966 # may vary 
399   
400      However, consider running from several randomized initializations and 
401      keeping the best result. 
402   
403      >>> res = min([quadratic_assignment(A, B, options=options) 
404      ...            for i in range(30)], key=lambda x: x.fun) 
405      >>> print(res.fun) 
406      46.55974835248574 # may vary 
407   
408      The '2-opt' method can be used to attempt to refine the results. 
409   
410      >>> options = {"partial_guess": np.array([np.arange(n), res.col_ind]).T, "rng": rng} 
411      >>> res = quadratic_assignment(A, B, method="2opt", options=options) 
412      >>> print(res.fun) 
413      46.55974835248574 # may vary 
414   
415      References 
416      ---------- 
417      .. [1] J.T. Vogelstein, J.M. Conroy, V. Lyzinski, L.J. Podrazik, 
418             S.G. Kratzer, E.T. Harley, D.E. Fishkind, R.J. Vogelstein, and 
419             C.E. Priebe, "Fast approximate quadratic programming for graph 
420             matching," PLOS one, vol. 10, no. 4, p. e0121002, 2015, 
421             :doi:`10.1371/journal.pone.0121002` 
422   
423      .. [2] D. Fishkind, S. Adali, H. Patsolic, L. Meng, D. Singh, V. Lyzinski, 
424             C. Priebe, "Seeded graph matching", Pattern Recognit. 87 (2019): 
425             203-215, :doi:`10.1016/j.patcog.2018.09.014` 
426   
427      .. [3] "Doubly stochastic Matrix," Wikipedia. 
428             https://en.wikipedia.org/wiki/Doubly_stochastic_matrix 
429   
430      """
431  
432      _check_unknown_options(unknown_options)
433  
434      maxiter = operator.index(maxiter)
435  
436      # ValueError check
437      A, B, partial_match = _common_input_validation(A, B, partial_match)
438  
439      msg = None
440      if isinstance(P0, str) and P0 not in {'barycenter', 'randomized'}:
441          msg = "Invalid 'P0' parameter string"
442      elif maxiter <= 0:
443          msg = "'maxiter' must be a positive integer"
444      elif tol <= 0:
445          msg = "'tol' must be a positive float"
446      if msg is not None:
447          raise ValueError(msg)
448  
449      rng = check_random_state(rng)
450      n = len(A)  # number of vertices in graphs
451      n_seeds = len(partial_match)  # number of seeds
452      n_unseed = n - n_seeds
453  
454      # [1] Algorithm 1 Line 1 - choose initialization
455      if not isinstance(P0, str):
456          P0 = np.atleast_2d(P0)
457          if P0.shape != (n_unseed, n_unseed):
458              msg = "`P0` matrix must have shape m' x m', where m'=n-m"
459          elif ((P0 < 0).any() or not np.allclose(np.sum(P0, axis=0), 1)
460                or not np.allclose(np.sum(P0, axis=1), 1)):
461              msg = "`P0` matrix must be doubly stochastic"
462          if msg is not None:
463              raise ValueError(msg)
464      elif P0 == 'barycenter':
465          P0 = np.ones((n_unseed, n_unseed)) / n_unseed
466      elif P0 == 'randomized':
467          J = np.ones((n_unseed, n_unseed)) / n_unseed
468          # generate a nxn matrix where each entry is a random number [0, 1]
469          # would use rand, but Generators don't have it
470          # would use random, but old mtrand.RandomStates don't have it
471          K = _doubly_stochastic(rng.uniform(size=(n_unseed, n_unseed)))
472          P0 = (J + K) / 2
473  
474      # check trivial cases
475      if n == 0 or n_seeds == n:
476          score = _calc_score(A, B, partial_match[:, 1])
477          res = {"col_ind": partial_match[:, 1], "fun": score, "nit": 0}
478          return OptimizeResult(res)
479  
480      obj_func_scalar = 1
481      if maximize:
482          obj_func_scalar = -1
483  
484      nonseed_B = np.setdiff1d(range(n), partial_match[:, 1])
485      if shuffle_input:
486          nonseed_B = rng.permutation(nonseed_B)
487  
488      nonseed_A = np.setdiff1d(range(n), partial_match[:, 0])
489      perm_A = np.concatenate([partial_match[:, 0], nonseed_A])
490      perm_B = np.concatenate([partial_match[:, 1], nonseed_B])
491  
492      # definitions according to Seeded Graph Matching [2].
493      A11, A12, A21, A22 = _split_matrix(A[perm_A][:, perm_A], n_seeds)
494      B11, B12, B21, B22 = _split_matrix(B[perm_B][:, perm_B], n_seeds)
495      const_sum = A21 @ B21.T + A12.T @ B12
496  
497      P = P0
498      # [1] Algorithm 1 Line 2 - loop while stopping criteria not met
499      for n_iter in range(1, maxiter+1):
500          # [1] Algorithm 1 Line 3 - compute the gradient of f(P) = -tr(APB^tP^t)
501          grad_fp = (const_sum + A22 @ P @ B22.T + A22.T @ P @ B22)
502          # [1] Algorithm 1 Line 4 - get direction Q by solving Eq. 8
503          _, cols = linear_sum_assignment(grad_fp, maximize=maximize)
504          Q = np.eye(n_unseed)[cols]
505  
506          # [1] Algorithm 1 Line 5 - compute the step size
507          # Noting that e.g. trace(Ax) = trace(A)*x, expand and re-collect
508          # terms as ax**2 + bx + c. c does not affect location of minimum
509          # and can be ignored. Also, note that trace(A@B) = (A.T*B).sum();
510          # apply where possible for efficiency.
511          R = P - Q
512          b21 = ((R.T @ A21) * B21).sum()
513          b12 = ((R.T @ A12.T) * B12.T).sum()
514          AR22 = A22.T @ R
515          BR22 = B22 @ R.T
516          b22a = (AR22 * B22.T[cols]).sum()
517          b22b = (A22 * BR22[cols]).sum()
518          a = (AR22.T * BR22).sum()
519          b = b21 + b12 + b22a + b22b
520          # critical point of ax^2 + bx + c is at x = -d/(2*e)
521          # if a * obj_func_scalar > 0, it is a minimum
522          # if minimum is not in [0, 1], only endpoints need to be considered
523          if a*obj_func_scalar > 0 and 0 <= -b/(2*a) <= 1:
524              alpha = -b/(2*a)
525          else:
526              alpha = np.argmin([0, (b + a)*obj_func_scalar])
527  
528          # [1] Algorithm 1 Line 6 - Update P
529          P_i1 = alpha * P + (1 - alpha) * Q
530          if np.linalg.norm(P - P_i1) / np.sqrt(n_unseed) < tol:
531              P = P_i1
532              break
533          P = P_i1
534      # [1] Algorithm 1 Line 7 - end main loop
535  
536      # [1] Algorithm 1 Line 8 - project onto the set of permutation matrices
537      _, col = linear_sum_assignment(P, maximize=True)
538      perm = np.concatenate((np.arange(n_seeds), col + n_seeds))
539  
540      unshuffled_perm = np.zeros(n, dtype=int)
541      unshuffled_perm[perm_A] = perm_B[perm]
542  
543      score = _calc_score(A, B, unshuffled_perm)
544      res = {"col_ind": unshuffled_perm, "fun": score, "nit": n_iter}
545      return OptimizeResult(res)

548  def _split_matrix(X, n):
549      # definitions according to Seeded Graph Matching [2].
550      upper, lower = X[:n], X[n:]
551      return upper[:, :n], upper[:, n:], lower[:, :n], lower[:, n:]

554  def _doubly_stochastic(P, tol=1e-3):
555      # Adapted from @btaba implementation
556      # https://github.com/btaba/sinkhorn_knopp
557      # of Sinkhorn-Knopp algorithm
558      # https://projecteuclid.org/euclid.pjm/1102992505
559  
560      max_iter = 1000
561      c = 1 / P.sum(axis=0)
562      r = 1 / (P @ c)
563      P_eps = P
564  
565      for it in range(max_iter):
566          if ((np.abs(P_eps.sum(axis=1) - 1) < tol).all() and
567                  (np.abs(P_eps.sum(axis=0) - 1) < tol).all()):
568              # All column/row sums ~= 1 within threshold
569              break
570  
571          c = 1 / (r @ P)
572          r = 1 / (P @ c)
573          P_eps = r[:, None] * P * c
574  
575      return P_eps
576

578  def _quadratic_assignment_2opt(A, B, maximize=False, rng=None,
579                                 partial_match=None,
580                                 partial_guess=None,
581                                 **unknown_options):
582      r"""Solve the quadratic assignment problem (approximately). 
583   
584      This function solves the Quadratic Assignment Problem (QAP) and the 
585      Graph Matching Problem (GMP) using the 2-opt algorithm [1]_. 
586   
587      Quadratic assignment solves problems of the following form: 
588   
589      .. math:: 
590   
591          \min_P & \ {\ \text{trace}(A^T P B P^T)}\\ 
592          \mbox{s.t. } & {P \ \epsilon \ \mathcal{P}}\\ 
593   
594      where :math:`\mathcal{P}` is the set of all permutation matrices, 
595      and :math:`A` and :math:`B` are square matrices. 
596   
597      Graph matching tries to *maximize* the same objective function. 
598      This algorithm can be thought of as finding the alignment of the 
599      nodes of two graphs that minimizes the number of induced edge 
600      disagreements, or, in the case of weighted graphs, the sum of squared 
601      edge weight differences. 
602   
603      Note that the quadratic assignment problem is NP-hard. The results given 
604      here are approximations and are not guaranteed to be optimal. 
605   
606      Parameters 
607      ---------- 
608      A : 2-D array, square 
609          The square matrix :math:`A` in the objective function above. 
610      B : 2-D array, square 
611          The square matrix :math:`B` in the objective function above. 
612      method :  str in {'faq', '2opt'} (default: 'faq') 
613          The algorithm used to solve the problem. This is the method-specific 
614          documentation for '2opt'. 
615          :ref:`'faq' <optimize.qap-faq>` is also available. 
616   
617      Options 
618      ------- 
619      maximize : bool (default: False) 
620          Maximizes the objective function if ``True``. 
621      rng : {None, int, `numpy.random.Generator`}, optional 
622          Pseudorandom number generator state. See `quadratic_assignment` for details. 
623      partial_match : 2-D array of integers, optional (default: None) 
624          Fixes part of the matching. Also known as a "seed" [2]_. 
625   
626          Each row of `partial_match` specifies a pair of matched nodes: node 
627          ``partial_match[i, 0]`` of `A` is matched to node 
628          ``partial_match[i, 1]`` of `B`. The array has shape ``(m, 2)``, 
629          where ``m`` is not greater than the number of nodes, :math:`n`. 
630   
631          .. note:: 
632               `partial_match` must be sorted by the first column. 
633   
634      partial_guess : 2-D array of integers, optional (default: None) 
635          A guess for the matching between the two matrices. Unlike 
636          `partial_match`, `partial_guess` does not fix the indices; they are 
637          still free to be optimized. 
638   
639          Each row of `partial_guess` specifies a pair of matched nodes: node 
640          ``partial_guess[i, 0]`` of `A` is matched to node 
641          ``partial_guess[i, 1]`` of `B`. The array has shape ``(m, 2)``, 
642          where ``m`` is not greater than the number of nodes, :math:`n`. 
643   
644          .. note:: 
645                  `partial_guess` must be sorted by the first column. 
646   
647      Returns 
648      ------- 
649      res : OptimizeResult 
650          `OptimizeResult` containing the following fields. 
651   
652          col_ind : 1-D array 
653              Column indices corresponding to the best permutation found of the 
654              nodes of `B`. 
655          fun : float 
656              The objective value of the solution. 
657          nit : int 
658              The number of iterations performed during optimization. 
659   
660      Notes 
661      ----- 
662      This is a greedy algorithm that works similarly to bubble sort: beginning 
663      with an initial permutation, it iteratively swaps pairs of indices to 
664      improve the objective function until no such improvements are possible. 
665   
666      References 
667      ---------- 
668      .. [1] "2-opt," Wikipedia. 
669             https://en.wikipedia.org/wiki/2-opt 
670   
671      .. [2] D. Fishkind, S. Adali, H. Patsolic, L. Meng, D. Singh, V. Lyzinski, 
672             C. Priebe, "Seeded graph matching", Pattern Recognit. 87 (2019): 
673             203-215, https://doi.org/10.1016/j.patcog.2018.09.014 
674   
675      """
676      _check_unknown_options(unknown_options)
677      rng = check_random_state(rng)
678      A, B, partial_match = _common_input_validation(A, B, partial_match)
679  
680      N = len(A)
681      # check trivial cases
682      if N == 0 or partial_match.shape[0] == N:
683          score = _calc_score(A, B, partial_match[:, 1])
684          res = {"col_ind": partial_match[:, 1], "fun": score, "nit": 0}
685          return OptimizeResult(res)
686  
687      if partial_guess is None:
688          partial_guess = np.array([[], []]).T
689      partial_guess = np.atleast_2d(partial_guess).astype(int)
690  
691      msg = None
692      if partial_guess.shape[0] > A.shape[0]:
693          msg = ("`partial_guess` can have only as "
694                 "many entries as there are nodes")
695      elif partial_guess.shape[1] != 2:
696          msg = "`partial_guess` must have two columns"
697      elif partial_guess.ndim != 2:
698          msg = "`partial_guess` must have exactly two dimensions"
699      elif (partial_guess < 0).any():
700          msg = "`partial_guess` must contain only positive indices"
701      elif (partial_guess >= len(A)).any():
702          msg = "`partial_guess` entries must be less than number of nodes"
703      elif (not len(set(partial_guess[:, 0])) == len(partial_guess[:, 0]) or
704            not len(set(partial_guess[:, 1])) == len(partial_guess[:, 1])):
705          msg = "`partial_guess` column entries must be unique"
706      if msg is not None:
707          raise ValueError(msg)
708  
709      fixed_rows = None
710      if partial_match.size or partial_guess.size:
711          # use partial_match and partial_guess for initial permutation,
712          # but randomly permute the rest.
713          guess_rows = np.zeros(N, dtype=bool)
714          guess_cols = np.zeros(N, dtype=bool)
715          fixed_rows = np.zeros(N, dtype=bool)
716          fixed_cols = np.zeros(N, dtype=bool)
717          perm = np.zeros(N, dtype=int)
718  
719          rg, cg = partial_guess.T
720          guess_rows[rg] = True
721          guess_cols[cg] = True
722          perm[guess_rows] = cg
723  
724          # match overrides guess
725          rf, cf = partial_match.T
726          fixed_rows[rf] = True
727          fixed_cols[cf] = True
728          perm[fixed_rows] = cf
729  
730          random_rows = ~fixed_rows & ~guess_rows
731          random_cols = ~fixed_cols & ~guess_cols
732          perm[random_rows] = rng.permutation(np.arange(N)[random_cols])
733      else:
734          perm = rng.permutation(np.arange(N))
735  
736      best_score = _calc_score(A, B, perm)
737  
738      i_free = np.arange(N)
739      if fixed_rows is not None:
740          i_free = i_free[~fixed_rows]
741  
742      better = operator.gt if maximize else operator.lt
743      n_iter = 0
744      done = False
745      while not done:
746          # equivalent to nested for loops i in range(N), j in range(i, N)
747          for i, j in itertools.combinations_with_replacement(i_free, 2):
748              n_iter += 1
749              perm[i], perm[j] = perm[j], perm[i]
750              score = _calc_score(A, B, perm)
751              if better(score, best_score):
752                  best_score = score
753                  break
754              # faster to swap back than to create a new list every time
755              perm[i], perm[j] = perm[j], perm[i]
756          else:  # no swaps made
757              done = True
758  
759      res = {"col_ind": perm, "fun": best_score, "nit": n_iter}
760      return OptimizeResult(res)
