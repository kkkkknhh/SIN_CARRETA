17   def theilslopes(y, x=None, alpha=0.95, method='separate'):
18       r""" 
19       Computes the Theil-Sen estimator for a set of points (x, y). 
20    
21       `theilslopes` implements a method for robust linear regression.  It 
22       computes the slope as the median of all slopes between paired values. 
23    
24       Parameters 
25       ---------- 
26       y : array_like 
27           Dependent variable. 
28       x : array_like or None, optional 
29           Independent variable. If None, use ``arange(len(y))`` instead. 
30       alpha : float, optional 
31           Confidence degree between 0 and 1. Default is 95% confidence. 
32           Note that `alpha` is symmetric around 0.5, i.e. both 0.1 and 0.9 are 
33           interpreted as "find the 90% confidence interval". 
34       method : {'joint', 'separate'}, optional 
35           Method to be used for computing estimate for intercept. 
36           Following methods are supported, 
37    
38               * 'joint': Uses np.median(y - slope * x) as intercept. 
39               * 'separate': Uses np.median(y) - slope * np.median(x) 
40                             as intercept. 
41    
42           The default is 'separate'. 
43    
44           .. versionadded:: 1.8.0 
45    
46       Returns 
47       ------- 
48       result : ``TheilslopesResult`` instance 
49           The return value is an object with the following attributes: 
50    
51           slope : float 
52               Theil slope. 
53           intercept : float 
54               Intercept of the Theil line. 
55           low_slope : float 
56               Lower bound of the confidence interval on `slope`. 
57           high_slope : float 
58               Upper bound of the confidence interval on `slope`. 
59    
60       See Also 
61       -------- 
62       siegelslopes : a similar technique using repeated medians 
63    
64       Notes 
65       ----- 
66       The implementation of `theilslopes` follows [1]_. The intercept is 
67       not defined in [1]_, and here it is defined as ``median(y) - 
68       slope*median(x)``, which is given in [3]_. Other definitions of 
69       the intercept exist in the literature such as  ``median(y - slope*x)`` 
70       in [4]_. The approach to compute the intercept can be determined by the 
71       parameter ``method``. A confidence interval for the intercept is not 
72       given as this question is not addressed in [1]_. 
73    
74       For compatibility with older versions of SciPy, the return value acts 
75       like a ``namedtuple`` of length 4, with fields ``slope``, ``intercept``, 
76       ``low_slope``, and ``high_slope``, so one can continue to write:: 
77    
78           slope, intercept, low_slope, high_slope = theilslopes(y, x) 
79    
80       References 
81       ---------- 
82       .. [1] P.K. Sen, "Estimates of the regression coefficient based on 
83              Kendall's tau", J. Am. Stat. Assoc., Vol. 63, pp. 1379-1389, 1968. 
84       .. [2] H. Theil, "A rank-invariant method of linear and polynomial 
85              regression analysis I, II and III",  Nederl. Akad. Wetensch., Proc. 
86              53:, pp. 386-392, pp. 521-525, pp. 1397-1412, 1950. 
87       .. [3] W.L. Conover, "Practical nonparametric statistics", 2nd ed., 
88              John Wiley and Sons, New York, pp. 493. 
89       .. [4] https://en.wikipedia.org/wiki/Theil%E2%80%93Sen_estimator 
90    
91       Examples 
92       -------- 
93       >>> import numpy as np 
94       >>> from scipy import stats 
95       >>> import matplotlib.pyplot as plt 
96    
97       >>> x = np.linspace(-5, 5, num=150) 
98       >>> y = x + np.random.normal(size=x.size) 
99       >>> y[11:15] += 10  # add outliers 
100      >>> y[-5:] -= 7 
101   
102      Compute the slope, intercept and 90% confidence interval.  For comparison, 
103      also compute the least-squares fit with `linregress`: 
104   
105      >>> res = stats.theilslopes(y, x, 0.90, method='separate') 
106      >>> lsq_res = stats.linregress(x, y) 
107   
108      Plot the results. The Theil-Sen regression line is shown in red, with the 
109      dashed red lines illustrating the confidence interval of the slope (note 
110      that the dashed red lines are not the confidence interval of the regression 
111      as the confidence interval of the intercept is not included). The green 
112      line shows the least-squares fit for comparison. 
113   
114      >>> fig = plt.figure() 
115      >>> ax = fig.add_subplot(111) 
116      >>> ax.plot(x, y, 'b.') 
117      >>> ax.plot(x, res[1] + res[0] * x, 'r-') 
118      >>> ax.plot(x, res[1] + res[2] * x, 'r--') 
119      >>> ax.plot(x, res[1] + res[3] * x, 'r--') 
120      >>> ax.plot(x, lsq_res[1] + lsq_res[0] * x, 'g-') 
121      >>> plt.show() 
122   
123      """
124      if method not in ['joint', 'separate']:
125          raise ValueError("method must be either 'joint' or 'separate'."
126                           f"'{method}' is invalid.")
127      # We copy both x and y so we can use _find_repeats.
128      y = np.array(y, dtype=float, copy=True).ravel()
129      if x is None:
130          x = np.arange(len(y), dtype=float)
131      else:
132          x = np.array(x, dtype=float, copy=True).ravel()
133          if len(x) != len(y):
134              raise ValueError(f"Incompatible lengths ! ({len(y)}<>{len(x)})")
135  
136      # Compute sorted slopes only when deltax > 0
137      deltax = x[:, np.newaxis] - x
138      deltay = y[:, np.newaxis] - y
139      slopes = deltay[deltax > 0] / deltax[deltax > 0]
140      if not slopes.size:
141          msg = "All `x` coordinates are identical."
142          warnings.warn(msg, RuntimeWarning, stacklevel=2)
143      slopes.sort()
144      medslope = np.median(slopes)
145      if method == 'joint':
146          medinter = np.median(y - medslope * x)
147      else:
148          medinter = np.median(y) - medslope * np.median(x)
149      # Now compute confidence intervals
150      if alpha > 0.5:
151          alpha = 1. - alpha
152  
153      z = distributions.norm.ppf(alpha / 2.)
154      # This implements (2.6) from Sen (1968)
155      _, nxreps = _find_repeats(x)
156      _, nyreps = _find_repeats(y)
157      nt = len(slopes)       # N in Sen (1968)
158      ny = len(y)            # n in Sen (1968)
159      # Equation 2.6 in Sen (1968):
160      sigsq = 1/18. * (ny * (ny-1) * (2*ny+5) -
161                       sum(k * (k-1) * (2*k + 5) for k in nxreps) -
162                       sum(k * (k-1) * (2*k + 5) for k in nyreps))
163      # Find the confidence interval indices in `slopes`
164      try:
165          sigma = np.sqrt(sigsq)
166          Ru = min(int(np.round((nt - z*sigma)/2.)), len(slopes)-1)
167          Rl = max(int(np.round((nt + z*sigma)/2.)) - 1, 0)
168          delta = slopes[[Rl, Ru]]
169      except (ValueError, IndexError):
170          delta = (np.nan, np.nan)
171  
172      return TheilslopesResult(slope=medslope, intercept=medinter,
173                               low_slope=delta[0], high_slope=delta[1])

176  def _find_repeats(arr):
177      # This function assumes it may clobber its input.
178      if len(arr) == 0:
179          return np.array(0, np.float64), np.array(0, np.intp)
180  
181      # XXX This cast was previously needed for the Fortran implementation,
182      # should we ditch it?
183      arr = np.asarray(arr, np.float64).ravel()
184      arr.sort()
185  
186      # Taken from NumPy 1.9's np.unique.
187      change = np.concatenate(([True], arr[1:] != arr[:-1]))
188      unique = arr[change]
189      change_idx = np.concatenate(np.nonzero(change) + ([arr.size],))
190      freq = np.diff(change_idx)
191      atleast2 = freq > 1
192      return unique[atleast2], freq[atleast2]

195  def siegelslopes(y, x=None, method="hierarchical"):
196      r""" 
197      Computes the Siegel estimator for a set of points (x, y). 
198   
199      `siegelslopes` implements a method for robust linear regression 
200      using repeated medians (see [1]_) to fit a line to the points (x, y). 
201      The method is robust to outliers with an asymptotic breakdown point 
202      of 50%. 
203   
204      Parameters 
205      ---------- 
206      y : array_like 
207          Dependent variable. 
208      x : array_like or None, optional 
209          Independent variable. If None, use ``arange(len(y))`` instead. 
210      method : {'hierarchical', 'separate'} 
211          If 'hierarchical', estimate the intercept using the estimated 
212          slope ``slope`` (default option). 
213          If 'separate', estimate the intercept independent of the estimated 
214          slope. See Notes for details. 
215   
216      Returns 
217      ------- 
218      result : ``SiegelslopesResult`` instance 
219          The return value is an object with the following attributes: 
220   
221          slope : float 
222              Estimate of the slope of the regression line. 
223          intercept : float 
224              Estimate of the intercept of the regression line. 
225   
226      See Also 
227      -------- 
228      theilslopes : a similar technique without repeated medians 
229   
230      Notes 
231      ----- 
232      With ``n = len(y)``, compute ``m_j`` as the median of 
233      the slopes from the point ``(x[j], y[j])`` to all other `n-1` points. 
234      ``slope`` is then the median of all slopes ``m_j``. 
235      Two ways are given to estimate the intercept in [1]_ which can be chosen 
236      via the parameter ``method``. 
237      The hierarchical approach uses the estimated slope ``slope`` 
238      and computes ``intercept`` as the median of ``y - slope*x``. 
239      The other approach estimates the intercept separately as follows: for 
240      each point ``(x[j], y[j])``, compute the intercepts of all the `n-1` 
241      lines through the remaining points and take the median ``i_j``. 
242      ``intercept`` is the median of the ``i_j``. 
243   
244      The implementation computes `n` times the median of a vector of size `n` 
245      which can be slow for large vectors. There are more efficient algorithms 
246      (see [2]_) which are not implemented here. 
247   
248      For compatibility with older versions of SciPy, the return value acts 
249      like a ``namedtuple`` of length 2, with fields ``slope`` and 
250      ``intercept``, so one can continue to write:: 
251   
252          slope, intercept = siegelslopes(y, x) 
253   
254      References 
255      ---------- 
256      .. [1] A. Siegel, "Robust Regression Using Repeated Medians", 
257             Biometrika, Vol. 69, pp. 242-244, 1982. 
258   
259      .. [2] A. Stein and M. Werman, "Finding the repeated median regression 
260             line", Proceedings of the Third Annual ACM-SIAM Symposium on 
261             Discrete Algorithms, pp. 409-413, 1992. 
262   
263      Examples 
264      -------- 
265      >>> import numpy as np 
266      >>> from scipy import stats 
267      >>> import matplotlib.pyplot as plt 
268   
269      >>> x = np.linspace(-5, 5, num=150) 
270      >>> y = x + np.random.normal(size=x.size) 
271      >>> y[11:15] += 10  # add outliers 
272      >>> y[-5:] -= 7 
273   
274      Compute the slope and intercept.  For comparison, also compute the 
275      least-squares fit with `linregress`: 
276   
277      >>> res = stats.siegelslopes(y, x) 
278      >>> lsq_res = stats.linregress(x, y) 
279   
280      Plot the results. The Siegel regression line is shown in red. The green 
281      line shows the least-squares fit for comparison. 
282   
283      >>> fig = plt.figure() 
284      >>> ax = fig.add_subplot(111) 
285      >>> ax.plot(x, y, 'b.') 
286      >>> ax.plot(x, res[1] + res[0] * x, 'r-') 
287      >>> ax.plot(x, lsq_res[1] + lsq_res[0] * x, 'g-') 
288      >>> plt.show() 
289   
290      """
291      if method not in ['hierarchical', 'separate']:
292          raise ValueError("method can only be 'hierarchical' or 'separate'")
293      y = np.asarray(y).ravel()
294      if x is None:
295          x = np.arange(len(y), dtype=float)
296      else:
297          x = np.asarray(x, dtype=float).ravel()
298          if len(x) != len(y):
299              raise ValueError(f"Incompatible lengths ! ({len(y)}<>{len(x)})")
300      dtype = np.result_type(x, y, np.float32)  # use at least float32
301      y, x = y.astype(dtype), x.astype(dtype)
302      medslope, medinter = siegelslopes_pythran(y, x, method)
303      return SiegelslopesResult(slope=medslope, intercept=medinter)
