76   def _concat_check(tup, dtype, out):
77       if tup == ():
78           raise ValueError("need at least one array to concatenate")
79   
80       """Check inputs in concatenate et al."""
81       if out is not None and dtype is not None:
82           # mimic numpy
83           raise TypeError(
84               "concatenate() only takes `out` or `dtype` as an "
85               "argument, but both were provided."
86           )

89   def _concat_cast_helper(tensors, out=None, dtype=None, casting="same_kind"):
90       """Figure out dtypes, cast if necessary."""
91   
92       if out is not None or dtype is not None:
93           # figure out the type of the inputs and outputs
94           out_dtype = out.dtype.torch_dtype if dtype is None else dtype
95       else:
96           out_dtype = _dtypes_impl.result_type_impl(*tensors)
97   
98       # cast input arrays if necessary; do not broadcast them agains `out`
99       tensors = _util.typecast_tensors(tensors, out_dtype, casting)
100  
101      return tensors

104  def _concatenate(
105      tensors, axis=0, out=None, dtype=None, casting: Optional[CastingModes] = "same_kind"
106  ):
107      # pure torch implementation, used below and in cov/corrcoef below
108      tensors, axis = _util.axis_none_flatten(*tensors, axis=axis)

110      return torch.cat(tensors, axis)

205  def _split_helper(tensor, indices_or_sections, axis, strict=False):
206      if isinstance(indices_or_sections, int):
207          return _split_helper_int(tensor, indices_or_sections, axis, strict)
208      elif isinstance(indices_or_sections, (list, tuple)):
209          # NB: drop split=..., it only applies to split_helper_int
210          return _split_helper_list(tensor, list(indices_or_sections), axis)
211      else:
212          raise TypeError("split_helper: ", type(indices_or_sections))
213  
214  
215  def _split_helper_int(tensor, indices_or_sections, axis, strict=False):
216      if not isinstance(indices_or_sections, int):
217          raise NotImplementedError("split: indices_or_sections")
218  
219      axis = _util.normalize_axis_index(axis, tensor.ndim)
220  
221      # numpy: l%n chunks of size (l//n + 1), the rest are sized l//n
222      l, n = tensor.shape[axis], indices_or_sections
223  
224      if n <= 0:
225          raise ValueError
226  
227      if l % n == 0:
228          num, sz = n, l // n
229          lst = [sz] * num
230      else:
231          if strict:
232              raise ValueError("array split does not result in an equal division")
233  
234          num, sz = l % n, l // n + 1
235          lst = [sz] * num
236  
237      lst += [sz - 1] * (n - num)
238  
239      return torch.split(tensor, lst, axis)
240  
241  
242  def _split_helper_list(tensor, indices_or_sections, axis):
243      if not isinstance(indices_or_sections, list):
244          raise NotImplementedError("split: indices_or_sections: list")
245      # numpy expects indices, while torch expects lengths of sections
246      # also, numpy appends zero-size arrays for indices above the shape[axis]
247      lst = [x for x in indices_or_sections if x <= tensor.shape[axis]]
248      num_extra = len(indices_or_sections) - len(lst)
249  
250      lst.append(tensor.shape[axis])
251      lst = [
252          lst[0],
253      ] + [a - b for a, b in zip(lst[1:], lst[:-1])]
254      lst += [0] * num_extra
255  
256      return torch.split(tensor, lst, axis)

314  def geomspace(
315      start: ArrayLike,
316      stop: ArrayLike,
317      num=50,
318      endpoint=True,
319      dtype: Optional[DTypeLike] = None,
320      axis=0,
321  ):
322      if axis != 0 or not endpoint:
323          raise NotImplementedError
324      base = torch.pow(stop / start, 1.0 / (num - 1))
325      logbase = torch.log(base)
326      return torch.logspace(
327          torch.log(start) / logbase,
328          torch.log(stop) / logbase,
329          num,
330          base=base,

348  def arange(
349      start: Optional[ArrayLikeOrScalar] = None,
350      stop: Optional[ArrayLikeOrScalar] = None,
351      step: Optional[ArrayLikeOrScalar] = 1,
352      dtype: Optional[DTypeLike] = None,
353      *,
354      like: NotImplementedType = None,
355  ):
356      if step == 0:
357          raise ZeroDivisionError
358      if stop is None and start is None:
359          raise TypeError
360      if stop is None:
361          # XXX: this breaks if start is passed as a kwarg:
362          # arange(start=4) should raise (no stop) but doesn't
363          start, stop = 0, start
364      if start is None:
365          start = 0
366  
367      # the dtype of the result
368      if dtype is None:
369          dtype = (
370              _dtypes_impl.default_dtypes().float_dtype
371              if any(_dtypes_impl.is_float_or_fp_tensor(x) for x in (start, stop, step))
372              else _dtypes_impl.default_dtypes().int_dtype
373          )
374      work_dtype = torch.float64 if dtype.is_complex else dtype
375  
376      # RuntimeError: "lt_cpu" not implemented for 'ComplexFloat'. Fall back to eager.
377      if any(_dtypes_impl.is_complex_or_complex_tensor(x) for x in (start, stop, step)):
378          raise NotImplementedError
379  
380      if (step > 0 and start > stop) or (step < 0 and start < stop):
381          # empty range
382          return torch.empty(0, dtype=dtype)
383  
384      result = torch.arange(start, stop, step, dtype=work_dtype)
385      result = _util.cast_if_needed(result, dtype)
386      return result

506  def _xy_helper_corrcoef(x_tensor, y_tensor=None, rowvar=True):
507      """Prepare inputs for cov and corrcoef."""
508  
509      # https://github.com/numpy/numpy/blob/v1.24.0/numpy/lib/function_base.py#L2636
510      if y_tensor is not None:
511          # make sure x and y are at least 2D
512          ndim_extra = 2 - x_tensor.ndim
513          if ndim_extra > 0:
514              x_tensor = x_tensor.view((1,) * ndim_extra + x_tensor.shape)
515          if not rowvar and x_tensor.shape[0] != 1:
516              x_tensor = x_tensor.mT
517          x_tensor = x_tensor.clone()
518  
519          ndim_extra = 2 - y_tensor.ndim
520          if ndim_extra > 0:
521              y_tensor = y_tensor.view((1,) * ndim_extra + y_tensor.shape)
522          if not rowvar and y_tensor.shape[0] != 1:
523              y_tensor = y_tensor.mT
524          y_tensor = y_tensor.clone()
525  
526          x_tensor = _concatenate((x_tensor, y_tensor), axis=0)
527  
528      return x_tensor

531  def corrcoef(
532      x: ArrayLike,
533      y: Optional[ArrayLike] = None,
534      rowvar=True,
535      bias=None,
536      ddof=None,
537      *,
538      dtype: Optional[DTypeLike] = None,
539  ):
540      if bias is not None or ddof is not None:
541          # deprecated in NumPy
542          raise NotImplementedError
543      xy_tensor = _xy_helper_corrcoef(x, y, rowvar)
544  
545      is_half = (xy_tensor.dtype == torch.float16) and xy_tensor.is_cpu
546      if is_half:
547          # work around torch's "addmm_impl_cpu_" not implemented for 'Half'"
548          dtype = torch.float32
549  
550      xy_tensor = _util.cast_if_needed(xy_tensor, dtype)
551      result = torch.corrcoef(xy_tensor)
552  
553      if is_half:
554          result = result.to(torch.float16)
555  
556      return result

559  def cov(
560      m: ArrayLike,
561      y: Optional[ArrayLike] = None,
562      rowvar=True,
563      bias=False,
564      ddof=None,
565      fweights: Optional[ArrayLike] = None,
566      aweights: Optional[ArrayLike] = None,
567      *,
568      dtype: Optional[DTypeLike] = None,
569  ):
570      m = _xy_helper_corrcoef(m, y, rowvar)
571  
572      if ddof is None:
573          ddof = 1 if bias == 0 else 0
574  
575      is_half = (m.dtype == torch.float16) and m.is_cpu
576      if is_half:
577          # work around torch's "addmm_impl_cpu_" not implemented for 'Half'"
578          dtype = torch.float32
579  
580      m = _util.cast_if_needed(m, dtype)
581      result = torch.cov(m, correction=ddof, aweights=aweights, fweights=fweights)
582  
583      if is_half:
584          result = result.to(torch.float16)
585  
586      return result

589  def _conv_corr_impl(a, v, mode):
590      dt = _dtypes_impl.result_type_impl(a, v)
591      a = _util.cast_if_needed(a, dt)
592      v = _util.cast_if_needed(v, dt)
593  
594      padding = v.shape[0] - 1 if mode == "full" else mode
595  
596      if padding == "same" and v.shape[0] % 2 == 0:
597          # UserWarning: Using padding='same' with even kernel lengths and odd
598          # dilation may require a zero-padded copy of the input be created
599          # (Triggered internally at pytorch/aten/src/ATen/native/Convolution.cpp:1010.)
600          raise NotImplementedError("mode='same' and even-length weights")
601  
602      # NumPy only accepts 1D arrays; PyTorch requires 2D inputs and 3D weights
603      aa = a[None, :]
604      vv = v[None, None, :]
605  
606      result = torch.nn.functional.conv1d(aa, vv, padding=padding)
607  
608      # torch returns a 2D result, numpy returns a 1D array
609      return result[0, :]

612  def convolve(a: ArrayLike, v: ArrayLike, mode="full"):
613      # NumPy: if v is longer than a, the arrays are swapped before computation
614      if a.shape[0] < v.shape[0]:
615          a, v = v, a
616  
617      # flip the weights since numpy does and torch does not
618      v = torch.flip(v, (0,))
619  
620      return _conv_corr_impl(a, v, mode)

642  def where(
643      condition: ArrayLike,
644      x: Optional[ArrayLikeOrScalar] = None,
645      y: Optional[ArrayLikeOrScalar] = None,
646      /,
647  ):
648      if (x is None) != (y is None):
649          raise ValueError("either both or neither of x and y should be given")
650  
651      if condition.dtype != torch.bool:
652          condition = condition.to(torch.bool)
653  
654      if x is None and y is None:
655          result = torch.where(condition)
656      else:
657          result = torch.where(condition, x, y)
658      return result

896      a: NDArray,
897      indices: ArrayLike,
898      values: ArrayLike,
899      mode: NotImplementedType = "raise",
900  ):
901      v = values.type(a.dtype)
902      # If indices is larger than v, expand v to at least the size of indices. Any
903      # unnecessary trailing elements are then trimmed.
904      if indices.numel() > v.numel():
905          ratio = (indices.numel() + v.numel() - 1) // v.numel()
906          v = v.unsqueeze(0).expand((ratio,) + v.shape)
907      # Trim unnecessary elements, regardless if v was expanded or not. Note
908      # np.put() trims v to match indices by default too.
909      if indices.numel() < v.numel():
910          v = v.flatten()
911          v = v[: indices.numel()]
912      a.put_(indices, v)
913      return None

1551 # NB: have to normalize *varargs manually
1552 def gradient(f: ArrayLike, *varargs, axis=None, edge_order=1):
1553     N = f.ndim  # number of dimensions
1554 
1555     varargs = _util.ndarrays_to_tensors(varargs)
1556 
1557     if axis is None:
1558         axes = tuple(range(N))
1559     else:
1560         axes = _util.normalize_axis_tuple(axis, N)
1561 
1562     len_axes = len(axes)
1563     n = len(varargs)
1564     if n == 0:
1565         # no spacing argument - use 1 in all axes
1566         dx = [1.0] * len_axes
1567     elif n == 1 and (_dtypes_impl.is_scalar(varargs[0]) or varargs[0].ndim == 0):
1568         # single scalar or 0D tensor for all axes (np.ndim(varargs[0]) == 0)
1569         dx = varargs * len_axes
1570     elif n == len_axes:
1571         # scalar or 1d array for each axis
1572         dx = list(varargs)
1573         for i, distances in enumerate(dx):
1574             distances = torch.as_tensor(distances)
1575             if distances.ndim == 0:
1576                 continue
1577             elif distances.ndim != 1:
1578                 raise ValueError("distances must be either scalars or 1d")
1579             if len(distances) != f.shape[axes[i]]:
1580                 raise ValueError(
1581                     "when 1d, distances must match "
1582                     "the length of the corresponding dimension"
1583                 )
1584             if not (distances.dtype.is_floating_point or distances.dtype.is_complex):
1585                 distances = distances.double()
1586 
1587             diffx = torch.diff(distances)
1588             # if distances are constant reduce to the scalar case
1589             # since it brings a consistent speedup
1590             if (diffx == diffx[0]).all():
1591                 diffx = diffx[0]
1592             dx[i] = diffx
1593     else:
1594         raise TypeError("invalid number of arguments")
1595 
1596     if edge_order > 2:
1597         raise ValueError("'edge_order' greater than 2 not supported")
1598 
1599     # use central differences on interior and one-sided differences on the
1600     # endpoints. This preserves second order-accuracy over the full domain.
1601 
1602     outvals = []
1603 
1604     # create slice objects --- initially all are [:, :, ..., :]
1605     slice1 = [slice(None)] * N
1606     slice2 = [slice(None)] * N
1607     slice3 = [slice(None)] * N
1608     slice4 = [slice(None)] * N
1609 
1610     otype = f.dtype
1611     if _dtypes_impl.python_type_for_torch(otype) in (int, bool):
1612         # Convert to floating point.
1613         # First check if f is a numpy integer type; if so, convert f to float64
1614         # to avoid modular arithmetic when computing the changes in f.
1615         f = f.double()
1616         otype = torch.float64
1617 
1618     for axis, ax_dx in zip(axes, dx):
1619         if f.shape[axis] < edge_order + 1:
1620             raise ValueError(
1621                 "Shape of array too small to calculate a numerical gradient, "
1622                 "at least (edge_order + 1) elements are required."
1623             )
1624         # result allocation
1625         out = torch.empty_like(f, dtype=otype)
1626 
1627         # spacing for the current axis (NB: np.ndim(ax_dx) == 0)
1628         uniform_spacing = _dtypes_impl.is_scalar(ax_dx) or ax_dx.ndim == 0
1629 
1630         # Numerical differentiation: 2nd order interior
1631         slice1[axis] = slice(1, -1)
1632         slice2[axis] = slice(None, -2)
1633         slice3[axis] = slice(1, -1)
1634         slice4[axis] = slice(2, None)
1635 
1636         if uniform_spacing:
1637             out[tuple(slice1)] = (f[tuple(slice4)] - f[tuple(slice2)]) / (2.0 * ax_dx)
1638         else:
1639             dx1 = ax_dx[0:-1]
1640             dx2 = ax_dx[1:]
1641             a = -(dx2) / (dx1 * (dx1 + dx2))
1642             b = (dx2 - dx1) / (dx1 * dx2)
1643             c = dx1 / (dx2 * (dx1 + dx2))
1644             # fix the shape for broadcasting
1645             shape = [1] * N
1646             shape[axis] = -1
1647             a = a.reshape(shape)
1648             b = b.reshape(shape)
1649             c = c.reshape(shape)
1650             # 1D equivalent -- out[1:-1] = a * f[:-2] + b * f[1:-1] + c * f[2:]
1651             out[tuple(slice1)] = (
1652                 a * f[tuple(slice2)] + b * f[tuple(slice3)] + c * f[tuple(slice4)]
1653             )
1654 
1655         # Numerical differentiation: 1st order edges
1656         if edge_order == 1:
1657             slice1[axis] = 0
1658             slice2[axis] = 1
1659             slice3[axis] = 0
1660             dx_0 = ax_dx if uniform_spacing else ax_dx[0]
1661             # 1D equivalent -- out[0] = (f[1] - f[0]) / (x[1] - x[0])
1662             out[tuple(slice1)] = (f[tuple(slice2)] - f[tuple(slice3)]) / dx_0
1663 
1664             slice1[axis] = -1
1665             slice2[axis] = -1
1666             slice3[axis] = -2
1667             dx_n = ax_dx if uniform_spacing else ax_dx[-1]
1668             # 1D equivalent -- out[-1] = (f[-1] - f[-2]) / (x[-1] - x[-2])
1669             out[tuple(slice1)] = (f[tuple(slice2)] - f[tuple(slice3)]) / dx_n
1670 
1671         # Numerical differentiation: 2nd order edges
1672         else:
1673             slice1[axis] = 0
1674             slice2[axis] = 0
1675             slice3[axis] = 1
1676             slice4[axis] = 2
1677             if uniform_spacing:
1678                 a = -1.5 / ax_dx
1679                 b = 2.0 / ax_dx
1680                 c = -0.5 / ax_dx
1681             else:
1682                 dx1 = ax_dx[0]
1683                 dx2 = ax_dx[1]
1684                 a = -(2.0 * dx1 + dx2) / (dx1 * (dx1 + dx2))
1685                 b = (dx1 + dx2) / (dx1 * dx2)
1686                 c = -dx1 / (dx2 * (dx1 + dx2))
1687             # 1D equivalent -- out[0] = a * f[0] + b * f[1] + c * f[2]
1688             out[tuple(slice1)] = (
1689                 a * f[tuple(slice2)] + b * f[tuple(slice3)] + c * f[tuple(slice4)]
1690             )
1691 
1692             slice1[axis] = -1
1693             slice2[axis] = -3
1694             slice3[axis] = -2
1695             slice4[axis] = -1
1696             if uniform_spacing:
1697                 a = 0.5 / ax_dx
1698                 b = -2.0 / ax_dx
1699                 c = 1.5 / ax_dx
1700             else:
1701                 dx1 = ax_dx[-2]
1702                 dx2 = ax_dx[-1]
1703                 a = (dx2) / (dx1 * (dx1 + dx2))
1704                 b = -(dx2 + dx1) / (dx1 * dx2)
1705                 c = (2.0 * dx2 + dx1) / (dx2 * (dx1 + dx2))
1706             # 1D equivalent -- out[-1] = a * f[-3] + b * f[-2] + c * f[-1]
1707             out[tuple(slice1)] = (
1708                 a * f[tuple(slice2)] + b * f[tuple(slice3)] + c * f[tuple(slice4)]
1709             )
1710 
1711         outvals.append(out)
1712 
1713         # reset the slice object in this dimension to ":"
1714         slice1[axis] = slice(None)
1715         slice2[axis] = slice(None)
1716         slice3[axis] = slice(None)
1717         slice4[axis] = slice(None)
1718 
1719     if len_axes == 1:
1720         return outvals[0]
1721     else:
1722         return outvals

1955     range: Optional[ArrayLike] = None,
1956     normed=None,
1957     weights: Optional[ArrayLike] = None,
1958     density=None,
1959 ):
1960     # have to normalize manually because `sample` interpretation differs
1961     # for a list of lists and a 2D array
1962     if normed is not None:
1963         raise ValueError("normed argument is deprecated, use density= instead")
1964

1966 
1967     if isinstance(sample, (list, tuple)):
1968         sample = normalize_array_like(sample).T

1971 
1972     sample = torch.atleast_2d(sample)
1973 
1974     if not (sample.dtype.is_floating_point or sample.dtype.is_complex):
1975         sample = sample.double()

1978     bins_is_array = not (

2020         dtype = torch.bool
2021 
2022     elif a.dtype.is_complex:
2023         fi = torch.finfo(torch.float32)
2024         fits_in_single = a.dtype == torch.complex64 or (
2025             fi.min <= a.real <= fi.max and fi.min <= a.imag <= fi.max
2026         )
2027         dtype = torch.complex64 if fits_in_single else torch.complex128
2028

2033                 dtype = dt

2039             ii = torch.iinfo(dt)
