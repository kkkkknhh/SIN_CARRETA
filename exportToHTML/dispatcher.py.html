63   def variadic_signature_matches_iter(types, full_signature):
64       """Check if a set of input types matches a variadic signature. 
65       Notes 
66       ----- 
67       The algorithm is as follows: 
68       Initialize the current signature to the first in the sequence 
69       For each type in `types`: 
70           If the current signature is variadic 
71               If the type matches the signature 
72                   yield True 
73               Else 
74                   Try to get the next signature 
75                   If no signatures are left we can't possibly have a match 
76                       so yield False 
77           Else 
78               yield True if the type matches the current signature 
79               Get the next signature 
80       """
81       sigiter = iter(full_signature)
82       sig = next(sigiter)
83       for typ in types:
84           matches = issubclass(typ, sig)
85           yield matches
86           if not isvariadic(sig):
87               # we're not matching a variadic argument, so move to the next
88               # element in the signature
89               sig = next(sigiter)
90       else:
91           try:
92               sig = next(sigiter)
93           except StopIteration:
94               assert isvariadic(sig)
95               yield True
96           else:
97               # We have signature items left over, so all of our arguments
98               # haven't matched
99               yield False
100

102  def variadic_signature_matches(types, full_signature):
103      # No arguments always matches a variadic signature
104      assert full_signature
105      return all(variadic_signature_matches_iter(types, full_signature))

108  class Dispatcher:
109      """Dispatch methods based on type signature 
110      Use ``dispatch`` to add implementations 
111      Examples 
112      -------- 
113      >>> # xdoctest: +SKIP("bad import name") 
114      >>> from multipledispatch import dispatch 
115      >>> @dispatch(int) 
116      ... def f(x): 
117      ...     return x + 1 
118      >>> @dispatch(float) 
119      ... def f(x): 
120      ...     return x - 1 
121      >>> f(3) 
122      4 
123      >>> f(3.0) 
124      2.0 
125      """
126  
127      __slots__ = "__name__", "name", "funcs", "_ordering", "_cache", "doc"
128  
129      def __init__(self, name, doc=None):
130          self.name = self.__name__ = name
131          self.funcs = {}
132          self.doc = doc
133  
134          self._cache = {}
135  
136      def register(self, *types, **kwargs):
137          """register dispatcher with new implementation 
138          >>> # xdoctest: +SKIP 
139          >>> f = Dispatcher("f") 
140          >>> @f.register(int) 
141          ... def inc(x): 
142          ...     return x + 1 
143          >>> @f.register(float) 
144          ... def dec(x): 
145          ...     return x - 1 
146          >>> @f.register(list) 
147          ... @f.register(tuple) 
148          ... def reverse(x): 
149          ...     return x[::-1] 
150          >>> f(1) 
151          2 
152          >>> f(1.0) 
153          0.0 
154          >>> f([1, 2, 3]) 
155          [3, 2, 1] 
156          """
157  
158          def _df(func):
159              self.add(types, func, **kwargs)  # type: ignore[call-arg]
160              return func
161  
162          return _df
163  
164      @classmethod
165      def get_func_params(cls, func):
166          if hasattr(inspect, "signature"):
167              sig = inspect.signature(func)
168              return sig.parameters.values()
169  
170      @classmethod
171      def get_func_annotations(cls, func):
172          """get annotations of function positional parameters"""
173          params = cls.get_func_params(func)
174          if params:
175              Parameter = inspect.Parameter
176  
177              params = (
178                  param
179                  for param in params
180                  if param.kind
181                  in (Parameter.POSITIONAL_ONLY, Parameter.POSITIONAL_OR_KEYWORD)
182              )
183  
184              annotations = tuple(param.annotation for param in params)
185  
186              if all(ann is not Parameter.empty for ann in annotations):
187                  return annotations

189      def add(self, signature, func):
190          """Add new types/method pair to dispatcher 
191          >>> # xdoctest: +SKIP 
192          >>> D = Dispatcher("add") 
193          >>> D.add((int, int), lambda x, y: x + y) 
194          >>> D.add((float, float), lambda x, y: x + y) 
195          >>> D(1, 2) 
196          3 
197          >>> D(1, 2.0) 
198          Traceback (most recent call last): 
199          ... 
200          NotImplementedError: Could not find signature for add: <int, float> 
201          >>> # When ``add`` detects a warning it calls the ``on_ambiguity`` callback 
202          >>> # with a dispatcher/itself, and a set of ambiguous type signature pairs 
203          >>> # as inputs.  See ``ambiguity_warn`` for an example. 
204          """
205          # Handle annotations
206          if not signature:
207              annotations = self.get_func_annotations(func)
208              if annotations:
209                  signature = annotations
210  
211          # Handle union types
212          if any(isinstance(typ, tuple) for typ in signature):
213              for typs in expand_tuples(signature):
214                  self.add(typs, func)
215              return
216  
217          new_signature = []
218  
219          for index, typ in enumerate(signature, start=1):
220              if not isinstance(typ, (type, list)):
221                  str_sig = ", ".join(
222                      c.__name__ if isinstance(c, type) else str(c) for c in signature
223                  )
224                  raise TypeError(
225                      f"Tried to dispatch on non-type: {typ}\n"
226                      f"In signature: <{str_sig}>\n"
227                      f"In function: {self.name}"
228                  )
229  
230              # handle variadic signatures
231              if isinstance(typ, list):
232                  if index != len(signature):
233                      raise TypeError("Variadic signature must be the last element")
234  
235                  if len(typ) != 1:
236                      raise TypeError(
237                          "Variadic signature must contain exactly one element. "
238                          "To use a variadic union type place the desired types "
239                          "inside of a tuple, e.g., [(int, str)]"
240                      )
241                  new_signature.append(Variadic[typ[0]])
242              else:
243                  new_signature.append(typ)
244  
245          self.funcs[tuple(new_signature)] = func
246          self._cache.clear()
247  
248          try:
249              del self._ordering
250          except AttributeError:
251              pass
252  
253      @property
254      def ordering(self):
255          try:
256              return self._ordering
257          except AttributeError:
258              return self.reorder()
259  
260      def reorder(self, on_ambiguity=ambiguity_warn):
261          self._ordering = od = ordering(self.funcs)
262          amb = ambiguities(self.funcs)
263          if amb:
264              on_ambiguity(self, amb)
265          return od
266  
267      def __call__(self, *args, **kwargs):
268          types = tuple([type(arg) for arg in args])
269          try:
270              func = self._cache[types]
271          except KeyError as e:
272              func = self.dispatch(*types)
273              if not func:
274                  raise NotImplementedError(
275                      f"Could not find signature for {self.name}: <{str_signature(types)}>"
276                  ) from e
277              self._cache[types] = func
278          try:
279              return func(*args, **kwargs)
280  
281          except MDNotImplementedError as e:
282              funcs = self.dispatch_iter(*types)
283              next(funcs)  # burn first
284              for func in funcs:
285                  try:
286                      return func(*args, **kwargs)
287                  except MDNotImplementedError:
288                      pass
289  
290              raise NotImplementedError(
291                  "Matching functions for "
292                  f"{self.name}: <{str_signature(types)}> found, but none completed successfully",
293              ) from e
294  
295      def __str__(self):
296          return f"<dispatched {self.name}>"
297  
298      __repr__ = __str__
299  
300      def dispatch(self, *types):
301          """Determine appropriate implementation for this type signature 
302          This method is internal.  Users should call this object as a function. 
303          Implementation resolution occurs within the ``__call__`` method. 
304          >>> # xdoctest: +SKIP 
305          >>> from multipledispatch import dispatch 
306          >>> @dispatch(int) 
307          ... def inc(x): 
308          ...     return x + 1 
309          >>> implementation = inc.dispatch(int) 
310          >>> implementation(3) 
311          4 
312          >>> print(inc.dispatch(float)) 
313          None 
314          See Also: 
315            ``multipledispatch.conflict`` - module to determine resolution order 
316          """
317  
318          if types in self.funcs:
319              return self.funcs[types]
320  
321          try:
322              return next(self.dispatch_iter(*types))
323          except StopIteration:
324              return None
325  
326      def dispatch_iter(self, *types):
327          n = len(types)
328          for signature in self.ordering:
329              if len(signature) == n and all(map(issubclass, types, signature)):
330                  result = self.funcs[signature]
331                  yield result
332              elif len(signature) and isvariadic(signature[-1]):
333                  if variadic_signature_matches(types, signature):
334                      result = self.funcs[signature]
335                      yield result

347      def __getstate__(self):
348          return {"name": self.name, "funcs": self.funcs}

350      def __setstate__(self, d):
351          self.name = d["name"]
352          self.funcs = d["funcs"]
353          self._ordering = ordering(self.funcs)
354          self._cache = {}

356      @property
357      def __doc__(self):  # type: ignore[override]
358          docs = [f"Multiply dispatched method: {self.name}"]
359  
360          if self.doc:
361              docs.append(self.doc)
362  
363          other = []
364          for sig in self.ordering[::-1]:
365              func = self.funcs[sig]
366              if func.__doc__:
367                  s = f"Inputs: <{str_signature(sig)}>\n"
368                  s += "-" * len(s) + "\n"
369                  s += func.__doc__.strip()
370                  docs.append(s)
371              else:
372                  other.append(str_signature(sig))
373  
374          if other:
375              docs.append("Other signatures:\n    " + "\n    ".join(other))
376  
377          return "\n\n".join(docs)

379      def _help(self, *args):
380          return self.dispatch(*map(type, args)).__doc__
381  
382      def help(self, *args, **kwargs):
383          """Print docstring for the function corresponding to inputs"""
384          print(self._help(*args))
385  
386      def _source(self, *args):
387          func = self.dispatch(*map(type, args))
388          if not func:
389              raise TypeError("No function found")
390          return source(func)
391  
392      def source(self, *args, **kwargs):
393          """Print source code for the function corresponding to inputs"""
394          print(self._source(*args))

398      s = f"File: {inspect.getsourcefile(func)}\n\n"
399      s = s + inspect.getsource(func)
400      return s

403  class MethodDispatcher(Dispatcher):
404      """Dispatch methods based on type signature 
405      See Also: 
406          Dispatcher 
407      """
408  
409      __slots__ = ("obj", "cls")
410  
411      @classmethod
412      def get_func_params(cls, func):
413          if hasattr(inspect, "signature"):
414              sig = inspect.signature(func)
415              return itl.islice(sig.parameters.values(), 1, None)
416  
417      def __get__(self, instance, owner):
418          self.obj = instance
419          self.cls = owner
420          return self
421  
422      def __call__(self, *args, **kwargs):
423          types = tuple([type(arg) for arg in args])
424          func = self.dispatch(*types)
425          if not func:
426              raise NotImplementedError(
427                  f"Could not find signature for {self.name}: <{str_signature(types)}>"
428              )
429          return func(self.obj, *args, **kwargs)

432  def str_signature(sig):
433      """String representation of type signature 
434      >>> str_signature((int, float)) 
435      'int, float' 
436      """
437      return ", ".join(cls.__name__ for cls in sig)

440  def warning_text(name, amb):
441      """The text for ambiguity warnings"""
442      text = f"\nAmbiguities exist in dispatched function {name}\n\n"
443      text += "The following signatures may result in ambiguous behavior:\n"
444      for pair in amb:
445          text += "\t" + ", ".join("[" + str_signature(s) + "]" for s in pair) + "\n"
446      text += "\n\nConsider making the following additions:\n\n"
447      text += "\n\n".join(
448          [
449              "@dispatch(" + str_signature(super_signature(s)) + f")\ndef {name}(...)"
450              for s in amb
451          ]
452      )
453      return text
