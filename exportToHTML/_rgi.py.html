17   def _check_points(points):
18       descending_dimensions = []
19       grid = []
20       for i, p in enumerate(points):
21           # early make points float
22           # see https://github.com/scipy/scipy/pull/17230
23           p = np.asarray(p, dtype=float)
24           if not np.all(p[1:] > p[:-1]):
25               if np.all(p[1:] < p[:-1]):
26                   # input is descending, so make it ascending
27                   descending_dimensions.append(i)
28                   p = np.flip(p)
29               else:
30                   raise ValueError(
31                       "The points in dimension %d must be strictly "
32                       "ascending or descending" % i)
33           # see https://github.com/scipy/scipy/issues/17716
34           p = np.ascontiguousarray(p)
35           grid.append(p)
36       return tuple(grid), tuple(descending_dimensions)

39   def _check_dimensionality(points, values):
40       if len(points) > values.ndim:
41           raise ValueError("There are %d point arrays, but values has %d "
42                            "dimensions" % (len(points), values.ndim))
43       for i, p in enumerate(points):
44           if not np.asarray(p).ndim == 1:
45               raise ValueError("The points in dimension %d must be "
46                                "1-dimensional" % i)
47           if not values.shape[i] == len(p):
48               raise ValueError("There are %d points and %d values in "
49                                "dimension %d" % (len(p), values.shape[i], i))

52   class RegularGridInterpolator:
53       """ 
54       Interpolator on a regular or rectilinear grid in arbitrary dimensions. 
55    
56       The data must be defined on a rectilinear grid; that is, a rectangular 
57       grid with even or uneven spacing. Linear, nearest-neighbor, spline 
58       interpolations are supported. After setting up the interpolator object, 
59       the interpolation method may be chosen at each evaluation. 
60    
61       Parameters 
62       ---------- 
63       points : tuple of ndarray of float, with shapes (m1, ), ..., (mn, ) 
64           The points defining the regular grid in n dimensions. The points in 
65           each dimension (i.e. every elements of the points tuple) must be 
66           strictly ascending or descending. 
67    
68       values : array_like, shape (m1, ..., mn, ...) 
69           The data on the regular grid in n dimensions. Complex data is 
70           accepted. 
71    
72       method : str, optional 
73           The method of interpolation to perform. Supported are "linear", 
74           "nearest", "slinear", "cubic", "quintic" and "pchip". This 
75           parameter will become the default for the object's ``__call__`` 
76           method. Default is "linear". 
77    
78       bounds_error : bool, optional 
79           If True, when interpolated values are requested outside of the 
80           domain of the input data, a ValueError is raised. 
81           If False, then `fill_value` is used. 
82           Default is True. 
83    
84       fill_value : float or None, optional 
85           The value to use for points outside of the interpolation domain. 
86           If None, values outside the domain are extrapolated. 
87           Default is ``np.nan``. 
88    
89       solver : callable, optional 
90           Only used for methods "slinear", "cubic" and "quintic". 
91           Sparse linear algebra solver for construction of the NdBSpline instance. 
92           Default is the iterative solver `scipy.sparse.linalg.gcrotmk`. 
93    
94           .. versionadded:: 1.13 
95    
96       solver_args: dict, optional 
97           Additional arguments to pass to `solver`, if any. 
98    
99           .. versionadded:: 1.13 
100   
101      Methods 
102      ------- 
103      __call__ 
104   
105      Attributes 
106      ---------- 
107      grid : tuple of ndarrays 
108          The points defining the regular grid in n dimensions. 
109          This tuple defines the full grid via 
110          ``np.meshgrid(*grid, indexing='ij')`` 
111      values : ndarray 
112          Data values at the grid. 
113      method : str 
114          Interpolation method. 
115      fill_value : float or ``None`` 
116          Use this value for out-of-bounds arguments to `__call__`. 
117      bounds_error : bool 
118          If ``True``, out-of-bounds argument raise a ``ValueError``. 
119   
120      Notes 
121      ----- 
122      Contrary to `LinearNDInterpolator` and `NearestNDInterpolator`, this class 
123      avoids expensive triangulation of the input data by taking advantage of the 
124      regular grid structure. 
125   
126      In other words, this class assumes that the data is defined on a 
127      *rectilinear* grid. 
128   
129      .. versionadded:: 0.14 
130   
131      The 'slinear'(k=1), 'cubic'(k=3), and 'quintic'(k=5) methods are 
132      tensor-product spline interpolators, where `k` is the spline degree, 
133      If any dimension has fewer points than `k` + 1, an error will be raised. 
134   
135      .. versionadded:: 1.9 
136   
137      If the input data is such that dimensions have incommensurate 
138      units and differ by many orders of magnitude, the interpolant may have 
139      numerical artifacts. Consider rescaling the data before interpolating. 
140   
141      **Choosing a solver for spline methods** 
142   
143      Spline methods, "slinear", "cubic" and "quintic" involve solving a 
144      large sparse linear system at instantiation time. Depending on data, 
145      the default solver may or may not be adequate. When it is not, you may 
146      need to experiment with an optional `solver` argument, where you may 
147      choose between the direct solver (`scipy.sparse.linalg.spsolve`) or 
148      iterative solvers from `scipy.sparse.linalg`. You may need to supply 
149      additional parameters via the optional `solver_args` parameter (for instance, 
150      you may supply the starting value or target tolerance). See the 
151      `scipy.sparse.linalg` documentation for the full list of available options. 
152   
153      Alternatively, you may instead use the legacy methods, "slinear_legacy", 
154      "cubic_legacy" and "quintic_legacy". These methods allow faster construction 
155      but evaluations will be much slower. 
156   
157      Examples 
158      -------- 
159      **Evaluate a function on the points of a 3-D grid** 
160   
161      As a first example, we evaluate a simple example function on the points of 
162      a 3-D grid: 
163   
164      >>> from scipy.interpolate import RegularGridInterpolator 
165      >>> import numpy as np 
166      >>> def f(x, y, z): 
167      ...     return 2 * x**3 + 3 * y**2 - z 
168      >>> x = np.linspace(1, 4, 11) 
169      >>> y = np.linspace(4, 7, 22) 
170      >>> z = np.linspace(7, 9, 33) 
171      >>> xg, yg ,zg = np.meshgrid(x, y, z, indexing='ij', sparse=True) 
172      >>> data = f(xg, yg, zg) 
173   
174      ``data`` is now a 3-D array with ``data[i, j, k] = f(x[i], y[j], z[k])``. 
175      Next, define an interpolating function from this data: 
176   
177      >>> interp = RegularGridInterpolator((x, y, z), data) 
178   
179      Evaluate the interpolating function at the two points 
180      ``(x,y,z) = (2.1, 6.2, 8.3)`` and ``(3.3, 5.2, 7.1)``: 
181   
182      >>> pts = np.array([[2.1, 6.2, 8.3], 
183      ...                 [3.3, 5.2, 7.1]]) 
184      >>> interp(pts) 
185      array([ 125.80469388,  146.30069388]) 
186   
187      which is indeed a close approximation to 
188   
189      >>> f(2.1, 6.2, 8.3), f(3.3, 5.2, 7.1) 
190      (125.54200000000002, 145.894) 
191   
192      **Interpolate and extrapolate a 2D dataset** 
193   
194      As a second example, we interpolate and extrapolate a 2D data set: 
195   
196      >>> x, y = np.array([-2, 0, 4]), np.array([-2, 0, 2, 5]) 
197      >>> def ff(x, y): 
198      ...     return x**2 + y**2 
199   
200      >>> xg, yg = np.meshgrid(x, y, indexing='ij') 
201      >>> data = ff(xg, yg) 
202      >>> interp = RegularGridInterpolator((x, y), data, 
203      ...                                  bounds_error=False, fill_value=None) 
204   
205      >>> import matplotlib.pyplot as plt 
206      >>> fig = plt.figure() 
207      >>> ax = fig.add_subplot(projection='3d') 
208      >>> ax.scatter(xg.ravel(), yg.ravel(), data.ravel(), 
209      ...            s=60, c='k', label='data') 
210   
211      Evaluate and plot the interpolator on a finer grid 
212   
213      >>> xx = np.linspace(-4, 9, 31) 
214      >>> yy = np.linspace(-4, 9, 31) 
215      >>> X, Y = np.meshgrid(xx, yy, indexing='ij') 
216   
217      >>> # interpolator 
218      >>> ax.plot_wireframe(X, Y, interp((X, Y)), rstride=3, cstride=3, 
219      ...                   alpha=0.4, color='m', label='linear interp') 
220   
221      >>> # ground truth 
222      >>> ax.plot_wireframe(X, Y, ff(X, Y), rstride=3, cstride=3, 
223      ...                   alpha=0.4, label='ground truth') 
224      >>> plt.legend() 
225      >>> plt.show() 
226   
227      Other examples are given 
228      :ref:`in the tutorial <tutorial-interpolate_regular_grid_interpolator>`. 
229   
230      See Also 
231      -------- 
232      NearestNDInterpolator : Nearest neighbor interpolator on *unstructured* 
233                              data in N dimensions 
234   
235      LinearNDInterpolator : Piecewise linear interpolator on *unstructured* data 
236                             in N dimensions 
237   
238      interpn : a convenience function which wraps `RegularGridInterpolator` 
239   
240      scipy.ndimage.map_coordinates : interpolation on grids with equal spacing 
241                                      (suitable for e.g., N-D image resampling) 
242

246             https://pypi.python.org/pypi/regulargrid/

259      _SPLINE_DEGREE_MAP = {"slinear": 1, "cubic": 3, "quintic": 5, 'pchip': 3,
260                            "slinear_legacy": 1, "cubic_legacy": 3, "quintic_legacy": 5,}
261      _SPLINE_METHODS_recursive = {"slinear_legacy", "cubic_legacy",
262                                  "quintic_legacy", "pchip"}
263      _SPLINE_METHODS_ndbspl = {"slinear", "cubic", "quintic"}
264      _SPLINE_METHODS = list(_SPLINE_DEGREE_MAP.keys())
265      _ALL_METHODS = ["linear", "nearest"] + _SPLINE_METHODS

267      def __init__(self, points, values, method="linear", bounds_error=True,
268                   fill_value=np.nan, *, solver=None, solver_args=None):
269          if method not in self._ALL_METHODS:
270              raise ValueError(f"Method '{method}' is not defined")
271          elif method in self._SPLINE_METHODS:
272              self._validate_grid_dimensions(points, method)
273          self.method = method
274          self._spline = None
275          self.bounds_error = bounds_error
276          self.grid, self._descending_dimensions = _check_points(points)
277          self.values = self._check_values(values)
278          self._check_dimensionality(self.grid, self.values)
279          self.fill_value = self._check_fill_value(self.values, fill_value)
280          if self._descending_dimensions:
281              self.values = np.flip(values, axis=self._descending_dimensions)
282          if self.method == "pchip" and np.iscomplexobj(self.values):
283              msg = ("`PchipInterpolator` only works with real values. If you are trying "
284                     "to use the real components of the passed array, use `np.real` on "
285                     "the array before passing to `RegularGridInterpolator`.")
286              raise ValueError(msg)
287          if method in self._SPLINE_METHODS_ndbspl:
288              if solver_args is None:
289                  solver_args = {}
290              self._spline = self._construct_spline(method, solver, **solver_args)
291          else:
292              if solver is not None or solver_args:
293                  raise ValueError(
294                      f"{method =} does not accept the 'solver' argument. Got "
295                      f" {solver = } and with arguments {solver_args}."
296                  )

298      def _construct_spline(self, method, solver=None, **solver_args):
299          if solver is None:
300              solver = ssl.gcrotmk
301          spl = make_ndbspl(
302                  self.grid, self.values, self._SPLINE_DEGREE_MAP[method],
303                  solver=solver, **solver_args
304                )
305          return spl

313      def _check_values(self, values):
314          if not hasattr(values, 'ndim'):
315              # allow reasonable duck-typed values
316              values = np.asarray(values)
317  
318          if hasattr(values, 'dtype') and hasattr(values, 'astype'):
319              if not np.issubdtype(values.dtype, np.inexact):
320                  values = values.astype(float)
321  
322          return values

324      def _check_fill_value(self, values, fill_value):
325          if fill_value is not None:
326              fill_value_dtype = np.asarray(fill_value).dtype
327              if (hasattr(values, 'dtype') and not
328                      np.can_cast(fill_value_dtype, values.dtype,
329                                  casting='same_kind')):
330                  raise ValueError("fill_value must be either 'None' or "
331                                   "of a type compatible with values")
332          return fill_value

334      def __call__(self, xi, method=None, *, nu=None):
335          """ 
336          Interpolation at coordinates. 
337   
338          Parameters 
339          ---------- 
340          xi : ndarray of shape (..., ndim) 
341              The coordinates to evaluate the interpolator at. 
342   
343          method : str, optional 
344              The method of interpolation to perform. Supported are "linear", 
345              "nearest", "slinear", "cubic", "quintic" and "pchip". Default is 
346              the method chosen when the interpolator was created. 
347   
348          nu : sequence of ints, length ndim, optional 
349              If not None, the orders of the derivatives to evaluate. 
350              Each entry must be non-negative. 
351              Only allowed for methods "slinear", "cubic" and "quintic". 
352   
353              .. versionadded:: 1.13 
354   
355          Returns 
356          ------- 
357          values_x : ndarray, shape xi.shape[:-1] + values.shape[ndim:] 
358              Interpolated values at `xi`. See notes for behaviour when 
359              ``xi.ndim == 1``. 
360   
361          Notes 
362          ----- 
363          In the case that ``xi.ndim == 1`` a new axis is inserted into 
364          the 0 position of the returned array, values_x, so its shape is 
365          instead ``(1,) + values.shape[ndim:]``. 
366   
367          Examples 
368          -------- 
369          Here we define a nearest-neighbor interpolator of a simple function 
370   
371          >>> import numpy as np 
372          >>> x, y = np.array([0, 1, 2]), np.array([1, 3, 7]) 
373          >>> def f(x, y): 
374          ...     return x**2 + y**2 
375          >>> data = f(*np.meshgrid(x, y, indexing='ij', sparse=True)) 
376          >>> from scipy.interpolate import RegularGridInterpolator 
377          >>> interp = RegularGridInterpolator((x, y), data, method='nearest') 
378   
379          By construction, the interpolator uses the nearest-neighbor 
380          interpolation 
381   
382          >>> interp([[1.5, 1.3], [0.3, 4.5]]) 
383          array([2., 9.]) 
384   
385          We can however evaluate the linear interpolant by overriding the 
386          `method` parameter 
387   
388          >>> interp([[1.5, 1.3], [0.3, 4.5]], method='linear') 
389          array([ 4.7, 24.3]) 
390          """
391          _spline = self._spline
392          method = self.method if method is None else method
393          is_method_changed = self.method != method
394          if method not in self._ALL_METHODS:
395              raise ValueError(f"Method '{method}' is not defined")
396          if is_method_changed and method in self._SPLINE_METHODS_ndbspl:
397              _spline = self._construct_spline(method)
398  
399          if nu is not None and method not in self._SPLINE_METHODS_ndbspl:
400              raise ValueError(
401                  f"Can only compute derivatives for methods "
402                  f"{self._SPLINE_METHODS_ndbspl}, got {method =}."
403              )
404  
405          xi, xi_shape, ndim, nans, out_of_bounds = self._prepare_xi(xi)
406  
407          if method == "linear":
408              indices, norm_distances = self._find_indices(xi.T)
409              if (ndim == 2 and hasattr(self.values, 'dtype') and
410                      self.values.ndim == 2 and self.values.flags.writeable and
411                      self.values.dtype in (np.float64, np.complex128) and
412                      self.values.dtype.byteorder == '='):
413                  # until cython supports const fused types, the fast path
414                  # cannot support non-writeable values
415                  # a fast path
416                  out = np.empty(indices.shape[1], dtype=self.values.dtype)
417                  result = evaluate_linear_2d(self.values,
418                                              indices,
419                                              norm_distances,
420                                              self.grid,
421                                              out)
422              else:
423                  result = self._evaluate_linear(indices, norm_distances)
424          elif method == "nearest":
425              indices, norm_distances = self._find_indices(xi.T)
426              result = self._evaluate_nearest(indices, norm_distances)
427          elif method in self._SPLINE_METHODS:
428              if is_method_changed:
429                  self._validate_grid_dimensions(self.grid, method)
430              if method in self._SPLINE_METHODS_recursive:
431                  result = self._evaluate_spline(xi, method)
432              else:
433                  result = _spline(xi, nu=nu)
434  
435          if not self.bounds_error and self.fill_value is not None:
436              result[out_of_bounds] = self.fill_value
437  
438          # f(nan) = nan, if any
439          if np.any(nans):
440              result[nans] = np.nan
441          return result.reshape(xi_shape[:-1] + self.values.shape[ndim:])
442  
443      def _prepare_xi(self, xi):
444          ndim = len(self.grid)
445          xi = _ndim_coords_from_arrays(xi, ndim=ndim)
446          if xi.shape[-1] != len(self.grid):
447              raise ValueError("The requested sample points xi have dimension "
448                               f"{xi.shape[-1]} but this "
449                               f"RegularGridInterpolator has dimension {ndim}")
450  
451          xi_shape = xi.shape
452          xi = xi.reshape(-1, xi_shape[-1])
453          xi = np.asarray(xi, dtype=float)
454  
455          # find nans in input
456          nans = np.any(np.isnan(xi), axis=-1)
457  
458          if self.bounds_error:
459              for i, p in enumerate(xi.T):
460                  if not np.logical_and(np.all(self.grid[i][0] <= p),
461                                        np.all(p <= self.grid[i][-1])):
462                      raise ValueError("One of the requested xi is out of bounds "
463                                       "in dimension %d" % i)
464              out_of_bounds = None
465          else:
466              out_of_bounds = self._find_out_of_bounds(xi.T)
467  
468          return xi, xi_shape, ndim, nans, out_of_bounds

470      def _evaluate_linear(self, indices, norm_distances):
471          # slice for broadcasting over trailing dimensions in self.values
472          vslice = (slice(None),) + (None,)*(self.values.ndim - len(indices))
473  
474          # Compute shifting up front before zipping everything together
475          shift_norm_distances = [1 - yi for yi in norm_distances]
476          shift_indices = [i + 1 for i in indices]
477  
478          # The formula for linear interpolation in 2d takes the form:
479          # values = self.values[(i0, i1)] * (1 - y0) * (1 - y1) + \
480          #          self.values[(i0, i1 + 1)] * (1 - y0) * y1 + \
481          #          self.values[(i0 + 1, i1)] * y0 * (1 - y1) + \
482          #          self.values[(i0 + 1, i1 + 1)] * y0 * y1
483          # We pair i with 1 - yi (zipped1) and i + 1 with yi (zipped2)
484          zipped1 = zip(indices, shift_norm_distances)
485          zipped2 = zip(shift_indices, norm_distances)
486  
487          # Take all products of zipped1 and zipped2 and iterate over them
488          # to get the terms in the above formula. This corresponds to iterating
489          # over the vertices of a hypercube.
490          hypercube = itertools.product(*zip(zipped1, zipped2))
491          value = np.array([0.])
492          for h in hypercube:
493              edge_indices, weights = zip(*h)
494              weight = np.array([1.])
495              for w in weights:
496                  weight = weight * w
497              term = np.asarray(self.values[edge_indices]) * weight[vslice]
498              value = value + term   # cannot use += because broadcasting
499          return value
500  
501      def _evaluate_nearest(self, indices, norm_distances):
502          idx_res = [np.where(yi <= .5, i, i + 1)
503                     for i, yi in zip(indices, norm_distances)]
504          return self.values[tuple(idx_res)]
505  
506      def _validate_grid_dimensions(self, points, method):
507          k = self._SPLINE_DEGREE_MAP[method]
508          for i, point in enumerate(points):
509              ndim = len(np.atleast_1d(point))
510              if ndim <= k:
511                  raise ValueError(f"There are {ndim} points in dimension {i},"
512                                   f" but method {method} requires at least "
513                                   f" {k+1} points per dimension.")
514  
515      def _evaluate_spline(self, xi, method):
516          # ensure xi is 2D list of points to evaluate (`m` is the number of
517          # points and `n` is the number of interpolation dimensions,
518          # ``n == len(self.grid)``.)
519          if xi.ndim == 1:
520              xi = xi.reshape((1, xi.size))
521          m, n = xi.shape
522  
523          # Reorder the axes: n-dimensional process iterates over the
524          # interpolation axes from the last axis downwards: E.g. for a 4D grid
525          # the order of axes is 3, 2, 1, 0. Each 1D interpolation works along
526          # the 0th axis of its argument array (for 1D routine it's its ``y``
527          # array). Thus permute the interpolation axes of `values` *and keep
528          # trailing dimensions trailing*.
529          axes = tuple(range(self.values.ndim))
530          axx = axes[:n][::-1] + axes[n:]
531          values = self.values.transpose(axx)
532  
533          if method == 'pchip':
534              _eval_func = self._do_pchip
535          else:
536              _eval_func = self._do_spline_fit
537          k = self._SPLINE_DEGREE_MAP[method]
538  
539          # Non-stationary procedure: difficult to vectorize this part entirely
540          # into numpy-level operations. Unfortunately this requires explicit
541          # looping over each point in xi.
542  
543          # can at least vectorize the first pass across all points in the
544          # last variable of xi.
545          last_dim = n - 1
546          first_values = _eval_func(self.grid[last_dim],
547                                    values,
548                                    xi[:, last_dim],
549                                    k)
550  
551          # the rest of the dimensions have to be on a per point-in-xi basis
552          shape = (m, *self.values.shape[n:])
553          result = np.empty(shape, dtype=self.values.dtype)
554          for j in range(m):
555              # Main process: Apply 1D interpolate in each dimension
556              # sequentially, starting with the last dimension.
557              # These are then "folded" into the next dimension in-place.
558              folded_values = first_values[j, ...]
559              for i in range(last_dim-1, -1, -1):
560                  # Interpolate for each 1D from the last dimensions.
561                  # This collapses each 1D sequence into a scalar.
562                  folded_values = _eval_func(self.grid[i],
563                                             folded_values,
564                                             xi[j, i],
565                                             k)
566              result[j, ...] = folded_values
567  
568          return result

571      def _do_spline_fit(x, y, pt, k):
572          local_interp = make_interp_spline(x, y, k=k, axis=0)
573          values = local_interp(pt)
574          return values
575  
576      @staticmethod
577      def _do_pchip(x, y, pt, k):
578          local_interp = PchipInterpolator(x, y, axis=0)
579          values = local_interp(pt)
580          return values
581  
582      def _find_indices(self, xi):
583          return find_indices(self.grid, xi)
584  
585      def _find_out_of_bounds(self, xi):
586          # check for out of bounds xi
587          out_of_bounds = np.zeros((xi.shape[1]), dtype=bool)
588          # iterate through dimensions
589          for x, grid in zip(xi, self.grid):
590              out_of_bounds += x < grid[0]
591              out_of_bounds += x > grid[-1]
592          return out_of_bounds

595  def interpn(points, values, xi, method="linear", bounds_error=True,
596              fill_value=np.nan):
597      """ 
598      Multidimensional interpolation on regular or rectilinear grids. 
599   
600      Strictly speaking, not all regular grids are supported - this function 
601      works on *rectilinear* grids, that is, a rectangular grid with even or 
602      uneven spacing. 
603   
604      Parameters 
605      ---------- 
606      points : tuple of ndarray of float, with shapes (m1, ), ..., (mn, ) 
607          The points defining the regular grid in n dimensions. The points in 
608          each dimension (i.e. every elements of the points tuple) must be 
609          strictly ascending or descending. 
610   
611      values : array_like, shape (m1, ..., mn, ...) 
612          The data on the regular grid in n dimensions. Complex data is 
613          accepted. 
614   
615          .. deprecated:: 1.13.0 
616              Complex data is deprecated with ``method="pchip"`` and will raise an 
617              error in SciPy 1.15.0. This is because ``PchipInterpolator`` only 
618              works with real values. If you are trying to use the real components of 
619              the passed array, use ``np.real`` on ``values``. 
620   
621      xi : ndarray of shape (..., ndim) 
622          The coordinates to sample the gridded data at 
623   
624      method : str, optional 
625          The method of interpolation to perform. Supported are "linear", 
626          "nearest", "slinear", "cubic", "quintic", "pchip", and "splinef2d". 
627          "splinef2d" is only supported for 2-dimensional data. 
628   
629      bounds_error : bool, optional 
630          If True, when interpolated values are requested outside of the 
631          domain of the input data, a ValueError is raised. 
632          If False, then `fill_value` is used. 
633   
634      fill_value : number, optional 
635          If provided, the value to use for points outside of the 
636          interpolation domain. If None, values outside 
637          the domain are extrapolated.  Extrapolation is not supported by method 
638          "splinef2d". 
639   
640      Returns 
641      ------- 
642      values_x : ndarray, shape xi.shape[:-1] + values.shape[ndim:] 
643          Interpolated values at `xi`. See notes for behaviour when 
644          ``xi.ndim == 1``. 
645   
646      See Also 
647      -------- 
648      NearestNDInterpolator : Nearest neighbor interpolation on unstructured 
649                              data in N dimensions 
650      LinearNDInterpolator : Piecewise linear interpolant on unstructured data 
651                             in N dimensions 
652      RegularGridInterpolator : interpolation on a regular or rectilinear grid 
653                                in arbitrary dimensions (`interpn` wraps this 
654                                class). 
655      RectBivariateSpline : Bivariate spline approximation over a rectangular mesh 
656      scipy.ndimage.map_coordinates : interpolation on grids with equal spacing 
657                                      (suitable for e.g., N-D image resampling) 
658   
659      Notes 
660      ----- 
661   
662      .. versionadded:: 0.14 
663   
664      In the case that ``xi.ndim == 1`` a new axis is inserted into 
665      the 0 position of the returned array, values_x, so its shape is 
666      instead ``(1,) + values.shape[ndim:]``. 
667   
668      If the input data is such that input dimensions have incommensurate 
669      units and differ by many orders of magnitude, the interpolant may have 
670      numerical artifacts. Consider rescaling the data before interpolation. 
671   
672      Examples 
673      -------- 
674      Evaluate a simple example function on the points of a regular 3-D grid: 
675   
676      >>> import numpy as np 
677      >>> from scipy.interpolate import interpn 
678      >>> def value_func_3d(x, y, z): 
679      ...     return 2 * x + 3 * y - z 
680      >>> x = np.linspace(0, 4, 5) 
681      >>> y = np.linspace(0, 5, 6) 
682      >>> z = np.linspace(0, 6, 7) 
683      >>> points = (x, y, z) 
684      >>> values = value_func_3d(*np.meshgrid(*points, indexing='ij')) 
685   
686      Evaluate the interpolating function at a point 
687   
688      >>> point = np.array([2.21, 3.12, 1.15]) 
689      >>> print(interpn(points, values, point)) 
690      [12.63] 
691   
692      """
693      # sanity check 'method' kwarg
694      if method not in ["linear", "nearest", "cubic", "quintic", "pchip",
695                        "splinef2d", "slinear",
696                        "slinear_legacy", "cubic_legacy", "quintic_legacy"]:
697          raise ValueError("interpn only understands the methods 'linear', "
698                           "'nearest', 'slinear', 'cubic', 'quintic', 'pchip', "
699                           f"and 'splinef2d'. You provided {method}.")
700  
701      if not hasattr(values, 'ndim'):
702          values = np.asarray(values)
703  
704      ndim = values.ndim
705      if ndim > 2 and method == "splinef2d":
706          raise ValueError("The method splinef2d can only be used for "
707                           "2-dimensional input data")
708      if not bounds_error and fill_value is None and method == "splinef2d":
709          raise ValueError("The method splinef2d does not support extrapolation.")
710  
711      # sanity check consistency of input dimensions
712      if len(points) > ndim:
713          raise ValueError("There are %d point arrays, but values has %d "
714                           "dimensions" % (len(points), ndim))
715      if len(points) != ndim and method == 'splinef2d':
716          raise ValueError("The method splinef2d can only be used for "
717                           "scalar data with one point per coordinate")
718  
719      grid, descending_dimensions = _check_points(points)
720      _check_dimensionality(grid, values)
721  
722      # sanity check requested xi
723      xi = _ndim_coords_from_arrays(xi, ndim=len(grid))
724      if xi.shape[-1] != len(grid):
725          raise ValueError("The requested sample points xi have dimension "
726                           "%d, but this RegularGridInterpolator has "
727                           "dimension %d" % (xi.shape[-1], len(grid)))
728  
729      if bounds_error:
730          for i, p in enumerate(xi.T):
731              if not np.logical_and(np.all(grid[i][0] <= p),
732                                    np.all(p <= grid[i][-1])):
733                  raise ValueError("One of the requested xi is out of bounds "
734                                   "in dimension %d" % i)
735  
736      # perform interpolation
737      if method in RegularGridInterpolator._ALL_METHODS:
738          interp = RegularGridInterpolator(points, values, method=method,
739                                           bounds_error=bounds_error,
740                                           fill_value=fill_value)
741          return interp(xi)
742      elif method == "splinef2d":
743          xi_shape = xi.shape
744          xi = xi.reshape(-1, xi.shape[-1])
745  
746          # RectBivariateSpline doesn't support fill_value; we need to wrap here
747          idx_valid = np.all((grid[0][0] <= xi[:, 0], xi[:, 0] <= grid[0][-1],
748                              grid[1][0] <= xi[:, 1], xi[:, 1] <= grid[1][-1]),
749                             axis=0)
750          result = np.empty_like(xi[:, 0])
751  
752          # make a copy of values for RectBivariateSpline
753          interp = RectBivariateSpline(points[0], points[1], values[:])
754          result[idx_valid] = interp.ev(xi[idx_valid, 0], xi[idx_valid, 1])
755          result[np.logical_not(idx_valid)] = fill_value
756  
757          return result.reshape(xi_shape[:-1])
758      else:
759          raise ValueError(f"unknown {method = }")
