9    def use_reflection(sign_n_even=None, reflection_fun=None):
10       # - If reflection_fun is not specified, reflects negative `z` and multiplies
11       #   output by appropriate sign (indicated by `sign_n_even`).
12       # - If reflection_fun is specified, calls `reflection_fun` instead of `fun`.
13       # See DLMF 10.47(v) https://dlmf.nist.gov/10.47
14       def decorator(fun):
15           def standard_reflection(n, z, derivative):
16               # sign_n_even indicates the sign when the order `n` is even
17               sign = np.where(n % 2 == 0, sign_n_even, -sign_n_even)
18               # By the chain rule, differentiation at `-z` adds a minus sign
19               sign = -sign if derivative else sign
20               # Evaluate at positive z (minus negative z) and adjust the sign
21               return fun(n, -z, derivative) * sign
22   
23           @wraps(fun)
24           def wrapper(n, z, derivative=False):
25               z = np.asarray(z)
26   
27               if np.issubdtype(z.dtype, np.complexfloating):
28                   return fun(n, z, derivative)  # complex dtype just works
29   
30               f2 = standard_reflection if reflection_fun is None else reflection_fun
31               return _lazywhere(z.real >= 0, (n, z),
32                                 f=lambda n, z: fun(n, z, derivative),
33                                 f2=lambda n, z: f2(n, z, derivative))[()]
34           return wrapper
35       return decorator

38   @use_reflection(+1)  # See DLMF 10.47(v) https://dlmf.nist.gov/10.47
39   def spherical_jn(n, z, derivative=False):
40       r"""Spherical Bessel function of the first kind or its derivative. 
41    
42       Defined as [1]_, 
43    
44       .. math:: j_n(z) = \sqrt{\frac{\pi}{2z}} J_{n + 1/2}(z), 
45    
46       where :math:`J_n` is the Bessel function of the first kind. 
47    
48       Parameters 
49       ---------- 
50       n : int, array_like 
51           Order of the Bessel function (n >= 0). 
52       z : complex or float, array_like 
53           Argument of the Bessel function. 
54       derivative : bool, optional 
55           If True, the value of the derivative (rather than the function 
56           itself) is returned. 
57    
58       Returns 
59       ------- 
60       jn : ndarray 
61    
62       Notes 
63       ----- 
64       For real arguments greater than the order, the function is computed 
65       using the ascending recurrence [2]_. For small real or complex 
66       arguments, the definitional relation to the cylindrical Bessel function 
67       of the first kind is used. 
68    
69       The derivative is computed using the relations [3]_, 
70    
71       .. math:: 
72           j_n'(z) = j_{n-1}(z) - \frac{n + 1}{z} j_n(z). 
73    
74           j_0'(z) = -j_1(z) 
75    
76    
77       .. versionadded:: 0.18.0 
78    
79       References 
80       ---------- 
81       .. [1] https://dlmf.nist.gov/10.47.E3 
82       .. [2] https://dlmf.nist.gov/10.51.E1 
83       .. [3] https://dlmf.nist.gov/10.51.E2 
84       .. [AS] Milton Abramowitz and Irene A. Stegun, eds. 
85           Handbook of Mathematical Functions with Formulas, 
86           Graphs, and Mathematical Tables. New York: Dover, 1972. 
87    
88       Examples 
89       -------- 
90       The spherical Bessel functions of the first kind :math:`j_n` accept 
91       both real and complex second argument. They can return a complex type: 
92    
93       >>> from scipy.special import spherical_jn 
94       >>> spherical_jn(0, 3+5j) 
95       (-9.878987731663194-8.021894345786002j) 
96       >>> type(spherical_jn(0, 3+5j)) 
97       <class 'numpy.complex128'> 
98    
99       We can verify the relation for the derivative from the Notes 
100      for :math:`n=3` in the interval :math:`[1, 2]`: 
101   
102      >>> import numpy as np 
103      >>> x = np.arange(1.0, 2.0, 0.01) 
104      >>> np.allclose(spherical_jn(3, x, True), 
105      ...             spherical_jn(2, x) - 4/x * spherical_jn(3, x)) 
106      True 
107   
108      The first few :math:`j_n` with real argument: 
109   
110      >>> import matplotlib.pyplot as plt 
111      >>> x = np.arange(0.0, 10.0, 0.01) 
112      >>> fig, ax = plt.subplots() 
113      >>> ax.set_ylim(-0.5, 1.5) 
114      >>> ax.set_title(r'Spherical Bessel functions $j_n$') 
115      >>> for n in np.arange(0, 4): 
116      ...     ax.plot(x, spherical_jn(n, x), label=rf'$j_{n}$') 
117      >>> plt.legend(loc='best') 
118      >>> plt.show() 
119   
120      """
121      n = np.asarray(n, dtype=np.dtype("long"))
122      if derivative:
123          return _spherical_jn_d(n, z)
124      else:
125          return _spherical_jn(n, z)
