15       def test_str(self):
16           svals = [0.0, -0.0, 1, -1, np.inf, -np.inf, np.nan]
17           styps = [np.float16, np.float32, np.float64, np.longdouble]
18           wanted = [
19                ['0.0',  '0.0',  '0.0',  '0.0' ],
20                ['-0.0', '-0.0', '-0.0', '-0.0'],
21                ['1.0',  '1.0',  '1.0',  '1.0' ],
22                ['-1.0', '-1.0', '-1.0', '-1.0'],
23                ['inf',  'inf',  'inf',  'inf' ],
24                ['-inf', '-inf', '-inf', '-inf'],
25                ['nan',  'nan',  'nan',  'nan']]
26   
27           for wants, val in zip(wanted, svals):
28               for want, styp in zip(wants, styps):
29                   msg = 'for str({}({}))'.format(np.dtype(styp).name, repr(val))
30                   assert_equal(str(styp(val)), want, err_msg=msg)

32       def test_scalar_cutoffs(self):
33           # test that both the str and repr of np.float64 behaves
34           # like python floats in python3.
35           def check(v):
36               assert_equal(str(np.float64(v)), str(v))
37               assert_equal(str(np.float64(v)), repr(v))
38               assert_equal(repr(np.float64(v)), f"np.float64({v!r})")
39               assert_equal(repr(np.float64(v)), f"np.float64({v})")
40   
41           # check we use the same number of significant digits
42           check(1.12345678901234567890)
43           check(0.0112345678901234567890)
44   
45           # check switch from scientific output to positional and back
46           check(1e-5)
47           check(1e-4)
48           check(1e15)
49           check(1e16)

51       def test_py2_float_print(self):
52           # gh-10753
53           # In python2, the python float type implements an obsolete method
54           # tp_print, which overrides tp_repr and tp_str when using "print" to
55           # output to a "real file" (ie, not a StringIO). Make sure we don't
56           # inherit it.
57           x = np.double(0.1999999999999)
58           with TemporaryFile('r+t') as f:
59               print(x, file=f)
60               f.seek(0)
61               output = f.read()
62           assert_equal(output, str(x) + '\n')
63           # In python2 the value float('0.1999999999999') prints with reduced
64           # precision as '0.2', but we want numpy's np.double('0.1999999999999')
65           # to print the unique value, '0.1999999999999'.
66   
67           # gh-11031
68           # Only in the python2 interactive shell and when stdout is a "real"
69           # file, the output of the last command is printed to stdout without
70           # Py_PRINT_RAW (unlike the print statement) so `>>> x` and `>>> print
71           # x` are potentially different. Make sure they are the same. The only
72           # way I found to get prompt-like output is using an actual prompt from
73           # the 'code' module. Again, must use tempfile to get a "real" file.
74   
75           # dummy user-input which enters one line and then ctrl-Ds.
76           def userinput():
77               yield 'np.sqrt(2)'
78               raise EOFError
79           gen = userinput()
80           input_func = lambda prompt="": next(gen)
81   
82           with TemporaryFile('r+t') as fo, TemporaryFile('r+t') as fe:
83               orig_stdout, orig_stderr = sys.stdout, sys.stderr
84               sys.stdout, sys.stderr = fo, fe
85   
86               code.interact(local={'np': np}, readfunc=input_func, banner='')
87   
88               sys.stdout, sys.stderr = orig_stdout, orig_stderr
89   
90               fo.seek(0)
91               capture = fo.read().strip()
92   
93           assert_equal(capture, repr(np.sqrt(2)))
