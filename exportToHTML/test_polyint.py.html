20   def check_shape(interpolator_cls, x_shape, y_shape, deriv_shape=None, axis=0,
21                   extra_args=None):
22       if extra_args is None:
23           extra_args = {}
24       rng = np.random.RandomState(1234)
25   
26       x = [-1, 0, 1, 2, 3, 4]
27       s = list(range(1, len(y_shape)+1))
28       s.insert(axis % (len(y_shape)+1), 0)
29       y = rng.rand(*((6,) + y_shape)).transpose(s)
30   
31       xi = np.zeros(x_shape)
32       if interpolator_cls is CubicHermiteSpline:
33           dydx = rng.rand(*((6,) + y_shape)).transpose(s)
34           yi = interpolator_cls(x, y, dydx, axis=axis, **extra_args)(xi)
35       else:
36           yi = interpolator_cls(x, y, axis=axis, **extra_args)(xi)
37   
38       target_shape = ((deriv_shape or ()) + y.shape[:axis]
39                       + x_shape + y.shape[axis:][1:])
40       assert yi.shape == target_shape
41   
42       # check it works also with lists
43       if x_shape and y.size > 0:
44           if interpolator_cls is CubicHermiteSpline:
45               interpolator_cls(list(x), list(y), list(dydx), axis=axis,
46                                **extra_args)(list(xi))
47           else:
48               interpolator_cls(list(x), list(y), axis=axis,
49                                **extra_args)(list(xi))
50   
51       # check also values
52       if xi.size > 0 and deriv_shape is None:
53           bs_shape = y.shape[:axis] + (1,)*len(x_shape) + y.shape[axis:][1:]
54           yv = y[((slice(None,),)*(axis % y.ndim)) + (1,)]
55           yv = yv.reshape(bs_shape)
56   
57           yi, y = np.broadcast_arrays(yi, yv)
58           xp_assert_close(yi, y)

486      def test_large_chebyshev(self):
487          # The weights for Chebyshev points of the second kind have analytically
488          # solvable weights. Naive calculation of barycentric weights will fail
489          # for large N because of numerical underflow and overflow. We test
490          # correctness for large N against analytical Chebyshev weights.
491  
492          # Without capacity scaling or permutation, n=800 fails,
493          # With just capacity scaling, n=1097 fails
494          # With both capacity scaling and random permutation, n=30000 succeeds
495          n = 1100
496          j = np.arange(n + 1).astype(np.float64)
497          x = np.cos(j * np.pi / n)
498

670  class TestCubicSpline:
671      @staticmethod
672      def check_correctness(S, bc_start='not-a-knot', bc_end='not-a-knot',
673                            tol=1e-14):
674          """Check that spline coefficients satisfy the continuity and boundary 
675          conditions."""
676          x = S.x
677          c = S.c
678          dx = np.diff(x)
679          dx = dx.reshape([dx.shape[0]] + [1] * (c.ndim - 2))
680          dxi = dx[:-1]
681  
682          # Check C2 continuity.
683          xp_assert_close(c[3, 1:], c[0, :-1] * dxi**3 + c[1, :-1] * dxi**2 +
684                          c[2, :-1] * dxi + c[3, :-1], rtol=tol, atol=tol)
685          xp_assert_close(c[2, 1:], 3 * c[0, :-1] * dxi**2 +
686                          2 * c[1, :-1] * dxi + c[2, :-1], rtol=tol, atol=tol)
687          xp_assert_close(c[1, 1:], 3 * c[0, :-1] * dxi + c[1, :-1],
688                          rtol=tol, atol=tol)
689  
690          # Check that we found a parabola, the third derivative is 0.
691          if x.size == 3 and bc_start == 'not-a-knot' and bc_end == 'not-a-knot':
692              xp_assert_close(c[0], np.zeros_like(c[0]), rtol=tol, atol=tol)
693              return
694  
695          # Check periodic boundary conditions.
696          if bc_start == 'periodic':
697              xp_assert_close(S(x[0], 0), S(x[-1], 0), rtol=tol, atol=tol)
698              xp_assert_close(S(x[0], 1), S(x[-1], 1), rtol=tol, atol=tol)
699              xp_assert_close(S(x[0], 2), S(x[-1], 2), rtol=tol, atol=tol)
700              return
701  
702          # Check other boundary conditions.
703          if bc_start == 'not-a-knot':
704              if x.size == 2:
705                  slope = (S(x[1]) - S(x[0])) / dx[0]
706                  slope = np.asarray(slope)
707                  xp_assert_close(S(x[0], 1), slope, rtol=tol, atol=tol)
708              else:
709                  xp_assert_close(c[0, 0], c[0, 1], rtol=tol, atol=tol)
710          elif bc_start == 'clamped':
711              xp_assert_close(
712                  S(x[0], 1), np.zeros_like(S(x[0], 1)), rtol=tol, atol=tol)
713          elif bc_start == 'natural':
714              xp_assert_close(
715                  S(x[0], 2), np.zeros_like(S(x[0], 2)), rtol=tol, atol=tol)
716          else:
717              order, value = bc_start
718              xp_assert_close(S(x[0], order), np.asarray(value), rtol=tol, atol=tol)
719  
720          if bc_end == 'not-a-knot':
721              if x.size == 2:
722                  slope = (S(x[1]) - S(x[0])) / dx[0]
723                  slope = np.asarray(slope)
724                  xp_assert_close(S(x[1], 1), slope, rtol=tol, atol=tol)
725              else:
726                  xp_assert_close(c[0, -1], c[0, -2], rtol=tol, atol=tol)
727          elif bc_end == 'clamped':
728              xp_assert_close(S(x[-1], 1), np.zeros_like(S(x[-1], 1)),
729                              rtol=tol, atol=tol)
730          elif bc_end == 'natural':
731              xp_assert_close(S(x[-1], 2), np.zeros_like(S(x[-1], 2)),
732                              rtol=2*tol, atol=2*tol)
733          else:
734              order, value = bc_end
735              xp_assert_close(S(x[-1], order), np.asarray(value), rtol=tol, atol=tol)
