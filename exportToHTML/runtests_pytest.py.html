runtests_pytest.py 
1    """Backwards compatible functions for running tests from SymPy using pytest. 
2     
3    SymPy historically had its own testing framework that aimed to: 
4    - be compatible with pytest; 
5    - operate similarly (or identically) to pytest; 
6    - not require any external dependencies; 
7    - have all the functionality in one file only; 
8    - have no magic, just import the test file and execute the test functions; and 
9    - be portable. 
10    
11   To reduce the maintenance burden of developing an independent testing framework 
12   and to leverage the benefits of existing Python testing infrastructure, SymPy 
13   now uses pytest (and various of its plugins) to run the test suite. 
14    
15   To maintain backwards compatibility with the legacy testing interface of SymPy, 
16   which implemented functions that allowed users to run the tests on their 
17   installed version of SymPy, the functions in this module are implemented to 
18   match the existing API while thinly wrapping pytest. 
19    
20   These two key functions are `test` and `doctest`. 
21    
22   """
23

97   def update_args_with_paths(
98       paths: List[str],
99       keywords: Optional[Tuple[str]],
100      args: List[str],
101  ) -> List[str]:
102      """Appends valid paths and flags to the args `list` passed to `pytest.main`. 
103   
104      The are three different types of "path" that a user may pass to the `paths` 
105      positional arguments, all of which need to be handled slightly differently: 
106   
107      1. Nothing is passed 
108          The paths to the `testpaths` defined in `pytest.ini` need to be appended 
109          to the arguments list. 
110      2. Full, valid paths are passed 
111          These paths need to be validated but can then be directly appended to 
112          the arguments list. 
113      3. Partial paths are passed. 
114          The `testpaths` defined in `pytest.ini` need to be recursed and any 
115          matches be appended to the arguments list. 
116   
117      """
118  
119      def find_paths_matching_partial(partial_paths):
120          partial_path_file_patterns = []
121          for partial_path in partial_paths:
122              if len(partial_path) >= 4:
123                  has_test_prefix = partial_path[:4] == 'test'
124                  has_py_suffix = partial_path[-3:] == '.py'
125              elif len(partial_path) >= 3:
126                  has_test_prefix = False
127                  has_py_suffix = partial_path[-3:] == '.py'
128              else:
129                  has_test_prefix = False
130                  has_py_suffix = False
131              if has_test_prefix and has_py_suffix:
132                  partial_path_file_patterns.append(partial_path)
133              elif has_test_prefix:
134                  partial_path_file_patterns.append(f'{partial_path}*.py')
135              elif has_py_suffix:
136                  partial_path_file_patterns.append(f'test*{partial_path}')
137              else:
138                  partial_path_file_patterns.append(f'test*{partial_path}*.py')
139          matches = []
140          for testpath in valid_testpaths_default:
141              for path, dirs, files in os.walk(testpath, topdown=True):
142                  zipped = zip(partial_paths, partial_path_file_patterns)
143                  for (partial_path, partial_path_file) in zipped:
144                      if fnmatch(path, f'*{partial_path}*'):
145                          matches.append(str(pathlib.Path(path)))
146                          dirs[:] = []
147                      else:
148                          for file in files:
149                              if fnmatch(file, partial_path_file):
150                                  matches.append(str(pathlib.Path(path, file)))
151          return matches
152  
153      def is_tests_file(filepath: str) -> bool:
154          path = pathlib.Path(filepath)
155          if not path.is_file():
156              return False
157          if not path.parts[-1].startswith('test_'):
158              return False
159          if not path.suffix == '.py':
160              return False
161          return True
162  
163      def find_tests_matching_keywords(keywords, filepath):
164          matches = []
165          source = pathlib.Path(filepath).read_text(encoding='utf-8')
166          for line in source.splitlines():
167              if line.lstrip().startswith('def '):
168                  for kw in keywords:
169                      if line.lower().find(kw.lower()) != -1:
170                          test_name = line.split(' ')[1].split('(')[0]
171                          full_test_path = filepath + '::' + test_name
172                          matches.append(full_test_path)
173          return matches
174  
175      valid_testpaths_default = []
176      for testpath in TESTPATHS_DEFAULT:
177          absolute_testpath = pathlib.Path(sympy_dir(), testpath)
178          if absolute_testpath.exists():
179              valid_testpaths_default.append(str(absolute_testpath))
180  
181      candidate_paths = []
182      if paths:
183          full_paths = []
184          partial_paths = []
185          for path in paths:
186              if pathlib.Path(path).exists():
187                  full_paths.append(str(pathlib.Path(sympy_dir(), path)))
188              else:
189                  partial_paths.append(path)
190          matched_paths = find_paths_matching_partial(partial_paths)
191          candidate_paths.extend(full_paths)
192          candidate_paths.extend(matched_paths)
193      else:
194          candidate_paths.extend(valid_testpaths_default)
195  
196      if keywords is not None and keywords != ():
197          matches = []
198          for path in candidate_paths:
199              if is_tests_file(path):
200                  test_matches = find_tests_matching_keywords(keywords, path)
201                  matches.extend(test_matches)
202              else:
203                  for root, dirnames, filenames in os.walk(path):
204                      for filename in filenames:
205                          absolute_filepath = str(pathlib.Path(root, filename))
206                          if is_tests_file(absolute_filepath):
207                              test_matches = find_tests_matching_keywords(
208                                  keywords,
209                                  absolute_filepath,
210                              )
211                              matches.extend(test_matches)
212          args.extend(matches)
213      else:
214          args.extend(candidate_paths)
215  
216      return args

219  def make_absolute_path(partial_path: str) -> str:
220      """Convert a partial path to an absolute path. 
221   
222      A path such a `sympy/core` might be needed. However, absolute paths should 
223      be used in the arguments to pytest in all cases as it avoids errors that 
224      arise from nonexistent paths. 
225   
226      This function assumes that partial_paths will be passed in such that they 
227      begin with the explicit `sympy` directory, i.e. `sympy/...`. 
228   
229      """
230  
231      def is_valid_partial_path(partial_path: str) -> bool:
232          """Assumption that partial paths are defined from the `sympy` root."""
233          return pathlib.Path(partial_path).parts[0] == 'sympy'
234  
235      if not is_valid_partial_path(partial_path):
236          msg = (
237              f'Partial path {dir(partial_path)} is invalid, partial paths are '
238              f'expected to be defined with the `sympy` directory as the root.'
239          )
240          raise ValueError(msg)
241  
242      absolute_path = str(pathlib.Path(sympy_dir(), partial_path))
243      return absolute_path

246  def test(*paths, subprocess=True, rerun=0, **kwargs):
247      """Interface to run tests via pytest compatible with SymPy's test runner. 
248   
249      Explanation 
250      =========== 
251   
252      Note that a `pytest.ExitCode`, which is an `enum`, is returned. This is 
253      different to the legacy SymPy test runner which would return a `bool`. If 
254      all tests successfully pass the `pytest.ExitCode.OK` with value `0` is 
255      returned, whereas the legacy SymPy test runner would return `True`. In any 
256      other scenario, a non-zero `enum` value is returned, whereas the legacy 
257      SymPy test runner would return `False`. Users need to, therefore, be careful 
258      if treating the pytest exit codes as booleans because 
259      `bool(pytest.ExitCode.OK)` evaluates to `False`, the opposite of legacy 
260      behaviour. 
261   
262      Examples 
263      ======== 
264   
265      >>> import sympy  # doctest: +SKIP 
266   
267      Run one file: 
268   
269      >>> sympy.test('sympy/core/tests/test_basic.py')  # doctest: +SKIP 
270      >>> sympy.test('_basic')  # doctest: +SKIP 
271   
272      Run all tests in sympy/functions/ and some particular file: 
273   
274      >>> sympy.test("sympy/core/tests/test_basic.py", 
275      ...            "sympy/functions")  # doctest: +SKIP 
276   
277      Run all tests in sympy/core and sympy/utilities: 
278   
279      >>> sympy.test("/core", "/util")  # doctest: +SKIP 
280   
281      Run specific test from a file: 
282   
283      >>> sympy.test("sympy/core/tests/test_basic.py", 
284      ...            kw="test_equality")  # doctest: +SKIP 
285   
286      Run specific test from any file: 
287   
288      >>> sympy.test(kw="subs")  # doctest: +SKIP 
289   
290      Run the tests using the legacy SymPy runner: 
291   
292      >>> sympy.test(use_sympy_runner=True)  # doctest: +SKIP 
293   
294      Note that this option is slated for deprecation in the near future and is 
295      only currently provided to ensure users have an alternative option while the 
296      pytest-based runner receives real-world testing. 
297   
298      Parameters 
299      ========== 
300      paths : first n positional arguments of strings 
301          Paths, both partial and absolute, describing which subset(s) of the test 
302          suite are to be run. 
303      subprocess : bool, default is True 
304          Legacy option, is currently ignored. 
305      rerun : int, default is 0 
306          Legacy option, is ignored. 
307      use_sympy_runner : bool or None, default is None 
308          Temporary option to invoke the legacy SymPy test runner instead of 
309          `pytest.main`. Will be removed in the near future. 
310      verbose : bool, default is False 
311          Sets the verbosity of the pytest output. Using `True` will add the 
312          `--verbose` option to the pytest call. 
313      tb : str, 'auto', 'long', 'short', 'line', 'native', or 'no' 
314          Sets the traceback print mode of pytest using the `--tb` option. 
315      kw : str 
316          Only run tests which match the given substring expression. An expression 
317          is a Python evaluatable expression where all names are substring-matched 
318          against test names and their parent classes. Example: -k 'test_method or 
319          test_other' matches all test functions and classes whose name contains 
320          'test_method' or 'test_other', while -k 'not test_method' matches those 
321          that don't contain 'test_method' in their names. -k 'not test_method and 
322          not test_other' will eliminate the matches. Additionally keywords are 
323          matched to classes and functions containing extra names in their 
324          'extra_keyword_matches' set, as well as functions which have names 
325          assigned directly to them. The matching is case-insensitive. 
326      pdb : bool, default is False 
327          Start the interactive Python debugger on errors or `KeyboardInterrupt`. 
328      colors : bool, default is True 
329          Color terminal output. 
330      force_colors : bool, default is False 
331          Legacy option, is ignored. 
332      sort : bool, default is True 
333          Run the tests in sorted order. pytest uses a sorted test order by 
334          default. Requires pytest-randomly. 
335      seed : int 
336          Seed to use for random number generation. Requires pytest-randomly. 
337      timeout : int, default is 0 
338          Timeout in seconds before dumping the stacks. 0 means no timeout. 
339          Requires pytest-timeout. 
340      fail_on_timeout : bool, default is False 
341          Legacy option, is currently ignored. 
342      slow : bool, default is False 
343          Run the subset of tests marked as `slow`. 
344      enhance_asserts : bool, default is False 
345          Legacy option, is currently ignored. 
346      split : string in form `<SPLIT>/<GROUPS>` or None, default is None 
347          Used to split the tests up. As an example, if `split='2/3' is used then 
348          only the middle third of tests are run. Requires pytest-split. 
349      time_balance : bool, default is True 
350          Legacy option, is currently ignored. 
351      blacklist : iterable of test paths as strings, default is BLACKLIST_DEFAULT 
352          Blacklisted test paths are ignored using the `--ignore` option. Paths 
353          may be partial or absolute. If partial then they are matched against 
354          all paths in the pytest tests path. 
355      parallel : bool, default is False 
356          Parallelize the test running using pytest-xdist. If `True` then pytest 
357          will automatically detect the number of CPU cores available and use them 
358          all. Requires pytest-xdist. 
359      store_durations : bool, False 
360          Store test durations into the file `.test_durations`. The is used by 
361          `pytest-split` to help determine more even splits when more than one 
362          test group is being used. Requires pytest-split. 
363   
364      """
365      # NOTE: to be removed alongside SymPy test runner
366      if kwargs.get('use_sympy_runner', False):
367          kwargs.pop('parallel', False)
368          kwargs.pop('store_durations', False)
369          kwargs.pop('use_sympy_runner', True)
370          if kwargs.get('slow') is None:
371              kwargs['slow'] = False
372          return test_sympy(*paths, subprocess=True, rerun=0, **kwargs)
373  
374      pytest_plugin_manager = PytestPluginManager()
375      if not pytest_plugin_manager.has_pytest:
376          pytest.main()
377  
378      args = []
379  
380      if kwargs.get('verbose', False):
381          args.append('--verbose')
382  
383      if tb := kwargs.get('tb'):
384          args.extend(['--tb', tb])
385  
386      if kwargs.get('pdb'):
387          args.append('--pdb')
388  
389      if not kwargs.get('colors', True):
390          args.extend(['--color', 'no'])
391  
392      if seed := kwargs.get('seed'):
393          if not pytest_plugin_manager.has_randomly:
394              msg = '`pytest-randomly` plugin required to control random seed.'
395              raise ModuleNotFoundError(msg)
396          args.extend(['--randomly-seed', str(seed)])
397  
398      if kwargs.get('sort', True) and pytest_plugin_manager.has_randomly:
399          args.append('--randomly-dont-reorganize')
400      elif not kwargs.get('sort', True) and not pytest_plugin_manager.has_randomly:
401          msg = '`pytest-randomly` plugin required to randomize test order.'
402          raise ModuleNotFoundError(msg)
403  
404      if timeout := kwargs.get('timeout', None):
405          if not pytest_plugin_manager.has_timeout:
406              msg = '`pytest-timeout` plugin required to apply timeout to tests.'
407              raise ModuleNotFoundError(msg)
408          args.extend(['--timeout', str(int(timeout))])
409  
410      # Skip slow tests by default and always skip tooslow tests
411      if kwargs.get('slow', False):
412          args.extend(['-m', 'slow and not tooslow'])
413      else:
414          args.extend(['-m', 'not slow and not tooslow'])
415  
416      if (split := kwargs.get('split')) is not None:
417          if not pytest_plugin_manager.has_split:
418              msg = '`pytest-split` plugin required to run tests as groups.'
419              raise ModuleNotFoundError(msg)
420          match = split_pattern.match(split)
421          if not match:
422              msg = ('split must be a string of the form a/b where a and b are '
423                     'positive nonzero ints')
424              raise ValueError(msg)
425          group, splits = map(str, match.groups())
426          args.extend(['--group', group, '--splits', splits])
427          if group > splits:
428              msg = (f'cannot have a group number {group} with only {splits} '
429                     'splits')
430              raise ValueError(msg)
431  
432      if blacklist := kwargs.get('blacklist', BLACKLIST_DEFAULT):
433          for path in blacklist:
434              args.extend(['--ignore', make_absolute_path(path)])
435  
436      if kwargs.get('parallel', False):
437          if not pytest_plugin_manager.has_xdist:
438              msg = '`pytest-xdist` plugin required to run tests in parallel.'
439              raise ModuleNotFoundError(msg)
440          args.extend(['-n', 'auto'])
441  
442      if kwargs.get('store_durations', False):
443          if not pytest_plugin_manager.has_split:
444              msg = '`pytest-split` plugin required to store test durations.'
445              raise ModuleNotFoundError(msg)
446          args.append('--store-durations')
447  
448      if (keywords := kwargs.get('kw')) is not None:
449          keywords = tuple(str(kw) for kw in keywords)
450      else:
451          keywords = ()
452  
453      args = update_args_with_paths(paths, keywords, args)
454      exit_code = pytest.main(args)
455      return exit_code
