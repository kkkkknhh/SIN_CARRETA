36   def _get_quantized_conv_bn_example_inputs_kwargs(
37       is_per_channel: bool,
38       has_bias: bool,
39       bias_is_quantized: bool,
40       is_cuda: bool,
41   ) -> dict[str, Any]:
42       """ 
43       Optional example inputs for quantized and folded conv-bn patterns 
44       used in convert, expressed as kwargs. 
45       """
46       kwargs = {}
47       # Per tensor quantization uses literals to represent scale and zero
48       # point, so there is no need to include them here as kwargs
49       if is_per_channel:
50           kwargs["weight_scale"] = torch.tensor([1], dtype=torch.float)
51           kwargs["weight_zero_point"] = torch.tensor([0], dtype=torch.int)
52           if has_bias and bias_is_quantized:

54               kwargs["bias_zero_point"] = torch.tensor([0], dtype=torch.int)
55       if has_bias:
56           kwargs["conv_bias"] = torch.randn(1)
57       if is_cuda:
58           for k, v in kwargs.items():
59               if isinstance(v, torch.Tensor):
60                   kwargs[k] = v.cuda()
61       return kwargs

84   def _get_qat_conv_bn_pattern(conv_fn: Callable) -> Callable:
85       def _qat_conv_bn_pattern(
86           x: torch.Tensor,
87           conv_weight: torch.Tensor,
88           conv_bias: torch.Tensor,
89           bn_weight: torch.Tensor,
90           bn_bias: torch.Tensor,
91           bn_running_mean: torch.Tensor,
92           bn_running_var: torch.Tensor,
93       ) -> torch.Tensor:
94           """ 
95           Approximated method to fuse conv and bn. It requires only one forward pass. 
96           conv_orig = conv / scale_factor where scale_factor = bn.weight / running_std. 
97           This is based on `nniqat.ConvBn2d._forward_approximate`. 
98           """
99           # TODO: allow setting eps
100          bn_eps = 1e-5
101          running_std = torch.sqrt(bn_running_var + bn_eps)
102          scale_factor = bn_weight / running_std
103          weight_shape = [1] * len(conv_weight.shape)
104          weight_in_channel_axis = 1 if _is_conv_transpose_fn(conv_fn) else 0
105          weight_shape[weight_in_channel_axis] = -1
106          bias_shape = [1] * len(conv_weight.shape)
107          bias_shape[1] = -1
108          scaled_weight = conv_weight * scale_factor.reshape(weight_shape)
109          zero_bias = torch.zeros_like(conv_bias, dtype=x.dtype)
110          x = conv_fn(x, scaled_weight, zero_bias)
111          x = x / scale_factor.reshape(bias_shape)
112          x = x + conv_bias.reshape(bias_shape)
113          x = F.batch_norm(
114              x,
115              bn_running_mean,
116              bn_running_var,
117              bn_weight,
118              bn_bias,
119              training=True,
120              eps=bn_eps,
121          )
122          return x
123  
124      return _WrapperModule(_qat_conv_bn_pattern)

167  def _append_qdq(x, is_per_channel, is_bias, kwargs):
168      """ 
169      Helper function to append q-dq ops after `x`, using dummy values for the qparams 
170      and qmin/qmax. We use dummy values here because we match with `ignore_literals=True` 
171      and will manually replace these values after subgraph rewriting. 
172   
173      Return the dq node. 
174      """
175      # Dummy args to be passed into q-dq ops
176      per_channel_axis = 0
177      scale_key = "bias_scale" if is_bias else "weight_scale"
178      zp_key = "bias_zero_point" if is_bias else "weight_zero_point"
179      scale = kwargs[scale_key] if is_per_channel else 1.0
180      zp = kwargs[zp_key] if is_per_channel else 0
181      qmin = -127
182      qmax = 127
183      dtype = torch.int8
184  
185      qd = torch.ops.quantized_decomposed
186      if is_per_channel:
187          x = qd.quantize_per_channel(x, scale, zp, per_channel_axis, qmin, qmax, dtype)
188          x = qd.dequantize_per_channel(x, scale, zp, per_channel_axis, qmin, qmax, dtype)
189      else:
190          x = qd.quantize_per_tensor(x, scale, zp, qmin, qmax, dtype)
191          x = qd.dequantize_per_tensor(x, scale, zp, qmin, qmax, dtype)
192      return x

195  def _get_quantized_qat_conv_bn_pattern(
196      is_per_channel: bool,
197      has_bias: bool,
198      bias_is_quantized: bool,
199      conv_fn: Callable,
200      bn_is_training: bool,
201  ) -> Callable:
202      """ 
203      Return the quantized version of QAT conv + BN pattern. 
204      This is based on `nniqat.ConvBn2d._forward_approximate`, 
205      used in QAT convert. We first match this pattern and replace 
206      it with the normal [conv - bn] pattern, then fold the BN 
207      weights into conv. 
208      """
209      # TODO: allow setting eps
210      bn_eps = 1e-5
211  
212      def _quantized_qat_conv_bn_pattern(
213          x: torch.Tensor,
214          conv_weight: torch.Tensor,
215          bn_weight: torch.Tensor,
216          bn_bias: torch.Tensor,
217          bn_running_mean: torch.Tensor,
218          bn_running_var: torch.Tensor,
219          **kwargs,
220      ) -> torch.Tensor:
221          running_std = torch.sqrt(bn_running_var + bn_eps)
222          scale_factor = bn_weight / running_std
223          weight_shape = [1] * len(conv_weight.shape)
224          weight_shape[0] = -1
225          bias_shape = [1] * len(conv_weight.shape)
226          bias_shape[1] = -1
227          scaled_weight = conv_weight * scale_factor.reshape(weight_shape)
228          scaled_weight = _append_qdq(
229              scaled_weight,
230              is_per_channel,
231              is_bias=False,
232              kwargs=kwargs,
233          )
234          if has_bias:
235              zero_bias = torch.zeros_like(kwargs["conv_bias"], dtype=x.dtype)
236              if bias_is_quantized:
237                  zero_bias = _append_qdq(
238                      zero_bias,
239                      is_per_channel,
240                      is_bias=True,
241                      kwargs=kwargs,
242                  )
243              x = conv_fn(x, scaled_weight, zero_bias)
244          else:
245              x = conv_fn(x, scaled_weight, None)
246          x = x / scale_factor.reshape(bias_shape)
247          if has_bias:
248              x = x + kwargs["conv_bias"].reshape(bias_shape)
249          x = F.batch_norm(
250              x,
251              bn_running_mean,
252              bn_running_var,
253              bn_weight,
254              bn_bias,
255              training=bn_is_training,
256              eps=bn_eps,
257          )
258          return x
259  
260      return _WrapperModule(_quantized_qat_conv_bn_pattern)

263  def _get_folded_quantized_qat_conv_bn_pattern(
264      is_per_channel: bool,
265      has_bias: bool,
266      bias_is_quantized: bool,
267      conv_fn: Callable,
268      bn_is_training: bool,
269  ) -> Callable:
270      """ 
271      Quantized QAT conv - bn pattern with bn weights being folded into conv. 
272      """
273      # TODO: allow setting eps
274      bn_eps = 1e-5
275  
276      def _folded_quantized_qat_conv_bn_pattern(
277          x: torch.Tensor,
278          conv_weight: torch.Tensor,
279          bn_weight: torch.Tensor,
280          bn_bias: torch.Tensor,
281          bn_running_mean: torch.Tensor,
282          bn_running_var: torch.Tensor,
283          **kwargs,
284      ) -> torch.Tensor:
285          conv_weight = _append_qdq(
286              conv_weight,
287              is_per_channel,
288              is_bias=False,
289              kwargs=kwargs,
290          )
291          if has_bias:
292              bias = kwargs["conv_bias"]
293              if bias_is_quantized:
294                  bias = _append_qdq(
295                      bias,
296                      is_per_channel,
297                      is_bias=True,
298                      kwargs=kwargs,
299                  )
300          else:
301              bias = None
302          x = conv_fn(x, conv_weight, bias)
303          x = F.batch_norm(
304              x,
305              bn_running_mean,
306              bn_running_var,
307              bn_weight,
308              bn_bias,
309              training=bn_is_training,
310              eps=bn_eps,
311          )
312          return x
313  
314      return _WrapperModule(_folded_quantized_qat_conv_bn_pattern)

317  def _has_conv_bias_filter(
318      match: "InternalMatch",
319      original_graph: Graph,
320      pattern_graph: Graph,
321  ) -> bool:
322      """ 
323      Match filter for the subgraph rewriter that returns True if the conv node in 
324      the original graph has bias. 
325      """
326      for n in match.nodes_map.values():
327          if _is_conv_or_conv_transpose_node(n):
328              return len(n.args) > 2 and n.args[2] is not None
329      raise ValueError("Could not find conv node in matched conv + bn pattern")

360  def _get_conv_bn_pattern_nodes(r: ReplacedPatterns) -> dict[str, tuple[Node, Node]]:
361      """ 
362      Helper function to extract the nodes in the conv-bn fusion pattern after 
363      subgraph rewriting, in the form of a map: 
364   
365          {name: (original_node, replacement_node)} 
366   
367      The following names must exist in the map: 
368   
369          "conv", "conv_weight", "conv_input", "bn", "getitem" 
370   
371      The following names may exist in the map: 
372   
373          "conv_weight_q", "conv_weight_dq", "conv_bias", 
374          "conv_bias_q", "conv_bias_dq" 
375      """
376  
377      def _get_nodes(nodes: list[Node]) -> tuple[Node, Node, Optional[Node]]:
378          """ 
379          Return a 3-tuple of (conv_node, bn_node, getitem_node). 
380          This asserts that the match contains exactly one of each node. 
381          """
382          conv_node, bn_node, getitem_node = None, None, None
383          for n in nodes:
384              if n.op != "call_function":
385                  continue
386              if _is_conv_or_conv_transpose_node(n):
387                  assert conv_node is None
388                  conv_node = n
389              if _is_bn_node(n):
390                  assert bn_node is None
391                  bn_node = n
392              if n.target == operator.getitem:
393                  assert getitem_node is None
394                  getitem_node = n
395          assert conv_node is not None
396          assert bn_node is not None
397          return (conv_node, bn_node, getitem_node)
398  
399      def _get_q_dq_nodes(n: Node) -> tuple[Node, Node, Node]:
400          """ 
401          Return a 3-tuple of (orig_node, q_node, dq_node). 
402          """
403          assert _is_dequantize(n)
404          q_node = n.args[0]
405          assert isinstance(q_node, Node)
406          assert _is_quantize(q_node)
407          orig_node = q_node.args[0]
408          assert isinstance(orig_node, Node)
409          return (orig_node, q_node, n)
410  
411      original_nodes = list(_filter_nodes_map(r.nodes_map).values())
412      o_conv, o_bn, o_getitem = _get_nodes(original_nodes)
413      r_conv, r_bn, r_getitem = _get_nodes(r.replacements)
414  
415      # Create the mapping from original node to replacement node
416      assert o_getitem is None
417      assert r_getitem is None
418      mapping = {
419          "conv": (o_conv, r_conv),
420          "bn": (o_bn, r_bn),
421      }
422  
423      # Extract conv input and weight
424      # Note: here we extract the original nodes indirectly through the pattern nodes
425      # because the args of the original nodes are no longer available after replacement
426      (p_conv, _, _) = _get_nodes(list(r.nodes_map.keys()))
427      (p_conv_input, p_conv_weight, *_) = p_conv.args
428      (r_conv_input, r_conv_weight, *_) = r_conv.args
429      assert isinstance(p_conv_input, Node)
430      assert isinstance(p_conv_weight, Node)
431      assert isinstance(r_conv_input, Node)
432      assert isinstance(r_conv_weight, Node)
433      o_conv_input = r.nodes_map[p_conv_input]
434      o_conv_weight = r.nodes_map[p_conv_weight]
435  
436      # If conv weight is quantized, extract the q - dq nodes
437      if _is_dequantize(p_conv_weight):
438          p_conv_weight, p_conv_weight_q, p_conv_weight_dq = _get_q_dq_nodes(
439              p_conv_weight
440          )
441          r_conv_weight, r_conv_weight_q, r_conv_weight_dq = _get_q_dq_nodes(
442              r_conv_weight
443          )
444          o_conv_weight = r.nodes_map[p_conv_weight]
445          o_conv_weight_q = r.nodes_map[p_conv_weight_q]
446          o_conv_weight_dq = r.nodes_map[p_conv_weight_dq]
447          mapping["conv_weight_q"] = (o_conv_weight_q, r_conv_weight_q)
448          mapping["conv_weight_dq"] = (o_conv_weight_dq, r_conv_weight_dq)
449      mapping["conv_input"] = (o_conv_input, r_conv_input)
450      mapping["conv_weight"] = (o_conv_weight, r_conv_weight)
451  
452      # Extract conv bias
453      if len(p_conv.args) > 2 and len(r_conv.args) > 2:
454          p_conv_bias = p_conv.args[2]
455          r_conv_bias = r_conv.args[2]
456          assert isinstance(p_conv_bias, Node)
457          assert isinstance(r_conv_bias, Node)
458          o_conv_bias = r.nodes_map[p_conv_bias]
459  
460          # If conv bias is quantized, extract the q - dq nodes
461          if _is_dequantize(p_conv_bias):
462              p_conv_bias, p_conv_bias_q, p_conv_bias_dq = _get_q_dq_nodes(p_conv_bias)
463              r_conv_bias, r_conv_bias_q, r_conv_bias_dq = _get_q_dq_nodes(r_conv_bias)
464              o_conv_bias = r.nodes_map[p_conv_bias]
465              o_conv_bias_q = r.nodes_map[p_conv_bias_q]
466              o_conv_bias_dq = r.nodes_map[p_conv_bias_dq]
467              mapping["conv_bias_q"] = (o_conv_bias_q, r_conv_bias_q)
468              mapping["conv_bias_dq"] = (o_conv_bias_dq, r_conv_bias_dq)
469          mapping["conv_bias"] = (o_conv_bias, r_conv_bias)
470      return mapping

473  def _filter_nodes_map(nodes_map: dict[Node, Node]) -> dict[Node, Node]:
474      """ 
475      Return a filtered `nodes_map` returned from the subgraph rewriter. 
476      The filtered `nodes_map` will contain only nodes that are actually 
477      matched in the pattern, excluding None or placeholder nodes. 
478      """
479      new_nodes_map: dict[Node, Node] = {}
480      for pattern_node, graph_node in nodes_map.items():
481          # bias can be None
482          if graph_node is None:
483              continue
484          # skip pattern placeholder nodes
485          if pattern_node.op == "placeholder":
486              continue
487          new_nodes_map[pattern_node] = graph_node
488      return new_nodes_map

491  # TODO: this is error prone, use the replace_literals_with_placeholders hack instead
492  def _copy_over_literal_conv_args(original_node: Node, new_node: Node):
493      """ 
494      Copy over literal args in conv, such as stride and padding, from the matched node 
495      in the original graph to its replacement in the new graph. 
496   
497      This is needed due to the following limitation in the subgraph rewriter when used 
498      with dynamo export: literal (non-tensor) args are not supported in the match and 
499      replacement patterns. This is because dynamo export automatically inlines these 
500      literal args, making them dead placeholder nodes. In the future, we should check 
501      if dynamo export can optionally disable this inlining, or if subgraph rewriter 
502      can do the copying for us. See https://github.com/pytorch/pytorch/issues/100419. 
503   
504      Note: Unlike other tensor args like conv weights and biases, literal args are 
505      preserved in the original nodes after replacement, so we can access them here. 
506      """
507      assert _is_conv_or_conv_transpose_node(original_node)
508      assert _is_conv_or_conv_transpose_node(new_node)
509      # x, weight, bias, [stride, padding, dilation, transposed, output_padding, groups]
510      new_args = list(new_node.args)
511      if len(new_args) < 3:
512          # bias is optional, when it is not present, it means it is None
513          new_args.append(None)
514      new_node.args = tuple(new_args[:3]) + original_node.args[3:]

517  def _update_conv_input_qspec_map_after_replacement(
518      original_node: Node, replacement_node: Node
519  ):
520      """ 
521      Update the `input_qspec_map` in the annotation after subgraph rewriting. 
522   
523      The original annotation referred to the nodes in the original graph, 
524      so the keys in the `input_qspec_map` will need to be updated to reflect 
525      the corresponding nodes in the replacement graph. 
526      """
527      assert _is_conv_or_conv_transpose_node(original_node)
528      assert _is_conv_or_conv_transpose_node(replacement_node)
529      if "quantization_annotation" not in original_node.meta:
530          return
531      original_input_qspec_map = original_node.meta[
532          "quantization_annotation"
533      ].input_qspec_map
534      input_qspec_map = {}
535      # get the list of configs, it should be ordered as input, weight, bias
536      # note: this is really hacky, we need a better solution, hopefully
537      # in subgraph_rewriter, issue tracking the problem: https://github.com/pytorch/pytorch/issues/101820
538      all_configs = list(original_input_qspec_map.items())
539      # input activation
540      input_qspec_map[replacement_node.args[0]] = all_configs[0][1]
541      # weight
542      input_qspec_map[replacement_node.args[1]] = all_configs[1][1]
543      # bias
544      if len(replacement_node.args) > 2 and len(all_configs) > 2:
545          input_qspec_map[replacement_node.args[2]] = all_configs[2][1]
546      replacement_node.meta["quantization_annotation"].input_qspec_map = input_qspec_map

549  def _update_special_qspecs_after_replacement(
550      node: Node,
551      original_to_replacement_node: dict[Node, Node],
552  ):
553      """ 
554      Update the `SharedQuantizationSpec`s and `DerivedQuantizationSpec`s 
555      used in `node`'s quantization annotation after subgraph rewriting. 
556   
557      The original annotation referred to the nodes in the original graph, 
558      so the nodes used in these special quantization specs will need to 
559      be updated to the corresponding nodes in the replacement graph. 
560      """
561  
562      def _get_new_edge_or_node(edge_or_node: EdgeOrNode):
563          if isinstance(edge_or_node, Node):
564              _node = edge_or_node
565              return original_to_replacement_node.get(_node, _node)
566          elif (
567              isinstance(edge_or_node, tuple)
568              and len(edge_or_node) == 2
569              and all(isinstance(x, Node) for x in edge_or_node)
570          ):
571              src, dest = edge_or_node
572              return (
573                  original_to_replacement_node.get(src, src),
574                  original_to_replacement_node.get(dest, dest),
575              )
576          else:
577              raise ValueError("unexpected type for edge_or_node: ", type(edge_or_node))
578  
579      def _get_new_qspec(qspec: QuantizationSpecBase):
580          if isinstance(qspec, SharedQuantizationSpec):
581              new_edge_or_node = _get_new_edge_or_node(qspec.edge_or_node)
582              return SharedQuantizationSpec(new_edge_or_node)
583          elif isinstance(qspec, DerivedQuantizationSpec):
584              new_derived_from = [_get_new_edge_or_node(x) for x in qspec.derived_from]
585              return dataclasses.replace(qspec, derived_from=new_derived_from)
586          else:
587              return qspec
588  
589      if "quantization_annotation" not in node.meta:
590          return

594      annotation.output_qspec = _get_new_qspec(annotation.output_qspec)

640  def _fuse_conv_bn_qat_helper(
641      m: GraphModule,
642      conv_fn: Callable,
643      example_inputs: tuple[Any, ...],
644      is_cuda: bool,
645  ) -> GraphModule:
646      """ 
647      Given a graph of decomposed aten ops, replace the (conv + bn) pattern with 
648      the fused QAT subgraph equivalent. The input graph should already be annotated. 
649      The annotations in the original nodes will be preserved in the corresponding 
650      nodes in the new subgraph. 
651   
652      Note: This also handles the (conv + bn + relu) pattern. 
653      """
654      m.graph.eliminate_dead_code()
655      m.recompile()
656  
657      conv_bn_pattern = _get_conv_bn_pattern(conv_fn)
658      match_pattern = _get_aten_graph_module_for_pattern(
659          conv_bn_pattern,
660          example_inputs,
661          is_cuda,
662      )
663  
664      # Step (1): Replace patterns with conv bias
665      #
666      # Here we do replacement separately for cases with and without conv bias, since
667      # the replacement patterns for these two cases are substantially different.
668      # TODO: use the public replace_pattern API once it also returns replacement nodes
669  
670      qat_conv_bn_pattern = _get_qat_conv_bn_pattern(conv_fn)
671      replacement_pattern_with_conv_bias = _get_aten_graph_module_for_pattern(
672          qat_conv_bn_pattern,
673          example_inputs,
674          is_cuda,
675      )
676      replacements_with_conv_bias = replace_pattern_with_filters(
677          m,
678          match_pattern,
679          replacement_pattern_with_conv_bias,
680          match_filters=[_has_conv_bias_filter],
681          ignore_literals=True,
682      )
683      m.recompile()
684  
685      # Step (2): Replace patterns without conv bias
686  
687      qat_conv_bn_pattern_no_conv_bias = _get_qat_conv_bn_pattern_no_conv_bias(conv_fn)
688      replacement_pattern_no_conv_bias = _get_aten_graph_module_for_pattern(
689          qat_conv_bn_pattern_no_conv_bias,
690          example_inputs,
691          is_cuda,
692      )
693      replacements_no_conv_bias = replace_pattern_with_filters(
694          m,
695          match_pattern,
696          replacement_pattern_no_conv_bias,
697          match_filters=[_no_conv_bias_filter],
698          ignore_literals=True,
699      )
700      m.recompile()
701  
702      # Step (3): Post processing
703      #
704      # Due to limited functionality in the subgraph rewriter, here we manually
705      # update the replacement graph as follows:
706      #
707      #   (a) Copy over metadata from original subgraph. This ensures the stack traces
708      #       and annotations are preserved in the new subgraph
709      #
710      #   (b) Copy over literal args for conv from the original subgraph
711      #       TODO: do this for literal args for batchnorm as well
712      #
713      #   (c) Update all references of the old nodes in the original subgraph to refer
714      #       to the corresponding nodes in the new subgraph in the annotations
715      #
716      # In the future, we should try to push as much of this functionality into the
717      # subgraph rewriter as possible, so we don't have to manually copy anything over.
718      # For more detail, see https://github.com/pytorch/pytorch/issues/100419.
719  
720      all_original_to_replacement_nodes = {}
721      for r in replacements_with_conv_bias + replacements_no_conv_bias:
722          replacement_dict = _get_conv_bn_pattern_nodes(r)
723          # The original conv node's "nn_module_stack"
724          conv_nn_module = replacement_dict["conv"][0].meta.get("nn_module_stack", None)
725          for k, node_tuple in replacement_dict.items():
726              original_node, replacement_node = node_tuple
727              # Step (3a): Copy over metadata for all nodes in [conv - bn - getitem]
728              replacement_node.meta = original_node.meta
729              # If original_node is a get_attr node, it doesn't have nn_module_stack.
730              # In this case, we copy nn_module_stack from the original conv node.
731              if (
732                  k in ["conv_input", "conv_weight"]
733                  and conv_nn_module
734                  and "nn_module_stack" not in replacement_node.meta
735              ):
736                  replacement_node.meta["nn_module_stack"] = copy.deepcopy(conv_nn_module)
737              if _is_conv_or_conv_transpose_node(original_node):
738                  # Step (3b): Copy over conv literal args
739                  _copy_over_literal_conv_args(original_node, replacement_node)
740                  # Step (3c): Update old references in the conv node's input_qspec_map
741                  _update_conv_input_qspec_map_after_replacement(
742                      original_node, replacement_node
743                  )
744              all_original_to_replacement_nodes[original_node] = replacement_node
745  
746      # Step (3c): Update old references in the special qspecs for all nodes in the graph
747      for n in m.graph.nodes:
748          _update_special_qspecs_after_replacement(n, all_original_to_replacement_nodes)
749  
750      return m

753  def _duplicate_dequantize_node(m: GraphModule):
754      """ 
755      Helper function to duplicate all dequantize nodes in the graph if the 
756      node has more than one user. For example: 
757   
758      Before: 
759        quantize -> dequantize -> a 
760                            \\--> b 
761                            \\--> c 
762   
763      After: 
764        quantize -> dequantize_1 -> a 
765              \\--> dequantize_2 -> b 
766              \\--> dequantize_3 -> c 
767   
768      This is useful for subgraph rewriting. E.g. if we wish to match the 
769      pattern [dequantize - a] above, subgraph matching would fail because 
770      the dequantize node has users outside the matched portion of the graph. 
771      Instead, we match [dequantize_1 - a], which is safe. 
772      """
773      dq_op = torch.ops.quantized_decomposed.dequantize_per_tensor
774      for n in m.graph.nodes:
775          if n.op != "call_function" or n.target != dq_op or len(n.users) == 1:
776              continue
777          for user in list(n.users):
778              with m.graph.inserting_before(n):
779                  new_node = m.graph.create_node("call_function", dq_op, n.args, n.kwargs)
780              user.replace_input_with(n, new_node)
781          m.graph.erase_node(n)
782      m.recompile()
783

785  def _remove_extra_dequantize(m: GraphModule):
786      """ 
787      Removes duplicate dequant nodes in the graph, for an operator that has 
788      multiple dequant nodes as a user, replace them with a single dequant node 
789      that can be shared across all the uses. This should be seen as the "reverse" 
790      of `_duplicate_dequantize_node`. 
791      """
792      dq_op = torch.ops.quantized_decomposed.dequantize_per_tensor
793      for n in m.graph.nodes:
794          dq_users = [
795              user
796              for user in n.users
797              if user.op == "call_function" and user.target == dq_op
798          ]
799          if len(dq_users) > 1:
800              with m.graph.inserting_after(dq_users[0]):
801                  new_node = m.graph.create_node(
802                      "call_function", dq_op, dq_users[0].args, {}
803                  )
804              for dq_user in dq_users:
805                  dq_user.replace_all_uses_with(new_node)
806                  m.graph.erase_node(dq_user)
807      m.recompile()

810  def _copy_over_q_dq_args(original_node: Node, replacement_node: Node):
811      """ 
812      Given a pair of quantize or dequantize nodes, copy over all literal args 
813      from the original node to the replacement node. 
814      """
815      # For quantize_per_tensor, scale and zp are literals and need to be copied
816      # For quantize_per_channel, scale and zp are get_attr nodes and should be skipped
817      assert original_node.target == replacement_node.target
818      if original_node.target in (
819          torch.ops.quantized_decomposed.quantize_per_tensor.default,
820          torch.ops.quantized_decomposed.dequantize_per_tensor.default,
821      ):
822          # Args: input, [scale, zp, qmin, qmax, dtype]
823          start_copy_arg_index = 1
824      elif original_node.target in (
825          torch.ops.quantized_decomposed.quantize_per_channel.default,
826          torch.ops.quantized_decomposed.dequantize_per_channel.default,
827      ):
828          # Args: input, scale, zp, [axis, qmin, qmax, dtype]
829          start_copy_arg_index = 3
830      else:
831          raise ValueError(
832              f"Expected quantize/dequantize nodes, got '{original_node.target}'"
833          )
834      replacement_node.args = (
835          replacement_node.args[:start_copy_arg_index]
836          + original_node.args[start_copy_arg_index:]
837      )

900  def _fold_conv_bn_qat_helper(
901      m: GraphModule,
902      conv_fn: Callable,
903      example_inputs: tuple[Any, ...],
904      is_cuda: bool,
905  ) -> GraphModule:
906      """ 
907      Replace the quantized (conv + bn) pattern with conv with bn weights folded into the weights of conv. 
908      """
909  
910      m.graph.eliminate_dead_code()
911      m.recompile()
912      _duplicate_dequantize_node(m)
913  
914      # Step (1): Replace QAT pattern with simple [conv - bn] pattern
915      replacements = []
916      replacement_options = itertools.product(
917          [True, False],  # is_per_channel
918          [True, False],  # has_bias
919          [True, False],  # bias_is_quantized
920          [True, False],  # bn_is_training
921      )
922      for (
923          is_per_channel,
924          has_bias,
925          bias_is_quantized,
926          bn_is_training,
927      ) in replacement_options:
928          # For the cases without bias, `bias_is_quantized` is irrelevant, so here we arbitrarily
929          # filter out one of the values for this flag to avoid having duplicate patterns
930          if not has_bias and bias_is_quantized:
931              continue
932          kwargs = _get_quantized_conv_bn_example_inputs_kwargs(
933              is_per_channel, has_bias, bias_is_quantized, is_cuda
934          )
935          match_pattern = _get_quantized_qat_conv_bn_pattern(
936              is_per_channel, has_bias, bias_is_quantized, conv_fn, bn_is_training
937          )
938          match_pattern = _get_aten_graph_module_for_pattern(
939              match_pattern,
940              example_inputs,
941              is_cuda,
942              **kwargs,
943          )
944          replacement_pattern = _get_folded_quantized_qat_conv_bn_pattern(
945              is_per_channel, has_bias, bias_is_quantized, conv_fn, bn_is_training
946          )
947          replacement_pattern = _get_aten_graph_module_for_pattern(
948              replacement_pattern,
949              example_inputs,
950              is_cuda,
951              **kwargs,
952          )
953          replacements.extend(
954              replace_pattern_with_filters(
955                  m,
956                  match_pattern,
957                  replacement_pattern,
958                  ignore_literals=True,
959              )
960          )
961      m.recompile()
962      _remove_extra_dequantize(m)
963  
964      for r in replacements:
965          node_map = _get_conv_bn_pattern_nodes(r)
966  
967          # Step (2): Copy over metadata from original subgraph
968          for original_node, replacement_node in node_map.values():
969              replacement_node.meta = original_node.meta
970  
971          # Step (3): Copy over args for weight (and optionally bias) q - dq nodes
972          _copy_over_q_dq_args(*node_map["conv_weight_q"])
973          _copy_over_q_dq_args(*node_map["conv_weight_dq"])
974          if "conv_bias_q" in node_map:
975              assert "conv_bias_dq" in node_map
976              _copy_over_q_dq_args(*node_map["conv_bias_q"])
977              _copy_over_q_dq_args(*node_map["conv_bias_dq"])
978  
979          # Step (4): Fold BN weights into conv
980          conv_bias = None
981          (_, conv_node) = node_map["conv"]
982          (_, bn_node) = node_map["bn"]
983          (_, conv_weight) = node_map["conv_weight"]
984          if "conv_bias" in node_map:
985              (_, conv_bias) = node_map["conv_bias"]
986          fold_bn_weights_into_conv_node(conv_node, conv_weight, conv_bias, bn_node, m)
987  
988          # Copy over literal args for conv
989          for original_node in _filter_nodes_map(r.nodes_map).values():
990              if _is_conv_or_conv_transpose_node(original_node):
991                  _copy_over_literal_conv_args(original_node, conv_node)
992  
993      m.graph.eliminate_dead_code()
994      m.recompile()
995      return m
