1682     def get_view(self, a):
1683         # Cannot view a StringDType as anything else directly, since
1684         # it has references. So, we use a stride trick hack.
1685         from numpy.lib._stride_tricks_impl import DummyArray
1686         interface = dict(a.__array_interface__)
1687         interface['descr'] = self.view_dtype.descr
1688         interface['typestr'] = self.view_dtype.str
1689         return np.asarray(DummyArray(interface, base=a))

1692         return self.get_view(a)['size_and_flags'] & 0xf0

1695         return self.get_flags(a) == self.INITIALIZED | self.OUTSIDE_ARENA

1699                                      | self.OUTSIDE_ARENA

1706         return (self.get_flags(a) & (self.INITIALIZED | self.OUTSIDE_ARENA)
