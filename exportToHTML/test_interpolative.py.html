100      def test_real_id_skel_and_interp_matrices(
101              self, A, L, eps, rank, rand, lin_op, rng):
102          k = rank
103          A_or_L = A if not lin_op else L
104  
105          idx, proj = pymatrixid.interp_decomp(A_or_L, k, rand=rand, rng=rng)
106          P = pymatrixid.reconstruct_interp_matrix(idx, proj)
107          B = pymatrixid.reconstruct_skel_matrix(A, k, idx)
108          assert_allclose(B, A[:, idx[:k]], rtol=eps, atol=1e-08)
109          assert_allclose(B @ P, A, rtol=eps, atol=1e-08)

133          k = rank
134

136          U, S, V = pymatrixid.id_to_svd(A[:, idx[:k]], idx, proj)

146          B = A.copy()

152      def test_rank_estimates_array(self, A, rng):
153          B = np.array([[1, 1, 0], [0, 0, 1], [0, 0, 1]], dtype=A.dtype)
154  
155          for M in [A, B]:
156              rank_tol = 1e-9
157              rank_np = np.linalg.matrix_rank(M, norm(M, 2) * rank_tol)
158              rank_est = pymatrixid.estimate_rank(M, rank_tol, rng=rng)
159              assert_(rank_est >= rank_np)
160              assert_(rank_est <= rank_np + 10)

162      def test_rank_estimates_lin_op(self, A, rng):
163          B = np.array([[1, 1, 0], [0, 0, 1], [0, 0, 1]], dtype=A.dtype)
164  
165          for M in [A, B]:
166              ML = aslinearoperator(M)
167              rank_tol = 1e-9
168              rank_np = np.linalg.matrix_rank(M, norm(M, 2) * rank_tol)
169              rank_est = pymatrixid.estimate_rank(ML, rank_tol, rng=rng)
170              assert_(rank_est >= rank_np - 4)
171              assert_(rank_est <= rank_np + 4)

184      def test_full_rank(self):
185          eps = 1.0e-12
186  
187          # fixed precision
188          A = np.random.rand(16, 8)
189          k, idx, proj = pymatrixid.interp_decomp(A, eps)
190          assert_equal(k, A.shape[1])
191  
192          P = pymatrixid.reconstruct_interp_matrix(idx, proj)
193          B = pymatrixid.reconstruct_skel_matrix(A, k, idx)
194          assert_allclose(A, B @ P)
195  
196          # fixed rank
197          idx, proj = pymatrixid.interp_decomp(A, k)
198  
199          P = pymatrixid.reconstruct_interp_matrix(idx, proj)
200          B = pymatrixid.reconstruct_skel_matrix(A, k, idx)
201          assert_allclose(A, B @ P)
