36   class ProbArg:
37       """Generate a set of probabilities on [0, 1]."""
38   
39       def __init__(self):
40           # Include the endpoints for compatibility with Arg et. al.
41           self.a = 0
42           self.b = 1
43   
44       def values(self, n):
45           """Return an array containing approximately n numbers."""
46           m = max(1, n//3)
47           v1 = np.logspace(-30, np.log10(0.3), m)
48           v2 = np.linspace(0.3, 0.7, m + 1, endpoint=False)[1:]
49           v3 = 1 - np.logspace(np.log10(0.3), -15, m)
50           v = np.r_[v1, v2, v3]
51           return np.unique(v)

54   class EndpointFilter:
55       def __init__(self, a, b, rtol, atol):
56           self.a = a
57           self.b = b
58           self.rtol = rtol
59           self.atol = atol
60   
61       def __call__(self, x):
62           mask1 = np.abs(x - self.a) < self.rtol*np.abs(self.a) + self.atol
63           mask2 = np.abs(x - self.b) < self.rtol*np.abs(self.b) + self.atol
64           return np.where(mask1 | mask2, False, True)

67   class _CDFData:
68       def __init__(self, spfunc, mpfunc, index, argspec, spfunc_first=True,
69                    dps=20, n=5000, rtol=None, atol=None,
70                    endpt_rtol=None, endpt_atol=None):
71           self.spfunc = spfunc
72           self.mpfunc = mpfunc
73           self.index = index
74           self.argspec = argspec
75           self.spfunc_first = spfunc_first
76           self.dps = dps
77           self.n = n
78           self.rtol = rtol
79           self.atol = atol
80   
81           if not isinstance(argspec, list):
82               self.endpt_rtol = None
83               self.endpt_atol = None
84           elif endpt_rtol is not None or endpt_atol is not None:
85               if isinstance(endpt_rtol, list):
86                   self.endpt_rtol = endpt_rtol
87               else:
88                   self.endpt_rtol = [endpt_rtol]*len(self.argspec)
89               if isinstance(endpt_atol, list):
90                   self.endpt_atol = endpt_atol
91               else:
92                   self.endpt_atol = [endpt_atol]*len(self.argspec)
93           else:
94               self.endpt_rtol = None
95               self.endpt_atol = None
96   
97       def idmap(self, *args):
98           if self.spfunc_first:
99               res = self.spfunc(*args)
100              if np.isnan(res):
101                  return np.nan
102              args = list(args)
103              args[self.index] = res
104              with mpmath.workdps(self.dps):
105                  res = self.mpfunc(*tuple(args))
106                  # Imaginary parts are spurious
107                  res = mpf2float(res.real)
108          else:
109              with mpmath.workdps(self.dps):
110                  res = self.mpfunc(*args)
111                  res = mpf2float(res.real)
112              args = list(args)
113              args[self.index] = res
114              res = self.spfunc(*tuple(args))
115          return res
116  
117      def get_param_filter(self):
118          if self.endpt_rtol is None and self.endpt_atol is None:
119              return None
120  
121          filters = []
122          for rtol, atol, spec in zip(self.endpt_rtol, self.endpt_atol, self.argspec):
123              if rtol is None and atol is None:
124                  filters.append(None)
125                  continue
126              elif rtol is None:
127                  rtol = 0.0
128              elif atol is None:
129                  atol = 0.0
130  
131              filters.append(EndpointFilter(spec.a, spec.b, rtol, atol))
132          return filters
133  
134      def check(self):
135          # Generate values for the arguments
136          args = get_args(self.argspec, self.n)
137          param_filter = self.get_param_filter()
138          param_columns = tuple(range(args.shape[1]))
139          result_columns = args.shape[1]
140          args = np.hstack((args, args[:, self.index].reshape(args.shape[0], 1)))
141          FuncData(self.idmap, args,
142                   param_columns=param_columns, result_columns=result_columns,
143                   rtol=self.rtol, atol=self.atol, vectorized=False,
144                   param_filter=param_filter).check()

152  def _binomial_cdf(k, n, p):
153      k, n, p = mpmath.mpf(k), mpmath.mpf(n), mpmath.mpf(p)
154      if k <= 0:
155          return mpmath.mpf(0)
156      elif k >= n:
157          return mpmath.mpf(1)
158  
159      onemp = mpmath.fsub(1, p, exact=True)
160      return mpmath.betainc(n - k, k + 1, x2=onemp, regularized=True)

163  def _f_cdf(dfn, dfd, x):
164      if x < 0:
165          return mpmath.mpf(0)
166      dfn, dfd, x = mpmath.mpf(dfn), mpmath.mpf(dfd), mpmath.mpf(x)
167      ub = dfn*x/(dfn*x + dfd)
168      res = mpmath.betainc(dfn/2, dfd/2, x2=ub, regularized=True)
169      return res

172  def _student_t_cdf(df, t, dps=None):
173      if dps is None:
174          dps = mpmath.mp.dps
175      with mpmath.workdps(dps):
176          df, t = mpmath.mpf(df), mpmath.mpf(t)
177          fac = mpmath.hyp2f1(0.5, 0.5*(df + 1), 1.5, -t**2/df)
178          fac *= t*mpmath.gamma(0.5*(df + 1))
179          fac /= mpmath.sqrt(mpmath.pi*df)*mpmath.gamma(0.5*df)
180          return 0.5 + fac

183  def _noncentral_chi_pdf(t, df, nc):
184      res = mpmath.besseli(df/2 - 1, mpmath.sqrt(nc*t))
185      res *= mpmath.exp(-(t + nc)/2)*(t/nc)**(df/4 - 1/2)/2
186      return res
187  
188  
189  def _noncentral_chi_cdf(x, df, nc, dps=None):
190      if dps is None:
191          dps = mpmath.mp.dps
192      x, df, nc = mpmath.mpf(x), mpmath.mpf(df), mpmath.mpf(nc)
193      with mpmath.workdps(dps):
194          res = mpmath.quad(lambda t: _noncentral_chi_pdf(t, df, nc), [0, x])
195          return res

198  def _tukey_lmbda_quantile(p, lmbda):
199      # For lmbda != 0
200      return (p**lmbda - (1 - p)**lmbda)/lmbda

442              # raise exceptions or cause hangs)
