53   def minimize(fun, x0, args=(), method=None, jac=None, hess=None,
54                hessp=None, bounds=None, constraints=(), tol=None,
55                callback=None, options=None):
56       """Minimization of scalar function of one or more variables. 
57    
58       Parameters 
59       ---------- 
60       fun : callable 
61           The objective function to be minimized:: 
62    
63               fun(x, *args) -> float 
64    
65           where ``x`` is a 1-D array with shape (n,) and ``args`` 
66           is a tuple of the fixed parameters needed to completely 
67           specify the function. 
68    
69           Suppose the callable has signature ``f0(x, *my_args, **my_kwargs)``, where 
70           ``my_args`` and ``my_kwargs`` are required positional and keyword arguments. 
71           Rather than passing ``f0`` as the callable, wrap it to accept 
72           only ``x``; e.g., pass ``fun=lambda x: f0(x, *my_args, **my_kwargs)`` as the 
73           callable, where ``my_args`` (tuple) and ``my_kwargs`` (dict) have been 
74           gathered before invoking this function. 
75       x0 : ndarray, shape (n,) 
76           Initial guess. Array of real elements of size (n,), 
77           where ``n`` is the number of independent variables. 
78       args : tuple, optional 
79           Extra arguments passed to the objective function and its 
80           derivatives (`fun`, `jac` and `hess` functions). 
81       method : str or callable, optional 
82           Type of solver.  Should be one of 
83    
84           - 'Nelder-Mead' :ref:`(see here) <optimize.minimize-neldermead>` 
85           - 'Powell'      :ref:`(see here) <optimize.minimize-powell>` 
86           - 'CG'          :ref:`(see here) <optimize.minimize-cg>` 
87           - 'BFGS'        :ref:`(see here) <optimize.minimize-bfgs>` 
88           - 'Newton-CG'   :ref:`(see here) <optimize.minimize-newtoncg>` 
89           - 'L-BFGS-B'    :ref:`(see here) <optimize.minimize-lbfgsb>` 
90           - 'TNC'         :ref:`(see here) <optimize.minimize-tnc>` 
91           - 'COBYLA'      :ref:`(see here) <optimize.minimize-cobyla>` 
92           - 'COBYQA'      :ref:`(see here) <optimize.minimize-cobyqa>` 
93           - 'SLSQP'       :ref:`(see here) <optimize.minimize-slsqp>` 
94           - 'trust-constr':ref:`(see here) <optimize.minimize-trustconstr>` 
95           - 'dogleg'      :ref:`(see here) <optimize.minimize-dogleg>` 
96           - 'trust-ncg'   :ref:`(see here) <optimize.minimize-trustncg>` 
97           - 'trust-exact' :ref:`(see here) <optimize.minimize-trustexact>` 
98           - 'trust-krylov' :ref:`(see here) <optimize.minimize-trustkrylov>` 
99           - custom - a callable object, see below for description. 
100   
101          If not given, chosen to be one of ``BFGS``, ``L-BFGS-B``, ``SLSQP``, 
102          depending on whether or not the problem has constraints or bounds. 
103      jac : {callable,  '2-point', '3-point', 'cs', bool}, optional 
104          Method for computing the gradient vector. Only for CG, BFGS, 
105          Newton-CG, L-BFGS-B, TNC, SLSQP, dogleg, trust-ncg, trust-krylov, 
106          trust-exact and trust-constr. 
107          If it is a callable, it should be a function that returns the gradient 
108          vector:: 
109   
110              jac(x, *args) -> array_like, shape (n,) 
111   
112          where ``x`` is an array with shape (n,) and ``args`` is a tuple with 
113          the fixed parameters. If `jac` is a Boolean and is True, `fun` is 
114          assumed to return a tuple ``(f, g)`` containing the objective 
115          function and the gradient. 
116          Methods 'Newton-CG', 'trust-ncg', 'dogleg', 'trust-exact', and 
117          'trust-krylov' require that either a callable be supplied, or that 
118          `fun` return the objective and gradient. 
119          If None or False, the gradient will be estimated using 2-point finite 
120          difference estimation with an absolute step size. 
121          Alternatively, the keywords  {'2-point', '3-point', 'cs'} can be used 
122          to select a finite difference scheme for numerical estimation of the 
123          gradient with a relative step size. These finite difference schemes 
124          obey any specified `bounds`. 
125      hess : {callable, '2-point', '3-point', 'cs', HessianUpdateStrategy}, optional 
126          Method for computing the Hessian matrix. Only for Newton-CG, dogleg, 
127          trust-ncg, trust-krylov, trust-exact and trust-constr. 
128          If it is callable, it should return the Hessian matrix:: 
129   
130              hess(x, *args) -> {LinearOperator, spmatrix, array}, (n, n) 
131   
132          where ``x`` is a (n,) ndarray and ``args`` is a tuple with the fixed 
133          parameters. 
134          The keywords {'2-point', '3-point', 'cs'} can also be used to select 
135          a finite difference scheme for numerical estimation of the hessian. 
136          Alternatively, objects implementing the `HessianUpdateStrategy` 
137          interface can be used to approximate the Hessian. Available 
138          quasi-Newton methods implementing this interface are: 
139   
140          - `BFGS` 
141          - `SR1` 
142   
143          Not all of the options are available for each of the methods; for 
144          availability refer to the notes. 
145      hessp : callable, optional 
146          Hessian of objective function times an arbitrary vector p. Only for 
147          Newton-CG, trust-ncg, trust-krylov, trust-constr. 
148          Only one of `hessp` or `hess` needs to be given. If `hess` is 
149          provided, then `hessp` will be ignored. `hessp` must compute the 
150          Hessian times an arbitrary vector:: 
151   
152              hessp(x, p, *args) ->  ndarray shape (n,) 
153   
154          where ``x`` is a (n,) ndarray, ``p`` is an arbitrary vector with 
155          dimension (n,) and ``args`` is a tuple with the fixed 
156          parameters. 
157      bounds : sequence or `Bounds`, optional 
158          Bounds on variables for Nelder-Mead, L-BFGS-B, TNC, SLSQP, Powell, 
159          trust-constr, COBYLA, and COBYQA methods. There are two ways to specify 
160          the bounds: 
161   
162          1. Instance of `Bounds` class. 
163          2. Sequence of ``(min, max)`` pairs for each element in `x`. None 
164             is used to specify no bound. 
165   
166      constraints : {Constraint, dict} or List of {Constraint, dict}, optional 
167          Constraints definition. Only for COBYLA, COBYQA, SLSQP and trust-constr. 
168   
169          Constraints for 'trust-constr' and 'cobyqa' are defined as a single object 
170          or a list of objects specifying constraints to the optimization problem. 
171          Available constraints are: 
172   
173          - `LinearConstraint` 
174          - `NonlinearConstraint` 
175   
176          Constraints for COBYLA, SLSQP are defined as a list of dictionaries. 
177          Each dictionary with fields: 
178   
179          type : str 
180              Constraint type: 'eq' for equality, 'ineq' for inequality. 
181          fun : callable 
182              The function defining the constraint. 
183          jac : callable, optional 
184              The Jacobian of `fun` (only for SLSQP). 
185          args : sequence, optional 
186              Extra arguments to be passed to the function and Jacobian. 
187   
188          Equality constraint means that the constraint function result is to 
189          be zero whereas inequality means that it is to be non-negative. 
190          Note that COBYLA only supports inequality constraints. 
191   
192      tol : float, optional 
193          Tolerance for termination. When `tol` is specified, the selected 
194          minimization algorithm sets some relevant solver-specific tolerance(s) 
195          equal to `tol`. For detailed control, use solver-specific 
196          options. 
197      options : dict, optional 
198          A dictionary of solver options. All methods except `TNC` accept the 
199          following generic options: 
200   
201          maxiter : int 
202              Maximum number of iterations to perform. Depending on the 
203              method each iteration may use several function evaluations. 
204   
205              For `TNC` use `maxfun` instead of `maxiter`. 
206          disp : bool 
207              Set to True to print convergence messages. 
208   
209          For method-specific options, see :func:`show_options()`. 
210      callback : callable, optional 
211          A callable called after each iteration. 
212   
213          All methods except TNC, SLSQP, and COBYLA support a callable with 
214          the signature:: 
215   
216              callback(intermediate_result: OptimizeResult) 
217   
218          where ``intermediate_result`` is a keyword parameter containing an 
219          `OptimizeResult` with attributes ``x`` and ``fun``, the present values 
220          of the parameter vector and objective function. Note that the name 
221          of the parameter must be ``intermediate_result`` for the callback 
222          to be passed an `OptimizeResult`. These methods will also terminate if 
223          the callback raises ``StopIteration``. 
224   
225          All methods except trust-constr (also) support a signature like:: 
226   
227              callback(xk) 
228   
229          where ``xk`` is the current parameter vector. 
230   
231          Introspection is used to determine which of the signatures above to 
232          invoke. 
233   
234      Returns 
235      ------- 
236      res : OptimizeResult 
237          The optimization result represented as a ``OptimizeResult`` object. 
238          Important attributes are: ``x`` the solution array, ``success`` a 
239          Boolean flag indicating if the optimizer exited successfully and 
240          ``message`` which describes the cause of the termination. See 
241          `OptimizeResult` for a description of other attributes.

249      Notes 
250      ----- 
251      This section describes the available solvers that can be selected by the 
252      'method' parameter. The default method is *BFGS*. 
253   
254      **Unconstrained minimization** 
255   
256      Method :ref:`CG <optimize.minimize-cg>` uses a nonlinear conjugate 
257      gradient algorithm by Polak and Ribiere, a variant of the 
258      Fletcher-Reeves method described in [5]_ pp.120-122. Only the 
259      first derivatives are used. 
260   
261      Method :ref:`BFGS <optimize.minimize-bfgs>` uses the quasi-Newton 
262      method of Broyden, Fletcher, Goldfarb, and Shanno (BFGS) [5]_ 
263      pp. 136. It uses the first derivatives only. BFGS has proven good 
264      performance even for non-smooth optimizations. This method also 
265      returns an approximation of the Hessian inverse, stored as 
266      `hess_inv` in the OptimizeResult object. 
267   
268      Method :ref:`Newton-CG <optimize.minimize-newtoncg>` uses a 
269      Newton-CG algorithm [5]_ pp. 168 (also known as the truncated 
270      Newton method). It uses a CG method to the compute the search 
271      direction. See also *TNC* method for a box-constrained 
272      minimization with a similar algorithm. Suitable for large-scale 
273      problems. 
274   
275      Method :ref:`dogleg <optimize.minimize-dogleg>` uses the dog-leg 
276      trust-region algorithm [5]_ for unconstrained minimization. This 
277      algorithm requires the gradient and Hessian; furthermore the 
278      Hessian is required to be positive definite. 
279   
280      Method :ref:`trust-ncg <optimize.minimize-trustncg>` uses the 
281      Newton conjugate gradient trust-region algorithm [5]_ for 
282      unconstrained minimization. This algorithm requires the gradient 
283      and either the Hessian or a function that computes the product of 
284      the Hessian with a given vector. Suitable for large-scale problems. 
285   
286      Method :ref:`trust-krylov <optimize.minimize-trustkrylov>` uses 
287      the Newton GLTR trust-region algorithm [14]_, [15]_ for unconstrained 
288      minimization. This algorithm requires the gradient 
289      and either the Hessian or a function that computes the product of 
290      the Hessian with a given vector. Suitable for large-scale problems. 
291      On indefinite problems it requires usually less iterations than the 
292      `trust-ncg` method and is recommended for medium and large-scale problems. 
293   
294      Method :ref:`trust-exact <optimize.minimize-trustexact>` 
295      is a trust-region method for unconstrained minimization in which 
296      quadratic subproblems are solved almost exactly [13]_. This 
297      algorithm requires the gradient and the Hessian (which is 
298      *not* required to be positive definite). It is, in many 
299      situations, the Newton method to converge in fewer iterations 
300      and the most recommended for small and medium-size problems. 
301   
302      **Bound-Constrained minimization** 
303   
304      Method :ref:`Nelder-Mead <optimize.minimize-neldermead>` uses the 
305      Simplex algorithm [1]_, [2]_. This algorithm is robust in many 
306      applications. However, if numerical computation of derivative can be 
307      trusted, other algorithms using the first and/or second derivatives 
308      information might be preferred for their better performance in 
309      general. 
310   
311      Method :ref:`L-BFGS-B <optimize.minimize-lbfgsb>` uses the L-BFGS-B 
312      algorithm [6]_, [7]_ for bound constrained minimization. 
313   
314      Method :ref:`Powell <optimize.minimize-powell>` is a modification 
315      of Powell's method [3]_, [4]_ which is a conjugate direction 
316      method. It performs sequential one-dimensional minimizations along 
317      each vector of the directions set (`direc` field in `options` and 
318      `info`), which is updated at each iteration of the main 
319      minimization loop. The function need not be differentiable, and no 
320      derivatives are taken. If bounds are not provided, then an 
321      unbounded line search will be used. If bounds are provided and 
322      the initial guess is within the bounds, then every function 
323      evaluation throughout the minimization procedure will be within 
324      the bounds. If bounds are provided, the initial guess is outside 
325      the bounds, and `direc` is full rank (default has full rank), then 
326      some function evaluations during the first iteration may be 
327      outside the bounds, but every function evaluation after the first 
328      iteration will be within the bounds. If `direc` is not full rank, 
329      then some parameters may not be optimized and the solution is not 
330      guaranteed to be within the bounds. 
331   
332      Method :ref:`TNC <optimize.minimize-tnc>` uses a truncated Newton 
333      algorithm [5]_, [8]_ to minimize a function with variables subject 
334      to bounds. This algorithm uses gradient information; it is also 
335      called Newton Conjugate-Gradient. It differs from the *Newton-CG* 
336      method described above as it wraps a C implementation and allows 
337      each variable to be given upper and lower bounds. 
338   
339      **Constrained Minimization** 
340   
341      Method :ref:`COBYLA <optimize.minimize-cobyla>` uses the 
342      Constrained Optimization BY Linear Approximation (COBYLA) method 
343      [9]_, [10]_, [11]_. The algorithm is based on linear 
344      approximations to the objective function and each constraint. The 
345      method wraps a FORTRAN implementation of the algorithm. The 
346      constraints functions 'fun' may return either a single number 
347      or an array or list of numbers. 
348   
349      Method :ref:`COBYQA <optimize.minimize-cobyqa>` uses the Constrained 
350      Optimization BY Quadratic Approximations (COBYQA) method [18]_. The 
351      algorithm is a derivative-free trust-region SQP method based on quadratic 
352      approximations to the objective function and each nonlinear constraint. The 
353      bounds are treated as unrelaxable constraints, in the sense that the 
354      algorithm always respects them throughout the optimization process. 
355   
356      Method :ref:`SLSQP <optimize.minimize-slsqp>` uses Sequential 
357      Least SQuares Programming to minimize a function of several 
358      variables with any combination of bounds, equality and inequality 
359      constraints. The method wraps the SLSQP Optimization subroutine 
360      originally implemented by Dieter Kraft [12]_. Note that the 
361      wrapper handles infinite values in bounds by converting them into 
362      large floating values. 
363   
364      Method :ref:`trust-constr <optimize.minimize-trustconstr>` is a 
365      trust-region algorithm for constrained optimization. It switches 
366      between two implementations depending on the problem definition. 
367      It is the most versatile constrained minimization algorithm 
368      implemented in SciPy and the most appropriate for large-scale problems. 
369      For equality constrained problems it is an implementation of Byrd-Omojokun 
370      Trust-Region SQP method described in [17]_ and in [5]_, p. 549. When 
371      inequality constraints are imposed as well, it switches to the trust-region 
372      interior point method described in [16]_. This interior point algorithm, 
373      in turn, solves inequality constraints by introducing slack variables 
374      and solving a sequence of equality-constrained barrier problems 
375      for progressively smaller values of the barrier parameter. 
376      The previously described equality constrained SQP method is 
377      used to solve the subproblems with increasing levels of accuracy 
378      as the iterate gets closer to a solution. 
379   
380      **Finite-Difference Options** 
381   
382      For Method :ref:`trust-constr <optimize.minimize-trustconstr>` 
383      the gradient and the Hessian may be approximated using 
384      three finite-difference schemes: {'2-point', '3-point', 'cs'}. 
385      The scheme 'cs' is, potentially, the most accurate but it 
386      requires the function to correctly handle complex inputs and to 
387      be differentiable in the complex plane. The scheme '3-point' is more 
388      accurate than '2-point' but requires twice as many operations. If the 
389      gradient is estimated via finite-differences the Hessian must be 
390      estimated using one of the quasi-Newton strategies. 
391   
392      **Method specific options for the** `hess` **keyword** 
393   
394      +--------------+------+----------+-------------------------+-----+ 
395      | method/Hess  | None | callable | '2-point/'3-point'/'cs' | HUS | 
396      +==============+======+==========+=========================+=====+ 
397      | Newton-CG    | x    | (n, n)   | x                       | x   | 
398      |              |      | LO       |                         |     | 
399      +--------------+------+----------+-------------------------+-----+ 
400      | dogleg       |      | (n, n)   |                         |     | 
401      +--------------+------+----------+-------------------------+-----+ 
402      | trust-ncg    |      | (n, n)   | x                       | x   | 
403      +--------------+------+----------+-------------------------+-----+ 
404      | trust-krylov |      | (n, n)   | x                       | x   | 
405      +--------------+------+----------+-------------------------+-----+ 
406      | trust-exact  |      | (n, n)   |                         |     | 
407      +--------------+------+----------+-------------------------+-----+ 
408      | trust-constr | x    | (n, n)   |  x                      | x   | 
409      |              |      | LO       |                         |     | 
410      |              |      | sp       |                         |     | 
411      +--------------+------+----------+-------------------------+-----+ 
412   
413      where LO=LinearOperator, sp=Sparse matrix, HUS=HessianUpdateStrategy

415      **Custom minimizers** 
416   
417      It may be useful to pass a custom minimization method, for example 
418      when using a frontend to this method such as `scipy.optimize.basinhopping` 
419      or a different library.  You can simply pass a callable as the ``method`` 
420      parameter. 
421   
422      The callable is called as ``method(fun, x0, args, **kwargs, **options)`` 
423      where ``kwargs`` corresponds to any other parameters passed to `minimize` 
424      (such as `callback`, `hess`, etc.), except the `options` dict, which has 
425      its contents also passed as `method` parameters pair by pair.  Also, if 
426      `jac` has been passed as a bool type, `jac` and `fun` are mangled so that 
427      `fun` returns just the function values and `jac` is converted to a function 
428      returning the Jacobian.  The method shall return an `OptimizeResult` 
429      object. 
430   
431      The provided `method` callable must be able to accept (and possibly ignore) 
432      arbitrary parameters; the set of parameters accepted by `minimize` may 
433      expand in future versions and then these parameters will be passed to 
434      the method.  You can find an example in the scipy.optimize tutorial.

492      Examples 
493      -------- 
494      Let us consider the problem of minimizing the Rosenbrock function. This 
495      function (and its respective derivatives) is implemented in `rosen` 
496      (resp. `rosen_der`, `rosen_hess`) in the `scipy.optimize`. 
497   
498      >>> from scipy.optimize import minimize, rosen, rosen_der 
499   
500      A simple application of the *Nelder-Mead* method is: 
501   
502      >>> x0 = [1.3, 0.7, 0.8, 1.9, 1.2] 
503      >>> res = minimize(rosen, x0, method='Nelder-Mead', tol=1e-6) 
504      >>> res.x 
505      array([ 1.,  1.,  1.,  1.,  1.]) 
506   
507      Now using the *BFGS* algorithm, using the first derivative and a few 
508      options: 
509   
510      >>> res = minimize(rosen, x0, method='BFGS', jac=rosen_der, 
511      ...                options={'gtol': 1e-6, 'disp': True}) 
512      Optimization terminated successfully. 
513               Current function value: 0.000000 
514               Iterations: 26 
515               Function evaluations: 31 
516               Gradient evaluations: 31 
517      >>> res.x 
518      array([ 1.,  1.,  1.,  1.,  1.]) 
519      >>> print(res.message) 
520      Optimization terminated successfully. 
521      >>> res.hess_inv 
522      array([ 
523          [ 0.00749589,  0.01255155,  0.02396251,  0.04750988,  0.09495377],  # may vary 
524          [ 0.01255155,  0.02510441,  0.04794055,  0.09502834,  0.18996269], 
525          [ 0.02396251,  0.04794055,  0.09631614,  0.19092151,  0.38165151], 
526          [ 0.04750988,  0.09502834,  0.19092151,  0.38341252,  0.7664427 ], 
527          [ 0.09495377,  0.18996269,  0.38165151,  0.7664427,   1.53713523] 
528      ]) 
529   
530   
531      Next, consider a minimization problem with several constraints (namely 
532      Example 16.4 from [5]_). The objective function is: 
533   
534      >>> fun = lambda x: (x[0] - 1)**2 + (x[1] - 2.5)**2 
535   
536      There are three constraints defined as: 
537   
538      >>> cons = ({'type': 'ineq', 'fun': lambda x:  x[0] - 2 * x[1] + 2}, 
539      ...         {'type': 'ineq', 'fun': lambda x: -x[0] - 2 * x[1] + 6}, 
540      ...         {'type': 'ineq', 'fun': lambda x: -x[0] + 2 * x[1] + 2}) 
541   
542      And variables must be positive, hence the following bounds: 
543   
544      >>> bnds = ((0, None), (0, None)) 
545   
546      The optimization problem is solved using the SLSQP method as: 
547   
548      >>> res = minimize(fun, (2, 0), method='SLSQP', bounds=bnds, 
549      ...                constraints=cons) 
550   
551      It should converge to the theoretical solution (1.4 ,1.7). 
552   
553      """
554      x0 = np.atleast_1d(np.asarray(x0))
555  
556      if x0.ndim != 1:
557          raise ValueError("'x0' must only have one dimension.")
558  
559      if x0.dtype.kind in np.typecodes["AllInteger"]:
560          x0 = np.asarray(x0, dtype=float)
561  
562      if not isinstance(args, tuple):
563          args = (args,)
564  
565      if method is None:
566          # Select automatically
567          if constraints:
568              method = 'SLSQP'
569          elif bounds is not None:
570              method = 'L-BFGS-B'
571          else:
572              method = 'BFGS'
573  
574      if callable(method):
575          meth = "_custom"
576      else:
577          meth = method.lower()
578  
579      if options is None:
580          options = {}
581      # check if optional parameters are supported by the selected method
582      # - jac
583      if meth in ('nelder-mead', 'powell', 'cobyla', 'cobyqa') and bool(jac):
584          warn(f'Method {method} does not use gradient information (jac).',
585               RuntimeWarning, stacklevel=2)
586      # - hess
587      if meth not in ('newton-cg', 'dogleg', 'trust-ncg', 'trust-constr',
588                      'trust-krylov', 'trust-exact', '_custom') and hess is not None:
589          warn(f'Method {method} does not use Hessian information (hess).',
590               RuntimeWarning, stacklevel=2)
591      # - hessp
592      if meth not in ('newton-cg', 'trust-ncg', 'trust-constr',
593                      'trust-krylov', '_custom') \
594         and hessp is not None:
595          warn(f'Method {method} does not use Hessian-vector product'
596               ' information (hessp).',
597               RuntimeWarning, stacklevel=2)
598      # - constraints or bounds
599      if (meth not in ('cobyla', 'cobyqa', 'slsqp', 'trust-constr', '_custom') and
600              np.any(constraints)):
601          warn(f'Method {method} cannot handle constraints.',
602               RuntimeWarning, stacklevel=2)
603      if meth not in (
604              'nelder-mead', 'powell', 'l-bfgs-b', 'cobyla', 'cobyqa', 'slsqp',
605              'tnc', 'trust-constr', '_custom') and bounds is not None:
606          warn(f'Method {method} cannot handle bounds.',
607               RuntimeWarning, stacklevel=2)
608      # - return_all
609      if (meth in ('l-bfgs-b', 'tnc', 'cobyla', 'cobyqa', 'slsqp') and
610              options.get('return_all', False)):
611          warn(f'Method {method} does not support the return_all option.',
612               RuntimeWarning, stacklevel=2)
613  
614      # check gradient vector
615      if callable(jac):
616          pass
617      elif jac is True:
618          # fun returns func and grad
619          fun = MemoizeJac(fun)
620          jac = fun.derivative
621      elif (jac in FD_METHODS and
622            meth in ['trust-constr', 'bfgs', 'cg', 'l-bfgs-b', 'tnc', 'slsqp']):
623          # finite differences with relative step
624          pass
625      elif meth in ['trust-constr']:
626          # default jac calculation for this method
627          jac = '2-point'
628      elif jac is None or bool(jac) is False:
629          # this will cause e.g. LBFGS to use forward difference, absolute step
630          jac = None
631      else:
632          # default if jac option is not understood
633          jac = None
634  
635      # set default tolerances
636      if tol is not None:
637          options = dict(options)
638          if meth == 'nelder-mead':
639              options.setdefault('xatol', tol)
640              options.setdefault('fatol', tol)
641          if meth in ('newton-cg', 'powell', 'tnc'):
642              options.setdefault('xtol', tol)
643          if meth in ('powell', 'l-bfgs-b', 'tnc', 'slsqp'):
644              options.setdefault('ftol', tol)
645          if meth in ('bfgs', 'cg', 'l-bfgs-b', 'tnc', 'dogleg',
646                      'trust-ncg', 'trust-exact', 'trust-krylov'):
647              options.setdefault('gtol', tol)
648          if meth in ('cobyla', '_custom'):
649              options.setdefault('tol', tol)
650          if meth == 'cobyqa':
651              options.setdefault('final_tr_radius', tol)
652          if meth == 'trust-constr':
653              options.setdefault('xtol', tol)
654              options.setdefault('gtol', tol)
655              options.setdefault('barrier_tol', tol)
656  
657      if meth == '_custom':
658          # custom method called before bounds and constraints are 'standardised'
659          # custom method should be able to accept whatever bounds/constraints
660          # are provided to it.
661          return method(fun, x0, args=args, jac=jac, hess=hess, hessp=hessp,
662                        bounds=bounds, constraints=constraints,
663                        callback=callback, **options)
664  
665      constraints = standardize_constraints(constraints, x0, meth)
666  
667      remove_vars = False
668      if bounds is not None:
669          # convert to new-style bounds so we only have to consider one case
670          bounds = standardize_bounds(bounds, x0, 'new')
671          bounds = _validate_bounds(bounds, x0, meth)
672  
673          if meth in {"tnc", "slsqp", "l-bfgs-b"}:
674              # These methods can't take the finite-difference derivatives they
675              # need when a variable is fixed by the bounds. To avoid this issue,
676              # remove fixed variables from the problem.
677              # NOTE: if this list is expanded, then be sure to update the
678              # accompanying tests and test_optimize.eb_data. Consider also if
679              # default OptimizeResult will need updating.
680  
681              # determine whether any variables are fixed
682              i_fixed = (bounds.lb == bounds.ub)
683  
684              if np.all(i_fixed):
685                  # all the parameters are fixed, a minimizer is not able to do
686                  # anything
687                  return _optimize_result_for_equal_bounds(
688                      fun, bounds, meth, args=args, constraints=constraints
689                  )
690  
691              # determine whether finite differences are needed for any grad/jac
692              fd_needed = (not callable(jac))
693              for con in constraints:
694                  if not callable(con.get('jac', None)):
695                      fd_needed = True
696  
697              # If finite differences are ever used, remove all fixed variables
698              # Always remove fixed variables for TNC; see gh-14565
699              remove_vars = i_fixed.any() and (fd_needed or meth == "tnc")
700              if remove_vars:
701                  x_fixed = (bounds.lb)[i_fixed]
702                  x0 = x0[~i_fixed]
703                  bounds = _remove_from_bounds(bounds, i_fixed)
704                  fun = _remove_from_func(fun, i_fixed, x_fixed)
705                  if callable(callback):
706                      callback = _remove_from_func(callback, i_fixed, x_fixed)
707                  if callable(jac):
708                      jac = _remove_from_func(jac, i_fixed, x_fixed, remove=1)
709  
710                  # make a copy of the constraints so the user's version doesn't
711                  # get changed. (Shallow copy is ok)
712                  constraints = [con.copy() for con in constraints]
713                  for con in constraints:  # yes, guaranteed to be a list
714                      con['fun'] = _remove_from_func(con['fun'], i_fixed,
715                                                     x_fixed, min_dim=1,
716                                                     remove=0)
717                      if callable(con.get('jac', None)):
718                          con['jac'] = _remove_from_func(con['jac'], i_fixed,
719                                                         x_fixed, min_dim=2,
720                                                         remove=1)
721          bounds = standardize_bounds(bounds, x0, meth)
722  
723      callback = _wrap_callback(callback, meth)
724  
725      if meth == 'nelder-mead':
726          res = _minimize_neldermead(fun, x0, args, callback, bounds=bounds,
727                                     **options)
728      elif meth == 'powell':
729          res = _minimize_powell(fun, x0, args, callback, bounds, **options)
730      elif meth == 'cg':
731          res = _minimize_cg(fun, x0, args, jac, callback, **options)
732      elif meth == 'bfgs':
733          res = _minimize_bfgs(fun, x0, args, jac, callback, **options)
734      elif meth == 'newton-cg':
735          res = _minimize_newtoncg(fun, x0, args, jac, hess, hessp, callback,
736                                   **options)
737      elif meth == 'l-bfgs-b':
738          res = _minimize_lbfgsb(fun, x0, args, jac, bounds,
739                                 callback=callback, **options)
740      elif meth == 'tnc':
741          res = _minimize_tnc(fun, x0, args, jac, bounds, callback=callback,
742                              **options)
743      elif meth == 'cobyla':
744          res = _minimize_cobyla(fun, x0, args, constraints, callback=callback,
745                                 bounds=bounds, **options)
746      elif meth == 'cobyqa':
747          res = _minimize_cobyqa(fun, x0, args, bounds, constraints, callback,
748                                 **options)
749      elif meth == 'slsqp':
750          res = _minimize_slsqp(fun, x0, args, jac, bounds,
751                                constraints, callback=callback, **options)
752      elif meth == 'trust-constr':
753          res = _minimize_trustregion_constr(fun, x0, args, jac, hess, hessp,
754                                             bounds, constraints,
755                                             callback=callback, **options)
756      elif meth == 'dogleg':
757          res = _minimize_dogleg(fun, x0, args, jac, hess,
758                                 callback=callback, **options)
759      elif meth == 'trust-ncg':
760          res = _minimize_trust_ncg(fun, x0, args, jac, hess, hessp,
761                                    callback=callback, **options)
762      elif meth == 'trust-krylov':
763          res = _minimize_trust_krylov(fun, x0, args, jac, hess, hessp,
764                                       callback=callback, **options)
765      elif meth == 'trust-exact':
766          res = _minimize_trustregion_exact(fun, x0, args, jac, hess,
767                                            callback=callback, **options)
768      else:
769          raise ValueError(f'Unknown solver {method}')
770  
771      if remove_vars:
772          res.x = _add_to_array(res.x, i_fixed, x_fixed)
773          res.jac = _add_to_array(res.jac, i_fixed, np.nan)
774          if "hess_inv" in res:
775              res.hess_inv = None  # unknown
776  
777      if getattr(callback, 'stop_iteration', False):
778          res.success = False
779          res.status = 99
780          res.message = "`callback` raised `StopIteration`."
781  
782      return res

785  def minimize_scalar(fun, bracket=None, bounds=None, args=(),
786                      method=None, tol=None, options=None):
787      """Local minimization of scalar function of one variable. 
788   
789      Parameters 
790      ---------- 
791      fun : callable 
792          Objective function. 
793          Scalar function, must return a scalar. 
794   
795          Suppose the callable has signature ``f0(x, *my_args, **my_kwargs)``, where 
796          ``my_args`` and ``my_kwargs`` are required positional and keyword arguments. 
797          Rather than passing ``f0`` as the callable, wrap it to accept 
798          only ``x``; e.g., pass ``fun=lambda x: f0(x, *my_args, **my_kwargs)`` as the 
799          callable, where ``my_args`` (tuple) and ``my_kwargs`` (dict) have been 
800          gathered before invoking this function. 
801   
802      bracket : sequence, optional 
803          For methods 'brent' and 'golden', `bracket` defines the bracketing 
804          interval and is required. 
805          Either a triple ``(xa, xb, xc)`` satisfying ``xa < xb < xc`` and 
806          ``func(xb) < func(xa) and  func(xb) < func(xc)``, or a pair 
807          ``(xa, xb)`` to be used as initial points for a downhill bracket search 
808          (see `scipy.optimize.bracket`). 
809          The minimizer ``res.x`` will not necessarily satisfy 
810          ``xa <= res.x <= xb``. 
811      bounds : sequence, optional 
812          For method 'bounded', `bounds` is mandatory and must have two finite 
813          items corresponding to the optimization bounds. 
814      args : tuple, optional 
815          Extra arguments passed to the objective function. 
816      method : str or callable, optional 
817          Type of solver.  Should be one of: 
818   
819          - :ref:`Brent <optimize.minimize_scalar-brent>` 
820          - :ref:`Bounded <optimize.minimize_scalar-bounded>` 
821          - :ref:`Golden <optimize.minimize_scalar-golden>` 
822          - custom - a callable object (added in version 0.14.0), see below 
823   
824          Default is "Bounded" if bounds are provided and "Brent" otherwise. 
825          See the 'Notes' section for details of each solver. 
826   
827      tol : float, optional 
828          Tolerance for termination. For detailed control, use solver-specific 
829          options. 
830      options : dict, optional 
831          A dictionary of solver options. 
832   
833          maxiter : int 
834              Maximum number of iterations to perform. 
835          disp : bool 
836              Set to True to print convergence messages. 
837   
838          See :func:`show_options()` for solver-specific options. 
839   
840      Returns 
841      ------- 
842      res : OptimizeResult 
843          The optimization result represented as a ``OptimizeResult`` object. 
844          Important attributes are: ``x`` the solution array, ``success`` a 
845          Boolean flag indicating if the optimizer exited successfully and 
846          ``message`` which describes the cause of the termination. See 
847          `OptimizeResult` for a description of other attributes. 
848   
849      See also 
850      -------- 
851      minimize : Interface to minimization algorithms for scalar multivariate 
852          functions 
853      show_options : Additional options accepted by the solvers 
854   
855      Notes 
856      ----- 
857      This section describes the available solvers that can be selected by the 
858      'method' parameter. The default method is the ``"Bounded"`` Brent method if 
859      `bounds` are passed and unbounded ``"Brent"`` otherwise. 
860   
861      Method :ref:`Brent <optimize.minimize_scalar-brent>` uses Brent's 
862      algorithm [1]_ to find a local minimum.  The algorithm uses inverse 
863      parabolic interpolation when possible to speed up convergence of 
864      the golden section method. 
865   
866      Method :ref:`Golden <optimize.minimize_scalar-golden>` uses the 
867      golden section search technique [1]_. It uses analog of the bisection 
868      method to decrease the bracketed interval. It is usually 
869      preferable to use the *Brent* method. 
870   
871      Method :ref:`Bounded <optimize.minimize_scalar-bounded>` can 
872      perform bounded minimization [2]_ [3]_. It uses the Brent method to find a 
873      local minimum in the interval x1 < xopt < x2. 
874   
875      Note that the Brent and Golden methods do not guarantee success unless a 
876      valid ``bracket`` triple is provided. If a three-point bracket cannot be 
877      found, consider `scipy.optimize.minimize`. Also, all methods are intended 
878      only for local minimization. When the function of interest has more than 
879      one local minimum, consider :ref:`global_optimization`. 
880   
881      **Custom minimizers** 
882   
883      It may be useful to pass a custom minimization method, for example 
884      when using some library frontend to minimize_scalar. You can simply 
885      pass a callable as the ``method`` parameter. 
886   
887      The callable is called as ``method(fun, args, **kwargs, **options)`` 
888      where ``kwargs`` corresponds to any other parameters passed to `minimize` 
889      (such as `bracket`, `tol`, etc.), except the `options` dict, which has 
890      its contents also passed as `method` parameters pair by pair.  The method 
891      shall return an `OptimizeResult` object. 
892   
893      The provided `method` callable must be able to accept (and possibly ignore) 
894      arbitrary parameters; the set of parameters accepted by `minimize` may 
895      expand in future versions and then these parameters will be passed to 
896      the method. You can find an example in the scipy.optimize tutorial. 
897   
898      .. versionadded:: 0.11.0 
899   
900      References 
901      ---------- 
902      .. [1] Press, W., S.A. Teukolsky, W.T. Vetterling, and B.P. Flannery. 
903             Numerical Recipes in C. Cambridge University Press. 
904      .. [2] Forsythe, G.E., M. A. Malcolm, and C. B. Moler. "Computer Methods 
905             for Mathematical Computations." Prentice-Hall Series in Automatic 
906             Computation 259 (1977). 
907      .. [3] Brent, Richard P. Algorithms for Minimization Without Derivatives. 
908             Courier Corporation, 2013. 
909   
910      Examples 
911      -------- 
912      Consider the problem of minimizing the following function. 
913   
914      >>> def f(x): 
915      ...     return (x - 2) * x * (x + 2)**2 
916   
917      Using the *Brent* method, we find the local minimum as: 
918   
919      >>> from scipy.optimize import minimize_scalar 
920      >>> res = minimize_scalar(f) 
921      >>> res.fun 
922      -9.9149495908 
923   
924      The minimizer is: 
925   
926      >>> res.x 
927      1.28077640403 
928   
929      Using the *Bounded* method, we find a local minimum with specified 
930      bounds as: 
931   
932      >>> res = minimize_scalar(f, bounds=(-3, -1), method='bounded') 
933      >>> res.fun  # minimum 
934      3.28365179850e-13 
935      >>> res.x  # minimizer 
936      -2.0000002026 
937   
938      """
939      if not isinstance(args, tuple):
940          args = (args,)
941  
942      if callable(method):
943          meth = "_custom"
944      elif method is None:
945          meth = 'brent' if bounds is None else 'bounded'
946      else:
947          meth = method.lower()
948      if options is None:
949          options = {}
950  
951      if bounds is not None and meth in {'brent', 'golden'}:
952          message = f"Use of `bounds` is incompatible with 'method={method}'."
953          raise ValueError(message)
954  
955      if tol is not None:
956          options = dict(options)
957          if meth == 'bounded' and 'xatol' not in options:
958              warn("Method 'bounded' does not support relative tolerance in x; "
959                   "defaulting to absolute tolerance.",
960                   RuntimeWarning, stacklevel=2)
961              options['xatol'] = tol
962          elif meth == '_custom':
963              options.setdefault('tol', tol)
964          else:
965              options.setdefault('xtol', tol)
966  
967      # replace boolean "disp" option, if specified, by an integer value.
968      disp = options.get('disp')
969      if isinstance(disp, bool):
970          options['disp'] = 2 * int(disp)
971  
972      if meth == '_custom':
973          res = method(fun, args=args, bracket=bracket, bounds=bounds, **options)
974      elif meth == 'brent':
975          res = _recover_from_bracket_error(_minimize_scalar_brent,
976                                            fun, bracket, args, **options)
977      elif meth == 'bounded':
978          if bounds is None:
979              raise ValueError('The `bounds` parameter is mandatory for '
980                               'method `bounded`.')
981          res = _minimize_scalar_bounded(fun, bounds, args, **options)
982      elif meth == 'golden':
983          res = _recover_from_bracket_error(_minimize_scalar_golden,
984                                            fun, bracket, args, **options)
985      else:
986          raise ValueError(f'Unknown solver {method}')
987  
988      # gh-16196 reported inconsistencies in the output shape of `res.x`. While
989      # fixing this, future-proof it for when the function is vectorized:
990      # the shape of `res.x` should match that of `res.fun`.
991      res.fun = np.asarray(res.fun)[()]
992      res.x = np.reshape(res.x, res.fun.shape)[()]
993      return res

996  def _remove_from_bounds(bounds, i_fixed):
997      """Removes fixed variables from a `Bounds` instance"""
998      lb = bounds.lb[~i_fixed]
999      ub = bounds.ub[~i_fixed]
1000     return Bounds(lb, ub)  # don't mutate original Bounds object
1001

1003 def _remove_from_func(fun_in, i_fixed, x_fixed, min_dim=None, remove=0):
1004     """Wraps a function such that fixed variables need not be passed in"""
1005     def fun_out(x_in, *args, **kwargs):
1006         x_out = np.zeros_like(i_fixed, dtype=x_in.dtype)
1007         x_out[i_fixed] = x_fixed
1008         x_out[~i_fixed] = x_in
1009         y_out = fun_in(x_out, *args, **kwargs)
1010         y_out = np.array(y_out)
1011 
1012         if min_dim == 1:
1013             y_out = np.atleast_1d(y_out)
1014         elif min_dim == 2:
1015             y_out = np.atleast_2d(y_out)
1016 
1017         if remove == 1:
1018             y_out = y_out[..., ~i_fixed]
1019         elif remove == 2:
1020             y_out = y_out[~i_fixed, ~i_fixed]
1021 
1022         return y_out
1023     return fun_out
1024

1026 def _add_to_array(x_in, i_fixed, x_fixed):
1027     """Adds fixed variables back to an array"""
1028     i_free = ~i_fixed
1029     if x_in.ndim == 2:
1030         i_free = i_free[:, None] @ i_free[None, :]
1031     x_out = np.zeros_like(i_free, dtype=x_in.dtype)
1032     x_out[~i_free] = x_fixed
1033     x_out[i_free] = x_in.ravel()
1034     return x_out

1037 def _validate_bounds(bounds, x0, meth):
1038     """Check that bounds are valid."""
1039 
1040     msg = "An upper bound is less than the corresponding lower bound."
1041     if np.any(bounds.ub < bounds.lb):
1042         raise ValueError(msg)
1043 
1044     msg = "The number of bounds is not compatible with the length of `x0`."
1045     try:
1046         bounds.lb = np.broadcast_to(bounds.lb, x0.shape)
1047         bounds.ub = np.broadcast_to(bounds.ub, x0.shape)
1048     except Exception as e:
1049         raise ValueError(msg) from e
1050 
1051     return bounds

1053 def standardize_bounds(bounds, x0, meth):
1054     """Converts bounds to the form required by the solver."""
1055     if meth in {'trust-constr', 'powell', 'nelder-mead', 'cobyla', 'cobyqa',
1056                 'new'}:
1057         if not isinstance(bounds, Bounds):
1058             lb, ub = old_bound_to_new(bounds)
1059             bounds = Bounds(lb, ub)
1060     elif meth in ('l-bfgs-b', 'tnc', 'slsqp', 'old'):
1061         if isinstance(bounds, Bounds):
1062             bounds = new_bounds_to_old(bounds.lb, bounds.ub, x0.shape[0])
1063     return bounds
1064

1066 def standardize_constraints(constraints, x0, meth):
1067     """Converts constraints to the form required by the solver."""
1068     all_constraint_types = (NonlinearConstraint, LinearConstraint, dict)
1069     new_constraint_types = all_constraint_types[:-1]
1070     if constraints is None:
1071         constraints = []
1072     elif isinstance(constraints, all_constraint_types):
1073         constraints = [constraints]
1074     else:
1075         constraints = list(constraints)  # ensure it's a mutable sequence
1076 
1077     if meth in ['trust-constr', 'cobyqa', 'new']:
1078         for i, con in enumerate(constraints):
1079             if not isinstance(con, new_constraint_types):
1080                 constraints[i] = old_constraint_to_new(i, con)
1081     else:
1082         # iterate over copy, changing original
1083         for i, con in enumerate(list(constraints)):
1084             if isinstance(con, new_constraint_types):
1085                 old_constraints = new_constraint_to_old(con, x0)
1086                 constraints[i] = old_constraints[0]
1087                 constraints.extend(old_constraints[1:])  # appends 1 if present
1088 
1089     return constraints

1092 def _optimize_result_for_equal_bounds(
1093         fun, bounds, method, args=(), constraints=()
1094 ):
1095     """ 
1096     Provides a default OptimizeResult for when a bounded minimization method 
1097     has (lb == ub).all(). 
1098  
1099     Parameters 
1100     ---------- 
1101     fun: callable 
1102     bounds: Bounds 
1103     method: str 
1104     constraints: Constraint 
1105     """
1106     success = True
1107     message = 'All independent variables were fixed by bounds.'
1108 
1109     # bounds is new-style
1110     x0 = bounds.lb
1111 
1112     if constraints:
1113         message = ("All independent variables were fixed by bounds at values"
1114                    " that satisfy the constraints.")
1115         constraints = standardize_constraints(constraints, x0, 'new')
1116 
1117     maxcv = 0
1118     for c in constraints:
1119         pc = PreparedConstraint(c, x0)
1120         violation = pc.violation(x0)
1121         if np.sum(violation):
1122             maxcv = max(maxcv, np.max(violation))
1123             success = False
1124             message = (f"All independent variables were fixed by bounds, but "
1125                        f"the independent variables do not satisfy the "
1126                        f"constraints exactly. (Maximum violation: {maxcv}).")
1127 
1128     return OptimizeResult(
1129         x=x0, fun=fun(x0, *args), success=success, message=message, nfev=1,
1130         njev=0, nhev=0,
1131     )
