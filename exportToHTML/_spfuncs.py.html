11   def estimate_blocksize(A,efficiency=0.7):
12       """Attempt to determine the blocksize of a sparse matrix 
13    
14       Returns a blocksize=(r,c) such that 
15           - A.nnz / A.tobsr( (r,c) ).nnz > efficiency 
16       """
17       if not (issparse(A) and A.format in ("csc", "csr")):
18           A = csr_array(A)
19   
20       if A.nnz == 0:
21           return (1,1)
22   
23       if not 0 < efficiency < 1.0:
24           raise ValueError('efficiency must satisfy 0.0 < efficiency < 1.0')
25   
26       high_efficiency = (1.0 + efficiency) / 2.0
27       nnz = float(A.nnz)
28       M,N = A.shape
29   
30       if M % 2 == 0 and N % 2 == 0:
31           e22 = nnz / (4 * count_blocks(A,(2,2)))
32       else:
33           e22 = 0.0
34   
35       if M % 3 == 0 and N % 3 == 0:
36           e33 = nnz / (9 * count_blocks(A,(3,3)))
37       else:
38           e33 = 0.0
39   
40       if e22 > high_efficiency and e33 > high_efficiency:
41           e66 = nnz / (36 * count_blocks(A,(6,6)))
42           if e66 > efficiency:
43               return (6,6)
44           else:
45               return (3,3)
46       else:
47           if M % 4 == 0 and N % 4 == 0:
48               e44 = nnz / (16 * count_blocks(A,(4,4)))
49           else:
50               e44 = 0.0
51   
52           if e44 > efficiency:
53               return (4,4)
54           elif e33 > efficiency:
55               return (3,3)
56           elif e22 > efficiency:
57               return (2,2)
58           else:
59               return (1,1)

62   def count_blocks(A,blocksize):
63       """For a given blocksize=(r,c) count the number of occupied 
64       blocks in a sparse matrix A 
65       """
66       r,c = blocksize
67       if r < 1 or c < 1:
68           raise ValueError('r and c must be positive')
69   
70       if issparse(A):
71           if A.format == "csr":
72               M,N = A.shape
73               return csr_count_blocks(M,N,r,c,A.indptr,A.indices)
74           elif A.format == "csc":
75               return count_blocks(A.T,(c,r))
76       return count_blocks(csr_array(A),blocksize)
