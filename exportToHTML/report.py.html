3    class ReportBase:
4        COLUMN_NAMES: list[str] = NotImplemented
5        COLUMN_WIDTHS: list[int] = NotImplemented
6        ITERATION_FORMATS: list[str] = NotImplemented
7    
8        @classmethod
9        def print_header(cls):
10           fmt = ("|"
11                  + "|".join([f"{{:^{x}}}" for x in cls.COLUMN_WIDTHS])
12                  + "|")
13           separators = ['-' * x for x in cls.COLUMN_WIDTHS]
14           print(fmt.format(*cls.COLUMN_NAMES))
15           print(fmt.format(*separators))
16   
17       @classmethod
18       def print_iteration(cls, *args):
19           iteration_format = [f"{{:{x}}}" for x in cls.ITERATION_FORMATS]
20           fmt = "|" + "|".join(iteration_format) + "|"
21           print(fmt.format(*args))
22   
23       @classmethod
24       def print_footer(cls):
25           print()
26
