61   # Like sympify, but supports less stuff, and also ensures that direct
62   # sympy expressions don't have free variables
63   def simple_sympify(e):
64       if isinstance(e, bool):
65           return sympy.true if e else sympy.false
66       elif isinstance(e, int):
67           return sympy.Integer(e)
68       elif isinstance(e, float):
69           # infinity is special; we use it to bracket integers as well
70           if math.isinf(e):
71               return sympy.oo if e > 0 else -sympy.oo
72           return sympy.Float(e)
73       elif isinstance(e, sympy.Expr):
74           assert e.is_number, e
75           # NaNs can occur when doing things like 0 * sympy.oo, but it is better
76           # if the operator notices this and takes care of it, because sometimes
77           # the NaN is inappropriate (for example, for ints, the [-oo, oo] range
78           # should go to zero when multiplied with [0, 0])
79           assert e != sympy.nan
80           return e
81       elif isinstance(e, BooleanAtom):
82           return e
83       else:
84           raise AssertionError(f"not simple sympy type {type(e)}: {e}")

87   # Sympy atomics only. Unlike <=, it also works on Sympy bools.
88   def sympy_generic_le(lower, upper):
89       if isinstance(lower, sympy.Expr):
90           assert isinstance(upper, sympy.Expr)
91           # instead of lower <= upper, we do upper >= lower since upper is mostly int_oo
92           # and we have better code paths there.
93           return upper >= lower
94       else:
95           # only negative condition is True > False
96           assert isinstance(lower, SympyBoolean) and isinstance(upper, SympyBoolean), (
97               lower,
98               upper,
99           )
100          return not (lower and not upper)

123  class ValueRanges(Generic[_T]):
124      if TYPE_CHECKING:
125          # ruff doesn't understand circular references but mypy does
126          ExprVR = ValueRanges[sympy.Expr]  # noqa: F821
127          BoolVR = ValueRanges[SympyBoolean]  # noqa: F821

131      # sympy expression, in practice the analysis here only works

136      is_int: bool

158      def __init__(self, lower: AllIn, upper: AllIn) -> None:
159          lower = simple_sympify(lower)
160          upper = simple_sympify(upper)
161          # TODO: when the bounds have free variables, this may be
162          # nontrivial to actually verify
163          try:
164              if not sympy_generic_le(lower, upper):
165                  raise ValueRangeError(f"Invalid ranges [{lower}:{upper}]")
166          except TypeError as e:
167              raise TypeError(f"Could not compare {lower} <= {upper}") from e
168  
169          is_bool_lower = isinstance(lower, SympyBoolean)
170          is_bool_upper = isinstance(upper, SympyBoolean)
171          assert is_bool_lower == is_bool_upper, (lower, upper)
172  
173          # Warning: is_int/is_float is best effort.  We do pretty well in
174          # Dynamo, but in Inductor these attributes are often wrong because we
175          # are not very rigorous in dtype analysis.  This is also why we need
176          # the flexible analysis for is_int: sometimes a sympy.oo pops in for
177          # an integer bound. I would /like/ for us not to do this, but it's
178          # too hard to push the invariant through right now.
179          if isinstance(lower, sympy.Integer) and upper == sympy.oo:
180              upper = int_oo
181          if isinstance(upper, sympy.Integer) and lower == -sympy.oo:
182              lower = -int_oo
183          # NB: [-int_oo, -int_oo] and [int_oo, int_oo] are allowed
184          integer_types = (sympy.Integer, NegativeIntInfinity, IntInfinity)
185          is_int_lower = isinstance(lower, integer_types)
186          is_int_upper = isinstance(upper, integer_types)
187  
188          # Because this is a frozen class
189          object.__setattr__(self, "lower", lower)
190          object.__setattr__(self, "upper", upper)
191          # Unlike bool/int in Python, we don't report bools are ints
192          #
193          # NB: is_bool_lower == is_bool_upper, so we only need to check one
194          object.__setattr__(self, "is_bool", is_bool_lower)
195          object.__setattr__(
196              self,
197              "is_int",
198              not self.is_bool and is_int_lower and is_int_upper,
199          )
200          """ 
201          # This assert is just impossible right now, too many sympy bugs 
202          if self.is_int: 
203              # NB: sympy will sometimes randomly lose the float-ness of zero, 
204              # so we also need to account for that in the assertion here. 
205              # See also https://github.com/sympy/sympy/issues/26620 
206              assert isinstance(lower, sympy.Integer) or lower in [-sympy.oo, 0], ( 
207                  lower, 
208                  upper, 
209              ) 
210              assert isinstance(upper, sympy.Integer) or upper in [sympy.oo, 0], (lower, upper) 
211          """
212          # NB: [-oo, oo] always advertises as float!
213          object.__setattr__(self, "is_float", not self.is_bool and not self.is_int)
214          assert self.is_bool or self.is_int or self.is_float, (lower, upper)

216      def boolify(self) -> ValueRanges[SympyBoolean]:
217          if vr_is_bool(self):
218              return self
219          elif self == ValueRanges.unknown():
220              return ValueRanges.unknown_bool()
221          else:
222              raise AssertionError(f"not bool like {self}")

224      def __contains__(self, x: AllIn) -> bool:
225          return ValueRanges.wrap(x).issubset(self)
226  
227      def issubset(self, other):
228          if other is self.unknown_int():
229              return True
230          return sympy_generic_le(other.lower, self.lower) and sympy_generic_le(
231              self.upper, other.upper
232          )

234      def tighten(self, other) -> ValueRanges:
235          """Given two ValueRanges, returns their intersection"""
236          return self & other

253      def __and__(self: AllVR, other: AllVR) -> AllVR:
254          if other in (ValueRanges.unknown(), ValueRanges.unknown_int()):
255              return self
256          if self in (ValueRanges.unknown(), ValueRanges.unknown_int()):
257              return other
258          assert self.is_bool == other.is_bool, (self, other)
259          assert self.is_int == other.is_int, (self, other)
260          assert self.is_float == other.is_float, (self, other)
261          if self.is_bool:
262              return ValueRanges(
263                  sympy.Or(self.lower, other.lower), sympy.And(self.upper, other.upper)
264              )
265          else:
266              return ValueRanges(
267                  sympy.Max(self.lower, other.lower), sympy.Min(self.upper, other.upper)
268              )

285      def __or__(self: AllVR, other: AllVR) -> AllVR:
286          if ValueRanges.unknown() in (self, other):
287              return ValueRanges.unknown()
288          assert self.is_bool == other.is_bool, (self, other)
289          assert self.is_int == other.is_int, (self, other)
290          assert self.is_float == other.is_float, (self, other)
291          if self.is_bool:
292              return ValueRanges(
293                  sympy.And(self.lower, other.lower), sympy.Or(self.upper, other.upper)
294              )
295          else:
296              return ValueRanges(
297                  sympy.Min(self.lower, other.lower), sympy.Max(self.upper, other.upper)
298              )

329      @staticmethod
330      def wrap(arg: Union[AllIn, AllVR]) -> AllVR:
331          if isinstance(arg, ValueRanges):
332              return arg
333          if isinstance(arg, float) and math.isnan(arg):
334              return ValueRanges.unknown()

338      @staticmethod
339      def increasing_map(x: Union[ExprIn, ExprVR], fn: ExprFn) -> ExprVR:
340          """Increasing: x <= y => f(x) <= f(y)."""
341          x = ValueRanges.wrap(x)
342          return ValueRanges(fn(x.lower), fn(x.upper))

354      @staticmethod
355      def decreasing_map(x: Union[AllIn, AllVR], fn: AllFn) -> AllVR:
356          """Decreasing: x <= y => f(x) >= f(y)."""
357          x = ValueRanges.wrap(x)
358          # consistently either Expr or Bool, but we don't know it here
359          return ValueRanges(fn(x.upper), fn(x.lower))  # type: ignore[arg-type]
360  
361      @staticmethod
362      def monotone_map(x: Union[ExprIn, ExprVR], fn: ExprFn) -> ExprVR:
363          """It's increasing or decreasing."""
364          x = ValueRanges.wrap(x)
365          l = fn(x.lower)
366          u = fn(x.upper)
367          return ValueRanges(min(l, u), max(l, u))
368  
369      @staticmethod
370      def convex_min_zero_map(x: Union[ExprIn, ExprVR], fn: ExprFn) -> ExprVR:
371          """Fn is convex and has a minimum at 0."""
372          x = ValueRanges.wrap(x)
373          if 0 in x:
374              upper = max(fn(x.lower), fn(x.upper))
375              upper = simple_sympify(upper)
376              if isinstance(upper, sympy.Float) or upper == sympy.oo:
377                  return ValueRanges(0.0, upper)
378              return ValueRanges(0, upper)
379          return ValueRanges.monotone_map(x, fn)

399      @staticmethod
400      def coordinatewise_increasing_map(
401          x: Union[AllIn, AllVR],
402          y: Union[AllIn, AllVR],
403          fn: AllFn2,
404      ) -> AllVR:
405          """ 
406          It's increasing on each coordinate. 
407   
408          Mathematically: 
409          For every 1 <= i <= n and x_i <= y_i we have that 
410          f(x1, .., xn) <= f(x1, , yi, ..., xn) 
411          """
412          x, y = ValueRanges.wrap(x), ValueRanges.wrap(y)
413          return ValueRanges(
414              fn(x.lower, y.lower),  # type: ignore[arg-type]
415              fn(x.upper, y.upper),  # type: ignore[arg-type]
416          )
417  
418      @classmethod
419      def coordinatewise_monotone_map(cls, x, y, fn):
420          """It's increasing or decreasing on each coordinate."""
421          x, y = cls.wrap(x), cls.wrap(y)
422          products = [
423              fn(a, b)
424              for a, b in itertools.product([x.lower, x.upper], [y.lower, y.upper])
425          ]
426          return ValueRanges(min(products), max(products))

431      It gives bounds on a SymPy operator given bounds on its arguments 
432      See the function `bound_sympy` for a function that applies this logic to a full SymPy expression

435      @staticmethod
436      def constant(value, dtype):
437          if isinstance(value, ValueRanges):
438              assert value.is_singleton()
439              value = value.lower
440          # NB: value is NOT a sympy expression, it's a constant!
441          is_python = isinstance(value, (int, float, bool))
442          assert is_python or isinstance(
443              value, (BooleanAtom, sympy.Integer, sympy.Number)
444          )
445  
446          # using nan makes subsequent computation throw, and for the purposes of optimization
447          # returning -math.inf - math.inf is equivalent to giving up
448          if isinstance(value, SupportsFloat) and math.isnan(value):
449              if dtype == torch.bool:
450                  return ValueRanges.unknown_bool()
451              elif dtype.is_floating_point:
452                  return ValueRanges.unknown()
453              else:
454                  return ValueRanges.unknown_int()
455  
456          if is_python:
457              type_ = dtype_to_type(dtype)
458              value = type_(value)
459          else:
460              # We do a type check on a best-effort basis
461              # We don't want to force a cast to sympy.Float if the value is Rational to avoid losing precision
462              if dtype == torch.bool:
463                  assert isinstance(value, BooleanAtom)
464              elif dtype.is_floating_point:
465                  assert not value.is_finite or value.is_real
466              else:
467                  # dtype is intXX
468                  assert value.is_integer
469  
470          r = ValueRanges.wrap(value)
471          return r

502      @staticmethod
503      def _bool_to_int(x):
504          if x.is_singleton():
505              return ValueRanges.wrap(sympy.Integer(1 if x.lower else 0))
506          else:
507              return ValueRanges(sympy.Integer(0), sympy.Integer(1))

509      @classmethod
510      def bitwise_and(cls, a, b):
511          a, b = ValueRanges.wrap(a), ValueRanges.wrap(b)
512          if a.is_bool and b.is_bool:
513              return cls.and_(a, b)
514          if a.is_bool:
515              a = cls._bool_to_int(a)
516          if b.is_bool:
517              b = cls._bool_to_int(b)
518          lower = min(a.lower, b.lower)
519          if lower < 0 and lower != -sympy.oo and lower != -int_oo:
520              # If both lower bounds are negative, then bits start like
521              # 1...10..., so the smallest possible value is 1...101...1.
522              # Thus, we need to find the next smallest power of 2 (inclusive).
523              try:
524                  lower = -(1 << int(-lower - 1).bit_length())
525              except Exception:
526                  lower = -int_oo
527          else:
528              lower = 0
529          return ValueRanges(lower, max(a.upper, b.upper))
530  
531      @classmethod
532      def bitwise_or(cls, a, b):
533          a, b = ValueRanges.wrap(a), ValueRanges.wrap(b)
534          if a.is_bool and b.is_bool:
535              return cls.or_(a, b)
536          if a.is_bool:
537              a = cls._bool_to_int(a)
538          if b.is_bool:
539              b = cls._bool_to_int(b)
540          upper = max(a.upper, b.upper)
541          if upper == 0:
542              upper = 0
543          elif upper > 0 and upper != sympy.oo and upper != int_oo:
544              # If both upper bounds are positive, then the largest
545              # possible value is 01...1, so we need to find
546              # next largest power of 2 (exclusive), minus 1
547              try:
548                  upper = (1 << int(upper).bit_length()) - 1
549              except Exception:
550                  upper = int_oo
551          elif upper < 0:
552              upper = -1
553          return ValueRanges(min(a.lower, b.lower), upper)
554  
555      @staticmethod
556      def eq(a, b):
557          a = ValueRanges.wrap(a)
558          b = ValueRanges.wrap(b)
559          if a.is_singleton() and b.is_singleton() and a.lower == b.lower:
560              return ValueRanges.wrap(sympy.true)
561          elif a.lower > b.upper or b.lower > a.upper:  # ranges disjoint
562              return ValueRanges.wrap(sympy.false)
563          return ValueRanges(sympy.false, sympy.true)

573      @classmethod
574      def lt(cls, a, b):
575          a = ValueRanges.wrap(a)
576          b = ValueRanges.wrap(b)
577          assert a.is_bool == b.is_bool
578          if a.is_bool:
579              return cls.and_(cls.not_(a), b)
580          else:
581              if a.upper < b.lower:
582                  return ValueRanges.wrap(sympy.true)
583              elif a.lower >= b.upper:
584                  return ValueRanges.wrap(sympy.false)
585              return ValueRanges(sympy.false, sympy.true)

599      @staticmethod
600      def add(a, b):
601          return ValueRanges.coordinatewise_increasing_map(

605      @classmethod
606      def mul(cls, a, b):
607          a = ValueRanges.wrap(a)
608          b = ValueRanges.wrap(b)
609  
610          assert a.is_bool == b.is_bool
611          if a.is_bool:
612              return cls.and_(a, b)
613  
614          def safe_mul(a, b):
615              # Make unknown() * wrap(0.0) == wrap(0.0)
616              if a == 0.0 or a == 0:
617                  return a
618              elif b == 0.0 or b == 0:
619                  return b
620              else:
621                  return a * b
622  
623          return ValueRanges.coordinatewise_monotone_map(a, b, _keep_float(safe_mul))

625      @staticmethod
626      def int_truediv(a, b):
627          a = ValueRanges.wrap(a)
628          b = ValueRanges.wrap(b)
629          if 0 in b or ((-int_oo in a or int_oo in a) and (-int_oo in b or int_oo in b)):
630              return ValueRanges.unknown()
631          else:
632              return ValueRanges.coordinatewise_monotone_map(
633                  a, b, _keep_float(IntTrueDiv)

636      @staticmethod
637      def truediv(a, b):
638          a = ValueRanges.wrap(a)
639          b = ValueRanges.wrap(b)
640          if 0 in b or (
641              (-sympy.oo in a or sympy.oo in a) and (-sympy.oo in b or sympy.oo in b)
642          ):
643              return ValueRanges.unknown()
644          else:
645              return ValueRanges.coordinatewise_monotone_map(
646                  a, b, _keep_float(FloatTrueDiv)
647              )

649      @staticmethod
650      def floordiv(a, b):
651          a = ValueRanges.wrap(a)
652          b = ValueRanges.wrap(b)
653          if 0 in b:
654              return ValueRanges.unknown_int()
655          products = []
656          for x, y in itertools.product([a.lower, a.upper], [b.lower, b.upper]):

672              ret = abs(a) % abs(b)
673              if a < 0:

678              x = a / b

688              if c_div(x.lower, y_val) == c_div(x.upper, y_val):
689                  return ValueRanges.increasing_map(x, lambda u: c_mod(u, y_val))

700                  return ValueRanges(lower, upper)
701          else:
702              # Too difficult, we bail out
703              upper = cls.abs(y).upper - 1
704              return ValueRanges(-upper, upper)

848      # will carry out the floor/ceil at too high precision and then something
849      # bad will happen when we convert it to float.

851      # For truncation, the implementation is clearly sound, because the desired
852      # target float is always exactly representable, since you're just chopping
853      # off bits the mantissa.  But what about ceil/floor?
854      #
855      # The important constraint here is that we're not defining floor on
856      # arbitrary real numbers, only representable float numbers.  So we can
857      # take advantage of the fact that before we reach the first
858      # unrepresentable integer in floating point space, we have the range of

860      # amounts.  floor/ceil is an identity operation in this case.  In the

1009 ) -> ValueRanges:
1010     log.debug(

1012         expr,

1014             lambda: (
1015                 "\n"
1016                 + "\n".join(
1017                     f"  {k}: {r}" for k, r in ranges.items() if k in expr.free_symbols

1030     context = torch._guards.TracingContext.try_get()
1031     if context and context.fake_mode.shape_env:
1032         if ranges:
1033             ranges = {**context.fake_mode.shape_env.var_to_range, **ranges}
1034         else:
1035             ranges = context.fake_mode.shape_env.var_to_range

1037     def missing_handler(s):

1039             if s.is_positive:  # type: ignore[attr-defined]

1042                 vr = ValueRanges(0, int_oo)
