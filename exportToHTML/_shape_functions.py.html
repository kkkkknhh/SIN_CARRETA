27   def broadcast(a: list[int], b: list[int]):
28       dimsA = len(a)
29       dimsB = len(b)
30       ndim = max(dimsA, dimsB)
31       expandedSizes: list[int] = []
32   
33       for i in range(ndim):
34           offset = ndim - 1 - i
35           dimA = dimsA - 1 - offset
36           dimB = dimsB - 1 - offset
37           sizeA = a[dimA] if (dimA >= 0) else 1
38           sizeB = b[dimB] if (dimB >= 0) else 1
39   
40           if sizeA != sizeB and sizeA != 1 and sizeB != 1:
41               # TODO: only assertion error is bound in C++ compilation right now
42               raise AssertionError(
43                   f"The size of tensor a {sizeA} must match the size of tensor b ({sizeB}) at non-singleton dimension {i}"
44               )
45   
46           expandedSizes.append(sizeB if sizeA == 1 else sizeA)
47   
48       return expandedSizes

84   def broadcast_inplace(a: list[int], b: list[int]):
85       dimsA = len(a)
86       dimsB = len(b)
87       if dimsB > dimsA:
88           raise AssertionError(
89               f"The dims of tensor b ({dimsB}) must be less than or equal tothe dims of tensor a ({dimsA}) "
90           )
91       for dimA in range(dimsA):
92           dimB = dimsB - dimsA + dimA
93           sizeA = a[dimA]
94           sizeB = b[dimB] if (dimB >= 0) else 1
95           if sizeA != sizeB and sizeB != 1:
96               # TODO: only assertion error is bound in C++ compilation right now
97               raise AssertionError(
98                   "The size of tensor a {} must match the size of tensor b ("
99                   "{}) at non-singleton dimension {}".format(sizeA, sizeB, dimA)
100              )
101      return _copy(a)

104  def expand(self: list[int], sizes: list[int]):
105      assert len(sizes) >= len(self)
106      ndim = len(sizes)
107      tensor_dim = len(self)
108      if ndim == 0:
109          return _copy(sizes)
110      out: list[int] = []
111      for i in range(ndim):
112          offset = ndim - 1 - i
113          dim = tensor_dim - 1 - offset
114          size = self[dim] if dim >= 0 else 1
115          targetSize = sizes[i]
116          if targetSize == -1:
117              assert dim >= 0
118              targetSize = size
119          if size != targetSize:
120              assert size == 1
121              size = targetSize
122          out.append(size)
123      return out

130  def infer_size_impl(shape: list[int], numel: int) -> list[int]:
131      newsize = 1
132      infer_dim: Optional[int] = None
133      for dim in range(len(shape)):
134          if shape[dim] == -1:
135              if infer_dim is not None:
136                  raise AssertionError("only one dimension can be inferred")
137              infer_dim = dim
138          elif shape[dim] >= 0:
139              newsize *= shape[dim]
140          else:
141              raise AssertionError("invalid shape dimensions")
142      if not (
143          numel == newsize
144          or (infer_dim is not None and newsize > 0 and numel % newsize == 0)
145      ):
146          raise AssertionError("invalid shape")
147      out = _copy(shape)
148      if infer_dim is not None:
149          out[infer_dim] = numel // newsize
150      return out

200  def pooling_output_shape_pad_lr(
201      inputSize: int,
202      kernelSize: int,
203      pad_l: int,
204      pad_r: int,
205      stride: int,
206      dilation: int,
207      ceil_mode: bool,
208  ):
209      outputSize = (
210          div_rtn(
211              inputSize
212              + pad_l
213              + pad_r
214              - dilation * (kernelSize - 1)
215              - 1
216              + (stride - 1 if ceil_mode else 0),
217              stride,
218          )
219          + 1
220      )
221      if ceil_mode:
222          if (outputSize - 1) * stride >= inputSize + pad_l:
223              outputSize = outputSize - 1
224      return outputSize

275  def max_pool2d(
276      input: list[int],
277      kernel_size: list[int],
278      stride: list[int],
279      padding: list[int],
280      dilation: list[int],
281      ceil_mode: bool,
282  ):
283      assert len(kernel_size) == 1 or len(kernel_size) == 2, (
284          "max_pool2d: kernel_size must either be a single int, or a tuple of two ints"
285      )
286      kH = kernel_size[0]
287      kW = kH if len(kernel_size) == 1 else kernel_size[1]
288  
289      assert len(stride) == 0 or len(stride) == 1 or len(stride) == 2, (
290          "max_pool2d: stride must either be omitted, a single int, or a tuple of two ints"
291      )
292      dH = kH if len(stride) == 0 else stride[0]
293      if len(stride) == 0:
294          dW = kW
295      elif len(stride) == 1:
296          dW = dH
297      else:
298          dW = stride[1]
299  
300      assert len(padding) == 1 or len(padding) == 2, (
301          "max_pool2d: padding must either be a single int, or a tuple of two ints"
302      )
303      padH = padding[0]
304      padW = padH if len(padding) == 1 else padding[1]
305  
306      assert len(dilation) == 1 or len(dilation) == 2, (
307          "max_pool2d: dilation must be either a single int, or a tuple of two ints"
308      )
309      dilationH = dilation[0]
310      dilationW = dilationH if len(dilation) == 1 else dilation[1]
311  
312      assert len(input) == 3 or len(input) == 4
313  
314      nbatch = input[-4] if len(input) == 4 else 1
315      nInputPlane = input[-3]
316      inputHeight = input[-2]
317      inputWidth = input[-1]
318  
319      outputHeight = pooling_output_shape(inputHeight, kH, padH, dH, dilationH, ceil_mode)
320      outputWidth = pooling_output_shape(inputWidth, kW, padW, dW, dilationW, ceil_mode)
321  
322      pool2d_shape_check(
323          input,
324          kH,
325          kW,
326          dH,
327          dW,
328          padH,
329          padW,
330          dilationH,
331          dilationW,
332          nInputPlane,
333          inputHeight,
334          inputWidth,
335          outputHeight,
336          outputWidth,
337      )
338  
339      if len(input) == 3:
340          return [nInputPlane, outputHeight, outputWidth]
341      else:
342          return [nbatch, nInputPlane, outputHeight, outputWidth]

486  def slice(
487      self: list[int], dim: int, start: Optional[int], end: Optional[int], step: int
488  ):
489      ndim = len(self)
490      assert ndim != 0
491      dim = maybe_wrap_dim(dim, ndim)
492      start_val = start if start is not None else 0
493      end_val = end if end is not None else max_int()
494      assert step > 0
495      if start_val == max_int():
496          start_val = 0
497      if start_val < 0:
498          start_val += self[dim]
499      if end_val < 0:
500          end_val += self[dim]
501      if start_val < 0:
502          start_val = 0
503      elif start_val > self[dim]:
504          start_val = self[dim]
505      if end_val < start_val:
506          end_val = start_val
507      elif end_val >= self[dim]:
508          end_val = self[dim]
509      slice_len = end_val - start_val
510      out = _copy(self)
511      out[dim] = (slice_len + step - 1) // step
512      return out

548  def cat(tensors: list[list[int]], dim: int):
549      check_cat_no_zero_dim(tensors)
550      dim = legacy_cat_wrap_dim(dim, tensors)
551      assert len(tensors) > 0
552      not_skipped_tensor: Optional[list[int]] = None
553      for tensor in tensors:
554          if not should_skip(tensor):
555              not_skipped_tensor = tensor
556      if not_skipped_tensor is None:
557          return [0]
558  
559      cat_dim_size = 0
560  
561      for i in range(len(tensors)):
562          tensor = tensors[i]
563          if not should_skip(tensor):
564              check_cat_shape_except_dim(not_skipped_tensor, tensor, dim, i)
565              cat_dim_size = cat_dim_size + tensor[dim]
566  
567      result_size = _copy(not_skipped_tensor)
568      result_size[dim] = cat_dim_size
569      return result_size

595  def matmul(tensor1: list[int], tensor2: list[int]):
596      dim_tensor1 = len(tensor1)
597      dim_tensor2 = len(tensor2)
598      if dim_tensor1 == 1 and dim_tensor2 == 1:
599          return dot(tensor1, tensor2)
600      elif dim_tensor1 == 2 and dim_tensor2 == 1:
601          return mv(tensor1, tensor2)
602      elif dim_tensor1 == 1 and dim_tensor2 == 2:
603          return squeeze(mm(unsqueeze(tensor1, 0), tensor2), 0)
604      elif dim_tensor1 == 2 and dim_tensor2 == 2:
605          return mm(tensor1, tensor2)
606      elif dim_tensor1 >= 1 and dim_tensor2 >= 1:
607          # We are multiplying b1 x n x m1 by x2 x m2 x p (where b1 can be a list);
608          # we track m1 vs m2 separately even though they must match for nicer error messages
609          n = tensor1[-2] if dim_tensor1 > 1 else 1
610          batch_tensor1: list[int] = []
611          # TODO: handling of slice
612          for i in range(dim_tensor1 - 2):
613              batch_tensor1.append(tensor1[i])
614          p = tensor2[-1]
615          batch_tensor2: list[int] = []
616          # TODO: handling of slice
617          for i in range(dim_tensor2 - 2):
618              batch_tensor2.append(tensor2[i])
619  
620          # expand the batch portion (i.e. cut off matrix dimensions and expand rest)
621          expand_batch_portion = broadcast(batch_tensor1, batch_tensor2)
622  
623          # todo: copy ?
624          output_shape = expand_batch_portion
625          if dim_tensor1 > 1:
626              output_shape.append(n)
627  
628          if dim_tensor2 > 1:
629              output_shape.append(p)
630  
631          return output_shape
632      else:
633          assert False, "both  arguments to matmul need to be at least 1D"

648  def transpose(self: list[int], dim0: int, dim1: int):
649      ndims = len(self)
650      dim0 = maybe_wrap_dim(dim0, ndims)
651      dim1 = maybe_wrap_dim(dim1, ndims)
652      if dim0 == dim1:
653          return _copy(self)
654      out: list[int] = []
655      for i in range(ndims):
656          if i == dim0:
657              out.append(self[dim1])
658          elif i == dim1:
659              out.append(self[dim0])
660          else:
661              out.append(self[i])
662      return out

706      assert bias is None or (len(bias) == 1 and bias[0] == weight_sizes[0])

708      for i in range(2, k):
709          assert (input[i] + 2 * padding[i - 2]) >= (
710              dilation[i - 2] * (weight_sizes[i] - 1) + 1
711          )

716  def conv_output_size(
717      input_size: list[int],
718      weight_size: list[int],
719      bias: Optional[list[int]],
720      stride: list[int],
721      padding: list[int],
722      dilation: list[int],
723      groups: int,
724  ):
725      check_shape_forward(
726          input_size, weight_size, bias, stride, padding, dilation, groups
727      )
728  
729      has_dilation = len(dilation) > 0
730      dim = len(input_size)
731      output_size: list[int] = []
732      input_batch_size_dim = 0
733      weight_output_channels_dim = 0
734      output_size.append(input_size[input_batch_size_dim])
735      output_size.append(weight_size[weight_output_channels_dim])
736  
737      for d in range(2, dim):
738          dilation_ = dilation[d - 2] if has_dilation else 1
739          kernel = dilation_ * (weight_size[d] - 1) + 1
740          output_size.append(
741              (input_size[d] + (2 * padding[d - 2]) - kernel) // stride[d - 2] + 1
742          )
743      return output_size

784  def conv_transpose2d_input(
785      input: list[int],
786      weight: list[int],
787      bias: Optional[list[int]] = None,
788      stride: Optional[list[int]] = None,
789      padding: Optional[list[int]] = None,
790      output_padding: Optional[list[int]] = None,
791      groups: int = 1,
792      dilation: Optional[list[int]] = None,
793  ) -> list[int]:
794      if stride is None:
795          stride = [1, 1]
796      if padding is None:
797          padding = [0, 0]
798      if output_padding is None:
799          output_padding = [0, 0]

801          dilation = [1, 1]

810      for d in range(2, dim):
811          dilation_ = dilation[d - 2] if has_dilation else 1
812          kernel = dilation_ * (weight[d] - 1)
813          output_size.append(
814              (input[d] - 1) * stride[d - 2]
815              - 2 * padding[d - 2]
816              + kernel
817              + output_padding[d - 2]
818              + 1
819          )
820      return output_size

823  def conv_forwards(
824      input: list[int],
825      weight: list[int],
826      bias: Optional[list[int]],
827      stride: list[int],
828      padding: list[int],
829      dilation: list[int],
830      transposed: bool,
831      output_padding: list[int],
832      groups: int,
833  ) -> list[int]:
834      has_dilation = len(dilation) > 0
835      has_output_padding = len(output_padding) > 0
836      dim = len(input)
837      output_size: list[int] = []
838      input_batch_size_dim = 0
839      weight_output_channels_dim = 1 if transposed else 0
840      output_size.append(input[input_batch_size_dim])
841      if transposed:
842          output_size.append(weight[weight_output_channels_dim] * groups)
843      else:
844          output_size.append(weight[weight_output_channels_dim])
845  
846      for d in range(2, dim):
847          dilation_ = dilation[d - 2] if has_dilation else 1
848          output_padding_ = output_padding[d - 2] if has_output_padding else 0
849          if transposed:
850              kernel = dilation_ * (weight[d] - 1)
851              output_size.append(
852                  (input[d] - 1) * stride[d - 2]
853                  - 2 * padding[d - 2]
854                  + kernel
855                  + output_padding_
856                  + 1
857              )
858          else:
859              kernel = dilation_ * (weight[d] - 1) + 1
860              output_size.append(
861                  (input[d] + (2 * padding[d - 2]) - kernel) // stride[d - 2] + 1
862              )
863      return output_size

973  def permute(input: list[int], dims: list[int]):
974      assert len(input) == len(dims)
975      ndim = len(dims)
976      seen_dims: list[int] = []
977      newSizes: list[int] = []
978      for i in range(ndim):
979          dim = maybe_wrap_dim(dims[i], ndim)
980          seen_dims.append(dim)
981          newSizes.append(input[dim])
982      for i in range(1, ndim):
983          for j in range(i):
984              assert seen_dims[i] != seen_dims[j]
985      return newSizes

988  def movedim(self: list[int], source: list[int], destination: list[int]) -> list[int]:
989      self_dim = len(self)
990      if self_dim <= 1:
991          return self
992      normalized_src: list[int] = []
993      normalized_dst: list[int] = []
994      for i in range(len(source)):
995          normalized_src.append(maybe_wrap_dim(source[i], self_dim))
996          normalized_dst.append(maybe_wrap_dim(destination[i], self_dim))
997      order = [-1 for i in range(self_dim)]
998      src_dims = [i for i in range(self_dim)]
999      dst_dims = [i for i in range(self_dim)]
1000 
1001     for i in range(len(source)):
1002         order[normalized_dst[i]] = normalized_src[i]
1003         src_dims[normalized_src[i]] = -1
1004         dst_dims[normalized_dst[i]] = -1
1005 
1006     source_dims: list[int] = []
1007     destination_dims: list[int] = []
1008     for ele in src_dims:
1009         if ele != -1:
1010             source_dims.append(ele)
1011     for ele in dst_dims:
1012         if ele != -1:
1013             destination_dims.append(ele)
1014 
1015     rest_dim = self_dim - len(source)
1016     for i in range(rest_dim):
1017         order[destination_dims[i]] = source_dims[i]
1018     return permute(self, order)

1021 def flatten(input: list[int], start_dim: int, end_dim: int):
1022     start_dim = maybe_wrap_dim(start_dim, len(input))
1023     end_dim = maybe_wrap_dim(end_dim, len(input))
1024     assert start_dim <= end_dim
1025     if len(input) == 0:
1026         return [1]
1027     if start_dim == end_dim:
1028         # TODO: return self
1029         out: list[int] = []
1030         for elem in input:
1031             out.append(elem)
1032         return out
1033     slice_numel = 1
1034     for i in range(start_dim, end_dim + 1):
1035         slice_numel *= input[i]
1036     # TODO: use slicing when slice optimization has landed
1037     # slice_numel = multiply_integers(input[start_dim:end_dim - start_dim + 1])
1038     shape: list[int] = []
1039     for i in range(start_dim):
1040         shape.append(input[i])
1041     shape.append(slice_numel)
1042     for i in range(end_dim + 1, len(input)):
1043         shape.append(input[i])
1044     return shape
