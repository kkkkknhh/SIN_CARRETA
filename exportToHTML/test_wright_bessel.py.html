37   @pytest.mark.parametrize('b', [0, 1e-6, 0.1, 0.5, 1, 10])
38   @pytest.mark.parametrize('x', [0, 1e-6, 0.1, 0.5, 1])
39   def test_wright_bessel_iv(b, x):
40       """Test relation of wright_bessel and modified bessel function iv. 
41    
42       iv(z) = (1/2*z)**v * Phi(1, v+1; 1/4*z**2). 
43       See https://dlmf.nist.gov/10.46.E2 
44       """
45       if x != 0:
46           v = b - 1
47           wb = wright_bessel(1, v + 1, x**2 / 4.)
48           # Note: iv(v, x) has precision of less than 1e-12 for some cases
49           # e.g v=1-1e-6 and x=1e-06)
50           assert_allclose(np.power(x / 2., v) * wb,
51                           sc.iv(v, x),
52                           rtol=1e-11, atol=1e-11)

55   @pytest.mark.parametrize('a', [0, 1e-6, 0.1, 0.5, 1, 10])
56   @pytest.mark.parametrize('b', [1, 1 + 1e-3, 2, 5, 10])
57   @pytest.mark.parametrize('x', [0, 1e-6, 0.1, 0.5, 1, 5, 10, 100])
58   def test_wright_functional(a, b, x):
59       """Test functional relation of wright_bessel. 
60    
61       Phi(a, b-1, z) = a*z*Phi(a, b+a, z) + (b-1)*Phi(a, b, z) 
62    
63       Note that d/dx Phi(a, b, x) = Phi(a, b-1, x) 
64       See Eq. (22) of 
65       B. Stankovic, On the Function of E. M. Wright, 
66       Publ. de l' Institut Mathematique, Beograd, 
67       Nouvelle S`er. 10 (1970), 113-124. 
68       """
69       assert_allclose(wright_bessel(a, b - 1, x),
70                       a * x * wright_bessel(a, b + a, x)
71                       + (b - 1) * wright_bessel(a, b, x),
72                       rtol=1e-8, atol=1e-8)
