88   def scale(
89       sample: "npt.ArrayLike",
90       l_bounds: "npt.ArrayLike",
91       u_bounds: "npt.ArrayLike",
92       *,
93       reverse: bool = False
94   ) -> np.ndarray:
95       r"""Sample scaling from unit hypercube to different bounds. 
96    
97       To convert a sample from :math:`[0, 1)` to :math:`[a, b), b>a`, 
98       with :math:`a` the lower bounds and :math:`b` the upper bounds. 
99       The following transformation is used: 
100   
101      .. math:: 
102   
103          (b - a) \cdot \text{sample} + a 
104   
105      Parameters 
106      ---------- 
107      sample : array_like (n, d) 
108          Sample to scale. 
109      l_bounds, u_bounds : array_like (d,) 
110          Lower and upper bounds (resp. :math:`a`, :math:`b`) of transformed 
111          data. If `reverse` is True, range of the original data to transform 
112          to the unit hypercube. 
113      reverse : bool, optional 
114          Reverse the transformation from different bounds to the unit hypercube. 
115          Default is False. 
116   
117      Returns 
118      ------- 
119      sample : array_like (n, d) 
120          Scaled sample. 
121   
122      Examples 
123      -------- 
124      Transform 3 samples in the unit hypercube to bounds: 
125   
126      >>> from scipy.stats import qmc 
127      >>> l_bounds = [-2, 0] 
128      >>> u_bounds = [6, 5] 
129      >>> sample = [[0.5 , 0.75], 
130      ...           [0.5 , 0.5], 
131      ...           [0.75, 0.25]] 
132      >>> sample_scaled = qmc.scale(sample, l_bounds, u_bounds) 
133      >>> sample_scaled 
134      array([[2.  , 3.75], 
135             [2.  , 2.5 ], 
136             [4.  , 1.25]]) 
137   
138      And convert back to the unit hypercube: 
139   
140      >>> sample_ = qmc.scale(sample_scaled, l_bounds, u_bounds, reverse=True) 
141      >>> sample_ 
142      array([[0.5 , 0.75], 
143             [0.5 , 0.5 ], 
144             [0.75, 0.25]]) 
145   
146      """
147      sample = np.asarray(sample)
148  
149      # Checking bounds and sample
150      if not sample.ndim == 2:
151          raise ValueError('Sample is not a 2D array')
152  
153      lower, upper = _validate_bounds(
154          l_bounds=l_bounds, u_bounds=u_bounds, d=sample.shape[1]
155      )
156  
157      if not reverse:
158          # Checking that sample is within the hypercube
159          if (sample.max() > 1.) or (sample.min() < 0.):
160              raise ValueError('Sample is not in unit hypercube')
161  
162          return sample * (upper - lower) + lower
163      else:
164          # Checking that sample is within the bounds
165          if not (np.all(sample >= lower) and np.all(sample <= upper)):
166              raise ValueError('Sample is out of bounds')
167  
168          return (sample - lower) / (upper - lower)

171  def _ensure_in_unit_hypercube(sample: "npt.ArrayLike") -> np.ndarray:
172      """Ensure that sample is a 2D array and is within a unit hypercube 
173   
174      Parameters 
175      ---------- 
176      sample : array_like (n, d) 
177          A 2D array of points. 
178   
179      Returns 
180      ------- 
181      np.ndarray 
182          The array interpretation of the input sample 
183   
184      Raises 
185      ------ 
186      ValueError 
187          If the input is not a 2D array or contains points outside of 
188          a unit hypercube. 
189      """
190      sample = np.asarray(sample, dtype=np.float64, order="C")
191  
192      if not sample.ndim == 2:
193          raise ValueError("Sample is not a 2D array")
194  
195      if (sample.max() > 1.) or (sample.min() < 0.):
196          raise ValueError("Sample is not in unit hypercube")
197  
198      return sample

201  def discrepancy(
202          sample: "npt.ArrayLike",
203          *,
204          iterative: bool = False,
205          method: Literal["CD", "WD", "MD", "L2-star"] = "CD",
206          workers: IntNumber = 1) -> float:
207      """Discrepancy of a given sample. 
208   
209      Parameters 
210      ---------- 
211      sample : array_like (n, d) 
212          The sample to compute the discrepancy from. 
213      iterative : bool, optional 
214          Must be False if not using it for updating the discrepancy. 
215          Default is False. Refer to the notes for more details. 
216      method : str, optional 
217          Type of discrepancy, can be ``CD``, ``WD``, ``MD`` or ``L2-star``. 
218          Refer to the notes for more details. Default is ``CD``. 
219      workers : int, optional 
220          Number of workers to use for parallel processing. If -1 is given all 
221          CPU threads are used. Default is 1. 
222   
223      Returns 
224      ------- 
225      discrepancy : float 
226          Discrepancy. 
227   
228      See Also 
229      -------- 
230      geometric_discrepancy 
231   
232      Notes 
233      ----- 
234      The discrepancy is a uniformity criterion used to assess the space filling 
235      of a number of samples in a hypercube. A discrepancy quantifies the 
236      distance between the continuous uniform distribution on a hypercube and the 
237      discrete uniform distribution on :math:`n` distinct sample points. 
238   
239      The lower the value is, the better the coverage of the parameter space is. 
240   
241      For a collection of subsets of the hypercube, the discrepancy is the 
242      difference between the fraction of sample points in one of those 
243      subsets and the volume of that subset. There are different definitions of 
244      discrepancy corresponding to different collections of subsets. Some 
245      versions take a root mean square difference over subsets instead of 
246      a maximum. 
247   
248      A measure of uniformity is reasonable if it satisfies the following 
249      criteria [1]_: 
250   
251      1. It is invariant under permuting factors and/or runs. 
252      2. It is invariant under rotation of the coordinates. 
253      3. It can measure not only uniformity of the sample over the hypercube, 
254         but also the projection uniformity of the sample over non-empty 
255         subset of lower dimension hypercubes. 
256      4. There is some reasonable geometric meaning. 
257      5. It is easy to compute. 
258      6. It satisfies the Koksma-Hlawka-like inequality. 
259      7. It is consistent with other criteria in experimental design. 
260   
261      Four methods are available: 
262   
263      * ``CD``: Centered Discrepancy - subspace involves a corner of the 
264        hypercube 
265      * ``WD``: Wrap-around Discrepancy - subspace can wrap around bounds 
266      * ``MD``: Mixture Discrepancy - mix between CD/WD covering more criteria 
267      * ``L2-star``: L2-star discrepancy - like CD BUT variant to rotation 
268   
269      See [2]_ for precise definitions of each method. 
270   
271      Lastly, using ``iterative=True``, it is possible to compute the 
272      discrepancy as if we had :math:`n+1` samples. This is useful if we want 
273      to add a point to a sampling and check the candidate which would give the 
274      lowest discrepancy. Then you could just update the discrepancy with 
275      each candidate using `update_discrepancy`. This method is faster than 
276      computing the discrepancy for a large number of candidates. 
277   
278      References 
279      ---------- 
280      .. [1] Fang et al. "Design and modeling for computer experiments". 
281         Computer Science and Data Analysis Series, 2006. 
282      .. [2] Zhou Y.-D. et al. "Mixture discrepancy for quasi-random point sets." 
283         Journal of Complexity, 29 (3-4) , pp. 283-301, 2013. 
284      .. [3] T. T. Warnock. "Computational investigations of low discrepancy 
285         point sets." Applications of Number Theory to Numerical 
286         Analysis, Academic Press, pp. 319-343, 1972. 
287   
288      Examples 
289      -------- 
290      Calculate the quality of the sample using the discrepancy: 
291   
292      >>> import numpy as np 
293      >>> from scipy.stats import qmc 
294      >>> space = np.array([[1, 3], [2, 6], [3, 2], [4, 5], [5, 1], [6, 4]]) 
295      >>> l_bounds = [0.5, 0.5] 
296      >>> u_bounds = [6.5, 6.5] 
297      >>> space = qmc.scale(space, l_bounds, u_bounds, reverse=True) 
298      >>> space 
299      array([[0.08333333, 0.41666667], 
300             [0.25      , 0.91666667], 
301             [0.41666667, 0.25      ], 
302             [0.58333333, 0.75      ], 
303             [0.75      , 0.08333333], 
304             [0.91666667, 0.58333333]]) 
305      >>> qmc.discrepancy(space) 
306      0.008142039609053464 
307   
308      We can also compute iteratively the ``CD`` discrepancy by using 
309      ``iterative=True``. 
310   
311      >>> disc_init = qmc.discrepancy(space[:-1], iterative=True) 
312      >>> disc_init 
313      0.04769081147119336 
314      >>> qmc.update_discrepancy(space[-1], space[:-1], disc_init) 
315      0.008142039609053513 
316   
317      """
318      sample = _ensure_in_unit_hypercube(sample)
319  
320      workers = _validate_workers(workers)
321  
322      methods = {
323          "CD": _cy_wrapper_centered_discrepancy,
324          "WD": _cy_wrapper_wrap_around_discrepancy,
325          "MD": _cy_wrapper_mixture_discrepancy,
326          "L2-star": _cy_wrapper_l2_star_discrepancy,
327      }
328  
329      if method in methods:
330          return methods[method](sample, iterative, workers=workers)
331      else:
332          raise ValueError(f"{method!r} is not a valid method. It must be one of"
333                           f" {set(methods)!r}")

336  def geometric_discrepancy(
337          sample: "npt.ArrayLike",
338          method: Literal["mindist", "mst"] = "mindist",
339          metric: str = "euclidean") -> float:
340      """Discrepancy of a given sample based on its geometric properties. 
341   
342      Parameters 
343      ---------- 
344      sample : array_like (n, d) 
345          The sample to compute the discrepancy from. 
346      method : {"mindist", "mst"}, optional 
347          The method to use. One of ``mindist`` for minimum distance (default) 
348          or ``mst`` for minimum spanning tree. 
349      metric : str or callable, optional 
350          The distance metric to use. See the documentation 
351          for `scipy.spatial.distance.pdist` for the available metrics and 
352          the default. 
353   
354      Returns 
355      ------- 
356      discrepancy : float 
357          Discrepancy (higher values correspond to greater sample uniformity). 
358   
359      See Also 
360      -------- 
361      discrepancy 
362   
363      Notes 
364      ----- 
365      The discrepancy can serve as a simple measure of quality of a random sample. 
366      This measure is based on the geometric properties of the distribution of points 
367      in the sample, such as the minimum distance between any pair of points, or 
368      the mean edge length in a minimum spanning tree. 
369   
370      The higher the value is, the better the coverage of the parameter space is. 
371      Note that this is different from `scipy.stats.qmc.discrepancy`, where lower 
372      values correspond to higher quality of the sample. 
373   
374      Also note that when comparing different sampling strategies using this function, 
375      the sample size must be kept constant. 
376   
377      It is possible to calculate two metrics from the minimum spanning tree: 
378      the mean edge length and the standard deviation of edges lengths. Using 
379      both metrics offers a better picture of uniformity than either metric alone, 
380      with higher mean and lower standard deviation being preferable (see [1]_ 
381      for a brief discussion). This function currently only calculates the mean 
382      edge length. 
383   
384      References 
385      ---------- 
386      .. [1] Franco J. et al. "Minimum Spanning Tree: A new approach to assess the quality 
387         of the design of computer experiments." Chemometrics and Intelligent Laboratory 
388         Systems, 97 (2), pp. 164-169, 2009. 
389   
390      Examples 
391      -------- 
392      Calculate the quality of the sample using the minimum euclidean distance 
393      (the defaults): 
394   
395      >>> import numpy as np 
396      >>> from scipy.stats import qmc 
397      >>> rng = np.random.default_rng(191468432622931918890291693003068437394) 
398      >>> sample = qmc.LatinHypercube(d=2, rng=rng).random(50) 
399      >>> qmc.geometric_discrepancy(sample) 
400      0.03708161435687876 
401   
402      Calculate the quality using the mean edge length in the minimum 
403      spanning tree: 
404   
405      >>> qmc.geometric_discrepancy(sample, method='mst') 
406      0.1105149978798376 
407   
408      Display the minimum spanning tree and the points with 
409      the smallest distance: 
410   
411      >>> import matplotlib.pyplot as plt 
412      >>> from matplotlib.lines import Line2D 
413      >>> from scipy.sparse.csgraph import minimum_spanning_tree 
414      >>> from scipy.spatial.distance import pdist, squareform 
415      >>> dist = pdist(sample) 
416      >>> mst = minimum_spanning_tree(squareform(dist)) 
417      >>> edges = np.where(mst.toarray() > 0) 
418      >>> edges = np.asarray(edges).T 
419      >>> min_dist = np.min(dist) 
420      >>> min_idx = np.argwhere(squareform(dist) == min_dist)[0] 
421      >>> fig, ax = plt.subplots(figsize=(10, 5)) 
422      >>> _ = ax.set(aspect='equal', xlabel=r'$x_1$', ylabel=r'$x_2$', 
423      ...            xlim=[0, 1], ylim=[0, 1]) 
424      >>> for edge in edges: 
425      ...     ax.plot(sample[edge, 0], sample[edge, 1], c='k') 
426      >>> ax.scatter(sample[:, 0], sample[:, 1]) 
427      >>> ax.add_patch(plt.Circle(sample[min_idx[0]], min_dist, color='red', fill=False)) 
428      >>> markers = [ 
429      ...     Line2D([0], [0], marker='o', lw=0, label='Sample points'), 
430      ...     Line2D([0], [0], color='k', label='Minimum spanning tree'), 
431      ...     Line2D([0], [0], marker='o', lw=0, markerfacecolor='w', markeredgecolor='r', 
432      ...            label='Minimum point-to-point distance'), 
433      ... ] 
434      >>> ax.legend(handles=markers, loc='center left', bbox_to_anchor=(1, 0.5)); 
435      >>> plt.show() 
436   
437      """
438      sample = _ensure_in_unit_hypercube(sample)
439      if sample.shape[0] < 2:
440          raise ValueError("Sample must contain at least two points")
441  
442      distances = distance.pdist(sample, metric=metric)  # type: ignore[call-overload]
443  
444      if np.any(distances == 0.0):
445          warnings.warn("Sample contains duplicate points.", stacklevel=2)
446  
447      if method == "mindist":
448          return np.min(distances[distances.nonzero()])
449      elif method == "mst":
450          fully_connected_graph = distance.squareform(distances)
451          mst = minimum_spanning_tree(fully_connected_graph)
452          distances = mst[mst.nonzero()]
453          # TODO consider returning both the mean and the standard deviation
454          # see [1] for a discussion
455          return np.mean(distances)
456      else:
457          raise ValueError(f"{method!r} is not a valid method. "
458                           f"It must be one of {{'mindist', 'mst'}}")

461  def update_discrepancy(
462          x_new: "npt.ArrayLike",
463          sample: "npt.ArrayLike",
464          initial_disc: DecimalNumber) -> float:
465      """Update the centered discrepancy with a new sample. 
466   
467      Parameters 
468      ---------- 
469      x_new : array_like (1, d) 
470          The new sample to add in `sample`. 
471      sample : array_like (n, d) 
472          The initial sample. 
473      initial_disc : float 
474          Centered discrepancy of the `sample`. 
475   
476      Returns 
477      ------- 
478      discrepancy : float 
479          Centered discrepancy of the sample composed of `x_new` and `sample`. 
480   
481      Examples 
482      -------- 
483      We can also compute iteratively the discrepancy by using 
484      ``iterative=True``. 
485   
486      >>> import numpy as np 
487      >>> from scipy.stats import qmc 
488      >>> space = np.array([[1, 3], [2, 6], [3, 2], [4, 5], [5, 1], [6, 4]]) 
489      >>> l_bounds = [0.5, 0.5] 
490      >>> u_bounds = [6.5, 6.5] 
491      >>> space = qmc.scale(space, l_bounds, u_bounds, reverse=True) 
492      >>> disc_init = qmc.discrepancy(space[:-1], iterative=True) 
493      >>> disc_init 
494      0.04769081147119336 
495      >>> qmc.update_discrepancy(space[-1], space[:-1], disc_init) 
496      0.008142039609053513 
497   
498      """
499      sample = np.asarray(sample, dtype=np.float64, order="C")
500      x_new = np.asarray(x_new, dtype=np.float64, order="C")
501  
502      # Checking that sample is within the hypercube and 2D
503      if not sample.ndim == 2:
504          raise ValueError('Sample is not a 2D array')
505  
506      if (sample.max() > 1.) or (sample.min() < 0.):
507          raise ValueError('Sample is not in unit hypercube')
508  
509      # Checking that x_new is within the hypercube and 1D
510      if not x_new.ndim == 1:
511          raise ValueError('x_new is not a 1D array')
512  
513      if not (np.all(x_new >= 0) and np.all(x_new <= 1)):
514          raise ValueError('x_new is not in unit hypercube')
515  
516      if x_new.shape[0] != sample.shape[1]:
517          raise ValueError("x_new and sample must be broadcastable")
518  
519      return _cy_wrapper_update_discrepancy(x_new, sample, initial_disc)

522  def _perturb_discrepancy(sample: np.ndarray, i1: int, i2: int, k: int,
523                           disc: float):
524      """Centered discrepancy after an elementary perturbation of a LHS. 
525   
526      An elementary perturbation consists of an exchange of coordinates between 
527      two points: ``sample[i1, k] <-> sample[i2, k]``. By construction, 
528      this operation conserves the LHS properties. 
529   
530      Parameters 
531      ---------- 
532      sample : array_like (n, d) 
533          The sample (before permutation) to compute the discrepancy from. 
534      i1 : int 
535          The first line of the elementary permutation. 
536      i2 : int 
537          The second line of the elementary permutation. 
538      k : int 
539          The column of the elementary permutation. 
540      disc : float 
541          Centered discrepancy of the design before permutation. 
542   
543      Returns 
544      ------- 
545      discrepancy : float 
546          Centered discrepancy of the design after permutation. 
547   
548      References 
549      ---------- 
550      .. [1] Jin et al. "An efficient algorithm for constructing optimal design 
551         of computer experiments", Journal of Statistical Planning and 
552         Inference, 2005. 
553   
554      """
555      n = sample.shape[0]
556  
557      z_ij = sample - 0.5
558  
559      # Eq (19)
560      c_i1j = (1. / n ** 2.
561               * np.prod(0.5 * (2. + abs(z_ij[i1, :])
562                                + abs(z_ij) - abs(z_ij[i1, :] - z_ij)), axis=1))
563      c_i2j = (1. / n ** 2.
564               * np.prod(0.5 * (2. + abs(z_ij[i2, :])
565                                + abs(z_ij) - abs(z_ij[i2, :] - z_ij)), axis=1))
566  
567      # Eq (20)
568      c_i1i1 = (1. / n ** 2 * np.prod(1 + abs(z_ij[i1, :]))
569                - 2. / n * np.prod(1. + 0.5 * abs(z_ij[i1, :])
570                                   - 0.5 * z_ij[i1, :] ** 2))
571      c_i2i2 = (1. / n ** 2 * np.prod(1 + abs(z_ij[i2, :]))
572                - 2. / n * np.prod(1. + 0.5 * abs(z_ij[i2, :])
573                                   - 0.5 * z_ij[i2, :] ** 2))
574  
575      # Eq (22), typo in the article in the denominator i2 -> i1
576      num = (2 + abs(z_ij[i2, k]) + abs(z_ij[:, k])
577             - abs(z_ij[i2, k] - z_ij[:, k]))
578      denum = (2 + abs(z_ij[i1, k]) + abs(z_ij[:, k])
579               - abs(z_ij[i1, k] - z_ij[:, k]))
580      gamma = num / denum
581  
582      # Eq (23)
583      c_p_i1j = gamma * c_i1j
584      # Eq (24)
585      c_p_i2j = c_i2j / gamma
586  
587      alpha = (1 + abs(z_ij[i2, k])) / (1 + abs(z_ij[i1, k]))
588      beta = (2 - abs(z_ij[i2, k])) / (2 - abs(z_ij[i1, k]))
589  
590      g_i1 = np.prod(1. + abs(z_ij[i1, :]))
591      g_i2 = np.prod(1. + abs(z_ij[i2, :]))
592      h_i1 = np.prod(1. + 0.5 * abs(z_ij[i1, :]) - 0.5 * (z_ij[i1, :] ** 2))
593      h_i2 = np.prod(1. + 0.5 * abs(z_ij[i2, :]) - 0.5 * (z_ij[i2, :] ** 2))
594  
595      # Eq (25), typo in the article g is missing
596      c_p_i1i1 = ((g_i1 * alpha) / (n ** 2) - 2. * alpha * beta * h_i1 / n)
597      # Eq (26), typo in the article n ** 2
598      c_p_i2i2 = ((g_i2 / ((n ** 2) * alpha)) - (2. * h_i2 / (n * alpha * beta)))
599  
600      # Eq (26)
601      sum_ = c_p_i1j - c_i1j + c_p_i2j - c_i2j
602  
603      mask = np.ones(n, dtype=bool)
604      mask[[i1, i2]] = False
605      sum_ = sum(sum_[mask])
606  
607      disc_ep = (disc + c_p_i1i1 - c_i1i1 + c_p_i2i2 - c_i2i2 + 2 * sum_)
608  
609      return disc_ep

613      """Prime numbers from 2 to *n*. 
614   
615      Parameters 
616      ----------

618          Sup bound with ``n >= 6``. 
619   
620      Returns 
621      ------- 
622      primes : list(int) 
623          Primes in ``2 <= p < n``.

683  def _van_der_corput_permutations(
684      base: IntNumber, *, rng: SeedType = None
685  ) -> np.ndarray:
686      """Permutations for scrambling a Van der Corput sequence. 
687   
688      Parameters 
689      ---------- 
690      base : int 
691          Base of the sequence. 
692      rng : `numpy.random.Generator`, optional 
693          Pseudorandom number generator state. When `rng` is None, a new 
694          `numpy.random.Generator` is created using entropy from the 
695          operating system. Types other than `numpy.random.Generator` are 
696          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
697   
698          .. versionchanged:: 1.15.0 
699   
700              As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
701              transition from use of `numpy.random.RandomState` to 
702              `numpy.random.Generator`, this keyword was changed from `seed` to 
703              `rng`. During the transition, the behavior documented above is not 
704              accurate; see `check_random_state` for actual behavior. After the 
705              transition, this admonition can be removed. 
706   
707      Returns 
708      ------- 
709      permutations : array_like 
710          Permutation indices. 
711   
712      Notes 
713      ----- 
714      In Algorithm 1 of Owen 2017, a permutation of `np.arange(base)` is 
715      created for each positive integer `k` such that ``1 - base**-k < 1`` 
716      using floating-point arithmetic. For double precision floats, the 
717      condition ``1 - base**-k < 1`` can also be written as ``base**-k > 
718      2**-54``, which makes it more apparent how many permutations we need 
719      to create. 
720      """
721      rng = check_random_state(rng)
722      count = math.ceil(54 / math.log2(base)) - 1
723      permutations = np.repeat(np.arange(base)[None], count, axis=0)
724      for perm in permutations:
725          rng.shuffle(perm)
726  
727      return permutations

730  def van_der_corput(
731          n: IntNumber,
732          base: IntNumber = 2,
733          *,
734          start_index: IntNumber = 0,
735          scramble: bool = False,
736          permutations: "npt.ArrayLike | None" = None,
737          rng: SeedType = None,
738          workers: IntNumber = 1) -> np.ndarray:
739      """Van der Corput sequence. 
740   
741      Pseudo-random number generator based on a b-adic expansion. 
742   
743      Scrambling uses permutations of the remainders (see [1]_). Multiple 
744      permutations are applied to construct a point. The sequence of 
745      permutations has to be the same for all points of the sequence. 
746   
747      Parameters 
748      ---------- 
749      n : int 
750          Number of element of the sequence. 
751      base : int, optional 
752          Base of the sequence. Default is 2. 
753      start_index : int, optional 
754          Index to start the sequence from. Default is 0. 
755      scramble : bool, optional 
756          If True, use Owen scrambling. Otherwise no scrambling is done. 
757          Default is True. 
758      permutations : array_like, optional 
759          Permutations used for scrambling. 
760      rng : `numpy.random.Generator`, optional 
761          Pseudorandom number generator state. When `rng` is None, a new 
762          `numpy.random.Generator` is created using entropy from the 
763          operating system. Types other than `numpy.random.Generator` are 
764          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
765      workers : int, optional 
766          Number of workers to use for parallel processing. If -1 is 
767          given all CPU threads are used. Default is 1. 
768   
769      Returns 
770      ------- 
771      sequence : list (n,) 
772          Sequence of Van der Corput. 
773   
774      References 
775      ---------- 
776      .. [1] A. B. Owen. "A randomized Halton algorithm in R", 
777         :arxiv:`1706.02808`, 2017. 
778   
779      """
780      if base < 2:
781          raise ValueError("'base' must be at least 2")
782  
783      if scramble:
784          if permutations is None:
785              permutations = _van_der_corput_permutations(
786                  base=base, rng=rng
787              )
788          else:
789              permutations = np.asarray(permutations)
790  
791          permutations = permutations.astype(np.int64)
792          return _cy_van_der_corput_scrambled(n, base, start_index,
793                                              permutations, workers)
794  
795      else:
796          return _cy_van_der_corput(n, base, start_index, workers)

799  class QMCEngine(ABC):
800      """A generic Quasi-Monte Carlo sampler class meant for subclassing. 
801   
802      QMCEngine is a base class to construct a specific Quasi-Monte Carlo 
803      sampler. It cannot be used directly as a sampler. 
804   
805      Parameters 
806      ---------- 
807      d : int 
808          Dimension of the parameter space. 
809      optimization : {None, "random-cd", "lloyd"}, optional 
810          Whether to use an optimization scheme to improve the quality after 
811          sampling. Note that this is a post-processing step that does not 
812          guarantee that all properties of the sample will be conserved. 
813          Default is None. 
814   
815          * ``random-cd``: random permutations of coordinates to lower the 
816            centered discrepancy. The best sample based on the centered 
817            discrepancy is constantly updated. Centered discrepancy-based 
818            sampling shows better space-filling robustness toward 2D and 3D 
819            subprojections compared to using other discrepancy measures. 
820          * ``lloyd``: Perturb samples using a modified Lloyd-Max algorithm. 
821            The process converges to equally spaced samples. 
822   
823          .. versionadded:: 1.10.0 
824   
825      rng : `numpy.random.Generator`, optional 
826          Pseudorandom number generator state. When `rng` is None, a new 
827          `numpy.random.Generator` is created using entropy from the 
828          operating system. Types other than `numpy.random.Generator` are 
829          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
830   
831          .. versionchanged:: 1.15.0 
832   
833              As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
834              transition from use of `numpy.random.RandomState` to 
835              `numpy.random.Generator`, this keyword was changed from `seed` to 
836              `rng`. For an interim period, both keywords will continue to work, although 
837              only one may be specified at a time. After the interim period, function 
838              calls using the `seed` keyword will emit warnings. Following a 
839              deprecation period, the `seed` keyword will be removed. 
840   
841      Notes 
842      ----- 
843      By convention samples are distributed over the half-open interval 
844      ``[0, 1)``. Instances of the class can access the attributes: ``d`` for 
845      the dimension; and ``rng`` for the random number generator. 
846   
847      **Subclassing** 
848   
849      When subclassing `QMCEngine` to create a new sampler,  ``__init__`` and 
850      ``random`` must be redefined. 
851   
852      * ``__init__(d, rng=None)``: at least fix the dimension. If the sampler 
853        does not take advantage of a ``rng`` (deterministic methods like 
854        Halton), this parameter can be omitted. 
855      * ``_random(n, *, workers=1)``: draw ``n`` from the engine. ``workers`` 
856        is used for parallelism. See `Halton` for example. 
857   
858      Optionally, two other methods can be overwritten by subclasses: 
859   
860      * ``reset``: Reset the engine to its original state. 
861      * ``fast_forward``: If the sequence is deterministic (like Halton 
862        sequence), then ``fast_forward(n)`` is skipping the ``n`` first draw. 
863   
864      Examples 
865      -------- 
866      To create a random sampler based on ``np.random.random``, we would do the 
867      following: 
868   
869      >>> from scipy.stats import qmc 
870      >>> class RandomEngine(qmc.QMCEngine): 
871      ...     def __init__(self, d, rng=None): 
872      ...         super().__init__(d=d, rng=rng) 
873      ... 
874      ... 
875      ...     def _random(self, n=1, *, workers=1): 
876      ...         return self.rng.random((n, self.d)) 
877      ... 
878      ... 
879      ...     def reset(self): 
880      ...         super().__init__(d=self.d, rng=self.rng_seed) 
881      ...         return self 
882      ... 
883      ... 
884      ...     def fast_forward(self, n): 
885      ...         self.random(n) 
886      ...         return self 
887   
888      After subclassing `QMCEngine` to define the sampling strategy we want to 
889      use, we can create an instance to sample from. 
890   
891      >>> engine = RandomEngine(2) 
892      >>> engine.random(5) 
893      array([[0.22733602, 0.31675834],  # random 
894             [0.79736546, 0.67625467], 
895             [0.39110955, 0.33281393], 
896             [0.59830875, 0.18673419], 
897             [0.67275604, 0.94180287]]) 
898   
899      We can also reset the state of the generator and resample again. 
900   
901      >>> _ = engine.reset() 
902      >>> engine.random(5) 
903      array([[0.22733602, 0.31675834],  # random 
904             [0.79736546, 0.67625467], 
905             [0.39110955, 0.33281393], 
906             [0.59830875, 0.18673419], 
907             [0.67275604, 0.94180287]]) 
908   
909      """
910  
911      @abstractmethod
912      @_transition_to_rng('seed', replace_doc=False)
913      def __init__(
914          self,
915          d: IntNumber,
916          *,
917          optimization: Literal["random-cd", "lloyd"] | None = None,
918          rng: SeedType = None
919      ) -> None:
920          self._initialize(d, optimization=optimization, rng=rng)
921  
922      # During SPEC 7 transition:
923      # `__init__` has to be wrapped with @_transition_to_rng decorator
924      # because it is public. Subclasses previously called `__init__`
925      # directly, but this was problematic because arguments passed to
926      # subclass `__init__` as `seed` would get passed to superclass
927      # `__init__` as `rng`, rejecting `RandomState` arguments.
928      def _initialize(
929          self,
930          d: IntNumber,
931          *,
932          optimization: Literal["random-cd", "lloyd"] | None = None,
933          rng: SeedType = None
934      ) -> None:
935          if not np.issubdtype(type(d), np.integer) or d < 0:
936              raise ValueError('d must be a non-negative integer value')
937  
938          self.d = d
939  
940          if isinstance(rng, np.random.Generator):
941              # Spawn a Generator that we can own and reset.
942              self.rng = _rng_spawn(rng, 1)[0]
943          else:
944              # Create our instance of Generator, does not need spawning
945              # Also catch RandomState which cannot be spawned
946              self.rng = check_random_state(rng)
947          self.rng_seed = copy.deepcopy(self.rng)
948  
949          self.num_generated = 0
950  
951          config = {
952              # random-cd
953              "n_nochange": 100,
954              "n_iters": 10_000,
955              "rng": self.rng,
956  
957              # lloyd
958              "tol": 1e-5,
959              "maxiter": 10,
960              "qhull_options": None,
961          }
962          self._optimization = optimization
963          self.optimization_method = _select_optimizer(optimization, config)

966      def _random(
967          self, n: IntNumber = 1, *, workers: IntNumber = 1
968      ) -> np.ndarray:
969          ...

971      def random(
972          self, n: IntNumber = 1, *, workers: IntNumber = 1
973      ) -> np.ndarray:
974          """Draw `n` in the half-open interval ``[0, 1)``. 
975   
976          Parameters 
977          ---------- 
978          n : int, optional 
979              Number of samples to generate in the parameter space. 
980              Default is 1. 
981          workers : int, optional 
982              Only supported with `Halton`. 
983              Number of workers to use for parallel processing. If -1 is 
984              given all CPU threads are used. Default is 1. It becomes faster 
985              than one worker for `n` greater than :math:`10^3`. 
986   
987          Returns 
988          ------- 
989          sample : array_like (n, d) 
990              QMC sample. 
991   
992          """
993          sample = self._random(n, workers=workers)
994          if self.optimization_method is not None:
995              sample = self.optimization_method(sample)
996  
997          self.num_generated += n
998          return sample

1000     def integers(
1001         self,
1002         l_bounds: "npt.ArrayLike",
1003         *,
1004         u_bounds: "npt.ArrayLike | None" = None,
1005         n: IntNumber = 1,
1006         endpoint: bool = False,
1007         workers: IntNumber = 1
1008     ) -> np.ndarray:
1009         r""" 
1010         Draw `n` integers from `l_bounds` (inclusive) to `u_bounds` 
1011         (exclusive), or if endpoint=True, `l_bounds` (inclusive) to 
1012         `u_bounds` (inclusive). 
1013  
1014         Parameters 
1015         ---------- 
1016         l_bounds : int or array-like of ints 
1017             Lowest (signed) integers to be drawn (unless ``u_bounds=None``, 
1018             in which case this parameter is 0 and this value is used for 
1019             `u_bounds`). 
1020         u_bounds : int or array-like of ints, optional 
1021             If provided, one above the largest (signed) integer to be drawn 
1022             (see above for behavior if ``u_bounds=None``). 
1023             If array-like, must contain integer values. 
1024         n : int, optional 
1025             Number of samples to generate in the parameter space. 
1026             Default is 1. 
1027         endpoint : bool, optional 
1028             If true, sample from the interval ``[l_bounds, u_bounds]`` instead 
1029             of the default ``[l_bounds, u_bounds)``. Defaults is False. 
1030         workers : int, optional 
1031             Number of workers to use for parallel processing. If -1 is 
1032             given all CPU threads are used. Only supported when using `Halton` 
1033             Default is 1. 
1034  
1035         Returns 
1036         ------- 
1037         sample : array_like (n, d) 
1038             QMC sample. 
1039  
1040         Notes 
1041         ----- 
1042         It is safe to just use the same ``[0, 1)`` to integer mapping 
1043         with QMC that you would use with MC. You still get unbiasedness, 
1044         a strong law of large numbers, an asymptotically infinite variance 
1045         reduction and a finite sample variance bound. 
1046  
1047         To convert a sample from :math:`[0, 1)` to :math:`[a, b), b>a`, 
1048         with :math:`a` the lower bounds and :math:`b` the upper bounds, 
1049         the following transformation is used: 
1050  
1051         .. math:: 
1052  
1053             \text{floor}((b - a) \cdot \text{sample} + a) 
1054  
1055         """
1056         if u_bounds is None:
1057             u_bounds = l_bounds
1058             l_bounds = 0
1059 
1060         u_bounds = np.atleast_1d(u_bounds)
1061         l_bounds = np.atleast_1d(l_bounds)
1062 
1063         if endpoint:
1064             u_bounds = u_bounds + 1
1065 
1066         if (not np.issubdtype(l_bounds.dtype, np.integer) or
1067                 not np.issubdtype(u_bounds.dtype, np.integer)):
1068             message = ("'u_bounds' and 'l_bounds' must be integers or"
1069                        " array-like of integers")
1070             raise ValueError(message)
1071 
1072         if isinstance(self, Halton):
1073             sample = self.random(n=n, workers=workers)
1074         else:
1075             sample = self.random(n=n)
1076 
1077         sample = scale(sample, l_bounds=l_bounds, u_bounds=u_bounds)
1078         sample = np.floor(sample).astype(np.int64)
1079 
1080         return sample

1236     def _initialize_permutations(self) -> None:
1237         """Initialize permutations for all Van der Corput sequences. 
1238

1240         """

1281 class LatinHypercube(QMCEngine):
1282     r"""Latin hypercube sampling (LHS). 
1283  
1284     A Latin hypercube sample [1]_ generates :math:`n` points in 
1285     :math:`[0,1)^{d}`. Each univariate marginal distribution is stratified, 
1286     placing exactly one point in :math:`[j/n, (j+1)/n)` for 
1287     :math:`j=0,1,...,n-1`. They are still applicable when :math:`n << d`. 
1288  
1289     Parameters 
1290     ---------- 
1291     d : int 
1292         Dimension of the parameter space. 
1293     scramble : bool, optional 
1294         When False, center samples within cells of a multi-dimensional grid. 
1295         Otherwise, samples are randomly placed within cells of the grid. 
1296  
1297         .. note:: 
1298             Setting ``scramble=False`` does not ensure deterministic output. 
1299             For that, use the `rng` parameter. 
1300  
1301         Default is True. 
1302  
1303         .. versionadded:: 1.10.0 
1304  
1305     optimization : {None, "random-cd", "lloyd"}, optional 
1306         Whether to use an optimization scheme to improve the quality after 
1307         sampling. Note that this is a post-processing step that does not 
1308         guarantee that all properties of the sample will be conserved. 
1309         Default is None. 
1310  
1311         * ``random-cd``: random permutations of coordinates to lower the 
1312           centered discrepancy. The best sample based on the centered 
1313           discrepancy is constantly updated. Centered discrepancy-based 
1314           sampling shows better space-filling robustness toward 2D and 3D 
1315           subprojections compared to using other discrepancy measures. 
1316         * ``lloyd``: Perturb samples using a modified Lloyd-Max algorithm. 
1317           The process converges to equally spaced samples. 
1318  
1319         .. versionadded:: 1.8.0 
1320         .. versionchanged:: 1.10.0 
1321             Add ``lloyd``. 
1322  
1323     strength : {1, 2}, optional 
1324         Strength of the LHS. ``strength=1`` produces a plain LHS while 
1325         ``strength=2`` produces an orthogonal array based LHS of strength 2 
1326         [7]_, [8]_. In that case, only ``n=p**2`` points can be sampled, 
1327         with ``p`` a prime number. It also constrains ``d <= p + 1``. 
1328         Default is 1. 
1329  
1330         .. versionadded:: 1.8.0 
1331  
1332     rng : `numpy.random.Generator`, optional 
1333         Pseudorandom number generator state. When `rng` is None, a new 
1334         `numpy.random.Generator` is created using entropy from the 
1335         operating system. Types other than `numpy.random.Generator` are 
1336         passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
1337  
1338         .. versionchanged:: 1.15.0 
1339  
1340             As part of the `SPEC-007 <https://scientific-python.org/specs/spec-0007/>`_ 
1341             transition from use of `numpy.random.RandomState` to 
1342             `numpy.random.Generator`, this keyword was changed from `seed` to 
1343             `rng`. For an interim period, both keywords will continue to work, although 
1344             only one may be specified at a time. After the interim period, function 
1345             calls using the `seed` keyword will emit warnings. Following a

1352     Notes 
1353     ----- 
1354  
1355     When LHS is used for integrating a function :math:`f` over :math:`n`, 
1356     LHS is extremely effective on integrands that are nearly additive [2]_. 
1357     With a LHS of :math:`n` points, the variance of the integral is always 
1358     lower than plain MC on :math:`n-1` points [3]_. There is a central limit 
1359     theorem for LHS on the mean and variance of the integral [4]_, but not 
1360     necessarily for optimized LHS due to the randomization. 
1361  
1362     :math:`A` is called an orthogonal array of strength :math:`t` if in each 
1363     n-row-by-t-column submatrix of :math:`A`: all :math:`p^t` possible 
1364     distinct rows occur the same number of times. The elements of :math:`A` 
1365     are in the set :math:`\{0, 1, ..., p-1\}`, also called symbols. 
1366     The constraint that :math:`p` must be a prime number is to allow modular 
1367     arithmetic. Increasing strength adds some symmetry to the sub-projections 
1368     of a sample. With strength 2, samples are symmetric along the diagonals of 
1369     2D sub-projections. This may be undesirable, but on the other hand, the 
1370     sample dispersion is improved. 
1371  
1372     Strength 1 (plain LHS) brings an advantage over strength 0 (MC) and 
1373     strength 2 is a useful increment over strength 1. Going to strength 3 is 
1374     a smaller increment and scrambled QMC like Sobol', Halton are more 
1375     performant [7]_. 
1376  
1377     To create a LHS of strength 2, the orthogonal array :math:`A` is 
1378     randomized by applying a random, bijective map of the set of symbols onto 
1379     itself. For example, in column 0, all 0s might become 2; in column 1, 
1380     all 0s might become 1, etc. 
1381     Then, for each column :math:`i` and symbol :math:`j`, we add a plain, 
1382     one-dimensional LHS of size :math:`p` to the subarray where 
1383     :math:`A^i = j`. The resulting matrix is finally divided by :math:`p`.

1541     def _random_lhs(self, n: IntNumber = 1) -> np.ndarray:
1542         """Base LHS algorithm."""
1543         if not self.scramble:
1544             samples: np.ndarray | float = 0.5
1545         else:
1546             samples = self.rng.uniform(size=(n, self.d))
1547 
1548         perms = np.tile(np.arange(1, n + 1),
1549                         (self.d, 1))  # type: ignore[arg-type]
1550         for i in range(self.d):
1551             self.rng.shuffle(perms[i, :])
1552         perms = perms.T
1553 
1554         samples = (perms - samples) / n
1555         return samples

1557     def _random_oa_lhs(self, n: IntNumber = 4) -> np.ndarray:
1558         """Orthogonal array based LHS of strength 2."""
1559         p = np.sqrt(n).astype(int)
1560         n_row = p**2
1561         n_col = p + 1
1562 
1563         primes = primes_from_2_to(p + 1)
1564         if p not in primes or n != n_row:
1565             raise ValueError(
1566                 "n is not the square of a prime number. Close"
1567                 f" values are {primes[-2:]**2}"
1568             )
1569         if self.d > p + 1:
1570             raise ValueError("n is too small for d. Must be n > (d-1)**2")
1571 
1572         oa_sample = np.zeros(shape=(n_row, n_col), dtype=int)
1573 
1574         # OA of strength 2
1575         arrays = np.tile(np.arange(p), (2, 1))
1576         oa_sample[:, :2] = np.stack(np.meshgrid(*arrays),
1577                                     axis=-1).reshape(-1, 2)
1578         for p_ in range(1, p):
1579             oa_sample[:, 2+p_-1] = np.mod(oa_sample[:, 0]
1580                                           + p_*oa_sample[:, 1], p)
1581 
1582         # scramble the OA
1583         oa_sample_ = np.empty(shape=(n_row, n_col), dtype=int)
1584         for j in range(n_col):
1585             perms = self.rng.permutation(p)
1586             oa_sample_[:, j] = perms[oa_sample[:, j]]
1587         
1588         oa_sample = oa_sample_
1589         # following is making a scrambled OA into an OA-LHS
1590         oa_lhs_sample = np.zeros(shape=(n_row, n_col))
1591         lhs_engine = LatinHypercube(d=1, scramble=self.scramble, strength=1,
1592                                     rng=self.rng)  # type: QMCEngine
1593         for j in range(n_col):
1594             for k in range(p):
1595                 idx = oa_sample[:, j] == k
1596                 lhs = lhs_engine.random(p).flatten()
1597                 oa_lhs_sample[:, j][idx] = lhs + oa_sample[:, j][idx]
1598 
1599         oa_lhs_sample /= p
1600 
1601         return oa_lhs_sample[:, :self.d]

1605     """Engine for generating (scrambled) Sobol' sequences. 
1606  
1607     Sobol' sequences are low-discrepancy, quasi-random numbers. Points 
1608     can be drawn using two methods: 
1609  
1610     * `random_base2`: safely draw :math:`n=2^m` points. This method 
1611       guarantees the balance properties of the sequence. 
1612     * `random`: draw an arbitrary number of points from the 
1613       sequence. See warning below. 
1614  
1615     Parameters 
1616     ---------- 
1617     d : int 
1618         Dimensionality of the sequence. Max dimensionality is 21201. 
1619     scramble : bool, optional 
1620         If True, use LMS+shift scrambling. Otherwise, no scrambling is done. 
1621         Default is True.

1624         that can be generated, which is ``2**bits``. Maximal value is 64. 
1625         It does not correspond to the return type, which is always 
1626         ``np.float64`` to prevent points from repeating themselves. 
1627         Default is None, which for backward compatibility, corresponds to 30.

1664     Sobol' sequences [1]_ provide :math:`n=2^m` low discrepancy points in 
1665     :math:`[0,1)^{d}`. Scrambling them [3]_ makes them suitable for singular 
1666     integrands, provides a means of error estimation, and can improve their

1668     (left) linear matrix scramble (LMS) followed by a digital random shift

1685        replicates does not have to be a power of 2.

1688        After :math:`2^B` points have been generated, the sequence would 
1689        repeat. Hence, an error is raised. 
1690        The number of bits can be controlled with the parameter `bits`.

2448     def _standard_normal_samples(self, n: IntNumber = 1) -> np.ndarray:
2449         """Draw `n` QMC samples from the standard Normal :math:`N(0, I_d)`. 
2450  
2451         Parameters 
2452         ---------- 
2453         n : int, optional 
2454             Number of samples to generate in the parameter space. Default is 1. 
2455  
2456         Returns 
2457         ------- 
2458         sample : array_like (n, d) 
2459             Sample. 
2460  
2461         """
2462         # get base samples
2463         samples = self.engine.random(n)
2464         if self._inv_transform:
2465             # apply inverse transform
2466             # (values to close to 0/1 result in inf values)
2467             return stats.norm.ppf(0.5 + (1 - 1e-10) * (samples - 0.5))  # type: ignore[attr-defined]  # noqa: E501
2468         else:
2469             # apply Box-Muller transform (note: indexes starting from 1)
2470             even = np.arange(0, samples.shape[-1], 2)
2471             Rs = np.sqrt(-2 * np.log(samples[:, even]))
2472             thetas = 2 * math.pi * samples[:, 1 + even]
2473             cos = np.cos(thetas)
2474             sin = np.sin(thetas)
2475             transf_samples = np.stack([Rs * cos, Rs * sin],
2476                                       -1).reshape(n, -1)
2477             # make sure we only return the number of dimension requested
2478             return transf_samples[:, : self._d]

2619 def _random_cd(
2620     best_sample: np.ndarray, n_iters: int, n_nochange: int, rng: GeneratorType,
2621     **kwargs: dict
2622 ) -> np.ndarray:
2623     """Optimal LHS on CD. 
2624  
2625     Create a base LHS and do random permutations of coordinates to 
2626     lower the centered discrepancy. 
2627     Because it starts with a normal LHS, it also works with the 
2628     `scramble` keyword argument. 
2629  
2630     Two stopping criterion are used to stop the algorithm: at most, 
2631     `n_iters` iterations are performed; or if there is no improvement 
2632     for `n_nochange` consecutive iterations. 
2633     """
2634     del kwargs  # only use keywords which are defined, needed by factory
2635 
2636     n, d = best_sample.shape
2637 
2638     if d == 0 or n == 0:
2639         return np.empty((n, d))
2640 
2641     if d == 1 or n == 1:
2642         # discrepancy measures are invariant under permuting factors and runs
2643         return best_sample
2644 
2645     best_disc = discrepancy(best_sample)
2646 
2647     bounds = ([0, d - 1],
2648               [0, n - 1],
2649               [0, n - 1])
2650 
2651     n_nochange_ = 0
2652     n_iters_ = 0
2653     while n_nochange_ < n_nochange and n_iters_ < n_iters:
2654         n_iters_ += 1
2655 
2656         col = rng_integers(rng, *bounds[0], endpoint=True)  # type: ignore[misc]
2657         row_1 = rng_integers(rng, *bounds[1], endpoint=True)  # type: ignore[misc]
2658         row_2 = rng_integers(rng, *bounds[2], endpoint=True)  # type: ignore[misc]
2659         disc = _perturb_discrepancy(best_sample,
2660                                     row_1, row_2, col,
2661                                     best_disc)
2662         if disc < best_disc:
2663             best_sample[row_1, col], best_sample[row_2, col] = (
2664                 best_sample[row_2, col], best_sample[row_1, col])
2665 
2666             best_disc = disc
2667             n_nochange_ = 0
2668         else:
2669             n_nochange_ += 1
2670 
2671     return best_sample

2678 def _lloyd_iteration(
2679     sample: np.ndarray,
2680     decay: float,
2681     qhull_options: str
2682 ) -> np.ndarray:
2683     """Lloyd-Max algorithm iteration. 
2684  
2685     Based on the implementation of Stfan van der Walt: 
2686  
2687     https://github.com/stefanv/lloyd 
2688  
2689     which is: 
2690  
2691         Copyright (c) 2021-04-21 Stfan van der Walt 
2692         https://github.com/stefanv/lloyd 
2693         MIT License 
2694  
2695     Parameters 
2696     ---------- 
2697     sample : array_like (n, d) 
2698         The sample to iterate on. 
2699     decay : float 
2700         Relaxation decay. A positive value would move the samples toward 
2701         their centroid, and negative value would move them away. 
2702         1 would move the samples to their centroid. 
2703     qhull_options : str 
2704         Additional options to pass to Qhull. See Qhull manual 
2705         for details. (Default: "Qbb Qc Qz Qj Qx" for ndim > 4 and 
2706         "Qbb Qc Qz Qj" otherwise.) 
2707  
2708     Returns 
2709     ------- 
2710     sample : array_like (n, d) 
2711         The sample after an iteration of Lloyd's algorithm. 
2712  
2713     """
2714     new_sample = np.empty_like(sample)
2715 
2716     voronoi = Voronoi(sample, qhull_options=qhull_options)
2717 
2718     for ii, idx in enumerate(voronoi.point_region):
2719         # the region is a series of indices into self.voronoi.vertices
2720         # remove samples at infinity, designated by index -1
2721         region = [i for i in voronoi.regions[idx] if i != -1]
2722 
2723         # get the vertices for this region
2724         verts = voronoi.vertices[region]
2725 
2726         # clipping would be wrong, we need to intersect
2727         # verts = np.clip(verts, 0, 1)
2728 
2729         # move samples towards centroids:
2730         # Centroid in n-D is the mean for uniformly distributed nodes
2731         # of a geometry.
2732         centroid = np.mean(verts, axis=0)
2733         new_sample[ii] = sample[ii] + (centroid - sample[ii]) * decay
2734 
2735     # only update sample to centroid within the region
2736     is_valid = np.all(np.logical_and(new_sample >= 0, new_sample <= 1), axis=1)
2737     sample[is_valid] = new_sample[is_valid]
2738 
2739     return sample

2742 def _lloyd_centroidal_voronoi_tessellation(
2743     sample: "npt.ArrayLike",
2744     *,
2745     tol: DecimalNumber = 1e-5,
2746     maxiter: IntNumber = 10,
2747     qhull_options: str | None = None,
2748     **kwargs: dict
2749 ) -> np.ndarray:
2750     """Approximate Centroidal Voronoi Tessellation. 
2751  
2752     Perturb samples in N-dimensions using Lloyd-Max algorithm. 
2753  
2754     Parameters 
2755     ---------- 
2756     sample : array_like (n, d) 
2757         The sample to iterate on. With ``n`` the number of samples and ``d`` 
2758         the dimension. Samples must be in :math:`[0, 1]^d`, with ``d>=2``. 
2759     tol : float, optional 
2760         Tolerance for termination. If the min of the L1-norm over the samples 
2761         changes less than `tol`, it stops the algorithm. Default is 1e-5. 
2762     maxiter : int, optional 
2763         Maximum number of iterations. It will stop the algorithm even if 
2764         `tol` is above the threshold. 
2765         Too many iterations tend to cluster the samples as a hypersphere. 
2766         Default is 10. 
2767     qhull_options : str, optional 
2768         Additional options to pass to Qhull. See Qhull manual 
2769         for details. (Default: "Qbb Qc Qz Qj Qx" for ndim > 4 and 
2770         "Qbb Qc Qz Qj" otherwise.) 
2771  
2772     Returns 
2773     ------- 
2774     sample : array_like (n, d) 
2775         The sample after being processed by Lloyd-Max algorithm. 
2776  
2777     Notes 
2778     ----- 
2779     Lloyd-Max algorithm is an iterative process with the purpose of improving 
2780     the dispersion of samples. For given sample: (i) compute a Voronoi 
2781     Tessellation; (ii) find the centroid of each Voronoi cell; (iii) move the 
2782     samples toward the centroid of their respective cell. See [1]_, [2]_. 
2783  
2784     A relaxation factor is used to control how fast samples can move at each 
2785     iteration. This factor is starting at 2 and ending at 1 after `maxiter` 
2786     following an exponential decay. 
2787  
2788     The process converges to equally spaced samples. It implies that measures 
2789     like the discrepancy could suffer from too many iterations. On the other 
2790     hand, L1 and L2 distances should improve. This is especially true with 
2791     QMC methods which tend to favor the discrepancy over other criteria. 
2792  
2793     .. note:: 
2794  
2795         The current implementation does not intersect the Voronoi Tessellation 
2796         with the boundaries. This implies that for a low number of samples, 
2797         empirically below 20, no Voronoi cell is touching the boundaries. 
2798         Hence, samples cannot be moved close to the boundaries. 
2799  
2800         Further improvements could consider the samples at infinity so that 
2801         all boundaries are segments of some Voronoi cells. This would fix 
2802         the computation of the centroid position. 
2803  
2804     .. warning:: 
2805  
2806        The Voronoi Tessellation step is expensive and quickly becomes 
2807        intractable with dimensions as low as 10 even for a sample 
2808        of size as low as 1000. 
2809  
2810     .. versionadded:: 1.9.0 
2811  
2812     References 
2813     ---------- 
2814     .. [1] Lloyd. "Least Squares Quantization in PCM". 
2815        IEEE Transactions on Information Theory, 1982. 
2816     .. [2] Max J. "Quantizing for minimum distortion". 
2817        IEEE Transactions on Information Theory, 1960. 
2818  
2819     Examples 
2820     -------- 
2821     >>> import numpy as np 
2822     >>> from scipy.spatial import distance 
2823     >>> from scipy.stats._qmc import _lloyd_centroidal_voronoi_tessellation 
2824     >>> rng = np.random.default_rng() 
2825     >>> sample = rng.random((128, 2)) 
2826  
2827     .. note:: 
2828  
2829         The samples need to be in :math:`[0, 1]^d`. `scipy.stats.qmc.scale` 
2830         can be used to scale the samples from their 
2831         original bounds to :math:`[0, 1]^d`. And back to their original bounds. 
2832  
2833     Compute the quality of the sample using the L1 criterion. 
2834  
2835     >>> def l1_norm(sample): 
2836     ...    return distance.pdist(sample, 'cityblock').min() 
2837  
2838     >>> l1_norm(sample) 
2839     0.00161...  # random 
2840  
2841     Now process the sample using Lloyd's algorithm and check the improvement 
2842     on the L1. The value should increase. 
2843  
2844     >>> sample = _lloyd_centroidal_voronoi_tessellation(sample) 
2845     >>> l1_norm(sample) 
2846     0.0278...  # random 
2847  
2848     """
2849     del kwargs  # only use keywords which are defined, needed by factory
2850 
2851     sample = np.asarray(sample).copy()
2852 
2853     if not sample.ndim == 2:
2854         raise ValueError('`sample` is not a 2D array')
2855 
2856     if not sample.shape[1] >= 2:
2857         raise ValueError('`sample` dimension is not >= 2')
2858 
2859     # Checking that sample is within the hypercube
2860     if (sample.max() > 1.) or (sample.min() < 0.):
2861         raise ValueError('`sample` is not in unit hypercube')
2862 
2863     if qhull_options is None:
2864         qhull_options = 'Qbb Qc Qz QJ'
2865 
2866         if sample.shape[1] >= 5:
2867             qhull_options += ' Qx'
2868 
2869     # Fit an exponential to be 2 at 0 and 1 at `maxiter`.
2870     # The decay is used for relaxation.
2871     # analytical solution for y=exp(-maxiter/x) - 0.1
2872     root = -maxiter / np.log(0.1)
2873     decay = [np.exp(-x / root)+0.9 for x in range(maxiter)]
2874 
2875     l1_old = _l1_norm(sample=sample)
2876     for i in range(maxiter):
2877         sample = _lloyd_iteration(
2878                 sample=sample, decay=decay[i],
2879                 qhull_options=qhull_options,
2880         )
2881 
2882         l1_new = _l1_norm(sample=sample)
2883 
2884         if abs(l1_new - l1_old) < tol:
2885             break
2886         else:
2887             l1_old = l1_new
2888 
2889     return sample
