12   class Var:
13       """Logic Variable"""
14   
15       _id = 1
16   
17       def __new__(cls, *token):
18           if len(token) == 0:
19               token = f"_{Var._id}"  # type: ignore[assignment]
20               Var._id += 1
21           elif len(token) == 1:
22               token = token[0]
23   
24           obj = object.__new__(cls)
25           obj.token = token  # type: ignore[attr-defined]
26           return obj
27   
28       def __str__(self):
29           return "~" + str(self.token)  # type: ignore[attr-defined]

31       __repr__ = __str__

34           return type(self) == type(other) and self.token == other.token  # type: ignore[attr-defined]

61   @contextmanager
62   def variables(*variables):
63       """ 
64       Context manager for logic variables 
65    
66       Example: 
67           >>> # xdoctest: +SKIP("undefined vars") 
68           >>> from __future__ import with_statement 
69           >>> with variables(1): 
70           ...     print(isvar(1)) 
71           True 
72           >>> print(isvar(1)) 
73           False 
74           >>> # Normal approach 
75           >>> from unification import unify 
76           >>> x = var("x") 
77           >>> unify(x, 1) 
78           {~x: 1} 
79           >>> # Context Manager approach 
80           >>> with variables("x"): 
81           ...     print(unify("x", 1)) 
82           {'x': 1} 
83       """
84       old_global_logic_variables = _global_logic_variables.copy()
85       _global_logic_variables.update(set(variables))
86       try:
87           yield
88       finally:
89           _global_logic_variables.clear()
90           _global_logic_variables.update(old_global_logic_variables)
