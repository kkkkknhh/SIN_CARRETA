22   class BosonOp(Operator):
23       """A bosonic operator that satisfies [a, Dagger(a)] == 1. 
24    
25       Parameters 
26       ========== 
27    
28       name : str 
29           A string that labels the bosonic mode. 
30    
31       annihilation : bool 
32           A bool that indicates if the bosonic operator is an annihilation (True, 
33           default value) or creation operator (False) 
34    
35       Examples 
36       ======== 
37    
38       >>> from sympy.physics.quantum import Dagger, Commutator 
39       >>> from sympy.physics.quantum.boson import BosonOp 
40       >>> a = BosonOp("a") 
41       >>> Commutator(a, Dagger(a)).doit() 
42       1 
43       """
44   
45       @property
46       def name(self):
47           return self.args[0]
48   
49       @property
50       def is_annihilation(self):
51           return bool(self.args[1])
52   
53       @classmethod
54       def default_args(self):
55           return ("a", True)
56   
57       def __new__(cls, *args, **hints):
58           if not len(args) in [1, 2]:
59               raise ValueError('1 or 2 parameters expected, got %s' % args)
60   
61           if len(args) == 1:
62               args = (args[0], S.One)
63   
64           if len(args) == 2:
65               args = (args[0], Integer(args[1]))
66   
67           return Operator.__new__(cls, *args)
68   
69       def _eval_commutator_BosonOp(self, other, **hints):
70           if self.name == other.name:
71               # [a^\dagger, a] = -1
72               if not self.is_annihilation and other.is_annihilation:
73                   return S.NegativeOne
74   
75           elif 'independent' in hints and hints['independent']:
76               # [a, b] = 0
77               return S.Zero
78   
79           return None
80   
81       def _eval_commutator_FermionOp(self, other, **hints):
82           return S.Zero
83   
84       def _eval_anticommutator_BosonOp(self, other, **hints):
85           if 'independent' in hints and hints['independent']:
86               # {a, b} = 2 * a * b, because [a, b] = 0
87               return 2 * self * other
88   
89           return None
90   
91       def _eval_adjoint(self):
92           return BosonOp(str(self.name), not self.is_annihilation)
93   
94       def _print_contents_latex(self, printer, *args):
95           if self.is_annihilation:
96               return r'{%s}' % str(self.name)
97           else:
98               return r'{{%s}^\dagger}' % str(self.name)
99   
100      def _print_contents(self, printer, *args):
101          if self.is_annihilation:
102              return r'%s' % str(self.name)
103          else:
104              return r'Dagger(%s)' % str(self.name)
105  
106      def _print_contents_pretty(self, printer, *args):
107          from sympy.printing.pretty.stringpict import prettyForm
108          pform = printer._print(self.args[0], *args)
109          if self.is_annihilation:
110              return pform
111          else:
112              return pform**prettyForm('\N{DAGGER}')

115  class BosonFockKet(Ket):
116      """Fock state ket for a bosonic mode. 
117   
118      Parameters 
119      ========== 
120   
121      n : Number 
122          The Fock state number. 
123   
124      """
125  
126      def __new__(cls, n):
127          return Ket.__new__(cls, n)
128  
129      @property
130      def n(self):
131          return self.label[0]
132  
133      @classmethod
134      def dual_class(self):
135          return BosonFockBra
136  
137      @classmethod
138      def _eval_hilbert_space(cls, label):
139          return FockSpace()
140  
141      def _eval_innerproduct_BosonFockBra(self, bra, **hints):
142          return KroneckerDelta(self.n, bra.n)
143  
144      def _apply_from_right_to_BosonOp(self, op, **options):
145          if op.is_annihilation:
146              return sqrt(self.n) * BosonFockKet(self.n - 1)
147          else:
148              return sqrt(self.n + 1) * BosonFockKet(self.n + 1)

178  class BosonCoherentKet(Ket):
179      """Coherent state ket for a bosonic mode. 
180   
181      Parameters 
182      ========== 
183   
184      alpha : Number, Symbol 
185          The complex amplitude of the coherent state. 
186   
187      """
188  
189      def __new__(cls, alpha):
190          return Ket.__new__(cls, alpha)
191  
192      @property
193      def alpha(self):
194          return self.label[0]
195  
196      @classmethod
197      def dual_class(self):
198          return BosonCoherentBra
199  
200      @classmethod
201      def _eval_hilbert_space(cls, label):
202          return HilbertSpace()
203  
204      def _eval_innerproduct_BosonCoherentBra(self, bra, **hints):
205          if self.alpha == bra.alpha:
206              return S.One
207          else:
208              return exp(-(abs(self.alpha)**2 + abs(bra.alpha)**2 - 2 * conjugate(bra.alpha) * self.alpha)/2)
209  
210      def _apply_from_right_to_BosonOp(self, op, **options):
211          if op.is_annihilation:
212              return self.alpha * self
213          else:
214              return None

217  class BosonCoherentBra(Bra):
218      """Coherent state bra for a bosonic mode. 
219   
220      Parameters 
221      ========== 
222   
223      alpha : Number, Symbol 
224          The complex amplitude of the coherent state. 
225   
226      """
227  
228      def __new__(cls, alpha):
229          return Bra.__new__(cls, alpha)
230  
231      @property
232      def alpha(self):
233          return self.label[0]
234  
235      @classmethod
236      def dual_class(self):
237          return BosonCoherentKet
238  
239      def _apply_operator_BosonOp(self, op, **options):
240          if not op.is_annihilation:
241              return self.alpha * self
242          else:
243              return None
