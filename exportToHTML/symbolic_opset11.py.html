116  @_onnx_symbolic("aten::clamp")
117  def clamp(g: jit_utils.GraphContext, self, min, max):
118      def _cast_if_not_none(tensor, dtype):
119          if tensor is not None and not symbolic_helper._is_none(tensor):
120              return g.op(
121                  "Cast",
122                  tensor,
123                  to_i=dtype.onnx_type(),
124              )
125          else:
126              return tensor
127  
128      scalar_type = _type_utils.JitScalarType.from_value(
129          self, _type_utils.JitScalarType.UNDEFINED
130      )
131      if scalar_type != _type_utils.JitScalarType.UNDEFINED:
132          min = _cast_if_not_none(min, scalar_type)
133          max = _cast_if_not_none(max, scalar_type)
134  
135      if symbolic_helper._is_none(min):
136          return clamp_max(g, self, max)
137      elif symbolic_helper._is_none(max):
138          return clamp_min(g, self, min)
139      else:
140          if (
141              symbolic_helper._get_tensor_rank(min) == 0
142              and symbolic_helper._get_tensor_rank(max) == 0
143          ):
144              return symbolic_helper._op_with_optional_float_cast(
145                  g, "Clip", self, min, max, opset_before=12
146              )
147          else:
148              return clamp_max(g, clamp_min(g, self, min), max)

205  @_onnx_symbolic("aten::index_put")
206  def index_put(
207      g: jit_utils.GraphContext, self, indices_list_value, values, accumulate=False
208  ):
209      if symbolic_helper._is_packed_list(indices_list_value):
210          indices_list = symbolic_helper._unpack_list(indices_list_value)
211      else:
212          indices_list = [indices_list_value]
213      accumulate = symbolic_helper._parse_arg(accumulate, "b")
214  
215      if len(indices_list) == 0:
216          return values
217  
218      if len(indices_list) > 1:
219          for idx_ in range(len(indices_list)):
220              if symbolic_helper._is_bool(indices_list[idx_]):
221                  indices_list[idx_] = g.op("NonZero", indices_list[idx_])
222          index = indices_list[0]
223  
224          for ind in indices_list[1:]:
225              index = opset9.add(g, index, ind)
226          broadcast_index_shape = g.op("Shape", index)
227          indices_list = [
228              symbolic_helper._unsqueeze_helper(
229                  g, opset9.expand(g, ind, broadcast_index_shape, None), [-1]
230              )
231              for ind in indices_list
232          ]
233          index = g.op("Concat", *indices_list, axis_i=-1)
234      else:
235          # Replace index_put node with masked_scatter or masked_fill
236          # when inputs to the index_put node contains a single boolean input.
237          #
238          # index_put -> masked_fill
239          #   * input index contains single tensor of Bool type (e.g.: %24 <- %23).
240          #   * input value contains single element (e.g.: %18).
241          #
242          # Torch IR
243          #   %mask : Float(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu) = aten::clone(%0, %6)
244          #   %16 : Bool(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu) =
245          #               aten::to(%8, %26, %27, %11, %12, %28, %29, %15)
246          #   %18 : Float(requires_grad=0, device=cpu) = prim::Constant[value={1}]()
247          #   %23 : Bool(8, strides=[1], device=cpu) = aten::view(%16, %22)
248          #   %24 : Tensor?[] = prim::ListConstruct(%23)
249          #   %25 : Float(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu) =
250          #                aten::index_put(%mask, %24, %18, %30)
251          #   return (%25)
252          #
253          #
254          # index_put -> masked_scatter
255          #   * input index contains single tensor of Bool type (e.g.: %32 <- %31).
256          #   * input value contains multiple elements (e.g.: %28).
257          #
258          # Torch IR
259          #   %mask : Float(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu) = aten::clone(%0, %6)
260          #   %28 : Float(8, strides=[1], requires_grad=0, device=cpu)
261          #                = prim::Constant[value= 1  1  1  1  1  1  1  1 [ CPUFloatType{8} ]]()
262          #   %15 : Bool(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu)
263          #                = aten::ne(%mask, %some_const)
264          #   %23 : Bool(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu)
265          #                = aten::to(%15, %34, %35, %18, %19, %36, %37, %22)
266          #   %38 : Long(requires_grad=0, device=cpu) = prim::Constant[value={0}]()
267          #   %30 : int[] = prim::Constant[value=[-1]]()
268          #   %31 : Bool(8, strides=[1], device=cpu) = aten::view(%23, %30)
269          #   %32 : Tensor?[] = prim::ListConstruct(%31)
270          #   %33 : Float(2, 2, 2, strides=[4, 2, 1], requires_grad=0, device=cpu)
271          #               = aten::index_put(%mask, %32, %28, %38)
272          #   return (%33)
273          index = indices_list[0]
274          bool_inp = index
275          if symbolic_helper._is_bool(bool_inp):
276              rank = symbolic_helper._get_tensor_rank(values)
277              if rank is not None and rank == 0:
278                  return opset9.masked_fill(g, self, bool_inp, values)
279              mask_rank = symbolic_helper._get_tensor_rank(bool_inp)
280              self_rank = symbolic_helper._get_tensor_rank(self)
281              if (
282                  mask_rank is not None
283                  and self_rank is not None
284                  and self_rank > mask_rank
285              ):
286                  # Unsqueeze 'bool_inp' to be broadcastable to shape of 'self'.
287                  bool_inp = symbolic_helper._unsqueeze_helper(
288                      g, bool_inp, list(range(mask_rank, self_rank))
289                  )
290              return masked_scatter(g, self, bool_inp, values)
291          broadcast_index_shape = g.op("Shape", index)
292          index = symbolic_helper._unsqueeze_helper(g, index, [-1])
293      sub_data_shape = symbolic_helper._slice_helper(
294          g, g.op("Shape", self), axes=[0], starts=[len(indices_list)], ends=[sys.maxsize]
295      )
296      values_shape = g.op("Concat", broadcast_index_shape, sub_data_shape, axis_i=0)
297      # Check if values is a singular value and expand accordingly
298      rank = symbolic_helper._get_tensor_rank(values)
299      if rank is not None and rank == 0:
300          values = opset9.expand(g, values, values_shape, None)
301      values = symbolic_helper._reshape_helper(g, values, values_shape)
302  
303      self_scalar_type = _type_utils.JitScalarType.from_value(
304          self, _type_utils.JitScalarType.UNDEFINED
305      )
306      if self_scalar_type != _type_utils.JitScalarType.UNDEFINED:
307          values_scalar_type = _type_utils.JitScalarType.from_value(
308              values, _type_utils.JitScalarType.UNDEFINED
309          )
310          if self_scalar_type != values_scalar_type:
311              values = g.op("Cast", values, to_i=self_scalar_type.onnx_type())
312      elif accumulate:
313          raise errors.SymbolicValueError("self does not have a valid scalar type.", self)
314  
315      if accumulate:
316          zeros = g.op(
317              "ConstantOfShape",
318              g.op("Shape", self),
319              value_t=torch.tensor([0], dtype=self_scalar_type.dtype()),
320          )
321          result = g.op("ScatterND", zeros, index, values)
322          result = add(g, self, result)
323      else:
324          result = g.op("ScatterND", self, index, values)
325  
326      return result

395  @_onnx_symbolic("aten::scatter")
396  @symbolic_helper.parse_args("v", "i", "v", "v")
397  def scatter(g: jit_utils.GraphContext, self, dim, index, src):
398      src_type = _type_utils.JitScalarType.from_value(src)
399      src = symbolic_helper._maybe_get_scalar(src)
400      if symbolic_helper._is_value(src):
401          return g.op("ScatterElements", self, index, src, axis_i=dim)
402      else:
403          # Check if scalar "src" has same type as self (PyTorch allows different
404          # type for scalar src (but not when src is tensor)). If not, insert Cast node.
405          if _type_utils.JitScalarType.from_value(self) != src_type:
406              src = g.op(
407                  "Cast",
408                  src,
409                  to_i=_type_utils.JitScalarType.from_value(self).onnx_type(),
410              )
411          return g.op(
412              "ScatterElements", self, index, opset9.expand_as(g, src, index), axis_i=dim
413          )

438  def masked_scatter(g: jit_utils.GraphContext, self, mask, source):
439      index = opset9.nonzero(g, opset9.expand_as(g, mask, self))
440      # NOTE: source can have more elements than needed.
441      # It could also have arbitrary shape.
442      # This is not supported by ONNX::ScatterND, so we need to flatten and slice source tensor.
443      source = symbolic_helper._reshape_helper(g, source, torch.LongTensor([-1]))
444      source = symbolic_helper._slice_helper(
445          g,
446          source,
447          axes=torch.LongTensor([0]),
448          starts=torch.LongTensor([0]),
449          ends=opset9.size(g, index, torch.LongTensor([0])),
450      )
451      return g.op("ScatterND", self, index, source)

489      if symbolic_helper._is_value(self) and symbolic_helper._is_tensor_list(self):
490          tensor_list_node = other.node()

492              return symbolic_helper._unimplemented(
493                  "add", "does not support adding dynamic tensor list to another"
494              )

499          return l

603  @symbolic_helper.parse_args("v", "v", "i", "i")
604  def split(g: jit_utils.GraphContext, self, split_size_or_sizes, dim, _outputs=None):
605      if not symbolic_helper._is_split_static(split_size_or_sizes, _outputs):
606          split_out = g.op("SplitToSequence", self, split_size_or_sizes, axis_i=dim)
607          if _outputs is None:
608              return split_out
609          # Convert to multiple slice nodes iff number of splits and number of outputs are statically known.
610          if (
611              symbolic_helper._is_packed_list(split_size_or_sizes)
612              and len(symbolic_helper._unpack_list(split_size_or_sizes)) == _outputs
613          ):
614              split_sizes = [
615                  symbolic_helper._unsqueeze_helper(g, v, [0])
616                  for v in symbolic_helper._unpack_list(split_size_or_sizes)
617              ]
618              start = g.op("Constant", value_t=torch.tensor([0], dtype=torch.long))
619              axis = g.op("Constant", value_t=torch.tensor([dim], dtype=torch.long))
620              res = []
621              for i in range(_outputs):
622                  end = g.op(
623                      "Add", start, split_sizes[i]
624                  )  # split_sizes is a list of same length as _outputs
625                  res.append(g.op("Slice", self, start, end, axis))
626                  start = end
627              return res
628          return [
629              g.op(
630                  "SequenceAt",
631                  split_out,
632                  g.op("Constant", value_t=torch.tensor([i], dtype=torch.long)),
633              )
634              for i in range(_outputs)
635          ]
636      else:
637          return opset9.split(g, self, split_size_or_sizes, dim, _outputs)

661  def _prepare_onnx_paddings(g: jit_utils.GraphContext, input, pad):
662      """Generate paddings in ONNX order based on pad in pytorch. 
663   
664      Args: 
665          input: the input tensor. 
666          pad: the paddings in pytorch. 
667              The order is dim_n_begin, dim_n_end, dim_n-1_begin, dim_n-1_end, ..., dim_m_begin, dim_m_end, 
668              where m is in range [0, n]. 
669      """
670      if (
671          not symbolic_helper._is_packed_list(pad)
672          and symbolic_helper._is_list(pad)
673          and symbolic_helper._is_scalar_list(pad)
674      ):
675          pad = g.op("ConcatFromSequence", pad, axis_i=0, new_axis_i=1)
676      # The desired order of paddings is
677      # dim_0_begin, dim_1_begin, ... , dim_0_end, ..., dim_n_end.
678      # n is the dimension of input.
679      # Assume zero-dimensions in the beginning, pad the "pad" sequence with zeros in the beginning
680      pad_len = opset9.size(g, pad, g.op("Constant", value_t=torch.tensor([0])))
681      # Set extension = [0] * (dim * 2 - len(pad))
682      rank = symbolic_helper._get_tensor_rank(input)
683      if rank is None:
684          rank = g.op("Size", g.op("Shape", input))
685      else:
686          rank = g.op("Constant", value_t=torch.tensor(rank, dtype=torch.int64))
687      extension = g.op(
688          "Sub",
689          g.op("Mul", rank, g.op("Constant", value_t=torch.tensor(2, dtype=torch.int64))),
690          pad_len,
691      )
692      # Concat pad with extension: paddings = [dim_n_begin, dim_n_end, dim_n-1_begin, dim_n-1_end, 0, 0, ... ]
693      # Currently ONNX only supports int64 type for Pad
694      pad = g.op("Cast", pad, to_i=_C_onnx.TensorProtoDataType.INT64)
695      paddings = g.op(
696          "Concat",
697          pad,
698          g.op(
699              "ConstantOfShape", extension, value_t=torch.tensor([0], dtype=torch.int64)
700          ),
701          axis_i=0,
702      )
703      # Reshape and reverse order and collate first beginnings and then ends
704      # paddings = [[..., 0, dim_n-1_begin, dim_n_begin],
705      #               [..., 0, dim_n-1_end, dim_n_end]]
706      # Reshape back to 1-D paddings = [..., 0, dim_n - 1_begin, dim_n_begin, ..., 0, dim_n - 1_end, dim_n_end]
707      paddings = symbolic_helper._reshape_helper(
708          g, paddings, g.op("Constant", value_t=torch.tensor([-1, 2]))
709      )
710      paddings = g.op("Transpose", opset10.flip(g, paddings, [0]), perm_i=[1, 0])
711      paddings = symbolic_helper._reshape_helper(
712          g, paddings, g.op("Constant", value_t=torch.tensor([-1]))
713      )
714      padding_c = g.op("Cast", paddings, to_i=_C_onnx.TensorProtoDataType.INT64)
715      return padding_c

777  def arange(g: jit_utils.GraphContext, *args):
778      def _get_arange_dtype(dtype):
779          dtype = symbolic_helper._maybe_get_const(dtype, "i")
780          return dtype
781  
782      if len(args) == 2 and all(isinstance(val, int) for val in args):
783          # aten::arange(Scalar start, Scalar end)
784          dtype = torch.int64
785          # Start index.
786          start = g.op(
787              "Constant",
788              value_t=torch.tensor(args[0], dtype=dtype),
789          )
790          # End (exclusive) index.
791          end = g.op(
792              "Constant",
793              value_t=torch.tensor(args[1], dtype=dtype),
794          )
795          # Step size from start to end indexes.
796          delta_default = g.op(
797              "Constant",
798              value_t=torch.tensor(1, dtype=dtype),
799          )
800          return g.op("Range", start, end, delta_default)
801      elif len(args) == 2 or len(args) == 5:
802          if len(args) == 2:
803              # aten::arange(Scalar end, Tensor out)
804              dtype = None
805          else:
806              # aten::arange(Scalar end, ScalarType dtype, Layout, Device, bool pin_memory)
807              dtype = _get_arange_dtype(args[1])
808          type_, end, start, step = symbolic_helper._arange_cast_helper(
809              g, end=args[0], dtype=dtype
810          )
811          start_default = g.op(
812              "Constant",
813              value_t=torch.tensor(0, dtype=type_.dtype()),
814          )
815          delta_default = g.op(
816              "Constant",
817              value_t=torch.tensor(1, dtype=type_.dtype()),
818          )
819          return g.op("Range", start_default, end, delta_default)
820      elif len(args) == 4 or len(args) == 7:
821          if len(args) == 4:
822              # aten::arange(Scalar start, Scalar end, Scalar step, Tensor out)
823              dtype = None
824          else:
825              # aten::arange(Scalar start, Scalar end, Scalar step, ScalarType dtype, Layout, Device, bool pin_memory)
826              dtype = _get_arange_dtype(args[3])

828              g, start=args[0], end=args[1], step=args[2], dtype=dtype

832          # aten::arange(Scalar start, Scalar end, ScalarType dtype, Layout, Device, bool pin_memory)

835              g, start=args[0], end=args[1], dtype=dtype

838              "Constant",
839              value_t=torch.tensor(1, dtype=type_.dtype()),
840          )
841          return g.op("Range", start, end, delta_default)
842      else:
843          return symbolic_helper._unimplemented(
844              "aten::arange", f"with {len(args)} arguments"
845          )

866  @_onnx_symbolic("aten::squeeze")
867  def squeeze(g: jit_utils.GraphContext, self, dim=None):
868      if dim is None:
869          return g.op("Squeeze", self)
870  
871      # dim as a tensor
872      if not symbolic_helper._is_constant(dim):
873          return symbolic_helper._squeeze_helper(g, self, [dim])
874  
875      dim = symbolic_helper._get_const(dim, "i", "dim")
876  
877      input_rank = symbolic_helper._get_tensor_rank(self)
878      adjusted_dim = dim
879      if input_rank is not None and dim < 0:
880          adjusted_dim += input_rank
881      dim_size = symbolic_helper._get_tensor_dim_size(self, adjusted_dim)
882      if (dim < 0 and input_rank is None) or dim_size is None:
883          # If onnx shape inference is not on, export always as dynamic.
884          # Because we cannot tell if observed static shape is also static at runtime.
885          # create "cond" node (condition is shape[i]==1)
886          dim_constant = g.op("Constant", value_t=torch.tensor([dim]))
887          size = symbolic_helper._size_helper(g, self, dim_constant)
888          const_one = g.op("Constant", value_t=torch.ones(1, dtype=torch.int64))
889          cond = g.op("Equal", size, const_one)
890          # create the "If" node and add the "then" and "else" blocks to it.
891          if_op, (if_context, else_context), _ = jit_utils.add_op_with_blocks(
892              g, "If", cond, n_blocks=2
893          )
894          squeeze_ = symbolic_helper._squeeze_helper(if_context, self, [dim])
895          utils._add_output_to_block(if_context.block, squeeze_)
896          identity_ = else_context.op("Identity", self)
897          utils._add_output_to_block(else_context.block, identity_)
898          return if_op
899  
900      # For static input shape
901      dim = adjusted_dim
902      if dim_size > 1:
903          warnings.warn(
904              "This model contains a squeeze operation on dimension "
905              + str(dim)
906              + ". The size of "
907              + "this dimension in the given input is "
908              + str(dim_size)
909              + ". The model will "
910              + "be exported without the squeeze node. If the model is intended to be used with dynamic "
911              + "input shapes, please export with dynamic_axes argument."
912          )
913          return self
914      return symbolic_helper._squeeze_helper(g, self, [dim])

1037 def _get_im2col_indices_along_dim(
1038     g: jit_utils.GraphContext, input_d, kernel_size_d, dilation_d, padding_d, stride_d
1039 ):
1040     # Input is always 4-D (N, C, H, W)
1041     # Calculate indices of sliding blocks along spatial dimension
1042     # Slide kernel over input each dim d:
1043     # each dimension d ranges from 0 to input[d]+2xpadding[d]-dilation[d]x(kernel_size[d]-1)
1044     # with steps = stride
1045 
1046     blocks_d = g.op(
1047         "Add", input_d, g.op("Constant", value_t=torch.tensor(padding_d * 2))
1048     )
1049     blocks_d = g.op(
1050         "Sub",
1051         blocks_d,
1052         g.op("Constant", value_t=torch.tensor(dilation_d * (kernel_size_d - 1))),
1053     )
1054 
1055     # Stride kernel over input and find starting indices along dim d
1056     blocks_d_indices = g.op(
1057         "Range",
1058         g.op("Constant", value_t=torch.tensor(0)),
1059         blocks_d,
1060         g.op("Constant", value_t=torch.tensor(stride_d)),
1061     )
1062 
1063     # Apply dilation on kernel and find its indices along dim d
1064     kernel_grid = torch.arange(0, kernel_size_d * dilation_d, dilation_d)
1065     kernel_grid = g.op("Constant", value_t=kernel_grid.unsqueeze(0))
1066 
1067     # Broadcast and add kernel staring positions (indices) with
1068     # kernel_grid along dim d, to get block indices along dim d
1069     blocks_d_indices = symbolic_helper._unsqueeze_helper(
1070         g, blocks_d_indices, [0]
1071     )  # Reshape to [1, -1]
1072     kernel_mask = symbolic_helper._reshape_helper(
1073         g, kernel_grid, g.op("Constant", value_t=torch.tensor([-1, 1]))
1074     )
1075     block_mask = g.op("Add", blocks_d_indices, kernel_mask)
1076 
1077     return block_mask

1105 @symbolic_helper.parse_args("v", "is", "is", "is", "is")
1106 def im2col(g: jit_utils.GraphContext, input, kernel_size, dilation, padding, stride):
1107     # Input is always 4-D tensor (N, C, H, W)
1108     # All other args are int[2]
1109 
1110     input_h = size(g, input, g.op("Constant", value_t=torch.tensor(2)))
1111     input_w = size(g, input, g.op("Constant", value_t=torch.tensor(3)))
1112 
1113     stride_h, stride_w = stride[0], stride[1]
1114     padding_h, padding_w = padding[0], padding[1]
1115     dilation_h, dilation_w = dilation[0], dilation[1]
1116     kernel_h, kernel_w = kernel_size[0], kernel_size[1]
1117 
1118     blocks_row_indices = _get_im2col_indices_along_dim(
1119         g, input_h, kernel_h, dilation_h, padding_h, stride_h
1120     )
1121     blocks_col_indices = _get_im2col_indices_along_dim(
1122         g, input_w, kernel_w, dilation_w, padding_w, stride_w
1123     )
1124 
1125     output_shape = _get_im2col_output_shape(g, input, kernel_h, kernel_w)
1126     padded_input = _get_im2col_padded_input(g, input, padding_h, padding_w)
1127 
1128     # For a 4D matrix of size (1, 1, 3, 3) as below with kernel_size=2, stride=1, and dilation=1
1129     # [[[[1., 2., 3.,],
1130     #    [4., 5., 6.,],
1131     #    [7., 8., 9.,]]]]
1132     # First gather indices along rows (dim=2) with blocks_row_indices = [[0,1], [1,2]] to get:
1133     # [[[[[1., 2., 3.],
1134     #     [4., 5., 6.]],
1135     #    [[4., 5., 6.],
1136     #     [7., 8., 9.]]]]]
1137     # And then gather along cols (dim=4) with blocks_row_indices = [[0,1], [1,2]] to get:
1138     # [[[[[[1., 2.],
1139     #      [4., 5.]],
1140     #     [[2., 3.],
1141     #      [5., 6]]],
1142     #    [[[4., 5.],
1143     #      [7., 8.]],
1144     #     [[5., 6.],
1145     #      [8., 9.]]]]]]
1146     # Transpose dims 3 (depth) and 4 (rows), and then reshape to output shape (1, 1, 4, 4) to get:
1147     #  [[[1., 2., 4., 5.],
1148     #    [2., 3., 5., 6.],
1149     #    [4., 5., 7., 8.],
1150     #    [5., 6., 8., 9.]]]
1151     output = g.op("Gather", padded_input, blocks_row_indices, axis_i=2)
1152     output = g.op("Gather", output, blocks_col_indices, axis_i=4)
1153     output = g.op("Transpose", output, perm_i=[0, 1, 2, 4, 3, 5])
1154     return symbolic_helper._reshape_helper(g, output, output_shape)

1232 @symbolic_helper.parse_args("v", "v", "f", "f")
1233 def embedding_renorm(g: jit_utils.GraphContext, weight, indices, max_norm, norm_type):
1234     unique_indices = g.op("Unique", indices)
1235     partial_weight = g.op("Gather", weight, unique_indices)
1236     norm_i = int(norm_type)
1237     if norm_i == 1:
1238         norm_type = "ReduceL1"
1239     elif norm_i == 2:

1244             "Only 1. and 2. are supported.",
1245             weight,
1246         )
1247     partial_weight_norm = g.op(norm_type, partial_weight, axes_i=[1], keepdims_i=1)

1250     partial_weight_norm_ = g.op(
1251         "Add", partial_weight_norm, g.op("Constant", value_t=torch.tensor(1e-7))
1252     )
1253     max_norm = torch.tensor(max_norm)
1254     scales = g.op("Div", max_norm, partial_weight_norm_)
1255     partial_weight_renorm = g.op("Mul", partial_weight, scales)
1256     partial_weight_renorm = g.op(
1257         "Where",
1258         g.op("Greater", partial_weight_norm, max_norm),
1259         partial_weight_renorm,
1260         partial_weight,
1261     )
1262     return g.op(
1263         "ScatterND",

1265         symbolic_helper._unsqueeze_helper(g, unique_indices, [1]),
