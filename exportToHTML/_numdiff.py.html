13   def _adjust_scheme_to_bounds(x0, h, num_steps, scheme, lb, ub):
14       """Adjust final difference scheme to the presence of bounds. 
15    
16       Parameters 
17       ---------- 
18       x0 : ndarray, shape (n,) 
19           Point at which we wish to estimate derivative. 
20       h : ndarray, shape (n,) 
21           Desired absolute finite difference steps. 
22       num_steps : int 
23           Number of `h` steps in one direction required to implement finite 
24           difference scheme. For example, 2 means that we need to evaluate 
25           f(x0 + 2 * h) or f(x0 - 2 * h) 
26       scheme : {'1-sided', '2-sided'} 
27           Whether steps in one or both directions are required. In other 
28           words '1-sided' applies to forward and backward schemes, '2-sided' 
29           applies to center schemes. 
30       lb : ndarray, shape (n,) 
31           Lower bounds on independent variables. 
32       ub : ndarray, shape (n,) 
33           Upper bounds on independent variables. 
34    
35       Returns 
36       ------- 
37       h_adjusted : ndarray, shape (n,) 
38           Adjusted absolute step sizes. Step size decreases only if a sign flip 
39           or switching to one-sided scheme doesn't allow to take a full step. 
40       use_one_sided : ndarray of bool, shape (n,) 
41           Whether to switch to one-sided scheme. Informative only for 
42           ``scheme='2-sided'``. 
43       """
44       if scheme == '1-sided':
45           use_one_sided = np.ones_like(h, dtype=bool)
46       elif scheme == '2-sided':
47           h = np.abs(h)
48           use_one_sided = np.zeros_like(h, dtype=bool)
49       else:
50           raise ValueError("`scheme` must be '1-sided' or '2-sided'.")
51   
52       if np.all((lb == -np.inf) & (ub == np.inf)):
53           return h, use_one_sided
54   
55       h_total = h * num_steps
56       h_adjusted = h.copy()
57   
58       lower_dist = x0 - lb
59       upper_dist = ub - x0
60   
61       if scheme == '1-sided':
62           x = x0 + h_total
63           violated = (x < lb) | (x > ub)
64           fitting = np.abs(h_total) <= np.maximum(lower_dist, upper_dist)
65           h_adjusted[violated & fitting] *= -1
66   
67           forward = (upper_dist >= lower_dist) & ~fitting
68           h_adjusted[forward] = upper_dist[forward] / num_steps
69           backward = (upper_dist < lower_dist) & ~fitting
70           h_adjusted[backward] = -lower_dist[backward] / num_steps
71       elif scheme == '2-sided':
72           central = (lower_dist >= h_total) & (upper_dist >= h_total)
73   
74           forward = (upper_dist >= lower_dist) & ~central
75           h_adjusted[forward] = np.minimum(
76               h[forward], 0.5 * upper_dist[forward] / num_steps)
77           use_one_sided[forward] = True
78   
79           backward = (upper_dist < lower_dist) & ~central
80           h_adjusted[backward] = -np.minimum(
81               h[backward], 0.5 * lower_dist[backward] / num_steps)
82           use_one_sided[backward] = True
83   
84           min_dist = np.minimum(upper_dist, lower_dist) / num_steps
85           adjusted_central = (~central & (np.abs(h_adjusted) <= min_dist))
86           h_adjusted[adjusted_central] = min_dist[adjusted_central]
87           use_one_sided[adjusted_central] = False
88   
89       return h_adjusted, use_one_sided

92   @functools.lru_cache
93   def _eps_for_method(x0_dtype, f0_dtype, method):
94       """ 
95       Calculates relative EPS step to use for a given data type 
96       and numdiff step method. 
97    
98       Progressively smaller steps are used for larger floating point types. 
99    
100      Parameters 
101      ---------- 
102      f0_dtype: np.dtype 
103          dtype of function evaluation 
104   
105      x0_dtype: np.dtype 
106          dtype of parameter vector 
107   
108      method: {'2-point', '3-point', 'cs'} 
109   
110      Returns 
111      ------- 
112      EPS: float 
113          relative step size. May be np.float16, np.float32, np.float64 
114   
115      Notes 
116      ----- 
117      The default relative step will be np.float64. However, if x0 or f0 are 
118      smaller floating point types (np.float16, np.float32), then the smallest 
119      floating point type is chosen. 
120      """
121      # the default EPS value
122      EPS = np.finfo(np.float64).eps
123  
124      x0_is_fp = False
125      if np.issubdtype(x0_dtype, np.inexact):
126          # if you're a floating point type then over-ride the default EPS
127          EPS = np.finfo(x0_dtype).eps
128          x0_itemsize = np.dtype(x0_dtype).itemsize
129          x0_is_fp = True
130  
131      if np.issubdtype(f0_dtype, np.inexact):
132          f0_itemsize = np.dtype(f0_dtype).itemsize
133          # choose the smallest itemsize between x0 and f0
134          if x0_is_fp and f0_itemsize < x0_itemsize:
135              EPS = np.finfo(f0_dtype).eps
136  
137      if method in ["2-point", "cs"]:
138          return EPS**0.5
139      elif method in ["3-point"]:
140          return EPS**(1/3)
141      else:
142          raise RuntimeError("Unknown step method, should be one of "
143                             "{'2-point', '3-point', 'cs'}")

146  def _compute_absolute_step(rel_step, x0, f0, method):
147      """ 
148      Computes an absolute step from a relative step for finite difference 
149      calculation. 
150   
151      Parameters 
152      ---------- 
153      rel_step: None or array-like 
154          Relative step for the finite difference calculation 
155      x0 : np.ndarray 
156          Parameter vector 
157      f0 : np.ndarray or scalar 
158      method : {'2-point', '3-point', 'cs'} 
159   
160      Returns 
161      ------- 
162      h : float 
163          The absolute step size 
164   
165      Notes 
166      ----- 
167      `h` will always be np.float64. However, if `x0` or `f0` are 
168      smaller floating point dtypes (e.g. np.float32), then the absolute 
169      step size will be calculated from the smallest floating point size. 
170      """
171      # this is used instead of np.sign(x0) because we need
172      # sign_x0 to be 1 when x0 == 0.
173      sign_x0 = (x0 >= 0).astype(float) * 2 - 1
174  
175      rstep = _eps_for_method(x0.dtype, f0.dtype, method)
176  
177      if rel_step is None:
178          abs_step = rstep * sign_x0 * np.maximum(1.0, np.abs(x0))
179      else:
180          # User has requested specific relative steps.
181          # Don't multiply by max(1, abs(x0) because if x0 < 1 then their
182          # requested step is not used.
183          abs_step = rel_step * sign_x0 * np.abs(x0)
184  
185          # however we don't want an abs_step of 0, which can happen if
186          # rel_step is 0, or x0 is 0. Instead, substitute a realistic step
187          dx = ((x0 + abs_step) - x0)
188          abs_step = np.where(dx == 0,
189                              rstep * sign_x0 * np.maximum(1.0, np.abs(x0)),
190                              abs_step)
191  
192      return abs_step

195  def _prepare_bounds(bounds, x0):
196      """ 
197      Prepares new-style bounds from a two-tuple specifying the lower and upper 
198      limits for values in x0. If a value is not bound then the lower/upper bound 
199      will be expected to be -np.inf/np.inf. 
200   
201      Examples 
202      -------- 
203      >>> _prepare_bounds([(0, 1, 2), (1, 2, np.inf)], [0.5, 1.5, 2.5]) 
204      (array([0., 1., 2.]), array([ 1.,  2., inf])) 
205      """
206      lb, ub = (np.asarray(b, dtype=float) for b in bounds)
207      if lb.ndim == 0:
208          lb = np.resize(lb, x0.shape)
209  
210      if ub.ndim == 0:
211          ub = np.resize(ub, x0.shape)
212  
213      return lb, ub

216  def group_columns(A, order=0):
217      """Group columns of a 2-D matrix for sparse finite differencing [1]_. 
218   
219      Two columns are in the same group if in each row at least one of them 
220      has zero. A greedy sequential algorithm is used to construct groups. 
221   
222      Parameters 
223      ---------- 
224      A : array_like or sparse matrix, shape (m, n) 
225          Matrix of which to group columns. 
226      order : int, iterable of int with shape (n,) or None 
227          Permutation array which defines the order of columns enumeration. 
228          If int or None, a random permutation is used with `order` used as 
229          a random seed. Default is 0, that is use a random permutation but 
230          guarantee repeatability. 
231   
232      Returns 
233      ------- 
234      groups : ndarray of int, shape (n,) 
235          Contains values from 0 to n_groups-1, where n_groups is the number 
236          of found groups. Each value ``groups[i]`` is an index of a group to 
237          which ith column assigned. The procedure was helpful only if 
238          n_groups is significantly less than n. 
239   
240      References 
241      ---------- 
242      .. [1] A. Curtis, M. J. D. Powell, and J. Reid, "On the estimation of 
243             sparse Jacobian matrices", Journal of the Institute of Mathematics 
244             and its Applications, 13 (1974), pp. 117-120. 
245      """
246      if issparse(A):
247          A = csc_matrix(A)
248      else:
249          A = np.atleast_2d(A)
250          A = (A != 0).astype(np.int32)
251  
252      if A.ndim != 2:
253          raise ValueError("`A` must be 2-dimensional.")
254  
255      m, n = A.shape
256  
257      if order is None or np.isscalar(order):
258          rng = np.random.RandomState(order)
259          order = rng.permutation(n)
260      else:
261          order = np.asarray(order)
262          if order.shape != (n,):
263              raise ValueError("`order` has incorrect shape.")
264  
265      A = A[:, order]
266  
267      if issparse(A):
268          groups = group_sparse(m, n, A.indices, A.indptr)
269      else:
270          groups = group_dense(m, n, A)
271  
272      groups[order] = groups.copy()
273  
274      return groups

277  def approx_derivative(fun, x0, method='3-point', rel_step=None, abs_step=None,
278                        f0=None, bounds=(-np.inf, np.inf), sparsity=None,
279                        as_linear_operator=False, args=(), kwargs=None):
280      """Compute finite difference approximation of the derivatives of a 
281      vector-valued function. 
282   
283      If a function maps from R^n to R^m, its derivatives form m-by-n matrix 
284      called the Jacobian, where an element (i, j) is a partial derivative of 
285      f[i] with respect to x[j]. 
286   
287      Parameters 
288      ---------- 
289      fun : callable 
290          Function of which to estimate the derivatives. The argument x 
291          passed to this function is ndarray of shape (n,) (never a scalar 
292          even if n=1). It must return 1-D array_like of shape (m,) or a scalar. 
293      x0 : array_like of shape (n,) or float 
294          Point at which to estimate the derivatives. Float will be converted 
295          to a 1-D array. 
296      method : {'3-point', '2-point', 'cs'}, optional 
297          Finite difference method to use: 
298              - '2-point' - use the first order accuracy forward or backward 
299                            difference. 
300              - '3-point' - use central difference in interior points and the 
301                            second order accuracy forward or backward difference 
302                            near the boundary. 
303              - 'cs' - use a complex-step finite difference scheme. This assumes 
304                       that the user function is real-valued and can be 
305                       analytically continued to the complex plane. Otherwise, 
306                       produces bogus results. 
307      rel_step : None or array_like, optional 
308          Relative step size to use. If None (default) the absolute step size is 
309          computed as ``h = rel_step * sign(x0) * max(1, abs(x0))``, with 
310          `rel_step` being selected automatically, see Notes. Otherwise 
311          ``h = rel_step * sign(x0) * abs(x0)``. For ``method='3-point'`` the 
312          sign of `h` is ignored. The calculated step size is possibly adjusted 
313          to fit into the bounds. 
314      abs_step : array_like, optional 
315          Absolute step size to use, possibly adjusted to fit into the bounds. 
316          For ``method='3-point'`` the sign of `abs_step` is ignored. By default 
317          relative steps are used, only if ``abs_step is not None`` are absolute 
318          steps used. 
319      f0 : None or array_like, optional 
320          If not None it is assumed to be equal to ``fun(x0)``, in this case 
321          the ``fun(x0)`` is not called. Default is None. 
322      bounds : tuple of array_like, optional 
323          Lower and upper bounds on independent variables. Defaults to no bounds. 
324          Each bound must match the size of `x0` or be a scalar, in the latter 
325          case the bound will be the same for all variables. Use it to limit the 
326          range of function evaluation. Bounds checking is not implemented 
327          when `as_linear_operator` is True. 
328      sparsity : {None, array_like, sparse matrix, 2-tuple}, optional 
329          Defines a sparsity structure of the Jacobian matrix. If the Jacobian 
330          matrix is known to have only few non-zero elements in each row, then 
331          it's possible to estimate its several columns by a single function 
332          evaluation [3]_. To perform such economic computations two ingredients 
333          are required: 
334   
335          * structure : array_like or sparse matrix of shape (m, n). A zero 
336            element means that a corresponding element of the Jacobian 
337            identically equals to zero. 
338          * groups : array_like of shape (n,). A column grouping for a given 
339            sparsity structure, use `group_columns` to obtain it. 
340   
341          A single array or a sparse matrix is interpreted as a sparsity 
342          structure, and groups are computed inside the function. A tuple is 
343          interpreted as (structure, groups). If None (default), a standard 
344          dense differencing will be used. 
345   
346          Note, that sparse differencing makes sense only for large Jacobian 
347          matrices where each row contains few non-zero elements. 
348      as_linear_operator : bool, optional 
349          When True the function returns an `scipy.sparse.linalg.LinearOperator`. 
350          Otherwise it returns a dense array or a sparse matrix depending on 
351          `sparsity`. The linear operator provides an efficient way of computing 
352          ``J.dot(p)`` for any vector ``p`` of shape (n,), but does not allow 
353          direct access to individual elements of the matrix. By default 
354          `as_linear_operator` is False. 
355      args, kwargs : tuple and dict, optional 
356          Additional arguments passed to `fun`. Both empty by default. 
357          The calling signature is ``fun(x, *args, **kwargs)``. 
358   
359      Returns 
360      ------- 
361      J : {ndarray, sparse matrix, LinearOperator} 
362          Finite difference approximation of the Jacobian matrix. 
363          If `as_linear_operator` is True returns a LinearOperator 
364          with shape (m, n). Otherwise it returns a dense array or sparse 
365          matrix depending on how `sparsity` is defined. If `sparsity` 
366          is None then a ndarray with shape (m, n) is returned. If 
367          `sparsity` is not None returns a csr_matrix with shape (m, n). 
368          For sparse matrices and linear operators it is always returned as 
369          a 2-D structure, for ndarrays, if m=1 it is returned 
370          as a 1-D gradient array with shape (n,). 
371   
372      See Also 
373      -------- 
374      check_derivative : Check correctness of a function computing derivatives. 
375   
376      Notes 
377      ----- 
378      If `rel_step` is not provided, it assigned as ``EPS**(1/s)``, where EPS is 
379      determined from the smallest floating point dtype of `x0` or `fun(x0)`, 
380      ``np.finfo(x0.dtype).eps``, s=2 for '2-point' method and 
381      s=3 for '3-point' method. Such relative step approximately minimizes a sum 
382      of truncation and round-off errors, see [1]_. Relative steps are used by 
383      default. However, absolute steps are used when ``abs_step is not None``. 
384      If any of the absolute or relative steps produces an indistinguishable 
385      difference from the original `x0`, ``(x0 + dx) - x0 == 0``, then a 
386      automatic step size is substituted for that particular entry. 
387   
388      A finite difference scheme for '3-point' method is selected automatically. 
389      The well-known central difference scheme is used for points sufficiently 
390      far from the boundary, and 3-point forward or backward scheme is used for 
391      points near the boundary. Both schemes have the second-order accuracy in 
392      terms of Taylor expansion. Refer to [2]_ for the formulas of 3-point 
393      forward and backward difference schemes. 
394   
395      For dense differencing when m=1 Jacobian is returned with a shape (n,), 
396      on the other hand when n=1 Jacobian is returned with a shape (m, 1). 
397      Our motivation is the following: a) It handles a case of gradient 
398      computation (m=1) in a conventional way. b) It clearly separates these two 
399      different cases. b) In all cases np.atleast_2d can be called to get 2-D 
400      Jacobian with correct dimensions. 
401   
402      References 
403      ---------- 
404      .. [1] W. H. Press et. al. "Numerical Recipes. The Art of Scientific 
405             Computing. 3rd edition", sec. 5.7. 
406   
407      .. [2] A. Curtis, M. J. D. Powell, and J. Reid, "On the estimation of 
408             sparse Jacobian matrices", Journal of the Institute of Mathematics 
409             and its Applications, 13 (1974), pp. 117-120. 
410   
411      .. [3] B. Fornberg, "Generation of Finite Difference Formulas on 
412             Arbitrarily Spaced Grids", Mathematics of Computation 51, 1988. 
413   
414      Examples 
415      -------- 
416      >>> import numpy as np 
417      >>> from scipy.optimize._numdiff import approx_derivative 
418      >>> 
419      >>> def f(x, c1, c2): 
420      ...     return np.array([x[0] * np.sin(c1 * x[1]), 
421      ...                      x[0] * np.cos(c2 * x[1])]) 
422      ... 
423      >>> x0 = np.array([1.0, 0.5 * np.pi]) 
424      >>> approx_derivative(f, x0, args=(1, 2)) 
425      array([[ 1.,  0.], 
426             [-1.,  0.]]) 
427   
428      Bounds can be used to limit the region of function evaluation. 
429      In the example below we compute left and right derivative at point 1.0. 
430   
431      >>> def g(x): 
432      ...     return x**2 if x >= 1 else x 
433      ... 
434      >>> x0 = 1.0 
435      >>> approx_derivative(g, x0, bounds=(-np.inf, 1.0)) 
436      array([ 1.]) 
437      >>> approx_derivative(g, x0, bounds=(1.0, np.inf)) 
438      array([ 2.]) 
439      """
440      if method not in ['2-point', '3-point', 'cs']:
441          raise ValueError(f"Unknown method '{method}'. ")
442  
443      xp = array_namespace(x0)
444      _x = xpx.atleast_nd(xp.asarray(x0), ndim=1, xp=xp)
445      _dtype = xp.float64
446      if xp.isdtype(_x.dtype, "real floating"):
447          _dtype = _x.dtype
448  
449      # promotes to floating
450      x0 = xp.astype(_x, _dtype)
451  
452      if x0.ndim > 1:
453          raise ValueError("`x0` must have at most 1 dimension.")
454  
455      lb, ub = _prepare_bounds(bounds, x0)
456  
457      if lb.shape != x0.shape or ub.shape != x0.shape:
458          raise ValueError("Inconsistent shapes between bounds and `x0`.")
459  
460      if as_linear_operator and not (np.all(np.isinf(lb))
461                                     and np.all(np.isinf(ub))):
462          raise ValueError("Bounds not supported when "
463                           "`as_linear_operator` is True.")
464  
465      if kwargs is None:
466          kwargs = {}
467  
468      def fun_wrapped(x):
469          # send user function same fp type as x0. (but only if cs is not being
470          # used
471          if xp.isdtype(x.dtype, "real floating"):
472              x = xp.astype(x, x0.dtype)
473  
474          f = np.atleast_1d(fun(x, *args, **kwargs))
475          if f.ndim > 1:
476              raise RuntimeError("`fun` return value has "
477                                 "more than 1 dimension.")
478          return f
479  
480      if f0 is None:
481          f0 = fun_wrapped(x0)
482      else:
483          f0 = np.atleast_1d(f0)
484          if f0.ndim > 1:
485              raise ValueError("`f0` passed has more than 1 dimension.")
486  
487      if np.any((x0 < lb) | (x0 > ub)):
488          raise ValueError("`x0` violates bound constraints.")
489  
490      if as_linear_operator:
491          if rel_step is None:
492              rel_step = _eps_for_method(x0.dtype, f0.dtype, method)
493  
494          return _linear_operator_difference(fun_wrapped, x0,
495                                             f0, rel_step, method)
496      else:
497          # by default we use rel_step
498          if abs_step is None:
499              h = _compute_absolute_step(rel_step, x0, f0, method)
500          else:
501              # user specifies an absolute step
502              sign_x0 = (x0 >= 0).astype(float) * 2 - 1
503              h = abs_step
504  
505              # cannot have a zero step. This might happen if x0 is very large
506              # or small. In which case fall back to relative step.
507              dx = ((x0 + h) - x0)
508              h = np.where(dx == 0,
509                           _eps_for_method(x0.dtype, f0.dtype, method) *
510                           sign_x0 * np.maximum(1.0, np.abs(x0)),
511                           h)
512  
513          if method == '2-point':
514              h, use_one_sided = _adjust_scheme_to_bounds(
515                  x0, h, 1, '1-sided', lb, ub)
516          elif method == '3-point':
517              h, use_one_sided = _adjust_scheme_to_bounds(
518                  x0, h, 1, '2-sided', lb, ub)
519          elif method == 'cs':
520              use_one_sided = False
521  
522          if sparsity is None:
523              return _dense_difference(fun_wrapped, x0, f0, h,
524                                       use_one_sided, method)
525          else:
526              if not issparse(sparsity) and len(sparsity) == 2:
527                  structure, groups = sparsity
528              else:
529                  structure = sparsity
530                  groups = group_columns(sparsity)
531  
532              if issparse(structure):
533                  structure = csc_matrix(structure)
534              else:
535                  structure = np.atleast_2d(structure)
536  
537              groups = np.atleast_1d(groups)
538              return _sparse_difference(fun_wrapped, x0, f0, h,
539                                        use_one_sided, structure,
540                                        groups, method)

543  def _linear_operator_difference(fun, x0, f0, h, method):
544      m = f0.size
545      n = x0.size
546  
547      if method == '2-point':
548          def matvec(p):
549              if np.array_equal(p, np.zeros_like(p)):
550                  return np.zeros(m)
551              dx = h / norm(p)
552              x = x0 + dx*p
553              df = fun(x) - f0
554              return df / dx
555  
556      elif method == '3-point':
557          def matvec(p):
558              if np.array_equal(p, np.zeros_like(p)):
559                  return np.zeros(m)
560              dx = 2*h / norm(p)
561              x1 = x0 - (dx/2)*p
562              x2 = x0 + (dx/2)*p
563              f1 = fun(x1)
564              f2 = fun(x2)
565              df = f2 - f1
566              return df / dx
567  
568      elif method == 'cs':
569          def matvec(p):
570              if np.array_equal(p, np.zeros_like(p)):
571                  return np.zeros(m)
572              dx = h / norm(p)
573              x = x0 + dx*p*1.j
574              f1 = fun(x)
575              df = f1.imag
576              return df / dx
577  
578      else:
579          raise RuntimeError("Never be here.")
580  
581      return LinearOperator((m, n), matvec)

584  def _dense_difference(fun, x0, f0, h, use_one_sided, method):
585      m = f0.size
586      n = x0.size
587      J_transposed = np.empty((n, m))
588      x1 = x0.copy()
589      x2 = x0.copy()
590      xc = x0.astype(complex, copy=True)
591  
592      for i in range(h.size):
593          if method == '2-point':
594              x1[i] += h[i]
595              dx = x1[i] - x0[i]  # Recompute dx as exactly representable number.
596              df = fun(x1) - f0
597          elif method == '3-point' and use_one_sided[i]:
598              x1[i] += h[i]
599              x2[i] += 2 * h[i]
600              dx = x2[i] - x0[i]
601              f1 = fun(x1)
602              f2 = fun(x2)
603              df = -3.0 * f0 + 4 * f1 - f2
604          elif method == '3-point' and not use_one_sided[i]:
605              x1[i] -= h[i]
606              x2[i] += h[i]
607              dx = x2[i] - x1[i]
608              f1 = fun(x1)
609              f2 = fun(x2)
610              df = f2 - f1
611          elif method == 'cs':
612              xc[i] += h[i] * 1.j
613              f1 = fun(xc)
614              df = f1.imag
615              dx = h[i]
616          else:
617              raise RuntimeError("Never be here.")
618  
619          J_transposed[i] = df / dx
620          x1[i] = x2[i] = xc[i] = x0[i]
621  
622      if m == 1:
623          J_transposed = np.ravel(J_transposed)
624  
625      return J_transposed.T

628  def _sparse_difference(fun, x0, f0, h, use_one_sided,
629                         structure, groups, method):
630      m = f0.size
631      n = x0.size
632      row_indices = []
633      col_indices = []
634      fractions = []
635  
636      n_groups = np.max(groups) + 1
637      for group in range(n_groups):
638          # Perturb variables which are in the same group simultaneously.
639          e = np.equal(group, groups)
640          h_vec = h * e
641          if method == '2-point':
642              x = x0 + h_vec
643              dx = x - x0
644              df = fun(x) - f0
645              # The result is  written to columns which correspond to perturbed
646              # variables.
647              cols, = np.nonzero(e)
648              # Find all non-zero elements in selected columns of Jacobian.
649              i, j, _ = find(structure[:, cols])
650              # Restore column indices in the full array.
651              j = cols[j]
652          elif method == '3-point':
653              # Here we do conceptually the same but separate one-sided
654              # and two-sided schemes.
655              x1 = x0.copy()
656              x2 = x0.copy()
657  
658              mask_1 = use_one_sided & e
659              x1[mask_1] += h_vec[mask_1]
660              x2[mask_1] += 2 * h_vec[mask_1]
661  
662              mask_2 = ~use_one_sided & e
663              x1[mask_2] -= h_vec[mask_2]
664              x2[mask_2] += h_vec[mask_2]
665  
666              dx = np.zeros(n)
667              dx[mask_1] = x2[mask_1] - x0[mask_1]
668              dx[mask_2] = x2[mask_2] - x1[mask_2]
669  
670              f1 = fun(x1)
671              f2 = fun(x2)
672  
673              cols, = np.nonzero(e)
674              i, j, _ = find(structure[:, cols])
675              j = cols[j]
676  
677              mask = use_one_sided[j]
678              df = np.empty(m)
679  
680              rows = i[mask]
681              df[rows] = -3 * f0[rows] + 4 * f1[rows] - f2[rows]
682  
683              rows = i[~mask]
684              df[rows] = f2[rows] - f1[rows]
685          elif method == 'cs':
686              f1 = fun(x0 + h_vec*1.j)
687              df = f1.imag
688              dx = h_vec
689              cols, = np.nonzero(e)
690              i, j, _ = find(structure[:, cols])
691              j = cols[j]
692          else:
693              raise ValueError("Never be here.")
694  
695          # All that's left is to compute the fraction. We store i, j and
696          # fractions as separate arrays and later construct coo_matrix.
697          row_indices.append(i)
698          col_indices.append(j)
699          fractions.append(df[i] / dx[j])
700  
701      row_indices = np.hstack(row_indices)
702      col_indices = np.hstack(col_indices)
703      fractions = np.hstack(fractions)
704      J = coo_matrix((fractions, (row_indices, col_indices)), shape=(m, n))
705      return csr_matrix(J)

708  def check_derivative(fun, jac, x0, bounds=(-np.inf, np.inf), args=(),
709                       kwargs=None):
710      """Check correctness of a function computing derivatives (Jacobian or 
711      gradient) by comparison with a finite difference approximation. 
712   
713      Parameters 
714      ---------- 
715      fun : callable 
716          Function of which to estimate the derivatives. The argument x 
717          passed to this function is ndarray of shape (n,) (never a scalar 
718          even if n=1). It must return 1-D array_like of shape (m,) or a scalar. 
719      jac : callable 
720          Function which computes Jacobian matrix of `fun`. It must work with 
721          argument x the same way as `fun`. The return value must be array_like 
722          or sparse matrix with an appropriate shape. 
723      x0 : array_like of shape (n,) or float 
724          Point at which to estimate the derivatives. Float will be converted 
725          to 1-D array. 
726      bounds : 2-tuple of array_like, optional 
727          Lower and upper bounds on independent variables. Defaults to no bounds. 
728          Each bound must match the size of `x0` or be a scalar, in the latter 
729          case the bound will be the same for all variables. Use it to limit the 
730          range of function evaluation. 
731      args, kwargs : tuple and dict, optional 
732          Additional arguments passed to `fun` and `jac`. Both empty by default. 
733          The calling signature is ``fun(x, *args, **kwargs)`` and the same 
734          for `jac`. 
735   
736      Returns 
737      ------- 
738      accuracy : float 
739          The maximum among all relative errors for elements with absolute values 
740          higher than 1 and absolute errors for elements with absolute values 
741          less or equal than 1. If `accuracy` is on the order of 1e-6 or lower, 
742          then it is likely that your `jac` implementation is correct. 
743   
744      See Also 
745      -------- 
746      approx_derivative : Compute finite difference approximation of derivative. 
747   
748      Examples 
749      -------- 
750      >>> import numpy as np 
751      >>> from scipy.optimize._numdiff import check_derivative 
752      >>> 
753      >>> 
754      >>> def f(x, c1, c2): 
755      ...     return np.array([x[0] * np.sin(c1 * x[1]), 
756      ...                      x[0] * np.cos(c2 * x[1])]) 
757      ... 
758      >>> def jac(x, c1, c2): 
759      ...     return np.array([ 
760      ...         [np.sin(c1 * x[1]),  c1 * x[0] * np.cos(c1 * x[1])], 
761      ...         [np.cos(c2 * x[1]), -c2 * x[0] * np.sin(c2 * x[1])] 
762      ...     ]) 
763      ... 
764      >>> 
765      >>> x0 = np.array([1.0, 0.5 * np.pi]) 
766      >>> check_derivative(f, jac, x0, args=(1, 2)) 
767      2.4492935982947064e-16 
768      """
769      if kwargs is None:
770          kwargs = {}
771      J_to_test = jac(x0, *args, **kwargs)
772      if issparse(J_to_test):
773          J_diff = approx_derivative(fun, x0, bounds=bounds, sparsity=J_to_test,
774                                     args=args, kwargs=kwargs)
775          J_to_test = csr_matrix(J_to_test)
776          abs_err = J_to_test - J_diff
777          i, j, abs_err_data = find(abs_err)
778          J_diff_data = np.asarray(J_diff[i, j]).ravel()
779          return np.max(np.abs(abs_err_data) /
780                        np.maximum(1, np.abs(J_diff_data)))
781      else:
782          J_diff = approx_derivative(fun, x0, bounds=bounds,
783                                     args=args, kwargs=kwargs)
784          abs_err = np.abs(J_to_test - J_diff)
785          return np.max(abs_err / np.maximum(1, np.abs(J_diff)))
