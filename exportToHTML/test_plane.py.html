259  def test_parameter_value():
260      t, u, v = symbols("t, u v")
261      p1, p2, p3 = Point(0, 0, 0), Point(0, 0, 1), Point(0, 1, 0)
262      p = Plane(p1, p2, p3)
263      assert p.parameter_value((0, -3, 2), t) == {t: asin(2*sqrt(13)/13)}
264      assert p.parameter_value((0, -3, 2), u, v) == {u: 3, v: 2}
265      assert p.parameter_value(p1, t) == p1
266      raises(ValueError, lambda: p.parameter_value((1, 0, 0), t))
267      raises(ValueError, lambda: p.parameter_value(Line(Point(0, 0), Point(1, 1)), t))
268      raises(ValueError, lambda: p.parameter_value((0, -3, 2), t, 1))
