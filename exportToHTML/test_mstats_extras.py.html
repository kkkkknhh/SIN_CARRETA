107          assert_almost_equal(ms.hdquantiles(data,[0., 1.]),
108                              [0.006514031, 0.995309248])

119          assert_almost_equal(hdq[...,-1],
120                              ms.hdquantiles(data[:,-1],[0.25,0.5,0.75], var=True))

122      def test_hdquantiles_sd(self):
123          # Standard deviation is a jackknife estimator, so we can check if
124          # the efficient version (hdquantiles_sd) matches a rudimentary,
125          # but clear version here.
126  
127          hd_std_errs = ms.hdquantiles_sd(self.data)
128  
129          # jacknnife standard error, Introduction to the Bootstrap Eq. 11.5
130          n = len(self.data)
131          jdata = np.broadcast_to(self.data, (n, n))
132          jselector = np.logical_not(np.eye(n))  # leave out one sample each row
133          jdata = jdata[jselector].reshape(n, n-1)
134          jdist = ms.hdquantiles(jdata, axis=1)
135          jdist_mean = np.mean(jdist, axis=0)
136          jstd = ((n-1)/n * np.sum((jdist - jdist_mean)**2, axis=0))**.5
137  
138          assert_almost_equal(hd_std_errs, jstd)
139          # Test actual values for good measure
140          assert_almost_equal(hd_std_errs, [0.0379258, 0.0380656, 0.0380013])
141  
142          two_data_points = ms.hdquantiles_sd([1, 2])
143          assert_almost_equal(two_data_points, [0.5, 0.5, 0.5])

151  def test_median_cihs():
152      # Basic test against R library EnvStats function `eqnpar`, e.g.
153      # library(EnvStats)
154      # options(digits=8)
155      # x = c(0.88612955, 0.35242375, 0.66240904, 0.94617974, 0.10929913,
156      #       0.76699506, 0.88550655, 0.62763754, 0.76818588, 0.68506508,
157      #       0.88043148, 0.03911248, 0.93805564, 0.95326961, 0.25291112,
158      #       0.16128487, 0.49784577, 0.24588924, 0.6597, 0.92239679)
159      # eqnpar(x, p=0.5,
160      #        ci.method = "interpolate", approx.conf.level = 0.95, ci = TRUE)
161      rng = np.random.default_rng(8824288259505800535)
162      x = rng.random(size=20)
163      assert_allclose(ms.median_cihs(x), (0.38663198, 0.88431272))
164  
165      # SciPy's 90% CI upper limit doesn't match that of EnvStats eqnpar. SciPy
166      # doesn't look wrong, and it agrees with a different reference,
167      # `median_confint_hs` from `hoehleatsu/quantileCI`.
168      # In (e.g.) Colab with R runtime:
169      # devtools::install_github("hoehleatsu/quantileCI")
170      # library(quantileCI)
171      # median_confint_hs(x=x, conf.level=0.90, interpolate=TRUE)
172      assert_allclose(ms.median_cihs(x, 0.1), (0.48319773366, 0.88094268050))
