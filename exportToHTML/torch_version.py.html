11   class TorchVersion(str):
12       """A string with magic powers to compare to both Version and iterables! 
13       Prior to 1.10.0 torch.__version__ was stored as a str and so many did 
14       comparisons against torch.__version__ as if it were a str. In order to not 
15       break them we have TorchVersion which masquerades as a str while also 
16       having the ability to compare against both packaging.version.Version as 
17       well as tuples of values, eg. (1, 2, 1) 
18       Examples: 
19           Comparing a TorchVersion object to a Version object 
20               TorchVersion('1.10.0a') > Version('1.10.0a') 
21           Comparing a TorchVersion object to a Tuple object 
22               TorchVersion('1.10.0a') > (1, 2)    # 1.2 
23               TorchVersion('1.10.0a') > (1, 2, 1) # 1.2.1 
24           Comparing a TorchVersion object against a string 
25               TorchVersion('1.10.0a') > '1.2' 
26               TorchVersion('1.10.0a') > '1.2.1' 
27       """
28   
29       __slots__ = ()
30   
31       # fully qualified type names here to appease mypy
32       def _convert_to_version(self, inp: Any) -> Any:
33           if isinstance(inp, Version):
34               return inp
35           elif isinstance(inp, str):
36               return Version(inp)
37           elif isinstance(inp, Iterable):
38               # Ideally this should work for most cases by attempting to group
39               # the version tuple, assuming the tuple looks (MAJOR, MINOR, ?PATCH)
40               # Examples:
41               #   * (1)         -> Version("1")
42               #   * (1, 20)     -> Version("1.20")
43               #   * (1, 20, 1)  -> Version("1.20.1")
44               return Version(".".join(str(item) for item in inp))
45           else:
46               raise InvalidVersion(inp)
47   
48       def _cmp_wrapper(self, cmp: Any, method: str) -> bool:
49           try:
50               return getattr(Version(self), method)(self._convert_to_version(cmp))
51           except BaseException as e:
52               if not isinstance(e, InvalidVersion):
53                   raise
54               # Fall back to regular string comparison if dealing with an invalid
55               # version like 'parrot'
56               return getattr(super(), method)(cmp)
57   
58   
59   for cmp_method in ["__gt__", "__lt__", "__eq__", "__ge__", "__le__"]:
60       setattr(
61           TorchVersion,
62           cmp_method,
63           lambda x, y, method=cmp_method: x._cmp_wrapper(y, method),
64       )
65   
66   __version__ = TorchVersion(internal_version)
