9    def dup_jacobi(n, a, b, K):
10       """Low-level implementation of Jacobi polynomials."""
11       if n < 1:
12           return [K.one]
13       m2, m1 = [K.one], [(a+b)/K(2) + K.one, (a-b)/K(2)]
14       for i in range(2, n+1):
15           den = K(i)*(a + b + i)*(a + b + K(2)*i - K(2))
16           f0 = (a + b + K(2)*i - K.one) * (a*a - b*b) / (K(2)*den)
17           f1 = (a + b + K(2)*i - K.one) * (a + b + K(2)*i - K(2)) * (a + b + K(2)*i) / (K(2)*den)
18           f2 = (a + i - K.one)*(b + i - K.one)*(a + b + K(2)*i) / den
19           p0 = dup_mul_ground(m1, f0, K)
20           p1 = dup_mul_ground(dup_lshift(m1, 1, K), f1, K)
21           p2 = dup_mul_ground(m2, f2, K)
22           m2, m1 = m1, dup_sub(dup_add(p0, p1, K), p2, K)
23       return m1

44   def dup_gegenbauer(n, a, K):
45       """Low-level implementation of Gegenbauer polynomials."""
46       if n < 1:
47           return [K.one]
48       m2, m1 = [K.one], [K(2)*a, K.zero]
49       for i in range(2, n+1):
50           p1 = dup_mul_ground(dup_lshift(m1, 1, K), K(2)*(a-K.one)/K(i) + K(2), K)
51           p2 = dup_mul_ground(m2, K(2)*(a-K.one)/K(i) + K.one, K)
52           m2, m1 = m1, dup_sub(p1, p2, K)
53       return m1

71   def dup_chebyshevt(n, K):
72       """Low-level implementation of Chebyshev polynomials of the first kind."""
73       if n < 1:
74           return [K.one]
75       # When n is small, it is faster to directly calculate the recurrence relation.
76       if n < 64: # The threshold serves as a heuristic
77           return _dup_chebyshevt_rec(n, K)
78       return _dup_chebyshevt_prod(n, K)
79   
80   def _dup_chebyshevt_rec(n, K):
81       r""" Chebyshev polynomials of the first kind using recurrence. 
82    
83       Explanation 
84       =========== 
85    
86       Chebyshev polynomials of the first kind are defined by the recurrence 
87       relation: 
88    
89       .. math:: 
90           T_0(x) &= 1\\ 
91           T_1(x) &= x\\ 
92           T_n(x) &= 2xT_{n-1}(x) - T_{n-2}(x) 
93    
94       This function calculates the Chebyshev polynomial of the first kind using 
95       the above recurrence relation. 
96    
97       Parameters 
98       ========== 
99    
100      n : int 
101          n is a nonnegative integer. 
102      K : domain 
103   
104      """
105      m2, m1 = [K.one], [K.one, K.zero]
106      for _ in range(n - 1):
107          m2, m1 = m1, dup_sub(dup_mul_ground(dup_lshift(m1, 1, K), K(2), K), m2, K)
108      return m1
109  
110  def _dup_chebyshevt_prod(n, K):
111      r""" Chebyshev polynomials of the first kind using recursive products. 
112   
113      Explanation 
114      =========== 
115   
116      Computes Chebyshev polynomials of the first kind using 
117   
118      .. math:: 
119          T_{2n}(x) &= 2T_n^2(x) - 1\\ 
120          T_{2n+1}(x) &= 2T_{n+1}(x)T_n(x) - x 
121   
122      This is faster than ``_dup_chebyshevt_rec`` for large ``n``. 
123   
124      Parameters 
125      ========== 
126   
127      n : int 
128          n is a nonnegative integer. 
129      K : domain 
130   
131      """
132      m2, m1 = [K.one, K.zero], [K(2), K.zero, -K.one]
133      for i in bin(n)[3:]:
134          c = dup_sub_term(dup_mul_ground(dup_mul(m1, m2, K), K(2), K), K.one, 1, K)
135          if  i  == '1':
136              m2, m1 = c, dup_sub_ground(dup_mul_ground(dup_sqr(m1, K), K(2), K), K.one, K)
137          else:
138              m2, m1 = dup_sub_ground(dup_mul_ground(dup_sqr(m2, K), K(2), K), K.one, K), c
139      return m2

141  def dup_chebyshevu(n, K):
142      """Low-level implementation of Chebyshev polynomials of the second kind."""
143      if n < 1:
144          return [K.one]
145      m2, m1 = [K.one], [K(2), K.zero]
146      for i in range(2, n+1):
147          m2, m1 = m1, dup_sub(dup_mul_ground(dup_lshift(m1, 1, K), K(2), K), m2, K)
148      return m1

182  def dup_hermite(n, K):
183      """Low-level implementation of Hermite polynomials."""
184      if n < 1:
185          return [K.one]
186      m2, m1 = [K.one], [K(2), K.zero]
187      for i in range(2, n+1):
188          a = dup_lshift(m1, 1, K)
189          b = dup_mul_ground(m2, K(i-1), K)
190          m2, m1 = m1, dup_mul_ground(dup_sub(a, b, K), K(2), K)
191      return m1

193  def dup_hermite_prob(n, K):
194      """Low-level implementation of probabilist's Hermite polynomials."""
195      if n < 1:
196          return [K.one]
197      m2, m1 = [K.one], [K.one, K.zero]
198      for i in range(2, n+1):
199          a = dup_lshift(m1, 1, K)
200          b = dup_mul_ground(m2, K(i-1), K)
201          m2, m1 = m1, dup_sub(a, b, K)
202      return m1

235  def dup_legendre(n, K):
236      """Low-level implementation of Legendre polynomials."""
237      if n < 1:
238          return [K.one]
239      m2, m1 = [K.one], [K.one, K.zero]
240      for i in range(2, n+1):
241          a = dup_mul_ground(dup_lshift(m1, 1, K), K(2*i-1, i), K)
242          b = dup_mul_ground(m2, K(i-1, i), K)
243          m2, m1 = m1, dup_sub(a, b, K)
244      return m1

261  def dup_laguerre(n, alpha, K):
262      """Low-level implementation of Laguerre polynomials."""
263      m2, m1 = [K.zero], [K.one]
264      for i in range(1, n+1):
265          a = dup_mul(m1, [-K.one/K(i), (alpha-K.one)/K(i) + K(2)], K)
266          b = dup_mul_ground(m2, (alpha-K.one)/K(i) + K.one, K)
267          m2, m1 = m1, dup_sub(a, b, K)
268      return m1

287  def dup_spherical_bessel_fn(n, K):
288      """Low-level implementation of fn(n, x)."""
289      if n < 1:
290          return [K.one, K.zero]
291      m2, m1 = [K.one], [K.one, K.zero]
292      for i in range(2, n+1):
293          m2, m1 = m1, dup_sub(dup_mul_ground(dup_lshift(m1, 1, K), K(2*i-1), K), m2, K)
294      return dup_lshift(m1, 1, K)

296  def dup_spherical_bessel_fn_minus(n, K):
297      """Low-level implementation of fn(-n, x)."""
298      m2, m1 = [K.one, K.zero], [K.zero]
299      for i in range(2, n+1):
300          m2, m1 = m1, dup_sub(dup_mul_ground(dup_lshift(m1, 1, K), K(3-2*i), K), m2, K)
301      return m1

303  def spherical_bessel_fn(n, x=None, polys=False):
304      """ 
305      Coefficients for the spherical Bessel functions. 
306   
307      These are only needed in the jn() function. 
308   
309      The coefficients are calculated from: 
310   
311      fn(0, z) = 1/z 
312      fn(1, z) = 1/z**2 
313      fn(n-1, z) + fn(n+1, z) == (2*n+1)/z * fn(n, z) 
314   
315      Parameters 
316      ========== 
317   
318      n : int 
319          Degree of the polynomial. 
320      x : optional 
321      polys : bool, optional 
322          If True, return a Poly, otherwise (default) return an expression. 
323   
324      Examples 
325      ======== 
326   
327      >>> from sympy.polys.orthopolys import spherical_bessel_fn as fn 
328      >>> from sympy import Symbol 
329      >>> z = Symbol("z") 
330      >>> fn(1, z) 
331      z**(-2) 
332      >>> fn(2, z) 
333      -1/z + 3/z**3 
334      >>> fn(3, z) 
335      -6/z**2 + 15/z**4 
336      >>> fn(4, z) 
337      1/z - 45/z**3 + 105/z**5 
338   
339      """
340      if x is None:
341          x = Dummy("x")
342      f = dup_spherical_bessel_fn_minus if n < 0 else dup_spherical_bessel_fn
343      return named_poly(abs(n), f, ZZ, "", (QQ(1)/x,), polys)
