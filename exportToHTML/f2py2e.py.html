197  def scaninputline(inputline):
198      files, skipfuncs, onlyfuncs, debug = [], [], [], []
199      f, f2, f3, f5, f6, f8, f9, f10 = 1, 0, 0, 0, 0, 0, 0, 0
200      verbose = 1
201      emptygen = True
202      dolc = -1
203      dolatexdoc = 0
204      dorestdoc = 0
205      wrapfuncs = 1
206      buildpath = '.'
207      include_paths, freethreading_compatible, inputline = get_newer_options(inputline)
208      signsfile, modulename = None, None
209      options = {'buildpath': buildpath,
210                 'coutput': None,
211                 'f2py_wrapper_output': None}
212      for l in inputline:
213          if l == '':
214              pass
215          elif l == 'only:':
216              f = 0
217          elif l == 'skip:':
218              f = -1
219          elif l == ':':
220              f = 1
221          elif l[:8] == '--debug-':
222              debug.append(l[8:])
223          elif l == '--lower':
224              dolc = 1
225          elif l == '--build-dir':
226              f6 = 1
227          elif l == '--no-lower':
228              dolc = 0
229          elif l == '--quiet':
230              verbose = 0
231          elif l == '--verbose':
232              verbose += 1
233          elif l == '--latex-doc':
234              dolatexdoc = 1
235          elif l == '--no-latex-doc':
236              dolatexdoc = 0
237          elif l == '--rest-doc':
238              dorestdoc = 1
239          elif l == '--no-rest-doc':
240              dorestdoc = 0
241          elif l == '--wrap-functions':
242              wrapfuncs = 1
243          elif l == '--no-wrap-functions':
244              wrapfuncs = 0
245          elif l == '--short-latex':
246              options['shortlatex'] = 1
247          elif l == '--coutput':
248              f8 = 1
249          elif l == '--f2py-wrapper-output':
250              f9 = 1
251          elif l == '--f2cmap':
252              f10 = 1
253          elif l == '--overwrite-signature':
254              options['h-overwrite'] = 1
255          elif l == '-h':
256              f2 = 1
257          elif l == '-m':
258              f3 = 1
259          elif l[:2] == '-v':
260              print(f2py_version)
261              sys.exit()
262          elif l == '--show-compilers':
263              f5 = 1
264          elif l[:8] == '-include':
265              cfuncs.outneeds['userincludes'].append(l[9:-1])
266              cfuncs.userincludes[l[9:-1]] = '#include ' + l[8:]
267          elif l == '--skip-empty-wrappers':
268              emptygen = False
269          elif l[0] == '-':
270              errmess('Unknown option %s\n' % repr(l))
271              sys.exit()
272          elif f2:
273              f2 = 0
274              signsfile = l
275          elif f3:
276              f3 = 0
277              modulename = l
278          elif f6:
279              f6 = 0
280              buildpath = l
281          elif f8:
282              f8 = 0
283              options["coutput"] = l
284          elif f9:
285              f9 = 0
286              options["f2py_wrapper_output"] = l
287          elif f10:
288              f10 = 0
289              options["f2cmap_file"] = l
290          elif f == 1:
291              try:
292                  with open(l):
293                      pass
294                  files.append(l)
295              except OSError as detail:
296                  errmess(f'OSError: {detail!s}. Skipping file "{l!s}".\n')
297          elif f == -1:
298              skipfuncs.append(l)
299          elif f == 0:
300              onlyfuncs.append(l)
301      if not f5 and not files and not modulename:
302          print(__usage__)
303          sys.exit()
304      if not os.path.isdir(buildpath):
305          if not verbose:
306              outmess('Creating build directory %s\n' % (buildpath))
307          os.mkdir(buildpath)
308      if signsfile:
309          signsfile = os.path.join(buildpath, signsfile)
310      if signsfile and os.path.isfile(signsfile) and 'h-overwrite' not in options:
311          errmess(
312              'Signature file "%s" exists!!! Use --overwrite-signature to overwrite.\n' % (signsfile))
313          sys.exit()
314  
315      options['emptygen'] = emptygen
316      options['debug'] = debug
317      options['verbose'] = verbose
318      if dolc == -1 and not signsfile:
319          options['do-lower'] = 0
320      else:
321          options['do-lower'] = dolc
322      if modulename:
323          options['module'] = modulename
324      if signsfile:
325          options['signsfile'] = signsfile
326      if onlyfuncs:
327          options['onlyfuncs'] = onlyfuncs
328      if skipfuncs:
329          options['skipfuncs'] = skipfuncs
330      options['dolatexdoc'] = dolatexdoc
331      options['dorestdoc'] = dorestdoc
332      options['wrapfuncs'] = wrapfuncs
333      options['buildpath'] = buildpath
334      options['include_paths'] = include_paths
335      options['requires_gil'] = not freethreading_compatible
336      options.setdefault('f2cmap_file', None)
337      return files, options

340  def callcrackfortran(files, options):
341      rules.options = options
342      crackfortran.debug = options['debug']
343      crackfortran.verbose = options['verbose']
344      if 'module' in options:
345          crackfortran.f77modulename = options['module']
346      if 'skipfuncs' in options:
347          crackfortran.skipfuncs = options['skipfuncs']
348      if 'onlyfuncs' in options:
349          crackfortran.onlyfuncs = options['onlyfuncs']
350      crackfortran.include_paths[:] = options['include_paths']
351      crackfortran.dolowercase = options['do-lower']
352      postlist = crackfortran.crackfortran(files)
353      if 'signsfile' in options:
354          outmess('Saving signatures to file "%s"\n' % (options['signsfile']))
355          pyf = crackfortran.crack2fortran(postlist)
356          if options['signsfile'][-6:] == 'stdout':
357              sys.stdout.write(pyf)
358          else:
359              with open(options['signsfile'], 'w') as f:
360                  f.write(pyf)
361      if options["coutput"] is None:
362          for mod in postlist:
363              mod["coutput"] = "%smodule.c" % mod["name"]
364      else:
365          for mod in postlist:
366              mod["coutput"] = options["coutput"]
367      if options["f2py_wrapper_output"] is None:
368          for mod in postlist:
369              mod["f2py_wrapper_output"] = "%s-f2pywrappers.f" % mod["name"]
370      else:
371          for mod in postlist:
372              mod["f2py_wrapper_output"] = options["f2py_wrapper_output"]
373      for mod in postlist:
374          if options["requires_gil"]:
375              mod['gil_used'] = 'Py_MOD_GIL_USED'
376          else:
377              mod['gil_used'] = 'Py_MOD_GIL_NOT_USED'
378      return postlist

381  def buildmodules(lst):
382      cfuncs.buildcfuncs()
383      outmess('Building modules...\n')
384      modules, mnames, isusedby = [], [], {}
385      for item in lst:
386          if '__user__' in item['name']:
387              cb_rules.buildcallbacks(item)
388          else:
389              if 'use' in item:
390                  for u in item['use'].keys():
391                      if u not in isusedby:
392                          isusedby[u] = []
393                      isusedby[u].append(item['name'])
394              modules.append(item)
395              mnames.append(item['name'])
396      ret = {}
397      for module, name in zip(modules, mnames):
398          if name in isusedby:
399              outmess('\tSkipping module "%s" which is used by %s.\n' % (
400                  name, ','.join('"%s"' % s for s in isusedby[name])))
401          else:
402              um = []
403              if 'use' in module:
404                  for u in module['use'].keys():
405                      if u in isusedby and u in mnames:
406                          um.append(modules[mnames.index(u)])
407                      else:
408                          outmess(
409                              f'\tModule "{name}" uses nonexisting "{u}" '
410                              'which will be ignored.\n')
411              ret[name] = {}
412              dict_append(ret[name], rules.buildmodule(module, um))
413      return ret

416  def dict_append(d_out, d_in):
417      for (k, v) in d_in.items():
418          if k not in d_out:
419              d_out[k] = []
420          if isinstance(v, list):
421              d_out[k] = d_out[k] + v
422          else:
423              d_out[k].append(v)

426  def run_main(comline_list):
427      """ 
428      Equivalent to running:: 
429   
430          f2py <args> 
431   
432      where ``<args>=string.join(<list>,' ')``, but in Python.  Unless 
433      ``-h`` is used, this function returns a dictionary containing 
434      information on generated modules and their dependencies on source 
435      files. 
436   
437      You cannot build extension modules with this function, that is, 
438      using ``-c`` is not allowed. Use the ``compile`` command instead. 
439   
440      Examples 
441      -------- 
442      The command ``f2py -m scalar scalar.f`` can be executed from Python as 
443      follows. 
444   
445      .. literalinclude:: ../../source/f2py/code/results/run_main_session.dat 
446          :language: python 
447   
448      """
449      crackfortran.reset_global_f2py_vars()
450      f2pydir = os.path.dirname(os.path.abspath(cfuncs.__file__))
451      fobjhsrc = os.path.join(f2pydir, 'src', 'fortranobject.h')
452      fobjcsrc = os.path.join(f2pydir, 'src', 'fortranobject.c')
453      # gh-22819 -- begin
454      parser = make_f2py_compile_parser()
455      args, comline_list = parser.parse_known_args(comline_list)
456      pyf_files, _ = filter_files("", "[.]pyf([.]src|)", comline_list)
457      # Checks that no existing modulename is defined in a pyf file
458      # TODO: Remove all this when scaninputline is replaced
459      if args.module_name:
460          if "-h" in comline_list:
461              modname = (
462                  args.module_name
463              )  # Directly use from args when -h is present
464          else:
465              modname = validate_modulename(
466                  pyf_files, args.module_name
467              )  # Validate modname when -h is not present
468          comline_list += ['-m', modname]  # needed for the rest of scaninputline
469      # gh-22819 -- end
470      files, options = scaninputline(comline_list)
471      auxfuncs.options = options
472      capi_maps.load_f2cmap_file(options['f2cmap_file'])
473      postlist = callcrackfortran(files, options)
474      isusedby = {}
475      for plist in postlist:
476          if 'use' in plist:
477              for u in plist['use'].keys():
478                  if u not in isusedby:
479                      isusedby[u] = []
480                  isusedby[u].append(plist['name'])
481      for plist in postlist:
482          if plist['block'] == 'python module' and '__user__' in plist['name']:
483              if plist['name'] in isusedby:
484                  # if not quiet:
485                  outmess(
486                      f'Skipping Makefile build for module "{plist["name"]}" '
487                      'which is used by {}\n'.format(
488                          ','.join(f'"{s}"' for s in isusedby[plist['name']])))
489      if 'signsfile' in options:
490          if options['verbose'] > 1:
491              outmess(
492                  'Stopping. Edit the signature file and then run f2py on the signature file: ')
493              outmess('%s %s\n' %
494                      (os.path.basename(sys.argv[0]), options['signsfile']))
495          return
496      for plist in postlist:
497          if plist['block'] != 'python module':
498              if 'python module' not in options:
499                  errmess(
500                      'Tip: If your original code is Fortran source then you must use -m option.\n')
501              raise TypeError('All blocks must be python module blocks but got %s' % (
502                  repr(plist['block'])))
503      auxfuncs.debugoptions = options['debug']
504      f90mod_rules.options = options
505      auxfuncs.wrapfuncs = options['wrapfuncs']
506  
507      ret = buildmodules(postlist)
508  
509      for mn in ret.keys():
510          dict_append(ret[mn], {'csrc': fobjcsrc, 'h': fobjhsrc})
511      return ret

514  def filter_files(prefix, suffix, files, remove_prefix=None):
515      """ 
516      Filter files by prefix and suffix. 
517      """
518      filtered, rest = [], []
519      match = re.compile(prefix + r'.*' + suffix + r'\Z').match
520      if remove_prefix:
521          ind = len(prefix)
522      else:
523          ind = 0
524      for file in [x.strip() for x in files]:
525          if match(file):
526              filtered.append(file[ind:])
527          else:
528              rest.append(file)
529      return filtered, rest

537  class CombineIncludePaths(argparse.Action):
538      def __call__(self, parser, namespace, values, option_string=None):
539          include_paths_set = set(getattr(namespace, 'include_paths', []) or [])
540          if option_string == "--include_paths":
541              outmess("Use --include-paths or -I instead of --include_paths which will be removed")
542          if option_string == "--include-paths" or option_string == "--include_paths":
543              include_paths_set.update(values.split(':'))
544          else:
545              include_paths_set.add(values)
546          namespace.include_paths = list(include_paths_set)

548  def f2py_parser():
549      parser = argparse.ArgumentParser(add_help=False)
550      parser.add_argument("-I", dest="include_paths", action=CombineIncludePaths)
551      parser.add_argument("--include-paths", dest="include_paths", action=CombineIncludePaths)
552      parser.add_argument("--include_paths", dest="include_paths", action=CombineIncludePaths)
553      parser.add_argument("--freethreading-compatible", dest="ftcompat", action=argparse.BooleanOptionalAction)
554      return parser

556  def get_newer_options(iline):
557      iline = (' '.join(iline)).split()
558      parser = f2py_parser()
559      args, remain = parser.parse_known_args(iline)
560      ipaths = args.include_paths

562          ipaths = []
563      return ipaths, args.ftcompat, remain

572  def preparse_sysargv():
573      # To keep backwards bug compatibility, newer flags are handled by argparse,
574      # and `sys.argv` is passed to the rest of `f2py` as is.
575      parser = make_f2py_compile_parser()
576  
577      args, remaining_argv = parser.parse_known_args()
578      sys.argv = [sys.argv[0]] + remaining_argv
579  
580      backend_key = args.backend
581      if MESON_ONLY_VER and backend_key == 'distutils':
582          outmess("Cannot use distutils backend with Python>=3.12,"
583                  " using meson backend instead.\n")
584          backend_key = "meson"
585  
586      return {
587          "dependencies": args.dependencies or [],
588          "backend": backend_key,
589          "modulename": args.module_name,
590      }

592  def run_compile():
593      """ 
594      Do it all in one call! 
595      """
596      import tempfile
597  
598      # Collect dependency flags, preprocess sys.argv
599      argy = preparse_sysargv()
600      modulename = argy["modulename"]
601      if modulename is None:
602          modulename = 'untitled'
603      dependencies = argy["dependencies"]
604      backend_key = argy["backend"]
605      build_backend = f2py_build_generator(backend_key)
606  
607      i = sys.argv.index('-c')
608      del sys.argv[i]
609  
610      remove_build_dir = 0
611      try:
612          i = sys.argv.index('--build-dir')
613      except ValueError:
614          i = None
615      if i is not None:
616          build_dir = sys.argv[i + 1]
617          del sys.argv[i + 1]
618          del sys.argv[i]
619      else:
620          remove_build_dir = 1
621          build_dir = tempfile.mkdtemp()
622  
623      _reg1 = re.compile(r'--link-')
624      sysinfo_flags = [_m for _m in sys.argv[1:] if _reg1.match(_m)]
625      sys.argv = [_m for _m in sys.argv if _m not in sysinfo_flags]
626      if sysinfo_flags:
627          sysinfo_flags = [f[7:] for f in sysinfo_flags]
628  
629      _reg2 = re.compile(
630          r'--((no-|)(wrap-functions|lower|freethreading-compatible)|debug-capi|quiet|skip-empty-wrappers)|-include')
631      f2py_flags = [_m for _m in sys.argv[1:] if _reg2.match(_m)]
632      sys.argv = [_m for _m in sys.argv if _m not in f2py_flags]
633      f2py_flags2 = []
634      fl = 0
635      for a in sys.argv[1:]:
636          if a in ['only:', 'skip:']:
637              fl = 1
638          elif a == ':':
639              fl = 0
640          if fl or a == ':':
641              f2py_flags2.append(a)
642      if f2py_flags2 and f2py_flags2[-1] != ':':
643          f2py_flags2.append(':')
644      f2py_flags.extend(f2py_flags2)
645      sys.argv = [_m for _m in sys.argv if _m not in f2py_flags2]
646      _reg3 = re.compile(
647          r'--((f(90)?compiler(-exec|)|compiler)=|help-compiler)')
648      flib_flags = [_m for _m in sys.argv[1:] if _reg3.match(_m)]
649      sys.argv = [_m for _m in sys.argv if _m not in flib_flags]
650      # TODO: Once distutils is dropped completely, i.e. min_ver >= 3.12, unify into --fflags
651      reg_f77_f90_flags = re.compile(r'--f(77|90)flags=')
652      reg_distutils_flags = re.compile(r'--((f(77|90)exec|opt|arch)=|(debug|noopt|noarch|help-fcompiler))')
653      fc_flags = [_m for _m in sys.argv[1:] if reg_f77_f90_flags.match(_m)]
654      distutils_flags = [_m for _m in sys.argv[1:] if reg_distutils_flags.match(_m)]
655      if not (MESON_ONLY_VER or backend_key == 'meson'):
656          fc_flags.extend(distutils_flags)
657      sys.argv = [_m for _m in sys.argv if _m not in (fc_flags + distutils_flags)]
658  
659      del_list = []
660      for s in flib_flags:
661          v = '--fcompiler='
662          if s[:len(v)] == v:
663              if MESON_ONLY_VER or backend_key == 'meson':
664                  outmess(
665                      "--fcompiler cannot be used with meson,"
666                      "set compiler with the FC environment variable\n"
667                      )
668              else:
669                  from numpy.distutils import fcompiler
670                  fcompiler.load_all_fcompiler_classes()
671                  allowed_keys = list(fcompiler.fcompiler_class.keys())
672                  nv = ov = s[len(v):].lower()
673                  if ov not in allowed_keys:
674                      vmap = {}  # XXX
675                      try:
676                          nv = vmap[ov]
677                      except KeyError:
678                          if ov not in vmap.values():
679                              print('Unknown vendor: "%s"' % (s[len(v):]))
680                      nv = ov
681                  i = flib_flags.index(s)
682                  flib_flags[i] = '--fcompiler=' + nv
683                  continue
684      for s in del_list:
685          i = flib_flags.index(s)
686          del flib_flags[i]
687      assert len(flib_flags) <= 2, repr(flib_flags)
688  
689      _reg5 = re.compile(r'--(verbose)')
690      setup_flags = [_m for _m in sys.argv[1:] if _reg5.match(_m)]
691      sys.argv = [_m for _m in sys.argv if _m not in setup_flags]
692  
693      if '--quiet' in f2py_flags:
694          setup_flags.append('--quiet')
695  
696      # Ugly filter to remove everything but sources
697      sources = sys.argv[1:]
698      f2cmapopt = '--f2cmap'
699      if f2cmapopt in sys.argv:
700          i = sys.argv.index(f2cmapopt)
701          f2py_flags.extend(sys.argv[i:i + 2])
702          del sys.argv[i + 1], sys.argv[i]
703          sources = sys.argv[1:]
704  
705      pyf_files, _sources = filter_files("", "[.]pyf([.]src|)", sources)
706      sources = pyf_files + _sources
707      modulename = validate_modulename(pyf_files, modulename)
708      extra_objects, sources = filter_files('', '[.](o|a|so|dylib)', sources)
709      library_dirs, sources = filter_files('-L', '', sources, remove_prefix=1)
710      libraries, sources = filter_files('-l', '', sources, remove_prefix=1)
711      undef_macros, sources = filter_files('-U', '', sources, remove_prefix=1)
712      define_macros, sources = filter_files('-D', '', sources, remove_prefix=1)
713      for i in range(len(define_macros)):
714          name_value = define_macros[i].split('=', 1)
715          if len(name_value) == 1:
716              name_value.append(None)
717          if len(name_value) == 2:
718              define_macros[i] = tuple(name_value)
719          else:
720              print('Invalid use of -D:', name_value)
721  
722      # Construct wrappers / signatures / things
723      if backend_key == 'meson':
724          if not pyf_files:
725              outmess('Using meson backend\nWill pass --lower to f2py\nSee https://numpy.org/doc/stable/f2py/buildtools/meson.html\n')
726              f2py_flags.append('--lower')
727              run_main(f" {' '.join(f2py_flags)} -m {modulename} {' '.join(sources)}".split())
728          else:
729              run_main(f" {' '.join(f2py_flags)} {' '.join(pyf_files)}".split())
730  
731      # Order matters here, includes are needed for run_main above
732      include_dirs, _, sources = get_newer_options(sources)
733      # Now use the builder
734      builder = build_backend(
735          modulename,
736          sources,
737          extra_objects,
738          build_dir,
739          include_dirs,
740          library_dirs,
741          libraries,
742          define_macros,
743          undef_macros,
744          f2py_flags,
745          sysinfo_flags,
746          fc_flags,
747          flib_flags,
748          setup_flags,
749          remove_build_dir,
750          {"dependencies": dependencies},
751      )
752  
753      builder.compile()
754

756  def validate_modulename(pyf_files, modulename='untitled'):
757      if len(pyf_files) > 1:
758          raise ValueError("Only one .pyf file per call")
759      if pyf_files:
760          pyff = pyf_files[0]
761          pyf_modname = auxfuncs.get_f2py_modulename(pyff)
762          if modulename != pyf_modname:
763              outmess(
764                  f"Ignoring -m {modulename}.\n"
765                  f"{pyff} defines {pyf_modname} to be the modulename.\n"
766              )
767              modulename = pyf_modname
768      return modulename

771      if '--help-link' in sys.argv[1:]:
772          sys.argv.remove('--help-link')
773          if MESON_ONLY_VER:
774              outmess("Use --dep for meson builds\n")
775          else:
776              from numpy.distutils.system_info import show_all
777              show_all()
778          return
779  
780      if '-c' in sys.argv[1:]:
781          run_compile()
782      else:
783          run_main(sys.argv[1:])
