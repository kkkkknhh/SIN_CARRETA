60           return res

155      def multi_fcn(self, B, x):
156          if (x < 0.0).any():
157              raise OdrStop
158          theta = pi*B[3]/2.
159          ctheta = np.cos(theta)
160          stheta = np.sin(theta)
161          omega = np.power(2.*pi*x*np.exp(-B[2]), B[3])
162          phi = np.arctan2((omega*stheta), (1.0 + omega*ctheta))
163          r = (B[0] - B[1]) * np.power(np.sqrt(np.power(1.0 + omega*ctheta, 2) +
164               np.power(omega*stheta, 2)), -B[4])
165          ret = np.vstack([B[1] + r*np.cos(B[4]*phi),
166                           r*np.sin(B[4]*phi)])
167          return ret

395          def func(beta, x):
396              y0 = beta[0] + beta[1] * x[0, :] + beta[2] * x[1, :]
397              y1 = beta[3] + beta[4] * x[0, :] + beta[5] * x[1, :]

407              dy1 = np.array([zeros, zeros, zeros, ones, x[0, :], x[1, :]])

412              nr_meas = np.shape(x)[1]

417              return np.stack((dy0, dy1))
418  
419          # do measurements with errors in independent and dependent variables

422          x_true = np.array([x0_true, x1_true])

424          y_true = func(beta_true, x_true)

427          y_meas = y_true + y_error
428  
429          # estimate model's parameters
