16   ROOT_SCALAR_METHODS = ['bisect', 'brentq', 'brenth', 'ridder', 'toms748',
17                          'newton', 'secant', 'halley']

20   class MemoizeDer:
21       """Decorator that caches the value and derivative(s) of function each 
22       time it is called. 
23    
24       This is a simplistic memoizer that calls and caches a single value 
25       of ``f(x, *args)``. 
26       It assumes that `args` does not change between invocations. 
27       It supports the use case of a root-finder where `args` is fixed, 
28       `x` changes, and only rarely, if at all, does x assume the same value 
29       more than once."""
30       def __init__(self, fun):
31           self.fun = fun
32           self.vals = None
33           self.x = None
34           self.n_calls = 0
35   
36       def __call__(self, x, *args):
37           r"""Calculate f or use cached value if available"""
38           # Derivative may be requested before the function itself, always check
39           if self.vals is None or x != self.x:
40               fg = self.fun(x, *args)
41               self.x = x
42               self.n_calls += 1
43               self.vals = fg[:]
44           return self.vals[0]
45   
46       def fprime(self, x, *args):
47           r"""Calculate f' or use a cached value if available"""
48           if self.vals is None or x != self.x:
49               self(x, *args)
50           return self.vals[1]
51   
52       def fprime2(self, x, *args):
53           r"""Calculate f'' or use a cached value if available"""
54           if self.vals is None or x != self.x:
55               self(x, *args)
56           return self.vals[2]
57   
58       def ncalls(self):
59           return self.n_calls
60

62   def root_scalar(f, args=(), method=None, bracket=None,
63                   fprime=None, fprime2=None,
64                   x0=None, x1=None,
65                   xtol=None, rtol=None, maxiter=None,
66                   options=None):
67       """ 
68       Find a root of a scalar function. 
69    
70       Parameters 
71       ---------- 
72       f : callable 
73           A function to find a root of. 
74    
75           Suppose the callable has signature ``f0(x, *my_args, **my_kwargs)``, where 
76           ``my_args`` and ``my_kwargs`` are required positional and keyword arguments. 
77           Rather than passing ``f0`` as the callable, wrap it to accept 
78           only ``x``; e.g., pass ``fun=lambda x: f0(x, *my_args, **my_kwargs)`` as the 
79           callable, where ``my_args`` (tuple) and ``my_kwargs`` (dict) have been 
80           gathered before invoking this function. 
81       args : tuple, optional 
82           Extra arguments passed to the objective function and its derivative(s). 
83       method : str, optional 
84           Type of solver.  Should be one of 
85    
86           - 'bisect'    :ref:`(see here) <optimize.root_scalar-bisect>` 
87           - 'brentq'    :ref:`(see here) <optimize.root_scalar-brentq>` 
88           - 'brenth'    :ref:`(see here) <optimize.root_scalar-brenth>` 
89           - 'ridder'    :ref:`(see here) <optimize.root_scalar-ridder>` 
90           - 'toms748'    :ref:`(see here) <optimize.root_scalar-toms748>` 
91           - 'newton'    :ref:`(see here) <optimize.root_scalar-newton>` 
92           - 'secant'    :ref:`(see here) <optimize.root_scalar-secant>` 
93           - 'halley'    :ref:`(see here) <optimize.root_scalar-halley>` 
94    
95       bracket: A sequence of 2 floats, optional 
96           An interval bracketing a root.  ``f(x, *args)`` must have different 
97           signs at the two endpoints. 
98       x0 : float, optional 
99           Initial guess. 
100      x1 : float, optional 
101          A second guess. 
102      fprime : bool or callable, optional 
103          If `fprime` is a boolean and is True, `f` is assumed to return the 
104          value of the objective function and of the derivative. 
105          `fprime` can also be a callable returning the derivative of `f`. In 
106          this case, it must accept the same arguments as `f`. 
107      fprime2 : bool or callable, optional 
108          If `fprime2` is a boolean and is True, `f` is assumed to return the 
109          value of the objective function and of the 
110          first and second derivatives. 
111          `fprime2` can also be a callable returning the second derivative of `f`. 
112          In this case, it must accept the same arguments as `f`. 
113      xtol : float, optional 
114          Tolerance (absolute) for termination. 
115      rtol : float, optional 
116          Tolerance (relative) for termination. 
117      maxiter : int, optional 
118          Maximum number of iterations. 
119      options : dict, optional 
120          A dictionary of solver options. E.g., ``k``, see 
121          :obj:`show_options()` for details. 
122   
123      Returns 
124      ------- 
125      sol : RootResults 
126          The solution represented as a ``RootResults`` object. 
127          Important attributes are: ``root`` the solution , ``converged`` a 
128          boolean flag indicating if the algorithm exited successfully and 
129          ``flag`` which describes the cause of the termination. See 
130          `RootResults` for a description of other attributes. 
131   
132      See also 
133      -------- 
134      show_options : Additional options accepted by the solvers 
135      root : Find a root of a vector function. 
136   
137      Notes 
138      ----- 
139      This section describes the available solvers that can be selected by the 
140      'method' parameter. 
141   
142      The default is to use the best method available for the situation 
143      presented. 
144      If a bracket is provided, it may use one of the bracketing methods. 
145      If a derivative and an initial value are specified, it may 
146      select one of the derivative-based methods. 
147      If no method is judged applicable, it will raise an Exception. 
148   
149      Arguments for each method are as follows (x=required, o=optional). 
150   
151      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
152      |                    method                     | f | args | bracket | x0 | x1 | fprime | fprime2 | xtol | rtol | maxiter | options | 
153      +===============================================+===+======+=========+====+====+========+=========+======+======+=========+=========+ 
154      | :ref:`bisect <optimize.root_scalar-bisect>`   | x |  o   |    x    |    |    |        |         |  o   |  o   |    o    |   o     | 
155      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
156      | :ref:`brentq <optimize.root_scalar-brentq>`   | x |  o   |    x    |    |    |        |         |  o   |  o   |    o    |   o     | 
157      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
158      | :ref:`brenth <optimize.root_scalar-brenth>`   | x |  o   |    x    |    |    |        |         |  o   |  o   |    o    |   o     | 
159      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
160      | :ref:`ridder <optimize.root_scalar-ridder>`   | x |  o   |    x    |    |    |        |         |  o   |  o   |    o    |   o     | 
161      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
162      | :ref:`toms748 <optimize.root_scalar-toms748>` | x |  o   |    x    |    |    |        |         |  o   |  o   |    o    |   o     | 
163      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
164      | :ref:`secant <optimize.root_scalar-secant>`   | x |  o   |         | x  | o  |        |         |  o   |  o   |    o    |   o     | 
165      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
166      | :ref:`newton <optimize.root_scalar-newton>`   | x |  o   |         | x  |    |   o    |         |  o   |  o   |    o    |   o     | 
167      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
168      | :ref:`halley <optimize.root_scalar-halley>`   | x |  o   |         | x  |    |   x    |    x    |  o   |  o   |    o    |   o     | 
169      +-----------------------------------------------+---+------+---------+----+----+--------+---------+------+------+---------+---------+ 
170   
171      Examples 
172      -------- 
173   
174      Find the root of a simple cubic 
175   
176      >>> from scipy import optimize 
177      >>> def f(x): 
178      ...     return (x**3 - 1)  # only one real root at x = 1 
179   
180      >>> def fprime(x): 
181      ...     return 3*x**2 
182   
183      The `brentq` method takes as input a bracket 
184   
185      >>> sol = optimize.root_scalar(f, bracket=[0, 3], method='brentq') 
186      >>> sol.root, sol.iterations, sol.function_calls 
187      (1.0, 10, 11) 
188   
189      The `newton` method takes as input a single point and uses the 
190      derivative(s). 
191   
192      >>> sol = optimize.root_scalar(f, x0=0.2, fprime=fprime, method='newton') 
193      >>> sol.root, sol.iterations, sol.function_calls 
194      (1.0, 11, 22) 
195   
196      The function can provide the value and derivative(s) in a single call. 
197   
198      >>> def f_p_pp(x): 
199      ...     return (x**3 - 1), 3*x**2, 6*x 
200   
201      >>> sol = optimize.root_scalar( 
202      ...     f_p_pp, x0=0.2, fprime=True, method='newton' 
203      ... ) 
204      >>> sol.root, sol.iterations, sol.function_calls 
205      (1.0, 11, 11) 
206   
207      >>> sol = optimize.root_scalar( 
208      ...     f_p_pp, x0=0.2, fprime=True, fprime2=True, method='halley' 
209      ... ) 
210      >>> sol.root, sol.iterations, sol.function_calls 
211      (1.0, 7, 8) 
212   
213   
214      """  # noqa: E501
215      if not isinstance(args, tuple):
216          args = (args,)
217  
218      if options is None:
219          options = {}
220  
221      # fun also returns the derivative(s)
222      is_memoized = False
223      if fprime2 is not None and not callable(fprime2):
224          if bool(fprime2):
225              f = MemoizeDer(f)
226              is_memoized = True
227              fprime2 = f.fprime2
228              fprime = f.fprime
229          else:
230              fprime2 = None
231      if fprime is not None and not callable(fprime):
232          if bool(fprime):
233              f = MemoizeDer(f)
234              is_memoized = True
235              fprime = f.fprime
236          else:
237              fprime = None
238  
239      # respect solver-specific default tolerances - only pass in if actually set
240      kwargs = {}
241      for k in ['xtol', 'rtol', 'maxiter']:
242          v = locals().get(k)
243          if v is not None:
244              kwargs[k] = v
245  
246      # Set any solver-specific options
247      if options:
248          kwargs.update(options)
249      # Always request full_output from the underlying method as _root_scalar
250      # always returns a RootResults object
251      kwargs.update(full_output=True, disp=False)
252  
253      # Pick a method if not specified.
254      # Use the "best" method available for the situation.
255      if not method:
256          if bracket is not None:
257              method = 'brentq'
258          elif x0 is not None:
259              if fprime:
260                  if fprime2:
261                      method = 'halley'
262                  else:
263                      method = 'newton'
264              elif x1 is not None:
265                  method = 'secant'
266              else:
267                  method = 'newton'
268      if not method:
269          raise ValueError('Unable to select a solver as neither bracket '
270                           'nor starting point provided.')
271  
272      meth = method.lower()
273      map2underlying = {'halley': 'newton', 'secant': 'newton'}
274  
275      try:
276          methodc = getattr(optzeros, map2underlying.get(meth, meth))
277      except AttributeError as e:
278          raise ValueError(f'Unknown solver {meth}') from e
279  
280      if meth in ['bisect', 'ridder', 'brentq', 'brenth', 'toms748']:
281          if not isinstance(bracket, (list, tuple, np.ndarray)):
282              raise ValueError(f'Bracket needed for {method}')
283  
284          a, b = bracket[:2]
285          try:
286              r, sol = methodc(f, a, b, args=args, **kwargs)
287          except ValueError as e:
288              # gh-17622 fixed some bugs in low-level solvers by raising an error
289              # (rather than returning incorrect results) when the callable
290              # returns a NaN. It did so by wrapping the callable rather than
291              # modifying compiled code, so the iteration count is not available.
292              if hasattr(e, "_x"):
293                  sol = optzeros.RootResults(root=e._x,
294                                             iterations=np.nan,
295                                             function_calls=e._function_calls,
296                                             flag=str(e), method=method)
297              else:
298                  raise
299  
300      elif meth in ['secant']:
301          if x0 is None:
302              raise ValueError(f'x0 must not be None for {method}')
303          if 'xtol' in kwargs:
304              kwargs['tol'] = kwargs.pop('xtol')
305          r, sol = methodc(f, x0, args=args, fprime=None, fprime2=None,
306                           x1=x1, **kwargs)
307      elif meth in ['newton']:
308          if x0 is None:
309              raise ValueError(f'x0 must not be None for {method}')
310          if not fprime:
311              # approximate fprime with finite differences
312  
313              def fprime(x, *args):
314                  # `root_scalar` doesn't actually seem to support vectorized
315                  # use of `newton`. In that case, `approx_derivative` will
316                  # always get scalar input. Nonetheless, it always returns an
317                  # array, so we extract the element to produce scalar output.
318                  # Similarly, `approx_derivative` always passes array input, so
319                  # we extract the element to ensure the user's function gets
320                  # scalar input.
321                  def f_wrapped(x, *args):
322                      return f(x[0], *args)
323                  return approx_derivative(f_wrapped, x, method='2-point', args=args)[0]
324  
325          if 'xtol' in kwargs:
326              kwargs['tol'] = kwargs.pop('xtol')
327          r, sol = methodc(f, x0, args=args, fprime=fprime, fprime2=None,
328                           **kwargs)
329      elif meth in ['halley']:
330          if x0 is None:
331              raise ValueError(f'x0 must not be None for {method}')
332          if not fprime:
333              raise ValueError(f'fprime must be specified for {method}')
334          if not fprime2:
335              raise ValueError(f'fprime2 must be specified for {method}')
336          if 'xtol' in kwargs:
337              kwargs['tol'] = kwargs.pop('xtol')
338          r, sol = methodc(f, x0, args=args, fprime=fprime, fprime2=fprime2, **kwargs)
339      else:
340          raise ValueError(f'Unknown solver {method}')
341  
342      if is_memoized:
343          # Replace the function_calls count with the memoized count.
344          # Avoids double and triple-counting.
345          n_calls = f.n_calls
346          sol.function_calls = n_calls
347  
348      return sol
