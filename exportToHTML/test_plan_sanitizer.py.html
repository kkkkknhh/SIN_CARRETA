34       @staticmethod
35       def test_sanitize_problematic_characters():
36           """Test sanitization of problematic characters."""
37           test_cases = [
38               # Slashes
39               ("Plan/Meta/Objetivo", "Plan - Meta - Objetivo"),
40               ("Plan\\Desarrollo\\2024", "Plan - Desarrollo - 2024"),
41               # Colons
42               ("Plan: Desarrollo Social", "Plan - Desarrollo Social"),
43               ("Proyecto: Meta 2024", "Proyecto - Meta 2024"),
44               # Asterisks and wildcards
45               ("Plan*Meta*2024", "PlanMeta2024"),
46               ("Proyecto?Meta", "ProyectoMeta"),
47               # Quotes
48               ('Plan "Importante"', "Plan 'Importante'"),
49               ("Plan 'Meta'", "Plan 'Meta'"),
50               # Angle brackets
51               ("Plan <urgente>", "Plan (urgente)"),
52               ("Meta <2024>", "Meta (2024)"),
53               # Pipes
54               ("Plan|Meta|2024", "Plan - Meta - 2024"),
55               # Combined problematic characters
56               ("Plan: Meta/Objetivo*2024?", "Plan - Meta - Objetivo2024"),
57               ('<Plan>:"Meta"/Objetivo*', "(Plan) - 'Meta' - Objetivo"),
58           ]
59   
60           for input_name, expected in test_cases:
61               result = PlanSanitizer.sanitize_plan_name(input_name)
62               assert result == expected, (
63                   f"Input: {input_name}, Expected: {expected}, Got: {result}"
64               )

117      @staticmethod
118      def test_sanitize_edge_cases():
119          """Test edge cases."""
120          # Empty or None names
121          assert PlanSanitizer.sanitize_plan_name("") == "plan_sin_nombre"
122          assert PlanSanitizer.sanitize_plan_name("   ") == "plan_sin_nombre"
123          assert PlanSanitizer.sanitize_plan_name(None) == "plan_sin_nombre"
124  
125          # Only problematic characters
126          assert PlanSanitizer.sanitize_plan_name(
127              "///**???") == "plan_sin_nombre"
128          assert PlanSanitizer.sanitize_plan_name("<<<>>>") == "()"
129  
130          # Leading/trailing problematic chars
131          assert PlanSanitizer.sanitize_plan_name(
132              "...Plan Meta...") == "Plan Meta"
133          assert PlanSanitizer.sanitize_plan_name("---Plan---") == "Plan"

421      @staticmethod
422      def test_extreme_plan_names():
423          """Test with extremely problematic plan names."""
424          extreme_cases = [
425              # All forbidden characters
426              '/<>:"|?*\\',
427              # Control characters
428              "\x00\x01\x02Plan\x03\x04\x05",
429              # Reserved names with extension
430              "CON.txt",
431              "PRN.doc",
432              # Mixed extreme case
433              '<Plan>:"Meta"/Obj*2024?\x00\x01',
434              # Unicode and special chars
435              "Plan 🚀 Meta ⭐ 2024 💫",
436              # Very long with problematic chars
437              ("Plan: Desarrollo/Nacional*Integral?" + "A" * 300),
438          ]
439  
440          for extreme_name in extreme_cases:
441              result = PlanSanitizer.sanitize_plan_name(extreme_name)
442  
443              # Should not contain any forbidden characters
444              for forbidden_char in PlanSanitizer.INVALID_CHARS:
445                  assert forbidden_char not in result, (
446                      f"Found forbidden char in: {result}"
447                  )
448  
449              # Should not be empty
450              assert len(result) > 0
451  
452              # Should be safe for directory creation
453              with tempfile.TemporaryDirectory() as temp_dir:
454                  try:
455                      safe_path = os.path.join(temp_dir, result)
456                      os.makedirs(safe_path, exist_ok=True)
457                      assert os.path.exists(safe_path)
458                  except (OSError, ValueError) as e:
459                      pytest.fail(
460                          f"Failed to create directory with name '{result}': {e}")
