16   template <> 
17   struct AtomicFPOp<at::Half> {
18     template <typename func_t> 
19     inline __device__ at::Half operator() (at::Half *address, at::Half val, const func_t& func) {
20       unsigned int * address_as_ui =
21         (unsigned int *) ((char *)address - ((size_t)address & 2));
22       unsigned int old = *address_as_ui;
23       unsigned int assumed;
24   
25       at::Half hsum;
26       do {
27         assumed = old;
28         hsum.x = (size_t)address & 2 ? (old >> 16) : (old & 0xffff);
29         hsum = func(hsum, val);
30         old = (size_t)address & 2 ? (old & 0xffff) | (hsum.x << 16) : (old & 0xffff0000) | hsum.x;
31         old = atomicCAS(address_as_ui, assumed, old);
32       } while (assumed != old);
33       hsum.x = (size_t)address & 2 ? (old >> 16) : (old & 0xffff);
34       return hsum;
35     }
36   };

60   template <> 
61   struct AtomicFPOp<double> {
62     template <typename func_t> 
63     inline __device__ double operator() (double * address, double val, const func_t& func) {
64       unsigned long long int* address_as_ull = (unsigned long long int*)address;
65       unsigned long long int old = *address_as_ull;
66       unsigned long long int assumed;
67   
68       do {
69         assumed = old;
70         old = atomicCAS(address_as_ull, assumed, func(val, assumed));
71         // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN) 
72       } while (assumed != old);
73   
74       return __longlong_as_double(old);
75     }
76   };

82   template<typename T>                                                                                                   \ 
83   struct Atomic##NAME##IntegerImpl<T, 1> {                                                                               \ 
84     template <typename func_t>                                                                                           \ 
85     inline __device__ void operator()(T *address, T val, const func_t& func) {                                           \ 
86       size_t offset = (size_t)address & 3;                                                                               \ 
87       uint32_t * address_as_ui = (uint32_t *)((char *)address - offset);                                                 \ 
88       uint32_t old = *address_as_ui;                                                                                     \ 
89       uint32_t shift = offset * 8;                                                                                       \ 
90       uint32_t old_byte;                                                                                                 \ 
91       uint32_t newval;                                                                                                   \ 
92       uint32_t assumed;                                                                                                  \ 
93                                                                                                                          \ 
94       do {                                                                                                               \ 
95         assumed = old;                                                                                                   \ 
96         old_byte = (old >> shift) & 0xff;                                                                                \ 
97         newval = static_cast<uint8_t>(func(val, static_cast<T>(old_byte)));                                              \ 
98         newval = (old & ~(0x000000ff << shift)) | (newval << shift);                                                     \ 
99         old = atomicCAS(address_as_ui, assumed, newval);                                                                 \ 
100      } while (assumed != old);                                                                                          \ 
101    }                                                                                                                    \ 
102  };                                                                                                                     \

319   * Due to compiler limitations, callers must opt-in to guarantee the optimized instruction.

334  /* Note [HIP unsafeAtomicAdd] 
335   * ~~~~~~~~~~~~~~~~~~~~~~~~~~ 
336   * Use unsafeAtomicAdd instead of atomicAdd for fp32 and fp64. 
337   * On HIP, atomicAdd is always correct but is a slow CAS loop. 
338   * unsafeAtomicAdd will use HW instructions and is much faster, 
339   * but the caller must guarantee the pointer is GPU memory. 
340   * If the pointer is system memory, the result is a silent no-op. 
341   * This guarantee is upheld by all PyTorch uses of unsafeAtomicAdd. 
342   * AMD HIP atomic header file is named amd_hip_atomic.h and is 
343   * under the LLVM compiler directory. 
344   */
345  #if defined(USE_ROCM)
346  inline __device__ void gpuAtomicAddNoReturn(float *address, float val) {
347  #if defined(__gfx908__)
348    atomicAddNoRet(address, val);
349  #else
350    (void)unsafeAtomicAdd(address, val);
351  #endif
352  }
353  inline __device__ void gpuAtomicAddNoReturn(double *address, double val) { (void)unsafeAtomicAdd(address, val); }
354  #else
355  inline __device__ void gpuAtomicAddNoReturn(float *address, float val) { gpuAtomicAdd(address, val); }
356  inline __device__ void gpuAtomicAddNoReturn(double *address, double val) { gpuAtomicAdd(address, val); }

390  inline __device__ float gpuAtomicMul (float * address, float val) {
391    unsigned int* address_as_ull = (unsigned int*)address;
392    unsigned int old = *address_as_ull;
393    unsigned int assumed;
394  
395    do {
396      assumed = old;
397      old = atomicCAS(address_as_ull, assumed,
398                      __float_as_int(val *
399                                     __int_as_float(assumed)));

401      // Note: uses integer comparison to avoid hang in case of NaN (since NaN != NaN) 
402    } while (assumed != old);
