88   def dup_trial_division(f, factors, K):
89       """ 
90       Determine multiplicities of factors for a univariate polynomial 
91       using trial division. 
92    
93       An error will be raised if any factor does not divide ``f``. 
94       """
95       result = []
96   
97       for factor in factors:
98           k = 0
99   
100          while True:
101              q, r = dup_div(f, factor, K)
102  
103              if not r:
104                  f, k = q, k + 1
105              else:
106                  break
107  
108          if k == 0:
109              raise RuntimeError("trial division failed")
110  
111          result.append((factor, k))
112  
113      return _sort_factors(result)

144  def dup_zz_mignotte_bound(f, K):
145      """ 
146      The Knuth-Cohen variant of Mignotte bound for 
147      univariate polynomials in ``K[x]``. 
148   
149      Examples 
150      ======== 
151   
152      >>> from sympy.polys import ring, ZZ 
153      >>> R, x = ring("x", ZZ) 
154   
155      >>> f = x**3 + 14*x**2 + 56*x + 64 
156      >>> R.dup_zz_mignotte_bound(f) 
157      152 
158   
159      By checking ``factor(f)`` we can see that max coeff is 8 
160   
161      Also consider a case that ``f`` is irreducible for example 
162      ``f = 2*x**2 + 3*x + 4``. To avoid a bug for these cases, we return the 
163      bound plus the max coefficient of ``f`` 
164   
165      >>> f = 2*x**2 + 3*x + 4 
166      >>> R.dup_zz_mignotte_bound(f) 
167      6 
168   
169      Lastly, to see the difference between the new and the old Mignotte bound 
170      consider the irreducible polynomial: 
171   
172      >>> f = 87*x**7 + 4*x**6 + 80*x**5 + 17*x**4 + 9*x**3 + 12*x**2 + 49*x + 26 
173      >>> R.dup_zz_mignotte_bound(f) 
174      744 
175   
176      The new Mignotte bound is 744 whereas the old one (SymPy 1.5.1) is 1937664. 
177   
178   
179      References 
180      ========== 
181   
182      ..[1] [Abbott13]_ 
183   
184      """
185      from sympy.functions.combinatorial.factorials import binomial
186      d = dup_degree(f)
187      delta = _ceil(d / 2)
188      delta2 = _ceil(delta / 2)
189  
190      # euclidean-norm
191      eucl_norm = K.sqrt( sum( cf**2 for cf in f ) )
192  
193      # biggest values of binomial coefficients (p. 538 of reference)
194      t1 = binomial(delta - 1, delta2)
195      t2 = binomial(delta - 1, delta2 - 1)
196  
197      lc = K.abs(dup_LC(f, K))   # leading coefficient
198      bound = t1 * eucl_norm + t2 * lc   # (p. 538 of reference)
199      bound += dup_max_norm(f, K) # add max coeff for irreducible polys
200      bound = _ceil(bound / 2) * 2   # round up to even integer
201  
202      return bound
203  
204  def dmp_zz_mignotte_bound(f, u, K):
205      """Mignotte bound for multivariate polynomials in `K[X]`. """
206      a = dmp_max_norm(f, u, K)
207      b = abs(dmp_ground_LC(f, u, K))
208      n = sum(dmp_degree_list(f, u))
209  
210      return K.sqrt(K(n + 1))*2**n*a*b

213  def dup_zz_hensel_step(m, f, g, h, s, t, K):
214      """ 
215      One step in Hensel lifting in `Z[x]`. 
216   
217      Given positive integer `m` and `Z[x]` polynomials `f`, `g`, `h`, `s` 
218      and `t` such that:: 
219   
220          f = g*h (mod m) 
221          s*g + t*h = 1 (mod m) 
222   
223          lc(f) is not a zero divisor (mod m) 
224          lc(h) = 1 
225   
226          deg(f) = deg(g) + deg(h) 
227          deg(s) < deg(h) 
228          deg(t) < deg(g) 
229   
230      returns polynomials `G`, `H`, `S` and `T`, such that:: 
231   
232          f = G*H (mod m**2) 
233          S*G + T*H = 1 (mod m**2) 
234   
235      References 
236      ========== 
237   
238      .. [1] [Gathen99]_ 
239   
240      """
241      M = m**2
242  
243      e = dup_sub_mul(f, g, h, K)
244      e = dup_trunc(e, M, K)
245  
246      q, r = dup_div(dup_mul(s, e, K), h, K)
247  
248      q = dup_trunc(q, M, K)
249      r = dup_trunc(r, M, K)
250  
251      u = dup_add(dup_mul(t, e, K), dup_mul(q, g, K), K)
252      G = dup_trunc(dup_add(g, u, K), M, K)
253      H = dup_trunc(dup_add(h, r, K), M, K)
254  
255      u = dup_add(dup_mul(s, G, K), dup_mul(t, H, K), K)
256      b = dup_trunc(dup_sub(u, [K.one], K), M, K)
257  
258      c, d = dup_div(dup_mul(s, b, K), H, K)
259  
260      c = dup_trunc(c, M, K)
261      d = dup_trunc(d, M, K)
262  
263      u = dup_add(dup_mul(t, b, K), dup_mul(c, G, K), K)
264      S = dup_trunc(dup_sub(s, d, K), M, K)
265      T = dup_trunc(dup_sub(t, u, K), M, K)
266  
267      return G, H, S, T

270  def dup_zz_hensel_lift(p, f, f_list, l, K):
271      r""" 
272      Multifactor Hensel lifting in `Z[x]`. 
273   
274      Given a prime `p`, polynomial `f` over `Z[x]` such that `lc(f)` 
275      is a unit modulo `p`, monic pair-wise coprime polynomials `f_i` 
276      over `Z[x]` satisfying:: 
277   
278          f = lc(f) f_1 ... f_r (mod p) 
279   
280      and a positive integer `l`, returns a list of monic polynomials 
281      `F_1,\ F_2,\ \dots,\ F_r` satisfying:: 
282   
283         f = lc(f) F_1 ... F_r (mod p**l) 
284   
285         F_i = f_i (mod p), i = 1..r 
286   
287      References 
288      ========== 
289   
290      .. [1] [Gathen99]_ 
291   
292      """
293      r = len(f_list)
294      lc = dup_LC(f, K)
295  
296      if r == 1:
297          F = dup_mul_ground(f, K.gcdex(lc, p**l)[0], K)
298          return [ dup_trunc(F, p**l, K) ]
299  
300      m = p
301      k = r // 2
302      d = int(_ceil(_log2(l)))
303  
304      g = gf_from_int_poly([lc], p)
305  
306      for f_i in f_list[:k]:
307          g = gf_mul(g, gf_from_int_poly(f_i, p), p, K)
308  
309      h = gf_from_int_poly(f_list[k], p)
310  
311      for f_i in f_list[k + 1:]:
312          h = gf_mul(h, gf_from_int_poly(f_i, p), p, K)
313  
314      s, t, _ = gf_gcdex(g, h, p, K)
315  
316      g = gf_to_int_poly(g, p)
317      h = gf_to_int_poly(h, p)
318      s = gf_to_int_poly(s, p)
319      t = gf_to_int_poly(t, p)
320  
321      for _ in range(1, d + 1):
322          (g, h, s, t), m = dup_zz_hensel_step(m, f, g, h, s, t, K), m**2
323  
324      return dup_zz_hensel_lift(p, g, f_list[:k], l, K) \
325          + dup_zz_hensel_lift(p, h, f_list[k:], l, K)

327  def _test_pl(fc, q, pl):
328      if q > pl // 2:
329          q = q - pl
330      if not q:
331          return True
332      return fc % q == 0

334  def dup_zz_zassenhaus(f, K):
335      """Factor primitive square-free polynomials in `Z[x]`. """
336      n = dup_degree(f)
337  
338      if n == 1:
339          return [f]
340  
341      from sympy.ntheory import isprime
342  
343      fc = f[-1]
344      A = dup_max_norm(f, K)
345      b = dup_LC(f, K)
346      B = int(abs(K.sqrt(K(n + 1))*2**n*A*b))
347      C = int((n + 1)**(2*n)*A**(2*n - 1))
348      gamma = int(_ceil(2*_log2(C)))
349      bound = int(2*gamma*_log(gamma))
350      a = []
351      # choose a prime number `p` such that `f` be square free in Z_p
352      # if there are many factors in Z_p, choose among a few different `p`
353      # the one with fewer factors
354      for px in range(3, bound + 1):
355          if not isprime(px) or b % px == 0:
356              continue
357  
358          px = K.convert(px)
359  
360          F = gf_from_int_poly(f, px)
361  
362          if not gf_sqf_p(F, px, K):
363              continue
364          fsqfx = gf_factor_sqf(F, px, K)[1]
365          a.append((px, fsqfx))
366          if len(fsqfx) < 15 or len(a) > 4:
367              break
368      p, fsqf = min(a, key=lambda x: len(x[1]))
369  
370      l = int(_ceil(_log(2*B + 1, p)))
371  
372      modular = [gf_to_int_poly(ff, p) for ff in fsqf]
373  
374      g = dup_zz_hensel_lift(p, f, modular, l, K)
375  
376      sorted_T = range(len(g))
377      T = set(sorted_T)
378      factors, s = [], 1
379      pl = p**l
380  
381      while 2*s <= len(T):
382          for S in subsets(sorted_T, s):
383              # lift the constant coefficient of the product `G` of the factors
384              # in the subset `S`; if it is does not divide `fc`, `G` does
385              # not divide the input polynomial
386  
387              if b == 1:
388                  q = 1
389                  for i in S:
390                      q = q*g[i][-1]
391                  q = q % pl
392                  if not _test_pl(fc, q, pl):
393                      continue
394              else:
395                  G = [b]
396                  for i in S:
397                      G = dup_mul(G, g[i], K)
398                  G = dup_trunc(G, pl, K)
399                  G = dup_primitive(G, K)[1]
400                  q = G[-1]
401                  if q and fc % q != 0:
402                      continue
403  
404              H = [b]
405              S = set(S)
406              T_S = T - S
407  
408              if b == 1:
409                  G = [b]
410                  for i in S:
411                      G = dup_mul(G, g[i], K)
412                  G = dup_trunc(G, pl, K)
413  
414              for i in T_S:
415                  H = dup_mul(H, g[i], K)
416  
417              H = dup_trunc(H, pl, K)
418  
419              G_norm = dup_l1_norm(G, K)
420              H_norm = dup_l1_norm(H, K)
421  
422              if G_norm*H_norm <= B:
423                  T = T_S
424                  sorted_T = [i for i in sorted_T if i not in S]
425  
426                  G = dup_primitive(G, K)[1]
427                  f = dup_primitive(H, K)[1]
428  
429                  factors.append(G)
430                  b = dup_LC(f, K)
431  
432                  break
433          else:
434              s += 1
435  
436      return factors + [f]

439  def dup_zz_irreducible_p(f, K):
440      """Test irreducibility using Eisenstein's criterion. """
441      lc = dup_LC(f, K)
442      tc = dup_TC(f, K)
443  
444      e_fc = dup_content(f[1:], K)
445  
446      if e_fc:
447          from sympy.ntheory import factorint
448          e_ff = factorint(int(e_fc))
449  
450          for p in e_ff.keys():
451              if (lc % p) and (tc % p**2):
452                  return True

455  def dup_cyclotomic_p(f, K, irreducible=False):
456      """ 
457      Efficiently test if ``f`` is a cyclotomic polynomial. 
458   
459      Examples 
460      ======== 
461   
462      >>> from sympy.polys import ring, ZZ 
463      >>> R, x = ring("x", ZZ) 
464   
465      >>> f = x**16 + x**14 - x**10 + x**8 - x**6 + x**2 + 1 
466      >>> R.dup_cyclotomic_p(f) 
467      False 
468   
469      >>> g = x**16 + x**14 - x**10 - x**8 - x**6 + x**2 + 1 
470      >>> R.dup_cyclotomic_p(g) 
471      True 
472   
473      References 
474      ========== 
475   
476      Bradford, Russell J., and James H. Davenport. "Effective tests for 
477      cyclotomic polynomials." In International Symposium on Symbolic and 
478      Algebraic Computation, pp. 244-251. Springer, Berlin, Heidelberg, 1988. 
479   
480      """
481      if K.is_QQ:
482          try:
483              K0, K = K, K.get_ring()
484              f = dup_convert(f, K0, K)
485          except CoercionFailed:
486              return False
487      elif not K.is_ZZ:
488          return False
489  
490      lc = dup_LC(f, K)
491      tc = dup_TC(f, K)
492  
493      if lc != 1 or (tc != -1 and tc != 1):
494          return False
495  
496      if not irreducible:
497          coeff, factors = dup_factor_list(f, K)
498  
499          if coeff != K.one or factors != [(f, 1)]:
500              return False
501  
502      n = dup_degree(f)
503      g, h = [], []
504  
505      for i in range(n, -1, -2):
506          g.insert(0, f[i])
507  
508      for i in range(n - 1, -1, -2):
509          h.insert(0, f[i])
510  
511      g = dup_sqr(dup_strip(g), K)
512      h = dup_sqr(dup_strip(h), K)
513  
514      F = dup_sub(g, dup_lshift(h, 1, K), K)
515  
516      if K.is_negative(dup_LC(F, K)):
517          F = dup_neg(F, K)
518  
519      if F == f:
520          return True
521  
522      g = dup_mirror(f, K)
523  
524      if K.is_negative(dup_LC(g, K)):
525          g = dup_neg(g, K)
526  
527      if F == g and dup_cyclotomic_p(g, K):
528          return True
529  
530      G = dup_sqf_part(F, K)
531  
532      if dup_sqr(G, K) == F and dup_cyclotomic_p(G, K):
533          return True
534  
535      return False

538  def dup_zz_cyclotomic_poly(n, K):
539      """Efficiently generate n-th cyclotomic polynomial. """
540      from sympy.ntheory import factorint
541      h = [K.one, -K.one]
542  
543      for p, k in factorint(n).items():
544          h = dup_quo(dup_inflate(h, p, K), h, K)
545          h = dup_inflate(h, p**(k - 1), K)
546  
547      return h

550  def _dup_cyclotomic_decompose(n, K):
551      from sympy.ntheory import factorint
552  
553      H = [[K.one, -K.one]]
554  
555      for p, k in factorint(n).items():
556          Q = [ dup_quo(dup_inflate(h, p, K), h, K) for h in H ]
557          H.extend(Q)
558  
559          for i in range(1, k):
560              Q = [ dup_inflate(q, p, K) for q in Q ]
561              H.extend(Q)
562  
563      return H

566  def dup_zz_cyclotomic_factor(f, K):
567      """ 
568      Efficiently factor polynomials `x**n - 1` and `x**n + 1` in `Z[x]`. 
569   
570      Given a univariate polynomial `f` in `Z[x]` returns a list of factors 
571      of `f`, provided that `f` is in the form `x**n - 1` or `x**n + 1` for 
572      `n >= 1`. Otherwise returns None. 
573   
574      Factorization is performed using cyclotomic decomposition of `f`, 
575      which makes this method much faster that any other direct factorization 
576      approach (e.g. Zassenhaus's). 
577   
578      References 
579      ========== 
580   
581      .. [1] [Weisstein09]_ 
582   
583      """
584      lc_f, tc_f = dup_LC(f, K), dup_TC(f, K)
585  
586      if dup_degree(f) <= 0:
587          return None
588  
589      if lc_f != 1 or tc_f not in [-1, 1]:
590          return None
591  
592      if any(bool(cf) for cf in f[1:-1]):
593          return None
594  
595      n = dup_degree(f)
596      F = _dup_cyclotomic_decompose(n, K)
597  
598      if not K.is_one(tc_f):
599          return F
600      else:
601          H = []
602  
603          for h in _dup_cyclotomic_decompose(2*n, K):
604              if h not in F:
605                  H.append(h)
606  
607          return H

610  def dup_zz_factor_sqf(f, K):
611      """Factor square-free (non-primitive) polynomials in `Z[x]`. """
612      cont, g = dup_primitive(f, K)
613  
614      n = dup_degree(g)
615  
616      if dup_LC(g, K) < 0:
617          cont, g = -cont, dup_neg(g, K)
618  
619      if n <= 0:
620          return cont, []
621      elif n == 1:
622          return cont, [g]
623  
624      if query('USE_IRREDUCIBLE_IN_FACTOR'):
625          if dup_zz_irreducible_p(g, K):
626              return cont, [g]
627  
628      factors = None
629  
630      if query('USE_CYCLOTOMIC_FACTOR'):
631          factors = dup_zz_cyclotomic_factor(g, K)
632  
633      if factors is None:
634          factors = dup_zz_zassenhaus(g, K)
635  
636      return cont, _sort_factors(factors, multiple=False)

639  def dup_zz_factor(f, K):
640      """ 
641      Factor (non square-free) polynomials in `Z[x]`. 
642   
643      Given a univariate polynomial `f` in `Z[x]` computes its complete 
644      factorization `f_1, ..., f_n` into irreducibles over integers:: 
645   
646                  f = content(f) f_1**k_1 ... f_n**k_n 
647   
648      The factorization is computed by reducing the input polynomial 
649      into a primitive square-free polynomial and factoring it using 
650      Zassenhaus algorithm. Trial division is used to recover the 
651      multiplicities of factors. 
652   
653      The result is returned as a tuple consisting of:: 
654   
655                (content(f), [(f_1, k_1), ..., (f_n, k_n)) 
656   
657      Examples 
658      ======== 
659   
660      Consider the polynomial `f = 2*x**4 - 2`:: 
661   
662          >>> from sympy.polys import ring, ZZ 
663          >>> R, x = ring("x", ZZ) 
664   
665          >>> R.dup_zz_factor(2*x**4 - 2) 
666          (2, [(x - 1, 1), (x + 1, 1), (x**2 + 1, 1)]) 
667   
668      In result we got the following factorization:: 
669   
670                   f = 2 (x - 1) (x + 1) (x**2 + 1) 
671   
672      Note that this is a complete factorization over integers, 
673      however over Gaussian integers we can factor the last term. 
674   
675      By default, polynomials `x**n - 1` and `x**n + 1` are factored 
676      using cyclotomic decomposition to speedup computations. To 
677      disable this behaviour set cyclotomic=False. 
678   
679      References 
680      ========== 
681   
682      .. [1] [Gathen99]_ 
683   
684      """
685      if GROUND_TYPES == 'flint':
686          f_flint = fmpz_poly(f[::-1])
687          cont, factors = f_flint.factor()
688          factors = [(fac.coeffs()[::-1], exp) for fac, exp in factors]
689          return cont, _sort_factors(factors)
690  
691      cont, g = dup_primitive(f, K)
692  
693      n = dup_degree(g)
694  
695      if dup_LC(g, K) < 0:

943      H = list(H)
944

991  def dmp_zz_wang(f, u, K, mod=None, seed=None):
992      r""" 
993      Factor primitive square-free polynomials in `Z[X]`. 
994   
995      Given a multivariate polynomial `f` in `Z[x_1,...,x_n]`, which is 
996      primitive and square-free in `x_1`, computes factorization of `f` into 
997      irreducibles over integers. 
998   
999      The procedure is based on Wang's Enhanced Extended Zassenhaus 
1000     algorithm. The algorithm works by viewing `f` as a univariate polynomial 
1001     in `Z[x_2,...,x_n][x_1]`, for which an evaluation mapping is computed:: 
1002  
1003                       x_2 -> a_2, ..., x_n -> a_n 
1004  
1005     where `a_i`, for `i = 2, \dots, n`, are carefully chosen integers.  The 
1006     mapping is used to transform `f` into a univariate polynomial in `Z[x_1]`, 
1007     which can be factored efficiently using Zassenhaus algorithm. The last 
1008     step is to lift univariate factors to obtain true multivariate 
1009     factors. For this purpose a parallel Hensel lifting procedure is used. 
1010  
1011     The parameter ``seed`` is passed to _randint and can be used to seed randint 
1012     (when an integer) or (for testing purposes) can be a sequence of numbers. 
1013  
1014     References 
1015     ========== 
1016  
1017     .. [1] [Wang78]_ 
1018     .. [2] [Geddes92]_ 
1019  
1020     """
1021     from sympy.ntheory import nextprime
1022 
1023     randint = _randint(seed)
1024 
1025     ct, T = dmp_zz_factor(dmp_LC(f, K), u - 1, K)
1026 
1027     b = dmp_zz_mignotte_bound(f, u, K)
1028     p = K(nextprime(b))
1029 
1030     if mod is None:
1031         if u == 1:
1032             mod = 2
1033         else:
1034             mod = 1
1035 
1036     history, configs, A, r = set(), [], [K.zero]*u, None
1037 
1038     try:
1039         cs, s, E = dmp_zz_wang_test_points(f, T, ct, A, u, K)
1040 
1041         _, H = dup_zz_factor_sqf(s, K)
1042 
1043         r = len(H)
1044 
1045         if r == 1:
1046             return [f]
1047 
1048         configs = [(s, cs, E, H, A)]
1049     except EvaluationFailed:
1050         pass
1051 
1052     eez_num_configs = query('EEZ_NUMBER_OF_CONFIGS')
1053     eez_num_tries = query('EEZ_NUMBER_OF_TRIES')
1054     eez_mod_step = query('EEZ_MODULUS_STEP')
1055 
1056     while len(configs) < eez_num_configs:
1057         for _ in range(eez_num_tries):
1058             A = [ K(randint(-mod, mod)) for _ in range(u) ]
1059 
1060             if tuple(A) not in history:
1061                 history.add(tuple(A))
1062             else:
1063                 continue
1064 
1065             try:
1066                 cs, s, E = dmp_zz_wang_test_points(f, T, ct, A, u, K)
1067             except EvaluationFailed:
1068                 continue
1069 
1070             _, H = dup_zz_factor_sqf(s, K)
1071 
1072             rr = len(H)
1073 
1074             if r is not None:
1075                 if rr != r:  # pragma: no cover
1076                     if rr < r:
1077                         configs, r = [], rr
1078                     else:
1079                         continue
1080             else:
1081                 r = rr
1082 
1083             if r == 1:
1084                 return [f]
1085 
1086             configs.append((s, cs, E, H, A))
1087 
1088             if len(configs) == eez_num_configs:
1089                 break
1090         else:
1091             mod += eez_mod_step
1092 
1093     s_norm, s_arg, i = None, 0, 0
1094 
1095     for s, _, _, _, _ in configs:
1096         _s_norm = dup_max_norm(s, K)
1097 
1098         if s_norm is not None:
1099             if _s_norm < s_norm:
1100                 s_norm = _s_norm
1101                 s_arg = i
1102         else:
1103             s_norm = _s_norm
1104 
1105         i += 1
1106 
1107     _, cs, E, H, A = configs[s_arg]
1108     orig_f = f
1109 
1110     try:
1111         f, H, LC = dmp_zz_wang_lead_coeffs(f, T, cs, E, H, A, u, K)
1112         factors = dmp_zz_wang_hensel_lifting(f, H, LC, A, p, u, K)
1113     except ExtraneousFactors:  # pragma: no cover
1114         if query('EEZ_RESTART_IF_NEEDED'):
1115             return dmp_zz_wang(orig_f, u, K, mod + 1)
1116         else:
1117             raise ExtraneousFactors(
1118                 "we need to restart algorithm with better parameters")
1119 
1120     result = []
1121 
1122     for f in factors:
1123         _, f = dmp_ground_primitive(f, u, K)
1124 
1125         if K.is_negative(dmp_ground_LC(f, u, K)):
1126             f = dmp_neg(f, u, K)
1127 
1128         result.append(f)
1129

1134     r""" 
1135     Factor (non square-free) polynomials in `Z[X]`. 
1136  
1137     Given a multivariate polynomial `f` in `Z[x]` computes its complete 
1138     factorization `f_1, \dots, f_n` into irreducibles over integers:: 
1139  
1140                  f = content(f) f_1**k_1 ... f_n**k_n 
1141  
1142     The factorization is computed by reducing the input polynomial 
1143     into a primitive square-free polynomial and factoring it using 
1144     Enhanced Extended Zassenhaus (EEZ) algorithm. Trial division 
1145     is used to recover the multiplicities of factors. 
1146  
1147     The result is returned as a tuple consisting of:: 
1148  
1149              (content(f), [(f_1, k_1), ..., (f_n, k_n)) 
1150  
1151     Consider polynomial `f = 2*(x**2 - y**2)`:: 
1152  
1153         >>> from sympy.polys import ring, ZZ 
1154         >>> R, x,y = ring("x,y", ZZ) 
1155  
1156         >>> R.dmp_zz_factor(2*x**2 - 2*y**2) 
1157         (2, [(x - y, 1), (x + y, 1)]) 
1158  
1159     In result we got the following factorization:: 
1160  
1161                     f = 2 (x - y) (x + y) 
1162  
1163     References 
1164     ========== 
1165  
1166     .. [1] [Gathen99]_ 
1167  
1168     """
1169     if not u:
1170         return dup_zz_factor(f, K)
1171 
1172     if dmp_zero_p(f, u):
1173         return K.zero, []
1174 
1175     cont, g = dmp_ground_primitive(f, u, K)
1176 
1177     if dmp_ground_LC(g, u, K) < 0:
1178         cont, g = -cont, dmp_neg(g, u, K)
1179 
1180     if all(d <= 0 for d in dmp_degree_list(g, u)):
1181         return cont, []
1182 
1183     G, g = dmp_primitive(g, u, K)
1184 
1185     factors = []
1186

1188         g = dmp_sqf_part(g, u, K)
1189         H = dmp_zz_wang(g, u, K)
1190         factors = dmp_trial_division(f, H, u, K)
1191 
1192     for g, k in dmp_zz_factor(G, u - 1, K)[1]:
1193         factors.insert(0, ([g], k))
1194 
1195     _dmp_check_degrees(f, u, factors)
1196 
1197     return cont, _sort_factors(factors)

1266 def dup_ext_factor(f, K):
1267     r"""Factor univariate polynomials over algebraic number fields. 
1268  
1269     The domain `K` must be an algebraic number field `k(a)` (see :ref:`QQ(a)`). 
1270  
1271     Examples 
1272     ======== 
1273  
1274     First define the algebraic number field `K = \mathbb{Q}(\sqrt{2})`: 
1275  
1276     >>> from sympy import QQ, sqrt 
1277     >>> from sympy.polys.factortools import dup_ext_factor 
1278     >>> K = QQ.algebraic_field(sqrt(2)) 
1279  
1280     We can now factorise the polynomial `x^2 - 2` over `K`: 
1281  
1282     >>> p = [K(1), K(0), K(-2)] # x^2 - 2 
1283     >>> p1 = [K(1), -K.unit]    # x - sqrt(2) 
1284     >>> p2 = [K(1), +K.unit]    # x + sqrt(2) 
1285     >>> dup_ext_factor(p, K) == (K.one, [(p1, 1), (p2, 1)]) 
1286     True 
1287  
1288     Usually this would be done at a higher level: 
1289  
1290     >>> from sympy import factor 
1291     >>> from sympy.abc import x 
1292     >>> factor(x**2 - 2, extension=sqrt(2)) 
1293     (x - sqrt(2))*(x + sqrt(2)) 
1294  
1295     Explanation 
1296     =========== 
1297  
1298     Uses Trager's algorithm. In particular this function is algorithm 
1299     ``alg_factor`` from [Trager76]_. 
1300  
1301     If `f` is a polynomial in `k(a)[x]` then its norm `g(x)` is a polynomial in 
1302     `k[x]`. If `g(x)` is square-free and has irreducible factors `g_1(x)`, 
1303     `g_2(x)`, `\cdots` then the irreducible factors of `f` in `k(a)[x]` are 
1304     given by `f_i(x) = \gcd(f(x), g_i(x))` where the GCD is computed in 
1305     `k(a)[x]`. 
1306  
1307     The first step in Trager's algorithm is to find an integer shift `s` so 
1308     that `f(x-sa)` has square-free norm. Then the norm is factorized in `k[x]` 
1309     and the GCD of (shifted) `f` with each factor gives the shifted factors of 
1310     `f`. At the end the shift is undone to recover the unshifted factors of `f` 
1311     in `k(a)[x]`. 
1312  
1313     The algorithm reduces the problem of factorization in `k(a)[x]` to 
1314     factorization in `k[x]` with the main additional steps being to compute the 
1315     norm (a resultant calculation in `k[x,y]`) and some polynomial GCDs in 
1316     `k(a)[x]`. 
1317  
1318     In practice in SymPy the base field `k` will be the rationals :ref:`QQ` and 
1319     this function factorizes a polynomial with coefficients in an algebraic 
1320     number field  like `\mathbb{Q}(\sqrt{2})`. 
1321  
1322     See Also 
1323     ======== 
1324  
1325     dmp_ext_factor: 
1326         Analogous function for multivariate polynomials over ``k(a)``. 
1327     dup_sqf_norm: 
1328         Subroutine ``sqfr_norm`` also from [Trager76]_. 
1329     sympy.polys.polytools.factor: 
1330         The high-level function that ultimately uses this function as needed. 
1331     """
1332     n, lc = dup_degree(f), dup_LC(f, K)
1333 
1334     f = dup_monic(f, K)
1335 
1336     if n <= 0:
1337         return lc, []
1338     if n == 1:
1339         return lc, [(f, 1)]
1340 
1341     f, F = dup_sqf_part(f, K), f
1342     s, g, r = dup_sqf_norm(f, K)
1343 
1344     factors = dup_factor_list_include(r, K.dom)
1345 
1346     if len(factors) == 1:
1347         return lc, [(f, n//dup_degree(f))]
1348 
1349     H = s*K.unit
1350 
1351     for i, (factor, _) in enumerate(factors):
1352         h = dup_convert(factor, K.dom, K)
1353         h, _, g = dup_inner_gcd(h, g, K)
1354         h = dup_shift(h, H, K)
1355         factors[i] = h
1356 
1357     factors = dup_trial_division(F, factors, K)
1358 
1359     _dup_check_degrees(F, factors)
1360 
1361     return lc, factors

1364 def dmp_ext_factor(f, u, K):
1365     r"""Factor multivariate polynomials over algebraic number fields. 
1366  
1367     The domain `K` must be an algebraic number field `k(a)` (see :ref:`QQ(a)`). 
1368  
1369     Examples 
1370     ======== 
1371  
1372     First define the algebraic number field `K = \mathbb{Q}(\sqrt{2})`: 
1373  
1374     >>> from sympy import QQ, sqrt 
1375     >>> from sympy.polys.factortools import dmp_ext_factor 
1376     >>> K = QQ.algebraic_field(sqrt(2)) 
1377  
1378     We can now factorise the polynomial `x^2 y^2 - 2` over `K`: 
1379  
1380     >>> p = [[K(1),K(0),K(0)], [], [K(-2)]] # x**2*y**2 - 2 
1381     >>> p1 = [[K(1),K(0)], [-K.unit]]       # x*y - sqrt(2) 
1382     >>> p2 = [[K(1),K(0)], [+K.unit]]       # x*y + sqrt(2) 
1383     >>> dmp_ext_factor(p, 1, K) == (K.one, [(p1, 1), (p2, 1)]) 
1384     True 
1385  
1386     Usually this would be done at a higher level: 
1387  
1388     >>> from sympy import factor 
1389     >>> from sympy.abc import x, y 
1390     >>> factor(x**2*y**2 - 2, extension=sqrt(2)) 
1391     (x*y - sqrt(2))*(x*y + sqrt(2)) 
1392  
1393     Explanation 
1394     =========== 
1395  
1396     This is Trager's algorithm for multivariate polynomials. In particular this 
1397     function is algorithm ``alg_factor`` from [Trager76]_. 
1398  
1399     See :func:`dup_ext_factor` for explanation. 
1400  
1401     See Also 
1402     ======== 
1403  
1404     dup_ext_factor: 
1405         Analogous function for univariate polynomials over ``k(a)``. 
1406     dmp_sqf_norm: 
1407         Multivariate version of subroutine ``sqfr_norm`` also from [Trager76]_. 
1408     sympy.polys.polytools.factor: 
1409         The high-level function that ultimately uses this function as needed. 
1410     """
1411     if not u:
1412         return dup_ext_factor(f, K)
1413 
1414     lc = dmp_ground_LC(f, u, K)
1415     f = dmp_ground_monic(f, u, K)
1416 
1417     if all(d <= 0 for d in dmp_degree_list(f, u)):
1418         return lc, []
1419 
1420     f, F = dmp_sqf_part(f, u, K), f
1421     s, g, r = dmp_sqf_norm(f, u, K)
1422 
1423     factors = dmp_factor_list_include(r, u, K.dom)
1424 
1425     if len(factors) == 1:
1426         factors = [f]
1427     else:
1428         for i, (factor, _) in enumerate(factors):
1429             h = dmp_convert(factor, u, K.dom, K)
1430             h, _, g = dmp_inner_gcd(h, g, u, K)
1431             a = [si*K.unit for si in s]
1432             h = dmp_shift(h, a, u, K)
1433             factors[i] = h
1434 
1435     result = dmp_trial_division(F, factors, u, K)
1436 
1437     _dmp_check_degrees(F, u, result)
1438 
1439     return lc, result

1459 def dup_factor_list(f, K0):
1460     """Factor univariate polynomials into irreducibles in `K[x]`. """
1461     j, f = dup_terms_gcd(f, K0)
1462     cont, f = dup_primitive(f, K0)

1467         coeff, factors = dup_ext_factor(f, K0)
1468     elif K0.is_GaussianRing:
1469         coeff, factors = dup_zz_i_factor(f, K0)
1470     elif K0.is_GaussianField:
1471         coeff, factors = dup_qq_i_factor(f, K0)
1472     else:
1473         if not K0.is_Exact:
1474             K0_inexact, K0 = K0, K0.get_exact()

1476         else:

1478 
1479         if K0.is_Field:
1480             K = K0.get_ring()
1481 
1482             denom, f = dup_clear_denoms(f, K0, K)
1483             f = dup_convert(f, K0, K)
1484         else:
1485             K = K0
1486

1488             coeff, factors = dup_zz_factor(f, K)
1489         elif K.is_Poly:
1490             f, u = dmp_inject(f, 0, K)
1491

1493

1495                 factors[i] = (dmp_eject(f, u, K), k)
1496

1498         else:  # pragma: no cover

1502             for i, (f, k) in enumerate(factors):
1503                 factors[i] = (dup_convert(f, K, K0), k)

1506             coeff = K0.quo(coeff, denom)
1507

1517                 K0 = K0_inexact
1518

1520         factors.insert(0, ([K0.one, K0.zero], j))
