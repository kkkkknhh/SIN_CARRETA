36   def get_io_paths(fname_inp, mname="untitled"):
37       """Takes in a temporary file for testing and returns the expected output and input paths 
38    
39       Here expected output is essentially one of any of the possible generated 
40       files. 
41    
42       ..note:: 
43    
44            Since this does not actually run f2py, none of these are guaranteed to 
45            exist, and module names are typically incorrect 
46    
47       Parameters 
48       ---------- 
49       fname_inp : str 
50                   The input filename 
51       mname : str, optional 
52                   The name of the module, untitled by default 
53    
54       Returns 
55       ------- 
56       genp : NamedTuple PPaths 
57               The possible paths which are generated, not all of which exist 
58       """
59       bpath = Path(fname_inp)
60       return PPaths(
61           finp=bpath.with_suffix(".f"),
62           f90inp=bpath.with_suffix(".f90"),
63           pyf=bpath.with_suffix(".pyf"),
64           wrap77=bpath.with_name(f"{mname}-f2pywrappers.f"),
65           wrap90=bpath.with_name(f"{mname}-f2pywrappers2.f90"),
66           cmodf=bpath.with_name(f"{mname}module.c"),
67       )
