5    import pytest
6    import numpy as np
7    from scipy.optimize._trlib import (get_trlib_quadratic_subproblem)
8    from numpy.testing import (assert_,
9                               assert_almost_equal,
10                              assert_equal, assert_array_almost_equal)
11   
12   KrylovQP = get_trlib_quadratic_subproblem(tol_rel_i=1e-8, tol_rel_b=1e-6)
13   KrylovQP_disp = get_trlib_quadratic_subproblem(tol_rel_i=1e-8, tol_rel_b=1e-6,
14                                                  disp=True)
15   
16   class TestKrylovQuadraticSubproblem:
17   
18       def test_for_the_easy_case(self):
19   
20           # `H` is chosen such that `g` is not orthogonal to the
21           # eigenvector associated with the smallest eigenvalue.
22           H = np.array([[1.0, 0.0, 4.0],
23                         [0.0, 2.0, 0.0],
24                         [4.0, 0.0, 3.0]])
25           g = np.array([5.0, 0.0, 4.0])
26   
27           # Trust Radius
28           trust_radius = 1.0
29   
30           # Solve Subproblem
31           subprob = KrylovQP(x=0,
32                              fun=lambda x: 0,
33                              jac=lambda x: g,
34                              hess=lambda x: None,
35                              hessp=lambda x, y: H.dot(y))
36           p, hits_boundary = subprob.solve(trust_radius)
37   
38           assert_array_almost_equal(p, np.array([-1.0, 0.0, 0.0]))
39           assert_equal(hits_boundary, True)
40           # check kkt satisfaction
41           assert_almost_equal(
42                   np.linalg.norm(H.dot(p) + subprob.lam * p + g),
43                   0.0)
44           # check trust region constraint
45           assert_almost_equal(np.linalg.norm(p), trust_radius)
46   
47           trust_radius = 0.5
48           p, hits_boundary = subprob.solve(trust_radius)
49   
50           assert_array_almost_equal(p,
51                   np.array([-0.46125446, 0., -0.19298788]))
52           assert_equal(hits_boundary, True)
53           # check kkt satisfaction
54           assert_almost_equal(
55                   np.linalg.norm(H.dot(p) + subprob.lam * p + g),
56                   0.0)
57           # check trust region constraint
58           assert_almost_equal(np.linalg.norm(p), trust_radius)
