3    """ 
4    Utilities for debugging and reproducing issues in Ahead of Time with Inductor (AOTI) compilation. 
5     
6    This file provides tools and utilities for: 
7    - Generating minimal reproducible test cases (minification) 
8    - Handling exported programs and graph modules 
9    - Creating debug repros for AOTI compilation issues 
10   - Supporting both accuracy testing and error reproduction 
11   - Managing configuration and environment for repro cases 
12    
13   The main components include: 
14   - Minification tools to reduce test cases while preserving errors 
15   - Repro generation utilities for exported programs 
16   - Error handling specific to AOTI compilation 
17   - Command-line interface for running and managing repros 
18   """

56   class AOTIMinifierError(Exception):
57       def __init__(self, original_exception):
58           additional_message = "This error is caused by a bug in the AOTI minifier, please report a bug to PyTorch"
59           full_message = f"{additional_message}: {str(original_exception)}"
60           super().__init__(full_message)
61           self.original_exception = original_exception

64   def dump_to_minify(
65       exported_program: ExportedProgram,
66       compiler_name: str,
67       command: str = "minify",
68       options: Optional[dict[str, Any]] = None,
69   ):
70       """ 
71       If command is "minify": 
72           Dump exported_program to `debug_dir/minifier/minifier_launcher.py`, with minify command. 
73       If command is "run": 
74           Dump exported_program to `cwd/repro.py`, with run command. 
75       """
76       assert command in ["minify", "run"]
77   
78       subdir = os.path.join(minifier_dir(), "checkpoints")
79       if not os.path.exists(subdir):
80           os.makedirs(subdir, exist_ok=True)
81   
82       if command == "minify":
83           out = io.StringIO()
84           save_graph_repro_ep(
85               out,
86               compiler_name,
87               exported_program=exported_program,
88               save_dir=subdir,
89               command="minify",
90               config_patches=options,
91           )
92           return helper_for_dump_minify(out.getvalue())
93       else:
94           curdir = os.getcwd()
95           file_name = os.path.join(curdir, "repro.py")
96           try:
97               with open(file_name, "w") as fd:
98                   save_graph_repro_ep(
99                       fd,
100                      compiler_name,
101                      exported_program=exported_program,
102                      config_patches=options,
103                      save_dir=subdir,
104                      command="run",
105                      module_in_comment=True,
106                  )
107              log.warning("Writing repro file to %s", file_name)
108              if use_buck:
109                  BuckTargetWriter(file_name).write()
110          except OSError:
111              log.warning("No write permissions for %s", file_name)

114  def get_module_string(gm):
115      def _convert_to_comment(s_):
116          s = s_.split("\n")
117          if len(s) == 1:
118              return "# " + s_
119          first = s.pop(0)
120          for i in range(len(s)):
121              line = s[i]
122              if line.strip() != "":
123                  s[i] = "# " + line
124              else:
125                  s[i] = ""
126          s = "\n".join(s)
127          s = first + "\n" + s
128          return s
129  
130      module_string = NNModuleToString.convert(gm)
131      return _convert_to_comment(module_string)

134  def save_graph_repro_ep(
135      fd,
136      compiler_name,
137      *,
138      exported_program: Optional[ExportedProgram] = None,
139      gm: Optional[torch.nn.Module] = None,
140      args: Optional[tuple[Any]] = None,
141      config_patches: Optional[dict[str, str]] = None,
142      stable_output=False,
143      save_dir=None,
144      command="run",
145      accuracy=None,
146      check_str=None,
147      module_in_comment=False,
148      strict=False,
149  ):
150      # Save graph for reproducing the error.
151      # Either exported_program or gm will be saved, depending on which one is defined.
152      # Only one of exported_program and gm should be defined.
153  
154      if exported_program is None and gm is None:
155          raise AOTIMinifierError("One of exported_program and gm must be defined")
156      if exported_program is not None and gm is not None:
157          raise AOTIMinifierError("Only one of exported_program and gm can be defined")
158      if gm is not None and args is None:
159          raise AOTIMinifierError("If gm is defined, args should also be defined")
160  
161      if exported_program is None:
162          assert gm is not None
163          assert args is not None
164          exported_program = torch.export.export(gm, args, strict=strict)
165      elif gm is None:
166          gm = exported_program.module()
167  
168      # save a graph preview using gm
169      module_string = get_module_string(gm)
170      fd.write(module_string)
171  
172      # save a graph repro using exported_program
173      fd.write(
174          generate_compiler_repro_exported_program(
175              exported_program,
176              options=config_patches,
177              stable_output=stable_output,
178              save_dir=save_dir,
179          )
180      )
181      if accuracy is None:
182          accuracy = "_accuracy" in compiler_name
183      fd.write("if __name__ == '__main__':\n")
184      fd.write("    from torch._dynamo.repro.aoti import run_repro\n")
185      fd.write(
186          f"    with torch.no_grad():\n"
187          f"        run_repro(exported_program, config_patches=config_patches, accuracy={accuracy!r}, command={command!r}, "
188          f"save_dir={save_dir!r}, check_str={check_str!r})\n"
189      )

192  def dump_compiler_graph_state(
193      gm,
194      args,
195      compiler_name,
196      *,
197      config_patches=None,
198      accuracy=None,
199      strict=False,
200  ):
201      subdir = os.path.join(minifier_dir(), "checkpoints")
202      if not os.path.exists(subdir):
203          os.makedirs(subdir, exist_ok=True)
204      file_name = os.path.join(subdir, f"{len(gm.graph.nodes)}.py")
205      log.warning(
206          "Writing checkpoint with %s nodes to %s", len(gm.graph.nodes), file_name
207      )
208      with open(file_name, "w") as fd:
209          save_graph_repro_ep(
210              fd,
211              compiler_name,
212              gm=gm,
213              args=tuple(args),
214              config_patches=config_patches,
215              save_dir=subdir,
216              accuracy=accuracy,
217              module_in_comment=True,
218              strict=strict,
219          )
220      curdir = os.getcwd()
221      repro_path = os.path.join(curdir, "repro.py")
222      try:
223          shutil.copyfile(file_name, repro_path)
224          log.warning("Copying repro file for convenience to %s", repro_path)
225          if use_buck:
226              BuckTargetWriter(file_name).write()
227      except OSError:
228          log.warning("No write permissions for %s", repro_path)

236  def generate_compiler_repro_exported_program(
237      exported_program,
238      *,
239      options: Optional[dict[str, str]] = None,
240      stable_output=False,
241      save_dir=None,
242  ):
243      model_str = textwrap.dedent(
244          f"""
245  {generate_env_vars_string(stable_output=stable_output)}
246  import torch
247  import torch._inductor.inductor_prims
248  
249  {generate_config_string(stable_output=stable_output)}
250  
251  isolate_fails_code_str = None
252  
253  {extra_imports}
254  
255          """
256      )
257      if not stable_output:
258          model_str += f"# torch version: {torch.version.__version__}\n"
259          if hasattr(torch.version, "cuda"):
260              model_str += f"# torch cuda version: {torch.version.cuda}\n"
261          if hasattr(torch.version, "git_version"):
262              model_str += f"# torch git version: {torch.version.git_version}\n\n\n"
263          model_str += _cuda_system_info_comment()
264  
265      ep_path = os.path.join(save_dir, "exported_program.pt2")
266      torch.export.save(exported_program, ep_path)
267  
268      model_str += f"exported_program = torch.export.load('{ep_path}')\n"
269      model_str += "# print(exported_program.graph)\n"
270      model_str += f"config_patches={options}\n"
271      return model_str

274  def repro_load_args(load_args, save_dir):
275      if not hasattr(load_args, "_version"):
276          log.warning(
277              "load_args does not have a _version attribute, please file a bug to PyTorch "
278              "and describe how you generate this repro script"
279          )
280      else:
281          if load_args._version > 0:
282              log.warning(
283                  "load_args is version %s, but this version of PyTorch only supports "
284                  "version 0.  We will try to run it anyway but there may be an incompatibility; "
285                  "if so, try upgrading your version of PyTorch.",
286                  load_args._version,
287              )
288  
289      nop_reader = NopInputReader()
290      load_args(nop_reader)
291  
292      with tqdm(desc="Loading inputs", total=nop_reader.total) as pbar:
293          input_reader = InputReader(save_dir=save_dir, pbar=pbar)
294          load_args(input_reader)
295          args = input_reader.args
296  
297      return tuple(args)

312  def repro_run(options, exported_program, config_patches):
313      from torch._inductor import _aoti_compile_and_package_inner
314  
315      gm, args, kwargs = repro_common(options, exported_program)
316  
317      from torch.cuda import synchronize
318  
319      _aoti_compile_and_package_inner(
320          gm,
321          args,
322          kwargs,
323          load_and_run=True,
324          check_accuracy=options.accuracy,
325          inductor_configs=config_patches,
326      )
327  
328      need_sync = False
329  
330      for arg in args:
331          if isinstance(arg, torch.Tensor) and arg.is_cuda:
332              need_sync = True
333              break
334  
335      if need_sync:
336          synchronize()  # ensure segfaults are surfaced

339  def export_for_aoti_minifier(
340      gm, tuple_inputs, strict=False, skip_export_error=True
341  ) -> Optional[torch.nn.Module]:
342      # Some graphs cannot be used for AOTI/export (illegal graphs), these should be
343      # considered as graphs that don't fail in the minifier, so the minifier keeps searching.
344      # In these case, we return None. Otherwise, we return the exported graph module.
345      # This won't affect the minifier result because the minifier is only responsible for catching
346      # errors in AOTI, not export.
347      #
348      # Please add to this list of illegal graphs if you change the implementation here.
349      # - graph output is not allowed by export
350      #
351      # If skip_export_error=True, then the errors in export will not be raised, and the minifier
352      # will keep exploring and ignore this graph.
353      from torch._dynamo.exc import UserError, UserErrorType
354  
355      try:
356          ep = torch.export.export(gm, tuple_inputs, strict=strict)
357          gm = ep.module()
358          return gm
359      except Exception as e:
360          if skip_export_error:
361              return None
362          if isinstance(e, UserError) and e.error_type == UserErrorType.INVALID_OUTPUT:
363              # graph output is not allowed by export when strict=True
364              return None
365          if isinstance(e, RuntimeError):
366              # graph output is not allowed by export when strict=False
367              pattern = r"Found .* in output, which is not a known type\."
368              if re.search(pattern, str(e)) is not None:
369                  return None
370          raise AOTIMinifierError(e) from e
371      # we should never reach here
372      return None

375  def repro_minify(options, exported_program, config_patches):
376      from functorch.compile import minifier
377      from torch._inductor import _aoti_compile_and_package_inner
378      from torch._inductor.compile_fx import _aoti_flatten_inputs
379  
380      mod, args, kwargs = repro_common(options, exported_program)
381  
382      # update serialized_in_spec and serialized_out_spec
383      flat_example_inputs, inductor_configs = _aoti_flatten_inputs(
384          mod, args, kwargs, options=config_patches
385      )
386      compiler_name = "aot_inductor"
387      assert options.minifier_export_mode in ["dynamo", "python"]
388      strict = options.minifier_export_mode == "dynamo"
389      skip_export_error = options.skip_export_error
390  
391      from torch.cuda import synchronize
392  
393      need_sync = False
394  
395      for arg in args:
396          if isinstance(arg, torch.Tensor) and arg.is_cuda:
397              need_sync = True
398              break
399  
400      def module_fails(gm, flat_example_inputs, check_str=None):
401          # Need to export first so the in_spec and out_spec are populated
402          tuple_inputs = tuple(flat_example_inputs)
403          gm = export_for_aoti_minifier(
404              gm, tuple_inputs, strict=strict, skip_export_error=skip_export_error
405          )
406  
407          # Some graphs cannot be used for AOTI/export (illegal graphs), these should be
408          # considered as graphs that don't fail in the minifier, so the minifier keeps searching.
409          if gm is None:
410              return False
411  
412          assert isinstance(gm, torch.fx.GraphModule)
413  
414          try:
415              _aoti_compile_and_package_inner(
416                  gm,
417                  tuple_inputs,
418                  load_and_run=True,
419                  check_accuracy=options.accuracy,
420                  inductor_configs=inductor_configs,
421              )
422              if need_sync:
423                  synchronize()  # ensure segfaults are surfaced
424              return False
425          except Exception as e:
426              if check_str is not None and check_str not in repr(e):
427                  return False
428              return True
429  
430      minifier(
431          mod,
432          flat_example_inputs,
433          module_fails=functools.partial(module_fails, check_str=options.check_str),
434          dump_state=functools.partial(
435              dump_compiler_graph_state,
436              compiler_name=compiler_name,
437              config_patches=config_patches,
438              accuracy=options.accuracy,
439              strict=strict,
440          ),
441          save_dir=options.save_dir,
442          offload_to_disk=options.offload_to_disk,
443          skip_offload=options.skip_saving_eager_intermediates,
444          skip_sanity=options.skip_sanity,
445          max_granularity=options.max_granularity,
446      )

449  def run_repro(
450      exported_program,
451      *,
452      config_patches: Optional[dict[str, str]] = None,
453      command="run",
454      accuracy: Union[bool, str] = "",
455      save_dir=None,
456      tracing_mode=None,
457      check_str=None,
458      minifier_export_mode="python",
459      skip_export_error=True,
460      **more_kwargs,
461  ):
462      for k in more_kwargs:
463          log.warning(
464              "Unrecognized kwarg %s; perhaps this repro was made on a newer version of PyTorch",
465              k,
466          )
467  
468      if accuracy is True:
469          accuracy = "accuracy"
470      elif accuracy is False:
471          accuracy = ""
472  
473      parser = argparse.ArgumentParser(
474          description=f"""\
475  An AOTI repro script, typically triggering a bug in PyTorch AOTInductor.
476  When run with no arguments, this script defaults to running '{command}'.
477  Extra flags may be available; to find out more, try '{command} --help'.
478  There are also alternate subcommands available, see below.
479  
480  default settings on this script:
481    {accuracy=}
482    {tracing_mode=}
483    {save_dir=}
484    {check_str=}
485  """,
486          formatter_class=argparse.RawTextHelpFormatter,
487      )
488  
489      def common_flags(parser):
490          accuracy_group = parser.add_mutually_exclusive_group()
491          accuracy_group.add_argument(
492              "--no-accuracy",
493              dest="accuracy",
494              action="store_const",
495              const="",
496              default=accuracy,
497              help="do not test accuracy, just run the module and see if it errors",
498          )
499          accuracy_group.add_argument(
500              "--accuracy",
501              action="store_const",
502              const="accuracy",
503              default=accuracy,
504              help="""\
505  test if the RMSE between the compiled module and the fp64 reference is greater 
506  than eager and the fp64 reference. This is usually more reliable than the 
507  standard allclose test, as we expect numeric differences from compiling, often 
508  improving accuracy over eager.  RMSE test allows for compiled module to 
509  diverge greatly from eager, as long as this divergence moves it closer to the 
510  'true' mathematical value of the network.  Caveats: (1) double precision can 
511  still suffer from rounding error, so it is not a perfect reference (see for 
512  example 'Herbie: Automatically Improving Floating Point Accuracy') for 
513  approaches that detect the necessary working precision and compute it in 
514  arbitrary precision floating point; unfortunately, this is not practical for 
515  tensor computation; (2) if there are not enough samples in the output being 
516  compared, we may get unlucky and have an unlucky greater RMSE than eager; this 
517  could be overcome by applying a more rigorous statistical test at some 
518  p-value, which we leave for future work. 
519  """,
520          )
521          accuracy_group.add_argument(
522              "--strict-accuracy",
523              dest="accuracy",
524              action="store_const",
525              const="strict_accuracy",
526              default=accuracy,
527              help="""\
528  by default, when doing accuracy minification we will reject reductions which 
529  change the divergence from a floating point divergence to a integral/boolean 
530  divergence.  This is because some operations like ReLU involve temporarily 
531  sharp boundaries that smooth out again afterwards; without requiring 
532  divergence on floating point, the minifier will often fixate on divergent 
533  boolean tensor even though this is not the true source of the divergence. 
534  However, rejecting these reductions makes it more difficult for the minifier 
535  to make process.  Using this option will let the minifier progress for ALL 
536  divergences--you just might not end up with a useful repro in the end.""",
537          )
538  
539          parser.add_argument(
540              "--save-dir",
541              type=str,
542              default=save_dir,
543              metavar="DIR",
544              help="directory where saved inputs live",
545          )
546          parser.add_argument(
547              "--no-save-dir",
548              dest="save_dir",
549              action="store_const",
550              const=None,
551              help="don't use any directory for saved inputs",
552          )
553  
554      subparsers = parser.add_subparsers(
555          dest="command", metavar="{run,minify}", required=True
556      )
557  
558      parser_run = subparsers.add_parser(
559          "run",
560          help="just run the repro",
561      )
562      common_flags(parser_run)
563  
564      parser_minify = subparsers.add_parser(
565          "minify", help="run the minifier on the repro"
566      )
567      common_flags(parser_minify)
568      parser_get_args = subparsers.add_parser("get_args", help="get the args")
569      common_flags(parser_get_args)
570      parser_minify.add_argument(
571          "--skip-saving-eager-intermediates",
572          action="store_true",
573          help="skip saving eager intermediates on --minify",
574      )
575      parser_minify.add_argument(
576          "--offload-to-disk",
577          action="store_true",
578          help="during minification, offload delta debugging intermediates to disk.  Use if you're OOMing",
579      )
580      parser_minify.add_argument(
581          "--skip-sanity",
582          action="store_true",
583          help="skip sanity check at beginning of minification on original graph",
584      )
585      parser_minify.add_argument(
586          "--max-granularity",
587          type=int,
588          default=None,
589          help="start at this granularity and work down; must be power of 2",
590      )
591      parser_minify.add_argument(
592          "--check-str",
593          type=str,
594          default=check_str,
595          help="require minified program to fail with error containing this string",
596      )
597      parser_minify.add_argument(
598          "--minifier-export-mode",
599          type=str,
600          default=minifier_export_mode,
601          help=(
602              "The export mode used in minifier, either dynamo or python."
603              "`dynamo` corresponds to strict=True, and `python` corresponds to strict=False."
604          ),
605      )
606      parser_minify.add_argument(
607          "--skip-export-error",
608          type=bool,
609          default=skip_export_error,
610          help="Skip intermediate graphs that cannot be exported.",
611      )
612  
613      # Run the repro in the context of minification, inverting exit code meaning
614      parser_minifier_query = subparsers.add_parser(
615          "minifier-query",
616      )
617      common_flags(parser_minifier_query)
618      parser_minifier_query.add_argument(
619          "--check-str",
620          type=str,
621          default=check_str,
622          help="require minified program to fail with error containing this string",
623      )
624  
625      args = None
626      if len(sys.argv) <= 1:
627          args = [command, *sys.argv[1:]]
628  
629      options = parser.parse_args(args)
630      COMMAND_FNS = {
631          "minify": repro_minify,
632          "run": repro_run,
633          "get_args": repro_get_args,
634      }
635      return COMMAND_FNS[options.command](
636          options, exported_program, config_patches=config_patches
637      )
