40   def test_subs():
41       A = ImmutableMatrix([[1, 2], [3, 4]])
42       B = ImmutableMatrix([[1, 2], [x, 4]])
43       C = ImmutableMatrix([[-x, x*y], [-(x + y), y**2]])
44       assert B.subs(x, 3) == A
45       assert (x*B).subs(x, 3) == 3*A
46       assert (x*eye(2) + B).subs(x, 3) == 3*eye(2) + A
47       assert C.subs([[x, -1], [y, -2]]) == A
48       assert C.subs([(x, -1), (y, -2)]) == A
49       assert C.subs({x: -1, y: -2}) == A
50       assert C.subs({x: y - 1, y: x - 1}, simultaneous=True) == \
51           ImmutableMatrix([[1 - y, (x - 1)*(y - 1)], [2 - x - y, (x - 1)**2]])

64   def test_function_return_types():
65       # Lets ensure that decompositions of immutable matrices remain immutable
66       # I.e. do MatrixBase methods return the correct class?
67       X = ImmutableMatrix([[1, 2], [3, 4]])
68       Y = ImmutableMatrix([[1], [0]])
69       q, r = X.QRdecomposition()
70       assert (type(q), type(r)) == (ImmutableMatrix, ImmutableMatrix)
71   
72       assert type(X.LUsolve(Y)) == ImmutableMatrix
73       assert type(X.QRsolve(Y)) == ImmutableMatrix
74   
75       X = ImmutableMatrix([[5, 2], [2, 7]])
76       assert X.T == X
77       assert X.is_symmetric
78       assert type(X.cholesky()) == ImmutableMatrix
79       L, D = X.LDLdecomposition()
80       assert (type(L), type(D)) == (ImmutableMatrix, ImmutableMatrix)

116  def test_Equality():
117      assert Equality(IM, IM) is S.true
118      assert Unequality(IM, IM) is S.false
119      assert Equality(IM, IM.subs(1, 2)) is S.false
120      assert Unequality(IM, IM.subs(1, 2)) is S.true
121      assert Equality(IM, 2) is S.false
122      assert Unequality(IM, 2) is S.true
123      M = ImmutableMatrix([x, y])
124      assert Equality(M, IM) is S.false
125      assert Unequality(M, IM) is S.true
126      assert Equality(M, M.subs(x, 2)).subs(x, 2) is S.true
127      assert Unequality(M, M.subs(x, 2)).subs(x, 2) is S.false
128      assert Equality(M, M.subs(x, 2)).subs(x, 3) is S.false
129      assert Unequality(M, M.subs(x, 2)).subs(x, 3) is S.true
