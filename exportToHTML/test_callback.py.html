124      def test_threadsafety(self):
125          # Segfaults if the callback handling is not threadsafe
126  
127          errors = []
128  
129          def cb():
130              # Sleep here to make it more likely for another thread
131              # to call their callback at the same time.
132              time.sleep(1e-3)
133  
134              # Check reentrancy
135              r = self.module.t(lambda: 123)
136              assert r == 123
137  
138              return 42
139  
140          def runner(name):
141              try:
142                  for j in range(50):
143                      r = self.module.t(cb)
144                      assert r == 42
145                      self.check_function(name)
146              except Exception:
147                  errors.append(traceback.format_exc())
148  
149          threads = [
150              threading.Thread(target=runner, args=(arg, ))
151              for arg in ("t", "t2") for n in range(20)
152          ]
153  
154          for t in threads:
155              t.start()
156  
157          for t in threads:
158              t.join()
159  
160          errors = "\n\n".join(errors)
161          if errors:
162              raise AssertionError(errors)

164      def test_hidden_callback(self):
165          try:
166              self.module.hidden_callback(2)
167          except Exception as msg:
168              assert str(msg).startswith("Callback global_f not defined")
169  
170          try:
171              self.module.hidden_callback2(2)
172          except Exception as msg:
173              assert str(msg).startswith("cb: Callback global_f not defined")
174  
175          self.module.global_f = lambda x: x + 1
176          r = self.module.hidden_callback(2)
177          assert r == 3
178  
179          self.module.global_f = lambda x: x + 2
180          r = self.module.hidden_callback(2)
181          assert r == 4
182  
183          del self.module.global_f
184          try:
185              self.module.hidden_callback(2)
186          except Exception as msg:
187              assert str(msg).startswith("Callback global_f not defined")
188  
189          self.module.global_f = lambda x=0: x + 3
190          r = self.module.hidden_callback(2)
191          assert r == 5
192  
193          # reproducer of gh18341
194          r = self.module.hidden_callback2(2)
195          assert r == 3
