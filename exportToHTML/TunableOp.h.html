30     public:
31       virtual ~Callable() = default;
32       virtual TuningStatus Call(const ParamsT*) {
33         return FAIL;

42   /** http://en.wikipedia.org/wiki/Algorithms_for_calculating_variance */
43   
44   class Stats {
45     public:
46       Stats() {
47         _n = 0UL;
48         _mean = 0.0;
49         _M2 = 0.0;
50         _sum = 0.0;
51         _min = 0.0;
52         _max = 0.0;
53       }
54   
55       void sample_value(const double x) {
56         double delta = 0;
57         _sum = _sum + x;
58         if (0UL == _n) {
59             _min = x;
60             _max = x;
61         }
62         else {
63             _min = _min < x ? _min : x;
64             _max = _max > x ? _max : x;
65         }
66         _n = _n + 1UL;
67         delta = x - _mean;
68         _mean = _mean + delta/_n;
69         _M2 = _M2 + delta * (x - _mean);
70       }
71   
72       double variance() const {
73         return _M2/(_n-1);
74       }
75   
76       double stddev() const {
77         return std::sqrt(variance());
78       }
79   
80       unsigned long _n;
81       double _mean;
82       double _M2;
83       double _sum;
84       double _min;
85       double _max;
86   };

88   class FixedSizeStack {
89     private:
90         std::deque<std::string> stack;
91         const size_t max_size;
92   
93     public:
94         FixedSizeStack(size_t size) : max_size(size) {}
95   
96         void push(const std::string& value) {
97             if (stack.size() >= max_size) {
98                 stack.pop_front(); // Remove the oldest entry
99             }
100            stack.push_back(value); // Add new entry
101        }
102  
103        auto rbegin() { return stack.rbegin(); }
104        auto rend() { return stack.rend(); }
105  };

109  template <typename ParamsT>
110  class TunableOp {
111    public:
112      virtual ~TunableOp() = default;
113  
114      TuningStatus operator()(const ParamsT* params) {
115        ResultEntry result = ResultEntry::Null();
116        TuningContext* ctx = getTuningContext();
117        if (ctx->IsTunableOpEnabled()) {
118          auto& mgr = ctx->GetTuningResultsManager();
119          auto op_sig = Signature();
120          auto params_sig = params->Signature();
121          auto blas_sig = params->BLASSignature();
122          result = mgr.Lookup(op_sig, params_sig);
123          // If there is not previous tuning result been found, we do the tuning iff tuning is enabled
124          if (result == ResultEntry::Null()) {
125            if (ctx->IsTuningEnabled()) {
126              result = FindFastest(params);
127              mgr.Add(op_sig, params_sig, result);
128            }
129            else if (ctx->IsRecordUntunedEnabled()) {
130              // or record the gemm into file
131              mgr.RecordUntuned(ctx->GetUntunedFile(), op_sig, params_sig, blas_sig);
132            }
133          }
134        }
135        else {
136          result = ResultEntry::Default();
137        }
138        if (result == ResultEntry::Null()) {
139          TUNABLE_LOG2("no result, using default");
140          result = ResultEntry::Default();
141        }
142        auto iter = ops_.find(result);
143        TORCH_CHECK(iter != ops_.end());
144        return iter->second->Call(params);
145      }
146  
147      virtual std::string Signature() {
148        // According to C++17 standard https://wg21.link/n4659 section 15.7.4
149        // > if the operand of typeid refers to the
150        // > object under construction or destruction, typeid yields the std::type_info object representing the constructor
151        // > or destructor’s class.
152        // So delay the op signature generation.
153        c10::call_once(signature_init_once_, [this]() { signature_ = CreateSignature(); });
154        return signature_;
155      }
156  
157    protected:
158      void RegisterOp(const std::string& name, std::unique_ptr<Callable<ParamsT>> op) {
159        this->op_names_.emplace_back(name);
160        this->ops_.emplace(name, std::move(op));
161      }
162  
163    private:
164      static void WarmUp(Callable<ParamsT> *op, const std::vector<ParamsT*> &param, size_t num_iter, size_t &offset) {
165        TuningContext* ctx = getTuningContext();
166        bool do_flush = ctx->IsICacheFlushEnabled();
167        for (size_t i = 0; i < num_iter; i++) {
168          if (do_flush) {
169            at::cuda::flush_icache();
170          }
171          TORCH_CHECK(op->Call(param[(i+offset++)%param.size()]) == OK);
172        }
173      }
174  
175      static double ProfileSimple(Callable<ParamsT> *op, const std::vector<ParamsT*> &param, size_t num_iter, size_t &offset) {
176        TuningContext* ctx = getTuningContext();
177        bool do_flush = ctx->IsICacheFlushEnabled();
178        StreamTimerNoSync timer{};
179  
180        // Small Mandatory Warmup
181        // Reduces outliers
182        for (size_t i = 0; i < 2; i++) {
183          TORCH_CHECK(op->Call(param[(i+offset++)%param.size()]) == OK);
184        }
185  
186        timer.Start();
187        for (size_t i = 0; i < num_iter; i++) {
188          if (do_flush) {
189            at::cuda::flush_icache();
190          }
191          TORCH_CHECK(op->Call(param[(i+offset++)%param.size()]) == OK);
192        }
193        timer.End();
194        return timer.Duration() / num_iter;
195      }
196  
197      static Stats ProfileStats(Callable<ParamsT> *op, const std::vector<ParamsT*> &param, size_t num_iter, size_t &offset) {
198        TuningContext* ctx = getTuningContext();
199        bool do_flush = ctx->IsICacheFlushEnabled();
200        std::vector<StreamTimerNoSync> timer(num_iter);
201  
202        // Small Mandatory Warmup
203        // Reduces outliers
204        for (size_t i = 0; i < 2; i++) {
205          TORCH_CHECK(op->Call(param[(i+offset++)%param.size()]) == OK);
206        }
207  
208        for (size_t i = 0; i < num_iter; i++) {
209          timer[i].Start();
210          TORCH_CHECK(op->Call(param[(i+offset++)%param.size()]) == OK);
211          timer[i].End();
212          if (do_flush) {
213            at::cuda::flush_icache();
214          }
215        }
216        Stats s;
217        for (size_t i = 0; i < num_iter; i++) {
218          s.sample_value(timer[i].Duration());
219        }
220        return s;
221      }
222  
223    protected:
224      virtual ResultEntry FindFastest(const ParamsT* params) {
225        TuningContext* ctx = getTuningContext();
226        auto op_sig = Signature();
227        auto params_sig = params->Signature();
228        auto blas_sig = params->BLASSignature();
229        TUNABLE_LOG2("finding fastest for ", op_sig, '(', params_sig, ')', " out of ", op_names_.size(), " candidates");
230        auto min_duration_ms = std::numeric_limits<double>::infinity();
231        std::string id_name = "Default";
232        ParamsT* reference_params = nullptr;
233        auto top_solns = FixedSizeStack(5);
234  
235        // numeric check option is controlled by non-static env var, so check it once per tuned operator
236        bool do_numerics_check = ctx->IsNumericsCheckEnabled();
237  
238        // calcaulte a reference answer for numerical check
239        if (do_numerics_check) {
240          reference_params = params->DeepCopy(false);
241          TORCH_CHECK(ops_[ResultEntry::Default()]->Call(reference_params) == OK);
242        }
243  
244        // need copies of params to reuse
245        // make as many copies as will fill the requested rotating buffer size, if requested
246        // rotating_size guaranteed to be >= 0 even though GetRotatingBufferSize() returns int
247        size_t rotating_size = ctx->GetRotatingBufferSize();
248        bool use_buffer_rotation = (rotating_size > 0);
249        size_t param_size = params->GetSize(use_buffer_rotation);
250        size_t param_count = (rotating_size / param_size) + 1;
251        constexpr size_t MB = 1024ull*1024;
252        if (use_buffer_rotation) {
253          TUNABLE_LOG2("Rotating buffer ", rotating_size/MB, " MiB. ",
254              "Needed Size: ", param_size/MB, " MiB. ",
255              "Needed number of param copies: ", param_count);
256        }
257        TORCH_CHECK(param_count > 0);
258  
259        std::vector<ParamsT*> reusable_params(param_count);
260        for (size_t i = 0; i < param_count; i++) {
261          reusable_params[i] = params->DeepCopy(use_buffer_rotation);
262        }
263  
264        // for rotating buffer
265        size_t offset = 0;
266  
267        for (size_t i = 0; i < op_names_.size(); i++) {
268          auto* candidate = ops_[op_names_[i]].get(); // borrow pointer
269  
270          if (do_numerics_check) {
271            ParamsT* numerical_params = params->DeepCopy(false);
272            auto status = candidate->Call(numerical_params);
273            if (status != OK) {
274              numerical_params->Delete();
275              TUNABLE_LOG3("├──unsupported id=", i, ", ", op_sig, '(', params_sig, ") ", op_names_[i]);
276              continue;
277            }
278            status = reference_params->NumericalCheck(numerical_params);
279            numerical_params->Delete();
280            if (status != OK) {
281              TUNABLE_LOG3("├──numerics check failed for id=", i, ", ", op_sig, '(', params_sig, ") ", op_names_[i]);
282              continue;
283            }
284          }
285          else {
286            auto status = candidate->Call(reusable_params[0]);
287            if (status != OK) {
288              TUNABLE_LOG3("├──unsupported id=", i, ", ", op_sig, '(', params_sig, ") ", op_names_[i]);
289              continue;
290            }
291          }
292  
293          // collect a small profile
294          int approx_num_iter = 3;
295          auto s = ProfileStats(candidate, reusable_params, approx_num_iter, offset);
296          double approx_duration = s._mean;
297          // bail if too slow
298          if (approx_duration > 1.5 * min_duration_ms) {
299            TUNABLE_LOG3("├──skip slow instance id=", i, ", ", op_sig, '(', params_sig, ") ", op_names_[i]);
300            continue;
301          }
302  
303          // 2nd phase skip, more aggressive
304          approx_num_iter = 10;
305          s = ProfileStats(candidate, reusable_params, approx_num_iter, offset);
306          approx_duration = s._mean;
307          // bail if too slow
308          if (approx_duration > 1.15 * min_duration_ms) {
309            TUNABLE_LOG3("├──2nd skip slow instance id=", i, ", ", op_sig, '(', params_sig, ") ", op_names_[i]);
310            continue;
311          }
312  
313          // for warmup does user set max duration, max iters, or both?
314          // warmup is skipped by default, i.e. warmup_iter = 0
315          // warmup will be set to the non-zero value of max_warmup_duration
316          // or max_warmup_iter
317          // if both are non-zero, we take the smaller of the two.
318          double max_warmup_duration = ctx->GetMaxWarmupDurationMs();
319          int max_warmup_iter = ctx->GetMaxWarmupIterations();
320          int warmup_iter = 0; // default
321          if (max_warmup_duration > 0) {
322            int duration_iters = max_warmup_duration / approx_duration;
323            if (max_warmup_iter > 0) {
324              warmup_iter = std::min(max_warmup_iter, duration_iters);
325            }
326            else {
327              warmup_iter = duration_iters;
328            }
329          }
330          else if (max_warmup_iter > 0) {
331            warmup_iter = max_warmup_iter;
332          }
333  
334          // for tuning does user set max duration, max iters, or both?
335          double max_tuning_duration = ctx->GetMaxTuningDurationMs();
336          int max_tuning_iter = ctx->GetMaxTuningIterations();
337          int tuning_iter = 100; // default
338          if (max_tuning_duration > 0) {
339            int duration_iters = max_tuning_duration / approx_duration;
340            if (max_tuning_iter > 0) {
341              tuning_iter = std::min(max_tuning_iter, duration_iters);
342            }
343            else {
344              tuning_iter = duration_iters;
345            }
346          }
347          else if (max_tuning_iter > 0) {
348            tuning_iter = max_tuning_iter;
349          }
350          // tuning must run at least 1 iteration
351          tuning_iter = std::max(1, tuning_iter);
352  
353          // do the full warmup followed by tuning
354          double warmup_ms = warmup_iter * approx_duration;
355          double tuning_ms = tuning_iter * approx_duration;
356          TUNABLE_LOG3("├──tuning using "
357              "warmup iters ", warmup_iter, " [", warmup_ms, " ms] "
358              "and tuning iters ", tuning_iter, " [", tuning_ms, " ms] ",
359              "instance id=", i, ", ", op_sig, "(", params_sig, ") ", op_names_[i]);
360          TUNABLE_LOG3("├──offset at ", offset);
361          WarmUp(candidate, reusable_params, warmup_iter, offset);
362          s = ProfileStats(candidate, reusable_params, tuning_iter, offset);
363          auto s_stddev = s.stddev();
364          // Assume normal distribution.
365          // Solution with smallest mean + 2*sigma will be a better solution?
366          // if ((s._mean + 2*s_stddev) < (min_duration_ms + 2*min_stddev_ms)) {
367          if (s._mean < min_duration_ms) {
368            TUNABLE_LOG3("├──found better instance id=", i, ". " , s._mean, "ms. ", op_names_[i],
369                  " min ", s._min,
370                  " max ", s._max,
371                  " mean ", s._mean,
372                  " std ", s_stddev);
373            min_duration_ms = s._mean;
374            id_name = op_names_[i];
375            std::string current_soln = std::to_string(s._mean) + " " + op_names_[i];
376            top_solns.push(current_soln);
377          }
378          else {
379            TUNABLE_LOG3("├──found slower instance id=", i, ". " , s._mean, "ms. ", op_names_[i],
380                  " min ", s._min,
381                  " max ", s._max,
382                  " mean ", s._mean,
383                  " std ", s_stddev);
384          }
385        }
386  
387        for (size_t i = 0; i < reusable_params.size(); i++) {
388          reusable_params[i]->Delete();
389        }
390        if (reference_params) {
391          reference_params->Delete();
392        }
393  
394        TUNABLE_LOG2("└──found fastest for ", op_sig, '(', params_sig, ") ", id_name);
395        TUNABLE_LOG2("└──top five solutions for ", op_sig, '(', params_sig, ") ");
396        for (auto it = top_solns.rbegin(); it != top_solns.rend(); ++it) {
397          TUNABLE_LOG2("   ", *it);
398        }
399        return ResultEntry(id_name, min_duration_ms, blas_sig);
400      }

402    private:
403      std::string CreateSignature() {
404  #ifndef _WIN32
405        const auto* name = typeid(*this).name();
406        // NOLINTNEXTLINE(*array*)
407        char buf[256];
408        size_t buf_len = 256;
409        abi::__cxa_demangle(name, buf, &buf_len, nullptr);
410        buf[255] = '\0';
411        return buf;
412  #else
413        return typeid(*this).name();
414  #endif
415      }
416  
417      mutable c10::once_flag signature_init_once_;
418      std::string signature_;
419  
420      std::unordered_map<std::string, std::unique_ptr<Callable<ParamsT>>> ops_;
421      std::vector<std::string> op_names_;
422  };
