15   class ActivationSparsifier:
16       r""" 
17       The Activation sparsifier class aims to sparsify/prune activations in a neural 
18       network. The idea is to attach the sparsifier to a layer (or layers) and it 
19       zeroes out the activations based on the mask_fn (or sparsification function) 
20       input by the user. 
21       The mask_fn is applied once all the inputs are aggregated and reduced i.e. 
22       mask = mask_fn(reduce_fn(aggregate_fn(activations))) 
23    
24       Note:: 
25           The sparsification mask is computed on the input **before it goes through the attached layer**. 
26    
27       Args: 
28           model (nn.Module): 
29               The model whose layers will be sparsified. The layers that needs to be 
30               sparsified should be added separately using the register_layer() function 
31           aggregate_fn (Optional, Callable): 
32               default aggregate_fn that is used if not specified while registering the layer. 
33               specifies how inputs should be aggregated over time. 
34               The aggregate_fn should usually take 2 torch tensors and return the aggregated tensor. 
35               Example 
36                   def add_agg_fn(tensor1, tensor2):  return tensor1 + tensor2 
37                   reduce_fn (Optional, Callable): 
38                       default reduce_fn that is used if not specified while registering the layer. 
39                       reduce_fn will be called on the aggregated tensor i.e. the tensor obtained after 
40                       calling agg_fn() on all inputs. 
41                       Example 
42                   def mean_reduce_fn(agg_tensor):    return agg_tensor.mean(dim=0) 
43                   mask_fn (Optional, Callable): 
44                       default mask_fn that is used to create the sparsification mask using the tensor obtained after 
45                       calling the reduce_fn(). This is used by default if a custom one is passed in the 
46                       register_layer(). 
47                       Note that the mask_fn() definition should contain the sparse arguments that is passed in sparse_config 
48                       arguments. 
49                   features (Optional, list): 
50                       default selected features to sparsify. 
51                       If this is non-empty, then the mask_fn will be applied for each feature of the input. 
52                       For example, 
53                   mask = [mask_fn(reduce_fn(aggregated_fn(input[feature])) for feature in features] 
54                   feature_dim (Optional, int): 
55                       default dimension of input features. Again, features along this dim will be chosen 
56                       for sparsification. 
57                   sparse_config (Dict): 
58                       Default configuration for the mask_fn. This config will be passed 
59                       with the mask_fn() 
60    
61       Example: 
62           >>> # xdoctest: +SKIP 
63           >>> model = SomeModel() 
64           >>> act_sparsifier = ActivationSparsifier(...)  # init activation sparsifier 
65           >>> # Initialize aggregate_fn 
66           >>> def agg_fn(x, y): 
67           >>>     return x + y 
68           >>> 
69           >>> # Initialize reduce_fn 
70           >>> def reduce_fn(x): 
71           >>>     return torch.mean(x, dim=0) 
72           >>> 
73           >>> # Initialize mask_fn 
74           >>> def mask_fn(data): 
75           >>>     return torch.eye(data.shape).to(data.device) 
76           >>> 
77           >>> 
78           >>> act_sparsifier.register_layer( 
79           ...     model.some_layer, 
80           ...     aggregate_fn=agg_fn, 
81           ...     reduce_fn=reduce_fn, 
82           ...     mask_fn=mask_fn, 
83           ... ) 
84           >>> 
85           >>> # start training process 
86           >>> for _ in [...]: 
87           >>> # epoch starts 
88           >>> # model.forward(), compute_loss() and model.backwards() 
89           >>> # epoch ends 
90           >>>     act_sparsifier.step() 
91           >>> # end training process 
92           >>> sparsifier.squash_mask() 
93       """

95       def __init__(
96           self,
97           model: nn.Module,
98           aggregate_fn=None,
99           reduce_fn=None,
100          mask_fn=None,
101          features=None,
102          feature_dim=None,
103          **sparse_config,
104      ):
105          self.model = model
106          self.defaults: dict[str, Any] = defaultdict()
107          self.defaults["sparse_config"] = sparse_config
108  
109          # functions
110          self.defaults["aggregate_fn"] = aggregate_fn
111          self.defaults["reduce_fn"] = reduce_fn
112          self.defaults["mask_fn"] = mask_fn
113  
114          # default feature and feature_dim
115          self.defaults["features"] = features
116          self.defaults["feature_dim"] = feature_dim
117  
118          self.data_groups: dict[str, dict] = defaultdict(
119              dict
120          )  # contains all relevant info w.r.t each registered layer
121  
122          self.state: dict[str, Any] = defaultdict(dict)  # layer name -> mask
123  
124      @staticmethod
125      def _safe_rail_checks(args):
126          """Makes sure that some of the functions and attributes are not passed incorrectly"""
127  
128          # if features are not None, then feature_dim must not be None
129          features, feature_dim = args["features"], args["feature_dim"]
130          if features is not None:
131              assert feature_dim is not None, "need feature dim to select features"
132  
133          # all the *_fns should be callable
134          fn_keys = ["aggregate_fn", "reduce_fn", "mask_fn"]
135          for key in fn_keys:
136              fn = args[key]
137              assert callable(fn), "function should be callable"

139      def _aggregate_hook(self, name):
140          """Returns hook that computes aggregate of activations passing through."""
141  
142          # gather some data
143          feature_dim = self.data_groups[name]["feature_dim"]
144          features = self.data_groups[name]["features"]
145          agg_fn = self.data_groups[name]["aggregate_fn"]
146  
147          def hook(module, input) -> None:
148              input_data = input[0]
149  
150              data = self.data_groups[name].get("data")  # aggregated data
151              if features is None:
152                  # no features associated, data should not be a list
153                  if data is None:
154                      data = torch.zeros_like(input_data)
155                      self.state[name]["mask"] = torch.ones_like(input_data)
156                  out_data = agg_fn(data, input_data)
157              else:
158                  # data should be a list [aggregated over each feature only]
159                  if data is None:
160                      out_data = [
161                          0 for _ in range(0, len(features))
162                      ]  # create one incase of 1st forward
163                      self.state[name]["mask"] = [0 for _ in range(0, len(features))]
164                  else:
165                      out_data = data  # a list
166  
167                  # compute aggregate over each feature
168                  for feature_idx in range(len(features)):
169                      # each feature is either a list or scalar, convert it to torch tensor
170                      feature_tensor = (
171                          torch.Tensor([features[feature_idx]])
172                          .long()
173                          .to(input_data.device)
174                      )
175                      data_feature = torch.index_select(
176                          input_data, feature_dim, feature_tensor
177                      )
178                      if data is None:
179                          curr_data = torch.zeros_like(data_feature)
180                          self.state[name]["mask"][feature_idx] = torch.ones_like(
181                              data_feature
182                          )
183                      else:
184                          curr_data = data[feature_idx]
185                      out_data[feature_idx] = agg_fn(curr_data, data_feature)
186              self.data_groups[name]["data"] = out_data
187  
188          return hook

190      def register_layer(
191          self,
192          layer: nn.Module,
193          aggregate_fn=None,
194          reduce_fn=None,
195          mask_fn=None,
196          features=None,
197          feature_dim=None,
198          **sparse_config,
199      ):
200          r""" 
201          Registers a layer for sparsification. The layer should be part of self.model. 
202          Specifically, registers a pre-forward hook to the layer. The hook will apply the aggregate_fn 
203          and store the aggregated activations that is input over each step. 
204   
205          Note:: 
206              - There is no need to pass in the name of the layer as it is automatically computed as per 
207                the fqn convention. 
208   
209              - All the functions (fn) passed as argument will be called at a dim, feature level. 
210          """
211          name = module_to_fqn(self.model, layer)
212          assert name is not None, "layer not found in the model"  # satisfy mypy
213  
214          if name in self.data_groups:  # unregister layer if already present
215              warnings.warn(
216                  "layer already attached to the sparsifier, deregistering the layer and registering with new config"
217              )
218              self.unregister_layer(name=name)
219  
220          local_args = copy.deepcopy(self.defaults)
221          update_dict = {
222              "aggregate_fn": aggregate_fn,
223              "reduce_fn": reduce_fn,
224              "mask_fn": mask_fn,
225              "features": features,
226              "feature_dim": feature_dim,
227              "layer": layer,
228          }
229          local_args.update(
230              (arg, val) for arg, val in update_dict.items() if val is not None
231          )
232          local_args["sparse_config"].update(sparse_config)
233  
234          self._safe_rail_checks(local_args)
235  
236          self.data_groups[name] = local_args
237          agg_hook = layer.register_forward_pre_hook(self._aggregate_hook(name=name))
238  
239          self.state[name]["mask"] = (
240              None  # mask will be created when model forward is called.
241          )
242  
243          # attach agg hook
244          self.data_groups[name]["hook"] = agg_hook
245  
246          # for serialization purposes, we know whether aggregate_hook is attached
247          # or sparsify_hook()
248          self.data_groups[name]["hook_state"] = "aggregate"  # aggregate hook is attached
249  
250      def get_mask(self, name: Optional[str] = None, layer: Optional[nn.Module] = None):
251          """ 
252          Returns mask associated to the layer. 
253   
254          The mask is 
255              - a torch tensor is features for that layer is None. 
256              - a list of torch tensors for each feature, otherwise 
257   
258          Note:: 
259              The shape of the mask is unknown until model.forward() is applied. 
260              Hence, if get_mask() is called before model.forward(), an 
261              error will be raised. 
262          """
263          assert name is not None or layer is not None, (
264              "Need at least name or layer obj to retrieve mask"
265          )
266  
267          if name is None:
268              assert layer is not None
269              name = module_to_fqn(self.model, layer)
270              assert name is not None, "layer not found in the specified model"
271  
272          if name not in self.state:
273              raise ValueError("Error: layer with the given name not found")
274  
275          mask = self.state[name].get("mask", None)
276  
277          if mask is None:
278              raise ValueError(
279                  "Error: shape unknown, call layer() routine at least once to infer mask"
280              )
281          return mask

283      def unregister_layer(self, name):
284          """Detaches the sparsifier from the layer"""
285  
286          # detach any hooks attached
287          self.data_groups[name]["hook"].remove()
288  
289          # pop from the state dict
290          self.state.pop(name)
291  
292          # pop from the data groups
293          self.data_groups.pop(name)
294  
295      def step(self):
296          """Internally calls the update_mask() function for each layer"""
297          with torch.no_grad():
298              for name, configs in self.data_groups.items():
299                  data = configs["data"]
300                  self.update_mask(name, data, configs)
301  
302                  self.data_groups[name].pop("data")  # reset the accumulated data
303  
304      def update_mask(self, name, data, configs):
305          """ 
306          Called for each registered layer and does the following- 
307              1. apply reduce_fn on the aggregated activations 
308              2. use mask_fn to compute the sparsification mask 
309   
310          Note: 
311              the reduce_fn and mask_fn is called for each feature, dim over the data 
312          """
313          mask = self.get_mask(name)
314          sparse_config = configs["sparse_config"]
315          features = configs["features"]
316          reduce_fn = configs["reduce_fn"]
317          mask_fn = configs["mask_fn"]
318          if features is None:
319              data = reduce_fn(data)
320              mask.data = mask_fn(data, **sparse_config)
321          else:
322              for feature_idx in range(len(features)):
323                  data_feature = reduce_fn(data[feature_idx])
324                  mask[feature_idx].data = mask_fn(data_feature, **sparse_config)
325  
326      def _sparsify_hook(self, name):
327          """Returns hook that applies sparsification mask to input entering the attached layer"""
328          mask = self.get_mask(name)
329          features = self.data_groups[name]["features"]
330          feature_dim = self.data_groups[name]["feature_dim"]
331  
332          def hook(module, input):
333              input_data = input[0]
334              if features is None:
335                  # apply to all the features
336                  return input_data * mask
337              else:
338                  # apply per feature, feature_dim
339                  for feature_idx in range(0, len(features)):
340                      feature = (
341                          torch.Tensor([features[feature_idx]])
342                          .long()
343                          .to(input_data.device)
344                      )
345                      sparsified = (
346                          torch.index_select(input_data, feature_dim, feature)
347                          * mask[feature_idx]
348                      )
349                      input_data.index_copy_(feature_dim, feature, sparsified)
350                  return input_data
351  
352          return hook

354      def squash_mask(self, attach_sparsify_hook=True, **kwargs):
355          """ 
356          Unregisters aggregate hook that was applied earlier and registers sparsification hooks if 
357          attach_sparsify_hook = True. 
358          """
359          for name, configs in self.data_groups.items():
360              # unhook agg hook
361              configs["hook"].remove()
362              configs.pop("hook")
363              self.data_groups[name]["hook_state"] = "None"
364              if attach_sparsify_hook:
365                  configs["hook"] = configs["layer"].register_forward_pre_hook(
366                      self._sparsify_hook(name)
367                  )
368              configs["hook_state"] = (
369                  "sparsify"  # signals that sparsify hook is now attached
370              )
371  
372      def _get_serializable_data_groups(self):
373          """Exclude hook and layer from the config keys before serializing 
374   
375          TODO: Might have to treat functions (reduce_fn, mask_fn etc) in a different manner while serializing. 
376                For time-being, functions are treated the same way as other attributes 
377          """
378          data_groups: dict[str, Any] = defaultdict()
379          for name, config in self.data_groups.items():
380              new_config = {
381                  key: value
382                  for key, value in config.items()
383                  if key not in ["hook", "layer"]
384              }
385              data_groups[name] = new_config
386          return data_groups

388      def _convert_mask(self, states_dict, sparse_coo=True):
389          r"""Converts the mask to sparse coo or dense depending on the `sparse_coo` argument. 
390          If `sparse_coo=True`, then the mask is stored as sparse coo else dense tensor 
391          """
392          states = copy.deepcopy(states_dict)
393          for state in states.values():
394              if state["mask"] is not None:
395                  if isinstance(state["mask"], list):
396                      for idx in range(len(state["mask"])):
397                          if sparse_coo:
398                              state["mask"][idx] = state["mask"][idx].to_sparse_coo()
399                          else:
400                              state["mask"][idx] = state["mask"][idx].to_dense()
401                  else:
402                      if sparse_coo:
403                          state["mask"] = state["mask"].to_sparse_coo()
404                      else:
405                          state["mask"] = state["mask"].to_dense()
406          return states

408      def state_dict(self) -> dict[str, Any]:
409          r"""Returns the state of the sparsifier as a :class:`dict`. 
410   
411          It contains: 
412          * state - contains name -> mask mapping. 
413          * data_groups - a dictionary containing all config information for each 
414              layer 
415          * defaults - the default config while creating the constructor 
416          """
417          data_groups = self._get_serializable_data_groups()
418          state = self._convert_mask(self.state)
419          return {"state": state, "data_groups": data_groups, "defaults": self.defaults}
420  
421      def load_state_dict(self, state_dict: dict[str, Any]) -> None:
422          r"""The load_state_dict() restores the state of the sparsifier based on the state_dict 
423   
424          Args: 
425          * state_dict - the dictionary that to which the current sparsifier needs to be restored to 
426          """
427          state = state_dict["state"]
428          data_groups, defaults = state_dict["data_groups"], state_dict["defaults"]
429  
430          self.__set_state__(
431              {"state": state, "data_groups": data_groups, "defaults": defaults}
432          )
433  
434      def __get_state__(self) -> dict[str, Any]:
435          data_groups = self._get_serializable_data_groups()
436          state = self._convert_mask(self.state)
437          return {
438              "defaults": self.defaults,
439              "state": state,
440              "data_groups": data_groups,
441          }
442  
443      def __set_state__(self, state: dict[str, Any]) -> None:
444          state["state"] = self._convert_mask(
445              state["state"], sparse_coo=False
446          )  # convert mask to dense tensor
447          self.__dict__.update(state)
448  
449          # need to attach layer and hook info into the data_groups
450          for name, config in self.data_groups.items():
451              # fetch layer
452              layer = fqn_to_module(self.model, name)
453              assert layer is not None  # satisfy mypy
454  
455              # if agg_mode is True, then layer in aggregate mode
456              if "hook_state" in config and config["hook_state"] == "aggregate":
457                  hook = layer.register_forward_pre_hook(self._aggregate_hook(name))
458  
459              elif "hook_state" in config and config["hook_state"] == "sparsify":
460                  hook = layer.register_forward_pre_hook(self._sparsify_hook(name))
461  
462              config["layer"] = layer
463              config["hook"] = hook  # type: ignore[possibly-undefined]

465      def __repr__(self):
466          format_string = self.__class__.__name__ + " ("
467          for name, config in self.data_groups.items():
468              format_string += "\n"
469              format_string += "\tData Group\n"
470              format_string += f"\t    name: {name}\n"
471              for key in sorted(config.keys()):
472                  if key in ["data", "hook", "reduce_fn", "mask_fn", "aggregate_fn"]:
473                      continue
474                  format_string += f"\t    {key}: {config[key]}\n"
475          format_string += ")"
476          return format_string
