1    """ 
2    Contains utility functions for working with nested python data structures. 
3     
4    A *pytree* is Python nested data structure. It is a tree in the sense that 
5    nodes are Python collections (e.g., list, tuple, dict) and the leaves are 
6    Python values. Furthermore, a pytree should not contain reference cycles. 
7     
8    pytrees are useful for working with nested collections of Tensors. For example, 
9    one can use `tree_map` to map a function over all Tensors inside some nested 
10   collection of Tensors and `tree_leaves` to get a flat list of all Tensors 
11   inside some nested collection. pytrees are helpful for implementing nested 
12   collection support for PyTorch APIs. 
13   """

116  def _reverse_args(func: UnflattenFunc) -> OpTreeUnflattenFunc:
117      @functools.wraps(func)
118      def wrapped(*args: Any, **kwargs: Any) -> Any:
119          return func(*reversed(args), **kwargs)
120  
121      return wrapped

124  def register_pytree_node(
125      cls: type[Any],
126      flatten_fn: FlattenFunc,
127      unflatten_fn: UnflattenFunc,
128      *,
129      serialized_type_name: Optional[str] = None,
130      to_dumpable_context: Optional[ToDumpableContextFn] = None,
131      from_dumpable_context: Optional[FromDumpableContextFn] = None,
132      flatten_with_keys_fn: Optional[FlattenWithKeysFunc] = None,
133  ) -> None:
134      """Register a container-like type as pytree node. 
135   
136      Args: 
137          cls (type): A Python type to treat as an internal pytree node. 
138          flatten_fn (callable): A function to be used during flattening, taking an instance of 
139              ``cls`` and returning a pair, with (1) an iterable for the children to be flattened 
140              recursively, and (2) some hashable auxiliary data to be stored in the treespec and to be 
141              passed to the ``unflatten_fn``. 
142          unflatten_fn (callable): A function taking two arguments: the auxiliary data that was 
143              returned by ``flatten_fn`` and stored in the treespec, and the unflattened children. 
144              The function should return an instance of ``cls``. 
145          serialized_type_name (str, optional): A keyword argument used to specify the fully 
146              qualified name used when serializing the tree spec. 
147          to_dumpable_context (callable, optional): An optional keyword argument to custom specify how 
148              to convert the context of the pytree to a custom json dumpable representation. This is 
149              used for json serialization, which is being used in :mod:`torch.export` right now. 
150          from_dumpable_context (callable, optional): An optional keyword argument to custom specify 
151              how to convert the custom json dumpable representation of the context back to the 
152              original context. This is used for json deserialization, which is being used in 
153              :mod:`torch.export` right now. 
154   
155      Example:: 
156   
157          >>> # xdoctest: +SKIP 
158          >>> # Registry a Python type with lambda functions 
159          >>> register_pytree_node( 
160          ...     set, 
161          ...     lambda s: (sorted(s), None, None), 
162          ...     lambda children, _: set(children), 
163          ... ) 
164      """
165      if flatten_with_keys_fn is not None:
166          raise NotImplementedError("KeyPaths are not yet supported in cxx_pytree.")
167  
168      _private_register_pytree_node(
169          cls,
170          flatten_fn,
171          unflatten_fn,
172          serialized_type_name=serialized_type_name,
173          to_dumpable_context=to_dumpable_context,
174          from_dumpable_context=from_dumpable_context,
175      )
176  
177      python_pytree._private_register_pytree_node(
178          cls,
179          flatten_fn,
180          unflatten_fn,
181          serialized_type_name=serialized_type_name,
182          to_dumpable_context=to_dumpable_context,
183          from_dumpable_context=from_dumpable_context,
184      )

192  def _register_pytree_node(
193      cls: type[Any],
194      flatten_fn: FlattenFunc,
195      unflatten_fn: UnflattenFunc,
196      *,
197      serialized_type_name: Optional[str] = None,
198      to_dumpable_context: Optional[ToDumpableContextFn] = None,
199      from_dumpable_context: Optional[FromDumpableContextFn] = None,
200  ) -> None:
201      """Register a container-like type as pytree node for the C++ pytree only. 
202   
203      The ``namespace`` argument is used to avoid collisions that occur when different libraries 
204      register the same Python type with different behaviors. It is recommended to add a unique prefix 
205      to the namespace to avoid conflicts with other libraries. Namespaces can also be used to specify 
206      the same class in different namespaces for different use cases. 
207   
208      .. warning:: 
209          For safety reasons, a ``namespace`` must be specified while registering a custom type. It is 
210          used to isolate the behavior of flattening and unflattening a pytree node type. This is to 
211          prevent accidental collisions between different libraries that may register the same type. 
212   
213      Args: 
214          cls (type): A Python type to treat as an internal pytree node. 
215          flatten_fn (callable): A function to be used during flattening, taking an instance of 
216              ``cls`` and returning a pair, with (1) an iterable for the children to be flattened 
217              recursively, and (2) some hashable auxiliary data to be stored in the treespec and to be 
218              passed to the ``unflatten_fn``. 
219          unflatten_fn (callable): A function taking two arguments: the auxiliary data that was 
220              returned by ``flatten_fn`` and stored in the treespec, and the unflattened children. 
221              The function should return an instance of ``cls``. 
222          serialized_type_name (str, optional): A keyword argument used to specify the fully 
223              qualified name used when serializing the tree spec. 
224          to_dumpable_context (callable, optional): An optional keyword argument to custom specify how 
225              to convert the context of the pytree to a custom json dumpable representation. This is 
226              used for json serialization, which is being used in :mod:`torch.export` right now. 
227          from_dumpable_context (callable, optional): An optional keyword argument to custom specify 
228              how to convert the custom json dumpable representation of the context back to the 
229              original context. This is used for json deserialization, which is being used in 
230              :mod:`torch.export` right now. 
231      """
232  
233      _private_register_pytree_node(
234          cls,
235          flatten_fn,
236          unflatten_fn,
237          serialized_type_name=serialized_type_name,
238          to_dumpable_context=to_dumpable_context,
239          from_dumpable_context=from_dumpable_context,
240      )

243  def _private_register_pytree_node(
244      cls: type[Any],
245      flatten_fn: FlattenFunc,
246      unflatten_fn: UnflattenFunc,
247      *,
248      serialized_type_name: Optional[str] = None,
249      to_dumpable_context: Optional[ToDumpableContextFn] = None,
250      from_dumpable_context: Optional[FromDumpableContextFn] = None,
251  ) -> None:
252      """This is an internal function that is used to register a pytree node type 
253      for the C++ pytree only. End-users should use :func:`register_pytree_node` 
254      instead. 
255      """
256      # TODO(XuehaiPan): remove this condition when we make Python pytree out-of-box support
257      # PyStructSequence types
258      if not optree.is_structseq_class(cls):
259          optree.register_pytree_node(
260              cls,
261              flatten_fn,
262              _reverse_args(unflatten_fn),
263              namespace="torch",
264          )

271  def tree_is_leaf(
272      tree: PyTree,
273      is_leaf: Optional[Callable[[PyTree], bool]] = None,
274  ) -> bool:
275      """Check if a pytree is a leaf. 
276   
277      >>> tree_is_leaf(1) 
278      True 
279      >>> tree_is_leaf(None) 
280      True 
281      >>> tree_is_leaf([1, 2, 3]) 
282      False 
283      >>> tree_is_leaf((1, 2, 3), is_leaf=lambda x: isinstance(x, tuple)) 
284      True 
285      >>> tree_is_leaf({'a': 1, 'b': 2, 'c': 3}) 
286      False 
287      >>> tree_is_leaf({'a': 1, 'b': 2, 'c': None}) 
288      False 
289   
290      Args: 
291          tree (pytree): A pytree to check if it is a leaf node. 
292          is_leaf (callable, optional): An extra leaf predicate function that will be called at each 
293              flattening step. The function should have a single argument with signature 
294              ``is_leaf(node) -> bool``. If it returns :data:`True`, the whole subtree being treated 
295              as a leaf. Otherwise, the default pytree registry will be used to determine a node is a 
296              leaf or not. If the function is not specified, the default pytree registry will be used. 
297   
298      Returns: 
299          A boolean indicating if the pytree is a leaf node. 
300      """
301      return optree.tree_is_leaf(
302          tree,
303          is_leaf=is_leaf,
304          none_is_leaf=True,
305          namespace="torch",
306      )

309  def tree_flatten(
310      tree: PyTree,
311      is_leaf: Optional[Callable[[PyTree], bool]] = None,
312  ) -> tuple[list[Any], TreeSpec]:
313      """Flatten a pytree. 
314   
315      See also :func:`tree_unflatten`. 
316   
317      The flattening order (i.e., the order of elements in the output list) is deterministic, 
318      corresponding to a left-to-right depth-first tree traversal. 
319   
320      >>> tree = {"b": (2, [3, 4]), "a": 1, "c": None, "d": 5} 
321      >>> tree_flatten(tree) 
322      ([2, 3, 4, 1, None, 5], PyTreeSpec({'b': (*, [*, *]), 'a': *, 'c': *, 'd': *}, NoneIsLeaf, namespace='torch')) 
323      >>> tree_flatten(1) 
324      ([1], PyTreeSpec(*, NoneIsLeaf, namespace='torch')) 
325      >>> tree_flatten(None) 
326      ([None], PyTreeSpec(*, NoneIsLeaf, namespace='torch')) 
327      >>> from collections import OrderedDict 
328      >>> tree = OrderedDict([("b", (2, [3, 4])), ("a", 1), ("c", None), ("d", 5)]) 
329      >>> tree_flatten(tree) 
330      ([2, 3, 4, 1, None, 5], PyTreeSpec(OrderedDict({'b': (*, [*, *]), 'a': *, 'c': *, 'd': *}), NoneIsLeaf, namespace='torch')) 
331   
332      Args: 
333          tree (pytree): A pytree to flatten. 
334          is_leaf (callable, optional): An extra leaf predicate function that will be called at each 
335              flattening step. The function should have a single argument with signature 
336              ``is_leaf(node) -> bool``. If it returns :data:`True`, the whole subtree being treated 
337              as a leaf. Otherwise, the default pytree registry will be used to determine a node is a 
338              leaf or not. If the function is not specified, the default pytree registry will be used. 
339   
340      Returns: 
341          A pair ``(leaves, treespec)`` where the first element is a list of leaf values and the 
342          second element is a treespec representing the structure of the pytree. 
343      """
344      return optree.tree_flatten(  # type: ignore[return-value]
345          tree,
346          is_leaf=is_leaf,
347          none_is_leaf=True,
348          namespace="torch",
349      )

352  def tree_unflatten(leaves: Iterable[Any], treespec: TreeSpec) -> PyTree:
353      """Reconstruct a pytree from the treespec and the leaves. 
354   
355      The inverse of :func:`tree_flatten`. 
356   
357      >>> tree = {"b": (2, [3, 4]), "a": 1, "c": None, "d": 5} 
358      >>> leaves, treespec = tree_flatten(tree) 
359      >>> tree == tree_unflatten(leaves, treespec) 
360      True 
361   
362      Args: 
363          leaves (iterable): The list of leaves to use for reconstruction. The list must match the 
364              number of leaves of the treespec. 
365          treespec (TreeSpec): The treespec to reconstruct. 
366   
367      Returns: 
368          The reconstructed pytree, containing the ``leaves`` placed in the structure described by 
369          ``treespec``. 
370      """
371      if not _is_pytreespec_instance(treespec):
372          raise TypeError(
373              f"tree_unflatten(leaves, treespec): Expected `treespec` to be instance of "
374              f"PyTreeSpec but got item of type {type(treespec)}."
375          )
376      return optree.tree_unflatten(treespec, leaves)  # type: ignore[arg-type]

484  def tree_map(
485      func: Callable[..., Any],
486      tree: PyTree,
487      *rests: PyTree,
488      is_leaf: Optional[Callable[[PyTree], bool]] = None,
489  ) -> PyTree:
490      """Map a multi-input function over pytree args to produce a new pytree. 
491   
492      See also :func:`tree_map_`. 
493   
494      >>> tree_map(lambda x: x + 1, {"x": 7, "y": (42, 64)}) 
495      {'x': 8, 'y': (43, 65)} 
496      >>> tree_map(lambda x: x is None, {"x": 7, "y": (42, 64), "z": None}) 
497      {'x': False, 'y': (False, False), 'z': True} 
498   
499      If multiple inputs are given, the structure of the tree is taken from the first input; 
500      subsequent inputs need only have ``tree`` as a prefix: 
501   
502      >>> tree_map(lambda x, y: [x] + y, [5, 6], [[7, 9], [1, 2]]) 
503      [[5, 7, 9], [6, 1, 2]] 
504   
505      Args: 
506          func (callable): A function that takes ``1 + len(rests)`` arguments, to be applied at the 
507              corresponding leaves of the pytrees. 
508          tree (pytree): A pytree to be mapped over, with each leaf providing the first positional 
509              argument to function ``func``. 
510          rests (tuple of pytree): A tuple of pytrees, each of which has the same structure as 
511              ``tree`` or has ``tree`` as a prefix. 
512          is_leaf (callable, optional): An extra leaf predicate function that will be called at each 
513              flattening step. The function should have a single argument with signature 
514              ``is_leaf(node) -> bool``. If it returns :data:`True`, the whole subtree being treated 
515              as a leaf. Otherwise, the default pytree registry will be used to determine a node is a 
516              leaf or not. If the function is not specified, the default pytree registry will be used. 
517   
518      Returns: 
519          A new pytree with the same structure as ``tree`` but with the value at each leaf given by 
520          ``func(x, *xs)`` where ``x`` is the value at the corresponding leaf in ``tree`` and ``xs`` 
521          is the tuple of values at corresponding nodes in ``rests``. 
522      """
523      return optree.tree_map(
524          func,
525          tree,
526          *rests,
527          is_leaf=is_leaf,
528          none_is_leaf=True,
529          namespace="torch",
530      )

533  def tree_map_(
534      func: Callable[..., Any],
535      tree: PyTree,
536      *rests: PyTree,
537      is_leaf: Optional[Callable[[PyTree], bool]] = None,
538  ) -> PyTree:
539      """Like :func:`tree_map`, but do an inplace call on each leaf and return the original tree. 
540   
541      See also :func:`tree_map`. 
542   
543      Args: 
544          func (callable): A function that takes ``1 + len(rests)`` arguments, to be applied at the 
545              corresponding leaves of the pytrees. 
546          tree (pytree): A pytree to be mapped over, with each leaf providing the first positional 
547              argument to function ``func``. 
548          rests (tuple of pytree): A tuple of pytrees, each of which has the same structure as 
549              ``tree`` or has ``tree`` as a prefix. 
550          is_leaf (callable, optional): An extra leaf predicate function that will be called at each 
551              flattening step. The function should have a single argument with signature 
552              ``is_leaf(node) -> bool``. If it returns :data:`True`, the whole subtree being treated 
553              as a leaf. Otherwise, the default pytree registry will be used to determine a node is a 
554              leaf or not. If the function is not specified, the default pytree registry will be used. 
555   
556      Returns: 
557          The original ``tree`` with the value at each leaf is given by the side-effect of function 
558          ``func(x, *xs)`` (not the return value) where ``x`` is the value at the corresponding leaf 
559          in ``tree`` and ``xs`` is the tuple of values at values at corresponding nodes in ``rests``. 
560      """
561      return optree.tree_map_(
562          func,
563          tree,
564          *rests,
565          is_leaf=is_leaf,
566          none_is_leaf=True,
567          namespace="torch",
568      )

614  def map_only(
615      type_or_types_or_pred: Union[TypeAny, Callable[[Any], bool]], /
616  ) -> MapOnlyFn[FnAny[Any]]:
617      """ 
618      Suppose you are writing a tree_map over tensors, leaving everything 
619      else unchanged.  Ordinarily you would have to write: 
620   
621          def go(t): 
622              if isinstance(t, Tensor): 
623                  return ... 
624              else: 
625                  return t 
626   
627      With this function, you only need to write: 
628   
629          @map_only(Tensor) 
630          def go(t): 
631              return ... 
632   
633      You can also directly use 'tree_map_only' 
634      """
635      if isinstance(type_or_types_or_pred, (type, tuple)) or (
636          sys.version_info >= (3, 10)
637          and isinstance(type_or_types_or_pred, types.UnionType)
638      ):
639  
640          def pred(x: Any) -> bool:
641              return isinstance(x, type_or_types_or_pred)  # type: ignore[arg-type]
642  
643      elif callable(type_or_types_or_pred):
644          pred = type_or_types_or_pred  # type: ignore[assignment]
645      else:
646          raise TypeError("Argument must be a type, a tuple of types, or a callable.")
647  
648      def wrapper(func: Callable[[T], Any]) -> Callable[[Any], Any]:
649          @functools.wraps(func)
650          def wrapped(x: T) -> Any:
651              if pred(x):
652                  return func(x)
653              return x
654  
655          return wrapped
656  
657      return wrapper

896  def broadcast_prefix(
897      prefix_tree: PyTree,
898      full_tree: PyTree,
899      is_leaf: Optional[Callable[[PyTree], bool]] = None,
900  ) -> list[Any]:
901      """Return a list of broadcasted leaves in ``prefix_tree`` to match the number of leaves in ``full_tree``. 
902   
903      If a ``prefix_tree`` is a prefix of a ``full_tree``, this means the ``full_tree`` can be 
904      constructed by replacing the leaves of ``prefix_tree`` with appropriate **subtrees**. 
905   
906      This function returns a list of leaves with the same size as ``full_tree``. The leaves are 
907      replicated from ``prefix_tree``. The number of replicas is determined by the corresponding 
908      subtree in ``full_tree``. 
909   
910      >>> broadcast_prefix(1, [1, 2, 3]) 
911      [1, 1, 1] 
912      >>> broadcast_prefix([1, 2, 3], [1, 2, 3]) 
913      [1, 2, 3] 
914      >>> broadcast_prefix([1, 2, 3], [1, 2, 3, 4]) 
915      Traceback (most recent call last): 
916          ... 
917      ValueError: list arity mismatch; expected: 3, got: 4; list: [1, 2, 3, 4]. 
918      >>> broadcast_prefix([1, 2, 3], [1, 2, (3, 4)]) 
919      [1, 2, 3, 3] 
920      >>> broadcast_prefix([1, 2, 3], [1, 2, {"a": 3, "b": 4, "c": (None, 5)}]) 
921      [1, 2, 3, 3, 3, 3] 
922   
923      Args: 
924          prefix_tree (pytree): A pytree with the same structure as a prefix of ``full_tree``. 
925          full_tree (pytree): A pytree with the same structure as a suffix of ``prefix_tree``. 
926          is_leaf (callable, optional): An extra leaf predicate function that will be called at each 
927              flattening step. The function should have a single argument with signature 
928              ``is_leaf(node) -> bool``. If it returns :data:`True`, the whole subtree being treated 
929              as a leaf. Otherwise, the default pytree registry will be used to determine a node is a 
930              leaf or not. If the function is not specified, the default pytree registry will be used. 
931   
932      Returns: 
933          A list of leaves in ``prefix_tree`` broadcasted to match the number of leaves in ``full_tree``. 
934      """
935      result: list[Any] = []
936  
937      def add_leaves(x: Any, subtree: PyTree) -> None:
938          subtreespec = tree_structure(subtree, is_leaf=is_leaf)
939          result.extend([x] * subtreespec.num_leaves)
940  
941      tree_map_(
942          add_leaves,
943          prefix_tree,
944          full_tree,
945          is_leaf=is_leaf,
946      )
947      return result
948

950  # Broadcasts a pytree to the provided TreeSpec and returns the flattened
951  # values. If this is not possible, then this function returns None.
952  #
953  # For example, given pytree=0 and spec=TreeSpec(list, None, [LeafSpec(), LeafSpec()]),
954  # would return [0, 0]. This is useful for part of the vmap implementation:
955  # a user can pass in vmap(fn, in_dims)(*inputs). `in_dims` should be
956  # broadcastable to the tree structure of `inputs` and we use
957  # _broadcast_to_and_flatten to check this.
958  def _broadcast_to_and_flatten(
959      tree: PyTree,
960      treespec: TreeSpec,
961      is_leaf: Optional[Callable[[PyTree], bool]] = None,
962  ) -> Optional[list[Any]]:
963      assert _is_pytreespec_instance(treespec)
964      full_tree = tree_unflatten([0] * treespec.num_leaves, treespec)
965      try:
966          return broadcast_prefix(tree, full_tree, is_leaf=is_leaf)
967      except ValueError:
968          return None

971  def treespec_dumps(treespec: TreeSpec, protocol: Optional[int] = None) -> str:
972      """Serialize a treespec to a JSON string."""
973      if not _is_pytreespec_instance(treespec):
974          raise TypeError(
975              f"treespec_dumps(treespec): Expected `treespec` to be instance of "
976              f"PyTreeSpec but got item of type {type(treespec)}."
977          )
978  
979      dummy_tree = tree_unflatten([0] * treespec.num_leaves, treespec)
980      orig_treespec = python_pytree.tree_structure(dummy_tree)
981      return python_pytree.treespec_dumps(orig_treespec, protocol=protocol)
