13   def test_parser():
14       assert Abs(parse_maxima('float(1/3)') - 0.333333333) < 10**(-5)
15       assert parse_maxima('13^26') == 91733330193268616658399616009
16       assert parse_maxima('sin(%pi/2) + cos(%pi/3)') == Rational(3, 2)
17       assert parse_maxima('log(%e)') == 1

22       # c created by parse_maxima

30   def test_maxima_functions():
31       assert parse_maxima('expand( (x+1)^2)') == x**2 + 2*x + 1
32       assert parse_maxima('factor( x**2 + 2*x + 1)') == (x + 1)**2
33       assert parse_maxima('2*cos(x)^2 + sin(x)^2') == 2*cos(x)**2 + sin(x)**2
34       assert parse_maxima('trigexpand(sin(2*x)+cos(2*x))') == \
35           -1 + 2*cos(x)**2 + 2*cos(x)*sin(x)
36       assert parse_maxima('solve(x^2-4,x)') == [-2, 2]
37       assert parse_maxima('limit((1+1/x)^x,x,inf)') == E
38       assert parse_maxima('limit(sqrt(-x)/x,x,0,minus)') is -oo
39       assert parse_maxima('diff(x^x, x)') == x**x*(1 + log(x))
40       assert parse_maxima('sum(k, k, 1, n)', name_dict={
41           "n": Symbol('n', integer=True),
42           "k": Symbol('k', integer=True)
43       }) == (n**2 + n)/2
44       assert parse_maxima('product(k, k, 1, n)', name_dict={
45           "n": Symbol('n', integer=True),
46           "k": Symbol('k', integer=True)
47       }) == factorial(n)
48       assert parse_maxima('ratsimp((x^2-1)/(x+1))') == x - 1
49       assert Abs( parse_maxima(
50           'float(sec(%pi/3) + csc(%pi/3))') - 3.154700538379252) < 10**(-5)
