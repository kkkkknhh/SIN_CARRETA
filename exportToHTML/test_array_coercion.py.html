21   def arraylikes():
22       """ 
23       Generator for functions converting an array into various array-likes. 
24       If full is True (default) it includes array-likes not capable of handling 
25       all dtypes. 
26       """
27       # base array:
28       def ndarray(a):
29           return a
30   
31       yield param(ndarray, id="ndarray")
32   
33       # subclass:
34       class MyArr(np.ndarray):
35           pass
36   
37       def subclass(a):
38           return a.view(MyArr)
39   
40       yield subclass
41   
42       class _SequenceLike:
43           # Older NumPy versions, sometimes cared whether a protocol array was
44           # also _SequenceLike.  This shouldn't matter, but keep it for now
45           # for __array__ and not the others.
46           def __len__(self):
47               raise TypeError
48   
49           def __getitem__(self):
50               raise TypeError
51   
52       # Array-interface
53       class ArrayDunder(_SequenceLike):
54           def __init__(self, a):
55               self.a = a
56   
57           def __array__(self, dtype=None, copy=None):
58               if dtype is None:
59                   return self.a
60               return self.a.astype(dtype)
61   
62       yield param(ArrayDunder, id="__array__")
63   
64       # memory-view
65       yield param(memoryview, id="memoryview")
66   
67       # Array-interface
68       class ArrayInterface:
69           def __init__(self, a):
70               self.a = a  # need to hold on to keep interface valid
71               self.__array_interface__ = a.__array_interface__
72   
73       yield param(ArrayInterface, id="__array_interface__")
74   
75       # Array-Struct
76       class ArrayStruct:
77           def __init__(self, a):
78               self.a = a  # need to hold on to keep struct valid
79               self.__array_struct__ = a.__array_struct__
80   
81       yield param(ArrayStruct, id="__array_struct__")

85       # Hard-coded list of scalar instances.
86       # Floats:

97           yield param(np.sqrt(np.clongdouble(2+3j)), id="clongdouble")

100      # XFAIL: Bool should be added, but has some bad properties when it
101      # comes to strings, see also gh-9875
102      # yield param(np.bool(0), id="bool")

104      # Integers:
105      yield param(np.int8(2), id="int8")
106      yield param(np.int16(2), id="int16")
107      yield param(np.int32(2), id="int32")
108      yield param(np.int64(2), id="int64")

110      yield param(np.uint8(2), id="uint8")
111      yield param(np.uint16(2), id="uint16")
112      yield param(np.uint32(2), id="uint32")
113      yield param(np.uint64(2), id="uint64")

116      if user_dtype:

140  def is_parametric_dtype(dtype):
141      """Returns True if the dtype is a parametric legacy dtype (itemsize 
142      is 0, or a datetime without units) 
143      """
144      if dtype.itemsize == 0:
145          return True
146      if issubclass(dtype.type, (np.datetime64, np.timedelta64)):
147          if dtype.name.endswith("64"):
148              # Generic time units
149              return True
150      return False

264      @pytest.mark.parametrize("scalar", scalar_instances())
265      def test_scalar_coercion(self, scalar):
266          # This tests various scalar coercion paths, mainly for the numerical
267          # types. It includes some paths not directly related to `np.array`.
268          if isinstance(scalar, np.inexact):
269              # Ensure we have a full-precision number if available
270              scalar = type(scalar)((scalar * 2)**0.5)
271  
272          if type(scalar) is rational:
273              # Rational generally fails due to a missing cast. In the future
274              # object casts should automatically be defined based on `setitem`.
275              pytest.xfail("Rational to object cast is undefined currently.")
276  
277          # Use casting from object:
278          arr = np.array(scalar, dtype=object).astype(scalar.dtype)
279  
280          # Test various ways to create an array containing this scalar:
281          arr1 = np.array(scalar).reshape(1)
282          arr2 = np.array([scalar])
283          arr3 = np.empty(1, dtype=scalar.dtype)
284          arr3[0] = scalar
285          arr4 = np.empty(1, dtype=scalar.dtype)
286          arr4[:] = [scalar]
287          # All of these methods should yield the same results
288          assert_array_equal(arr, arr1)
289          assert_array_equal(arr, arr2)
290          assert_array_equal(arr, arr3)
291          assert_array_equal(arr, arr4)

293      @pytest.mark.xfail(IS_PYPY, reason="`int(np.complex128(3))` fails on PyPy")
294      @pytest.mark.filterwarnings("ignore::numpy.exceptions.ComplexWarning")
295      @pytest.mark.parametrize("cast_to", scalar_instances())
296      def test_scalar_coercion_same_as_cast_and_assignment(self, cast_to):
297          """ 
298          Test that in most cases: 
299             * `np.array(scalar, dtype=dtype)` 
300             * `np.empty((), dtype=dtype)[()] = scalar` 
301             * `np.array(scalar).astype(dtype)` 
302          should behave the same.  The only exceptions are parametric dtypes 
303          (mainly datetime/timedelta without unit) and void without fields. 
304          """
305          dtype = cast_to.dtype  # use to parametrize only the target dtype
306  
307          for scalar in scalar_instances(times=False):
308              scalar = scalar.values[0]
309  
310              if dtype.type == np.void:
311                 if scalar.dtype.fields is not None and dtype.fields is None:
312                      # Here, coercion to "V6" works, but the cast fails.
313                      # Since the types are identical, SETITEM takes care of
314                      # this, but has different rules than the cast.
315                      with pytest.raises(TypeError):
316                          np.array(scalar).astype(dtype)
317                      np.array(scalar, dtype=dtype)
318                      np.array([scalar], dtype=dtype)
319                      continue
320  
321              # The main test, we first try to use casting and if it succeeds
322              # continue below testing that things are the same, otherwise
323              # test that the alternative paths at least also fail.
324              try:
325                  cast = np.array(scalar).astype(dtype)
326              except (TypeError, ValueError, RuntimeError):
327                  # coercion should also raise (error type may change)
328                  with pytest.raises(Exception):
329                      np.array(scalar, dtype=dtype)
330  
331                  if (isinstance(scalar, rational) and
332                          np.issubdtype(dtype, np.signedinteger)):
333                      return
334  
335                  with pytest.raises(Exception):
336                      np.array([scalar], dtype=dtype)
337                  # assignment should also raise
338                  res = np.zeros((), dtype=dtype)
339                  with pytest.raises(Exception):
340                      res[()] = scalar
341  
342                  return
343  
344              # Non error path:
345              arr = np.array(scalar, dtype=dtype)
346              assert_array_equal(arr, cast)
347              # assignment behaves the same
348              ass = np.zeros((), dtype=dtype)
349              ass[()] = scalar
350              assert_array_equal(ass, cast)

519      @pytest.mark.parametrize("arraylike", arraylikes())
520      def test_nested_arraylikes(self, arraylike):
521          # We try storing an array like into an array, but the array-like
522          # will have too many dimensions.  This means the shape discovery
523          # decides that the array-like must be treated as an object (a special
524          # case of ragged discovery).  The result will be an array with one
525          # dimension less than the maximum dimensions, and the array being
526          # assigned to it (which does work for object or if `float(arraylike)`
527          # works).
528          initial = arraylike(np.ones((1, 1)))
529  
530          nested = initial
531          for i in range(ncu.MAXDIMS - 1):
532              nested = [nested]
533  
534          with pytest.raises(ValueError, match=".*would exceed the maximum"):
535              # It will refuse to assign the array into
536              np.array(nested, dtype="float64")
537  
538          # If this is object, we end up assigning a (1, 1) array into (1,)
539          # (due to running out of dimensions), this is currently supported but
540          # a special case which is not ideal.
541          arr = np.array(nested, dtype=object)
542          assert arr.shape == (1,) * ncu.MAXDIMS
543          assert arr.item() == np.array(initial).item()

777  class TestAsArray:
778      """Test expected behaviors of ``asarray``."""
779  
780      def test_dtype_identity(self):
781          """Confirm the intended behavior for *dtype* kwarg. 
782   
783          The result of ``asarray()`` should have the dtype provided through the 
784          keyword argument, when used. This forces unique array handles to be 
785          produced for unique np.dtype objects, but (for equivalent dtypes), the 
786          underlying data (the base object) is shared with the original array 
787          object. 
788   
789          Ref https://github.com/numpy/numpy/issues/1468 
790          """
791          int_array = np.array([1, 2, 3], dtype='i')
792          assert np.asarray(int_array) is int_array
793  
794          # The character code resolves to the singleton dtype object provided
795          # by the numpy package.
796          assert np.asarray(int_array, dtype='i') is int_array
797  
798          # Derive a dtype from n.dtype('i'), but add a metadata object to force
799          # the dtype to be distinct.
800          unequal_type = np.dtype('i', metadata={'spam': True})
801          annotated_int_array = np.asarray(int_array, dtype=unequal_type)
802          assert annotated_int_array is not int_array
803          assert annotated_int_array.base is int_array
804          # Create an equivalent descriptor with a new and distinct dtype
805          # instance.
806          equivalent_requirement = np.dtype('i', metadata={'spam': True})
807          annotated_int_array_alt = np.asarray(annotated_int_array,
808                                               dtype=equivalent_requirement)
809          assert unequal_type == equivalent_requirement
810          assert unequal_type is not equivalent_requirement
811          assert annotated_int_array_alt is not annotated_int_array
812          assert annotated_int_array_alt.dtype is equivalent_requirement
813  
814          # Check the same logic for a pair of C types whose equivalence may vary
815          # between computing environments.
816          # Find an equivalent pair.
817          integer_type_codes = ('i', 'l', 'q')
818          integer_dtypes = [np.dtype(code) for code in integer_type_codes]
819          typeA = None
820          typeB = None
821          for typeA, typeB in permutations(integer_dtypes, r=2):
822              if typeA == typeB:
823                  assert typeA is not typeB
824                  break
825          assert isinstance(typeA, np.dtype) and isinstance(typeB, np.dtype)
826  
827          # These ``asarray()`` calls may produce a new view or a copy,
828          # but never the same object.
829          long_int_array = np.asarray(int_array, dtype='l')
830          long_long_int_array = np.asarray(int_array, dtype='q')
831          assert long_int_array is not int_array
832          assert long_long_int_array is not int_array
833          assert np.asarray(long_int_array, dtype='q') is not long_int_array
834          array_a = np.asarray(int_array, dtype=typeA)
835          assert typeA == typeB
836          assert typeA is not typeB
837          assert array_a.dtype is typeA
838          assert array_a is not np.asarray(array_a, dtype=typeB)
839          assert np.asarray(array_a, dtype=typeB).dtype is typeB
840          assert array_a is np.asarray(array_a, dtype=typeB).base
