17   template <typename T>
18   inline float erf(T x) {
19     const auto a = static_cast<float>(x);
20     const auto t = ::metal::abs(a);
21     const auto s = a * a;
22     if (t > 0.927734375f) {
23       // maximum error 0.99527 ulp
24       auto r = ::metal::fma(
25           -1.72853470e-5f, t, 3.83197126e-4f); // -0x1.220000p-16,0x1.91cfb2p-12
26       const auto u = ::metal::fma(
27           -3.88396438e-3f, t, 2.42546219e-2f); // -0x1.fd1438p-9, 0x1.8d6342p-6
28       r = ::metal::fma(r, s, u);
29       r = ::metal::fma(r, t, -1.06777877e-1f); // -0x1.b55cb8p-4
30       r = ::metal::fma(r, t, -6.34846687e-1f); // -0x1.450aa0p-1
31       r = ::metal::fma(r, t, -1.28717512e-1f); // -0x1.079d0cp-3
32       r = ::metal::fma(r, t, -t);
33       // TODO, replace with expm1 when implemented
34       r = 1.0f - ::metal::exp(r);
35       r = ::metal::copysign(r, a);
36       return r;
37     }
38   
39     // maximum error 0.98929 ulp
40     auto r = -5.96761703e-4f; // -0x1.38e000p-11
41     r = ::metal::fma(r, s, 4.99119423e-3f); //  0x1.471a58p-8
42     r = ::metal::fma(r, s, -2.67681349e-2f); // -0x1.b691b2p-6
43     r = ::metal::fma(r, s, 1.12819925e-1f); //  0x1.ce1c44p-4
44     r = ::metal::fma(r, s, -3.76125336e-1f); // -0x1.812700p-2
45     r = ::metal::fma(r, s, 1.28379166e-1f); //  0x1.06eba8p-3
46     r = ::metal::fma(r, a, a);
47     return r;
48   }

50   template <typename T>
51   inline float erfinv(T y) {
52     /* coefficients in rational expansion */
53     constexpr float a[4] = {0.886226899, -1.645349621, 0.914624893, -0.140543331};
54     constexpr float b[4] = {-2.118377725, 1.442710462, -0.329097515, 0.012229801};
55     constexpr float c[4] = {-1.970840454, -1.624906493, 3.429567803, 1.641345311};
56     constexpr float d[2] = {3.543889200, 1.637067800};
57   
58     float x, z, num, dem; /*working variables */
59   
60     float y_abs = ::metal::abs(static_cast<float>(y));
61     if (y_abs >= 1.0f) {
62       return y_abs > 1.0f ? NAN
63                           : ::metal::copysign(INFINITY, static_cast<float>(y));
64     }
65     if (y_abs <= 0.7f) {
66       z = y * y;
67       num = ((a[3] * z + a[2]) * z + a[1]) * z + a[0];
68       dem = (((b[3] * z + b[2]) * z + b[1]) * z + b[0]) * z + 1.0f;
69       x = y * num / dem;
70     } else {
71       z = ::metal::sqrt(-1.0f * ::metal::log((1.0 - y_abs) / 2.0));
72       num = ((c[3] * z + c[2]) * z + c[1]) * z + c[0];
73       dem = (d[1] * z + d[0]) * z + 1.0f;
74       x = ::metal::copysign(num, static_cast<float>(y)) / dem;
75     }
76   
77     return x;
78   }

85   template <typename T>
86   inline T chbevl(T x, const float array[], const int len) {
87     T b0, b1, b2;
88   
89     b0 = array[0];
90     b1 = 0;
91   
92     for (int i = 1; i < len; ++i) {
93       b2 = b1;
94       b1 = b0;
95       b0 = x * b1 - b2 + array[i];
96     }
97   
98     return T{0.5} * (b0 - b2);
99   }

311  template <typename T>
312  inline float gamma(const T x) {
313    if (x < 0.001) {
314      constexpr float EULER_MASCHERONI = 0.577215664901532860606512090;
315      // For small x, 1/gamma(x) has power series x + gamma x^2  - ...
316      // So in this range, 1/gamma(x) = x + gamma x^2 with error on the order of
317      // x^3. The relative error over this interval is less than 6e-7.
318  
319      return 1.0 / (x * (1.0 + EULER_MASCHERONI * x));
320    }
321    if (x >= 12.0) {
322      return ::metal::exp(log_gamma(x));
323    }
324    // The algorithm directly approximates gamma over (1,2) and uses
325    // reduction identities to reduce other arguments to this interval.
326    // numerator coefficients for gamma approximation over the interval (1,2)
327    constexpr float GAMMA_NUMERATOR_COEF[8] = {
328        -1.71618513886549492533811E+0,
329        2.47656508055759199108314E+1,
330        -3.79804256470945635097577E+2,
331        6.29331155312818442661052E+2,
332        8.66966202790413211295064E+2,
333        -3.14512729688483675254357E+4,
334        -3.61444134186911729807069E+4,
335        6.64561438202405440627855E+4};
336  
337    // denominator coefficients for gamma approximation over the interval (1,2)
338    constexpr float GAMMA_DENOMINATOR_COEF[8] = {
339        -3.08402300119738975254353E+1,
340        3.15350626979604161529144E+2,
341        -1.01515636749021914166146E+3,
342        -3.10777167157231109440444E+3,
343        2.25381184209801510330112E+4,
344        4.75584627752788110767815E+3,
345        -1.34659959864969306392456E+5,
346        -1.15132259675553483497211E+5};
347  
348    // Add or subtract integers as necessary to bring y into (1,2)
349    float y = 1.0 + ::metal::fract(x);
350  
351    float num = 0.0;
352    float den = 1.0;
353  
354    float z = y - 1;
355    for (int i = 0; i < 8; i++) {
356      num = (num + GAMMA_NUMERATOR_COEF[i]) * z;
357      den = den * z + GAMMA_DENOMINATOR_COEF[i];
358    }
359    float result = num / den + 1.0;
360  
361    // Apply correction if argument was not initially in (1,2)
362    if (x < 1.0) {
363      // identity gamma(z) = gamma(z+1)/z
364      result /= (y - 1.0);
365    } else {
366      // identity gamma(z+n) = z*(z+1)* ... *(z+n-1)*gamma(z)
367      auto n = static_cast<int>(::metal::floor(x));
368      for (int i = 1; i < n; i++) {
369        result *= y++;
370      }
371    }
372  
373    return result;
374  }

376  template <typename T>
377  inline float log_gamma(const T x) {
378    constexpr float LOG_PI = 1.14472988584940017414342735135305;
379    constexpr float HALF_LOG_TWO_PI = 0.91893853320467274178032973640562;
380    constexpr float LGAMMA_EXPANSION_COEF[8] = {
381        1.0 / 12.0,
382        -1.0 / 360.0,
383        1.0 / 1260.0,
384        -1.0 / 1680.0,
385        1.0 / 1188.0,
386        -691.0 / 360360.0,
387        1.0 / 156.0,
388        -3617.0 / 122400.0};
389  
390    float rc;
391  
392    const auto abs_x = ::metal::abs(static_cast<float>(x));
393    if (abs_x == 0) {
394      return INFINITY;
395    }
396    if (abs_x < 12.0) {
397      rc = ::metal::log(::metal::abs(gamma(abs_x)));
398    } else {
399      // Abramowitz and Stegun 6.1.41
400      // Asymptotic series should be good to at least 11 or 12 figures
401      // For error analysis, see Whittiker and Watson
402      // A Course in Modern Analysis (1927), page 252
403  
404      float z = 1.0 / (abs_x * abs_x);
405      float sum = LGAMMA_EXPANSION_COEF[7];
406  
407      for (int i = 6; i >= 0; i--) {
408        sum *= z;
409        sum += LGAMMA_EXPANSION_COEF[i];
410      }
411      float series = sum / abs_x;
412  
413      rc = (abs_x - 0.5) * ::metal::log(abs_x) - abs_x + HALF_LOG_TWO_PI + series;
414    }
415  
416    if (x >= 0) {
417      return rc;
418    }

420    // Reflection formula

496  inline float calc_digamma_positive_domain(float x) {
497    constexpr float DIGAMMA_COEF[7] = {
498        8.33333333333333333333E-2,
499        -2.10927960927960927961E-2,
500        7.57575757575757575758E-3,
501        -4.16666666666666666667E-3,
502        3.96825396825396825397E-3,
503        -8.33333333333333333333E-3,
504        8.33333333333333333333E-2,
505    };

507    // Push x to be >= 10
508    float result = 0;
509    while (x < 10) {
510      result -= 1 / x;
511      x += 1;
512    }

515      return result + PSI_10;
516    }

518    // Compute asymptotic digamma
519    float y = 0;

525      y *= z;
526    }

530  template <typename T0>
531  inline float digamma(T0 x) {
532    if (x < 0.0f) {
533      if (x == ::metal::trunc(x)) {
534        // As per C++ standard for gamma related functions and SciPy,
535        // If the argument is a negative integer, NaN is returned
536        return NAN;
537      } else {
538        // Extracts the fractional part of x as r, since tan(pi * r) is more
539        // numerically accurate than tan(pi * x). While these operations are
540        // mathematically equivalent since both x and r are in radians and tan()
541        // has a periodicity of pi, in practice the computation of pi * x is a
542        // source of error (when |x| > 1).
543        float r = ::metal::fract(x);
544        return calc_digamma_positive_domain(1.0f - x) -
545            M_PI_F / ::metal::tan(M_PI_F * r);
546      }
547    } else if (x == 0.0f) {
548      // As per C++ standard for gamma related functions and SciPy,
549      // If the argument is 0,  is returned
550      return ::metal::copysign(INFINITY, static_cast<float>(-x));
551    } else {
552      return calc_digamma_positive_domain(x);
553    }
554  }

557  inline float polygamma(const int64_t order, const T0 input) {
558    // Filter out n == 0.
559    if (order == 0) {
560      return digamma(input);
561    }

563    float x = input;

565    float sgn = ((order % 2) ? 1 : -1);
566    return sgn * gamma(n + 1) * zeta(n + 1, x);

581    auto a = static_cast<float2>(inp) * M_PI_F;

583    if (a2 == 0) {

587    float sinx = ::metal::sincos(a.x, cosx);
588    float sinhy = ::metal::sinh(a.y);
589    float coshy = ::metal::cosh(a.y);
590    auto re = sinx * coshy * a.x + cosx * sinhy * a.y;
591    auto im = cosx * sinhy * a.x - sinx * coshy * a.y;
592    return T(re, im) / a2;
