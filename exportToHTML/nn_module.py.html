3    """ 
4    This module implements variable tracking for PyTorch nn.Module instances during Dynamo tracing. 
5     
6    It provides specialized handling for different types of nn.Module instances through several key classes: 
7     
8    - NNModuleVariable: Handles instance-specific module tracing, specializing on module id() and placing 
9      parameters directly on the torch.fx.GraphModule. This creates one graph per module instance. 
10    
11   - UnspecializedNNModuleVariable: Provides class-level module tracing, treating nn.Modules like other 
12     user-defined objects and passing parameters as inputs to the FX graph. This creates one graph per 
13     module class. 
14    
15   - UnspecializedBuiltinNNModuleVariable: Specifically handles built-in PyTorch modules (e.g. nn.Linear) 
16     with appropriate optimizations. 
17    
18   - FSDPManagedNNModuleVariable: Special handling for FSDP-wrapped modules with modified guarding behavior 
19     and parameter handling. 
20    
21   The module integrates with Dynamo's broader tracing functionality to handle module method calls, 
22   parameter access, hooks, and other nn.Module behaviors while maintaining proper scoping and guarding 
23   of module state. 
24   """

79   def initialize_lazy_module(tx: "InstructionTranslator", mod, args, kwargs):
80       """ 
81       Fairly coupled helper used by NNModuleVariable and UnspecializedNNModuleVariable. 
82    
83       Used to cause lazy module to be initialized (and delete its init hook) before tracing. Especially 
84       useful now that 'allowed' modules graph-break on hooks, calling this first ensures there is no hook 
85       by the time we trace __call__ and thus no graph-break for lazy allowed modules. 
86       """
87       if hasattr(mod, "_initialize_hook"):
88   
89           def convert_to_fake(x):
90               if is_namedtuple(x):
91                   return type(x)(*(convert_to_fake(elem) for elem in x))
92               elif isinstance(x, dict):
93                   return {k: convert_to_fake(v) for k, v in x.items()}
94               elif isinstance(x, (list, tuple, set)):
95                   return type(x)(convert_to_fake(elem) for elem in x)
96               elif isinstance(x, torch.fx.Proxy):
97                   return get_fake_value(x.node, tx)
98               else:
99                   return x
100  
101          proxy_args, proxy_kwargs = proxy_args_kwargs(args, kwargs)
102          fake_args = [convert_to_fake(arg) for arg in proxy_args]
103          fake_kwargs = {k: convert_to_fake(v) for k, v in proxy_kwargs.items()}
104          mod._infer_parameters(mod, fake_args, fake_kwargs)

107  @contextmanager
108  def record_nn_module_stack(module_key: str, source, tx, mod: torch.nn.Module):
109      fully_qualified_name = source.name()
110      num_calls = tx.num_calls.get(fully_qualified_name, 0)
111      module_key = f"{module_key}@{num_calls}" if num_calls > 0 else module_key
112      try:
113          tx.nn_module_stack[module_key] = (fully_qualified_name, mod.__class__)
114          tx.num_calls[fully_qualified_name] = num_calls + 1
115          yield
116      finally:
117          del tx.nn_module_stack[module_key]

120  def guard_to_detect_forward_monkeypatching(source, mod):
121      # Users sometimes patch the forward method of a nn module instance to
122      # perform optimizations like quantization. Though this is not a good
123      # software practice, but python allows this and Dynamo needs to detect
124      # this patching.
125      #
126      # One way to do this is to add an ID_MATCH guard on every function
127      # getting inlined (https://github.com/pytorch/pytorch/pull/124975). But
128      # this increased guard overhead by around 20%.
129      #
130      # To keep the guard overhead down, we just guard on the `forward` being
131      # not present in the mod __dict__. The common case of patching forward
132      # method adds `forward` in the instance __dict__, whereas the unpatched
133      # `forward` sits in the type(mod).__dict__
134      if source:
135          if "forward" in mod.__dict__ and callable(mod.__dict__["forward"]):
136              # Monkeypatched forward method, add an ID_MATCH guard on forward function
137              fwd = mod.__dict__["forward"]
138              forward_source = AttrSource(source, "forward")
139              if type(fwd) is types.MethodType:
140                  forward_source = AttrSource(forward_source, "__func__")
141              install_guard(forward_source.make_guard(GuardBuilder.CLOSURE_MATCH))
142          else:
143              # Common case - check that the forward key is absent in mod __dict__
144              install_guard(
145                  source.make_guard(
146                      functools.partial(
147                          GuardBuilder.NOT_PRESENT_IN_GENERIC_DICT, attr="forward"
148                      )
149                  )
150              )

154      _nonvar_fields = {
155          "module_type",
156          "module_key",
157          "value",
158          "nn_module_stack_source",

162      def __init__(
163          self, module_type: type, module_key: str, value: torch.nn.Module, **kwargs
164      ) -> None:
165          super().__init__(**kwargs)
166          self.module_type = module_type
167          self.module_key = module_key
168          self.value = value
169          assert self.source
170          self.nn_module_stack_source = self.source

179          return self.module_type

186      def unpack_var_sequence(self, tx):
187          # implement list/iter/tuple/etc calls
188          base = tx.output.get_submodule(self.module_key)
189          if isinstance(base, torch.nn.ModuleDict):
190              result = []
191              for name, submod in base.items():
192                  name_var = variables.ConstantVariable.create(name)
193                  tx.output.register_attr_or_module(
194                      submod,
195                      self.module_key,
196                      name,
197                      source=NNModuleSource(GetItemSource(self.source, name)),
198                  )
199                  result.append(name_var)
200              return result
201  
202          assert isinstance(
203              base, (torch.nn.ModuleList, torch.nn.ParameterList, torch.nn.Sequential)
204          ), typestr(base)
205          assert self.source
206          result = []
207          for idx, submod in enumerate(base):
208              result.append(
209                  tx.output.register_attr_or_module(
210                      submod,
211                      self.module_key,
212                      idx,
213                      source=NNModuleSource(GetItemSource(self.source, idx)),
214                  )
215              )
216          return result

221          mod = tx.output.get_submodule(self.module_key)

227          )
228          return variables.ConstantVariable.create(result)

234      def convert_to_unspecialized(self, tx):
235          """Restart analysis treating this module as an UnspecializedNNModuleVariable"""
236          mod = tx.output.get_submodule(self.module_key)
237          GenerationTracker.tag(mod)
238  
239          # Mark the class dynamic unless its module initialization
240          if tx.f_code.co_name != "__init__":
241              GenerationTracker.mark_class_dynamic(type(mod))
242          raise UnspecializeRestartAnalysis

244      def has_key_in_generic_dict(self, tx: "InstructionTranslator", key):
245          base = tx.output.get_submodule(self.module_key)
246  
247          if object_has_getattribute(base):
248              unimplemented_v2(
249                  gb_type="Custom __getattribute__ in nn.Module dict key check",
250                  context=f"has_key_in_generic_dict {self} {key}",
251                  explanation="Dynamo does not support checking key existence "
252                  "on `nn.Module` instances that have a custom "
253                  "`__getattribute__` method defined.",
254                  hints=[
255                      "Avoid defining `__getattribute__` in your module.",
256                      *graph_break_hints.SUPPORTABLE,
257                  ],
258              )
259  
260          if tx.output.side_effects.has_pending_mutation_of_attr(self, key):
261              mutated_attr = tx.output.side_effects.load_attr(self, key, deleted_ok=True)
262              return not isinstance(mutated_attr, variables.DeletedVariable)
263  
264          base_dict = object.__getattribute__(base, "__dict__")
265          return key in base_dict

267      def _custom_getattr_fallback(self, base, tx, name, obj_source):
268          """Check for a __getattr__ and handle it specially if it is implemented"""
269          if object_has_getattribute(base):
270              unimplemented_v2(
271                  gb_type="Custom __getattribute__ in nn.Module attribute access",
272                  context=f"var_getattr {self} {name}",
273                  explanation="Dynamo does not support checking key existence "
274                  "on `nn.Module` instances that have a custom "
275                  "`__getattribute__` method defined.",
276                  hints=[
277                      "Avoid defining `__getattribute__` in your module.",
278                      *graph_break_hints.SUPPORTABLE,
279                  ],
280              )
281  
282          getattr_fn = get_custom_getattr(base, ignore_nn_module_getattr=True)
283          if getattr_fn is None:
284              return None
285  
286          if not isinstance(getattr_fn, types.FunctionType):
287              unimplemented_v2(
288                  gb_type="torch.nn.Module with a non-function custom __getattr__",
289                  context=f"var_getattr {self} {name}",
290                  explanation=(
291                      "Dynamo detected a nn.Module object with a custom "
292                      "`__getattr__` method, but this method is not a standard "
293                      "Python function (e.g., it might be implemented in C/C++). "
294                      "Dynamo cannot currently trace into such non-standard "
295                      "`__getattr__` methods."
296                  ),
297                  hints=[
298                      "Avoid using objects with non-standard __getattr__ methods "
299                      "within the compiled region. If possible, implement "
300                      "__getattr__ as a standard Python function.",
301                      *graph_break_hints.SUPPORTABLE,
302                  ],
303              )
304  
305          options = {"source": AttrSource(obj_source, "__getattr__")}
306          return variables.UserMethodVariable(getattr_fn, self, **options).call_function(
307              tx, [variables.ConstantVariable.create(name)], {}
308          )

310      def var_getattr(self, tx: "InstructionTranslator", name):
311          source = self.source and AttrSource(self.source, name)
312  
313          base = tx.output.get_submodule(self.module_key)
314          base_dict = object.__getattribute__(base, "__dict__")
315          object_member = True
316          all_class_attribute_names = set()
317          for x in inspect.getmro(base.__class__):
318              all_class_attribute_names.update(x.__dict__.keys())
319  
320          if not self.source:
321              unimplemented_v2(
322                  gb_type="getattr with no source",
323                  context=f"var_getattr {self} {name}",
324                  explanation="Dynamo does not know how to access an attribute "
325                  "on an `nn.Module` instance that lacks a source. This is "
326                  "usually an internal error in Dynamo.",
327                  hints=[*graph_break_hints.DYNAMO_BUG],
328              )
329  
330          if name == "__dict__":
331              return variables.GetAttrVariable(self, name, source=source)
332  
333          if name in base_dict:
334              subobj = base_dict[name]
335          elif (
336              "_modules" in base_dict
337              and name in base_dict["_modules"]
338              and name not in all_class_attribute_names
339          ):
340              subobj = base_dict["_modules"][name]
341          elif "_parameters" in base_dict and name in base_dict["_parameters"]:
342              subobj = base_dict["_parameters"][name]
343          elif "_buffers" in base_dict and name in base_dict["_buffers"]:
344              subobj = base_dict["_buffers"][name]
345          else:
346              try:
347                  subobj = inspect.getattr_static(base, name)
348                  object_member = False
349              except AttributeError:
350                  # see if we can fallback to __getattr__, which is not checked by getattr_static
351                  result = self._custom_getattr_fallback(
352                      base=base, tx=tx, name=name, obj_source=self.source
353                  )
354                  if result is not None:
355                      return result
356                  # if we can't find a __getattr__, we can't parse this, raise attribute error
357                  raise_observed_exception(
358                      AttributeError,
359                      tx,
360                  )
361  
362          if name == "forward":
363              guard_to_detect_forward_monkeypatching(self.source, base)
364  
365          if name == "__class__" and not object_member:
366              return variables.UserDefinedClassVariable(base.__class__, source=source)
367  
368          if object_member:
369              out = VariableTracker.build(tx, subobj, NNModuleSource(source))
370  
371              if isinstance(out, (NNModuleVariable, UnspecializedNNModuleVariable)):
372                  # nn_module_stack source is BC surface area. Ensure that
373                  # mod._modules["linear"] is reflected as mod.linear for
374                  # nn_module_stack.
375                  out.set_nn_module_stack_source(
376                      AttrSource(self.get_nn_module_stack_source(), name)
377                  )
378              return out
379  
380          else:
381              if istype(subobj, property):
382                  if self.source:
383                      # Read the class attribute to reach the property
384                      source = AttrSource(AttrSource(self.source, "__class__"), name)
385                      # Get the getter function
386                      source = AttrSource(source, "fget")
387                  return variables.UserFunctionVariable(
388                      subobj.fget,
389                      source=source,
390                  ).call_function(tx, [(self)], {})
391              elif istype(subobj, classmethod):
392                  return variables.UserMethodVariable(
393                      subobj.__func__,
394                      variables.UserDefinedObjectVariable(type(base)),
395                      source=source,
396                  )
397              elif istype(subobj, staticmethod):
398                  return variables.UserFunctionVariable(
399                      subobj.__get__(base), source=source
400                  )
401              elif istype(subobj, types.FunctionType):
402                  return variables.UserMethodVariable(subobj, self, source=source)
403              elif is_safe_constant(subobj) or istensor(subobj):
404                  # Support possibly common cases of class members
405                  return VariableTracker.build(tx, subobj, NNModuleSource(source))
406              else:
407                  unimplemented_v2(
408                      gb_type="Unsupported nn.Module attribute type",
409                      context=f"nn.Module subclass: {typestr(base)}, name: {name}, attribute type: {typestr(subobj)}",
410                      explanation=f"Dynamo does not support tracing nn.Module attributes of type `{typestr(subobj)}`",
411                      hints=[
412                          f"Refactor your code so that `{name}` (type `{typestr(subobj)}`) is not an attribute of `{typestr(base)}`",
413                          "Currently supported attribute types are methods, classmethods, staticmethods, "
414                          "properties, constants, and tensors.",
415                          *graph_break_hints.SUPPORTABLE,
416                      ],
417                  )
418  
419          return variables.GetAttrVariable(self, name, source=source)

421      def call_function(
422          self,
423          tx,
424          args: "list[VariableTracker]",
425          kwargs: "dict[str, VariableTracker]",
426      ) -> "VariableTracker":
427          mod = tx.output.get_submodule(self.module_key)
428  
429          with record_nn_module_stack(
430              self.module_key, self.get_nn_module_stack_source(), tx, mod
431          ):
432              is_lazy = is_lazy_module(mod)
433              if (
434                  isinstance(mod, torch.nn.Sequential)
435                  and mod.__class__.forward is torch.nn.Sequential.forward
436              ):
437                  if nnmodule_has_hooks(mod):
438                      # We do not want to unroll sequential if it has hooks, since evaporating it
439                      # will cause hooks to not fire!
440                      # This terminates and restart the tracing process
441                      self.convert_to_unspecialized(tx)
442  
443                  # Unroll sequential
444                  assert not is_lazy, (
445                      "Expected lazy sequential isn't a valid combination?"
446                  )
447                  assert not kwargs
448                  (arg,) = args
449                  # TODO: Use named_children when it supports remove_duplicate=False.
450                  for child_name, submod in mod._modules.items():
451                      tx.call_function(
452                          tx.output.register_attr_or_module(
453                              submod,
454                              self.module_key,
455                              child_name,
456                              source=NNModuleSource(AttrSource(self.source, child_name)),
457                          ),
458                          [arg],
459                          {},
460                      )
461                      arg = tx.pop()
462                  return arg
463  
464              if is_lazy:
465                  # The module type will change after it is called
466                  if mod.cls_to_become is not None:
467                      self.module_type = mod.cls_to_become
468  
469                  # The pre-hook runs to initialize the module shapes, then deletes itself.  After this,
470                  # the module is more or less not lazy and can be treated as a normal module regardless of
471                  # is_allowed or other variations.
472                  initialize_lazy_module(tx, mod, args, kwargs)
473  
474              # If we are tracing the higher order op, we want Dynamo to step
475              # inside the module call so that Dynamo can see the underlying
476              # parameters and buffers and raise them as inputs to the graph.
477              #
478              # NB: torch.nn.utils.parametrize changes the class type of a
479              # parametrized module such that its __module__ points to
480              # "torch.nn.utils.parametrize".
481              if (
482                  tx.output.is_root_tracer()
483                  and mod.__module__.startswith(("torch.nn.", "torch.ao."))
484                  and mod.__module__ != "torch.nn.utils.parametrize"
485              ):
486                  if nnmodule_has_hooks(
487                      mod, check_forward_hooks=True, check_backward_hooks=True
488                  ):
489                      # End of fn, this bubbles up and restarts tracing.
490                      self.convert_to_unspecialized(tx)
491  
492                  from .builder import wrap_fx_proxy
493  
494                  return wrap_fx_proxy(
495                      tx=tx,
496                      proxy=tx.output.create_proxy(
497                          "call_module",
498                          self.module_key,
499                          *proxy_args_kwargs(args, kwargs),
500                      ),
501                  )
502              else:
503                  assert self.source, (
504                      "Must provide a valid source in order to inline, "
505                      "since inlined function may have default args which must be guarded."
506                  )
507                  if isinstance(mod, torch.fx.GraphModule):
508                      # TODO: do we want to support __call__ for GM's?
509                      # If so at least some changes are needed, we don't allow inlining
510                      # the call_wrapped currently, and maybe other issues too
511                      fn = mod.forward
512                      fn_source = AttrSource(self.source, "forward")
513                  else:
514                      fn = mod._call_impl
515                      fn_source = AttrSource(self.source, "_call_impl")
516                  if istype(fn, types.MethodType):
517                      fn = fn.__func__
518                      fn_source = AttrSource(fn_source, "__func__")
519                      args = [self] + args
520                  else:
521                      assert istype(fn, types.FunctionType)
522                  return tx.inline_user_function_return(
523                      variables.UserFunctionVariable(fn, source=fn_source),
524                      args,
525                      kwargs,
526                  )

528      def call_method(
529          self,
530          tx,
531          name,
532          args: "list[VariableTracker]",
533          kwargs: "dict[str, VariableTracker]",
534          constant=False,
535      ) -> "VariableTracker":
536          from . import ConstantVariable, ListIteratorVariable, TupleVariable
537  
538          key = self.module_key
539          module = tx.output.get_submodule(key)
540  
541          def generic_call_method_helper(name):
542              # Helper function to put a `call_method` node in FX graph,
543              # with nn.Module as the first arg.
544              mod_proxy = tx.output.create_proxy(
545                  "get_attr",
546                  self.module_key,
547                  (),
548                  {},
549              )
550              set_example_value(mod_proxy.node, module)
551  
552              proxy_args, proxy_kwargs = proxy_args_kwargs(args, kwargs)
553  
554              from .builder import wrap_fx_proxy
555  
556              return wrap_fx_proxy(
557                  tx=tx,
558                  proxy=tx.output.create_proxy(
559                      "call_method",
560                      name,
561                      args=(mod_proxy, *proxy_args),
562                      kwargs=proxy_kwargs,
563                  ),
564              )
565  
566          if name in ["_call_impl", "_wrapped_call_impl"]:
567              # Example: `self.layer.__call__(x)`
568              # This is used for explicit calling `__call__` in a forward function.
569              # Dynamo inlines `__call__`, includes hooks.
570              return self.call_function(tx, args, kwargs)
571          elif name == "forward":
572              # Example: `self.layer.forward(x)`
573              # This is used for explicit calling `forward` in a forward function.
574              # Dynamo puts `call_method` node in FX, doesn't trigger hooks.
575              with record_nn_module_stack(
576                  self.module_key, self.get_nn_module_stack_source(), tx, module
577              ):
578                  return generic_call_method_helper(name)
579  
580          if name == "_check_input_dim" and trace_rules.is_torch_inline_allowed(
581              inspect.getfile(module.__class__._check_input_dim)
582          ):
583              return ConstantVariable.create(True)
584  
585          if name == "_get_item_by_idx":
586              assert args[1].is_python_constant()
587              assert isinstance(args[0], TupleVariable)
588              mod_var = args[0].items[args[1].value]
589              if isinstance(mod_var, UnspecializedNNModuleVariable):
590                  return mod_var
591              key = mod_var.module_key
592              submod = tx.output.get_submodule(key)
593              return tx.output.register_attr_or_module(
594                  submod,
595                  key,
596                  key,
597                  source=NNModuleSource(GetItemSource(self.source, key)),
598              )
599  
600          if constant:
601              fn = getattr(module, name)
602              name = f"{module.__class__.__name__}_{name}_result"
603              return invoke_and_store_as_constant(tx, fn, name, args, kwargs)
604  
605          def assert_all_args_kwargs_const():
606              if not all(
607                  x.is_python_constant() for x in itertools.chain(args, kwargs.values())
608              ):
609                  unimplemented_v2(
610                      gb_type="non-const argument in nn.Module method",
611                      context=f"call_method: {self} {name} {args} {kwargs}",
612                      explanation="Dynamo does not support calling "
613                      f"method `{name}` of ``nn.Module`` {module} with non-constant arguments.",
614                      hints=[],
615                  )
616  
617          def get_kwargs(*names):
618              assert_all_args_kwargs_const()
619              fn = getattr(module, name)
620              bound_args = inspect.signature(fn).bind(
621                  *([x.as_python_constant() for x in args]),
622                  **{k: v.as_python_constant() for k, v in kwargs.items()},
623              )
624              bound_args.apply_defaults()
625              bound_args = bound_args.arguments
626              return {k: bound_args[k] for k in names}
627  
628          def wrap_values(items):
629              result = []
630              for name, submod in items:
631                  result.append(
632                      tx.output.register_attr_or_module(
633                          submod,
634                          key,
635                          name,
636                          source=NNModuleSource(gen_source(self.source, name)),
637                      )
638                  )
639              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
640  
641          def named_embed(name, obj):
642              return TupleVariable(
643                  [
644                      ConstantVariable.create(name),
645                      tx.output.register_attr_or_module(
646                          obj,
647                          key,
648                          name,
649                          source=NNModuleSource(gen_source(self.source, name)),
650                      ),
651                  ]
652              )
653  
654          def gen_source(source, name):
655              name_split = name.split(".")
656              if name_split[0] == "":
657                  return source
658              while len(name_split) > 0:
659                  x = name_split.pop(0)
660                  source = AttrSource(source, x)
661              return source
662  
663          if name == "named_children":
664              tx.output.guard_on_key_order.add(AttrSource(self.source, "_modules"))
665              assert not (args or kwargs)
666              result = []
667              for name, submod in module.named_children():
668                  result.append(named_embed(name, submod))
669              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
670          elif name == "named_parameters":
671              tx.output.guard_on_key_order.add(AttrSource(self.source, "_parameters"))
672              result = []
673              for name, param in module.named_parameters(
674                  **get_kwargs("prefix", "recurse")
675              ):
676                  result.append(named_embed(name, param))
677              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
678          elif name == "named_buffers":
679              tx.output.guard_on_key_order.add(AttrSource(self.source, "_buffers"))
680              result = []
681              for name, buffer in module.named_buffers(
682                  **get_kwargs("prefix", "recurse", "remove_duplicate")
683              ):
684                  result.append(named_embed(name, buffer))
685              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
686          elif name == "named_modules":
687              tx.output.guard_on_key_order.add(AttrSource(self.source, "_modules"))
688              result = []
689              for name, submod in module.named_modules(
690                  **get_kwargs("memo", "prefix", "remove_duplicate")
691              ):
692                  result.append(named_embed(name, submod))
693              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
694          elif name == "children":
695              tx.output.guard_on_key_order.add(AttrSource(self.source, "_modules"))
696              assert not (args or kwargs)
697              return wrap_values(module.named_children())
698          elif name == "modules":
699              tx.output.guard_on_key_order.add(AttrSource(self.source, "_modules"))
700              return wrap_values(module.named_modules())
701          elif name == "parameters":
702              tx.output.guard_on_key_order.add(AttrSource(self.source, "_parameters"))
703              return wrap_values(module.named_parameters(**get_kwargs("recurse")))
704          elif name == "buffers":
705              tx.output.guard_on_key_order.add(AttrSource(self.source, "_buffers"))
706              return wrap_values(module.named_buffers(**get_kwargs("recurse")))
707          elif name == "keys":
708              assert not (args or kwargs)
709              result = []
710              for name in module.keys():
711                  result.append(ConstantVariable.create(name))
712              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
713          elif name == "values":
714              assert not (args or kwargs)
715              return wrap_values(module.items())
716          elif name == "items":
717              assert not (args or kwargs)
718              result = []
719              for name, submod in module.items():
720                  result.append(named_embed(name, submod))
721              return ListIteratorVariable(result, mutation_type=ValueMutationNew())
722          elif name == "__len__":
723              assert not (args or kwargs)
724              return ConstantVariable.create(len(module))
725          elif (
726              name == "__contains__"
727              and isinstance(module, (torch.nn.ModuleDict, torch.nn.ParameterDict))
728              and args
729              and args[0].is_python_constant()
730          ):
731              return ConstantVariable.create(
732                  args[0].as_python_constant() in module._modules
733              )
734          elif name == "__getitem__":
735              assert not kwargs and len(args) == 1
736              builtin_supported = (
737                  torch.nn.ModuleDict.__getitem__,
738                  torch.nn.ModuleList.__getitem__,
739                  torch.nn.ParameterDict.__getitem__,
740                  torch.nn.ParameterList.__getitem__,
741                  torch.nn.Sequential.__getitem__,
742              )
743  
744              if type(module).__getitem__ not in builtin_supported:
745                  assert isinstance(args[0], variables.ConstantVariable), typestr(args[0])
746                  key = args[0].as_python_constant()
747                  assert isinstance(key, (str, int))
748                  fn = getattr(module, name).__func__
749  
750                  assert isinstance(fn, types.FunctionType)
751  
752                  src = AttrSource(AttrSource(self.source, name), "__func__")
753                  return tx.inline_user_function_return(
754                      variables.UserFunctionVariable(fn, source=src),
755                      [self] + list(args),
756                      kwargs,
757                  )
758  
759              assert self.source
760  
761              if isinstance(args[0], SliceVariable):
762                  # TODO(anijain2305,export-team) - Remove this if condition when inlining of inbuilt nn modules is
763                  # enabled for export.
764                  if tx.output.export:
765                      # Build a TupleVariable of NNModules
766                      result = []
767  
768                      # Turn the slice into the list of integers
769                      keys = list(range(len(module)))[args[0].as_python_constant()]
770                      for idx, submod in enumerate(module[args[0].as_python_constant()]):
771                          key = keys[idx]
772                          src = NNModuleSource(GetItemSource(self.source, key))
773                          result.append(
774                              tx.output.register_attr_or_module(
775                                  submod,
776                                  key,
777                                  source=src,
778                              )
779                          )
780  
781                      new_module = module[args[0].as_python_constant()]
782                      new_module_variable = tx.output.register_attr_or_module(
783                          new_module,
784                          f"{self}.__getitem__(slice)",
785                          source=NNModuleSource(
786                              GetItemSource(self.source, args[0].as_python_constant())
787                          ),
788                      )
789                      return new_module_variable
790                  else:
791                      # slice on nn module results in a creation of new module instance, so we need to make it sourceless.
792                      # Convert to unspecialized so that UnspecializedNNModule variable can take care of it.
793                      self.convert_to_unspecialized(tx)
794  
795              from .tensor import SymNodeVariable
796  
797              if isinstance(args[0], SymNodeVariable):
798                  key = args[0].evaluate_expr(tx.output)
799              elif args[0].is_python_constant():
800                  key = args[0].as_python_constant()
801              else:
802                  unimplemented_v2(
803                      gb_type="Unsupported key type for nn.Module.__getitem__",
804                      context=f"call_method: {self} {name} {args} {kwargs}",
805                      explanation="Dynamo does not support getitem on "
806                      "`nn.Module` with non-constant key.",
807                      hints=[],
808                  )
809  
810              submod = module[key]
811              return tx.output.register_attr_or_module(
812                  submod,
813                  self.module_key,
814                  key,
815                  source=NNModuleSource(GetItemSource(self.source, key)),
816              )
817          elif (
818              name == "_get_abs_string_index"
819              or (
820                  isinstance(module, torch.nn.modules.conv._ConvNd)
821                  and name == "_conv_forward"
822              )
823              or (
824                  isinstance(module, torch.nn.modules.conv._ConvTransposeNd)
825                  and name == "_output_padding"
826              )
827          ):
828              # Inline the function
829              fn = getattr(module, name).__func__
830              fn_source = AttrSource(AttrSource(self.source, name), "__func__")
831              return tx.inline_user_function_return(
832                  variables.UserFunctionVariable(fn, source=fn_source),
833                  [self] + args,
834                  kwargs,
835              )
836          # A loose heuristic, but seems to be generally good before we drop into the
837          # manual handling of inputs
838          elif (
839              name in module.__class__.__dict__
840              and callable(module.__class__.__dict__[name])
841              and all(
842                  isinstance(x, variables.TensorVariable)
843                  for x in itertools.chain(args, kwargs.values())
844              )
845          ):
846              return generic_call_method_helper(name)
847          else:
848              return super().call_method(tx, name, args, kwargs)

851  class UnspecializedNNModuleVariable(UserDefinedObjectVariable):
852      _nonvar_fields = {
853          "value_type",
854          "is_state_mutated",
855          "nn_module_stack_source",
856          *UserDefinedObjectVariable._nonvar_fields,
857      }

859      """ 
860      The above class will specialize on the id() of a module and place 
861      parameters on the torch.fx.GraphModule.  Giving one graph per 
862      module instance.  This version treats nn.Modules() like other user 
863      defined objects and will pass parameters into the FX graph as inputs. 
864      Giving one graph per module class. 
865      """

867      def __init__(self, value, **kwargs) -> None:
868          if type(value) is torch.jit._script.RecursiveScriptModule:
869              raise Unsupported(
870                  "ScriptModules aren't supported in UnspecializedNNModuleVariable"
871                  " because their .forward function isn't a static member of their type"
872              )
873          if "value_type" in kwargs:
874              lazy_value_to_become = getattr(kwargs["value_type"], "cls_to_become", None)
875              if type(value) is lazy_value_to_become:
876                  # We may have cloned a variabletracker for a LazyModule earlier (e.g. tracking side-effects)
877                  # and then later we called and mutated the LazyModule into a MaterializedModule.
878                  # We do not do the mutation upon first seeing a LazyModule since we preserve eager semantics to only
879                  # mutate upon first call, but this requires we update multiple copies of the VariableTracker post-mutation.
880                  kwargs["value_type"] = type(value)
881  
882          super().__init__(value=value, **kwargs)
883          self.is_state_mutated = False
884          # nn_module_stack_source is used to ensure BC for nn_module_stack.
885          # Downstream users prefer mod.linear instead of mod._modules['linear']
886          # as the module stack. When Dynamo inlines the __getattr__ method, we
887          # cannot use self.source for nn_module_stack because it will be similar
888          # to mod._modules['linear']. In these cases, we set the
889          # nn_module_stack_source appropriately to resemble mod.linear.
890          self.nn_module_stack_source = self.source

892      def _wrap_source(self, attr_source):
893          # the vt is already wrapped with UnspecializedNNModuleSource
894          return attr_source

896      def get_nn_module_stack_source(self):
897          return self.nn_module_stack_source or self.source

899      def set_nn_module_stack_source(self, source):
900          self.nn_module_stack_source = source

906          supported = {torch.nn.Module.__setattr__, torch.nn.Module.__init__}
907          return {

914          try:
915              fn = inspect.getattr_static(self.value_type, "__iter__")
916          except AttributeError as e:
917              raise NotImplementedError from e
918  
919          if fn in (
920              torch.nn.ModuleList.__iter__,
921              torch.nn.ParameterList.__iter__,
922              torch.nn.Sequential.__iter__,
923          ):

925              # will not reflect the mutations. So, trace through the `__iter__`
926              # function to reflect any tracked mutations.
927              return tx.inline_user_function_return(
928                  variables.UserFunctionVariable(fn),

930                      self,
931                  ],
932                  {},

937      def call_function(
938          self,
939          tx: "InstructionTranslator",
940          args: "list[VariableTracker]",
941          kwargs: "dict[str, VariableTracker]",
942      ) -> "VariableTracker":
943          mod = self.value
944          # see comment on lazy module handling in NNModuleVariable.call_function for context
945          if is_lazy_module(mod):
946              if mod.cls_to_become is not None:
947                  self.value_type = mod.cls_to_become
948              initialize_lazy_module(tx, mod, args, kwargs)
949  
950          if (
951              not isinstance(mod, torch.fx.GraphModule)
952              and mod.__call__.__func__ is not unpatched_nn_module_call
953          ):
954              name = "__call__"
955              fn = getattr(self.value_type, name)
956          else:
957              name = "_call_impl"
958              fn = getattr(self.value_type, name)
959  
960          # Check if we can short circuit nn.Module._call_impl to the forward
961          # method.  NB - This is done to reduce the compile time of Dynamo.
962          if (
963              istype(mod.__call__, types.MethodType)
964              and istype(mod._call_impl, types.MethodType)
965              and mod.__call__.__func__ is unpatched_nn_module_call
966              and mod._call_impl.__func__ is unpatched_nn_module_call_impl
967              and "forward" not in mod.__dict__
968          ):
969              forward_method = inspect.getattr_static(mod, "forward")
970              if isinstance(forward_method, types.FunctionType):
971                  globals_vt = tx.nn_modules_globals_vt
972                  if not (
973                      self.var_getattr(tx, "_backward_hooks").realize().len()
974                      or self.var_getattr(tx, "_backward_pre_hooks").realize().len()
975                      or self.var_getattr(tx, "_forward_hooks").realize().len()
976                      or self.var_getattr(tx, "_forward_pre_hooks").realize().len()
977                      or globals_vt.var_getattr(tx, "_global_backward_pre_hooks").len()
978                      or globals_vt.var_getattr(tx, "_global_backward_hooks").len()
979                      or globals_vt.var_getattr(tx, "_global_forward_hooks").len()
980                      or globals_vt.var_getattr(tx, "_global_forward_pre_hooks").len()
981                  ):
982                      name = "forward"
983                      fn = self.value_type.forward
984  
985          if self.source:
986              source = AttrSource(AttrSource(self.source, "__class__"), name)
987          else:
988              source = None
989  
990          guard_to_detect_forward_monkeypatching(self.source, mod)
991  
992          ctx = (
993              record_nn_module_stack(
994                  str(id(mod)), self.get_nn_module_stack_source(), tx, mod
995              )
996              if self.source
997              else nullcontext()
998          )
999          with ctx:
1000             return variables.UserFunctionVariable(fn, source=source).call_function(
1001                 tx, [self] + list(args), kwargs
1002             )

1004     def call_method(
1005         self,
1006         tx,
1007         name,
1008         args: "list[VariableTracker]",
1009         kwargs: "dict[str, VariableTracker]",
1010     ) -> "VariableTracker":
1011         if name in ["_call_impl", "_wrapped_call_impl"]:
1012             fn = getattr(self.value_type, name)
1013             if self.source:
1014                 source = AttrSource(AttrSource(self.source, "__class__"), name)
1015             else:
1016                 source = None
1017 
1018             return variables.UserFunctionVariable(fn, source=source).call_function(
1019                 tx, [self] + list(args), kwargs
1020             )
1021 
1022         if name not in getattr(self.value, "__dict__", {}):
1023             try:
1024                 method = inspect.getattr_static(type(self.value), name)
1025             except AttributeError:
1026                 method = None
1027 
1028             if isinstance(method, staticmethod):
1029                 source = AttrSource(
1030                     AttrSource(AttrSource(self.source, "__class__"), name), "__func__"
1031                 )
1032                 return tx.inline_user_function_return(
1033                     variables.UserFunctionVariable(method.__func__, source=source),
1034                     args,
1035                     kwargs,
1036                 )
1037 
1038             if (
1039                 hasattr(method, "__code__")
1040                 and id(method.__code__) in self._nn_module_method_ids()
1041             ):
1042                 unimplemented_v2(
1043                     gb_type="UnspecializedNNModuleVariable missing method",
1044                     context=f"call_method: {self} {name} {args} {kwargs}",
1045                     explanation=f"Dynamo does not support tracing method {name} of nn.Module {self.value}",
1046                     hints=[
1047                         "Dynamo does not really define unspecialized nn.Module very well.",
1048                         *graph_break_hints.DIFFICULT,
1049                     ],
1050                 )
1051 
1052             # "_parameters" in self.value.__dict__ checks that module is initialized
1053             if name == "__setattr__" and "_parameters" in self.value.__dict__:
1054                 # Record if mutations happens on parameters/buffers/modules. The
1055                 # mutations on these are not tracked by base class
1056                 # UserDefinedObject vt. This will be used later to graph break
1057                 # on seeing a parameters() and family calls.
1058                 # TODO(anijain2305) - This might not be needed if we let Dynamo
1059                 # inline both getattr and setattr. In that case, it should see
1060                 # the lowest level dicts - _parameters and family and
1061                 # automatically track mutations on those. Investigate if that
1062                 # can be done.
1063                 attr_name = args[0].as_python_constant()
1064                 value = args[1]
1065 
1066                 # This is reverse engineered by looking at nn module __setattr__
1067                 # logic.
1068                 if (
1069                     isinstance(value, variables.TensorVariable)
1070                     and value.python_type() is torch.nn.Parameter
1071                 ) or attr_name in self.value.__dict__["_parameters"]:
1072                     # Handle parameters
1073                     self.is_state_mutated = True
1074                 elif attr_name in self.value.__dict__["_buffers"]:
1075                     # Handle buffers
1076                     self.is_state_mutated = True
1077                 elif (
1078                     isinstance(
1079                         value,
1080                         (
1081                             variables.NNModuleVariable,
1082                             variables.UnspecializedNNModuleVariable,
1083                         ),
1084                     )
1085                     or attr_name in self.value.__dict__["_modules"]
1086                 ):
1087                     # Handle submodules
1088                     self.is_state_mutated = True
1089 
1090             if method is torch.nn.Module.__setattr__ and isinstance(
1091                 args[1], variables.DeletedVariable
1092             ):
1093                 # Trace through __delattr__ to track mutations on the module
1094                 # members like `_modules``.
1095                 return tx.inline_user_function_return(
1096                     variables.UserFunctionVariable(torch.nn.Module.__delattr__),
1097                     [self, args[0]],
1098                     kwargs,
1099                 )
1100 
1101         return super().call_method(tx, name, args, kwargs)

1103     def getattr_helper(self, tx: "InstructionTranslator", field, name_vt):

1109     def var_getattr(self, tx: "InstructionTranslator", name):
1110         # Allow skipping of empty hook dict guards on inbuilt nn modules
1111         if name in (
1112             "_backward_hooks",
1113             "_backward_pre_hooks",
1114             "_forward_hooks",
1115             "_forward_pre_hooks",
1116         ):
1117             # For empty hooks, make an EMPTY_NN_MODULE_HOOKS_DICT. This allows us to control the installation of empty
1118             # hooks guard via skip_nnmodule_hook_guards
1119             if not tx.output.side_effects.has_pending_mutation_of_attr(self, name):
1120                 hooks_dict = getattr(self.value, name)
1121                 if isinstance(hooks_dict, dict) and len(hooks_dict) == 0:
1122                     if self.source:
1123                         hooks_source = AttrSource(self.source, name)
1124                         install_guard(
1125                             hooks_source.make_guard(
1126                                 GuardBuilder.EMPTY_NN_MODULE_HOOKS_DICT
1127                             )
1128                         )
1129                     return variables.ConstDictVariable({})
1130 
1131         # For non-empty hook dicts, one way is to just fallback to VariableTracker.build() and create a ConstDictVariable.
1132         # However, ConstDictVariable guards on keys. This can cause recompiles when the same hook is installed for
1133         # different nn module instances, because the key keeps changing (look more into RemovableHandle to understand why
1134         # key changes - also related https://github.com/pytorch/pytorch/issues/125836). Here, we carefully craft a
1135         # NNModuleHooksDictVariable (a subclass of ConstDictVariable) to avoid any guard on the keys.
1136         if (
1137             self.source
1138             and name
1139             in (
1140                 "_forward_pre_hooks",
1141                 "_forward_hooks",
1142             )
1143             and not tx.output.side_effects.has_pending_mutation_of_attr(self, name)
1144         ):
1145             hooks_dict = getattr(self.value, name)
1146             hooks_dict_source = AttrSource(self.source, name)
1147             install_guard(hooks_dict_source.make_guard(GuardBuilder.SEQUENCE_LENGTH))
1148             tx.output.guard_on_key_order.add(hooks_dict_source)
1149 
1150             def build_key_value(i, k, v):
1151                 # Make key sourceless to avoid any guard on it
1152                 key = variables.ConstantVariable.create(k)
1153 
1154                 # Instead of using dict[key] to access the value, use a dict[dict.keys()[index]] to access the
1155                 # value. This removes the reliance on the actual key value.
1156                 source_key = ConstDictKeySource(hooks_dict_source, i)
1157                 source_value = DictGetItemSource(hooks_dict_source, source_key)
1158                 value = LazyVariableTracker.create(v, source_value)
1159                 return key, value
1160 
1161             result = dict(
1162                 build_key_value(i, k, v) for i, (k, v) in enumerate(hooks_dict.items())
1163             )
1164 
1165             return variables.NNModuleHooksDictVariable(
1166                 result, type(hooks_dict), source=hooks_dict_source
1167             )
1168         return super().var_getattr(tx, name)
1169 
1170     def manually_trace_nn_module_getattr(self, tx: "InstructionTranslator", name):
1171         """ 
1172         Dynamo tracing of nn.Module __getattr__ can be expensive if the model 
1173         has deep submodule hierarchy. Since the __getattr__ is stable, we can 
1174         directly look into the underlying datastructures. This saves a lot of 
1175         compilation time. 
1176         """
1177         name_vt = variables.ConstantVariable(name)
1178         out = self.getattr_helper(tx, "_parameters", name_vt)
1179         if out is None:
1180             out = self.getattr_helper(tx, "_modules", name_vt)
1181         if out is None:
1182             out = self.getattr_helper(tx, "_buffers", name_vt)
1183         if out is None:
1184             raise_observed_exception(AttributeError, tx)
1185         return out
