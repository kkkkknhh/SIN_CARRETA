10   /** 
11    * This is a simple bitset class with sizeof(long long int) bits. 
12    * You can set bits, unset bits, query bits by index, 
13    * and query for the first set bit. 
14    * Before using this class, please also take a look at std::bitset, 
15    * which has more functionality and is more generic. It is probably 
16    * a better fit for your use case. The sole reason for c10::utils::bitset 
17    * to exist is that std::bitset misses a find_first_set() method. 
18    */
19   struct bitset final {
20    private:
21   #if defined(_MSC_VER)
22     // MSVCs _BitScanForward64 expects int64_t
23     using bitset_type = int64_t;
24   #else
25     // POSIX ffsll expects long long int
26     using bitset_type = long long int;
27   #endif
28    public:
29     static constexpr size_t NUM_BITS() {
30       return 8 * sizeof(bitset_type);
31     }
32   
33     constexpr bitset() noexcept = default;
34     constexpr bitset(const bitset&) noexcept = default;
35     constexpr bitset(bitset&&) noexcept = default;
36     // there is an issure for gcc 5.3.0 when define default function as constexpr
37     // see https://gcc.gnu.org/bugzilla/show_bug.cgi?id=68754.
38     bitset& operator=(const bitset&) noexcept = default;
39     bitset& operator=(bitset&&) noexcept = default;
40     ~bitset() = default;
41   
42     constexpr void set(size_t index) noexcept {
43       bitset_ |= (static_cast<long long int>(1) << index);
44     }
45   
46     constexpr void unset(size_t index) noexcept {
47       bitset_ &= ~(static_cast<long long int>(1) << index);
48     }
49   
50     constexpr bool get(size_t index) const noexcept {
51       return bitset_ & (static_cast<long long int>(1) << index);
52     }
53   
54     constexpr bool is_entirely_unset() const noexcept {
55       return 0 == bitset_;
56     }
57   
58     // Call the given functor with the index of each bit that is set
59     template <class Func>
60     // NOLINTNEXTLINE(cppcoreguidelines-missing-std-forward)
61     void for_each_set_bit(Func&& func) const {
62       bitset cur = *this;
63       size_t index = cur.find_first_set();
64       while (0 != index) {
65         // -1 because find_first_set() is not one-indexed.
66         index -= 1;
67         func(index);
68         cur.unset(index);
69         index = cur.find_first_set();
70       }
71     }
72   
73    private:
74     // Return the index of the first set bit. The returned index is one-indexed
75     // (i.e. if the very first bit is set, this function returns '1'), and a
76     // return of '0' means that there was no bit set.
77     size_t find_first_set() const {
78   #if defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
79       unsigned long result;
80       bool has_bits_set = (0 != _BitScanForward64(&result, bitset_));
81       if (!has_bits_set) {
82         return 0;
83       }
84       return result + 1;
85   #elif defined(_MSC_VER) && defined(_M_IX86)
86       unsigned long result;
87       if (static_cast<uint32_t>(bitset_) != 0) {
88         bool has_bits_set =
89             (0 != _BitScanForward(&result, static_cast<uint32_t>(bitset_)));
90         if (!has_bits_set) {
91           return 0;
92         }
93         return result + 1;
94       } else {
95         bool has_bits_set =
96             (0 != _BitScanForward(&result, static_cast<uint32_t>(bitset_ >> 32)));
97         if (!has_bits_set) {
98           return 32;
99         }
100        return result + 33;
101      }
102  #else
103      return __builtin_ffsll(bitset_);
104  #endif
105    }
106  
107    friend bool operator==(bitset lhs, bitset rhs) noexcept {
108      return lhs.bitset_ == rhs.bitset_;
109    }
110  
111    bitset_type bitset_{0};
112  };
113  
114  inline bool operator!=(bitset lhs, bitset rhs) noexcept {
115    return !(lhs == rhs);
116  }
117  
118  } // namespace c10::utils
