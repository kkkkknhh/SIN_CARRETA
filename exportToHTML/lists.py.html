3    """ 
4    Variable tracking implementations for list-like data structures in Dynamo. 
5     
6    This module provides specialized variable tracking for various collection types: 
7    - Lists and list subclasses (including torch.nn.ModuleList, ParameterList) 
8    - Tuples and named tuples 
9    - Ranges and slices 
10   - Collections.deque 
11   - torch.Size with special proxy handling 
12    
13   The implementations support both mutable and immutable collections, iteration, 
14   and common sequence operations. Each collection type has a dedicated Variable 
15   class that handles its unique behaviors while integrating with Dynamo's 
16   variable tracking system. 
17   """

53   class BaseListVariable(VariableTracker):
54       @staticmethod
55       def cls_for_instance(obj):
56           return BaseListVariable.cls_for(type(obj))
57   
58       @staticmethod
59       def cls_for(obj):
60           return {
61               iter: ListIteratorVariable,
62               list: ListVariable,
63               slice: SliceVariable,
64               torch.Size: SizeVariable,
65               tuple: TupleVariable,
66               odict_values: ListVariable,
67               torch.nn.ParameterList: ListVariable,
68               torch.nn.ModuleList: ListVariable,
69               collections.deque: DequeVariable,
70           }[obj]
71   
72       def __init__(
73           self,
74           items: list[VariableTracker],
75           **kwargs,
76       ) -> None:
77           super().__init__(**kwargs)
78           assert isinstance(items, list)
79           assert all(isinstance(x, VariableTracker) for x in items)
80           self.items: list[VariableTracker] = items
81   
82       def _as_proxy(self):
83           return [x.as_proxy() for x in self.items]
84   
85       def modified(self, items, **kwargs):
86           return type(self)(items, **kwargs)
87   
88       @property
89       def value(self):
90           return self.as_python_constant()
91   
92       def debug_repr_helper(self, prefix, suffix):
93           return prefix + ", ".join(i.debug_repr() for i in self.items) + suffix
94   
95       def as_python_constant(self):
96           return self.python_type()([x.as_python_constant() for x in self.items])
97   
98       def as_proxy(self):
99           assert self.python_type() is not SizeVariable
100          return self.python_type()(self._as_proxy())
101  
102      def getitem_const(self, tx: "InstructionTranslator", arg: VariableTracker):
103          from .tensor import SymNodeVariable
104  
105          if isinstance(arg, SymNodeVariable):
106              index = arg.sym_num
107          else:
108              index = arg.as_python_constant()
109  
110          if isinstance(index, slice):
111              # Set source to None because slicing a list gives a new local
112              return self.clone(
113                  items=self.items[index],
114                  source=None,
115                  mutation_type=ValueMutationNew() if self.mutation_type else None,
116              )
117          else:
118              assert isinstance(index, (int, torch.SymInt))
119              try:
120                  return self.items[index]
121              except IndexError:
122                  raise_observed_exception(
123                      IndexError, tx, args=["list index out of range"]
124                  )
125  
126      def unpack_var_sequence(self, tx):
127          return list(self.items)
128  
129      def call_method(
130          self,
131          tx,
132          name,
133          args: list["VariableTracker"],
134          kwargs: dict[str, "VariableTracker"],
135      ) -> "VariableTracker":
136          if name == "__getitem__":
137              from .tensor import TensorVariable
138  
139              assert not kwargs and len(args) == 1
140              if isinstance(args[0], TensorVariable):
141                  value = get_fake_value(args[0].as_proxy().node, tx)
142                  if value.constant is not None and value.constant.numel() == 1:
143                      value = variables.ConstantVariable.create(value.constant.item())
144                  else:
145                      unimplemented_v2(
146                          gb_type="Indexing list with non-scalar tensor",
147                          context=f"call_method {self} {name} {args} {kwargs}",
148                          explanation=(
149                              "Attempted to index list-like object with tensor with > 1 element."
150                          ),
151                          hints=[*graph_break_hints.USER_ERROR],
152                      )
153              else:
154                  value = args[0]
155              return self.getitem_const(tx, value)
156          elif name == "__contains__":
157              assert len(args) == 1
158              assert not kwargs
159              return iter_contains(self.unpack_var_sequence(tx), args[0], tx)
160          elif name == "index":
161              return tx.inline_user_function_return(
162                  VariableTracker.build(tx, polyfills.index),
163                  [self] + list(args),
164                  kwargs,
165              )
166          elif name in cmp_name_to_op_mapping:
167              left = self
168              right = args[0]
169              # TODO this type check logic mirrors the following
170              # https://github.com/python/cpython/blob/a1c52d1265c65bcf0d9edf87e143843ad54f9b8f/Objects/object.c#L991-L1007
171              # But we should probably move it up the stack to so that we don't
172              # need to duplicate it for different VTs.
173              if not isinstance(left, BaseListVariable) or not isinstance(
174                  right, BaseListVariable
175              ):
176                  if name == "__eq__":
177                      return variables.BuiltinVariable(operator.is_).call_function(
178                          tx, (left, right), {}
179                      )
180                  elif name == "__ne__":
181                      return variables.BuiltinVariable(operator.is_not).call_function(
182                          tx, (left, right), {}
183                      )
184                  else:
185                      op_str = cmp_name_to_op_str_mapping[name]
186                      left_ty = left.python_type_name()
187                      right_ty = right.python_type_name()
188                      msg = f"{op_str} not supported between instances of '{left_ty}' and '{right_ty}'"
189                      raise_observed_exception(TypeError, tx, args=[msg])
190  
191              return variables.UserFunctionVariable(polyfills.list_cmp).call_function(
192                  tx,
193                  [variables.BuiltinVariable(cmp_name_to_op_mapping[name]), left, right],
194                  {},
195              )
196  
197          return super().call_method(tx, name, args, kwargs)

200  class RangeVariable(BaseListVariable):
201      def __init__(self, items, **kwargs) -> None:
202          items_to_map = items
203          start = variables.ConstantVariable.create(0)
204          stop = None
205          step = variables.ConstantVariable.create(1)
206  
207          if len(items_to_map) == 1:
208              (stop,) = items_to_map
209          elif len(items_to_map) == 2:
210              start, stop = items_to_map
211          elif len(items_to_map) == 3:
212              start, stop, step = items_to_map
213          else:
214              raise AssertionError
215  
216          assert stop is not None
217          super().__init__([start, stop, step], **kwargs)
218  
219      def debug_repr(self):
220          return self.debug_repr_helper("range(", ")")
221  
222      def python_type(self):
223          return range
224  
225      def start(self):
226          return self.items[0].as_python_constant()
227  
228      def stop(self):
229          return self.items[1].as_python_constant()
230  
231      def step(self):
232          return self.items[2].as_python_constant()
233  
234      def range_length(self):
235          lo = self.start()
236          hi = self.stop()
237          step = self.step()
238  
239          assert step != 0
240          if step > 0 and lo < hi:
241              return 1 + (hi - 1 - lo) // step
242          elif step < 0 and lo > hi:
243              return 1 + (lo - 1 - hi) // (0 - step)
244          else:
245              return 0
246  
247      def _get_slice_indices(self, length, slice):
248          step_is_negative = 0
249  
250          if slice.step is None:
251              step = 1
252              step_is_negative = False
253          else:
254              step = slice.step
255              step_is_negative = slice.step < 0
256  
257          # Find lower and upper bounds for start and stop.
258          if step_is_negative:
259              lower = -1
260              upper = length + lower
261          else:
262              lower = 0
263              upper = length
264  
265          # Compute start
266          if slice.start is None:
267              start = upper if step_is_negative else lower
268          else:
269              start = slice.start
270  
271          if start < 0:
272              start += length
273              if start < lower:
274                  start = lower
275          else:
276              if start > upper:
277                  start = upper
278  
279          # Compute stop.
280          if slice.stop is None:
281              stop = lower if step_is_negative else upper
282  
283          else:
284              stop = slice.stop
285  
286              if stop < 0:
287                  stop += length
288                  if stop < lower:
289                      stop = lower
290              else:
291                  if stop > upper:
292                      stop = upper
293  
294          return [start, stop, step]
295  
296      def apply_index(self, index):
297          length = self.range_length()
298          if index < 0:
299              index = length + index
300  
301          if index < 0 or index >= length:
302              raise IndexError(f"index {index} is out of range")
303  
304          return variables.ConstantVariable.create(self.start() + (index * self.step()))
305  
306      def apply_slice(self, slice):
307          (slice_start, slice_stop, slice_step) = self._get_slice_indices(
308              self.range_length(), slice
309          )
310  
311          def compute_item(index):
312              return self.start() + (index * self.step())
313  
314          sub_step = self.step() * slice_step
315          sub_start = compute_item(slice_start)
316          sub_stop = compute_item(slice_stop)
317  
318          result = RangeVariable(
319              [
320                  variables.ConstantVariable.create(x)
321                  for x in [sub_start, sub_stop, sub_step]
322              ],
323              mutation_type=ValueMutationNew() if self.mutation_type else None,
324          )
325          return result
326  
327      def as_python_constant(self):
328          return range(*[x.as_python_constant() for x in self.items])
329  
330      def getitem_const(self, tx: "InstructionTranslator", arg: VariableTracker):
331          # implementations mimics https://github.com/python/cpython/blob/main/Objects/rangeobject.c
332          index = arg.as_python_constant()
333  
334          if isinstance(index, slice):
335              return self.apply_slice(index)
336          else:
337              return self.apply_index(index)
338  
339      def as_proxy(self):
340          return self.python_type()(*self._as_proxy())
341  
342      def unpack_var_sequence(self, tx=None):
343          return [variables.ConstantVariable.create(x) for x in self.as_python_constant()]
344  
345      def reconstruct(self, codegen: "PyCodegen") -> None:
346          assert "range" not in codegen.tx.f_globals
347          codegen.add_push_null(
348              lambda: codegen.append_output(codegen.create_load_python_module(range))
349          )
350          codegen.foreach(self.items)
351          codegen.extend_output(create_call_function(3, False))
352  
353      def var_getattr(self, tx: "InstructionTranslator", name):
354          fields = ["start", "stop", "step"]
355          if name not in fields:
356              unimplemented_v2(
357                  gb_type="Unsupported attribute for range() object",
358                  context=f"var_getattr {self} {name}",
359                  explanation=f"Expected attribute to be one of {','.join(fields)} "
360                  f"but got {name}",
361                  hints=[*graph_break_hints.USER_ERROR],
362              )
363          return self.items[fields.index(name)]

366  class CommonListMethodsVariable(BaseListVariable):
367      """ 
368      Implement methods common to List and other List-like things 
369      """
370  
371      def call_method(
372          self,
373          tx,
374          name,
375          args: list["VariableTracker"],
376          kwargs: dict[str, "VariableTracker"],
377      ) -> "VariableTracker":
378          from .tensor import SymNodeVariable
379  
380          if name == "append" and self.is_mutable():
381              assert not kwargs
382              (arg,) = args
383              tx.output.side_effects.mutation(self)
384              self.items.append(arg)
385              return ConstantVariable.create(None)
386          elif (
387              name == "extend"
388              and self.is_mutable()
389              and args
390              and args[0].has_force_unpack_var_sequence(tx)
391          ):
392              assert not kwargs
393              (arg,) = args
394              arg.force_apply_to_var_sequence(
395                  tx, lambda item: self.call_method(tx, "append", [item], {})
396              )
397              return ConstantVariable.create(None)
398          elif name == "insert" and self.is_mutable():
399              assert not kwargs
400              idx, value = args
401              if isinstance(idx, SymNodeVariable):
402                  const_idx = idx.evaluate_expr()
403              else:
404                  const_idx = idx.as_python_constant()
405              tx.output.side_effects.mutation(self)
406              self.items.insert(const_idx, value)
407              return ConstantVariable.create(None)
408          elif name == "pop" and self.is_mutable():
409              assert not kwargs
410              tx.output.side_effects.mutation(self)
411              return self.items.pop(*[a.as_python_constant() for a in args])
412          elif name == "clear" and self.is_mutable():
413              assert not kwargs and not args
414              tx.output.side_effects.mutation(self)
415              self.items.clear()
416              return ConstantVariable.create(None)
417          elif (
418              name == "__setitem__"
419              and self.is_mutable()
420              and args
421              and args[0].is_python_constant()
422          ):
423              assert not kwargs
424              key, value = args
425              tx.output.side_effects.mutation(self)
426              if isinstance(key, SliceVariable):
427                  self.items[key.as_python_constant()] = list(value.items)
428              else:
429                  self.items[key.as_python_constant()] = value
430              return ConstantVariable.create(None)
431          elif name == "copy":
432              # List copy() doesn't have args and kwargs
433              assert not kwargs
434              assert not args
435              items = list(self.items)
436              return self.modified(items, mutation_type=ValueMutationNew())
437          elif name == "reverse" and self.is_mutable():
438              assert not kwargs
439              assert not args
440              self.items.reverse()
441              tx.output.side_effects.mutation(self)
442              return ConstantVariable.create(None)
443          else:
444              return super().call_method(tx, name, args, kwargs)

447  class ListVariable(CommonListMethodsVariable):
448      def python_type(self):
449          return list
450  
451      def __repr__(self) -> str:
452          return f"{self.__class__.__name__}(length={len(self.items)})"
453  
454      def debug_repr(self):
455          return self.debug_repr_helper("[", "]")
456  
457      def reconstruct(self, codegen: "PyCodegen") -> None:
458          codegen.foreach(self.items)
459          codegen.append_output(create_instruction("BUILD_LIST", arg=len(self.items)))
460  
461      def call_method(
462          self,
463          tx,
464          name,
465          args: list["VariableTracker"],
466          kwargs: dict[str, "VariableTracker"],
467      ) -> "VariableTracker":
468          if (
469              name == "__setitem__"
470              and self.is_mutable()
471              and args
472              and args[0].is_python_constant()
473          ):
474              assert not kwargs
475              key, value = args
476              tx.output.side_effects.mutation(self)
477              if isinstance(key, SliceVariable):
478                  if not value.has_force_unpack_var_sequence(tx):
479                      unimplemented_v2(
480                          gb_type="Unsupported conversion for slice assignment",
481                          context=f"call_method {self} {name} {args}",
482                          explanation=f"Missing dynamo support for converting {value} into a list for slice assignment.",
483                          hints=[*graph_break_hints.SUPPORTABLE],
484                      )
485                  self.items[key.as_python_constant()] = value.force_unpack_var_sequence(
486                      tx
487                  )
488              else:
489                  self.items[key.as_python_constant()] = value
490              return ConstantVariable.create(None)
491  
492          if name == "sort" and self.is_mutable():
493              assert len(args) == 0
494              key_fn_var = kwargs.pop("key", ConstantVariable.create(None))
495              reverse = kwargs.pop(
496                  "reverse", ConstantVariable.create(False)
497              ).as_python_constant()
498              assert len(kwargs) == 0
499  
500              if (
501                  key_fn_var.is_python_constant()
502                  and key_fn_var.as_python_constant() is None
503              ):
504                  keys = self.items.copy()
505              else:
506                  keys = [key_fn_var.call_function(tx, [x], {}) for x in self.items]
507  
508              if not all(k.is_python_constant() for k in keys):
509                  first_non_constant_key = None
510                  for k in keys:
511                      if not k.is_python_constant():
512                          first_non_constant_key = k
513                  assert first_non_constant_key is not None
514  
515                  try:
516                      python_type = first_non_constant_key.python_type()
517                  except NotImplementedError:
518                      python_type = "unknown"
519  
520                  unimplemented_v2(
521                      gb_type="sort with non-constant keys",
522                      context=str(first_non_constant_key),
523                      explanation=(
524                          f"Cannot perform sort with non-constant key. "
525                          f"First non-constant key type: {python_type}. "
526                          f"Most notably, we cannot sort with Tensor or SymInt keys, but we can "
527                          f"sort ints."
528                      ),
529                      hints=["Use something else as the key."],
530                  )
531  
532              tx.output.side_effects.mutation(self)
533              sorted_items_with_keys = sorted(
534                  (
535                      (
536                          x,
537                          k.as_python_constant(),
538                          -i if reverse else i,  # extra key to ensure stable sort
539                      )
540                      for i, (k, x) in enumerate(zip(keys, self.items))
541                  ),
542                  key=operator.itemgetter(1, 2),
543                  reverse=reverse,
544              )
545              self.items[:] = [x for x, *_ in sorted_items_with_keys]
546              return ConstantVariable.create(None)
547  
548          if name == "__init__" and self.is_mutable():
549              assert not kwargs
550              if len(args) == 0:
551                  return ConstantVariable.create(None)
552              elif len(args) == 1 and args[0].has_force_unpack_var_sequence(tx):
553                  (arg,) = args
554                  tx.output.side_effects.mutation(self)
555                  self.items[:] = arg.force_unpack_var_sequence(tx)
556                  return ConstantVariable.create(None)
557  
558          return super().call_method(tx, name, args, kwargs)
559  
560      def var_getattr(self, tx, name):
561          if name == "__class__":
562              source = AttrSource(self.source, name) if self.source else None
563              class_type = self.python_type()
564              if class_type is list:
565                  return variables.BuiltinVariable(class_type, source=source)
566              else:
567                  return variables.UserDefinedClassVariable(class_type, source=source)
568          return super().var_getattr(tx, name)
569  
570      def call_obj_hasattr(
571          self, tx: "InstructionTranslator", name: str
572      ) -> "VariableTracker":
573          if self.python_type() is not list:
574              return super().call_obj_hasattr(tx, name)
575          return variables.ConstantVariable.create(hasattr([], name))

578  class DequeVariable(CommonListMethodsVariable):
579      def __init__(self, items, maxlen=None, **kwargs) -> None:
580          if maxlen is None:
581              maxlen = ConstantVariable.create(None)
582          assert maxlen.is_python_constant(), (
583              f"maxlen must be a constant, got: {maxlen.debug_repr()}"
584          )
585          self.maxlen = maxlen
586          items = list(items)
587          if self.maxlen.as_python_constant() is not None:
588              items = items[-maxlen.as_python_constant() :]
589          super().__init__(items, **kwargs)
590  
591      def python_type(self):
592          return collections.deque
593  
594      def debug_repr(self):
595          if self.maxlen.as_python_constant() is None:
596              return self.debug_repr_helper(
597                  "deque([", "], maxlen=" + self.maxlen.debug_repr() + ")"
598              )
599          return self.debug_repr_helper("deque([", "])")
600  
601      def as_python_constant(self):
602          return self.python_type()(
603              [x.as_python_constant() for x in self.items],
604              maxlen=self.maxlen.as_python_constant(),
605          )
606  
607      def reconstruct(self, codegen: "PyCodegen") -> None:
608          codegen.add_push_null(
609              lambda: codegen.append_output(
610                  codegen.create_load_python_module(collections.deque)
611              )
612          )
613          codegen.foreach(self.items)
614          codegen.extend_output([create_instruction("BUILD_LIST", arg=len(self.items))])
615          codegen(self.maxlen)
616          codegen.extend_output(codegen.create_call_function_kw(2, ("maxlen",), False))
617  
618      def var_getattr(self, tx: "InstructionTranslator", name):
619          if name == "maxlen":
620              return self.maxlen
621          return super().var_getattr(tx, name)
622  
623      def call_method(
624          self,
625          tx,
626          name,
627          args: list["VariableTracker"],
628          kwargs: dict[str, "VariableTracker"],
629      ) -> "VariableTracker":
630          if (
631              name == "__setitem__"
632              and self.is_mutable()
633              and args
634              and args[0].is_python_constant()
635          ):
636              assert len(args) == 2
637              assert not kwargs
638              key, value = args
639              assert key.is_python_constant()
640              assert isinstance(key.as_python_constant(), int)
641              tx.output.side_effects.mutation(self)
642              self.items[key.as_python_constant()] = value
643              return ConstantVariable.create(None)
644  
645          maxlen = self.maxlen.as_python_constant()
646          if maxlen is not None:
647              slice_within_maxlen = slice(-maxlen, None)
648          else:
649              slice_within_maxlen = None
650  
651          if (
652              name == "extendleft"
653              and self.is_mutable()
654              and len(args) > 0
655              and args[0].has_force_unpack_var_sequence(tx)
656          ):
657              assert len(args) == 1
658              assert not kwargs
659              # NOTE this is inefficient, but the alternative is to represent self.items
660              # as a deque, which is a more intrusive change.
661              args[0].force_apply_to_var_sequence(
662                  tx, lambda item: self.call_method(tx, "appendleft", [item], {})
663              )
664              slice_within_maxlen = slice(None, maxlen)
665              result = ConstantVariable.create(None)
666          elif name == "popleft" and self.is_mutable():
667              assert not args
668              assert not kwargs
669              tx.output.side_effects.mutation(self)
670              result, *self.items[:] = self.items
671          elif name == "appendleft" and len(args) > 0 and self.is_mutable():
672              assert len(args) == 1
673              assert not kwargs
674              tx.output.side_effects.mutation(self)
675              self.items[:] = [args[0], *self.items]
676              slice_within_maxlen = slice(None, maxlen)
677              result = ConstantVariable.create(None)
678          elif name == "insert" and len(args) > 0 and self.is_mutable():
679              assert len(args) == 2
680              assert not kwargs
681              if maxlen is not None and len(self.items) == maxlen:
682                  raise_observed_exception(
683                      IndexError, tx, args=["deque already at its maximum size"]
684                  )
685              result = super().call_method(tx, name, args, kwargs)
686          else:
687              result = super().call_method(tx, name, args, kwargs)
688  
689          if (
690              slice_within_maxlen is not None
691              and maxlen is not None
692              and len(self.items) > maxlen
693          ):
694              self.items[:] = self.items[slice_within_maxlen]

739  class SizeVariable(TupleVariable):
740      """torch.Size(...)"""
741  
742      _nonvar_fields = {
743          "proxy",
744          *TupleVariable._nonvar_fields,
745      }
746  
747      def __init__(
748          self,
749          items: list[VariableTracker],
750          proxy: Optional[torch.fx.Proxy] = None,
751          **kwargs,
752      ) -> None:
753          self.proxy = proxy
754          super().__init__(items, **kwargs)
755  
756      def debug_repr(self):
757          return self.debug_repr_helper("torch.Size([", "])")
758  
759      def python_type(self):
760          return torch.Size
761  
762      def as_proxy(self):
763          if self.proxy is not None:
764              return self.proxy
765  
766          # torch.Size needs special handling.  Normally, we pun a list-like
767          # container to directly contain Proxy/Node objects from FX, and FX
768          # knows to look inside containers (via map_aggregate).  But torch.Size
769          # is weird; although it subclasses from tuple, it doesn't allow
770          # members which aren't int-like (rejecting Proxy and Node).  This
771          # means we can't use the normal representation trick
772          # torch.Size([proxy0, proxy1]).  I looked into seeing if I could
773          # relax torch.Size in PyTorch proper, but if torch.Size constructor
774          # sees a type that it doesn't recognize, it will try to call
775          # __index__() on it, so there is no BC way to actually change this
776          # behavior (though it occurs to me that I could have just added a
777          # YOLO no checking alternate constructor.)
778          #
779          # To work around this problem, I represent a torch.Size proxy as
780          # a straight up proxy, that would have been constructed by taking
781          # the constituent proxies as arguments.  This trick can be generally
782          # used for any construct that we need a proxy for but we can't
783          # directly represent as an aggregate; I don't see very many examples
784          # of this in torchdynamo though!
785  
786          # Look for a proxy.  If there are none, do the legacy behavior
787          tracer = None
788          proxies = self._as_proxy()
789          for proxy in proxies:
790              if isinstance(proxy, torch.fx.Proxy):
791                  tracer = proxy.tracer
792                  break
793  
794          if tracer is None:
795              return torch.Size(proxies)
796  
797          proxy = tracer.create_proxy("call_function", torch.Size, (proxies,), {})
798          set_example_value(
799              proxy.node,
800              torch.Size(
801                  [
802                      p.node.meta["example_value"] if not isinstance(p, int) else p
803                      for p in proxies
804                  ]
805              ),
806          )
807          return proxy
808  
809      def reconstruct(self, codegen: "PyCodegen") -> None:
810          codegen.add_push_null(lambda: codegen.load_import_from("torch", "Size"))
811          codegen.foreach(self.items)
812          build_torch_size = [
813              create_instruction("BUILD_TUPLE", arg=len(self.items)),
814          ] + create_call_function(1, False)
815          codegen.extend_output(build_torch_size)
816  
817      def unpack_var_sequence(self, tx):
818          return list(self.items)
819  
820      def numel(self, tx):
821          from .builtin import BuiltinVariable
822          from .tensor import SymNodeVariable
823  
824          const_result = 1
825          sym_sizes = []
826  
827          for v in self.items:
828              if isinstance(v, ConstantVariable):
829                  const_result *= v.value
830              else:
831                  assert isinstance(v, SymNodeVariable), type(v)
832                  # Delay proxy calls  until we know it will be necessary
833                  sym_sizes.append(v)
834  
835          result = ConstantVariable.create(const_result)
836          if sym_sizes and const_result == 1:
837              # Skip multiplying by 1
838              result, *sym_sizes = sym_sizes
839  
840          if not sym_sizes or const_result == 0:
841              return result
842  
843          mul = BuiltinVariable(operator.mul)
844          for v in sym_sizes:
845              result = mul.call_function(tx, [result, v], {})
846          return result
847  
848      def call_method(
849          self,
850          tx,
851          name,
852          args: list["VariableTracker"],
853          kwargs: dict[str, "VariableTracker"],
854      ) -> "VariableTracker":
855          if name == "__getitem__":
856              assert not kwargs and len(args) == 1
857              out = self.get_item_dyn(tx, args[0])
858              return out
859          elif name == "numel":
860              assert not args and not kwargs
861              return self.numel(tx)
862  
863          return super().call_method(tx, name, args, kwargs)
864  
865      def get_item_dyn(self, tx: "InstructionTranslator", arg: VariableTracker):
866          from .tensor import SymNodeVariable
867  
868          if isinstance(arg, SymNodeVariable):
869              index = arg.sym_num
870          else:
871              index = arg.as_python_constant()
872  
873          if isinstance(index, slice):
874              return SizeVariable(self.items[index])
875          else:
876              assert isinstance(index, (int, torch.SymInt))
877              return self.items[index]
878  
879      def call_obj_hasattr(
880          self, tx: "InstructionTranslator", name: str
881      ) -> "VariableTracker":
882          return variables.ConstantVariable.create(hasattr(torch.Size, name))

885  class NamedTupleVariable(TupleVariable):
886      _nonvar_fields = {
887          "tuple_cls",
888          "dynamic_attributes",
889          *TupleVariable._nonvar_fields,
890      }
891  
892      def __init__(self, items, tuple_cls, **kwargs) -> None:
893          super().__init__(items, **kwargs)
894          self.tuple_cls = tuple_cls
895          self.dynamic_attributes = {}
896  
897      def is_namedtuple(self):
898          return isinstance(getattr(self.tuple_cls, "_fields", None), tuple) and callable(
899              getattr(self.tuple_cls, "_make", None)
900          )
901  
902      def is_structseq(self):
903          return not self.is_namedtuple()
904  
905      def fields(self):
906          return namedtuple_fields(self.tuple_cls)
907  
908      def debug_repr(self):
909          if self.is_structseq():
910              # StructSequenceType(iterable)
911              return repr(self.tuple_cls([Lit(x.debug_repr()) for x in self.items]))
912          # NamedTupleType(*iterable)
913          return repr(self.tuple_cls(*(Lit(x.debug_repr()) for x in self.items)))
914  
915      def python_type(self):
916          return self.tuple_cls
917  
918      def as_python_constant(self):
919          if self.is_structseq():
920              # StructSequenceType(iterable)
921              return self.python_type()([x.as_python_constant() for x in self.items])
922          # NamedTupleType(*iterable)
923          return self.python_type()(*[x.as_python_constant() for x in self.items])
924  
925      def as_proxy(self):
926          assert self.python_type() is not SizeVariable
927          if self.is_structseq():
928              # StructSequenceType(iterable)
929              return self.python_type()(self._as_proxy())
930          # NamedTupleType(*iterable)
931          return self.python_type()(*self._as_proxy())
932  
933      def reconstruct(self, codegen: "PyCodegen") -> None:
934          # Constructors:
935          #   StructSequenceType(iterable)
936          #   NamedTupleType(*iterable)
937          #   NamedTupleType._make(iterable)
938          create_fn = self.tuple_cls if self.is_structseq() else self.tuple_cls._make
939          codegen.add_push_null(
940              lambda: codegen.append_output(
941                  codegen.create_load_const_unchecked(create_fn)
942              )
943          )
944          codegen.foreach(self.items)
945          codegen.extend_output(
946              [
947                  create_instruction("BUILD_TUPLE", arg=len(self.items)),
948              ]
949              + create_call_function(1, False)
950          )
951  
952      def call_method(
953          self,
954          tx,
955          name,
956          args: list[VariableTracker],
957          kwargs: dict[str, VariableTracker],
958      ) -> VariableTracker:
959          if name == "__setattr__":
960              assert len(args) == 2
961              assert len(kwargs) == 0
962              attr, value = args
963              attr = attr.as_python_constant()
964              if (
965                  # structseq is immutable
966                  self.is_structseq()
967                  # namedtuple directly created by `collections.namedtuple` is immutable
968                  or self.tuple_cls.__bases__ == (tuple,)
969                  # fields are immutable
970                  or attr in self.fields()
971              ):
972                  raise_observed_exception(AttributeError, tx)
973              # Subclass of namedtuple type can have dynamic attributes
974              tx.output.side_effects.mutation(self)
975              self.dynamic_attributes[attr] = value
976              return ConstantVariable.create(None)
977          return super().call_method(tx, name, args, kwargs)
978  
979      def var_getattr(self, tx: "InstructionTranslator", name):
980          def check_and_create_method():
981              method = inspect.getattr_static(self.tuple_cls, name, None)
982              if isinstance(method, classmethod):
983                  # We need the unbounded cls method to avoid the inline __self__
984                  return UserMethodVariable(
985                      method.__func__,
986                      variables.UserDefinedClassVariable(self.tuple_cls),
987                  )
988              elif isinstance(method, staticmethod):
989                  return UserFunctionVariable(method.__func__)
990              elif inspect.isfunction(method):
991                  return UserMethodVariable(method, self)
992              else:
993                  return None
994  
995          if name in self.dynamic_attributes:
996              return self.dynamic_attributes[name]
997  
998          fields = self.fields()
999          if name not in fields:
1000             method = check_and_create_method()
1001             if not method:
1002                 return super().var_getattr(tx, name)
1003             return method
1004         return self.items[fields.index(name)]
1005 
1006     def call_obj_hasattr(
1007         self, tx: "InstructionTranslator", name: str
1008     ) -> "VariableTracker":
1009         return variables.ConstantVariable.create(
1010             name in self.dynamic_attributes or hasattr(self.tuple_cls, name)
1011         )
