19   @dataclasses.dataclass
20   class ExportStatus:
21       # Whether torch.export.export(..., strict=True) succeeds
22       torch_export_strict: bool | None = None
23       # Whether torch.export.export(..., strict=False) succeeds
24       torch_export_non_strict: bool | None = None
25       # Whether torch.export._draft_export.draft_export() succeeds
26       torch_export_draft_export: bool | None = None
27       # Whether decomposition succeeds
28       decomposition: bool | None = None
29       # Whether ONNX translation succeeds
30       onnx_translation: bool | None = None
31       # Whether ONNX model passes onnx.checker.check_model
32       onnx_checker: bool | None = None
33       # Whether ONNX model runs successfully with ONNX Runtime
34       onnx_runtime: bool | None = None
35       # Whether the output of the ONNX model is accurate
36       output_accuracy: bool | None = None

39   def _status_emoji(status: bool | None) -> str:
40       if status is None:
41           return "⚪"
42       return "✅" if status else "❌"

45   def _format_export_status(status: ExportStatus) -> str:
46       return (
47           f"```\n"
48           f"{_status_emoji(status.torch_export_non_strict)} Obtain model graph with `torch.export.export(..., strict=False)`\n"
49           f"{_status_emoji(status.torch_export_strict)} Obtain model graph with `torch.export.export(..., strict=True)`\n"
50           f"{_status_emoji(status.torch_export_draft_export)} Obtain model graph with `torch.export._draft_export.draft_export`\n"
51           f"{_status_emoji(status.decomposition)} Decompose operators for ONNX compatibility\n"
52           f"{_status_emoji(status.onnx_translation)} Translate the graph into ONNX\n"
53           f"{_status_emoji(status.onnx_checker)} Run `onnx.checker` on the ONNX model\n"
54           f"{_status_emoji(status.onnx_runtime)} Execute the model with ONNX Runtime\n"
55           f"{_status_emoji(status.output_accuracy)} Validate model output accuracy\n"
56           f"```\n\n"
57       )

60   def _strip_color_from_string(text: str) -> str:
61       # This regular expression matches ANSI escape codes
62       # https://github.com/pytorch/pytorch/blob/9554a9af8788c57e1c5222c39076a5afcf0998ae/torch/_dynamo/utils.py#L2785-L2788
63       ansi_escape = re.compile(r"\x1B[@-_][0-?]*[ -/]*[@-~]")
64       return ansi_escape.sub("", text)

67   def _format_exported_program(exported_program: torch.export.ExportedProgram) -> str:
68       # Adapted from https://github.com/pytorch/pytorch/pull/128476
69       # to remove colors
70       # Even though we can call graph_module.print_readable directly, since the
71       # colored option was added only recently, we can't guarantee that the
72       # version of PyTorch used by the user has this option. Therefore, we
73       # still call str(ExportedProgram)
74       text = f"```python\n{_strip_color_from_string(str(exported_program))}\n```\n\n"
75       return text

78   def construct_report_file_name(timestamp: str, status: ExportStatus) -> str:
79       # Status could be None. So we need to check for False explicitly.
80       if not (
81           status.torch_export_non_strict
82           or status.torch_export_strict
83           or status.torch_export_draft_export
84       ):
85           # All strategies failed
86           postfix = "pt_export"
87       elif status.decomposition is False:
88           postfix = "decomp"
89       elif status.onnx_translation is False:
90           postfix = "conversion"
91       elif status.onnx_checker is False:
92           postfix = "checker"
93       elif status.onnx_runtime is False:
94           postfix = "runtime"
95       elif status.output_accuracy is False:
96           postfix = "accuracy"
97       elif (
98           status.torch_export_strict is False
99           or status.torch_export_non_strict is False
100          or status.torch_export_draft_export is False
101      ):
102          # Some strategies failed
103          postfix = "strategies"
104      else:
105          postfix = "success"
106      return f"onnx_export_{timestamp}_{postfix}.md"

109  def format_decomp_comparison(
110      pre_decomp_unique_ops: set[str],
111      post_decomp_unique_ops: set[str],
112  ) -> str:
113      """Format the decomposition comparison result. 
114   
115      Args: 
116          unique_ops_in_a: The unique ops in the first program. 
117          unique_ops_in_b: The unique ops in the second program. 
118   
119      Returns: 
120          The formatted comparison result. 
121      """
122      return (
123          f"Ops exist only in the ExportedProgram before decomposition: `{sorted(pre_decomp_unique_ops)}`\n\n"
124          f"Ops exist only in the ExportedProgram after decomposition: `{sorted(post_decomp_unique_ops)}`\n"
125      )

128  def format_verification_infos(
129      verification_infos: list[_verification.VerificationInfo],
130  ) -> str:
131      """Format the verification result. 
132   
133      Args: 
134          verification_infos: The verification result. 
135   
136      Returns: 
137          The formatted verification result. 
138      """
139      return "\n".join(
140          f"`{info.name}`: `max_abs_diff={info.max_abs_diff:e}`, `max_rel_diff={info.max_rel_diff:e}`, "
141          f"`abs_diff_hist={info.abs_diff_hist}`, `rel_diff_hist={info.rel_diff_hist}`"
142          for info in verification_infos
143      )

146  def create_torch_export_error_report(
147      filename: str | os.PathLike,
148      formatted_traceback: str,
149      *,
150      export_status: ExportStatus,
151      profile_result: str | None,
152  ):
153      with open(filename, "w", encoding="utf-8") as f:
154          f.write("# PyTorch ONNX Conversion Error Report\n\n")
155          f.write(_format_export_status(export_status))
156          f.write("Error message:\n\n")
157          f.write("```pytb\n")
158          f.write(formatted_traceback)
159          f.write("```\n\n")
160          if profile_result is not None:
161              f.write("## Profiling result\n\n")
162              f.write("```\n")
163              f.write(profile_result)
164              f.write("```\n")

167  def create_onnx_export_report(
168      filename: str | os.PathLike,
169      formatted_traceback: str,
170      program: torch.export.ExportedProgram,
171      *,
172      decomp_comparison: str | None = None,
173      export_status: ExportStatus,
174      profile_result: str | None,
175      model: ir.Model | None = None,
176      registry: _registration.ONNXRegistry | None = None,
177      verification_result: str | None = None,
178  ):
179      with open(filename, "w", encoding="utf-8") as f:
180          f.write("# PyTorch ONNX Conversion Report\n\n")
181          f.write(_format_export_status(export_status))
182          f.write("## Error messages\n\n")
183          f.write("```pytb\n")
184          f.write(formatted_traceback)
185          f.write("\n```\n\n")
186          f.write("## Exported program\n\n")
187          f.write(_format_exported_program(program))
188          if model is not None:
189              f.write("## ONNX model\n\n")
190              f.write("```python\n")
191              f.write(str(model))
192              f.write("\n```\n\n")
193          f.write("## Analysis\n\n")
194          _analysis.analyze(program, file=f, registry=registry)
195          if decomp_comparison is not None:
196              f.write("\n## Decomposition comparison\n\n")
197              f.write(decomp_comparison)
198              f.write("\n")
199          if verification_result is not None:
200              f.write("\n## Verification results\n\n")
201              f.write(verification_result)
202              f.write("\n")
203          if profile_result is not None:
204              f.write("\n## Profiling result\n\n")
205              f.write("```\n")
206              f.write(profile_result)
207              f.write("```\n")
