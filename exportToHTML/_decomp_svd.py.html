13   def svd(a, full_matrices=True, compute_uv=True, overwrite_a=False,
14           check_finite=True, lapack_driver='gesdd'):
15       """ 
16       Singular Value Decomposition. 
17    
18       Factorizes the matrix `a` into two unitary matrices ``U`` and ``Vh``, and 
19       a 1-D array ``s`` of singular values (real, non-negative) such that 
20       ``a == U @ S @ Vh``, where ``S`` is a suitably shaped matrix of zeros with 
21       main diagonal ``s``. 
22    
23       Parameters 
24       ---------- 
25       a : (M, N) array_like 
26           Matrix to decompose. 
27       full_matrices : bool, optional 
28           If True (default), `U` and `Vh` are of shape ``(M, M)``, ``(N, N)``. 
29           If False, the shapes are ``(M, K)`` and ``(K, N)``, where 
30           ``K = min(M, N)``. 
31       compute_uv : bool, optional 
32           Whether to compute also ``U`` and ``Vh`` in addition to ``s``. 
33           Default is True. 
34       overwrite_a : bool, optional 
35           Whether to overwrite `a`; may improve performance. 
36           Default is False. 
37       check_finite : bool, optional 
38           Whether to check that the input matrix contains only finite numbers. 
39           Disabling may give a performance gain, but may result in problems 
40           (crashes, non-termination) if the inputs do contain infinities or NaNs. 
41       lapack_driver : {'gesdd', 'gesvd'}, optional 
42           Whether to use the more efficient divide-and-conquer approach 
43           (``'gesdd'``) or general rectangular approach (``'gesvd'``) 
44           to compute the SVD. MATLAB and Octave use the ``'gesvd'`` approach. 
45           Default is ``'gesdd'``. 
46    
47       Returns 
48       ------- 
49       U : ndarray 
50           Unitary matrix having left singular vectors as columns. 
51           Of shape ``(M, M)`` or ``(M, K)``, depending on `full_matrices`. 
52       s : ndarray 
53           The singular values, sorted in non-increasing order. 
54           Of shape (K,), with ``K = min(M, N)``. 
55       Vh : ndarray 
56           Unitary matrix having right singular vectors as rows. 
57           Of shape ``(N, N)`` or ``(K, N)`` depending on `full_matrices`. 
58    
59       For ``compute_uv=False``, only ``s`` is returned. 
60    
61       Raises 
62       ------ 
63       LinAlgError 
64           If SVD computation does not converge. 
65    
66       See Also 
67       -------- 
68       svdvals : Compute singular values of a matrix. 
69       diagsvd : Construct the Sigma matrix, given the vector s. 
70    
71       Examples 
72       -------- 
73       >>> import numpy as np 
74       >>> from scipy import linalg 
75       >>> rng = np.random.default_rng() 
76       >>> m, n = 9, 6 
77       >>> a = rng.standard_normal((m, n)) + 1.j*rng.standard_normal((m, n)) 
78       >>> U, s, Vh = linalg.svd(a) 
79       >>> U.shape,  s.shape, Vh.shape 
80       ((9, 9), (6,), (6, 6)) 
81    
82       Reconstruct the original matrix from the decomposition: 
83    
84       >>> sigma = np.zeros((m, n)) 
85       >>> for i in range(min(m, n)): 
86       ...     sigma[i, i] = s[i] 
87       >>> a1 = np.dot(U, np.dot(sigma, Vh)) 
88       >>> np.allclose(a, a1) 
89       True 
90    
91       Alternatively, use ``full_matrices=False`` (notice that the shape of 
92       ``U`` is then ``(m, n)`` instead of ``(m, m)``): 
93    
94       >>> U, s, Vh = linalg.svd(a, full_matrices=False) 
95       >>> U.shape, s.shape, Vh.shape 
96       ((9, 6), (6,), (6, 6)) 
97       >>> S = np.diag(s) 
98       >>> np.allclose(a, np.dot(U, np.dot(S, Vh))) 
99       True 
100   
101      >>> s2 = linalg.svd(a, compute_uv=False) 
102      >>> np.allclose(s, s2) 
103      True 
104   
105      """
106      a1 = _asarray_validated(a, check_finite=check_finite)
107      if len(a1.shape) != 2:
108          raise ValueError('expected matrix')
109      m, n = a1.shape
110  
111      # accommodate empty matrix
112      if a1.size == 0:
113          u0, s0, v0 = svd(np.eye(2, dtype=a1.dtype))
114  
115          s = np.empty_like(a1, shape=(0,), dtype=s0.dtype)
116          if full_matrices:
117              u = np.empty_like(a1, shape=(m, m), dtype=u0.dtype)
118              u[...] = np.identity(m)
119              v = np.empty_like(a1, shape=(n, n), dtype=v0.dtype)
120              v[...] = np.identity(n)
121          else:
122              u = np.empty_like(a1, shape=(m, 0), dtype=u0.dtype)
123              v = np.empty_like(a1, shape=(0, n), dtype=v0.dtype)
124          if compute_uv:
125              return u, s, v
126          else:
127              return s
128  
129      overwrite_a = overwrite_a or (_datacopied(a1, a))
130  
131      if not isinstance(lapack_driver, str):
132          raise TypeError('lapack_driver must be a string')
133      if lapack_driver not in ('gesdd', 'gesvd'):
134          message = f'lapack_driver must be "gesdd" or "gesvd", not "{lapack_driver}"'
135          raise ValueError(message)
136  
137      if compute_uv:
138          # XXX: revisit int32 when ILP64 lapack becomes a thing
139          max_mn, min_mn = (m, n) if m > n else (n, m)
140          if full_matrices:
141              if max_mn*max_mn > np.iinfo(np.int32).max:
142                  raise ValueError(f"Indexing a matrix size {max_mn} x {max_mn} "
143                                    "would incur integer overflow in LAPACK. "
144                                    "Try using numpy.linalg.svd instead.")
145          else:
146              sz = max(m * min_mn, n * min_mn)
147              if max(m * min_mn, n * min_mn) > np.iinfo(np.int32).max:
148                  raise ValueError(f"Indexing a matrix of {sz} elements would "
149                                    "incur an in integer overflow in LAPACK. "
150                                    "Try using numpy.linalg.svd instead.")
151  
152      funcs = (lapack_driver, lapack_driver + '_lwork')
153      # XXX: As of 1.14.1 it isn't possible to build SciPy with ILP64,
154      # so the following line always yields a LP64 (32-bit pointer size) variant
155      gesXd, gesXd_lwork = get_lapack_funcs(funcs, (a1,), ilp64="preferred")
156  
157      # compute optimal lwork
158      lwork = _compute_lwork(gesXd_lwork, a1.shape[0], a1.shape[1],
159                             compute_uv=compute_uv, full_matrices=full_matrices)
160  
161      # perform decomposition
162      u, s, v, info = gesXd(a1, compute_uv=compute_uv, lwork=lwork,
163                            full_matrices=full_matrices, overwrite_a=overwrite_a)
164  
165      if info > 0:
166          raise LinAlgError("SVD did not converge")
167      if info < 0:
168          raise ValueError('illegal value in %dth argument of internal gesdd'
169                           % -info)
170      if compute_uv:
171          return u, s, v
172      else:
173          return s

176  def svdvals(a, overwrite_a=False, check_finite=True):
177      """ 
178      Compute singular values of a matrix. 
179   
180      Parameters 
181      ---------- 
182      a : (M, N) array_like 
183          Matrix to decompose. 
184      overwrite_a : bool, optional 
185          Whether to overwrite `a`; may improve performance. 
186          Default is False. 
187      check_finite : bool, optional 
188          Whether to check that the input matrix contains only finite numbers. 
189          Disabling may give a performance gain, but may result in problems 
190          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
191   
192      Returns 
193      ------- 
194      s : (min(M, N),) ndarray 
195          The singular values, sorted in decreasing order. 
196   
197      Raises 
198      ------ 
199      LinAlgError 
200          If SVD computation does not converge. 
201   
202      See Also 
203      -------- 
204      svd : Compute the full singular value decomposition of a matrix. 
205      diagsvd : Construct the Sigma matrix, given the vector s. 
206   
207      Examples 
208      -------- 
209      >>> import numpy as np 
210      >>> from scipy.linalg import svdvals 
211      >>> m = np.array([[1.0, 0.0], 
212      ...               [2.0, 3.0], 
213      ...               [1.0, 1.0], 
214      ...               [0.0, 2.0], 
215      ...               [1.0, 0.0]]) 
216      >>> svdvals(m) 
217      array([ 4.28091555,  1.63516424]) 
218   
219      We can verify the maximum singular value of `m` by computing the maximum 
220      length of `m.dot(u)` over all the unit vectors `u` in the (x,y) plane. 
221      We approximate "all" the unit vectors with a large sample. Because 
222      of linearity, we only need the unit vectors with angles in [0, pi]. 
223   
224      >>> t = np.linspace(0, np.pi, 2000) 
225      >>> u = np.array([np.cos(t), np.sin(t)]) 
226      >>> np.linalg.norm(m.dot(u), axis=0).max() 
227      4.2809152422538475 
228   
229      `p` is a projection matrix with rank 1. With exact arithmetic, 
230      its singular values would be [1, 0, 0, 0]. 
231   
232      >>> v = np.array([0.1, 0.3, 0.9, 0.3]) 
233      >>> p = np.outer(v, v) 
234      >>> svdvals(p) 
235      array([  1.00000000e+00,   2.02021698e-17,   1.56692500e-17, 
236               8.15115104e-34]) 
237   
238      The singular values of an orthogonal matrix are all 1. Here, we 
239      create a random orthogonal matrix by using the `rvs()` method of 
240      `scipy.stats.ortho_group`. 
241   
242      >>> from scipy.stats import ortho_group 
243      >>> orth = ortho_group.rvs(4) 
244      >>> svdvals(orth) 
245      array([ 1.,  1.,  1.,  1.]) 
246   
247      """
248      return svd(a, compute_uv=0, overwrite_a=overwrite_a,
249                 check_finite=check_finite)

252  def diagsvd(s, M, N):
253      """ 
254      Construct the sigma matrix in SVD from singular values and size M, N. 
255   
256      Parameters 
257      ---------- 
258      s : (M,) or (N,) array_like 
259          Singular values 
260      M : int 
261          Size of the matrix whose singular values are `s`. 
262      N : int 
263          Size of the matrix whose singular values are `s`. 
264   
265      Returns 
266      ------- 
267      S : (M, N) ndarray 
268          The S-matrix in the singular value decomposition 
269   
270      See Also 
271      -------- 
272      svd : Singular value decomposition of a matrix 
273      svdvals : Compute singular values of a matrix. 
274   
275      Examples 
276      -------- 
277      >>> import numpy as np 
278      >>> from scipy.linalg import diagsvd 
279      >>> vals = np.array([1, 2, 3])  # The array representing the computed svd 
280      >>> diagsvd(vals, 3, 4) 
281      array([[1, 0, 0, 0], 
282             [0, 2, 0, 0], 
283             [0, 0, 3, 0]]) 
284      >>> diagsvd(vals, 4, 3) 
285      array([[1, 0, 0], 
286             [0, 2, 0], 
287             [0, 0, 3], 
288             [0, 0, 0]]) 
289   
290      """
291      part = diag(s)
292      typ = part.dtype.char
293      MorN = len(s)
294      if MorN == M:
295          return np.hstack((part, zeros((M, N - M), dtype=typ)))
296      elif MorN == N:
297          return r_[part, zeros((M - N, N), dtype=typ)]
298      else:
299          raise ValueError("Length of s must be M or N.")

304  def orth(A, rcond=None):
305      """ 
306      Construct an orthonormal basis for the range of A using SVD 
307   
308      Parameters 
309      ---------- 
310      A : (M, N) array_like 
311          Input array 
312      rcond : float, optional 
313          Relative condition number. Singular values ``s`` smaller than 
314          ``rcond * max(s)`` are considered zero. 
315          Default: floating point eps * max(M,N). 
316   
317      Returns 
318      ------- 
319      Q : (M, K) ndarray 
320          Orthonormal basis for the range of A. 
321          K = effective rank of A, as determined by rcond 
322   
323      See Also 
324      -------- 
325      svd : Singular value decomposition of a matrix 
326      null_space : Matrix null space 
327   
328      Examples 
329      -------- 
330      >>> import numpy as np 
331      >>> from scipy.linalg import orth 
332      >>> A = np.array([[2, 0, 0], [0, 5, 0]])  # rank 2 array 
333      >>> orth(A) 
334      array([[0., 1.], 
335             [1., 0.]]) 
336      >>> orth(A.T) 
337      array([[0., 1.], 
338             [1., 0.], 
339             [0., 0.]]) 
340   
341      """
342      u, s, vh = svd(A, full_matrices=False)
343      M, N = u.shape[0], vh.shape[1]
344      if rcond is None:
345          rcond = np.finfo(s.dtype).eps * max(M, N)
346      tol = np.amax(s, initial=0.) * rcond
347      num = np.sum(s > tol, dtype=int)
348      Q = u[:, :num]
349      return Q

352  def null_space(A, rcond=None, *, overwrite_a=False, check_finite=True,
353                 lapack_driver='gesdd'):
354      """ 
355      Construct an orthonormal basis for the null space of A using SVD 
356   
357      Parameters 
358      ---------- 
359      A : (M, N) array_like 
360          Input array 
361      rcond : float, optional 
362          Relative condition number. Singular values ``s`` smaller than 
363          ``rcond * max(s)`` are considered zero. 
364          Default: floating point eps * max(M,N). 
365      overwrite_a : bool, optional 
366          Whether to overwrite `a`; may improve performance. 
367          Default is False. 
368      check_finite : bool, optional 
369          Whether to check that the input matrix contains only finite numbers. 
370          Disabling may give a performance gain, but may result in problems 
371          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
372      lapack_driver : {'gesdd', 'gesvd'}, optional 
373          Whether to use the more efficient divide-and-conquer approach 
374          (``'gesdd'``) or general rectangular approach (``'gesvd'``) 
375          to compute the SVD. MATLAB and Octave use the ``'gesvd'`` approach. 
376          Default is ``'gesdd'``. 
377   
378      Returns 
379      ------- 
380      Z : (N, K) ndarray 
381          Orthonormal basis for the null space of A. 
382          K = dimension of effective null space, as determined by rcond 
383   
384      See Also 
385      -------- 
386      svd : Singular value decomposition of a matrix 
387      orth : Matrix range 
388   
389      Examples 
390      -------- 
391      1-D null space: 
392   
393      >>> import numpy as np 
394      >>> from scipy.linalg import null_space 
395      >>> A = np.array([[1, 1], [1, 1]]) 
396      >>> ns = null_space(A) 
397      >>> ns * np.copysign(1, ns[0,0])  # Remove the sign ambiguity of the vector 
398      array([[ 0.70710678], 
399             [-0.70710678]]) 
400   
401      2-D null space: 
402   
403      >>> from numpy.random import default_rng 
404      >>> rng = default_rng() 
405      >>> B = rng.random((3, 5)) 
406      >>> Z = null_space(B) 
407      >>> Z.shape 
408      (5, 2) 
409      >>> np.allclose(B.dot(Z), 0) 
410      True 
411   
412      The basis vectors are orthonormal (up to rounding error): 
413   
414      >>> Z.T.dot(Z) 
415      array([[  1.00000000e+00,   6.92087741e-17], 
416             [  6.92087741e-17,   1.00000000e+00]]) 
417   
418      """
419      u, s, vh = svd(A, full_matrices=True, overwrite_a=overwrite_a,
420                     check_finite=check_finite, lapack_driver=lapack_driver)
421      M, N = u.shape[0], vh.shape[1]
422      if rcond is None:
423          rcond = np.finfo(s.dtype).eps * max(M, N)
424      tol = np.amax(s, initial=0.) * rcond
425      num = np.sum(s > tol, dtype=int)
426      Q = vh[num:,:].T.conj()
427      return Q

430  def subspace_angles(A, B):
431      r""" 
432      Compute the subspace angles between two matrices. 
433   
434      Parameters 
435      ---------- 
436      A : (M, N) array_like 
437          The first input array. 
438      B : (M, K) array_like 
439          The second input array. 
440   
441      Returns 
442      ------- 
443      angles : ndarray, shape (min(N, K),) 
444          The subspace angles between the column spaces of `A` and `B` in 
445          descending order. 
446   
447      See Also 
448      -------- 
449      orth 
450      svd 
451   
452      Notes 
453      ----- 
454      This computes the subspace angles according to the formula 
455      provided in [1]_. For equivalence with MATLAB and Octave behavior, 
456      use ``angles[0]``. 
457   
458      .. versionadded:: 1.0 
459   
460      References 
461      ---------- 
462      .. [1] Knyazev A, Argentati M (2002) Principal Angles between Subspaces 
463             in an A-Based Scalar Product: Algorithms and Perturbation 
464             Estimates. SIAM J. Sci. Comput. 23:2008-2040. 
465   
466      Examples 
467      -------- 
468      An Hadamard matrix, which has orthogonal columns, so we expect that 
469      the suspace angle to be :math:`\frac{\pi}{2}`: 
470   
471      >>> import numpy as np 
472      >>> from scipy.linalg import hadamard, subspace_angles 
473      >>> rng = np.random.default_rng() 
474      >>> H = hadamard(4) 
475      >>> print(H) 
476      [[ 1  1  1  1] 
477       [ 1 -1  1 -1] 
478       [ 1  1 -1 -1] 
479       [ 1 -1 -1  1]] 
480      >>> np.rad2deg(subspace_angles(H[:, :2], H[:, 2:])) 
481      array([ 90.,  90.]) 
482   
483      And the subspace angle of a matrix to itself should be zero: 
484   
485      >>> subspace_angles(H[:, :2], H[:, :2]) <= 2 * np.finfo(float).eps 
486      array([ True,  True], dtype=bool) 
487   
488      The angles between non-orthogonal subspaces are in between these extremes: 
489   
490      >>> x = rng.standard_normal((4, 3)) 
491      >>> np.rad2deg(subspace_angles(x[:, :2], x[:, [2]])) 
492      array([ 55.832])  # random 
493      """
494      # Steps here omit the U and V calculation steps from the paper
495  
496      # 1. Compute orthonormal bases of column-spaces
497      A = _asarray_validated(A, check_finite=True)
498      if len(A.shape) != 2:
499          raise ValueError(f'expected 2D array, got shape {A.shape}')
500      QA = orth(A)
501      del A
502  
503      B = _asarray_validated(B, check_finite=True)
504      if len(B.shape) != 2:
505          raise ValueError(f'expected 2D array, got shape {B.shape}')
506      if len(B) != len(QA):
507          raise ValueError('A and B must have the same number of rows, got '
508                           f'{QA.shape[0]} and {B.shape[0]}')
509      QB = orth(B)
510      del B
511  
512      # 2. Compute SVD for cosine
513      QA_H_QB = dot(QA.T.conj(), QB)
514      sigma = svdvals(QA_H_QB)
515  
516      # 3. Compute matrix B
517      if QA.shape[1] >= QB.shape[1]:
518          B = QB - dot(QA, QA_H_QB)
519      else:
520          B = QA - dot(QB, QA_H_QB.T.conj())
521      del QA, QB, QA_H_QB
522  
523      # 4. Compute SVD for sine
524      mask = sigma ** 2 >= 0.5
525      if mask.any():
526          mu_arcsin = arcsin(clip(svdvals(B, overwrite_a=True), -1., 1.))
527      else:
528          mu_arcsin = 0.
529  
530      # 5. Compute the principal angles
531      # with reverse ordering of sigma because smallest sigma belongs to largest
532      # angle theta
533      theta = where(mask, mu_arcsin, arccos(clip(sigma[::-1], -1., 1.)))
534      return theta
