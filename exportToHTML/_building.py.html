42   # Logic for adapting inputs from general Python or PyTorch inputs to ONNX ir.Value
43   def _construct_named_inputs_and_attrs(
44       signature: _schemas.OpSignature,
45       args: Sequence[AllowedArgType],
46       kwargs: Mapping[str, AllowedArgType],
47   ) -> tuple[dict[str, AllowedArgType], dict[str, ValidAttributeType]]:
48       """Construct two mappings: name to inputs and named to attributes based on the signature and args/kwargs. 
49    
50       This function uses the OpSignature to determine which argument in args and kwargs corresponds to 
51       which parameter in the signature. ONNX node inputs are stored in named_inputs, and attributes are 
52       stored in named_attrs. If an _optional input_ is not provided, it is filled with None. 
53    
54       Args: 
55           signature: The OpSignature for the node. 
56           args: The positional arguments for the node. 
57           kwargs: The keyword arguments for the node. 
58    
59       Returns: 
60           A tuple of two mappings: named_inputs and named_attrs. 
61    
62       Raises: 
63           ValueError: If a required parameter is not provided. 
64       """
65       # 1. Construct the (named_inputs, named_attrs) mapping based on (args, kwargs) and the signature.
66       #   a. Loop over all parameters in the signature and args together
67       #   b. Depending on param.is_input, Record named_inputs[param.name] = arg or named_attrs[param.name] = arg
68       #   c. Handle kwargs as well
69       #   d. Fill in None if the input is not provided
70       named_inputs: dict[str, Any] = {}
71       named_attrs: dict[str, Any] = {}
72       reversed_args_stack = list(reversed(args))
73       for param in signature.params:
74           if isinstance(param, _schemas.Parameter):
75               # Handle inputs
76               if reversed_args_stack:
77                   # First exhaust the positional arguments
78                   if param.variadic:
79                       # Handle variadic arguments
80                       named_inputs[param.name] = tuple(args)
81                       reversed_args_stack.clear()
82                   else:
83                       named_inputs[param.name] = reversed_args_stack.pop()  # type: ignore[assignment]
84               elif param.name in kwargs:
85                   named_inputs[param.name] = kwargs[param.name]  # type: ignore[assignment]
86               elif param.required:
87                   raise ValueError(
88                       f"Required parameter '{param.name}' is not provided. "
89                       f"Signature: {signature}. Args: {args}. Kwargs: {kwargs}."
90                   )
91               else:
92                   logger.debug(
93                       "Optional parameter '%s' is not provided. Added as None. Signature: %s",
94                       param.name,
95                       signature,
96                   )
97                   named_inputs[param.name] = None  # type: ignore[assignment]
98           else:
99               # Handle attributes
100              attribute: ValidAttributeType | ir.Attr
101              assert isinstance(param, _schemas.AttributeParameter), (
102                  f"Expected AttributeParameter, got {type(param)}"
103              )
104              if reversed_args_stack:
105                  # First exhaust the positional arguments
106                  attribute = reversed_args_stack.pop()  # type: ignore[assignment]
107              elif param.name in kwargs:
108                  attribute = kwargs[param.name]  # type: ignore[assignment]
109              elif param.default is not None:
110                  attribute = param.default
111              else:
112                  attribute = None
113  
114              if attribute is None:
115                  if param.required:
116                      raise ValueError(
117                          f"Required attribute '{param.name}' is not provided. "
118                          f"Signature: {signature}. Args: {args}. Kwargs: {kwargs}."
119                      )
120                  else:
121                      logger.debug(
122                          "Optional attribute '%s' is None. Dropped. Signature: %s",
123                          param.name,
124                          signature,
125                      )
126                      continue
127  
128              if isinstance(attribute, ir.Attr):
129                  # Turn the attribute from an default value into an actual parameter for the node
130                  attr_copied = copy.copy(attribute)
131                  # Make sure the name is the same as the parameter name and not the name of the default parameter
132                  attr_copied.name = param.name
133                  attribute = attr_copied
134  
135              if isinstance(attribute, int) and param.type == ir.AttributeType.FLOAT:
136                  # Convert the attribute to float if needed. This happens in PyTorch
137                  # where an attribute marked as float can be passed as an int.
138                  attribute = float(attribute)
139              named_attrs[param.name] = attribute
140      return named_inputs, named_attrs  # type: ignore[return-value]

143  def _resolve_parameter_dtypes(
144      signature: _schemas.OpSignature, named_inputs: Mapping[str, AllowedArgType]
145  ) -> Mapping[_schemas.TypeConstraintParam, ir.TypeProtocol]:
146      """Determine which parameter takes which type. 
147   
148      Handle non-tensor input corner cases and type promotion. 
149   
150      Requires: 
151          All ir.Value in name_inputs should have type set. Their type should be 
152          compatible with the type_constraint of the corresponding parameter in the signature. 
153   
154      Args: 
155          signature: The OpSignature for the node. 
156          named_inputs: The mapping of parameter names to their arguments. 
157   
158      Returns: 
159          A mapping of Constraint names to ir.TypeProtocol. 
160      """
161      #   a. Create type_binding: dict[str, ir.TypeProtocol]
162      #   b. Iterate over all named_inputs
163      #   b0. Find the corresponding parameter in the signature
164      #   b1. If the argument is a Python constant, skip.
165      #   b2. If the argument is a ir.Value, Bind {constraint: arg.type}.
166      type_binding = {}
167      for name, arg in named_inputs.items():
168          param = signature.params_map[name]
169          assert isinstance(param, _schemas.Parameter), (
170              f"Expected Parameter, got {type(param)}"
171          )
172          if isinstance(arg, (int, float, bool, str, Sequence, torch.Tensor)):
173              # Skip the Python constants because we do not know what dtype they should take yet
174              continue
175          elif isinstance(arg, ir.Value):
176              if arg.type is None:
177                  # Skip the ir.Value if the type is not set
178                  continue
179              # NOTE: We assume arg.type is compatible with the type_constraint
180              assert arg.type is not None, f"Expected type to be set for {arg}"
181              # TODO(justinchuby): Implement type promotion logic here.
182              type_binding[param.type_constraint] = arg.type
183      return type_binding

186  def _determine_input_dtype(
187      param: _schemas.Parameter,
188      arg: AllowedArgType,
189      type_binding: Mapping[_schemas.TypeConstraintParam, ir.TypeProtocol],
190  ) -> ir.DataType:
191      """Determine the dtype of the input that is a mix of Python constants and ir.Value."""
192      if param.type_constraint in type_binding:
193          # A known dtype is available because it was resolved
194          return type_binding[param.type_constraint].dtype
195      if len(param.type_constraint.allowed_types) == 1:
196          # Only one type is allowed by the type constraint
197          return next(iter(param.type_constraint.allowed_types)).dtype
198  
199      # No dtype information available. Infer from the Python constant or (in the Sequence case)
200      # from a mix of Python constants and ir.Value
201      if isinstance(arg, bool):
202          return ir.DataType.BOOL
203      if isinstance(arg, float):
204          return ir.DataType.FLOAT
205      if isinstance(arg, int):
206          return ir.DataType.INT64
207      if isinstance(arg, str):
208          return ir.DataType.STRING
209      if isinstance(arg, (ir.Tensor, ir.TensorProtocol)):
210          return arg.dtype
211      if isinstance(arg, complex):
212          return ir.DataType.FLOAT
213      if arg is None:
214          return ir.DataType.UNDEFINED
215  
216      # Handle sequences
217      if isinstance(arg, (tuple, list)):
218          if len(arg) == 0:
219              # Special case: Treat empty sequence as INT64 as they are typically used for shape
220              return ir.DataType.INT64
221  
222          # Try to obtain the dtype from one of the values
223          for val in arg:
224              if isinstance(val, ir.Value) and val.dtype is not None:
225                  return val.dtype
226  
227          if any(isinstance(val, float) for val in arg):
228              # If any float is present, the dtype is float
229              return ir.DataType.FLOAT
230          elif any(isinstance(val, int) for val in arg):
231              # Otherwise if any int is present, the dtype is int
232              return ir.DataType.INT64
233  
234      raise ValueError(
235          f"Could not determine the dtype for the input '{param.name}'. "
236          f"param={param}, arg={arg}, param_type_constraint={param.type_constraint}, "
237          f"type_binding={type_binding}"
238      )

241  def _allowed_types_are_sequence_types(allowed_types: Iterable[ir.TypeProtocol]) -> bool:
242      """Check if all allowed types are Sequence types."""
243      return all(isinstance(t, ir.SequenceType) for t in allowed_types)

246  def _get_or_create_constant(
247      constant_farm: dict[
248          tuple[
249              bool | int | float | str | tuple[int] | tuple[float],
250              ir.DataType,
251          ],
252          ir.Value,
253      ],
254      arg: bool
255      | int
256      | float
257      | str
258      | tuple[int]
259      | tuple[float]
260      | tuple[bool]
261      | list[int]
262      | list[float]
263      | list[bool],
264      dtype: ir.DataType,
265      opset: onnxscript.values.Opset,
266  ) -> ir.Value:
267      # float representation of complex numbers
268      if isinstance(arg, complex):
269          # Convert the complex number to a float
270          arg = (arg.real, arg.imag)
271  
272      if isinstance(arg, list):
273          # Make the arg hashable
274          arg = tuple(arg)  # type: ignore[assignment]
275  
276      constant_value = constant_farm.get((arg, dtype))  # type: ignore[arg-type]
277      if constant_value is None:
278          constant_tensor = ir.tensor(value=arg, dtype=dtype)  # type: ignore[arg-type]
279          constant_value = opset.Constant(value=constant_tensor)
280          constant_farm[(arg, dtype)] = constant_value  # type: ignore[arg-type,index]
281      return constant_value

284  def _process_python_constants(
285      signature: _schemas.OpSignature,
286      named_inputs: dict[str, AllowedArgType],
287      type_binding: Mapping[_schemas.TypeConstraintParam, ir.TypeProtocol],
288      constant_farm: dict[
289          tuple[
290              bool | int | float | str | tuple[int] | tuple[float],
291              ir.DataType,
292          ],
293          ir.Value,
294      ],
295      opset: onnxscript.values.Opset,
296  ) -> dict[str, ir.Value | None]:
297      """Convert Python constants to Constant nodes and list to Sequence nodes based on the dtype information. 
298   
299      The added constants will be replacing values in named_inputs in place. 
300   
301      Args: 
302          signature: The OpSignature for the node. 
303          named_inputs: The mapping of parameter names to their arguments. 
304          type_binding: A mapping of Constraint names to ir.DataType. 
305          constant_farm: A dictionary of {(py_value, ir.DataType): ir.Value} to store the deduplicated constants. 
306          opset: The Opset to use for creating Constant nodes. 
307   
308      Returns: 
309          A mapping of parameter names to Python constants converted to constant Nodes. 
310      """
311      # 3. Convert Python constants to Constant nodes based on the dtype information;
312      #    construct sequences
313      #   a. Iterate over all parameters in the signature the second time
314      #   b. If the parameter is in to_resolve_type:
315      #       - If param.constraint in type_binding,
316      #         Get the constant from constant_farm (deduplicated);
317      #            otherwise set named_inputs[param.name] = Constant(value, dtype=type_binding[param.constraint])
318      #       - Otherwise, set named_inputs[param.name] = Constant(value)
319      for name, arg in named_inputs.items():
320          param = signature.params_map[name]
321          assert isinstance(param, _schemas.Parameter), (
322              f"Expected Parameter, got {type(param)}"
323          )
324  
325          if isinstance(arg, ir.Value):
326              # TODO(justinchuby): Cast the ir.Value here if needed
327              continue
328  
329          if (
330              isinstance(arg, Sequence)
331              and len(arg) > 0
332              and any(isinstance(val, ir.Value) for val in arg)
333          ):
334              # Skip the sequence of ir.Value. This is a variadic input or a Sequence input
335              # It will be handled by _process_python_sequences
336              continue
337          if param.variadic:
338              # Handled by _process_python_sequences
339              continue
340          if _allowed_types_are_sequence_types(param.type_constraint.allowed_types):
341              # Handled by _process_python_sequences
342              continue
343  
344          dtype = _determine_input_dtype(param, arg, type_binding)
345  
346          if arg is None:
347              constant_value = None
348          elif isinstance(arg, (ir.Tensor, ir.TensorProtocol)):
349              constant_value = opset.Constant(value=arg)
350          else:
351              # Deduplicate the constants
352              constant_value = _get_or_create_constant(constant_farm, arg, dtype, opset)  # type: ignore[arg-type]
353  
354          named_inputs[param.name] = constant_value
355      return named_inputs  # type: ignore[return-value]

358  def _reshape_to_1d_tensor(opset: onnxscript.values.Opset, arg: ir.Value) -> ir.Value:
359      """Reshape the input to a 1D tensor."""
360  
361      return opset.Reshape(
362          arg, opset.Constant(value=ir.tensor([-1], dtype=ir.DataType.INT64))
363      )

366  def _process_python_sequences(
367      signature: _schemas.OpSignature,
368      named_inputs: dict[str, AllowedArgType],
369      type_binding: Mapping[_schemas.TypeConstraintParam, ir.TypeProtocol],
370      constant_farm: dict[
371          tuple[
372              bool | int | float | str | ir.TensorProtocol | tuple[int] | tuple[float],
373              ir.DataType,
374          ],
375          ir.Value,
376      ],
377      opset: onnxscript.values.Opset,
378  ):
379      """Handle three types of sequences. 
380   
381      1. Variadic inputs 
382      2. Sequence input of ir.Value, 
383      3. Sequence of Python constants that contains ir.Value 
384      """
385      for name, arg in named_inputs.items():
386          param = signature.params_map[name]
387          assert isinstance(param, _schemas.Parameter), (
388              f"Expected Parameter, got {type(param)}"
389          )
390  
391          if not isinstance(arg, (tuple, list)):
392              continue
393  
394          if len(arg) == 0:
395              # Skip empty sequences
396              continue
397  
398          # 1. Sequence input of ir.Value
399          if _allowed_types_are_sequence_types(param.type_constraint.allowed_types):
400              # Turn the list into a Sequence node
401              # Constant op creation will be handled by the variadic case below when calling
402              # the SequenceConstruct op.
403              named_inputs[name] = opset.SequenceConstruct(*arg)
404              continue
405  
406          # 2. Variadic inputs
407          # NOTE: Variadic operators like Max can be called with mixed ir.Value and Python constants
408          # like `Max(0, ir.Value())`
409          # We need to convert the Python constants to Constant nodes
410          if param.variadic:
411              if all(isinstance(val, ir.Value) for val in arg):
412                  # Skip the variadic input if all values are ir.Value
413                  continue
414  
415              dtype = _determine_input_dtype(param, arg, type_binding)
416              new_args = []
417              for val in arg:
418                  if isinstance(val, ir.Value):
419                      new_args.append(val)
420                  else:
421                      constant_tensor = ir.tensor(value=val, dtype=dtype)  # type: ignore[arg-type]
422                      constant_value = opset.Constant(value=constant_tensor)
423                      new_args.append(constant_value)
424              named_inputs[name] = new_args
425              continue
426          else:
427              # 3. Concat the list as a single input
428              # E.g. [Value, 42] should be converted to op.Concat(Value, Constant(42))
429              # when the expected input type is INT64
430              # We assume this only happens for 0D cases
431              if all(isinstance(val, ir.Value) for val in arg):
432                  expanded_args = [_reshape_to_1d_tensor(opset, val) for val in arg]
433                  named_inputs[name] = opset.Concat(*expanded_args, axis=0)
434                  continue
435  
436              dtype = _determine_input_dtype(param, arg, type_binding)
437              new_args = []
438              for val in arg:
439                  if isinstance(val, ir.Value):
440                      new_args.append(_reshape_to_1d_tensor(opset, val))
441                  elif val is None:
442                      # Skip None values
443                      continue
444                  elif isinstance(val, (ir.Tensor, ir.TensorProtocol)):
445                      new_args.append(
446                          _reshape_to_1d_tensor(opset, opset.Constant(value=val))
447                      )
448                  else:
449                      # Turn the Python constant into 1D tensor for the constant
450                      assert isinstance(val, (bool, int, float)), (
451                          f"Expected int or float, got {type(val)}"
452                      )
453                      new_args.append(
454                          _get_or_create_constant(constant_farm, [val], dtype, opset)  # type: ignore[arg-type]
455                      )
456              named_inputs[name] = opset.Concat(*new_args, axis=0)
457              continue
458      return named_inputs

461  def _determine_output_number(
462      signature: _schemas.OpSignature, named_attrs: Mapping[str, ValidAttributeType]
463  ) -> int:
464      """Determine the number of outputs for the node with heuristics."""
465      if signature.domain == "":
466          if signature.name == "BatchNormalization":
467              if not named_attrs.get("training_mode", 0):
468                  return 1
469          if signature.name == "Split":
470              num_outputs = named_attrs.get("num_outputs")
471              if num_outputs is not None and isinstance(num_outputs, int):
472                  return num_outputs
473              else:
474                  raise ValueError(
475                      "Could not determine the number of outputs for Split. "
476                      "num_outputs must be provided"
477                  )
478      return len(signature.outputs)

481  def _construct_node(
482      signature: _schemas.OpSignature,
483      named_inputs: Mapping[str, ir.Value | None],
484      named_attrs: Mapping[str, ValidAttributeType],
485      opset: onnxscript.values.Opset,
486      num_outputs: int,
487  ) -> ir.Node:
488      """Construct the node with the inputs and attributes. 
489   
490      Variadic inputs are flattened. 
491   
492      Args: 
493          signature: The OpSignature for the node. 
494          named_inputs: The mapping of parameter names to their arguments. When we 
495              do not have the schema of an operator, we do not know the names of 
496              the inputs, in which case the names can be anything because they 
497              are not used in this function. The data structure is passed in for 
498              consistency with the other functions. 
499          named_attrs: The mapping of attribute names to their values. 
500          num_outputs: The number of outputs for the node. 
501      """
502      inputs: list[ir.Value | None] = []
503      # Flatten variadic inputs
504      for value in named_inputs.values():
505          if isinstance(value, Sequence):
506              inputs.extend(value)
507          else:
508              inputs.append(value)
509  
510      # If final inputs are None, strip them from the node inputs
511      for input in reversed(inputs):
512          if input is not None:
513              break
514          inputs.pop()
515  
516      # Construct and filter out None attributes
517      attributes = [
518          attr
519          for attr in ir_convenience.convert_attributes(named_attrs)
520          if attr.value is not None
521      ]
522      outputs = [_tensors.SymbolicTensor(opset) for _ in range(num_outputs)]
523      return ir.Node(
524          signature.domain,
525          signature.name,
526          inputs=inputs,
527          attributes=attributes,
528          outputs=outputs,
529          version=signature.opset_version,
530      )

533  class OpRecorder(evaluator.Evaluator):
534      """An onnxscript Evaluator that captures the graph into ONNX IR."""
535  
536      def __init__(
537          self, opset: onnxscript.values.Opset, constant_farm: dict[Any, ir.Value]
538      ):
539          self.nodes: list[ir.Node] = []
540          self.opset = opset
541          self.functions: dict[
542              ir.OperatorIdentifier, onnxscript.OnnxFunction | ir.Function
543          ] = {}
544          self.constant_farm = constant_farm
545  
546      def _call_op(
547          self,
548          op_signature: _schemas.OpSignature,
549          named_inputs: dict[str, AllowedArgType],
550          named_attrs: dict[str, ValidAttributeType],
551          num_outputs: int,
552      ) -> Sequence[_tensors.SymbolicTensor]:
553          """Record nodes for the given opschema and arguments. 
554   
555          Args: 
556              op_signature: The OpSchema containing the node signature. 
557              named_inputs: The mapping of parameter names to their arguments. 
558              named_attrs: The mapping of attribute names to their values. 
559          """
560          type_binding = _resolve_parameter_dtypes(op_signature, named_inputs)
561          try:
562              converted_named_inputs = _process_python_constants(
563                  op_signature, named_inputs, type_binding, self.constant_farm, self.opset
564              )
565              converted_named_inputs = _process_python_sequences(
566                  op_signature,
567                  converted_named_inputs,  # type: ignore[arg-type]
568                  type_binding,
569                  self.constant_farm,
570                  self.opset,
571              )
572  
573          except Exception as e:
574              raise _errors.GraphConstructionError(
575                  f"Error processing Python constants for operator '{op_signature.domain}::{op_signature.name}'. "
576                  f"named_inputs={named_inputs}, named_attrs={named_attrs}, opset={self.opset}, op_signature={op_signature}."
577              ) from e
578  
579          try:
580              self.nodes.append(
581                  node := _construct_node(
582                      op_signature,
583                      converted_named_inputs,
584                      named_attrs,
585                      self.opset,
586                      num_outputs,
587                  )
588              )
589          except Exception as e:
590              raise _errors.GraphConstructionError(
591                  f"Error constructing node for operator '{op_signature.domain}::{op_signature.name}'. "
592                  f"named_inputs={named_inputs}, converted_named_inputs={converted_named_inputs}, "
593                  f"named_attrs={named_attrs}, opset={self.opset}, op_signature={op_signature}."
594              ) from e
595          return node.outputs  # type: ignore[return-value]

597      def eval(
598          self,
599          schema: onnx.defs.OpSchema,
600          args: Sequence[AllowedArgType],  # type: ignore[override]
601          kwargs: Mapping[str, AllowedArgType],
602      ) -> _tensors.SymbolicTensor | Sequence[_tensors.SymbolicTensor]:
603          try:
604              op_signature = _schemas.OpSignature.from_opschema(schema)
605              named_inputs, named_attrs = _construct_named_inputs_and_attrs(
606                  op_signature, args, kwargs
607              )
608              # TODO(justinchuby): Handle cast
609              if schema.name == "CastLike":
610                  assert len(named_inputs) == 2
611                  # Skip CastLike if the input and output types are the same
612                  src_input = named_inputs["input"]
613                  target_type = named_inputs["target_type"]
614  
615                  if (
616                      isinstance(src_input, ir.Value)
617                      and isinstance(target_type, ir.Value)
618                      and src_input.dtype is not None
619                      and target_type.dtype is not None
620                  ):
621                      # dtypes are available
622                      if src_input.dtype == target_type.dtype:
623                          # Same type. No cast needed
624                          return src_input  # type: ignore[return-value]
625                      else:
626                          # Create a Cast node
627                          return self.opset.Cast(src_input, to=target_type.dtype)  # type: ignore[union-attr,return-value]
628  
629              num_outputs = _determine_output_number(op_signature, named_attrs)
630              outputs = self._call_op(
631                  op_signature, named_inputs, named_attrs, num_outputs
632              )
633              if len(outputs) == 1:
634                  return outputs[0]
635              return outputs
636          except Exception as e:
637              raise _errors.GraphConstructionError(
638                  f"Error calling operator '{schema.name}' with args {args} and kwargs {kwargs}."
639              ) from e

641      def eval_function(  # type: ignore[override]
642          self,
643          function: onnxscript.OnnxFunction,
644          args: Sequence[AllowedArgType],
645          kwargs: Mapping[str, AllowedArgType],
646      ) -> _tensors.SymbolicTensor | Sequence[_tensors.SymbolicTensor] | bool | int:
647          try:
648              # TODO(justinchuby): Remove this once IsScalar and Rank are removed
649              # Special cases for handling IsScalar and Rank
650              if function.name == "IsScalar":
651                  if len(args) != 1:
652                      raise TypeError(
653                          f"Expected 1 positional argument for function '{function}', got {len(args)}."
654                      )
655                  if isinstance(args[0], _tensors.SymbolicTensor):
656                      if args[0].rank is not None:
657                          return args[0].rank == 0
658                      else:
659                          # Fall to call add_function_call
660                          pass
661                  elif isinstance(args[0], Sequence):
662                      return False
663                  else:
664                      # Python constants are scalars
665                      return True
666              if function.name == "Rank":
667                  if len(args) != 1:
668                      raise TypeError(
669                          f"Expected 1 positional argument for function '{function}', got {len(args)}."
670                      )
671                  if isinstance(args[0], _tensors.SymbolicTensor):
672                      if args[0].rank is not None:
673                          return args[0].rank
674                      else:
675                          # Fall to call add_function_call
676                          pass
677                  elif isinstance(args[0], Sequence):
678                      if all(isinstance(arg, (int, float)) for arg in args[0]):
679                          return 1
680                      else:
681                          # Fall to call add_function_call
682                          pass
683                  else:
684                      # Python constants are scalars
685                      return 0
686  
687              # NOTE: signature should be written to function in the registration process
688              if hasattr(function, "_pt_onnx_signature"):
689                  op_signature = function._pt_onnx_signature  # type: ignore[attr-defined]
690              else:
691                  op_signature = _schemas.OpSignature.from_function(
692                      function,
693                      function.function_ir.domain,
694                      function.name,
695                      opset_version=function.opset.version,
696                  )
697                  function._pt_onnx_signature = op_signature  # type: ignore[attr-defined]
698  
699              named_inputs, named_attrs = _construct_named_inputs_and_attrs(
700                  op_signature, args, kwargs
701              )
702  
703              # TODO(after torchlib migration): Remove traceable function handling
704              # NOTE: We need to call traceable functions after the _construct_named_inputs_and_attrs
705              # call because it will filter out the unexpected kwargs for us.
706              if function.traceable:
707                  # Trace the function call instead of adding the function as a node
708                  # Turn the ir.Attr objects into Python constants first
709                  named_attrs = {
710                      name: attr.value if isinstance(attr, ir.Attr) else attr
711                      for name, attr in named_attrs.items()
712                  }
713  
714                  # Use the type binding to resolve the dtypes of the inputs, and
715                  # convert Python constants to Constant nodes
716                  type_binding = _resolve_parameter_dtypes(op_signature, named_inputs)
717                  try:
718                      # _process_python_sequences is not here because we want to preserve python list
719                      # properties for the function call
720                      converted_named_inputs = _process_python_constants(
721                          op_signature,
722                          named_inputs,
723                          type_binding,
724                          self.constant_farm,
725                          self.opset,
726                      )
727  
728                  except Exception as e:
729                      raise _errors.GraphConstructionError(
730                          f"Error processing Python constants for operator '{op_signature.domain}::{op_signature.name}'. "
731                          f"named_inputs={named_inputs}, named_attrs={named_attrs}, opset={self.opset}, op_signature={op_signature}."
732                      ) from e
733  
734                  return function.function(**converted_named_inputs, **named_attrs)
735  
736              outputs = self._call_op(
737                  op_signature,
738                  named_inputs,
739                  named_attrs,
740                  len(op_signature.outputs),
741              )
742  
743              self.functions[(function.function_ir.domain, function.name, "")] = function
744              if len(outputs) == 1:
745                  return outputs[0]
746              return outputs
747          except Exception as e:
748              try:
749                  source_file = inspect.getsourcefile(function.function)
750                  _, lineno = inspect.getsourcelines(function.function)
751              except Exception:
752                  source_file = lineno = None
753              raise _errors.GraphConstructionError(
754                  f"Error calling function '{function.name}' with args {args} and kwargs {kwargs}."
755                  + f" The function is defined at '{source_file}:{lineno}'."
756                  if source_file
757                  else ""
758              ) from e
