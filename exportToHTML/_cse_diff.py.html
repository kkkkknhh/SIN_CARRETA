7    def _remove_cse_from_derivative(replacements, reduced_expressions):
8        """ 
9        This function is designed to postprocess the output of a common subexpression 
10       elimination (CSE) operation. Specifically, it removes any CSE replacement 
11       symbols from the arguments of ``Derivative`` terms in the expression. This 
12       is necessary to ensure that the forward Jacobian function correctly handles 
13       derivative terms. 
14    
15       Parameters 
16       ========== 
17    
18       replacements : list of (Symbol, expression) pairs 
19           Replacement symbols and relative common subexpressions that have been 
20           replaced during a CSE operation. 
21    
22       reduced_expressions : list of SymPy expressions 
23           The reduced expressions with all the replacements from the 
24           replacements list above. 
25    
26       Returns 
27       ======= 
28    
29       processed_replacements : list of (Symbol, expression) pairs 
30           Processed replacement list, in the same format of the 
31           ``replacements`` input list. 
32    
33       processed_reduced : list of SymPy expressions 
34           Processed reduced list, in the same format of the 
35           ``reduced_expressions`` input list. 
36       """
37   
38       def traverse(node, repl_dict):
39           if isinstance(node, Derivative):
40               return replace_all(node, repl_dict)
41           if not node.args:
42               return node
43           new_args = [traverse(arg, repl_dict) for arg in node.args]
44           return node.func(*new_args)
45   
46       def replace_all(node, repl_dict):
47           result = node
48           while True:
49               free_symbols = result.free_symbols
50               symbols_dict = {k: repl_dict[k] for k in free_symbols if k in repl_dict}
51               if not symbols_dict:
52                   break
53               result = result.xreplace(symbols_dict)
54           return result
55   
56       repl_dict = dict(replacements)
57       processed_replacements = [
58           (rep_sym, traverse(sub_exp, repl_dict))
59           for rep_sym, sub_exp in replacements
60       ]
61       processed_reduced = [
62           red_exp.__class__([traverse(exp, repl_dict) for exp in red_exp])
63           for red_exp in reduced_expressions
64       ]
65   
66       return processed_replacements, processed_reduced

69   def _forward_jacobian_cse(replacements, reduced_expr, wrt):
70       """ 
71       Core function to compute the Jacobian of an input Matrix of expressions 
72       through forward accumulation. Takes directly the output of a CSE operation 
73       (replacements and reduced_expr), and an iterable of variables (wrt) with 
74       respect to which to differentiate the reduced expression and returns the 
75       reduced Jacobian matrix and the ``replacements`` list. 
76    
77       The function also returns a list of precomputed free symbols for each 
78       subexpression, which are useful in the substitution process. 
79    
80       Parameters 
81       ========== 
82    
83       replacements : list of (Symbol, expression) pairs 
84           Replacement symbols and relative common subexpressions that have been 
85           replaced during a CSE operation. 
86    
87       reduced_expr : list of SymPy expressions 
88           The reduced expressions with all the replacements from the 
89           replacements list above. 
90    
91       wrt : iterable 
92           Iterable of expressions with respect to which to compute the 
93           Jacobian matrix. 
94    
95       Returns 
96       ======= 
97    
98       replacements : list of (Symbol, expression) pairs 
99           Replacement symbols and relative common subexpressions that have been 
100          replaced during a CSE operation. Compared to the input replacement list, 
101          the output one doesn't contain replacement symbols inside 
102          ``Derivative``'s arguments. 
103   
104      jacobian : list of SymPy expressions 
105          The list only contains one element, which is the Jacobian matrix with 
106          elements in reduced form (replacement symbols are present). 
107   
108      precomputed_fs: list 
109          List of sets, which store the free symbols present in each sub-expression. 
110          Useful in the substitution process. 
111      """
112  
113      if not isinstance(reduced_expr[0], MatrixBase):
114          raise TypeError("``expr`` must be of matrix type")
115  
116      if not (reduced_expr[0].shape[0] == 1 or reduced_expr[0].shape[1] == 1):
117          raise TypeError("``expr`` must be a row or a column matrix")
118  
119      if not iterable(wrt):
120          raise TypeError("``wrt`` must be an iterable of variables")
121  
122      elif not isinstance(wrt, MatrixBase):
123          wrt = Matrix(wrt)
124  
125      if not (wrt.shape[0] == 1 or wrt.shape[1] == 1):
126          raise TypeError("``wrt`` must be a row or a column matrix")
127  
128      replacements, reduced_expr = _remove_cse_from_derivative(replacements, reduced_expr)
129  
130      if replacements:
131          rep_sym, sub_expr = map(Matrix, zip(*replacements))
132      else:
133          rep_sym, sub_expr = Matrix([]), Matrix([])
134  
135      l_sub, l_wrt, l_red = len(sub_expr), len(wrt), len(reduced_expr[0])
136  
137      f1 = reduced_expr[0].__class__.from_dok(l_red, l_wrt,
138          {
139              (i, j): diff_value
140              for i, r in enumerate(reduced_expr[0])
141              for j, w in enumerate(wrt)
142              if (diff_value := r.diff(w)) != 0
143          },
144      )
145  
146      if not replacements:
147          return [], [f1], []
148  
149      f2 = Matrix.from_dok(l_red, l_sub,
150          {
151              (i, j): diff_value
152              for i, (r, fs) in enumerate([(r, r.free_symbols) for r in reduced_expr[0]])
153              for j, s in enumerate(rep_sym)
154              if s in fs and (diff_value := r.diff(s)) != 0
155          },
156      )
157  
158      rep_sym_set = set(rep_sym)
159      precomputed_fs = [s.free_symbols & rep_sym_set for s in sub_expr ]
160  
161      c_matrix = Matrix.from_dok(1, l_wrt,
162                                 {(0, j): diff_value for j, w in enumerate(wrt)
163                                  if (diff_value := sub_expr[0].diff(w)) != 0})
164  
165      for i in range(1, l_sub):
166  
167          bi_matrix = Matrix.from_dok(1, i,
168                                      {(0, j): diff_value for j in range(i + 1)
169                                       if rep_sym[j] in precomputed_fs[i]
170                                       and (diff_value := sub_expr[i].diff(rep_sym[j])) != 0})
171  
172          ai_matrix = Matrix.from_dok(1, l_wrt,
173                                      {(0, j): diff_value for j, w in enumerate(wrt)
174                                       if (diff_value := sub_expr[i].diff(w)) != 0})
175  
176          if bi_matrix._rep.nnz():
177              ci_matrix = bi_matrix.multiply(c_matrix).add(ai_matrix)
178              c_matrix = Matrix.vstack(c_matrix, ci_matrix)
179          else:
180              c_matrix = Matrix.vstack(c_matrix, ai_matrix)
181  
182      jacobian = f2.multiply(c_matrix).add(f1)
183      jacobian = [reduced_expr[0].__class__(jacobian)]
184  
185      return replacements, jacobian, precomputed_fs

188  def _forward_jacobian_norm_in_cse_out(expr, wrt):
189      """ 
190      Function to compute the Jacobian of an input Matrix of expressions through 
191      forward accumulation. Takes a sympy Matrix of expressions (expr) as input 
192      and an iterable of variables (wrt) with respect to which to compute the 
193      Jacobian matrix. The matrix is returned in reduced form (containing 
194      replacement symbols) along with the ``replacements`` list. 
195   
196      The function also returns a list of precomputed free symbols for each 
197      subexpression, which are useful in the substitution process. 
198   
199      Parameters 
200      ========== 
201   
202      expr : Matrix 
203          The vector to be differentiated. 
204   
205      wrt : iterable 
206          The vector with respect to which to perform the differentiation. 
207          Can be a matrix or an iterable of variables. 
208   
209      Returns 
210      ======= 
211   
212      replacements : list of (Symbol, expression) pairs 
213          Replacement symbols and relative common subexpressions that have been 
214          replaced during a CSE operation. The output replacement list doesn't 
215          contain replacement symbols inside ``Derivative``'s arguments. 
216   
217      jacobian : list of SymPy expressions 
218          The list only contains one element, which is the Jacobian matrix with 
219          elements in reduced form (replacement symbols are present). 
220   
221      precomputed_fs: list 
222          List of sets, which store the free symbols present in each 
223          sub-expression. Useful in the substitution process. 
224      """
225  
226      replacements, reduced_expr = cse(expr)
227      replacements, jacobian, precomputed_fs = _forward_jacobian_cse(replacements, reduced_expr, wrt)
228  
229      return replacements, jacobian, precomputed_fs

232  def _forward_jacobian(expr, wrt):
233      """ 
234      Function to compute the Jacobian of an input Matrix of expressions through 
235      forward accumulation. Takes a sympy Matrix of expressions (expr) as input 
236      and an iterable of variables (wrt) with respect to which to compute the 
237      Jacobian matrix. 
238   
239      Explanation 
240      =========== 
241   
242      Expressions often contain repeated subexpressions. Using a tree structure, 
243      these subexpressions are duplicated and differentiated multiple times, 
244      leading to inefficiency. 
245   
246      Instead, if a data structure called a directed acyclic graph (DAG) is used 
247      then each of these repeated subexpressions will only exist a single time. 
248      This function uses a combination of representing the expression as a DAG and 
249      a forward accumulation algorithm (repeated application of the chain rule 
250      symbolically) to more efficiently calculate the Jacobian matrix of a target 
251      expression ``expr`` with respect to an expression or set of expressions 
252      ``wrt``. 
253   
254      Note that this function is intended to improve performance when 
255      differentiating large expressions that contain many common subexpressions. 
256      For small and simple expressions it is likely less performant than using 
257      SymPy's standard differentiation functions and methods. 
258   
259      Parameters 
260      ========== 
261   
262      expr : Matrix 
263          The vector to be differentiated. 
264   
265      wrt : iterable 
266          The vector with respect to which to do the differentiation. 
267          Can be a matrix or an iterable of variables. 
268   
269      See Also 
270      ======== 
271   
272      Direct Acyclic Graph : https://en.wikipedia.org/wiki/Directed_acyclic_graph 
273      """
274  
275      replacements, reduced_expr = cse(expr)
276  
277      if replacements:
278          rep_sym, _ = map(Matrix, zip(*replacements))
279      else:
280          rep_sym = Matrix([])
281  
282      replacements, jacobian, precomputed_fs = _forward_jacobian_cse(replacements, reduced_expr, wrt)
283  
284      if not replacements: return jacobian[0]
285  
286      sub_rep = dict(replacements)
287      for i, ik in enumerate(precomputed_fs):
288          sub_dict = {j: sub_rep[j] for j in ik}
289          sub_rep[rep_sym[i]] = sub_rep[rep_sym[i]].xreplace(sub_dict)
290  
291      return jacobian[0].xreplace(sub_rep)
