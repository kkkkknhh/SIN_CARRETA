61   class MemoizeJac:
62       """Decorator that caches the return values of a function returning ``(fun, grad)`` 
63       each time it is called."""
64   
65       def __init__(self, fun):
66           self.fun = fun
67           self.jac = None
68           self._value = None
69           self.x = None
70   
71       def _compute_if_needed(self, x, *args):
72           if not np.all(x == self.x) or self._value is None or self.jac is None:
73               self.x = np.asarray(x).copy()
74               fg = self.fun(x, *args)
75               self.jac = fg[1]
76               self._value = fg[0]
77   
78       def __call__(self, x, *args):
79           """ returns the function value """
80           self._compute_if_needed(x, *args)
81           return self._value
82   
83       def derivative(self, x, *args):
84           self._compute_if_needed(x, *args)
85           return self.jac

88   def _wrap_callback(callback, method=None):
89       """Wrap a user-provided callback so that attributes can be attached."""
90       if callback is None or method in {'tnc', 'slsqp', 'cobyla', 'cobyqa'}:
91           return callback  # don't wrap
92   
93       sig = inspect.signature(callback)
94   
95       if set(sig.parameters) == {'intermediate_result'}:
96           def wrapped_callback(res):
97               return callback(intermediate_result=res)
98       elif method == 'trust-constr':
99           def wrapped_callback(res):
100              return callback(np.copy(res.x), res)
101      elif method == 'differential_evolution':
102          def wrapped_callback(res):
103              return callback(np.copy(res.x), res.convergence)
104      else:
105          def wrapped_callback(res):
106              return callback(np.copy(res.x))
107  
108      wrapped_callback.stop_iteration = False
109      return wrapped_callback

160  def _check_positive_definite(Hk):
161      def is_pos_def(A):
162          if issymmetric(A):
163              try:
164                  cholesky(A)
165                  return True
166              except LinAlgError:
167                  return False
168          else:
169              return False
170      if Hk is not None:
171          if not is_pos_def(Hk):
172              raise ValueError("'hess_inv0' matrix isn't positive definite.")

194  def vecnorm(x, ord=2):
195      if ord == np.inf:
196          return np.amax(np.abs(x))
197      elif ord == -np.inf:
198          return np.amin(np.abs(x))
199      else:
200          return np.sum(np.abs(x)**ord, axis=0)**(1.0 / ord)

203  def _prepare_scalar_function(fun, x0, jac=None, args=(), bounds=None,
204                               epsilon=None, finite_diff_rel_step=None,
205                               hess=None):
206      """ 
207      Creates a ScalarFunction object for use with scalar minimizers 
208      (BFGS/LBFGSB/SLSQP/TNC/CG/etc). 
209   
210      Parameters 
211      ---------- 
212      fun : callable 
213          The objective function to be minimized. 
214   
215              ``fun(x, *args) -> float`` 
216   
217          where ``x`` is an 1-D array with shape (n,) and ``args`` 
218          is a tuple of the fixed parameters needed to completely 
219          specify the function. 
220      x0 : ndarray, shape (n,) 
221          Initial guess. Array of real elements of size (n,), 
222          where 'n' is the number of independent variables. 
223      jac : {callable,  '2-point', '3-point', 'cs', None}, optional 
224          Method for computing the gradient vector. If it is a callable, it 
225          should be a function that returns the gradient vector: 
226   
227              ``jac(x, *args) -> array_like, shape (n,)`` 
228   
229          If one of `{'2-point', '3-point', 'cs'}` is selected then the gradient 
230          is calculated with a relative step for finite differences. If `None`, 
231          then two-point finite differences with an absolute step is used. 
232      args : tuple, optional 
233          Extra arguments passed to the objective function and its 
234          derivatives (`fun`, `jac` functions). 
235      bounds : sequence, optional 
236          Bounds on variables. 'new-style' bounds are required. 
237      eps : float or ndarray 
238          If ``jac is None`` the absolute step size used for numerical 
239          approximation of the jacobian via forward differences. 
240      finite_diff_rel_step : None or array_like, optional 
241          If ``jac in ['2-point', '3-point', 'cs']`` the relative step size to 
242          use for numerical approximation of the jacobian. The absolute step 
243          size is computed as ``h = rel_step * sign(x0) * max(1, abs(x0))``, 
244          possibly adjusted to fit into the bounds. For ``jac='3-point'`` 
245          the sign of `h` is ignored. If None (default) then step is selected 
246          automatically. 
247      hess : {callable,  '2-point', '3-point', 'cs', None} 
248          Computes the Hessian matrix. If it is callable, it should return the 
249          Hessian matrix: 
250   
251              ``hess(x, *args) -> {LinearOperator, spmatrix, array}, (n, n)`` 
252   
253          Alternatively, the keywords {'2-point', '3-point', 'cs'} select a 
254          finite difference scheme for numerical estimation. 
255          Whenever the gradient is estimated via finite-differences, the Hessian 
256          cannot be estimated with options {'2-point', '3-point', 'cs'} and needs 
257          to be estimated using one of the quasi-Newton strategies. 
258   
259      Returns 
260      ------- 
261      sf : ScalarFunction 
262      """
263      if callable(jac):
264          grad = jac
265      elif jac in FD_METHODS:
266          # epsilon is set to None so that ScalarFunction is made to use
267          # rel_step
268          epsilon = None
269          grad = jac
270      else:
271          # default (jac is None) is to do 2-point finite differences with
272          # absolute step size. ScalarFunction has to be provided an
273          # epsilon value that is not None to use absolute steps. This is
274          # normally the case from most _minimize* methods.
275          grad = '2-point'
276          epsilon = epsilon
277  
278      if hess is None:
279          # ScalarFunction requires something for hess, so we give a dummy
280          # implementation here if nothing is provided, return a value of None
281          # so that downstream minimisers halt. The results of `fun.hess`
282          # should not be used.
283          def hess(x, *args):
284              return None
285  
286      if bounds is None:
287          bounds = (-np.inf, np.inf)
288  
289      # ScalarFunction caches. Reuse of fun(x) during grad
290      # calculation reduces overall function evaluations.
291      sf = ScalarFunction(fun, x0, args, grad, hess,
292                          finite_diff_rel_step, bounds, epsilon=epsilon)
293  
294      return sf

321  def rosen(x):
322      """ 
323      The Rosenbrock function. 
324   
325      The function computed is:: 
326   
327          sum(100.0*(x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0) 
328   
329      Parameters 
330      ---------- 
331      x : array_like 
332          1-D array of points at which the Rosenbrock function is to be computed. 
333   
334      Returns 
335      ------- 
336      f : float 
337          The value of the Rosenbrock function. 
338   
339      See Also 
340      -------- 
341      rosen_der, rosen_hess, rosen_hess_prod 
342   
343      Examples 
344      -------- 
345      >>> import numpy as np 
346      >>> from scipy.optimize import rosen 
347      >>> X = 0.1 * np.arange(10) 
348      >>> rosen(X) 
349      76.56 
350   
351      For higher-dimensional input ``rosen`` broadcasts. 
352      In the following example, we use this to plot a 2D landscape. 
353      Note that ``rosen_hess`` does not broadcast in this manner. 
354   
355      >>> import matplotlib.pyplot as plt 
356      >>> from mpl_toolkits.mplot3d import Axes3D 
357      >>> x = np.linspace(-1, 1, 50) 
358      >>> X, Y = np.meshgrid(x, x) 
359      >>> ax = plt.subplot(111, projection='3d') 
360      >>> ax.plot_surface(X, Y, rosen([X, Y])) 
361      >>> plt.show() 
362      """
363      xp = array_namespace(x)
364      x = xp.asarray(x)
365      if xp.isdtype(x.dtype, 'integral'):
366          x = xp.astype(x, xp.asarray(1.).dtype)
367      r = xp.sum(100.0 * (x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0,
368                 axis=0, dtype=x.dtype)
369      return r

372  def rosen_der(x):
373      """ 
374      The derivative (i.e. gradient) of the Rosenbrock function. 
375   
376      Parameters 
377      ---------- 
378      x : array_like 
379          1-D array of points at which the derivative is to be computed. 
380   
381      Returns 
382      ------- 
383      rosen_der : (N,) ndarray 
384          The gradient of the Rosenbrock function at `x`. 
385   
386      See Also 
387      -------- 
388      rosen, rosen_hess, rosen_hess_prod 
389   
390      Examples 
391      -------- 
392      >>> import numpy as np 
393      >>> from scipy.optimize import rosen_der 
394      >>> X = 0.1 * np.arange(9) 
395      >>> rosen_der(X) 
396      array([ -2. ,  10.6,  15.6,  13.4,   6.4,  -3. , -12.4, -19.4,  62. ]) 
397   
398      """
399      xp = array_namespace(x)
400      x = xp.asarray(x)
401      if xp.isdtype(x.dtype, 'integral'):
402          x = xp.astype(x, xp.asarray(1.).dtype)
403      xm = x[1:-1]
404      xm_m1 = x[:-2]
405      xm_p1 = x[2:]
406      der = xp.zeros_like(x)
407      der[1:-1] = (200 * (xm - xm_m1**2) -
408                   400 * (xm_p1 - xm**2) * xm - 2 * (1 - xm))
409      der[0] = -400 * x[0] * (x[1] - x[0]**2) - 2 * (1 - x[0])
410      der[-1] = 200 * (x[-1] - x[-2]**2)
411      return der

414  def rosen_hess(x):
415      """ 
416      The Hessian matrix of the Rosenbrock function. 
417   
418      Parameters 
419      ---------- 
420      x : array_like 
421          1-D array of points at which the Hessian matrix is to be computed. 
422   
423      Returns 
424      ------- 
425      rosen_hess : ndarray 
426          The Hessian matrix of the Rosenbrock function at `x`. 
427   
428      See Also 
429      -------- 
430      rosen, rosen_der, rosen_hess_prod 
431   
432      Examples 
433      -------- 
434      >>> import numpy as np 
435      >>> from scipy.optimize import rosen_hess 
436      >>> X = 0.1 * np.arange(4) 
437      >>> rosen_hess(X) 
438      array([[-38.,   0.,   0.,   0.], 
439             [  0., 134., -40.,   0.], 
440             [  0., -40., 130., -80.], 
441             [  0.,   0., -80., 200.]]) 
442   
443      """
444      xp = array_namespace(x)
445      x = xpx.atleast_nd(x, ndim=1, xp=xp)
446      if xp.isdtype(x.dtype, 'integral'):
447          x = xp.astype(x, xp.asarray(1.).dtype)
448      H = (xpx.create_diagonal(-400 * x[:-1], offset=1, xp=xp) 
449           - xpx.create_diagonal(400 * x[:-1], offset=-1, xp=xp))
450      diagonal = xp.zeros(x.shape[0], dtype=x.dtype)
451      diagonal[0] = 1200 * x[0]**2 - 400 * x[1] + 2
452      diagonal[-1] = 200
453      diagonal[1:-1] = 202 + 1200 * x[1:-1]**2 - 400 * x[2:]
454      return H + xpx.create_diagonal(diagonal, xp=xp)

457  def rosen_hess_prod(x, p):
458      """ 
459      Product of the Hessian matrix of the Rosenbrock function with a vector. 
460   
461      Parameters 
462      ---------- 
463      x : array_like 
464          1-D array of points at which the Hessian matrix is to be computed. 
465      p : array_like 
466          1-D array, the vector to be multiplied by the Hessian matrix. 
467   
468      Returns 
469      ------- 
470      rosen_hess_prod : ndarray 
471          The Hessian matrix of the Rosenbrock function at `x` multiplied 
472          by the vector `p`. 
473   
474      See Also 
475      -------- 
476      rosen, rosen_der, rosen_hess 
477   
478      Examples 
479      -------- 
480      >>> import numpy as np 
481      >>> from scipy.optimize import rosen_hess_prod 
482      >>> X = 0.1 * np.arange(9) 
483      >>> p = 0.5 * np.arange(9) 
484      >>> rosen_hess_prod(X, p) 
485      array([  -0.,   27.,  -10.,  -95., -192., -265., -278., -195., -180.]) 
486   
487      """
488      xp = array_namespace(x, p)
489      x = xpx.atleast_nd(x, ndim=1, xp=xp)
490      if xp.isdtype(x.dtype, 'integral'):
491          x = xp.astype(x, xp.asarray(1.).dtype)
492      p = xp.asarray(p, dtype=x.dtype)
493      Hp = xp.zeros(x.shape[0], dtype=x.dtype)
494      Hp[0] = (1200 * x[0]**2 - 400 * x[1] + 2) * p[0] - 400 * x[0] * p[1]
495      Hp[1:-1] = (-400 * x[:-2] * p[:-2] +
496                  (202 + 1200 * x[1:-1]**2 - 400 * x[2:]) * p[1:-1] -
497                  400 * x[1:-1] * p[2:])
498      Hp[-1] = -400 * x[-2] * p[-2] + 200*p[-1]
499      return Hp

502  def _wrap_scalar_function(function, args):
503      # wraps a minimizer function to count number of evaluations
504      # and to easily provide an args kwd.
505      ncalls = [0]
506      if function is None:
507          return ncalls, None
508  
509      def function_wrapper(x, *wrapper_args):
510          ncalls[0] += 1
511          # A copy of x is sent to the user function (gh13740)
512          fx = function(np.copy(x), *(wrapper_args + args))
513          # Ideally, we'd like to a have a true scalar returned from f(x). For
514          # backwards-compatibility, also allow np.array([1.3]), np.array([[1.3]]) etc.
515          if not np.isscalar(fx):
516              try:
517                  fx = np.asarray(fx).item()
518              except (TypeError, ValueError) as e:
519                  raise ValueError("The user-provided objective function "
520                                   "must return a scalar value.") from e
521          return fx
522  
523      return ncalls, function_wrapper

557  def fmin(func, x0, args=(), xtol=1e-4, ftol=1e-4, maxiter=None, maxfun=None,
558           full_output=0, disp=1, retall=0, callback=None, initial_simplex=None):
559      """ 
560      Minimize a function using the downhill simplex algorithm. 
561   
562      This algorithm only uses function values, not derivatives or second 
563      derivatives. 
564   
565      Parameters 
566      ---------- 
567      func : callable func(x,*args) 
568          The objective function to be minimized. 
569      x0 : ndarray 
570          Initial guess. 
571      args : tuple, optional 
572          Extra arguments passed to func, i.e., ``f(x,*args)``. 
573      xtol : float, optional 
574          Absolute error in xopt between iterations that is acceptable for 
575          convergence. 
576      ftol : number, optional 
577          Absolute error in func(xopt) between iterations that is acceptable for 
578          convergence. 
579      maxiter : int, optional 
580          Maximum number of iterations to perform. 
581      maxfun : number, optional 
582          Maximum number of function evaluations to make. 
583      full_output : bool, optional 
584          Set to True if fopt and warnflag outputs are desired. 
585      disp : bool, optional 
586          Set to True to print convergence messages. 
587      retall : bool, optional 
588          Set to True to return list of solutions at each iteration. 
589      callback : callable, optional 
590          Called after each iteration, as callback(xk), where xk is the 
591          current parameter vector. 
592      initial_simplex : array_like of shape (N + 1, N), optional 
593          Initial simplex. If given, overrides `x0`. 
594          ``initial_simplex[j,:]`` should contain the coordinates of 
595          the jth vertex of the ``N+1`` vertices in the simplex, where 
596          ``N`` is the dimension. 
597   
598      Returns 
599      ------- 
600      xopt : ndarray 
601          Parameter that minimizes function. 
602      fopt : float 
603          Value of function at minimum: ``fopt = func(xopt)``. 
604      iter : int 
605          Number of iterations performed. 
606      funcalls : int 
607          Number of function calls made. 
608      warnflag : int 
609          1 : Maximum number of function evaluations made. 
610          2 : Maximum number of iterations reached. 
611      allvecs : list 
612          Solution at each iteration. 
613   
614      See also 
615      -------- 
616      minimize: Interface to minimization algorithms for multivariate 
617          functions. See the 'Nelder-Mead' `method` in particular. 
618   
619      Notes 
620      ----- 
621      Uses a Nelder-Mead simplex algorithm to find the minimum of function of 
622      one or more variables. 
623   
624      This algorithm has a long history of successful use in applications. 
625      But it will usually be slower than an algorithm that uses first or 
626      second derivative information. In practice, it can have poor 
627      performance in high-dimensional problems and is not robust to 
628      minimizing complicated functions. Additionally, there currently is no 
629      complete theory describing when the algorithm will successfully 
630      converge to the minimum, or how fast it will if it does. Both the ftol and 
631      xtol criteria must be met for convergence. 
632   
633      Examples 
634      -------- 
635      >>> def f(x): 
636      ...     return x**2 
637   
638      >>> from scipy import optimize 
639   
640      >>> minimum = optimize.fmin(f, 1) 
641      Optimization terminated successfully. 
642               Current function value: 0.000000 
643               Iterations: 17 
644               Function evaluations: 34 
645      >>> minimum[0] 
646      -8.8817841970012523e-16 
647   
648      References 
649      ---------- 
650      .. [1] Nelder, J.A. and Mead, R. (1965), "A simplex method for function 
651             minimization", The Computer Journal, 7, pp. 308-313 
652   
653      .. [2] Wright, M.H. (1996), "Direct Search Methods: Once Scorned, Now 
654             Respectable", in Numerical Analysis 1995, Proceedings of the

682  def _minimize_neldermead(func, x0, args=(), callback=None,
683                           maxiter=None, maxfev=None, disp=False,
684                           return_all=False, initial_simplex=None,
685                           xatol=1e-4, fatol=1e-4, adaptive=False, bounds=None,
686                           **unknown_options):
687      """ 
688      Minimization of scalar function of one or more variables using the 
689      Nelder-Mead algorithm. 
690   
691      Options 
692      ------- 
693      disp : bool 
694          Set to True to print convergence messages. 
695      maxiter, maxfev : int 
696          Maximum allowed number of iterations and function evaluations. 
697          Will default to ``N*200``, where ``N`` is the number of 
698          variables, if neither `maxiter` or `maxfev` is set. If both 
699          `maxiter` and `maxfev` are set, minimization will stop at the 
700          first reached. 
701      return_all : bool, optional 
702          Set to True to return a list of the best solution at each of the 
703          iterations. 
704      initial_simplex : array_like of shape (N + 1, N) 
705          Initial simplex. If given, overrides `x0`. 
706          ``initial_simplex[j,:]`` should contain the coordinates of 
707          the jth vertex of the ``N+1`` vertices in the simplex, where 
708          ``N`` is the dimension. 
709      xatol : float, optional 
710          Absolute error in xopt between iterations that is acceptable for 
711          convergence. 
712      fatol : number, optional 
713          Absolute error in func(xopt) between iterations that is acceptable for 
714          convergence. 
715      adaptive : bool, optional 
716          Adapt algorithm parameters to dimensionality of problem. Useful for 
717          high-dimensional minimization [1]_. 
718      bounds : sequence or `Bounds`, optional 
719          Bounds on variables. There are two ways to specify the bounds: 
720   
721          1. Instance of `Bounds` class. 
722          2. Sequence of ``(min, max)`` pairs for each element in `x`. None 
723             is used to specify no bound. 
724   
725          Note that this just clips all vertices in simplex based on 
726          the bounds. 
727   
728      References 
729      ---------- 
730      .. [1] Gao, F. and Han, L. 
731         Implementing the Nelder-Mead simplex algorithm with adaptive 
732         parameters. 2012. Computational Optimization and Applications. 
733         51:1, pp. 259-277 
734   
735      """
736      _check_unknown_options(unknown_options)
737      maxfun = maxfev
738      retall = return_all
739  
740      x0 = np.atleast_1d(x0).flatten()
741      dtype = x0.dtype if np.issubdtype(x0.dtype, np.inexact) else np.float64
742      x0 = np.asarray(x0, dtype=dtype)
743  
744      if adaptive:
745          dim = float(len(x0))
746          rho = 1
747          chi = 1 + 2/dim
748          psi = 0.75 - 1/(2*dim)
749          sigma = 1 - 1/dim
750      else:
751          rho = 1
752          chi = 2
753          psi = 0.5
754          sigma = 0.5
755  
756      nonzdelt = 0.05
757      zdelt = 0.00025
758  
759      if bounds is not None:
760          lower_bound, upper_bound = bounds.lb, bounds.ub
761          # check bounds
762          if (lower_bound > upper_bound).any():
763              raise ValueError("Nelder Mead - one of the lower bounds "
764                               "is greater than an upper bound.")
765          if np.any(lower_bound > x0) or np.any(x0 > upper_bound):
766              warnings.warn("Initial guess is not within the specified bounds",
767                            OptimizeWarning, stacklevel=3)
768  
769      if bounds is not None:
770          x0 = np.clip(x0, lower_bound, upper_bound)
771  
772      if initial_simplex is None:
773          N = len(x0)
774  
775          sim = np.empty((N + 1, N), dtype=x0.dtype)
776          sim[0] = x0
777          for k in range(N):
778              y = np.array(x0, copy=True)
779              if y[k] != 0:
780                  y[k] = (1 + nonzdelt)*y[k]
781              else:
782                  y[k] = zdelt
783              sim[k + 1] = y
784      else:
785          sim = np.atleast_2d(initial_simplex).copy()
786          dtype = sim.dtype if np.issubdtype(sim.dtype, np.inexact) else np.float64
787          sim = np.asarray(sim, dtype=dtype)
788          if sim.ndim != 2 or sim.shape[0] != sim.shape[1] + 1:
789              raise ValueError("`initial_simplex` should be an array of shape (N+1,N)")
790          if len(x0) != sim.shape[1]:
791              raise ValueError("Size of `initial_simplex` is not consistent with `x0`")
792          N = sim.shape[1]
793  
794      if retall:
795          allvecs = [sim[0]]
796  
797      # If neither are set, then set both to default
798      if maxiter is None and maxfun is None:
799          maxiter = N * 200
800          maxfun = N * 200
801      elif maxiter is None:
802          # Convert remaining Nones, to np.inf, unless the other is np.inf, in
803          # which case use the default to avoid unbounded iteration
804          if maxfun == np.inf:
805              maxiter = N * 200
806          else:
807              maxiter = np.inf
808      elif maxfun is None:
809          if maxiter == np.inf:
810              maxfun = N * 200
811          else:
812              maxfun = np.inf
813  
814      if bounds is not None:
815          # The default simplex construction may make all entries (for a given
816          # parameter) greater than an upper bound if x0 is very close to the
817          # upper bound. If one simply clips the simplex to the bounds this could
818          # make the simplex entries degenerate. If that occurs reflect into the
819          # interior.
820          msk = sim > upper_bound
821          # reflect into the interior
822          sim = np.where(msk, 2*upper_bound - sim, sim)
823          # but make sure the reflection is no less than the lower_bound
824          sim = np.clip(sim, lower_bound, upper_bound)
825  
826      one2np1 = list(range(1, N + 1))
827      fsim = np.full((N + 1,), np.inf, dtype=float)
828  
829      fcalls, func = _wrap_scalar_function_maxfun_validation(func, args, maxfun)
830  
831      try:
832          for k in range(N + 1):
833              fsim[k] = func(sim[k])
834      except _MaxFuncCallError:
835          pass
836      finally:
837          ind = np.argsort(fsim)
838          sim = np.take(sim, ind, 0)
839          fsim = np.take(fsim, ind, 0)
840  
841      ind = np.argsort(fsim)
842      fsim = np.take(fsim, ind, 0)
843      # sort so sim[0,:] has the lowest function value
844      sim = np.take(sim, ind, 0)
845  
846      iterations = 1
847  
848      while (fcalls[0] < maxfun and iterations < maxiter):
849          try:
850              if (np.max(np.ravel(np.abs(sim[1:] - sim[0]))) <= xatol and
851                      np.max(np.abs(fsim[0] - fsim[1:])) <= fatol):
852                  break
853  
854              xbar = np.add.reduce(sim[:-1], 0) / N
855              xr = (1 + rho) * xbar - rho * sim[-1]
856              if bounds is not None:
857                  xr = np.clip(xr, lower_bound, upper_bound)
858              fxr = func(xr)
859              doshrink = 0
860  
861              if fxr < fsim[0]:
862                  xe = (1 + rho * chi) * xbar - rho * chi * sim[-1]
863                  if bounds is not None:
864                      xe = np.clip(xe, lower_bound, upper_bound)
865                  fxe = func(xe)
866  
867                  if fxe < fxr:
868                      sim[-1] = xe
869                      fsim[-1] = fxe
870                  else:
871                      sim[-1] = xr
872                      fsim[-1] = fxr
873              else:  # fsim[0] <= fxr
874                  if fxr < fsim[-2]:
875                      sim[-1] = xr
876                      fsim[-1] = fxr
877                  else:  # fxr >= fsim[-2]
878                      # Perform contraction
879                      if fxr < fsim[-1]:
880                          xc = (1 + psi * rho) * xbar - psi * rho * sim[-1]
881                          if bounds is not None:
882                              xc = np.clip(xc, lower_bound, upper_bound)
883                          fxc = func(xc)
884  
885                          if fxc <= fxr:
886                              sim[-1] = xc
887                              fsim[-1] = fxc
888                          else:
889                              doshrink = 1
890                      else:
891                          # Perform an inside contraction
892                          xcc = (1 - psi) * xbar + psi * sim[-1]
893                          if bounds is not None:
894                              xcc = np.clip(xcc, lower_bound, upper_bound)
895                          fxcc = func(xcc)
896  
897                          if fxcc < fsim[-1]:
898                              sim[-1] = xcc
899                              fsim[-1] = fxcc
900                          else:
901                              doshrink = 1
902  
903                      if doshrink:
904                          for j in one2np1:
905                              sim[j] = sim[0] + sigma * (sim[j] - sim[0])
906                              if bounds is not None:
907                                  sim[j] = np.clip(
908                                      sim[j], lower_bound, upper_bound)
909                              fsim[j] = func(sim[j])
910              iterations += 1
911          except _MaxFuncCallError:
912              pass
913          finally:
914              ind = np.argsort(fsim)
915              sim = np.take(sim, ind, 0)
916              fsim = np.take(fsim, ind, 0)
917              if retall:
918                  allvecs.append(sim[0])
919              intermediate_result = OptimizeResult(x=sim[0], fun=fsim[0])
920              if _call_callback_maybe_halt(callback, intermediate_result):
921                  break
922  
923      x = sim[0]
924      fval = np.min(fsim)
925      warnflag = 0
926  
927      if fcalls[0] >= maxfun:
928          warnflag = 1
929          msg = _status_message['maxfev']
930          if disp:
931              warnings.warn(msg, RuntimeWarning, stacklevel=3)
932      elif iterations >= maxiter:
933          warnflag = 2
934          msg = _status_message['maxiter']
935          if disp:
936              warnings.warn(msg, RuntimeWarning, stacklevel=3)
937      else:
938          msg = _status_message['success']
939          if disp:
940              print(msg)
941              print(f"         Current function value: {fval:f}")
942              print("         Iterations: %d" % iterations)
943              print("         Function evaluations: %d" % fcalls[0])
944  
945      result = OptimizeResult(fun=fval, nit=iterations, nfev=fcalls[0],
946                              status=warnflag, success=(warnflag == 0),
947                              message=msg, x=x, final_simplex=(sim, fsim))
948      if retall:
949          result['allvecs'] = allvecs
950      return result

953  def approx_fprime(xk, f, epsilon=_epsilon, *args):
954      """Finite difference approximation of the derivatives of a 
955      scalar or vector-valued function. 
956   
957      If a function maps from :math:`R^n` to :math:`R^m`, its derivatives form 
958      an m-by-n matrix 
959      called the Jacobian, where an element :math:`(i, j)` is a partial 
960      derivative of f[i] with respect to ``xk[j]``. 
961   
962      Parameters 
963      ---------- 
964      xk : array_like 
965          The coordinate vector at which to determine the gradient of `f`. 
966      f : callable 
967          Function of which to estimate the derivatives of. Has the signature 
968          ``f(xk, *args)`` where `xk` is the argument in the form of a 1-D array 
969          and `args` is a tuple of any additional fixed parameters needed to 
970          completely specify the function. The argument `xk` passed to this 
971          function is an ndarray of shape (n,) (never a scalar even if n=1). 
972          It must return a 1-D array_like of shape (m,) or a scalar. 
973   
974          Suppose the callable has signature ``f0(x, *my_args, **my_kwargs)``, where 
975          ``my_args`` and ``my_kwargs`` are required positional and keyword arguments. 
976          Rather than passing ``f0`` as the callable, wrap it to accept 
977          only ``x``; e.g., pass ``fun=lambda x: f0(x, *my_args, **my_kwargs)`` as the 
978          callable, where ``my_args`` (tuple) and ``my_kwargs`` (dict) have been 
979          gathered before invoking this function. 
980   
981          .. versionchanged:: 1.9.0 
982              `f` is now able to return a 1-D array-like, with the :math:`(m, n)` 
983              Jacobian being estimated. 
984   
985      epsilon : {float, array_like}, optional 
986          Increment to `xk` to use for determining the function gradient. 
987          If a scalar, uses the same finite difference delta for all partial 
988          derivatives. If an array, should contain one value per element of 
989          `xk`. Defaults to ``sqrt(np.finfo(float).eps)``, which is approximately 
990          1.49e-08. 
991      \\*args : args, optional 
992          Any other arguments that are to be passed to `f`. 
993   
994      Returns 
995      ------- 
996      jac : ndarray 
997          The partial derivatives of `f` to `xk`. 
998   
999      See Also 
1000     -------- 
1001     check_grad : Check correctness of gradient function against approx_fprime. 
1002  
1003     Notes 
1004     ----- 
1005     The function gradient is determined by the forward finite difference 
1006     formula:: 
1007  
1008                  f(xk[i] + epsilon[i]) - f(xk[i]) 
1009         f'[i] = --------------------------------- 
1010                             epsilon[i] 
1011  
1012     Examples 
1013     -------- 
1014     >>> import numpy as np 
1015     >>> from scipy import optimize 
1016     >>> def func(x, c0, c1): 
1017     ...     "Coordinate vector `x` should be an array of size two." 
1018     ...     return c0 * x[0]**2 + c1*x[1]**2 
1019  
1020     >>> x = np.ones(2) 
1021     >>> c0, c1 = (1, 200) 
1022     >>> eps = np.sqrt(np.finfo(float).eps) 
1023     >>> optimize.approx_fprime(x, func, [eps, np.sqrt(200) * eps], c0, c1) 
1024     array([   2.        ,  400.00004208]) 
1025  
1026     """
1027     xk = np.asarray(xk, float)
1028     f0 = f(xk, *args)
1029 
1030     return approx_derivative(f, xk, method='2-point', abs_step=epsilon,
1031                              args=args, f0=f0)

1035 def check_grad(func, grad, x0, *args, epsilon=_epsilon,
1036                 direction='all', rng=None):
1037     r"""Check the correctness of a gradient function by comparing it against a 
1038     (forward) finite-difference approximation of the gradient. 
1039  
1040     Parameters 
1041     ---------- 
1042     func : callable ``func(x0, *args)`` 
1043         Function whose derivative is to be checked. 
1044     grad : callable ``grad(x0, *args)`` 
1045         Jacobian of `func`. 
1046     x0 : ndarray 
1047         Points to check `grad` against forward difference approximation of grad 
1048         using `func`. 
1049     args : \\*args, optional 
1050         Extra arguments passed to `func` and `grad`. 
1051     epsilon : float, optional 
1052         Step size used for the finite difference approximation. It defaults to 
1053         ``sqrt(np.finfo(float).eps)``, which is approximately 1.49e-08. 
1054     direction : str, optional 
1055         If set to ``'random'``, then gradients along a random vector 
1056         are used to check `grad` against forward difference approximation 
1057         using `func`. By default it is ``'all'``, in which case, all 
1058         the one hot direction vectors are considered to check `grad`. 
1059         If `func` is a vector valued function then only ``'all'`` can be used. 
1060     rng : `numpy.random.Generator`, optional 
1061         Pseudorandom number generator state. When `rng` is None, a new 
1062         `numpy.random.Generator` is created using entropy from the 
1063         operating system. Types other than `numpy.random.Generator` are 
1064         passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
1065  
1066         The random numbers generated affect the random vector along which gradients 
1067         are computed to check ``grad``. Note that `rng` is only used when `direction` 
1068         argument is set to `'random'`. 
1069  
1070     Returns 
1071     ------- 
1072     err : float 
1073         The square root of the sum of squares (i.e., the 2-norm) of the 
1074         difference between ``grad(x0, *args)`` and the finite difference 
1075         approximation of `grad` using func at the points `x0`. 
1076  
1077     See Also 
1078     -------- 
1079     approx_fprime 
1080  
1081     Examples 
1082     -------- 
1083     >>> import numpy as np 
1084     >>> def func(x): 
1085     ...     return x[0]**2 - 0.5 * x[1]**3 
1086     >>> def grad(x): 
1087     ...     return [2 * x[0], -1.5 * x[1]**2] 
1088     >>> from scipy.optimize import check_grad 
1089     >>> check_grad(func, grad, [1.5, -1.5]) 
1090     2.9802322387695312e-08  # may vary 
1091     >>> rng = np.random.default_rng() 
1092     >>> check_grad(func, grad, [1.5, -1.5], 
1093     ...             direction='random', seed=rng) 
1094     2.9802322387695312e-08 
1095  
1096     """
1097     step = epsilon
1098     x0 = np.asarray(x0)
1099 
1100     def g(w, func, x0, v, *args):
1101         return func(x0 + w*v, *args)
1102 
1103     if direction == 'random':
1104         _grad = np.asanyarray(grad(x0, *args))
1105         if _grad.ndim > 1:
1106             raise ValueError("'random' can only be used with scalar valued"
1107                              " func")
1108         rng_gen = check_random_state(rng)
1109         v = rng_gen.standard_normal(size=(x0.shape))
1110         _args = (func, x0, v) + args
1111         _func = g
1112         vars = np.zeros((1,))
1113         analytical_grad = np.dot(_grad, v)
1114     elif direction == 'all':
1115         _args = args
1116         _func = func
1117         vars = x0
1118         analytical_grad = grad(x0, *args)
1119     else:
1120         raise ValueError(f"{direction} is not a valid string for "
1121                          "``direction`` argument")
1122 
1123     return np.sqrt(np.sum(np.abs(
1124         (analytical_grad - approx_fprime(vars, _func, step, *_args))**2
1125     )))

1184 def fmin_bfgs(f, x0, fprime=None, args=(), gtol=1e-5, norm=np.inf,
1185               epsilon=_epsilon, maxiter=None, full_output=0, disp=1,
1186               retall=0, callback=None, xrtol=0, c1=1e-4, c2=0.9,
1187               hess_inv0=None):
1188     """ 
1189     Minimize a function using the BFGS algorithm. 
1190  
1191     Parameters 
1192     ---------- 
1193     f : callable ``f(x,*args)`` 
1194         Objective function to be minimized. 
1195     x0 : ndarray 
1196         Initial guess, shape (n,) 
1197     fprime : callable ``f'(x,*args)``, optional 
1198         Gradient of f. 
1199     args : tuple, optional 
1200         Extra arguments passed to f and fprime. 
1201     gtol : float, optional 
1202         Terminate successfully if gradient norm is less than `gtol` 
1203     norm : float, optional 
1204         Order of norm (Inf is max, -Inf is min) 
1205     epsilon : int or ndarray, optional 
1206         If `fprime` is approximated, use this value for the step size. 
1207     callback : callable, optional 
1208         An optional user-supplied function to call after each 
1209         iteration. Called as ``callback(xk)``, where ``xk`` is the 
1210         current parameter vector. 
1211     maxiter : int, optional 
1212         Maximum number of iterations to perform. 
1213     full_output : bool, optional 
1214         If True, return ``fopt``, ``func_calls``, ``grad_calls``, and 
1215         ``warnflag`` in addition to ``xopt``. 
1216     disp : bool, optional 
1217         Print convergence message if True. 
1218     retall : bool, optional 
1219         Return a list of results at each iteration if True. 
1220     xrtol : float, default: 0 
1221         Relative tolerance for `x`. Terminate successfully if step 
1222         size is less than ``xk * xrtol`` where ``xk`` is the current 
1223         parameter vector. 
1224     c1 : float, default: 1e-4 
1225         Parameter for Armijo condition rule. 
1226     c2 : float, default: 0.9 
1227         Parameter for curvature condition rule. 
1228     hess_inv0 : None or ndarray, optional`` 
1229         Initial inverse hessian estimate, shape (n, n). If None (default) then 
1230         the identity matrix is used. 
1231  
1232     Returns 
1233     ------- 
1234     xopt : ndarray 
1235         Parameters which minimize f, i.e., ``f(xopt) == fopt``. 
1236     fopt : float 
1237         Minimum value. 
1238     gopt : ndarray 
1239         Value of gradient at minimum, f'(xopt), which should be near 0. 
1240     Bopt : ndarray 
1241         Value of 1/f''(xopt), i.e., the inverse Hessian matrix. 
1242     func_calls : int 
1243         Number of function_calls made. 
1244     grad_calls : int 
1245         Number of gradient calls made. 
1246     warnflag : integer 
1247         1 : Maximum number of iterations exceeded. 
1248         2 : Gradient and/or function calls not changing. 
1249         3 : NaN result encountered. 
1250     allvecs : list 
1251         The value of `xopt` at each iteration. Only returned if `retall` is 
1252         True. 
1253  
1254     Notes

1328 def _minimize_bfgs(fun, x0, args=(), jac=None, callback=None,
1329                    gtol=1e-5, norm=np.inf, eps=_epsilon, maxiter=None,
1330                    disp=False, return_all=False, finite_diff_rel_step=None,
1331                    xrtol=0, c1=1e-4, c2=0.9,
1332                    hess_inv0=None, **unknown_options):
1333     """ 
1334     Minimization of scalar function of one or more variables using the 
1335     BFGS algorithm. 
1336  
1337     Options 
1338     ------- 
1339     disp : bool 
1340         Set to True to print convergence messages. 
1341     maxiter : int 
1342         Maximum number of iterations to perform. 
1343     gtol : float 
1344         Terminate successfully if gradient norm is less than `gtol`. 
1345     norm : float 
1346         Order of norm (Inf is max, -Inf is min). 
1347     eps : float or ndarray 
1348         If `jac is None` the absolute step size used for numerical 
1349         approximation of the jacobian via forward differences. 
1350     return_all : bool, optional 
1351         Set to True to return a list of the best solution at each of the 
1352         iterations. 
1353     finite_diff_rel_step : None or array_like, optional 
1354         If ``jac in ['2-point', '3-point', 'cs']`` the relative step size to 
1355         use for numerical approximation of the jacobian. The absolute step 
1356         size is computed as ``h = rel_step * sign(x) * max(1, abs(x))``, 
1357         possibly adjusted to fit into the bounds. For ``jac='3-point'`` 
1358         the sign of `h` is ignored. If None (default) then step is selected 
1359         automatically. 
1360     xrtol : float, default: 0 
1361         Relative tolerance for `x`. Terminate successfully if step size is 
1362         less than ``xk * xrtol`` where ``xk`` is the current parameter vector. 
1363     c1 : float, default: 1e-4 
1364         Parameter for Armijo condition rule. 
1365     c2 : float, default: 0.9 
1366         Parameter for curvature condition rule. 
1367     hess_inv0 : None or ndarray, optional 
1368         Initial inverse hessian estimate, shape (n, n). If None (default) then 
1369         the identity matrix is used. 
1370  
1371     Notes 
1372     ----- 
1373     Parameters `c1` and `c2` must satisfy ``0 < c1 < c2 < 1``. 
1374  
1375     If minimization doesn't complete successfully, with an error message of 
1376     ``Desired error not necessarily achieved due to precision loss``, then 
1377     consider setting `gtol` to a higher value. This precision loss typically 
1378     occurs when the (finite difference) numerical differentiation cannot provide 
1379     sufficient precision to satisfy the `gtol` termination criterion. 
1380     This can happen when working in single precision and a callable jac is not 
1381     provided. For single precision problems a `gtol` of 1e-3 seems to work. 
1382     """
1383     _check_unknown_options(unknown_options)
1384     _check_positive_definite(hess_inv0)
1385     retall = return_all
1386 
1387     x0 = asarray(x0).flatten()
1388     if x0.ndim == 0:
1389         x0.shape = (1,)
1390     if maxiter is None:
1391         maxiter = len(x0) * 200
1392 
1393     sf = _prepare_scalar_function(fun, x0, jac, args=args, epsilon=eps,
1394                                   finite_diff_rel_step=finite_diff_rel_step)
1395 
1396     f = sf.fun
1397     myfprime = sf.grad
1398 
1399     old_fval = f(x0)
1400     gfk = myfprime(x0)
1401 
1402     k = 0
1403     N = len(x0)
1404     I = np.eye(N, dtype=int)
1405     Hk = I if hess_inv0 is None else hess_inv0
1406 
1407     # Sets the initial step guess to dx ~ 1
1408     old_old_fval = old_fval + np.linalg.norm(gfk) / 2
1409 
1410     xk = x0
1411     if retall:
1412         allvecs = [x0]
1413     warnflag = 0
1414     gnorm = vecnorm(gfk, ord=norm)
1415     while (gnorm > gtol) and (k < maxiter):
1416         pk = -np.dot(Hk, gfk)
1417         try:
1418             alpha_k, fc, gc, old_fval, old_old_fval, gfkp1 = \
1419                      _line_search_wolfe12(f, myfprime, xk, pk, gfk,
1420                                           old_fval, old_old_fval, amin=1e-100,
1421                                           amax=1e100, c1=c1, c2=c2)
1422         except _LineSearchError:
1423             # Line search failed to find a better solution.
1424             warnflag = 2
1425             break
1426 
1427         sk = alpha_k * pk
1428         xkp1 = xk + sk
1429 
1430         if retall:
1431             allvecs.append(xkp1)
1432         xk = xkp1
1433         if gfkp1 is None:
1434             gfkp1 = myfprime(xkp1)
1435 
1436         yk = gfkp1 - gfk
1437         gfk = gfkp1
1438         k += 1
1439         intermediate_result = OptimizeResult(x=xk, fun=old_fval)
1440         if _call_callback_maybe_halt(callback, intermediate_result):
1441             break
1442         gnorm = vecnorm(gfk, ord=norm)
1443         if (gnorm <= gtol):
1444             break
1445 
1446         #  See Chapter 5 in  P.E. Frandsen, K. Jonasson, H.B. Nielsen,
1447         #  O. Tingleff: "Unconstrained Optimization", IMM, DTU.  1999.
1448         #  These notes are available here:
1449         #  http://www2.imm.dtu.dk/documents/ftp/publlec.html
1450         if (alpha_k*vecnorm(pk) <= xrtol*(xrtol + vecnorm(xk))):
1451             break
1452 
1453         if not np.isfinite(old_fval):
1454             # We correctly found +-Inf as optimal value, or something went
1455             # wrong.
1456             warnflag = 2
1457             break
1458 
1459         rhok_inv = np.dot(yk, sk)
1460         # this was handled in numeric, let it remains for more safety
1461         # Cryptic comment above is preserved for posterity. Future reader:
1462         # consider change to condition below proposed in gh-1261/gh-17345.
1463         if rhok_inv == 0.:
1464             rhok = 1000.0
1465             if disp:
1466                 msg = "Divide-by-zero encountered: rhok assumed large"
1467                 _print_success_message_or_warn(True, msg)
1468         else:
1469             rhok = 1. / rhok_inv
1470 
1471         A1 = I - sk[:, np.newaxis] * yk[np.newaxis, :] * rhok
1472         A2 = I - yk[:, np.newaxis] * sk[np.newaxis, :] * rhok
1473         Hk = np.dot(A1, np.dot(Hk, A2)) + (rhok * sk[:, np.newaxis] *
1474                                                  sk[np.newaxis, :])
1475 
1476     fval = old_fval
1477 
1478     if warnflag == 2:
1479         msg = _status_message['pr_loss']
1480     elif k >= maxiter:
1481         warnflag = 1
1482         msg = _status_message['maxiter']
1483     elif np.isnan(gnorm) or np.isnan(fval) or np.isnan(xk).any():
1484         warnflag = 3
1485         msg = _status_message['nan']
1486     else:
1487         msg = _status_message['success']
1488 
1489     if disp:
1490         _print_success_message_or_warn(warnflag, msg)
1491         print(f"         Current function value: {fval:f}")
1492         print("         Iterations: %d" % k)
1493         print("         Function evaluations: %d" % sf.nfev)
1494         print("         Gradient evaluations: %d" % sf.ngev)
1495 
1496     result = OptimizeResult(fun=fval, jac=gfk, hess_inv=Hk, nfev=sf.nfev,
1497                             njev=sf.ngev, status=warnflag,
1498                             success=(warnflag == 0), message=msg, x=xk,
1499                             nit=k)
1500     if retall:
1501         result['allvecs'] = allvecs
1502     return result

1512 def fmin_cg(f, x0, fprime=None, args=(), gtol=1e-5, norm=np.inf,
1513             epsilon=_epsilon, maxiter=None, full_output=0, disp=1, retall=0,
1514             callback=None, c1=1e-4, c2=0.4):
1515     """ 
1516     Minimize a function using a nonlinear conjugate gradient algorithm. 
1517  
1518     Parameters 
1519     ---------- 
1520     f : callable, ``f(x, *args)`` 
1521         Objective function to be minimized. Here `x` must be a 1-D array of 
1522         the variables that are to be changed in the search for a minimum, and 
1523         `args` are the other (fixed) parameters of `f`. 
1524     x0 : ndarray 
1525         A user-supplied initial estimate of `xopt`, the optimal value of `x`. 
1526         It must be a 1-D array of values. 
1527     fprime : callable, ``fprime(x, *args)``, optional 
1528         A function that returns the gradient of `f` at `x`. Here `x` and `args` 
1529         are as described above for `f`. The returned value must be a 1-D array. 
1530         Defaults to None, in which case the gradient is approximated 
1531         numerically (see `epsilon`, below). 
1532     args : tuple, optional 
1533         Parameter values passed to `f` and `fprime`. Must be supplied whenever 
1534         additional fixed parameters are needed to completely specify the 
1535         functions `f` and `fprime`. 
1536     gtol : float, optional 
1537         Stop when the norm of the gradient is less than `gtol`. 
1538     norm : float, optional 
1539         Order to use for the norm of the gradient 
1540         (``-np.inf`` is min, ``np.inf`` is max). 
1541     epsilon : float or ndarray, optional 
1542         Step size(s) to use when `fprime` is approximated numerically. Can be a 
1543         scalar or a 1-D array. Defaults to ``sqrt(eps)``, with eps the 
1544         floating point machine precision.  Usually ``sqrt(eps)`` is about 
1545         1.5e-8. 
1546     maxiter : int, optional 
1547         Maximum number of iterations to perform. Default is ``200 * len(x0)``. 
1548     full_output : bool, optional 
1549         If True, return `fopt`, `func_calls`, `grad_calls`, and `warnflag` in 
1550         addition to `xopt`.  See the Returns section below for additional 
1551         information on optional return values. 
1552     disp : bool, optional 
1553         If True, return a convergence message, followed by `xopt`. 
1554     retall : bool, optional 
1555         If True, add to the returned values the results of each iteration. 
1556     callback : callable, optional 
1557         An optional user-supplied function, called after each iteration. 
1558         Called as ``callback(xk)``, where ``xk`` is the current value of `x0`. 
1559     c1 : float, default: 1e-4 
1560         Parameter for Armijo condition rule. 
1561     c2 : float, default: 0.4 
1562         Parameter for curvature condition rule. 
1563  
1564     Returns 
1565     ------- 
1566     xopt : ndarray 
1567         Parameters which minimize f, i.e., ``f(xopt) == fopt``. 
1568     fopt : float, optional 
1569         Minimum value found, f(xopt). Only returned if `full_output` is True. 
1570     func_calls : int, optional 
1571         The number of function_calls made. Only returned if `full_output` 
1572         is True. 
1573     grad_calls : int, optional 
1574         The number of gradient calls made. Only returned if `full_output` is 
1575         True. 
1576     warnflag : int, optional 
1577         Integer value with warning status, only returned if `full_output` is 
1578         True. 
1579  
1580         0 : Success. 
1581  
1582         1 : The maximum number of iterations was exceeded. 
1583  
1584         2 : Gradient and/or function calls were not changing. May indicate 
1585             that precision was lost, i.e., the routine did not converge. 
1586  
1587         3 : NaN result encountered. 
1588  
1589     allvecs : list of ndarray, optional 
1590         List of arrays, containing the results at each iteration. 
1591         Only returned if `retall` is True. 
1592  
1593     See Also 
1594     -------- 
1595     minimize : common interface to all `scipy.optimize` algorithms for 
1596                unconstrained and constrained minimization of multivariate 
1597                functions. It provides an alternative way to call 
1598                ``fmin_cg``, by specifying ``method='CG'``. 
1599  
1600     Notes 
1601     ----- 
1602     This conjugate gradient algorithm is based on that of Polak and Ribiere 
1603     [1]_. 
1604  
1605     Conjugate gradient methods tend to work better when: 
1606  
1607     1. `f` has a unique global minimizing point, and no local minima or 
1608        other stationary points, 
1609     2. `f` is, at least locally, reasonably well approximated by a 
1610        quadratic function of the variables, 
1611     3. `f` is continuous and has a continuous gradient, 
1612     4. `fprime` is not too large, e.g., has a norm less than 1000, 
1613     5. The initial guess, `x0`, is reasonably close to `f` 's global 
1614        minimizing point, `xopt`. 
1615  
1616     Parameters `c1` and `c2` must satisfy ``0 < c1 < c2 < 1``. 
1617  
1618     References 
1619     ---------- 
1620     .. [1] Wright & Nocedal, "Numerical Optimization", 1999, pp. 120-122. 
1621  
1622     Examples 
1623     -------- 
1624     Example 1: seek the minimum value of the expression 
1625     ``a*u**2 + b*u*v + c*v**2 + d*u + e*v + f`` for given values 
1626     of the parameters and an initial guess ``(u, v) = (0, 0)``. 
1627  
1628     >>> import numpy as np 
1629     >>> args = (2, 3, 7, 8, 9, 10)  # parameter values 
1630     >>> def f(x, *args): 
1631     ...     u, v = x 
1632     ...     a, b, c, d, e, f = args 
1633     ...     return a*u**2 + b*u*v + c*v**2 + d*u + e*v + f 
1634     >>> def gradf(x, *args): 
1635     ...     u, v = x 
1636     ...     a, b, c, d, e, f = args 
1637     ...     gu = 2*a*u + b*v + d     # u-component of the gradient 
1638     ...     gv = b*u + 2*c*v + e     # v-component of the gradient 
1639     ...     return np.asarray((gu, gv)) 
1640     >>> x0 = np.asarray((0, 0))  # Initial guess. 
1641     >>> from scipy import optimize 
1642     >>> res1 = optimize.fmin_cg(f, x0, fprime=gradf, args=args) 
1643     Optimization terminated successfully. 
1644              Current function value: 1.617021 
1645              Iterations: 4 
1646              Function evaluations: 8 
1647              Gradient evaluations: 8 
1648     >>> res1 
1649     array([-1.80851064, -0.25531915]) 
1650  
1651     Example 2: solve the same problem using the `minimize` function. 
1652     (This `myopts` dictionary shows all of the available options, 
1653     although in practice only non-default values would be needed. 
1654     The returned value will be a dictionary.)

1656     >>> opts = {'maxiter' : None,    # default value. 
1657     ...         'disp' : True,    # non-default value. 
1658     ...         'gtol' : 1e-5,    # default value. 
1659     ...         'norm' : np.inf,  # default value. 
1660     ...         'eps' : 1.4901161193847656e-08}  # default value. 
1661     >>> res2 = optimize.minimize(f, x0, jac=gradf, args=args, 
1662     ...                          method='CG', options=opts) 
1663     Optimization terminated successfully. 
1664             Current function value: 1.617021 
1665             Iterations: 4 
1666             Function evaluations: 8 
1667             Gradient evaluations: 8 
1668     >>> res2.x  # minimum found

1673             'norm': norm,
1674             'eps': epsilon,
1675             'disp': disp,
1676             'maxiter': maxiter,
1677             'return_all': retall}

1681                        **opts)
1682

1684         retlist = res['x'], res['fun'], res['nfev'], res['njev'], res['status']
1685         if retall:
1686             retlist += (res['allvecs'], )

1695 def _minimize_cg(fun, x0, args=(), jac=None, callback=None,
1696                  gtol=1e-5, norm=np.inf, eps=_epsilon, maxiter=None,
1697                  disp=False, return_all=False, finite_diff_rel_step=None,
1698                  c1=1e-4, c2=0.4, **unknown_options):
1699     """ 
1700     Minimization of scalar function of one or more variables using the 
1701     conjugate gradient algorithm. 
1702  
1703     Options 
1704     ------- 
1705     disp : bool 
1706         Set to True to print convergence messages. 
1707     maxiter : int 
1708         Maximum number of iterations to perform. 
1709     gtol : float 
1710         Gradient norm must be less than `gtol` before successful 
1711         termination. 
1712     norm : float 
1713         Order of norm (Inf is max, -Inf is min). 
1714     eps : float or ndarray 
1715         If `jac is None` the absolute step size used for numerical 
1716         approximation of the jacobian via forward differences. 
1717     return_all : bool, optional 
1718         Set to True to return a list of the best solution at each of the 
1719         iterations. 
1720     finite_diff_rel_step : None or array_like, optional 
1721         If ``jac in ['2-point', '3-point', 'cs']`` the relative step size to 
1722         use for numerical approximation of the jacobian. The absolute step 
1723         size is computed as ``h = rel_step * sign(x) * max(1, abs(x))``, 
1724         possibly adjusted to fit into the bounds. For ``jac='3-point'`` 
1725         the sign of `h` is ignored. If None (default) then step is selected 
1726         automatically. 
1727     c1 : float, default: 1e-4 
1728         Parameter for Armijo condition rule. 
1729     c2 : float, default: 0.4 
1730         Parameter for curvature condition rule. 
1731  
1732     Notes 
1733     ----- 
1734     Parameters `c1` and `c2` must satisfy ``0 < c1 < c2 < 1``. 
1735     """
1736     _check_unknown_options(unknown_options)
1737 
1738     retall = return_all
1739 
1740     x0 = asarray(x0).flatten()
1741     if maxiter is None:
1742         maxiter = len(x0) * 200
1743 
1744     sf = _prepare_scalar_function(fun, x0, jac=jac, args=args, epsilon=eps,
1745                                   finite_diff_rel_step=finite_diff_rel_step)
1746 
1747     f = sf.fun
1748     myfprime = sf.grad
1749 
1750     old_fval = f(x0)
1751     gfk = myfprime(x0)
1752 
1753     k = 0
1754     xk = x0
1755     # Sets the initial step guess to dx ~ 1
1756     old_old_fval = old_fval + np.linalg.norm(gfk) / 2
1757 
1758     if retall:
1759         allvecs = [xk]
1760     warnflag = 0
1761     pk = -gfk
1762     gnorm = vecnorm(gfk, ord=norm)
1763 
1764     sigma_3 = 0.01
1765 
1766     while (gnorm > gtol) and (k < maxiter):
1767         deltak = np.dot(gfk, gfk)
1768 
1769         cached_step = [None]
1770 
1771         def polak_ribiere_powell_step(alpha, gfkp1=None):
1772             xkp1 = xk + alpha * pk
1773             if gfkp1 is None:
1774                 gfkp1 = myfprime(xkp1)
1775             yk = gfkp1 - gfk
1776             beta_k = max(0, np.dot(yk, gfkp1) / deltak)
1777             pkp1 = -gfkp1 + beta_k * pk
1778             gnorm = vecnorm(gfkp1, ord=norm)
1779             return (alpha, xkp1, pkp1, gfkp1, gnorm)
1780 
1781         def descent_condition(alpha, xkp1, fp1, gfkp1):
1782             # Polak-Ribiere+ needs an explicit check of a sufficient
1783             # descent condition, which is not guaranteed by strong Wolfe.
1784             #
1785             # See Gilbert & Nocedal, "Global convergence properties of
1786             # conjugate gradient methods for optimization",
1787             # SIAM J. Optimization 2, 21 (1992).
1788             cached_step[:] = polak_ribiere_powell_step(alpha, gfkp1)
1789             alpha, xk, pk, gfk, gnorm = cached_step
1790 
1791             # Accept step if it leads to convergence.
1792             if gnorm <= gtol:
1793                 return True
1794 
1795             # Accept step if sufficient descent condition applies.
1796             return np.dot(pk, gfk) <= -sigma_3 * np.dot(gfk, gfk)
1797 
1798         try:
1799             alpha_k, fc, gc, old_fval, old_old_fval, gfkp1 = \
1800                      _line_search_wolfe12(f, myfprime, xk, pk, gfk, old_fval,
1801                                           old_old_fval, c1=c1, c2=c2, amin=1e-100,
1802                                           amax=1e100, extra_condition=descent_condition)
1803         except _LineSearchError:
1804             # Line search failed to find a better solution.
1805             warnflag = 2
1806             break
1807 
1808         # Reuse already computed results if possible
1809         if alpha_k == cached_step[0]:
1810             alpha_k, xk, pk, gfk, gnorm = cached_step
1811         else:
1812             alpha_k, xk, pk, gfk, gnorm = polak_ribiere_powell_step(alpha_k, gfkp1)
1813 
1814         if retall:
1815             allvecs.append(xk)
1816         k += 1
1817         intermediate_result = OptimizeResult(x=xk, fun=old_fval)
1818         if _call_callback_maybe_halt(callback, intermediate_result):
1819             break
1820

2459     def optimize(self):
2460         # set up for optimization
2461         func = self.func
2462         xa, xb, xc, fa, fb, fc, funcalls = self.get_bracket_info()
2463         _mintol = self._mintol
2464         _cg = self._cg
2465         #################################
2466         #BEGIN CORE ALGORITHM
2467         #################################
2468         x = w = v = xb
2469         fw = fv = fx = fb
2470         if (xa < xc):
2471             a = xa
2472             b = xc
2473         else:
2474             a = xc
2475             b = xa
2476         deltax = 0.0
2477         iter = 0
2478 
2479         if self.disp > 2:
2480             print(" ")
2481             print(f"{'Func-count':^12} {'x':^12} {'f(x)': ^12}")
2482             print(f"{funcalls:^12g} {x:^12.6g} {fx:^12.6g}")
2483 
2484         while (iter < self.maxiter):
2485             tol1 = self.tol * np.abs(x) + _mintol
2486             tol2 = 2.0 * tol1
2487             xmid = 0.5 * (a + b)
2488             # check for convergence
2489             if np.abs(x - xmid) < (tol2 - 0.5 * (b - a)):
2490                 break
2491             # XXX In the first iteration, rat is only bound in the true case
2492             # of this conditional. This used to cause an UnboundLocalError
2493             # (gh-4140). It should be set before the if (but to what?).
2494             if (np.abs(deltax) <= tol1):
2495                 if (x >= xmid):
2496                     deltax = a - x       # do a golden section step
2497                 else:
2498                     deltax = b - x
2499                 rat = _cg * deltax
2500             else:                              # do a parabolic step
2501                 tmp1 = (x - w) * (fx - fv)
2502                 tmp2 = (x - v) * (fx - fw)
2503                 p = (x - v) * tmp2 - (x - w) * tmp1
2504                 tmp2 = 2.0 * (tmp2 - tmp1)
2505                 if (tmp2 > 0.0):
2506                     p = -p
2507                 tmp2 = np.abs(tmp2)
2508                 dx_temp = deltax
2509                 deltax = rat
2510                 # check parabolic fit
2511                 if ((p > tmp2 * (a - x)) and (p < tmp2 * (b - x)) and
2512                         (np.abs(p) < np.abs(0.5 * tmp2 * dx_temp))):
2513                     rat = p * 1.0 / tmp2        # if parabolic step is useful.
2514                     u = x + rat
2515                     if ((u - a) < tol2 or (b - u) < tol2):
2516                         if xmid - x >= 0:
2517                             rat = tol1
2518                         else:
2519                             rat = -tol1
2520                 else:
2521                     if (x >= xmid):
2522                         deltax = a - x  # if it's not do a golden section step
2523                     else:
2524                         deltax = b - x
2525                     rat = _cg * deltax
2526 
2527             if (np.abs(rat) < tol1):            # update by at least tol1
2528                 if rat >= 0:
2529                     u = x + tol1
2530                 else:
2531                     u = x - tol1
2532             else:
2533                 u = x + rat
2534             fu = func(*((u,) + self.args))      # calculate new output value
2535             funcalls += 1
2536 
2537             if (fu > fx):                 # if it's bigger than current
2538                 if (u < x):
2539                     a = u
2540                 else:
2541                     b = u
2542                 if (fu <= fw) or (w == x):
2543                     v = w
2544                     w = u
2545                     fv = fw
2546                     fw = fu
2547                 elif (fu <= fv) or (v == x) or (v == w):
2548                     v = u
2549                     fv = fu
2550             else:
2551                 if (u >= x):
2552                     a = x
2553                 else:
2554                     b = x
2555                 v = w
2556                 w = x
2557                 x = u
2558                 fv = fw
2559                 fw = fx
2560                 fx = fu
2561 
2562             if self.disp > 2:
2563                 print(f"{funcalls:^12g} {x:^12.6g} {fx:^12.6g}")
2564 
2565             iter += 1
2566         #################################
2567         #END CORE ALGORITHM
2568         #################################
2569

2571         self.fval = fx
2572         self.iter = iter
2573         self.funcalls = funcalls

2916 def bracket(func, xa=0.0, xb=1.0, args=(), grow_limit=110.0, maxiter=1000):
2917     """ 
2918     Bracket the minimum of a function. 
2919  
2920     Given a function and distinct initial points, search in the 
2921     downhill direction (as defined by the initial points) and return 
2922     three points that bracket the minimum of the function. 
2923  
2924     Parameters 
2925     ---------- 
2926     func : callable f(x,*args) 
2927         Objective function to minimize. 
2928     xa, xb : float, optional 
2929         Initial points. Defaults `xa` to 0.0, and `xb` to 1.0. 
2930         A local minimum need not be contained within this interval. 
2931     args : tuple, optional 
2932         Additional arguments (if present), passed to `func`. 
2933     grow_limit : float, optional 
2934         Maximum grow limit.  Defaults to 110.0 
2935     maxiter : int, optional 
2936         Maximum number of iterations to perform. Defaults to 1000. 
2937  
2938     Returns 
2939     ------- 
2940     xa, xb, xc : float 
2941         Final points of the bracket. 
2942     fa, fb, fc : float 
2943         Objective function values at the bracket points. 
2944     funcalls : int 
2945         Number of function evaluations made. 
2946  
2947     Raises 
2948     ------ 
2949     BracketError 
2950         If no valid bracket is found before the algorithm terminates. 
2951         See notes for conditions of a valid bracket. 
2952  
2953     Notes 
2954     ----- 
2955     The algorithm attempts to find three strictly ordered points (i.e. 
2956     :math:`x_a < x_b < x_c` or :math:`x_c < x_b < x_a`) satisfying 
2957     :math:`f(x_b)  f(x_a)` and :math:`f(x_b)  f(x_c)`, where one of the 
2958     inequalities must be satisfied strictly and all :math:`x_i` must be 
2959     finite. 
2960  
2961     Examples 
2962     -------- 
2963     This function can find a downward convex region of a function: 
2964  
2965     >>> import numpy as np 
2966     >>> import matplotlib.pyplot as plt 
2967     >>> from scipy.optimize import bracket 
2968     >>> def f(x): 
2969     ...     return 10*x**2 + 3*x + 5 
2970     >>> x = np.linspace(-2, 2) 
2971     >>> y = f(x) 
2972     >>> init_xa, init_xb = 0.1, 1 
2973     >>> xa, xb, xc, fa, fb, fc, funcalls = bracket(f, xa=init_xa, xb=init_xb) 
2974     >>> plt.axvline(x=init_xa, color="k", linestyle="--") 
2975     >>> plt.axvline(x=init_xb, color="k", linestyle="--") 
2976     >>> plt.plot(x, y, "-k") 
2977     >>> plt.plot(xa, fa, "bx") 
2978     >>> plt.plot(xb, fb, "rx") 
2979     >>> plt.plot(xc, fc, "bx") 
2980     >>> plt.show() 
2981  
2982     Note that both initial points were to the right of the minimum, and the 
2983     third point was found in the "downhill" direction: the direction 
2984     in which the function appeared to be decreasing (to the left). 
2985     The final points are strictly ordered, and the function value 
2986     at the middle point is less than the function values at the endpoints; 
2987     it follows that a minimum must lie within the bracket. 
2988  
2989     """
2990     _gold = 1.618034  # golden ratio: (1.0+sqrt(5.0))/2.0
2991     _verysmall_num = 1e-21
2992     # convert to numpy floats if not already
2993     xa, xb = np.asarray([xa, xb])
2994     fa = func(*(xa,) + args)
2995     fb = func(*(xb,) + args)
2996     if (fa < fb):                      # Switch so fa > fb
2997         xa, xb = xb, xa
2998         fa, fb = fb, fa
2999     xc = xb + _gold * (xb - xa)
3000     fc = func(*((xc,) + args))
3001     funcalls = 3
3002     iter = 0
3003     while (fc < fb):
3004         tmp1 = (xb - xa) * (fb - fc)
3005         tmp2 = (xb - xc) * (fb - fa)
3006         val = tmp2 - tmp1
3007         if np.abs(val) < _verysmall_num:
3008             denom = 2.0 * _verysmall_num
3009         else:
3010             denom = 2.0 * val
3011         w = xb - ((xb - xc) * tmp2 - (xb - xa) * tmp1) / denom
3012         wlim = xb + grow_limit * (xc - xb)
3013         msg = ("No valid bracket was found before the iteration limit was "
3014                "reached. Consider trying different initial points or "
3015                "increasing `maxiter`.")
3016         if iter > maxiter:
3017             raise RuntimeError(msg)
3018         iter += 1
3019         if (w - xc) * (xb - w) > 0.0:
3020             fw = func(*((w,) + args))
3021             funcalls += 1
3022             if (fw < fc):
3023                 xa = xb
3024                 xb = w
3025                 fa = fb
3026                 fb = fw
3027                 break
3028             elif (fw > fb):
3029                 xc = w
3030                 fc = fw
3031                 break
3032             w = xc + _gold * (xc - xb)
3033             fw = func(*((w,) + args))
3034             funcalls += 1
3035         elif (w - wlim)*(wlim - xc) >= 0.0:
3036             w = wlim
3037             fw = func(*((w,) + args))
3038             funcalls += 1
3039         elif (w - wlim)*(xc - w) > 0.0:
3040             fw = func(*((w,) + args))
3041             funcalls += 1
3042             if (fw < fc):
3043                 xb = xc
3044                 xc = w
3045                 w = xc + _gold * (xc - xb)
3046                 fb = fc
3047                 fc = fw
3048                 fw = func(*((w,) + args))
3049                 funcalls += 1
3050         else:
3051             w = xc + _gold * (xc - xb)
3052             fw = func(*((w,) + args))
3053             funcalls += 1
3054         xa = xb
3055         xb = xc
3056         xc = w
3057         fa = fb
3058         fb = fc
3059         fc = fw
3060 
3061     # three conditions for a valid bracket

3079 def _recover_from_bracket_error(solver, fun, bracket, args, **options):
3080     # `bracket` was originally written without checking whether the resulting
3081     # bracket is valid. `brent` and `golden` built on top of it without
3082     # checking the returned bracket for validity, and their output can be
3083     # incorrect without warning/error if the original bracket is invalid.
3084     # gh-14858 noticed the problem, and the following is the desired
3085     # behavior:
3086     # - `scipy.optimize.bracket`, `scipy.optimize.brent`, and
3087     #   `scipy.optimize.golden` should raise an error if the bracket is
3088     #   invalid, as opposed to silently returning garbage
3089     # - `scipy.optimize.minimize_scalar` should return with `success=False`
3090     #   and other information
3091     # The changes that would be required to achieve this the traditional
3092     # way (`return`ing all the required information from bracket all the way
3093     # up to `minimizer_scalar`) are extensive and invasive. (See a6aa40d.)
3094     # We can achieve the same thing by raising the error in `bracket`, but
3095     # storing the information needed by `minimize_scalar` in the error object,
3096     # and intercepting it here.
3097     try:
3098         res = solver(fun, bracket, args, **options)
3099     except BracketError as e:
3100         msg = str(e)
3101         xa, xb, xc, fa, fb, fc, funcalls = e.data
3102         xs, fs = [xa, xb, xc], [fa, fb, fc]
3103         if np.any(np.isnan([xs, fs])):
3104             x, fun = np.nan, np.nan
3105         else:
3106             imin = np.argmin(fs)
3107             x, fun = xs[imin], fs[imin]
3108         return OptimizeResult(fun=fun, nfev=funcalls, x=x,
3109                               nit=0, success=False, message=msg)
3110     return res

3113 def _line_for_search(x0, alpha, lower_bound, upper_bound):
3114     """ 
3115     Given a parameter vector ``x0`` with length ``n`` and a direction 
3116     vector ``alpha`` with length ``n``, and lower and upper bounds on 
3117     each of the ``n`` parameters, what are the bounds on a scalar 
3118     ``l`` such that ``lower_bound <= x0 + alpha * l <= upper_bound``. 
3119  
3120  
3121     Parameters 
3122     ---------- 
3123     x0 : np.array. 
3124         The vector representing the current location. 
3125         Note ``np.shape(x0) == (n,)``. 
3126     alpha : np.array. 
3127         The vector representing the direction. 
3128         Note ``np.shape(alpha) == (n,)``. 
3129     lower_bound : np.array. 
3130         The lower bounds for each parameter in ``x0``. If the ``i``th 
3131         parameter in ``x0`` is unbounded below, then ``lower_bound[i]`` 
3132         should be ``-np.inf``. 
3133         Note ``np.shape(lower_bound) == (n,)``. 
3134     upper_bound : np.array. 
3135         The upper bounds for each parameter in ``x0``. If the ``i``th 
3136         parameter in ``x0`` is unbounded above, then ``upper_bound[i]`` 
3137         should be ``np.inf``. 
3138         Note ``np.shape(upper_bound) == (n,)``. 
3139  
3140     Returns 
3141     ------- 
3142     res : tuple ``(lmin, lmax)`` 
3143         The bounds for ``l`` such that 
3144             ``lower_bound[i] <= x0[i] + alpha[i] * l <= upper_bound[i]`` 
3145         for all ``i``. 
3146  
3147     """
3148     # get nonzero indices of alpha so we don't get any zero division errors.
3149     # alpha will not be all zero, since it is called from _linesearch_powell
3150     # where we have a check for this.
3151     nonzero, = alpha.nonzero()
3152     lower_bound, upper_bound = lower_bound[nonzero], upper_bound[nonzero]
3153     x0, alpha = x0[nonzero], alpha[nonzero]
3154     low = (lower_bound - x0) / alpha
3155     high = (upper_bound - x0) / alpha
3156 
3157     # positive and negative indices
3158     pos = alpha > 0
3159 
3160     lmin_pos = np.where(pos, low, 0)
3161     lmin_neg = np.where(pos, 0, high)
3162     lmax_pos = np.where(pos, high, 0)
3163     lmax_neg = np.where(pos, 0, low)
3164 
3165     lmin = np.max(lmin_pos + lmin_neg)
3166     lmax = np.min(lmax_pos + lmax_neg)
3167 
3168     # if x0 is outside the bounds, then it is possible that there is
3169     # no way to get back in the bounds for the parameters being updated
3170     # with the current direction alpha.
3171     # when this happens, lmax < lmin.
3172     # If this is the case, then we can just return (0, 0)
3173     return (lmin, lmax) if lmax >= lmin else (0, 0)

3176 def _linesearch_powell(func, p, xi, tol=1e-3,
3177                        lower_bound=None, upper_bound=None, fval=None):
3178     """Line-search algorithm using fminbound. 
3179  
3180     Find the minimum of the function ``func(x0 + alpha*direc)``. 
3181  
3182     lower_bound : np.array. 
3183         The lower bounds for each parameter in ``x0``. If the ``i``th 
3184         parameter in ``x0`` is unbounded below, then ``lower_bound[i]`` 
3185         should be ``-np.inf``. 
3186         Note ``np.shape(lower_bound) == (n,)``. 
3187     upper_bound : np.array. 
3188         The upper bounds for each parameter in ``x0``. If the ``i``th 
3189         parameter in ``x0`` is unbounded above, then ``upper_bound[i]`` 
3190         should be ``np.inf``. 
3191         Note ``np.shape(upper_bound) == (n,)``. 
3192     fval : number. 
3193         ``fval`` is equal to ``func(p)``, the idea is just to avoid 
3194         recomputing it so we can limit the ``fevals``. 
3195  
3196     """
3197     def myfunc(alpha):
3198         return func(p + alpha*xi)
3199 
3200     # if xi is zero, then don't optimize
3201     if not np.any(xi):
3202         return ((fval, p, xi) if fval is not None else (func(p), p, xi))
3203     elif lower_bound is None and upper_bound is None:
3204         # non-bounded minimization
3205         res = _recover_from_bracket_error(_minimize_scalar_brent,
3206                                           myfunc, None, tuple(), xtol=tol)
3207         alpha_min, fret = res.x, res.fun
3208         xi = alpha_min * xi
3209         return fret, p + xi, xi
3210     else:
3211         bound = _line_for_search(p, xi, lower_bound, upper_bound)
3212         if np.isneginf(bound[0]) and np.isposinf(bound[1]):
3213             # equivalent to unbounded
3214             return _linesearch_powell(func, p, xi, fval=fval, tol=tol)
3215         elif not np.isneginf(bound[0]) and not np.isposinf(bound[1]):
3216             # we can use a bounded scalar minimization
3217             res = _minimize_scalar_bounded(myfunc, bound, xatol=tol / 100)
3218             xi = res.x * xi
3219             return res.fun, p + xi, xi
3220         else:
3221             # only bounded on one side. use the tangent function to convert
3222             # the infinity bound to a finite bound. The new bounded region
3223             # is a subregion of the region bounded by -np.pi/2 and np.pi/2.
3224             bound = np.arctan(bound[0]), np.arctan(bound[1])
3225             res = _minimize_scalar_bounded(
3226                 lambda x: myfunc(np.tan(x)),
3227                 bound,
3228                 xatol=tol / 100)
3229             xi = np.tan(res.x) * xi
3230             return res.fun, p + xi, xi

3645     function. The value and type of the function evaluation returned when 
3646     ``full_output=True`` are affected in addition by the ``finish`` argument 
3647     (see Notes). 
3648  
3649     The brute force approach is inefficient because the number of grid points 
3650     increases exponentially - the number of grid points to evaluate is 
3651     ``Ns ** len(x)``. Consequently, even with coarse grid spacing, even 
3652     moderately sized problems can take a long time to run, and/or run into 
3653     memory limitations. 
3654  
3655     Parameters 
3656     ---------- 
3657     func : callable 
3658         The objective function to be minimized. Must be in the 
3659         form ``f(x, *args)``, where ``x`` is the argument in 
3660         the form of a 1-D array and ``args`` is a tuple of any 
3661         additional fixed parameters needed to completely specify 
3662         the function. 
3663     ranges : tuple 
3664         Each component of the `ranges` tuple must be either a 
3665         "slice object" or a range tuple of the form ``(low, high)``. 
3666         The program uses these to create the grid of points on which 
3667         the objective function will be computed. See `Note 2` for 
3668         more detail. 
3669     args : tuple, optional 
3670         Any additional fixed parameters needed to completely specify 
3671         the function. 
3672     Ns : int, optional 
3673         Number of grid points along the axes, if not otherwise 
3674         specified. See `Note2`. 
3675     full_output : bool, optional 
3676         If True, return the evaluation grid and the objective function's 
3677         values on it. 
3678     finish : callable, optional 
3679         An optimization function that is called with the result of brute force 
3680         minimization as initial guess. `finish` should take `func` and 
3681         the initial guess as positional arguments, and take `args` as 
3682         keyword arguments. It may additionally take `full_output` 
3683         and/or `disp` as keyword arguments. Use None if no "polishing" 
3684         function is to be used. See Notes for more details. 
3685     disp : bool, optional 
3686         Set to True to print convergence messages from the `finish` callable. 
3687     workers : int or map-like callable, optional 
3688         If `workers` is an int the grid is subdivided into `workers` 
3689         sections and evaluated in parallel (uses 
3690         `multiprocessing.Pool <multiprocessing>`). 
3691         Supply `-1` to use all cores available to the Process. 
3692         Alternatively supply a map-like callable, such as 
3693         `multiprocessing.Pool.map` for evaluating the grid in parallel. 
3694         This evaluation is carried out as ``workers(func, iterable)``. 
3695         Requires that `func` be pickleable. 
3696  
3697         .. versionadded:: 1.3.0 
3698  
3699     Returns 
3700     ------- 
3701     x0 : ndarray 
3702         A 1-D array containing the coordinates of a point at which the 
3703         objective function had its minimum value. (See `Note 1` for 
3704         which point is returned.) 
3705     fval : float 
3706         Function value at the point `x0`. (Returned when `full_output` is 
3707         True.) 
3708     grid : tuple 
3709         Representation of the evaluation grid. It has the same 
3710         length as `x0`. (Returned when `full_output` is True.) 
3711     Jout : ndarray 
3712         Function values at each point of the evaluation 
3713         grid, i.e., ``Jout = func(*grid)``. (Returned 
3714         when `full_output` is True.)

3716     See Also 
3717     -------- 
3718     basinhopping, differential_evolution

3720     Notes 
3721     ----- 
3722     *Note 1*: The program finds the gridpoint at which the lowest value 
3723     of the objective function occurs. If `finish` is None, that is the 
3724     point returned. When the global minimum occurs within (or not very far 
3725     outside) the grid's boundaries, and the grid is fine enough, that 
3726     point will be in the neighborhood of the global minimum. 
3727  
3728     However, users often employ some other optimization program to 
3729     "polish" the gridpoint values, i.e., to seek a more precise 
3730     (local) minimum near `brute's` best gridpoint. 
3731     The `brute` function's `finish` option provides a convenient way to do 
3732     that. Any polishing program used must take `brute's` output as its 
3733     initial guess as a positional argument, and take `brute's` input values 
3734     for `args` as keyword arguments, otherwise an error will be raised. 
3735     It may additionally take `full_output` and/or `disp` as keyword arguments. 
3736  
3737     `brute` assumes that the `finish` function returns either an 
3738     `OptimizeResult` object or a tuple in the form: 
3739     ``(xmin, Jmin, ... , statuscode)``, where ``xmin`` is the minimizing 
3740     value of the argument, ``Jmin`` is the minimum value of the objective 
3741     function, "..." may be some other returned values (which are not used 
3742     by `brute`), and ``statuscode`` is the status code of the `finish` program. 
3743  
3744     Note that when `finish` is not None, the values returned are those 
3745     of the `finish` program, *not* the gridpoint ones. Consequently, 
3746     while `brute` confines its search to the input grid points, 
3747     the `finish` program's results usually will not coincide with any 
3748     gridpoint, and may fall outside the grid's boundary. Thus, if a 
3749     minimum only needs to be found over the provided grid points, make 
3750     sure to pass in ``finish=None``. 
3751  
3752     *Note 2*: The grid of points is a `numpy.mgrid` object. 
3753     For `brute` the `ranges` and `Ns` inputs have the following effect. 
3754     Each component of the `ranges` tuple can be either a slice object or a 
3755     two-tuple giving a range of values, such as (0, 5). If the component is a 
3756     slice object, `brute` uses it directly. If the component is a two-tuple 
3757     range, `brute` internally converts it to a slice object that interpolates 
3758     `Ns` points from its low-value to its high-value, inclusive. 
3759  
3760     Examples 
3761     -------- 
3762     We illustrate the use of `brute` to seek the global minimum of a function 
3763     of two variables that is given as the sum of a positive-definite 
3764     quadratic and two deep "Gaussian-shaped" craters. Specifically, define 
3765     the objective function `f` as the sum of three other functions, 
3766     ``f = f1 + f2 + f3``. We suppose each of these has a signature 
3767     ``(z, *params)``, where ``z = (x, y)``,  and ``params`` and the functions 
3768     are as defined below. 
3769  
3770     >>> import numpy as np 
3771     >>> params = (2, 3, 7, 8, 9, 10, 44, -1, 2, 26, 1, -2, 0.5) 
3772     >>> def f1(z, *params): 
3773     ...     x, y = z 
3774     ...     a, b, c, d, e, f, g, h, i, j, k, l, scale = params 
3775     ...     return (a * x**2 + b * x * y + c * y**2 + d*x + e*y + f) 
3776  
3777     >>> def f2(z, *params): 
3778     ...     x, y = z 
3779     ...     a, b, c, d, e, f, g, h, i, j, k, l, scale = params 
3780     ...     return (-g*np.exp(-((x-h)**2 + (y-i)**2) / scale)) 
3781  
3782     >>> def f3(z, *params): 
3783     ...     x, y = z 
3784     ...     a, b, c, d, e, f, g, h, i, j, k, l, scale = params 
3785     ...     return (-j*np.exp(-((x-k)**2 + (y-l)**2) / scale)) 
3786  
3787     >>> def f(z, *params): 
3788     ...     return f1(z, *params) + f2(z, *params) + f3(z, *params) 
3789  
3790     Thus, the objective function may have local minima near the minimum 
3791     of each of the three functions of which it is composed. To 
3792     use `fmin` to polish its gridpoint result, we may then continue as 
3793     follows: 
3794  
3795     >>> rranges = (slice(-4, 4, 0.25), slice(-4, 4, 0.25)) 
3796     >>> from scipy import optimize 
3797     >>> resbrute = optimize.brute(f, rranges, args=params, full_output=True, 
3798     ...                           finish=optimize.fmin) 
3799     >>> resbrute[0]  # global minimum 
3800     array([-1.05665192,  1.80834843]) 
3801     >>> resbrute[1]  # function value at global minimum 
3802     -3.4085818767 
3803  
3804     Note that if `finish` had been set to None, we would have gotten the 
3805     gridpoint [-1.0 1.75] where the rounded function value is -2.892. 
3806  
3807     """
3808     N = len(ranges)
3809     if N > 40:
3810         raise ValueError("Brute Force not possible with more "
3811                          "than 40 variables.")

3813     for k in range(N):
3814         if not isinstance(lrange[k], slice):
3815             if len(lrange[k]) < 3:
3816                 lrange[k] = tuple(lrange[k]) + (complex(Ns),)
3817             lrange[k] = slice(*lrange[k])
3818     if (N == 1):
3819         lrange = lrange[0]
3820 
3821     grid = np.mgrid[lrange]
3822 
3823     # obtain an array of parameters that is iterable by a map-like callable
3824     inpt_shape = grid.shape
3825     if (N > 1):
3826         grid = np.reshape(grid, (inpt_shape[0], np.prod(inpt_shape[1:]))).T
3827 
3828     if not np.iterable(args):
3829         args = (args,)
3830

3832 
3833     # iterate over input arrays, possibly in parallel

3835         Jout = np.array(list(mapper(wrapped_func, grid)))

3844 
3845     indx = argmin(Jout.ravel(), axis=-1)
3846     Nindx = np.empty(N, int)
3847     xmin = np.empty(N, float)
3848     for k in range(N - 1, -1, -1):
3849         thisN = Nshape[k]
3850         Nindx[k] = indx % Nshape[k]
3851         indx = indx // thisN
3852     for k in range(N):
3853         xmin[k] = grid[k][tuple(Nindx)]
3854 
3855     Jmin = Jout[tuple(Nindx)]
3856     if (N == 1):
3857         grid = grid[0]
3858         xmin = xmin[0]
3859 
3860     if callable(finish):
3861         # set up kwargs for `finish` function
3862         finish_args = _getfullargspec(finish).args
3863         finish_kwargs = dict()
3864         if 'full_output' in finish_args:
3865             finish_kwargs['full_output'] = 1
3866         if 'disp' in finish_args:

3868         elif 'options' in finish_args:
3869             # pass 'disp' as `options`
3870             # (e.g., if `finish` is `minimize`)
3871             finish_kwargs['options'] = {'disp': disp}
3872 
3873         # run minimizer
3874         res = finish(func, xmin, args=args, **finish_kwargs)
3875 
3876         if isinstance(res, OptimizeResult):
3877             xmin = res.x
3878             Jmin = res.fun
3879             success = res.success
3880         else:
3881             xmin = res[0]
3882             Jmin = res[1]

3884         if not success:

3886                 warnings.warn("Either final optimization did not succeed or `finish` "
3887                               "does not return `statuscode` as its last argument.",
3888                               RuntimeWarning, stacklevel=2)
3889 
3890     if full_output:

3892     else:
