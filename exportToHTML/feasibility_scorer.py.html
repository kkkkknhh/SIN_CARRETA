383      def detect_components(self, text: str) -> List[DetectionResult]:
384          """ 
385          Detect all components in the given text using comprehensive regex patterns. 
386   
387          Args: 
388              text (str): Text to analyze for indicator components 
389   
390          Returns: 
391              List[DetectionResult]: List of detected components with confidence scores 
392                                    and position information 
393   
394          Note: 
395              Applies Unicode normalization before pattern matching to ensure 
396              consistent detection across different text encodings and formats. 
397   
398          """
399          results = []
400          # Apply Unicode normalization before processing
401          normalized_text = FeasibilityScorer._normalize_text(text)
402          text_lower = normalized_text.lower()
403  
404          for component_type, patterns in self.detection_patterns.items():
405              for pattern_info in patterns:
406                  pattern = pattern_info["pattern"]
407                  confidence = pattern_info["confidence"]
408  
409                  matches = re.finditer(pattern, text_lower, re.IGNORECASE)
410                  for match in matches:
411                      result = DetectionResult(
412                          component_type=component_type,
413                          matched_text=match.group(),
414                          confidence=confidence,
415                          position=match.start(),
416                      )
417                      results.append(result)
418  
419          return results

421      def _has_quantitative_component(
422          self, text: str, component_type: ComponentType
423      ) -> bool:
424          """ 
425          Check if a component has quantitative elements in nearby context. 
426   
427          Analyzes text around component mentions to determine if quantitative 
428          values are associated with baseline or target components. 
429   
430          Args: 
431              text (str): Text to analyze 
432              component_type (ComponentType): Component type to check for quantitative association 
433   
434          Returns: 
435              bool: True if quantitative elements are found within 30 characters of component mentions 
436   
437          Note: 
438              Uses a 30-character context window around component mentions to identify 
439              associated numerical values, percentages, or quantitative indicators. 
440   
441          """
442          # Apply Unicode normalization before processing
443          normalized_text = FeasibilityScorer._normalize_text(text)
444          text_lower = normalized_text.lower()
445  
446          # Find component mentions
447          component_positions = []
448          for pattern_info in self.detection_patterns[component_type]:
449              pattern = pattern_info["pattern"]
450              matches = re.finditer(pattern, text_lower, re.IGNORECASE)
451              component_positions.extend([match.start() for match in matches])
452  
453          if not component_positions:
454              return False
455  
456          # Check for numerical values within 30 characters of component mentions
457          numerical_patterns = self.detection_patterns[ComponentType.NUMERICAL]
458          for pos in component_positions:
459              context_start = max(0, pos - 30)
460              context_end = min(len(text), pos + 30)
461              context = text_lower[context_start:context_end]
462  
463              for pattern_info in numerical_patterns:
464                  if re.search(pattern_info["pattern"], context, re.IGNORECASE):
465                      return True
466  
467          return False

470          """ 
471          Calculate comprehensive feasibility score based on detected components and quality.

474          bonus components for comprehensive indicator quality assessment. 
475   
476          Args:

483              - Must have both baseline and target components for positive score

617   
618          Args:

620              compare_backends: If True, compare performance between threading and loky backends. 
621              use_parallel: Legacy parameter for backward compatibility. 
622   
623          Returns: 
624              List of IndicatorScore results in the same order as input indicators. 
625          """
626          if not indicators:
627              return []
628  
629          # Validate evidencia_soporte_list length if provided
630          if evidencia_soporte_list is not None and len(evidencia_soporte_list) != len(
631              indicators

633              raise ValueError(
634                  "evidencia_soporte_list must have the same length as indicators"
635              )
636  
637          # For small batches, use sequential processing to avoid overhead
638          if len(indicators) < 10 or not self.enable_parallel or not JOBLIB_AVAILABLE:
639              return self._batch_score_sequential(indicators, evidencia_soporte_list)

648      def _batch_score_sequential(
649          self,
650          indicators: List[str],
651          evidencia_soporte_list: Optional[List[Optional[int]]] = None,
652      ) -> List[IndicatorScore]:
653          """Sequential batch scoring."""
654          start_time = time.time()
655          results = []
656          for i, indicator in enumerate(indicators):
657              evidencia_soporte = (
658                  evidencia_soporte_list[i] if evidencia_soporte_list else None
659              )
660              results.append(
661                  self.calculate_feasibility_score(indicator, evidencia_soporte)
662              )
663          elapsed = time.time() - start_time
664  
665          self.logger.info(
666              f"Sequential processing: {len(indicators)} indicators in {elapsed:.3f}s "
667              f"({elapsed / len(indicators) * 1000:.1f}ms per indicator)"
668          )
669          return results

671      def _batch_score_parallel(
672          self,
673          indicators: List[str],
674          backend: str,
675          evidencia_soporte_list: Optional[List[Optional[int]]] = None,
676      ) -> List[IndicatorScore]:
677          """Parallel batch scoring using specified backend."""
678          if not JOBLIB_AVAILABLE:
679              raise RuntimeError("joblib not available for parallel processing")
680  
681          start_time = time.time()
682  
683          # Create a picklable instance for parallel processing
684          scorer_copy = self._create_picklable_copy()
685  
686          with Parallel(n_jobs=self.n_jobs, backend=backend) as parallel:
687              results = parallel(
688                  delayed(scorer_copy._score_single_indicator)(
689                      indicator,
690                      evidencia_soporte_list[i] if evidencia_soporte_list else None,
691                  )
692                  for i, indicator in enumerate(indicators)
693              )
694  
695          elapsed = time.time() - start_time
696          self.logger.info(
697              f"Parallel processing ({backend}): {len(indicators)} indicators in {elapsed:.3f}s "
698              f"({elapsed / len(indicators) * 1000:.1f}ms per indicator, {self.n_jobs} workers)"
699          )
700          return results

878      def calcular_calidad_evidencia(self, fragment: str) -> float:
879          """ 
880          Calculate evidence quality score for text fragments (0.0 to 1.0). 
881   
882          Higher scores for fragments containing: 
883          - Numerical values with monetary amounts (COP, $, millones) 
884          - Dates (YYYY format, quarters Q1-Q4, months) 
885          - Measurement terminology (baseline, meta, periodicidad) 
886   
887          Penalties applied for: 
888          - Indicators appearing only in titles/bullet points without values 
889          - Empty or malformed content 
890   
891          Args: 
892              fragment (str): Text fragment to analyze 
893   
894          Returns: 
895              float: Quality score between 0.0 and 1.0 
896          """
897          if not fragment or not fragment.strip():
898              return 0.0
899  
900          # Normalize Unicode text using NFKC
901          normalized_text = unicodedata.normalize("NFKC", fragment.strip())
902          text_lower = normalized_text.lower()
903  
904          # Initialize scoring components
905          scores = {
906              "monetary": 0.0,
907              "dates": 0.0,
908              "terminology": 0.0,
909              "structure_penalty": 0.0,
910          }
911  
912          # Weights for different scoring components
913          weights = {
914              "monetary": 0.35,
915              "dates": 0.25,
916              "terminology": 0.25,
917              "structure_penalty": -0.15,
918          }
919  
920          # 1. Monetary amount detection
921          scores["monetary"] = self._detect_monetary_values(text_lower)
922  
923          # 2. Date detection
924          scores["dates"] = self._detect_temporal_indicators(text_lower)
925  
926          # 3. Measurement terminology detection
927          scores["terminology"] = self._detect_measurement_terminology(
928              text_lower)
929  
930          # 4. Structure penalty for title-only indicators
931          scores["structure_penalty"] = self._calculate_structure_penalty(
932              normalized_text)
933  
934          # Calculate weighted final score
935          final_score = sum(
936              scores[component] * weights[component] for component in scores.keys()
937          )
938  
939          # Ensure score is between 0.0 and 1.0
940          return max(0.0, min(1.0, final_score))
