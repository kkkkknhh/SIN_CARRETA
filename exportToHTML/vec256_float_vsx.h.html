141    static Vectorized<float> C10_ALWAYS_INLINE blendv(
142        const Vectorized<float>& a,
143        const Vectorized<float>& b,
144        const Vectorized<float>& mask) {
145      // the mask used here returned by comparision of vec256
146      // assuming this we can use the same mask directly with vec_sel

149          vec_sel(a._vec1, b._vec1, mask._vecb1)};

153    static Vectorized<float> arange(
154        float base = 0.f,
155        step_t step = static_cast<step_t>(1)) {
156      return Vectorized<float>(
157          base,
158          base + step,
159          base + 2 * step,
160          base + 3 * step,
161          base + 4 * step,
162          base + 5 * step,
163          base + 6 * step,
164          base + 7 * step);

166    static Vectorized<float> set(
167        const Vectorized<float>& a,
168        const Vectorized<float>& b,
169        size_t count = size()) {
170      switch (count) {
171        case 0:
172          return a;
173        case 1:
174          return blend<1>(a, b);
175        case 2:
176          return blend<3>(a, b);
177        case 3:
178          return blend<7>(a, b);
179        case 4:
180          return blend<15>(a, b);
181        case 5:
182          return blend<31>(a, b);
183        case 6:
184          return blend<63>(a, b);
185        case 7:
186          return blend<127>(a, b);
187      }
188  
189      return b;
190    }
191    static Vectorized<value_type> C10_ALWAYS_INLINE
192    loadu(const void* ptr, int count = size()) {
193      if (count == size()) {
194        return {
195            vec_vsx_ld(offset0, reinterpret_cast<const value_type*>(ptr)),
196            vec_vsx_ld(offset16, reinterpret_cast<const value_type*>(ptr))};
197      }
198  
199      __at_align__ value_type tmp_values[size()] = {};
200      std::memcpy(tmp_values, ptr, std::min(count, size()) * sizeof(value_type));
201  
202      return {vec_vsx_ld(offset0, tmp_values), vec_vsx_ld(offset16, tmp_values)};
203    }
204    void C10_ALWAYS_INLINE store(void* ptr, int count = size()) const {
205      if (count == size()) {
206        vec_vsx_st(_vec0, offset0, reinterpret_cast<value_type*>(ptr));
207        vec_vsx_st(_vec1, offset16, reinterpret_cast<value_type*>(ptr));
208      } else if (count > 0) {
209        __at_align__ value_type tmp_values[size()];
210        vec_vsx_st(_vec0, offset0, tmp_values);
211        vec_vsx_st(_vec1, offset16, tmp_values);
212        std::memcpy(
213            ptr, tmp_values, std::min(count, size()) * sizeof(value_type));
214      }
215    }

220    Vectorized<float> map(float (*const f)(float)) const {
221      Vectorized<float> ret;
222      for (int i = 0; i < size() / 2; i++) {
223        ret._vec0[i] = f(_vec0[i]);
224      }
225      for (int i = 0; i < size() / 2; i++) {
226        ret._vec1[i] = f(_vec1[i]);
227      }
228      return ret;
229    }

234      Vectorized<float> ret;
235      for (int i = 0; i < size() / 2; i++) {
236        ret._vec0[i] = f(_vec0[i], other._vec0[i]);

238      for (int i = 0; i < size() / 2; i++) {
239        ret._vec1[i] = f(_vec1[i], other._vec1[i]);
240      }
241      return ret;

254    bool has_inf_nan() const {
255      for (const auto i : c10::irange(size() / 2)) {
256        if (_isnan(_vec0[i]) || _isinf(_vec0[i])) {
257          return true;
258        }
259      }
260      for (const auto i : c10::irange(size() / 2)) {
261        if (_isnan(_vec1[i]) || _isinf(_vec1[i])) {
262          return true;
263        }
264      }
265      return false;
266    }

268    int zero_mask() const {
269      // returns an integer mask where all zero elements are translated to 1-bit
270      // and others are translated to 0-bit
271      //__m256 cmp = _mm256_cmp_ps(values, _mm256_set1_ps(0.0f), _CMP_EQ_OQ);
272      auto cmp = (*this == zero);
273      // return _mm256_movemask_ps(cmp);
274      // possible simulation  //mask= lvsl ( 0 ) vbpermq( vec, mask <<5)
275      vuint64 result0 = vec_vbpermq((vuint8)cmp._vecb0, mask_zero_bits);
276      vuint64 result1 = vec_vbpermq((vuint8)cmp._vecb1, mask_zero_bits);
277      return (result0[1] >> 12 | (result1[1] >> 8));
278    }
