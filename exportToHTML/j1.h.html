5    /*                                                     j1.c 
6     * 
7     *     Bessel function of order one 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, y, j1(); 
14    * 
15    * y = j1( x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Returns Bessel function of order one of the argument. 
22    * 
23    * The domain is divided into the intervals [0, 8] and 
24    * (8, infinity). In the first interval a 24 term Chebyshev 
25    * expansion is used. In the second, the asymptotic 
26    * trigonometric representation is employed using two 
27    * rational functions of degree 5/5. 
28    * 
29    * 
30    * 
31    * ACCURACY: 
32    * 
33    *                      Absolute error: 
34    * arithmetic   domain      # trials      peak         rms 
35    *    IEEE      0, 30       30000       2.6e-16     1.1e-16 
36    * 
37    * 
38    */

43    * 
44    * 
45    * SYNOPSIS:

47    * double x, y, y1(); 
48    *

50    * 
51    * 
52    * 
53    * DESCRIPTION: 
54    * 
55    * Returns Bessel function of the second kind of order one

57    *

65    *

148      XSF_HOST_DEVICE inline double j1(double x) {
149          double w, z, p, q, xn;
150  
151          w = x;
152          if (x < 0) {
153              return -j1(-x);
154          }
155  
156          if (w <= 5.0) {
157              z = x * x;
158              w = polevl(z, detail::j1_RP, 3) / p1evl(z, detail::j1_RQ, 8);
159              w = w * x * (z - detail::j1_Z1) * (z - detail::j1_Z2);
160              return (w);
161          }
162  
163          w = 5.0 / x;
164          z = w * w;
165          p = polevl(z, detail::j1_PP, 6) / polevl(z, detail::j1_PQ, 6);
166          q = polevl(z, detail::j1_QP, 7) / p1evl(z, detail::j1_QQ, 7);
167          xn = x - detail::THPIO4;
168          p = p * std::cos(xn) - w * q * std::sin(xn);
169          return (p * detail::SQRT2OPI / std::sqrt(x));
170      }
171  
172      XSF_HOST_DEVICE inline double y1(double x) {
173          double w, z, p, q, xn;
174  
175          if (x <= 5.0) {
176              if (x == 0.0) {
177                  set_error("y1", SF_ERROR_SINGULAR, NULL);
178                  return -std::numeric_limits<double>::infinity();
179              } else if (x <= 0.0) {
180                  set_error("y1", SF_ERROR_DOMAIN, NULL);
181                  return std::numeric_limits<double>::quiet_NaN();
182              }
183              z = x * x;
184              w = x * (polevl(z, detail::j1_YP, 5) / p1evl(z, detail::j1_YQ, 8));
185              w += M_2_PI * (j1(x) * std::log(x) - 1.0 / x);
186              return (w);
187          }
188  
189          w = 5.0 / x;
190          z = w * w;
191          p = polevl(z, detail::j1_PP, 6) / polevl(z, detail::j1_PQ, 6);
192          q = polevl(z, detail::j1_QP, 7) / p1evl(z, detail::j1_QQ, 7);
193          xn = x - detail::THPIO4;
194          p = p * std::sin(xn) + w * q * std::cos(xn);
195          return (p * detail::SQRT2OPI / std::sqrt(x));
196      }
