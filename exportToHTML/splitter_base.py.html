102  @compatibility(is_backward_compatible=False)
103  class FxNetAccNodesFinder:
104      """ 
105      Finds a set of nodes that can be supported on ACC, excluding nodes that have non-tensor 
106      input/output to cpu nodes to prevent non-tensor data flow between backends and cpu. 
107   
108      I.e. if we have a chain: 
109   
110      ACC_NODE_1 -> ACC_NODE_2 -> ACC_NODE_3 -> CPU_NODE_1 
111   
112      where every ACC node produces non-tensor output, then they all should be treated as CPU nodes. 
113   
114      This behavior can be turned off by passing allow_non_tensor=True. 
115      """
116  
117      def __init__(
118          self,
119          module: torch.fx.GraphModule,
120          operator_support: OperatorSupportBase,
121          allow_non_tensor: bool,
122      ):
123          self.module = module
124          self.operator_support = operator_support
125          self.allow_non_tensor = allow_non_tensor
126          self.acc_nodes: NodeSet = set()
127  
128      def reduce_acc_nodes_non_tensor_input_helper(self, cpu_worklist: NodeList):
129          """ 
130          Transitively excludes nodes from ACC supported set. 
131          For every node in the worklist: 
132          - removes its downstream ACC nodes from ACC supported set, 
133          - if any downstream ACC node produces non-tensor output, 
134            then it gets added into the worklist. 
135          """
136          while cpu_worklist:
137              node = cpu_worklist.pop(0)
138  
139              for user in node.users:
140                  if user in self.acc_nodes:
141                      self.acc_nodes.remove(user)
142                      if not is_node_output_tensor(user):
143                          cpu_worklist.append(user)
144  
145      def reduce_acc_nodes_non_tensor_input(self):
146          """ 
147          Excludes nodes from ACC supported set that have direct 
148          upstream CPU nodes that produce non-tensor outputs. 
149          """
150          non_tensor_cpu_nodes: NodeList = []
151  
152          for node in self.module.graph.nodes:
153              if node.op not in CALLABLE_NODE_OPS:
154                  continue
155              if node in self.acc_nodes:
156                  continue
157              if is_node_output_tensor(node):
158                  continue
159              non_tensor_cpu_nodes.append(node)
160  
161          self.reduce_acc_nodes_non_tensor_input_helper(non_tensor_cpu_nodes)
162  
163      def reduce_acc_nodes_non_tensor_output(self):
164          """ 
165          Excludes nodes from ACC supported set that produce non-tensor 
166          outputs and have downstream CPU nodes. 
167          """
168          while True:
169              new_cpu_nodes: NodeList = []
170  
171              for acc_node in self.acc_nodes:
172                  if is_node_output_tensor(acc_node):
173                      continue
174                  for user in acc_node.users:
175                      if user not in self.acc_nodes:
176                          new_cpu_nodes.append(acc_node)
177                          break
178  
179              if not new_cpu_nodes:
180                  break
181  
182              for new_cpu_node in new_cpu_nodes:
183                  self.acc_nodes.remove(new_cpu_node)
184  
185              self.reduce_acc_nodes_non_tensor_input_helper(new_cpu_nodes)
186  
187      def __call__(self) -> NodeSet:
188          submodules = dict(self.module.named_modules())
189          self.acc_nodes = {
190              n
191              for n in self.module.graph.nodes
192              if n.op in CALLABLE_NODE_OPS
193              and self.operator_support.is_node_supported(submodules, n)
194          }
195  
196          if not self.allow_non_tensor:
197              self.reduce_acc_nodes_non_tensor_input()
198              self.reduce_acc_nodes_non_tensor_output()
199  
200          return self.acc_nodes

233  @compatibility(is_backward_compatible=False)
234  def generate_inputs_for_submodules(
235      model: torch.nn.Module,
236      inputs: Sequence[Any],
237      target_submodules: Iterable[str],
238      deepcopy: bool = False,
239  ) -> dict[str, Any]:
240      """ 
241      Generate inputs for targeting submdoules in the given model. Note that if two submodules refer to the same obj, this 
242      function doesn't work. 
243   
244      Args: 
245          model: root model. 
246          inputs: inputs to the root model. 
247          target_submodules: submodules that we want to generate inputs for. 
248   
249      Returns: 
250          A dict that maps from submodule name to its inputs. 
251      """
252  
253      handles = []
254      results = {}
255      submodule_to_names = {mod: name for name, mod in model.named_modules()}
256  
257      def pre_forward(module, module_inputs):
258          results[submodule_to_names[module]] = (
259              copy.deepcopy(module_inputs) if deepcopy else module_inputs
260          )
261  
262      for name, mod in model.named_modules():
263          if name in target_submodules:
264              handles.append(mod.register_forward_pre_hook(pre_forward))
265  
266      def clean_up_handles():
267          for h in handles:
268              h.remove()
269  
270      try:
271          with torch.no_grad():
272              model(*inputs)
273      except Exception as e:
274          clean_up_handles()
275          raise e
276  
277      clean_up_handles()
278      return results

281  class _SplitterBase:
282      """ 
283      Splits a GraphModule into sub-GraphModules for execution on CPU or the accelerator. 
284      Output is a GraphModule with supported and unsupported operators grouped into as few sub-GraphModules as possible. 
285      Assumes that only "call_module", "call_function" and "call_method" from FX IR can potentially be executed on the accelerator. 
286   
287      Given the following graph: 
288            ==> b ==> 
289          //         \\ 
290         a             d 
291          \\         // 
292            ==> c ==> 
293   
294      class SimpleModule(torch.nn.Module): 
295          def forward(self, a): 
296              b = torch.sin(a) 
297              c = torch.cos(a) 
298              d = b + c 
299              return d 
300   
301      and providing "operator_support" that indicates that 'b' and 'c' can be executed on the accelerator, 
302      we will get the following split result: 
303   
304      main: 
305      def forward(self, a): 
306          run_on_acc_0_0 = self._run_on_acc_0_0(a) 
307          getitem = run_on_acc_0_0[0] 
308          getitem_1 = run_on_acc_0_0[1] 
309          run_on_cpu_1_1 = self._run_on_cpu_1_1(getitem, getitem_1) 
310          return run_on_cpu_1_1 
311   
312      _run_on_acc_0_0: 
313      def forward(self, a): 
314          sin_1 = torch.sin(a) 
315          cos_1 = torch.cos(a) 
316          return (sin_1, cos_1) 
317   
318      _run_on_cpu_1_1: 
319      def forward(self, sin_1, cos_1): 
320          add_1 = sin_1 + cos_1 
321          return add_1 
322      """
323  
324      # PCIe bandwidth for the backend, default to 100 GB/s
325      PCIe_BW = 100 * 2**30
326  
327      def __init__(
328          self,
329          module: torch.fx.GraphModule,
330          sample_input: Sequence[Any],
331          operator_support: OperatorSupportBase,
332          settings: _SplitterSettingBase,
333          non_acc_submodule_name: str = "_run_on_cpu_",
334          return_tuple: bool = False,
335          nodes_finder: Optional[FxNetAccNodesFinder] = None,
336      ):
337          """ 
338          Preprocesses graph before splitting: 
339          - finds nodes supported by ACC, 
340          - finds fusion groups for ACC nodes having non-tensor IO, 
341          - builds a graph of direct dependencies, 
342          - builds a map of fused nodes to their fusions. 
343          As a result we get self.acc_nodes, self.deps and self.fusions. 
344          """
345          assert isinstance(module, torch.fx.GraphModule)
346  
347          self.module = module
348          ShapeProp(self.module).propagate(*sample_input)
349  
350          self.settings = settings
351          self.operator_support = operator_support
352          self.sample_input = sample_input
353          if nodes_finder is None:
354              nodes_finder = FxNetAccNodesFinder(
355                  self.module, self.operator_support, self.settings.allow_non_tensor
356              )
357          self.acc_nodes = nodes_finder()
358  
359          if self.settings.skip_fusion:
360              self.fusions = {}
361          else:
362              self.fusions = FxNetAccFusionsFinder(module, self.acc_nodes)()
363  
364          # Modify deps to add more deps for fused nodes
365          self.deps = self.find_deps()
366          self.update_deps_for_fusions()
367  
368          self.non_acc_submodule_name = non_acc_submodule_name
369          self._node_submodule_map: dict[str, str] = {}
370          self._return_tuple = return_tuple
371  
372          self.tags: list[str] = []
373

378      def get_node_submodule_map(self) -> dict[str, str]:
379          """Returns a map from node name to submodule name, e.g. 
380          node: main_module_impl_impl_over_arch_unary_multiple_embedding 
381            _pooling_embedding_pooling_sparse_entity_equivalence_key 
382            _proxy_embedding_bag 
383          maps to submodule name of: _run_on_acc_1 
384          """
385          return self._node_submodule_map
386  
387      def find_deps(self) -> dict[torch.fx.Node, NodeSet]:
388          """ 
389          Builds a graph of node dependencies. Leaf nodes don't have any 
390          dependencies and the "output" node doesn't have nodes depending on it. 
391   
392          Resulting graph has only direct dependencies, i.e. there are no 
393          transitive dependencies. 
394          """
395          deps: dict[torch.fx.Node, NodeSet] = defaultdict(set)
396          for node in self.module.graph.nodes:
397              if node.op not in CALLABLE_NODE_OPS:
398                  continue
399  
400              for user in node.users:
401                  if user.op != "output":
402                      deps[user].add(node)
403          return deps
404  
405      def update_deps_for_fusions(self):
406          """ 
407          Updates graph of dependencies so that: 
408          - nodes from the same fusion depend on the same set of outer nodes, 
409          - outer nodes depending on a fusion depend on all nodes in that fusion. 
410          """
411          for node in self.fusions:
412              fusion = self.fusions[node]
413              for fused_neighbor in fusion:
414                  self.deps[node].update(self.deps[fused_neighbor] - fusion)
415  
416                  for user in fused_neighbor.users:
417                      if user not in fusion:
418                          self.deps[user].add(node)

467      def node_support_preview(self, dump_graph: bool = False):
468          submodules = dict(self.module.named_modules())
469  
470          supported_nodes: NodeList = []
471          supported_node_types = defaultdict(set)
472          unsupported_node_types = defaultdict(set)
473  
474          def get_dtype(arg):
475              tensor_meta = arg.meta.get("tensor_meta")
476              return getattr(tensor_meta, "dtype", None)
477  
478          for node in self.module.graph.nodes:
479              if node.op not in CALLABLE_NODE_OPS:
480                  continue
481  
482              target = get_node_target(submodules, node)
483  
484              # Store dtype of arg in node.args. If arg doesn't have dtype, i.e. not a tensor, we'll store None.
485              arg_dtypes = [
486                  get_dtype(arg) if isinstance(arg, torch.fx.Node) else None
487                  for arg in node.args
488              ]
489  
490              # Find last non-None element. If all elements are None, return max_len.
491              last_index = len(arg_dtypes) - next(
492                  (
493                      i
494                      for i, dtype in enumerate(reversed(arg_dtypes))
495                      if dtype is not None
496                  ),
497                  len(arg_dtypes),
498              )
499  
500              # Strip None elements at the end.
501              arg_dtypes_tuple = tuple(arg_dtypes[:last_index])
502              kwarg_dtypes_tuple = tuple(
503                  (k, get_dtype(arg))
504                  for k, arg in node.kwargs.items()
505                  if isinstance(arg, torch.fx.Node)
506              )
507  
508              if self.operator_support.is_node_supported(submodules, node):
509                  supported_nodes.append(node)
510                  supported_node_types[target].add((arg_dtypes_tuple, kwarg_dtypes_tuple))
511              else:
512                  unsupported_node_types[target].add(
513                      (arg_dtypes_tuple, kwarg_dtypes_tuple)
514                  )
515  
516          if dump_graph:
517              self._draw_graph_based_on_node_support(self.module, supported_nodes)
518  
519          reports = "\nSupported node types in the model:\n"
520          for t, dtypes in supported_node_types.items():
521              for arg_dtypes_tuple, kwarg_dtypes_tuple in dtypes:
522                  reports += f"{t}: ({arg_dtypes_tuple}, {dict(kwarg_dtypes_tuple)})\n"
523  
524          reports += "\nUnsupported node types in the model:\n"
525          for t, dtypes in unsupported_node_types.items():
526              for arg_dtypes_tuple, kwarg_dtypes_tuple in dtypes:
527                  reports += f"{t}: ({arg_dtypes_tuple}, {dict(kwarg_dtypes_tuple)})\n"
528  
529          print(reports)
530  
531          # Return reports for testing purpose
532          return reports

534      def split_preview(self, dump_graph: bool = False):
535          reports = ""
536          subgraphs = self.put_nodes_into_subgraphs()
537          acc_subgraphs_num = len([g for g in subgraphs if g.is_acc])
538          cpu_subgraphs_num = len(subgraphs) - acc_subgraphs_num
539          reports += f"Before removing small acc subgraphs, total {len(subgraphs)} subgraphs are created:"
540          reports += f" {acc_subgraphs_num} acc subgraphs and {cpu_subgraphs_num} cpu subgraphs.\n"
541  
542          subgraphs = self.remove_small_acc_subgraphs(subgraphs)
543          acc_subgraphs_num = len([g for g in subgraphs if g.is_acc])
544          cpu_subgraphs_num = len(subgraphs) - acc_subgraphs_num
545          reports += f"After removing small acc subgraphs, total {len(subgraphs)} subgraphs are created:"
546          reports += f" {acc_subgraphs_num} acc subgraphs and {cpu_subgraphs_num} cpu subgraphs.\n"
547  
548          for i, subgraph in enumerate(subgraphs):
549              reports += (
550                  f"_run_on_acc_{i}: "
551                  if subgraph.is_acc
552                  else f"{self.non_acc_submodule_name}{i}: "
553              )
554              reports += f"{len(subgraph.nodes)} node(s)\n"
555  
556          self.tag(subgraphs)
557          split_mod = self.split(remove_tag=True)
558          split_mod.eval()
559  
560          if dump_graph:
561              drawer = FxGraphDrawer(split_mod, "preview", ignore_getattr=True)
562              dot_graphs = drawer.get_all_dot_graphs()
563              for name, dot_graph in dot_graphs.items():
564                  # pyre-fixme[16]: `pydot.Dot` has no attribute `write_raw`.
565                  dot_graph.write_raw(f"{name}.dot")  # type: ignore[attr-defined]
566  
567          max_qps: float = self.PCIe_BW
568          bottleneck_module = ""
569  
570          for node in split_mod.graph.nodes:
571              if node.op == "call_module" and "acc" in node.target:
572                  reports += f"\nProcessing acc submodule {node.target}\n"
573  
574                  submod = getattr(split_mod, node.target)
575  
576                  def get_submod_inputs(main_mod, submod, example_inputs):
577                      sub_inputs = None
578  
579                      def get_inputs(self, inputs):
580                          nonlocal sub_inputs
581                          sub_inputs = inputs
582  
583                      handle = submod.register_forward_pre_hook(get_inputs)
584                      main_mod(*example_inputs)
585                      handle.remove()
586                      return sub_inputs
587  
588                  submod_inputs = get_submod_inputs(split_mod, submod, self.sample_input)
589                  ShapeProp(submod).propagate(*submod_inputs)
590  
591                  total_input_bytes = 0
592                  total_output_bytes = 0
593  
594                  reports += "Checking inputs...\n"
595                  for n in submod.graph.nodes:
596                      if n.op == "placeholder":
597                          if not is_node_output_tensor(n):
598                              reports += f"Input {n.name} is not a tensor, this might cause problems during lowering!\n"
599                          else:
600                              total_input_bytes += get_size_of_node(submod, n)[0]
601                      if n.op == "output":
602                          output_node = n
603  
604                  reports += "Checking outputs...\n"
605  
606                  def get_bytes(node: torch.fx.Node):
607                      nonlocal total_output_bytes
608                      nonlocal reports
609                      if not is_node_output_tensor(node):
610                          reports += f"Output {node.name} is not a tensor, this might cause problems during lowering!\n"
611                      else:
612                          total_output_bytes += get_size_of_node(submod, node)[0]
613  
614                  map_arg(output_node.args, get_bytes)  # type: ignore[possibly-undefined]
615                  qps = self.PCIe_BW / max(total_input_bytes, total_output_bytes)
616                  reports += f"Total input size in bytes is {total_input_bytes}, total output size in bytes is {total_output_bytes},"
617                  reports += f" theoretical max qps (bounds by PCIe bandwidth) for this submodule is {qps}.\n"
618  
619                  if qps < max_qps:
620                      max_qps = qps
621                      bottleneck_module = node.target
622  
623                  try:
624                      lowered_submod = self._lower_model_to_backend(submod, submod_inputs)
625                  except RuntimeError:
626                      reports += "Run into an error during lowering!\n"
627                      reports += self._find_culprit(submod, submod_inputs)
628                      continue
629  
630                  try:
631                      lowered_submod(*submod_inputs)
632                  except RuntimeError:
633                      reports += "Run into an error during inference!\n"
634                      reports += self._find_culprit(submod, submod_inputs)
635                  else:
636                      reports += "Lowering and running succeed!\n"
637  
638          reports += f"\nTheoretical max qps (bounds by PCIe bandwidth) for this model is {max_qps},"
639          reports += f" bottleneck is submodule {bottleneck_module}."
640          print(reports)
641  
642          # return the reports for testing purposes
643          return reports

649      def find_reverse_deps(
650          self, tag_id: Optional[int] = None
651      ) -> dict[torch.fx.Node, NodeSet]:
652          """ 
653          Builds reversed topological node dependencies, if tag_id is specified, 
654          we ignore nodes that are in later subgraph i.e. nodes have greater tag_id. 
655          """
656          result: dict[torch.fx.Node, NodeSet] = defaultdict(set)
657  
658          for node in self.module.graph.nodes:
659              if node.op not in CALLABLE_NODE_OPS:
660                  continue
661  
662              for user in node.users:
663                  if user.op not in CALLABLE_NODE_OPS:
664                      continue
665  
666                  if tag_id is None or (int(user.tag.split("_")[-1]) < tag_id):
667                      result[node].add(user)
668  
669          return result
670  
671      def update_reverse_deps_for_fusions(self, deps: dict[torch.fx.Node, NodeSet]):
672          processed_node = set()
673  
674          for node, fusion in self.fusions.items():
675              if node in processed_node:
676                  continue
677  
678              new_dep = set()
679  
680              # Create a new dependency set which include all the
681              # dependencies of the nodes in the fusion group
682              for n in fusion:
683                  new_dep.update(deps[n])
684  
685              # Exclude nodes in the fusion
686              new_dep.difference_update(fusion)
687  
688              # Update dependency
689              for n in fusion:
690                  deps[n] = new_dep
691  
692                  for arg in n.all_input_nodes:
693                      if arg not in fusion:
694                          deps[arg].update(fusion)
695  
696                  processed_node.add(n)
697  
698      def find_parent_nodes_of_subgraph(self, tag: str) -> NodeSet:
699          """ 
700          Finds parent nodes of the `tag` subgraph. 
701   
702          Traverse the inputs of nodes in the subgraph, if input doesn't belong to the subgraph 
703          and is not a placeholder, we consider it as the parent node of the subgraph. 
704          """
705          parent_nodes = set()
706  
707          for node in self.module.graph.nodes:
708              if node.op in CALLABLE_NODE_OPS and node.tag == tag:
709                  for arg in node.all_input_nodes:
710                      if arg.op in CALLABLE_NODE_OPS and arg.tag != tag:
711                          parent_nodes.add(arg)
712  
713          return parent_nodes
714  
715      def extend_acc_subgraph(self, tag: str):
716          """ 
717          Extend the acc subgraph with `tag` going the reversed topological direction. 
718          """
719          # Dict that maps node to its users and ignore users that
720          # are in the subgraph that has greater tag
721          deps = self.find_reverse_deps(tag_id=int(tag.split("_")[-1]))
722          self.update_reverse_deps_for_fusions(deps)
723  
724          # Parent nodes of the subgraph
725          parent_nodes = self.find_parent_nodes_of_subgraph(tag)
726  
727          visited_nodes: NodeSet = set()
728  
729          while parent_nodes:
730              node = None
731  
732              # Find a acc node that depends on visited nodes only
733              for n in parent_nodes:
734                  if deps[n] <= visited_nodes and n in self.acc_nodes:
735                      node = n
736                      break
737  
738              if node is None:
739                  break
740  
741              # Put the node into `tag` subgraph
742              node.tag = tag  # type: ignore[attr-defined]
743              parent_nodes.remove(node)
744              visited_nodes.add(node)
745  
746              # If node is in a fusion group, add all fusion buddies to parent nodes
747              if node in self.fusions:
748                  for fusion_node in self.fusions[node]:
749                      if fusion_node not in visited_nodes:
750                          parent_nodes.add(fusion_node)
751  
752              # Add inputs of the node to parent nodes
753              for arg in node.all_input_nodes:
754                  if arg.op in CALLABLE_NODE_OPS and arg not in visited_nodes:
755                      parent_nodes.add(arg)

761      def starter_nodes(self) -> tuple[NodeSet, NodeSet]:
762          """ 
763          Finds nodes that consume module inputs or get_attr nodes. 
764          """
765          starter_cpu_nodes: NodeSet = set()
766          starter_acc_nodes: NodeSet = set()
767          for node in self.module.graph.nodes:
768              if node.op not in {"placeholder", "get_attr"}:
769                  continue
770              for user in node.users:
771                  if user in self.acc_nodes:
772                      starter_acc_nodes.add(user)
773                  else:
774                      starter_cpu_nodes.add(user)
775          return starter_cpu_nodes, starter_acc_nodes
776  
777      def put_nodes_into_subgraphs(self) -> list[Subgraph]:
778          # We start graph traversal from leaf nodes
779          current_cpu_nodes, current_acc_nodes = self.starter_nodes()
780          visited_nodes: NodeSet = set()
781  
782          # Determine which subgraph to start from based on which subgraph has
783          # 0-dep node
784          acc_subgraph: bool = not any(len(self.deps[n]) == 0 for n in current_cpu_nodes)
785  
786          current_subgraph_nodes: NodeList = []
787  
788          # Result accumulator
789          subgraphs: list[Subgraph] = []
790          while current_cpu_nodes or current_acc_nodes:
791              # Find the first node that should belong to the current subgraph and has all dependencies resolved
792              current_nodes = current_acc_nodes if acc_subgraph else current_cpu_nodes
793              node = next(
794                  (n for n in current_nodes if self.deps[n] <= visited_nodes),
795                  None,
796              )
797  
798              # If nothing was found, then it's time to flip the mode and start a new subgraph
799              if node is None:
800                  if not current_subgraph_nodes:
801                      raise FxNetSplitterInternalError("Subgraph can't be empty")
802  
803                  subgraphs.append(
804                      Subgraph(is_acc=acc_subgraph, nodes=current_subgraph_nodes)
805                  )
806                  acc_subgraph = not acc_subgraph
807                  current_subgraph_nodes = []
808                  continue
809  
810              current_nodes.remove(node)
811              visited_nodes.add(node)
812              current_subgraph_nodes.append(node)
813  
814              # Add fusion buddies
815              if node in self.fusions:
816                  if node in self.acc_nodes:
817                      current_acc_nodes.update(self.fusions[node] - visited_nodes)
818                  else:
819                      current_cpu_nodes.update(self.fusions[node] - visited_nodes)
820  
821              # Put depending nodes into the queue
822              for user in node.users:
823                  if user.op not in CALLABLE_NODE_OPS:
824                      continue
825  
826                  # Add downstream nodes
827                  if user in self.acc_nodes:
828                      current_acc_nodes.add(user)
829                  else:
830                      current_cpu_nodes.add(user)
831  
832          # Check if the last subgraph was not created
833          if current_subgraph_nodes:
834              subgraphs.append(
835                  Subgraph(is_acc=acc_subgraph, nodes=current_subgraph_nodes)
836              )
837  
838          if not subgraphs:
839              raise FxNetSplitterInternalError("Couldn't create subgraphs")
840  
841          return subgraphs
842  
843      def remove_small_acc_subgraphs(self, subgraphs: list[Subgraph]) -> list[Subgraph]:
844          """ 
845          This pass finds ACC submodules with less than specified size and merges 
846          them with adjacent CPU submodules. 
847          """
848          result: list[Subgraph] = []
849          for subgraph in subgraphs:
850              if subgraph.is_acc:
851                  if len(subgraph.nodes) >= self.settings.min_acc_module_size:
852                      result.append(subgraph)
853                  else:
854                      print(
855                          "Eliminating acc subgraph because it's smaller than the threshold: "
856                          f"{len(subgraph.nodes)} < {self.settings.min_acc_module_size}"
857                      )
858                      if result:
859                          result[-1].nodes.extend(subgraph.nodes)
860                      else:
861                          subgraph.is_acc = False
862                          result.append(subgraph)
863              else:
864                  if result and not result[-1].is_acc:
865                      result[-1].nodes.extend(subgraph.nodes)
866                  else:
867                      result.append(subgraph)
868          return result

870      def tag(self, subgraphs: list[Subgraph]):
871          self.tags = []
872          for subgraph in subgraphs:
873              tag = (
874                  f"_run_on_acc_{len(self.tags)}"
875                  if subgraph.is_acc
876                  else f"{self.non_acc_submodule_name}{len(self.tags)}"
877              )
878              self.tags.append(tag)
879              for node in subgraph.nodes:
880                  if hasattr(node, "tag"):
881                      raise FxNetSplitterInternalError(f"Node {node} was already tagged")
882  
883                  node.tag = tag  # type: ignore[attr-defined]
884                  self._node_submodule_map[node.name] = tag
885  
886      def split(self, remove_tag: bool = False) -> torch.fx.GraphModule:
887          split_module = split_by_tags(
888              self.module, self.tags, return_tuple=self._return_tuple
889          )
890          if remove_tag:
891              for node in self.module.graph.nodes:
892                  if hasattr(node, "tag"):
893                      del node.tag
894          return split_module  # type: ignore[return-value]
895  
896      def __call__(self) -> torch.fx.GraphModule:
897          subgraphs = self.put_nodes_into_subgraphs()
898          subgraphs = self.remove_small_acc_subgraphs(subgraphs)
899          acc_subgraphs_count = len([s for s in subgraphs if s.is_acc])
900          non_acc_subgraphs_count = len(subgraphs) - acc_subgraphs_count
901          print(
902              f"Got {acc_subgraphs_count} acc subgraphs and {non_acc_subgraphs_count} non-acc subgraphs"
903          )
904          self.tag(subgraphs)
905          return self.split()

907      def generate_split_results(self) -> SplitResult:
908          split_module = self()
909          submodule_names = []
910          for name, _mod in split_module.named_children():
911              submodule_names.append(name)
912          if (
913              self.settings.max_acc_splits > 0
914              and len(submodule_names) > self.settings.max_acc_splits
915          ):
916              raise ValueError(
917                  "Cannot fulfill max_acc_splits limit. "
918                  "This may cause split fragmentation and "
919                  "result in performance issues."
920              )
921  
922          submodule_inputs = generate_inputs_for_submodules(
923              split_module, self.sample_input, submodule_names
924          )
925          return SplitResult(split_module, submodule_inputs, self.non_acc_submodule_name)
