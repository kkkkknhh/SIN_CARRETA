3     */

10   namespace xsf {
11   namespace cephes {
12   
13       namespace detail {
14   
15           XSF_HOST_DEVICE inline double is_nonpos_int(double x) {
16               return x <= 0 && x == std::ceil(x) && std::abs(x) < 1e13;
17           }
18       } // namespace detail
19   
20       XSF_HOST_DEVICE inline double poch(double a, double m) {
21           double r = 1.0;
22   
23           /* 
24            * 1. Reduce magnitude of `m` to |m| < 1 by using recurrence relations. 
25            * 
26            * This may end up in over/underflow, but then the function itself either 
27            * diverges or goes to zero. In case the remainder goes to the opposite 
28            * direction, we end up returning 0*INF = NAN, which is OK. 
29            */
30   
31           /* Recurse down */
32           while (m >= 1.0) {
33               if (a + m == 1) {
34                   break;
35               }
36               m -= 1.0;
37               r *= (a + m);
38               if (!std::isfinite(r) || r == 0) {
39                   break;
40               }
41           }
42   
43           /* Recurse up */
44           while (m <= -1.0) {
45               if (a + m == 0) {
46                   break;
47               }
48               r /= (a + m);
49               m += 1.0;
50               if (!std::isfinite(r) || r == 0) {
51                   break;
52               }
53           }
54   
55           /* 
56            * 2. Evaluate function with reduced `m` 
57            * 
58            * Now either `m` is not big, or the `r` product has over/underflown. 
59            * If so, the function itself does similarly. 
60            */
61   
62           if (m == 0) {
63               /* Easy case */
64               return r;
65           } else if (a > 1e4 && std::abs(m) <= 1) {
66               /* Avoid loss of precision */
67               return r * std::pow(a, m) *
68                      (1 + m * (m - 1) / (2 * a) + m * (m - 1) * (m - 2) * (3 * m - 1) / (24 * a * a) +
69                       m * m * (m - 1) * (m - 1) * (m - 2) * (m - 3) / (48 * a * a * a));
70           }
71   
72           /* Check for infinity */
73           if (detail::is_nonpos_int(a + m) && !detail::is_nonpos_int(a) && a + m != m) {
74               return std::numeric_limits<double>::infinity();
75           }
76   
77           /* Check for zero */
78           if (!detail::is_nonpos_int(a + m) && detail::is_nonpos_int(a)) {
79               return 0;
80           }
81   
82           return r * std::exp(lgam(a + m) - lgam(a)) * gammasgn(a + m) * gammasgn(a);
83       }
