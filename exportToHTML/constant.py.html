27   class ConstantVariable(VariableTracker):
28       """ 
29       Variable tracker for Python literals and basic immutable types, with automatic 
30       routing support for collection types (lists, tuples, sets, etc.). 
31    
32       The create() method intelligently constructs appropriate variable types for 
33       nested collections. 
34       """
35   
36       @staticmethod
37       def create(value, **kwargs) -> VariableTracker:
38           """ 
39           Create a `ConstantVariable` based on the given value, and supports 
40           automatic routing for collection types like `tuple` (in which case we'd 
41           create `ConstantVariable` for the leaf items). 
42    
43           NOTE: the caller must install the proper guards if needed; most often 
44           the guard will be `CONSTANT_MATCH`. 
45           """
46           source = kwargs.get("source", None)
47   
48           # Routing for supported collection literals.
49           if isinstance(value, set):
50               items = [ConstantVariable.create(x) for x in value]
51               return variables.SetVariable(items, **kwargs)
52           elif isinstance(value, frozenset):
53               items = [ConstantVariable.create(x) for x in value]
54               return variables.FrozensetVariable(items, **kwargs)
55           elif isinstance(value, (list, tuple)):
56               items = []
57               for i, x in enumerate(value):
58                   item_source = GetItemSource(source, i) if source else None
59                   items.append(
60                       ConstantVariable.create(
61                           x,
62                           source=item_source,
63                       )
64                   )
65               return variables.BaseListVariable.cls_for(type(value))(items, **kwargs)
66   
67           return ConstantVariable(value, **kwargs)
68   
69       def __init__(self, value, **kwargs) -> None:
70           super().__init__(**kwargs)
71           assert ConstantVariable.is_base_literal(value), f"""
72   Cannot construct `ConstantVariable` for value of type {type(value)}.
73   
74   This failure likely due to PyTorch-internal use of `ConstantVariable` on
75   non-literal python values, please try using `VariableTracker.build` instead. If
76   you believe it's a necessary and legitimate use case (the value is immutable and
77   can't easily be represented with another `VariableTracker` class), please add
78   its type to `common_constant_types`.
79   """
80           if np is not None and isinstance(value, np.number):
81               self.value = value.item()
82           else:
83               self.value = value
84   
85       def as_proxy(self):
86           return self.value
87   
88       def __repr__(self) -> str:
89           return f"ConstantVariable({type(self.value).__name__}: {repr(self.value)})"
90   
91       def as_python_constant(self):
92           return self.value
93   
94       def is_python_constant(self):
95           return True
96   
97       @property
98       def items(self):
99           """ 
100          Need this when adding a BaseListVariable and a ConstantVariable together. 
101          Happens in detectron2. 
102          """
103          return self.unpack_var_sequence(tx=None)
104  
105      def getitem_const(self, tx: "InstructionTranslator", arg: VariableTracker):
106          return ConstantVariable.create(
107              self.value[arg.as_python_constant()],
108          )
109  
110      @staticmethod
111      def is_base_literal(obj):
112          return type(obj) in common_constant_types
113  
114      @staticmethod
115      def is_literal(obj):
116          if type(obj) in (list, tuple, set, frozenset, torch.Size):
117              return all(ConstantVariable.is_literal(x) for x in obj)
118          return ConstantVariable.is_base_literal(obj)
119  
120      def unpack_var_sequence(self, tx):
121          try:
122              return [ConstantVariable.create(x) for x in self.as_python_constant()]
123          except TypeError as e:
124              raise NotImplementedError from e
125  
126      def const_getattr(self, tx: "InstructionTranslator", name):
127          if not hasattr(self.value, name):
128              raise NotImplementedError
129          member = getattr(self.value, name)
130          if callable(member):
131              raise NotImplementedError
132          return member
133  
134      def call_method(
135          self,
136          tx: "InstructionTranslator",
137          name,
138          args: "list[VariableTracker]",
139          kwargs: "dict[str, VariableTracker]",
140      ) -> "VariableTracker":
141          from .tensor import SymNodeVariable
142  
143          if name == "format" and istype(self.value, str):
144              return variables.BuiltinVariable(str.format).call_function(
145                  tx, [self, *args], kwargs
146              )
147          elif name == "join" and istype(self.value, str):
148              assert len(args) == 1 and len(kwargs) == 0
149              arg_unpacked = args[0].force_unpack_var_sequence(tx)
150              try:
151                  arg_const = [x.as_python_constant() for x in arg_unpacked]
152                  return ConstantVariable.create(self.value.join(arg_const))
153              except NotImplementedError:
154                  return super().call_method(tx, name, args, kwargs)
155  
156          if any(isinstance(x, SymNodeVariable) for x in args):
157              # Promote to SymNodeVariable for operations involving dynamic shapes.
158              return variables.SymNodeVariable(self.as_proxy(), self.value).call_method(
159                  tx, name, args, kwargs
160              )
161  
162          try:
163              const_args = [a.as_python_constant() for a in args]
164              const_kwargs = {k: v.as_python_constant() for k, v in kwargs.items()}
165          except NotImplementedError:
166              return super().call_method(tx, name, args, kwargs)
167  
168          if isinstance(self.value, str) and name in str.__dict__.keys():
169              method = getattr(self.value, name)
170              try:
171                  return ConstantVariable.create(method(*const_args, **const_kwargs))
172              except Exception as e:
173                  raise_observed_exception(type(e), tx)
174          elif isinstance(self.value, (float, int)):
175              if not (args or kwargs):
176                  return ConstantVariable.create(getattr(self.value, name)())
177              if (
178                  hasattr(operator, name)
179                  and len(args) == 1
180                  and args[0].is_python_constant()
181              ):
182                  add_target = const_args[0]
183                  op = getattr(operator, name)
184                  if isinstance(
185                      add_target, (torch.SymBool, torch.SymFloat, torch.SymInt)
186                  ):
187                      # Addition between a non sym and sym makes a sym
188                      proxy = tx.output.create_proxy(
189                          "call_function", op, (self.value, add_target), {}
190                      )
191                      return SymNodeVariable.create(tx, proxy, add_target)
192                  else:
193                      try:
194                          return ConstantVariable.create(op(self.value, add_target))
195                      except Exception as e:
196                          raise_observed_exception(
197                              type(e), tx, args=list(map(ConstantVariable.create, e.args))
198                          )
199          elif isinstance(self.value, bytes) and name == "decode":
200              method = getattr(self.value, name)
201              return ConstantVariable.create(method(*const_args, **const_kwargs))
202  
203          if name == "__len__" and not (args or kwargs):
204              return ConstantVariable.create(len(self.value))
205          elif name == "__round__" and len(args) == 1 and args[0].is_python_constant():
206              return ConstantVariable.create(
207                  round(self.value, args[0].as_python_constant())
208              )
209          elif name == "__contains__" and len(args) == 1 and args[0].is_python_constant():
210              assert not kwargs
211              search = args[0].as_python_constant()
212              result = search in self.value
213              return ConstantVariable.create(result)
214          return super().call_method(tx, name, args, kwargs)
215  
216      def call_obj_hasattr(
217          self, tx: "InstructionTranslator", name: str
218      ) -> "VariableTracker":
219          result = hasattr(self.value, name)
220          return variables.ConstantVariable.create(result)

223  class EnumVariable(VariableTracker):
224      """VariableTracker for enum.Enum and enum.IntEnum instances 
225   
226      Provides specialized handling for Python enum types, supporting 
227      both standard Enum and IntEnum with proper value tracking and comparison. 
228      """
229  
230      def __init__(self, value, **kwargs) -> None:
231          super().__init__(**kwargs)
232          self.value = value
233  
234      @classmethod
235      def create(cls, cls_type, value_vt, options):
236          if isinstance(value_vt, variables.ConstantVariable):
237              for member in list(cls_type):
238                  if member.value == value_vt.as_python_constant():
239                      return cls(member, **options)
240          unimplemented_v2(
241              gb_type="Failed to construct Enum variable",
242              context=f"value: {value_vt}, allowed enum values: {list(cls_type)}",
243              explanation="Attempted to construct an Enum value that is non-constant (e.g. int, string) "
244              "or is not an acceptable value for the Enum. "
245              f"Acceptable values for Enum `{cls_type}`: {list(cls_type)}.",
246              hints=[*graph_break_hints.USER_ERROR, *graph_break_hints.SUPPORTABLE],
247          )
248  
249      def as_proxy(self):
250          if isinstance(self.value, int):
251              return int(self.value)  # convert IntEnum to a normal int
252          return self.value
253  
254      def __repr__(self) -> str:
255          return f"EnumVariable({type(self.value)})"
256  
257      def as_python_constant(self):
258          return self.value
259  
260      def var_getattr(self, tx: "InstructionTranslator", name):
261          if not hasattr(self.value, name):
262              raise NotImplementedError
263          if name in cmp_name_to_op_mapping:
264              return variables.GetAttrVariable(self, name)
265          member = getattr(self.value, name)
266          source = self.source and AttrSource(self.source, name)
267          return VariableTracker.build(tx, member, source=source)
