56   def register_transformation_rule(call_target):
57       def register(fn):
58           if call_target in _TRANSFORMATION_RULES:
59               raise RuntimeError(
60                   f"Transformation rule already registered for {call_target}!"
61               )
62           _TRANSFORMATION_RULES[call_target] = fn
63           return fn
64   
65       return register

68   def valid_index(index, dims):
69       """ 
70       Given a list of dimensions, checks if an index is valid in the list 
71       """
72       try:
73           dims[index]
74           return T()
75       except IndexError:
76           return F()

79   @register_transformation_rule(Transpose)
80   def transform_transpose(constraint, counter):
81       """ 
82       Similar to a sequence of two index-selects 
83       """
84       dims, counter = gen_tensor_dims(constraint.tensor_size, counter)
85       is_valid_index1 = valid_index(constraint.index1, dims)
86       is_valid_index2 = valid_index(constraint.index2, dims)
87       new_dims = copy.deepcopy(dims)
88       nat_constraints = gen_nat_constraints(dims)
89   
90       if is_valid_index1 == T() and is_valid_index2 == T():
91           new_dims[constraint.index1] = dims[constraint.index2]
92           new_dims[constraint.index2] = dims[constraint.index1]
93   
94       transformed_constraint = Conj(
95           [
96               BinConstraintT(constraint.input_var, TensorType(dims), op_eq),
97               *nat_constraints,
98               is_valid_index1,
99               is_valid_index2,
100              BinConstraintT(constraint.output, TensorType(new_dims), op_eq),
101          ]
102      )
103      return transformed_constraint, counter

106  @register_transformation_rule(IndexSelect)
107  def transform_index_select(constraint, counter):
108      """ 
109      The constraints consider the given tensor size, checks if the index is valid 
110      and if so, generates a constraint for replacing the input dimension 
111      with the required dimension 
112      """
113      dims, counter = gen_tensor_dims(constraint.tensor_size, counter)
114      is_valid_index = valid_index(constraint.index, dims)
115      nat_constraints = gen_nat_constraints(dims)
116  
117      # if the index is valid then replace the input dimension with the new dimension
118      # otherwise the dimension will not be replaced and the clause will contain False
119      if is_valid_index == T():
120          new_dims = copy.deepcopy(dims)
121          new_dims[constraint.index] = constraint.dim_replace
122  
123      transformed_constraint = Conj(
124          [
125              BinConstraintT(constraint.input_var, TensorType(dims), op_eq),
126              *nat_constraints,
127              is_valid_index,
128              BinConstraintT(constraint.output, TensorType(new_dims), op_eq),
129          ]
130      )
131  
132      # print(constraints)
133      return transformed_constraint, counter

136  @register_transformation_rule(GetItem)
137  def transform_get_item(constraint, counter):
138      """ 
139      generate an equality of the form: 
140      t = [a1, ..., an] 
141      then generate constraints that check if the given index is valid 
142      given this particular tensor size. 
143      If the index is valid, generate a constraint to get the item 
144      Note that we already handled the Dyn input case in the previous 
145      step. 
146      Args: 
147          constraint: GetItem which assumes we are getting an item from a tensor (not Dyn) 
148          counter: variable tracking 
149      Returns: simplified constraints for GetItem 
150   
151      """
152      dims, counter = gen_tensor_dims(constraint.tensor_size, counter)
153      nat_constraints = gen_nat_constraints(dims)
154  
155      is_valid_index = valid_index(constraint.index, dims)
156  
157      all_constraints = [
158          BinConstraintT(constraint.input_var, TensorType(dims), op_eq),
159          *nat_constraints,
160          is_valid_index,
161      ]
162  
163      # if the index is valid, we generate a constraint for getting an item
164      # otherwise this clause will have been UNSAT due to the wrong index
165      if is_valid_index == T():
166          all_constraints.append(
167              BinConstraintD(constraint.res, dims[constraint.index], op_eq)
168          )
169  
170      return Conj(all_constraints), counter

173  def valid_index_tensor(index, dims):
174      """ 
175      if the slice instances exceed the length of the dimensions 
176      then this is a type error so we return False 
177      """
178      slice_count = 0
179      for s in index:
180          if isinstance(s, slice):
181              slice_count += 1
182      if slice_count > len(dims):
183          return F()
184      else:
185          return T()

188  @register_transformation_rule(GetItemTensor)
189  def transform_get_item_tensor(constraint, counter):
190      """ 
191      When the index is a tuple, then the output will be a tensor 
192      TODO: we have to check if this is the case for all HF models 
193   
194      The cases we are covering here are a tuple with one of: 
195       - slice with default argument 
196       - None 
197   
198       None appends 1 to the input tensor dimensions 
199       so each occurrence of 'None' increases the rank by 1 
200   
201       slice with default arguments does not change the rank 
202      """
203      assert isinstance(constraint.index_tuple, tuple)
204  
205      # generate a result tensor of the expected size
206      dims, counter = gen_tensor_dims(constraint.tensor_size, counter)
207      nat_constraints = gen_nat_constraints(dims)
208  
209      # generate a place-holder list of the right rank
210      # where "slice" does not contribute to the rank and "None" does
211      none_c = constraint.index_tuple.count(None)
212      resulting_tensor_dims = (none_c + len(dims)) * [None]
213  
214      dim_index = 0
215      for i in range(len(constraint.index_tuple)):
216          # append 1 to the right location of the resulting tensor
217          if constraint.index_tuple[i] is None:
218              resulting_tensor_dims[i] = 1
219  
220          elif constraint.index_tuple[i] == slice(None, None, None):
221              pass
222  
223          else:
224              raise NotImplementedError("Method not yet implemented")
225  
226      # append the remaining dimensions to the right location
227      dim_index = 0
228      for i in range(len(resulting_tensor_dims)):
229          if resulting_tensor_dims[i] is None:
230              resulting_tensor_dims[i] = dims[dim_index]
231              dim_index += 1
232  
233      # check if the index is valid
234      is_valid_index = valid_index_tensor(constraint.index_tuple, dims)
235  
236      # check if the resulting tensor is within bounds
237      if len(resulting_tensor_dims) > 4:
238          return F(), counter
239  
240      else:
241          constraints = [
242              BinConstraintT(constraint.input_var, TensorType(dims), op_eq),
243              BinConstraintT(constraint.res, TensorType(resulting_tensor_dims), op_eq),
244              *nat_constraints,
245              is_valid_index,
246          ]
247          return Conj(constraints), counter

250  @register_transformation_rule(BinConstraintT)
251  def generate_binconstraint_t(constraint, counter):
252      """ 
253      Transform binary constraints for tensors 
254      """
255  
256      # precision constraints
257      if constraint.op == op_precision:
258          if constraint.lhs == Dyn:
259              return T(), counter
260          elif isinstance(constraint.lhs, TensorType):
261              is_fully_static = all(d != Dyn for d in constraint.lhs.__args__)
262              if is_fully_static:
263                  return BinConstraintT(constraint.lhs, constraint.rhs, op_eq), counter
264              else:
265                  new_dims = []
266  
267                  for _ in range(len(constraint.lhs.__args__)):
268                      dim, counter = gen_dvar(counter)
269                      new_dims.append(dim)
270  
271                  new_dim_constraints = (
272                      [
273                          BinConstraintD(old_dim, new_dim, op_precision)
274                          for new_dim, old_dim in zip(new_dims, constraint.lhs.__args__)
275                      ]
276                      + [BinConstraintT(constraint.rhs, TensorType(new_dims), op_eq)]
277                      + [BinConstraintD(1, new_dim, op_leq) for new_dim in new_dims]
278                  )
279                  return Conj(new_dim_constraints), counter
280  
281      # matching
282      elif constraint.op == op_matching:
283          assert isinstance(constraint.rhs, TensorType)
284          d1 = constraint.rhs.__args__[0]
285          d2 = constraint.rhs.__args__[1]
286          d3 = constraint.rhs.__args__[2]
287          d4 = constraint.rhs.__args__[3]
288  
289          conj = [
290              BinConstraintT(constraint.lhs, Dyn, op_eq),
291              BinConstraintD(d1, Dyn, op_eq),
292              BinConstraintD(d2, Dyn, op_eq),
293              BinConstraintD(d3, Dyn, op_eq),
294              BinConstraintD(d4, Dyn, op_eq),
295          ]
296          return (
297              Disj(
298                  [
299                      Conj(conj),
300                      BinConstraintT(constraint.lhs, TensorType([d1, d2, d3, d4]), op_eq),
301                  ]
302              ),
303              counter,
304          )
305  
306      elif constraint.op == op_consistency:
307          c_dyn = Disj(
308              [
309                  BinConstraintT(constraint.lhs, Dyn, op_eq),
310                  BinConstraintT(constraint.rhs, Dyn, op_eq),
311              ]
312          )
313          (
314              (
315                  c_tensor_1,
316                  c_tensor_2,
317                  c_tensor_3,
318                  c_tensor_4,
319              ),
320              counter,
321          ) = gen_consistency_constraints(constraint, counter)
322  
323          return Disj([c_dyn, c_tensor_1, c_tensor_2, c_tensor_3, c_tensor_4]), counter
324  
325      elif constraint.op == op_leq:
326          assert isinstance(constraint.rhs, int)
327          disj = [BinConstraintT(constraint.lhs, Dyn, op_eq)]
328          for i in range(1, constraint.rhs + 1):
329              dims = []
330              for _ in range(1, i + 1):
331                  dim_var, counter = gen_dvar(counter)
332                  dims.append(dim_var)
333              disj.append(BinConstraintT(constraint.lhs, TensorType(dims), op_eq))
334          return Disj(disj), counter
335      else:
336          return constraint, counter

339  @register_transformation_rule(BinConstraintD)
340  def generate_binconstraint_d(constraint, counter):
341      """ 
342      Transform binary constraints for dimensions 
343      """
344      if constraint.op == op_precision:
345          if isinstance(constraint.lhs, int):
346              return BinConstraintD(constraint.lhs, constraint.rhs, op_eq), counter
347          elif constraint.lhs == Dyn:
348              return T(), counter
349  
350      elif constraint.op == op_consistency:
351          return (
352              Disj(
353                  [
354                      BinConstraintD(constraint.lhs, constraint.rhs, op_eq),
355                      BinConstraintD(constraint.rhs, Dyn, op_eq),
356                      BinConstraintD(constraint.lhs, Dyn, op_eq),
357                  ]
358              ),
359              counter,
360          )
361  
362      else:
363          return constraint, counter

367  def generate_conj(constraint, counter):
368      """ 
369      Transform conjunctions 
370      """
371      new = []
372      for c in constraint.conjucts:

374          new.append(new_c)

415      """

439  @register_transformation_rule(CalcConv)
440  def generate_calc_conv(constraint, counter):
441      d, counter = gen_tensor_dims(4, counter)
442      conv_result = TensorType([d[0], d[1], d[2], d[3]])
443  
444      # the convolution result is a tensor of size 4
445      c1 = BinConstraintT(constraint.conv_result, conv_result, op_eq)
446  
447      # the second dimension of the output is equal to the output channels
448      c2 = Conj(
449          [
450              BinConstraintD(d[1], constraint.c_out, op_eq),
451              BinConstraintD(d[1], Dyn, op_neq),
452          ]
453      )
454  
455      # the input corresponds to the output in the first dimension of the convolution
456      c3 = BinConstraintD(constraint.matching_constraint[0], d[0], op_eq)
457  
458      c4, c5 = calc_last_two_dims(constraint, d)
459  
460      leq_constraints = Conj(
461          [
462              BinConstraintD(0, d[0], op_leq),
463              BinConstraintD(0, d[1], op_leq),
464              BinConstraintD(0, d[2], op_leq),
465              BinConstraintD(0, d[3], op_leq),
466          ]
467      )
468  
469      return Conj([c1, c2, c3, c4, c5, leq_constraints]), counter

500  @register_transformation_rule(CalcProduct)
501  def generate_calc_product(constraint, counter):
502      """ 
503      Transform flatten constraints 
504      """
505      start = constraint.start
506      end = constraint.end
507      dims = constraint.dims_to_flatten
508      flattened = constraint.flattened
509      n = len(constraint.dims_to_flatten)
510  
511      # this will be evaluated right here
512      boundary_check = 0 <= start and start < end and end <= n
513  
514      c_boundary = T() if boundary_check else F()
515  
516      lhs = dims[0:start]
517      rhs = dims[end:]
518      mid = dims[start:end]
519  
520      all_possibilities = generate_all_int_dyn_dim_possibilities(mid)
521  
522      all_constraints = []
523  
524      for p in all_possibilities:
525          p = list(p)
526          # this tells us there is a dynamic variable
527          contains_dyn = not all(constraint.op == op_neq for constraint in p)
528          if contains_dyn:
529              mid_var = [Dyn]
530              total_constraints = lhs + mid_var + rhs
531              if len(total_constraints) > 4:
532                  all_constraints.append(F())
533              else:
534                  all_constraints.append(
535                      Conj(
536                          [
537                              BinConstraintT(
538                                  flattened, TensorType(lhs + mid_var + rhs), op_eq
539                              )
540                          ]
541                          + p
542                      )
543                  )
544          else:
545              new_var, counter = gen_dvar(counter)
546              mid_eq_prod = Conj(
547                  [
548                      BinConstraintD(new_var, Prod(mid), op_eq),
549                      BinConstraintD(new_var, Dyn, op_neq),
550                  ]
551              )
552              mid_var = [new_var]
553              total_constraints = lhs + mid_var + rhs
554              if len(total_constraints) > 4:
555                  all_constraints.append(F())
556              else:
557                  all_constraints.append(
558                      Conj(
559                          [
560                              BinConstraintT(
561                                  flattened, TensorType(lhs + mid_var + rhs), op_eq
562                              ),
563                              mid_eq_prod,
564                          ]
565                          + p
566                      )
567                  )
568  
569      return Conj([Disj(all_constraints), c_boundary]), counter

572  @register_transformation_rule(CanReshape)
573  def generate_reshape(constraint, counter):
574      """ 
575      Transform reshape constraints 
576      """
577      d, counter = gen_tensor_dims(4, counter)
578  
579      d1 = d[0]
580      d2 = d[1]
581      d3 = d[2]
582      d4 = d[3]
583  
584      target = constraint.target.__args__
585  
586      is_fully_static = all(d != Dyn for d in target)
587  
588      # dynamic tensor
589      c1_dyn = BinConstraintT(constraint.src, Dyn, op_eq)
590      c2_tensor1 = BinConstraintT(constraint.src, TensorType([d1]), op_eq)
591      c2_tensor2 = BinConstraintT(constraint.src, TensorType([d1, d2]), op_eq)
592      c2_tensor3 = BinConstraintT(constraint.src, TensorType([d1, d2, d3]), op_eq)
593      c2_tensor4 = BinConstraintT(constraint.src, TensorType([d1, d2, d3, d4]), op_eq)
594  
595      d1_eq_dyn = BinConstraintD(d1, Dyn, op_eq)
596      d1_neq_dyn = BinConstraintD(d1, Dyn, op_neq)
597  
598      d2_eq_dyn = BinConstraintD(d2, Dyn, op_eq)
599      d2_neq_dyn = BinConstraintD(d2, Dyn, op_neq)
600  
601      d3_eq_dyn = BinConstraintD(d3, Dyn, op_eq)
602      d3_neq_dyn = BinConstraintD(d3, Dyn, op_neq)
603  
604      d4_eq_dyn = BinConstraintD(d3, Dyn, op_eq)
605      d4_neq_dyn = BinConstraintD(d3, Dyn, op_neq)
606  
607      nat_d1 = BinConstraintD(0, d1, op_leq)
608      nat_d2 = BinConstraintD(0, d2, op_leq)
609      nat_d3 = BinConstraintD(0, d3, op_leq)
610      nat_d4 = BinConstraintD(0, d4, op_leq)
611  
612      if is_fully_static:
613          # size 1 tensor
614          c3_tensor1 = Disj(
615              [d1_eq_dyn, (Conj([d1_neq_dyn, BinConstraintD(d1, Prod(target), op_eq)]))]
616          )
617          all_tensor_1 = Conj([c2_tensor1, c3_tensor1])
618  
619          # size 2 tensor
620          all_tensor_2 = Conj(
621              [c2_tensor2, gen_all_reshape_possibilities([d1, d2], target)]
622          )
623  
624          # size 3 tensor
625          all_tensor_3 = Conj(
626              [c2_tensor3, gen_all_reshape_possibilities([d1, d2, d3], target)]
627          )
628  
629          # size 4 tensor
630          all_tensor_4 = Conj(
631              [c2_tensor4, gen_all_reshape_possibilities([d1, d2, d3, d4], target)]
632          )
633  
634          return (
635              Conj(
636                  [
637                      Disj(
638                          [c1_dyn, all_tensor_1, all_tensor_2, all_tensor_3, all_tensor_4]
639                      ),
640                      nat_d1,
641                      nat_d2,
642                      nat_d3,
643                      nat_d4,
644                  ]
645              ),
646              counter,
647          )
648  
649      # then there must be exactly one occurrence of dyn
650      else:
651          new_target = [n for n in target if n != Dyn]
652  
653          # tensor 1
654          c3_tensor1 = Disj(
655              [d1_eq_dyn, (Conj([d1_neq_dyn, is_dim_div_by_target(new_target, d1)]))]
656          )
657          all_tensor_1 = Conj([c2_tensor1, c3_tensor1])
658  
659          # tensor 2
660          c21 = Disj([d1_eq_dyn, d2_eq_dyn])
661          c22 = Conj(
662              [d1_neq_dyn, d2_neq_dyn, is_dim_div_by_target(new_target, Prod([d1, d2]))]
663          )
664          all_tensor_2 = Conj([c2_tensor2, Disj([c21, c22])])
665  
666          # tensor 3
667          c31 = Disj([d1_eq_dyn, d2_eq_dyn, d3_eq_dyn])
668          c32 = Conj(
669              [
670                  d1_neq_dyn,
671                  d2_neq_dyn,
672                  d3_neq_dyn,
673                  is_dim_div_by_target(new_target, Prod([d1, d2, d3])),
674              ]
675          )
676          all_tensor_3 = Conj([c2_tensor3, Disj([c31, c32])])
677  
678          # tensor 4
679          c41 = Disj([d1_eq_dyn, d2_eq_dyn, d3_eq_dyn, d4_eq_dyn])
680          c42 = Conj(
681              [
682                  d1_neq_dyn,
683                  d2_neq_dyn,
684                  d3_neq_dyn,
685                  d4_neq_dyn,
686                  is_dim_div_by_target(new_target, Prod([d1, d2, d3, d4])),
687              ]
688          )
689          all_tensor_4 = Conj([c2_tensor4, Disj([c41, c42])])
690  
691          return (
692              Conj(
693                  [
694                      Disj(
695                          [c1_dyn, all_tensor_1, all_tensor_2, all_tensor_3, all_tensor_4]
696                      ),
697                      nat_d1,
698                      nat_d2,
699                      nat_d3,

707  @register_transformation_rule(ApplyBroadcasting)
708  def generate_broadcasting(constraint, counter):
709      """ 
710      Transform broadcasting constraints 
711      """
712      e11, e12 = constraint.res1, constraint.res2
713      e1, e2 = constraint.input1, constraint.input2
714  
715      e1_dyn = BinConstraintT(e1, Dyn, op_eq)
716      e2_dyn = BinConstraintT(e2, Dyn, op_eq)
717  
718      # Introduce dimensions
719      e1_equal_e11 = BinConstraintT(e1, e11, op_eq)
720      e2_equal_e12 = BinConstraintT(e2, e12, op_eq)
721  
722      # dyn possibility
723      e1_dyn_constraint = Conj([e1_dyn, e1_equal_e11, e2_equal_e12])
724      e2_dyn_constraint = Conj([e2_dyn, e1_equal_e11, e2_equal_e12])
725  
726      # tensor possibility
727      # generate dimensions to create tensors of size 1
728      final_tensor_1_constraint, _, _, nat_dims_1, counter = gen_broadcasting_constraints(
729          e1, e2, e11, e12, 1, counter
730      )
731  
732      # generate dimensions to create tensors of size 2
733      (
734          final_tensor_2_constraint_no_padding,
735          final_tensor_2_constraint_padding_arg1,
736          final_tensor_2_constraint_padding_arg2,
737          nat_dims_2,
738          counter,
739      ) = gen_broadcasting_constraints(e1, e2, e11, e12, 2, counter)
740  
741      # generate dimensions to create tensors of size 3
742      (
743          final_tensor_3_constraint_no_padding,
744          final_tensor_3_constraint_padding_arg1,
745          final_tensor_3_constraint_padding_arg2,
746          nat_dims_3,
747          counter,
748      ) = gen_broadcasting_constraints(e1, e2, e11, e12, 3, counter)
749  
750      # generate dimensions to create tensors of size 4
751      (
752          final_tensor_4_constraint_no_padding,
753          final_tensor_4_constraint_padding_arg1,
754          final_tensor_4_constraint_padding_arg2,
755          nat_dims_4,
756          counter,
757      ) = gen_broadcasting_constraints(e1, e2, e11, e12, 4, counter)
758  
759      final_result = Disj(
760          [
761              e1_dyn_constraint,
762              e2_dyn_constraint,
763              final_tensor_1_constraint,
764              final_tensor_2_constraint_no_padding,
765              final_tensor_2_constraint_padding_arg1,
766              final_tensor_2_constraint_padding_arg2,
767              final_tensor_3_constraint_no_padding,
768              final_tensor_3_constraint_padding_arg1,
769              final_tensor_3_constraint_padding_arg2,
770              final_tensor_4_constraint_no_padding,
771              final_tensor_4_constraint_padding_arg1,
772              final_tensor_4_constraint_padding_arg2,
773          ]
774      )
775  
776      return (
777          Conj([final_result, *nat_dims_1, *nat_dims_2, *nat_dims_3, *nat_dims_4]),
778          counter,
779      )

800  def calc_last_two_dims(constraint, d: list[DVar]):
801      """ 
802      Generates constraints for the last two dimensions of a convolution or a maxpool output 
803      Args: 
804          constraint: CalcConv or CalcMaxPool 
805          d: The list of output dimensions 
806   
807      Returns: Constraints for calculating the last two dimensions of the output 
808   
809      """
810  
811      assert isinstance(constraint, (CalcConv, CalcMaxPool))
812  
813      b3 = constraint.matching_constraint[2]
814      b4 = constraint.matching_constraint[3]
815  
816      b3_dyn = Conj([BinConstraintD(d[2], Dyn, op_eq), BinConstraintD(b3, Dyn, op_eq)])
817      b4_dyn = Conj([BinConstraintD(d[3], Dyn, op_eq), BinConstraintD(b4, Dyn, op_eq)])
818  
819      d3_not_dyn = Conj(
820          [BinConstraintD(d[2], Dyn, op_neq), BinConstraintD(b3, Dyn, op_neq)]
821      )
822      d4_not_dyn = Conj(
823          [BinConstraintD(d[3], Dyn, op_neq), BinConstraintD(b4, Dyn, op_neq)]
824      )
825  
826      # transform parameters into tuples incase they are not already
827      padding = (
828          (constraint.padding, constraint.padding)
829          if isinstance(constraint.padding, int)
830          else constraint.padding
831      )
832      kernel = (
833          (constraint.kernel, constraint.kernel)
834          if isinstance(constraint.kernel, int)
835          else constraint.kernel
836      )
837      stride = (
838          (constraint.stride, constraint.stride)
839          if isinstance(constraint.stride, int)
840          else constraint.stride
841      )
842      dilation = (
843          (constraint.dilation, constraint.dilation)
844          if isinstance(constraint.dilation, int)
845          else constraint.dilation
846      )
847  
848      f1 = BinConstraintD(b3, BinConstraintD(2, padding[0], op_mul), op_add)
849      f2 = BinConstraintD(dilation[0], BinConstraintD(kernel[0], 1, op_sub), op_mul)
850      f3 = BinConstraintD(
851          BinConstraintD(BinConstraintD(f1, f2, op_sub), 1, op_sub), stride[0], op_div
852      )
853      f4 = BinConstraintD(f3, 1, op_add)
854  
855      c4 = Disj([b3_dyn, Conj([d3_not_dyn, BinConstraintD(d[2], f4, op_eq)])])
856  
857      f11 = BinConstraintD(b4, BinConstraintD(2, padding[1], op_mul), op_add)
858      f22 = BinConstraintD(dilation[1], BinConstraintD(kernel[1], 1, op_sub), op_mul)
859      f33 = BinConstraintD(
860          BinConstraintD(BinConstraintD(f11, f22, op_sub), 1, op_sub), stride[1], op_div
861      )
862      f44 = BinConstraintD(f33, 1, op_add)
863  
864      c5 = Disj([b4_dyn, Conj([d4_not_dyn, BinConstraintD(d[3], f44, op_eq)])])
865  
866      return c4, c5

869  def generate_all_int_dyn_dim_possibilities(my_list: list[DVar]):
870      """ 
871      Generate all possibilities of being equal or not equal to dyn for my_list 
872      Args: 
873          my_list: List of tensor dimensions 
874   
875      Returns: A list of a list of constraints. Each list of constraints corresponds to 
876      one possibility about the values of the dimension variables 
877      """
878      # generate all possibilities of being equal or not equal to dyn for my_list
879      eq_possibilities = [
880          BinConstraintD(my_list[i], Dyn, op_eq) for i in range(len(my_list))
881      ]
882      neq_possibilities = [
883          BinConstraintD(my_list[i], Dyn, op_neq) for i in range(len(my_list))
884      ]
885  
886      d_possibilities = [list(i) for i in zip(eq_possibilities, neq_possibilities)]
887      all_possibilities = list(itertools.product(*d_possibilities))
888      return all_possibilities

917  def gen_all_reshape_possibilities(list_of_dims, target):
918      """ 
919      Consider all possibilities what the input dimensions could be (number or dynamic) 
920      Then generate the appropriate constraints using multiplication or mod depending on the possibility 
921      The possibilities we consider here are the cross product of being equal to dyn or not equal to dyn 
922      for the input. Target is fixed because at most one dimension could be dyn. 
923      We have different cases for this. 
924   
925      Args: 
926          list_of_dims: The input list of dimensions 
927          target: The tensor we want to reshape to 
928   
929      Returns: A disjunction of transformed reshape constraints 
930   
931      """
932      all_possibilities = generate_all_int_dyn_dim_possibilities(list_of_dims)
933  
934      all_constraints = []
935  
936      for p in all_possibilities:
937          to_multiply = []
938  
939          p = list(p)
940  
941          for constraint in p:
942              assert isinstance(constraint, BinConstraintD)
943              if constraint.op == op_neq:
944                  to_multiply.append(constraint.lhs)
945  
946          if not to_multiply:
947              all_constraints.append(Conj(p))
948  
949          elif len(to_multiply) < len(list_of_dims):
950              all_constraints.append(
951                  Conj(p + [is_target_div_by_dim(target, Prod(to_multiply))])
952              )
953          else:
954              all_constraints.append(
955                  Conj(p + [BinConstraintD(Prod(list_of_dims), Prod(target), op_eq)])
956              )
957  
958      return Disj(all_constraints)

961  def broadcast_dim(tensor_input1, tensor_input2, res1, res2, index, padding=False):
962      """ 
963      Apply broadcasting to the 'index' dimension of tensor_input1. 
964      Args: 
965          tensor_input1: should represent [d1, ..., d_index, ...] where d_index = 1 
966          tensor_input2: represents the second input 
967          res1: broadcasted result 1 
968          res2: broadcasted result 2 
969          index: the index to broadcast 
970          padding: If padding was used, then tensor_input1[index] does not exist 
971   
972      Returns: 
973   
974      """
975      if tensor_input1[index] is None:
976          assert padding
977  
978      if not padding:
979          # then the inputs are the same length so they all have dimensions at "index"
980          return Conj(
981              [
982                  BinConstraintD(tensor_input1[index], 1, op_eq),
983                  BinConstraintD(res1[index], res2[index], op_eq),
984                  BinConstraintD(res2[index], tensor_input2[index], op_eq),
985              ]
986          )
987  
988      else:
989          # we don't set the input dimension to 1, since it doesn't exist.
990          return Conj(
991              [
992                  BinConstraintD(res1[index], res2[index], op_eq),
993                  BinConstraintD(res2[index], tensor_input2[index], op_eq),
994              ]
995          )

998  def apply_padding(
999      e1_var: TVar,
1000     e11: BinConstraintT,
1001     e2: BinConstraintT,
1002     e12: BinConstraintT,
1003     d2: list[DVar],
1004     d11: list[DVar],
1005     d12: list[DVar],
1006     counter: int,
1007 ):
1008     """ 
1009     We are considering the possibility where one input has less dimensions than 
1010     another input, so we apply padding to the broadcasted results 
1011  
1012     Args: 
1013         e1_var: Variable representing the first input where padding will be 
1014         e11: constraint of the form e11 = Tensortype[d1, ..., dn] 
1015         e2:  constraint of the form e2 = Tensortype[d1, ..., dn] 
1016         e12: constraint of the form e11 = Tensortype[d1, ..., dn] 
1017         d2: Tensor variables for the second input 
1018         d11: Tensor variables for the broadcasted first input 
1019         d12: Tensor variables for the broadcasted second input 
1020         counter: variable tracking 
1021  
1022     Returns: A new constraint whose goal is to apply padding to the broadcasted result 
1023  
1024     """
1025 
1026     res = []
1027 
1028     # pad the shorter input with None so we can pass it to the broadcasting helper function
1029     for i in range(1, len(d2)):
1030         d1, counter = gen_tensor_dims(i, counter)
1031 
1032         nat_constraints = gen_nat_constraints(d1 + d2 + d11 + d12)
1033 
1034         e1 = BinConstraintT(e1_var, TensorType(d1), op_eq)
1035 
1036         simulate_padding = [None] * (len(d2) - i)
1037 
1038         assert len(simulate_padding + d1) == len(d2)
1039 
1040         # for every padding size, we also consider broadcasting
1041         broadcast_padding = [
1042             broadcast_dim(simulate_padding, d2, d11, d12, j, True)
1043             for j in range(len(d2) - i)
1044         ]
1045 
1046         # we consider the possibilities for broadcasting for every dimension. Since we already
1047         # padded d1, we do not consider it while broadcasting
1048         all_broadcasting_possibilities = (
1049             generate_all_broadcasting_possibilities_no_padding(
1050                 d1, d2[(len(d2) - i) :], d11[(len(d2) - i) :], d12[(len(d2) - i) :]
1051             )
1052         )
1053         # combine all constraints into a conjunction
1054         c = Conj(
1055             [
1056                 e1,
1057                 e11,
1058                 e2,
1059                 e12,
1060                 *broadcast_padding,
1061                 all_broadcasting_possibilities,
1062                 *nat_constraints,
1063             ]
1064         )
1065         res.append(c)
1066 
1067     return Disj(res), counter

1268 def gen_broadcasting_constraints(
1269     e1: TVar, e2: TVar, e11: TVar, e12: TVar, i: int, counter: int
1270 ):
1271     """ 
1272     Simulates broadcasting on e1 and e2 and returns the results 
1273     respectively in e11 and e12. Because of gradual types, 
1274     e1 and e2 may not be equal. Similarly, e11 and e12 may not 
1275     be equal. e11 and e12 should be guaranteed to be consistent 
1276     as they represent the shapes of the tensors to be added after 
1277     broadcasting. 
1278     Args: 
1279         e1: TVar representing the type of input 1 
1280         e2: TVar representing the type of input 2 
1281         e11: TVar representing the representing broadcasted input 1 
1282         e12: TVar representing the representing broadcasted input 2 
1283         i: The rank of the resulting type of addition 
1284         counter: for variable tracking 
1285  
1286     Returns: Simplified broadcasting constraints 
1287  
1288     """
1289     dims, counter = gen_lists_of_dims(4, i, counter)
1290     [d1, d2, d3, d4] = dims
1291     nat_dims_i = gen_nat_constraints(list(itertools.chain.from_iterable(dims)))
1292 
1293     initialize_tensors_constraints = create_equality_constraints_for_broadcasting(
1294         e1, e2, e11, e12, d1, d2, d3, d4
1295     )
1296 
1297     [e1_tensor, e11_tensor, e2_tensor, e12_tensor] = initialize_tensors_constraints
1298 
1299     # without padding, broadcast all possibilities for tensors of size i
1300     final_tensor_constraint_no_padding = Conj(
1301         [
1302             *initialize_tensors_constraints,
1303             generate_all_broadcasting_possibilities_no_padding(d1, d2, d3, d4),
1304         ]
1305     )
1306 
1307     # with padding, broadcast all possibilities for tensors of size i
1308     final_tensor_constraint_padding_arg1, counter = apply_padding(
1309         e1, e11_tensor, e2_tensor, e12_tensor, d2, d3, d4, counter
1310     )
1311 
1312     final_tensor_constraint_padding_arg2, counter = apply_padding(
1313         e2, e12_tensor, e1_tensor, e11_tensor, d1, d4, d3, counter
1314     )
1315 
1316     return (
1317         final_tensor_constraint_no_padding,
1318         final_tensor_constraint_padding_arg1,
1319         final_tensor_constraint_padding_arg2,
1320         nat_dims_i,
1321         counter,
1322     )
1323
