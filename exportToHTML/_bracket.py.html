75   def _bracket_root(func, xl0, xr0=None, *, xmin=None, xmax=None, factor=None,
76                     args=(), maxiter=1000):
77       """Bracket the root of a monotonic scalar function of one variable 
78    
79       This function works elementwise when `xl0`, `xr0`, `xmin`, `xmax`, `factor`, and 
80       the elements of `args` are broadcastable arrays. 
81    
82       Parameters 
83       ---------- 
84       func : callable 
85           The function for which the root is to be bracketed. 
86           The signature must be:: 
87    
88               func(x: ndarray, *args) -> ndarray 
89    
90           where each element of ``x`` is a finite real and ``args`` is a tuple, 
91           which may contain an arbitrary number of arrays that are broadcastable 
92           with `x`. ``func`` must be an elementwise function: each element 
93           ``func(x)[i]`` must equal ``func(x[i])`` for all indices ``i``. 
94       xl0, xr0: float array_like 
95           Starting guess of bracket, which need not contain a root. If `xr0` is 
96           not provided, ``xr0 = xl0 + 1``. Must be broadcastable with one another. 
97       xmin, xmax : float array_like, optional 
98           Minimum and maximum allowable endpoints of the bracket, inclusive. Must 
99           be broadcastable with `xl0` and `xr0`. 
100      factor : float array_like, default: 2 
101          The factor used to grow the bracket. See notes for details. 
102      args : tuple, optional 
103          Additional positional arguments to be passed to `func`.  Must be arrays 
104          broadcastable with `xl0`, `xr0`, `xmin`, and `xmax`. If the callable to be 
105          bracketed requires arguments that are not broadcastable with these 
106          arrays, wrap that callable with `func` such that `func` accepts 
107          only `x` and broadcastable arrays. 
108      maxiter : int, optional 
109          The maximum number of iterations of the algorithm to perform. 
110   
111      Returns 
112      ------- 
113      res : _RichResult 
114          An instance of `scipy._lib._util._RichResult` with the following 
115          attributes. The descriptions are written as though the values will be 
116          scalars; however, if `func` returns an array, the outputs will be 
117          arrays of the same shape. 
118   
119          xl, xr : float 
120              The lower and upper ends of the bracket, if the algorithm 
121              terminated successfully. 
122          fl, fr : float 
123              The function value at the lower and upper ends of the bracket. 
124          nfev : int 
125              The number of function evaluations required to find the bracket. 
126              This is distinct from the number of times `func` is *called* 
127              because the function may evaluated at multiple points in a single 
128              call. 
129          nit : int 
130              The number of iterations of the algorithm that were performed. 
131          status : int 
132              An integer representing the exit status of the algorithm. 
133   
134              - ``0`` : The algorithm produced a valid bracket. 
135              - ``-1`` : The bracket expanded to the allowable limits without finding a bracket. 
136              - ``-2`` : The maximum number of iterations was reached. 
137              - ``-3`` : A non-finite value was encountered. 
138              - ``-4`` : Iteration was terminated by `callback`. 
139              - ``-5``: The initial bracket does not satisfy `xmin <= xl0 < xr0 < xmax`. 
140              - ``1`` : The algorithm is proceeding normally (in `callback` only). 
141              - ``2`` : A bracket was found in the opposite search direction (in `callback` only). 
142   
143          success : bool 
144              ``True`` when the algorithm terminated successfully (status ``0``). 
145   
146      Notes 
147      ----- 
148      This function generalizes an algorithm found in pieces throughout 
149      `scipy.stats`. The strategy is to iteratively grow the bracket ``(l, r)`` 
150       until ``func(l) < 0 < func(r)``. The bracket grows to the left as follows. 
151   
152      - If `xmin` is not provided, the distance between `xl0` and `l` is iteratively 
153        increased by `factor`. 
154      - If `xmin` is provided, the distance between `xmin` and `l` is iteratively 
155        decreased by `factor`. Note that this also *increases* the bracket size. 
156   
157      Growth of the bracket to the right is analogous. 
158   
159      Growth of the bracket in one direction stops when the endpoint is no longer 
160      finite, the function value at the endpoint is no longer finite, or the 
161      endpoint reaches its limiting value (`xmin` or `xmax`). Iteration terminates 
162      when the bracket stops growing in both directions, the bracket surrounds 
163      the root, or a root is found (accidentally). 
164   
165      If two brackets are found - that is, a bracket is found on both sides in 
166      the same iteration, the smaller of the two is returned. 
167      If roots of the function are found, both `l` and `r` are set to the 
168      leftmost root. 
169   
170      """  # noqa: E501

188      # The approach is to treat the left and right searches as though they were
189      # (almost) totally independent one-sided bracket searches. (The interaction
190      # is considered when checking for termination and preparing the result
191      # object.)
192      # `x` is the "moving" end of the bracket
193      x = xp.concat(xs)
194      f = xp.concat(fs)
195      invalid_bracket = xp.concat((invalid_bracket, invalid_bracket))
196      n = x.shape[0] // 2

198      # `x_last` is the previous location of the moving end of the bracket. If
199      # the signs of `f` and `f_last` are different, `x` and `x_last` form a
200      # bracket.
201      x_last = xp.concat((x[n:], x[:n]))
202      f_last = xp.concat((f[n:], f[:n]))
203      # `x0` is the "fixed" end of the bracket.
204      x0 = x_last
205      # We don't need to retain the corresponding function value, since the
206      # fixed end of the bracket is only needed to compute the new value of the
207      # moving end; it is never returned.
208      limit = xp.concat((xmin, xmax))

210      factor = xp_ravel(xp.broadcast_to(factor, shape), xp=xp)
211      factor = xp.astype(factor, dtype, copy=False)
212      factor = xp.concat((factor, factor))

214      active = xp.arange(2*n)
215      args = [xp.concat((arg, arg)) for arg in args]

217      # This is needed due to inner workings of `eim._loop`.
218      # We're abusing it a tiny bit.

221      # `d` is for "distance".
222      # For searches without a limit, the distance between the fixed end of the
223      # bracket `x0` and the moving end `x` will grow by `factor` each iteration.
224      # For searches with a limit, the distance between the `limit` and moving
225      # end of the bracket `x` will shrink by `factor` each iteration.
226      i = xp.isinf(limit)
227      ni = ~i
228      d = xp.zeros_like(x)
229      d[i] = x[i] - x0[i]
230      d[ni] = limit[ni] - x[ni]
231  
232      status = xp.full_like(x, eim._EINPROGRESS, dtype=xp.int32)  # in progress

234      nit, nfev = 0, 1  # one function evaluation per side performed above
235  
236      work = _RichResult(x=x, x0=x0, f=f, limit=limit, factor=factor,
237                         active=active, d=d, x_last=x_last, f_last=f_last,
238                         nit=nit, nfev=nfev, status=status, args=args,
239                         xl=xp.nan, xr=xp.nan, fl=xp.nan, fr=xp.nan, n=n)
240      res_work_pairs = [('status', 'status'), ('xl', 'xl'), ('xr', 'xr'),
241                        ('nit', 'nit'), ('nfev', 'nfev'), ('fl', 'fl'),
242                        ('fr', 'fr'), ('x', 'x'), ('f', 'f'),
243                        ('x_last', 'x_last'), ('f_last', 'f_last')]

245      def pre_func_eval(work):
246          # Initialize moving end of bracket
247          x = xp.zeros_like(work.x)
248  
249          # Unlimited brackets grow by `factor` by increasing distance from fixed
250          # end to moving end.
251          i = xp.isinf(work.limit)  # indices of unlimited brackets
252          work.d[i] *= work.factor[i]
253          x[i] = work.x0[i] + work.d[i]
254  
255          # Limited brackets grow by decreasing the distance from the limit to
256          # the moving end.
257          ni = ~i  # indices of limited brackets
258          work.d[ni] /= work.factor[ni]
259          x[ni] = work.limit[ni] - work.d[ni]
260  
261          return x

263      def post_func_eval(x, f, work):
264          # Keep track of the previous location of the moving end so that we can
265          # return a narrower bracket. (The alternative is to remember the
266          # original fixed end, but then the bracket would be wider than needed.)
267          work.x_last = work.x
268          work.f_last = work.f
269          work.x = x
270          work.f = f

272      def check_termination(work):
273          # Condition 0: initial bracket is invalid
274          stop = (work.status == eim._EINPUTERR)
275  
276          # Condition 1: a valid bracket (or the root itself) has been found
277          sf = xp.sign(work.f)
278          sf_last = xp.sign(work.f_last)
279          i = ((sf_last == -sf) | (sf_last == 0) | (sf == 0)) & ~stop
280          work.status[i] = eim._ECONVERGED
281          stop[i] = True
282  
283          # Condition 2: the other side's search found a valid bracket.
284          # (If we just found a bracket with the rightward search, we can stop
285          #  the leftward search, and vice-versa.)
286          # To do this, we need to set the status of the other side's search;
287          # this is tricky because `work.status` contains only the *active*
288          # elements, so we don't immediately know the index of the element we
289          # need to set - or even if it's still there. (That search may have
290          # terminated already, e.g. by reaching its `limit`.)
291          # To facilitate this, `work.active` contains a unit integer index of
292          # each search. Index `k` (`k < n)` and `k + n` correspond with a
293          # leftward and rightward search, respectively. Elements are removed
294          # from `work.active` just as they are removed from `work.status`, so
295          # we use `work.active` to help find the right location in
296          # `work.status`.
297          # Get the integer indices of the elements that can also stop
298          also_stop = (work.active[i] + work.n) % (2*work.n)
299          # Check whether they are still active. We want to find the indices
300          # in work.active where the associated values in work.active are
301          # contained in also_stop. xp.searchsorted let's us take advantage
302          # of work.active being sorted, but requires some hackery because
303          # searchsorted solves the separate but related problem of finding
304          # the indices where the values in also_stop should be added to
305          # maintain sorted order.
306          j = xp.searchsorted(work.active, also_stop)
307          # If the location exceeds the length of the `work.active`, they are
308          # not there. This happens when a value in also_stop is larger than
309          # the greatest value in work.active. This case needs special handling
310          # because we cannot simply check that also_stop == work.active[j].
311          mask = j < work.active.shape[0]
312          # Note that we also have to use the mask to filter also_stop to ensure
313          # that also_stop and j will still have the same shape.
314          j, also_stop = j[mask], also_stop[mask]
315          j = j[also_stop == work.active[j]]
316          # Now convert these to boolean indices to use with `work.status`.
317          i = xp.zeros_like(stop)
318          i[j] = True  # boolean indices of elements that can also stop
319          i = i & ~stop
320          work.status[i] = _ESTOPONESIDE
321          stop[i] = True
322  
323          # Condition 3: moving end of bracket reaches limit
324          i = (work.x == work.limit) & ~stop
325          work.status[i] = _ELIMITS
326          stop[i] = True
327  
328          # Condition 4: non-finite value encountered
329          i = ~(xp.isfinite(work.x) & xp.isfinite(work.f)) & ~stop
330          work.status[i] = eim._EVALUEERR
331          stop[i] = True
332  
333          return stop

338      def customize_result(res, shape):
339          n = res['x'].shape[0] // 2
340  
341          # To avoid ambiguity, below we refer to `xl0`, the initial left endpoint
342          # as `a` and `xr0`, the initial right endpoint, as `b`.
343          # Because we treat the two one-sided searches as though they were
344          # independent, what we keep track of in `work` and what we want to
345          # return in `res` look quite different. Combine the results from the
346          # two one-sided searches before reporting the results to the user.
347          # - "a" refers to the leftward search (the moving end started at `a`)
348          # - "b" refers to the rightward search (the moving end started at `b`)
349          # - "l" refers to the left end of the bracket (closer to -oo)
350          # - "r" refers to the right end of the bracket (closer to +oo)
351          xal = res['x'][:n]
352          xar = res['x_last'][:n]
353          xbl = res['x_last'][n:]
354          xbr = res['x'][n:]
355  
356          fal = res['f'][:n]
357          far = res['f_last'][:n]
358          fbl = res['f_last'][n:]
359          fbr = res['f'][n:]
360  
361          # Initialize the brackets and corresponding function values to return
362          # to the user. Brackets may not be valid (e.g. there is no root,
363          # there weren't enough iterations, NaN encountered), but we still need
364          # to return something. One option would be all NaNs, but what I've
365          # chosen here is the left- and right-most points at which the function
366          # has been evaluated. This gives the user some information about what
367          # interval of the real line has been searched and shows that there is
368          # no sign change between the two ends.
369          xl = xp.asarray(xal, copy=True)
370          fl = xp.asarray(fal, copy=True)
371          xr = xp.asarray(xbr, copy=True)
372          fr = xp.asarray(fbr, copy=True)
373  
374          # `status` indicates whether the bracket is valid or not. If so,
375          # we want to adjust the bracket we return to be the narrowest possible
376          # given the points at which we evaluated the function.
377          # For example if bracket "a" is valid and smaller than bracket "b" OR
378          # if bracket "a" is valid and bracket "b" is not valid, we want to
379          # return bracket "a" (and vice versa).
380          sa = res['status'][:n]
381          sb = res['status'][n:]
382  
383          da = xar - xal
384          db = xbr - xbl
385  
386          i1 = ((da <= db) & (sa == 0)) | ((sa == 0) & (sb != 0))
387          i2 = ((db <= da) & (sb == 0)) | ((sb == 0) & (sa != 0))
388  
389          xr[i1] = xar[i1]
390          fr[i1] = far[i1]
391          xl[i2] = xbl[i2]
392          fl[i2] = fbl[i2]
393  
394          # Finish assembling the result object
395          res['xl'] = xl
396          res['xr'] = xr
397          res['fl'] = fl
398          res['fr'] = fr
399  
400          res['nit'] = xp.maximum(res['nit'][:n], res['nit'][n:])
401          res['nfev'] = res['nfev'][:n] + res['nfev'][n:]
402          # If the status on one side is zero, the status is zero. In any case,
403          # report the status from one side only.
404          res['status'] = xp.where(sa == 0, sa, sb)
405          res['success'] = (res['status'] == 0)
406  
407          del res['x']
408          del res['f']
409          del res['x_last']
410          del res['f_last']
411  
412          return shape[:-1]
413  
414      return eim._loop(work, callback, shape, maxiter, func, args, dtype,
415                       pre_func_eval, post_func_eval, check_termination,
416                       post_termination_check, customize_result, res_work_pairs,
417                       xp)

502  def _bracket_minimum(func, xm0, *, xl0=None, xr0=None, xmin=None, xmax=None,
503                       factor=None, args=(), maxiter=1000):
504      """Bracket the minimum of a unimodal scalar function of one variable 
505   
506      This function works elementwise when `xm0`, `xl0`, `xr0`, `xmin`, `xmax`, 
507      and the elements of `args` are broadcastable arrays. 
508   
509      Parameters 
510      ---------- 
511      func : callable 
512          The function for which the minimum is to be bracketed. 
513          The signature must be:: 
514   
515              func(x: ndarray, *args) -> ndarray 
516   
517          where each element of ``x`` is a finite real and ``args`` is a tuple, 
518          which may contain an arbitrary number of arrays that are broadcastable 
519          with ``x``. `func` must be an elementwise function: each element 
520          ``func(x)[i]`` must equal ``func(x[i])`` for all indices `i`. 
521      xm0: float array_like 
522          Starting guess for middle point of bracket. 
523      xl0, xr0: float array_like, optional 
524          Starting guesses for left and right endpoints of the bracket. Must be 
525          broadcastable with one another and with `xm0`. 
526      xmin, xmax : float array_like, optional 
527          Minimum and maximum allowable endpoints of the bracket, inclusive. Must 
528          be broadcastable with `xl0`, `xm0`, and `xr0`. 
529      factor : float array_like, optional 
530          Controls expansion of bracket endpoint in downhill direction. Works 
531          differently in the cases where a limit is set in the downhill direction 
532          with `xmax` or `xmin`. See Notes. 
533      args : tuple, optional 
534          Additional positional arguments to be passed to `func`.  Must be arrays 
535          broadcastable with `xl0`, `xm0`, `xr0`, `xmin`, and `xmax`. If the 
536          callable to be bracketed requires arguments that are not broadcastable 
537          with these arrays, wrap that callable with `func` such that `func` 
538          accepts only ``x`` and broadcastable arrays. 
539      maxiter : int, optional 
540          The maximum number of iterations of the algorithm to perform. The number 
541          of function evaluations is three greater than the number of iterations. 
542   
543      Returns 
544      ------- 
545      res : _RichResult 
546          An instance of `scipy._lib._util._RichResult` with the following 
547          attributes. The descriptions are written as though the values will be 
548          scalars; however, if `func` returns an array, the outputs will be 
549          arrays of the same shape. 
550   
551          xl, xm, xr : float 
552              The left, middle, and right points of the bracket, if the algorithm 
553              terminated successfully. 
554          fl, fm, fr : float 
555              The function value at the left, middle, and right points of the bracket. 
556          nfev : int 
557              The number of function evaluations required to find the bracket. 
558          nit : int 
559              The number of iterations of the algorithm that were performed. 
560          status : int 
561              An integer representing the exit status of the algorithm. 
562   
563              - ``0`` : The algorithm produced a valid bracket. 
564              - ``-1`` : The bracket expanded to the allowable limits. Assuming 
565                         unimodality, this implies the endpoint at the limit is a 
566                         minimizer. 
567              - ``-2`` : The maximum number of iterations was reached. 
568              - ``-3`` : A non-finite value was encountered. 
569              - ``-4`` : ``None`` shall pass. 
570              - ``-5`` : The initial bracket does not satisfy 
571                         `xmin <= xl0 < xm0 < xr0 <= xmax`. 
572   
573          success : bool 
574              ``True`` when the algorithm terminated successfully (status ``0``). 
575   
576      Notes 
577      ----- 
578      Similar to `scipy.optimize.bracket`, this function seeks to find real 
579      points ``xl < xm < xr`` such that ``f(xl) >= f(xm)`` and ``f(xr) >= f(xm)``, 
580      where at least one of the inequalities is strict. Unlike `scipy.optimize.bracket`, 
581      this function can operate in a vectorized manner on array input, so long as 
582      the input arrays are broadcastable with each other. Also unlike 
583      `scipy.optimize.bracket`, users may specify minimum and maximum endpoints 
584      for the desired bracket. 
585   
586      Given an initial trio of points ``xl = xl0``, ``xm = xm0``, ``xr = xr0``, 
587      the algorithm checks if these points already give a valid bracket. If not, 
588      a new endpoint, ``w`` is chosen in the "downhill" direction, ``xm`` becomes the new 
589      opposite endpoint, and either `xl` or `xr` becomes the new middle point, 
590      depending on which direction is downhill. The algorithm repeats from here. 
591   
592      The new endpoint `w` is chosen differently depending on whether or not a 
593      boundary `xmin` or `xmax` has been set in the downhill direction. Without 
594      loss of generality, suppose the downhill direction is to the right, so that 
595      ``f(xl) > f(xm) > f(xr)``. If there is no boundary to the right, then `w` 
596      is chosen to be ``xr + factor * (xr - xm)`` where `factor` is controlled by 
597      the user (defaults to 2.0) so that step sizes increase in geometric proportion. 
598      If there is a boundary, `xmax` in this case, then `w` is chosen to be 
599      ``xmax - (xmax - xr)/factor``, with steps slowing to a stop at 
600      `xmax`. This cautious approach ensures that a minimum near but distinct from 
601      the boundary isn't missed while also detecting whether or not the `xmax` is 
602      a minimizer when `xmax` is reached after a finite number of steps. 
603      """  # noqa: E501
604      callback = None  # works; I just don't want to test it

606      temp = _bracket_minimum_iv(func, xm0, xl0, xr0, xmin, xmax, factor, args, maxiter)
607      func, xm0, xl0, xr0, xmin, xmax, factor, args, maxiter, xp = temp
608  
609      xs = (xl0, xm0, xr0)
610      temp = eim._initialize(func, xs, args)
611      func, xs, fs, args, shape, dtype, xp = temp
612  
613      xl0, xm0, xr0 = xs
614      fl0, fm0, fr0 = fs
615      xmin = xp.astype(xp.broadcast_to(xmin, shape), dtype, copy=False)
616      xmin = xp_ravel(xmin, xp=xp)
617      xmax = xp.astype(xp.broadcast_to(xmax, shape), dtype, copy=False)
618      xmax = xp_ravel(xmax, xp=xp)
619      invalid_bracket = ~((xmin <= xl0) & (xl0 < xm0) & (xm0 < xr0) & (xr0 <= xmax))
620      # We will modify factor later on so make a copy. np.broadcast_to returns
621      # a read-only view.
622      factor = xp.astype(xp.broadcast_to(factor, shape), dtype, copy=True)
623      factor = xp_ravel(factor)
624  
625      # To simplify the logic, swap xl and xr if f(xl) < f(xr). We should always be
626      # marching downhill in the direction from xl to xr.
627      comp = fl0 < fr0
628      xl0[comp], xr0[comp] = xr0[comp], xl0[comp]
629      fl0[comp], fr0[comp] = fr0[comp], fl0[comp]
630      # We only need the boundary in the direction we're traveling.
631      limit = xp.where(comp, xmin, xmax)
632  
633      unlimited = xp.isinf(limit)
634      limited = ~unlimited
635      step = xp.empty_like(xl0)
636  
637      step[unlimited] = (xr0[unlimited] - xm0[unlimited])
638      step[limited] = (limit[limited] - xr0[limited])
639  
640      # Step size is divided by factor for case where there is a limit.
641      factor[limited] = 1 / factor[limited]
642  
643      status = xp.full_like(xl0, eim._EINPROGRESS, dtype=xp.int32)
644      status[invalid_bracket] = eim._EINPUTERR
645      nit, nfev = 0, 3
646  
647      work = _RichResult(xl=xl0, xm=xm0, xr=xr0, xr0=xr0, fl=fl0, fm=fm0, fr=fr0,
648                         step=step, limit=limit, limited=limited, factor=factor, nit=nit,
649                         nfev=nfev, status=status, args=args)

651      res_work_pairs = [('status', 'status'), ('xl', 'xl'), ('xm', 'xm'), ('xr', 'xr'),
652                        ('nit', 'nit'), ('nfev', 'nfev'), ('fl', 'fl'), ('fm', 'fm'),
653                        ('fr', 'fr')]

655      def pre_func_eval(work):
656          work.step *= work.factor
657          x = xp.empty_like(work.xr)
658          x[~work.limited] = work.xr0[~work.limited] + work.step[~work.limited]
659          x[work.limited] = work.limit[work.limited] - work.step[work.limited]
660          # Since the new bracket endpoint is calculated from an offset with the
661          # limit, it may be the case that the new endpoint equals the old endpoint,
662          # when the old endpoint is sufficiently close to the limit. We use the
663          # limit itself as the new endpoint in these cases.
664          x[work.limited] = xp.where(
665              x[work.limited] == work.xr[work.limited],
666              work.limit[work.limited],
667              x[work.limited],
668          )
669          return x
670  
671      def post_func_eval(x, f, work):
672          work.xl, work.xm, work.xr = work.xm, work.xr, x
673          work.fl, work.fm, work.fr = work.fm, work.fr, f
674  
675      def check_termination(work):
676          # Condition 0: Initial bracket is invalid.
677          stop = (work.status == eim._EINPUTERR)
678  
679          # Condition 1: A valid bracket has been found.
680          i = (
681              (work.fl >= work.fm) & (work.fr > work.fm)
682              | (work.fl > work.fm) & (work.fr >= work.fm)
683          ) & ~stop
684          work.status[i] = eim._ECONVERGED
685          stop[i] = True
686  
687          # Condition 2: Moving end of bracket reaches limit.
688          i = (work.xr == work.limit) & ~stop
689          work.status[i] = _ELIMITS
690          stop[i] = True
691  
692          # Condition 3: non-finite value encountered
693          i = ~(xp.isfinite(work.xr) & xp.isfinite(work.fr)) & ~stop
694          work.status[i] = eim._EVALUEERR
695          stop[i] = True
696  
697          return stop
698  
699      def post_termination_check(work):
700          pass
701  
702      def customize_result(res, shape):
703          # Reorder entries of xl and xr if they were swapped due to f(xl0) < f(xr0).
704          comp = res['xl'] > res['xr']
705          res['xl'][comp], res['xr'][comp] = res['xr'][comp], res['xl'][comp]
706          res['fl'][comp], res['fr'][comp] = res['fr'][comp], res['fl'][comp]
707          return shape
708  
709      return eim._loop(work, callback, shape,
710                       maxiter, func, args, dtype,
711                       pre_func_eval, post_func_eval,
712                       check_termination, post_termination_check,
713                       customize_result, res_work_pairs, xp)
