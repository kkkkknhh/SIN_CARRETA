24   class MultiUFunc:
25       def __init__(self, ufunc_or_ufuncs, doc=None, *,
26                    force_complex_output=False, **default_kwargs):
27           if not isinstance(ufunc_or_ufuncs, np.ufunc):
28               if isinstance(ufunc_or_ufuncs, collections.abc.Mapping):
29                   ufuncs_iter = ufunc_or_ufuncs.values()
30               elif isinstance(ufunc_or_ufuncs, collections.abc.Iterable):
31                   ufuncs_iter = ufunc_or_ufuncs
32               else:
33                   raise ValueError("ufunc_or_ufuncs should be a ufunc or a"
34                                    " ufunc collection")
35   
36               # Perform input validation to ensure all ufuncs in ufuncs are
37               # actually ufuncs and all take the same input types.
38               seen_input_types = set()
39               for ufunc in ufuncs_iter:
40                   if not isinstance(ufunc, np.ufunc):
41                       raise ValueError("All ufuncs must have type `numpy.ufunc`."
42                                        f" Received {ufunc_or_ufuncs}")
43                   seen_input_types.add(frozenset(x.split("->")[0] for x in ufunc.types))
44               if len(seen_input_types) > 1:
45                   raise ValueError("All ufuncs must take the same input types.")
46   
47           self._ufunc_or_ufuncs = ufunc_or_ufuncs
48           self.__doc = doc
49           self.__force_complex_output = force_complex_output
50           self._default_kwargs = default_kwargs
51           self._resolve_out_shapes = None
52           self._finalize_out = None
53           self._key = None
54           self._ufunc_default_args = lambda *args, **kwargs: ()
55           self._ufunc_default_kwargs = lambda *args, **kwargs: {}
56   
57       @property
58       def __doc__(self):
59           return self.__doc
60   
61       def _override_key(self, func):
62           """Set `key` method by decorating a function. 
63           """
64           self._key = func
65   
66       def _override_ufunc_default_args(self, func):
67           self._ufunc_default_args = func
68   
69       def _override_ufunc_default_kwargs(self, func):
70           self._ufunc_default_kwargs = func
71   
72       def _override_resolve_out_shapes(self, func):
73           """Set `resolve_out_shapes` method by decorating a function."""
74           if func.__doc__ is None:
75               func.__doc__ = \
76                   """Resolve to output shapes based on relevant inputs."""
77           func.__name__ = "resolve_out_shapes"
78           self._resolve_out_shapes = func
79   
80       def _override_finalize_out(self, func):
81           self._finalize_out = func
82   
83       def _resolve_ufunc(self, **kwargs):
84           """Resolve to a ufunc based on keyword arguments."""
85   
86           if isinstance(self._ufunc_or_ufuncs, np.ufunc):
87               return self._ufunc_or_ufuncs
88   
89           ufunc_key = self._key(**kwargs)
90           return self._ufunc_or_ufuncs[ufunc_key]
91   
92       def __call__(self, *args, **kwargs):
93           kwargs = self._default_kwargs | kwargs
94   
95           args += self._ufunc_default_args(**kwargs)
96   
97           ufunc = self._resolve_ufunc(**kwargs)
98   
99           # array arguments to be passed to the ufunc
100          ufunc_args = [np.asarray(arg) for arg in args[-ufunc.nin:]]
101  
102          ufunc_kwargs = self._ufunc_default_kwargs(**kwargs)
103  
104          if (self._resolve_out_shapes is not None):
105              ufunc_arg_shapes = tuple(np.shape(ufunc_arg) for ufunc_arg in ufunc_args)
106              ufunc_out_shapes = self._resolve_out_shapes(*args[:-ufunc.nin],
107                                                          *ufunc_arg_shapes, ufunc.nout,
108                                                          **kwargs)
109  
110              ufunc_arg_dtypes = tuple(ufunc_arg.dtype if hasattr(ufunc_arg, 'dtype')
111                                       else np.dtype(type(ufunc_arg))
112                                       for ufunc_arg in ufunc_args)
113  
114              if hasattr(ufunc, 'resolve_dtypes'):
115                  ufunc_dtypes = ufunc_arg_dtypes + ufunc.nout * (None,)
116                  ufunc_dtypes = ufunc.resolve_dtypes(ufunc_dtypes)
117                  ufunc_out_dtypes = ufunc_dtypes[-ufunc.nout:]
118              else:
119                  ufunc_out_dtype = np.result_type(*ufunc_arg_dtypes)
120                  if (not np.issubdtype(ufunc_out_dtype, np.inexact)):
121                      ufunc_out_dtype = np.float64
122  
123                  ufunc_out_dtypes = ufunc.nout * (ufunc_out_dtype,)
124  
125              if self.__force_complex_output:
126                  ufunc_out_dtypes = tuple(np.result_type(1j, ufunc_out_dtype)
127                                           for ufunc_out_dtype in ufunc_out_dtypes)
128  
129              out = tuple(np.empty(ufunc_out_shape, dtype=ufunc_out_dtype)
130                          for ufunc_out_shape, ufunc_out_dtype
131                          in zip(ufunc_out_shapes, ufunc_out_dtypes))
132  
133              ufunc_kwargs['out'] = out
134  
135          out = ufunc(*ufunc_args, **ufunc_kwargs)
136          if (self._finalize_out is not None):
137              out = self._finalize_out(out)
138  
139          return out

144      r"""sph_legendre_p(n, m, theta, *, diff_n=0) 
145   
146      Spherical Legendre polynomial of the first kind. 
147   
148      Parameters 
149      ---------- 
150      n : ArrayLike[int] 
151          Degree of the spherical Legendre polynomial. Must have ``n >= 0``. 
152      m : ArrayLike[int] 
153          Order of the spherical Legendre polynomial. 
154      theta : ArrayLike[float] 
155          Input value. 
156      diff_n : Optional[int] 
157          A non-negative integer. Compute and return all derivatives up 
158          to order ``diff_n``. Default is 0. 
159   
160      Returns 
161      ------- 
162      p : ndarray or tuple[ndarray] 
163          Spherical Legendre polynomial with ``diff_n`` derivatives. 
164   
165      Notes 
166      ----- 
167      The spherical counterpart of an (unnormalized) associated Legendre polynomial has 
168      the additional factor 
169   
170      .. math:: 
171   
172          \sqrt{\frac{(2 n + 1) (n - m)!}{4 \pi (n + m)!}} 
173   
174      It is the same as the spherical harmonic :math:`Y_{n}^{m}(\theta, \phi)` 
175      with :math:`\phi = 0`. 
176      """, diff_n=0
177  )

184          raise ValueError(
185              "diff_n is currently only implemented for orders 0, 1, and 2,"
186              f" received: {diff_n}."
187          )
188      return diff_n

465  sph_harm_y = MultiUFunc(
466      sph_harm_y,
467      r"""sph_harm_y(n, m, theta, phi, *, diff_n=0) 
468   
469      Spherical harmonics. They are defined as 
470   
471      .. math:: 
472   
473          Y_n^m(\theta,\phi) = \sqrt{\frac{2 n + 1}{4 \pi} \frac{(n - m)!}{(n + m)!}} 
474              P_n^m(\cos(\theta)) e^{i m \phi} 
475   
476      where :math:`P_n^m` are the (unnormalized) associated Legendre polynomials. 
477   
478      Parameters 
479      ---------- 
480      n : ArrayLike[int] 
481          Degree of the harmonic. Must have ``n >= 0``. This is 
482          often denoted by ``l`` (lower case L) in descriptions of 
483          spherical harmonics. 
484      m : ArrayLike[int] 
485          Order of the harmonic. 
486      theta : ArrayLike[float] 
487          Polar (colatitudinal) coordinate; must be in ``[0, pi]``. 
488      phi : ArrayLike[float] 
489          Azimuthal (longitudinal) coordinate; must be in ``[0, 2*pi]``. 
490      diff_n : Optional[int] 
491          A non-negative integer. Compute and return all derivatives up 
492          to order ``diff_n``. Default is 0. 
493   
494      Returns 
495      ------- 
496      y : ndarray[complex] or tuple[ndarray[complex]] 
497         Spherical harmonics with ``diff_n`` derivatives. 
498   
499      Notes 
500      ----- 
501      There are different conventions for the meanings of the input 
502      arguments ``theta`` and ``phi``. In SciPy ``theta`` is the 
503      polar angle and ``phi`` is the azimuthal angle. It is common to 
504      see the opposite convention, that is, ``theta`` as the azimuthal angle 
505      and ``phi`` as the polar angle. 
506   
507      Note that SciPy's spherical harmonics include the Condon-Shortley 
508      phase [2]_ because it is part of `sph_legendre_p`. 
509   
510      With SciPy's conventions, the first several spherical harmonics 
511      are 
512   
513      .. math:: 
514   
515          Y_0^0(\theta, \phi) &= \frac{1}{2} \sqrt{\frac{1}{\pi}} \\ 
516          Y_1^{-1}(\theta, \phi) &= \frac{1}{2} \sqrt{\frac{3}{2\pi}} 
517                                      e^{-i\phi} \sin(\theta) \\ 
518          Y_1^0(\theta, \phi) &= \frac{1}{2} \sqrt{\frac{3}{\pi}} 
519                                   \cos(\theta) \\ 
520          Y_1^1(\theta, \phi) &= -\frac{1}{2} \sqrt{\frac{3}{2\pi}} 
521                                   e^{i\phi} \sin(\theta). 
522   
523      References 
524      ---------- 
525      .. [1] Digital Library of Mathematical Functions, 14.30. 
526             https://dlmf.nist.gov/14.30 
527      .. [2] https://en.wikipedia.org/wiki/Spherical_harmonics#Condon.E2.80.93Shortley_phase 
528      """, force_complex_output=True, diff_n=0
529  )
530
