11   def test_native():
12       native_is_le = sys.byteorder == 'little'
13       assert_(sibc.sys_is_le == native_is_le)
14   
15   
16   def test_to_numpy():
17       if sys.byteorder == 'little':
18           assert_(sibc.to_numpy_code('native') == '<')
19           assert_(sibc.to_numpy_code('swapped') == '>')
20       else:
21           assert_(sibc.to_numpy_code('native') == '>')
22           assert_(sibc.to_numpy_code('swapped') == '<')
23       assert_(sibc.to_numpy_code('native') == sibc.to_numpy_code('='))
24       assert_(sibc.to_numpy_code('big') == '>')
25       for code in ('little', '<', 'l', 'L', 'le'):
26           assert_(sibc.to_numpy_code(code) == '<')
27       for code in ('big', '>', 'b', 'B', 'be'):
28           assert_(sibc.to_numpy_code(code) == '>')
29       assert_raises(ValueError, sibc.to_numpy_code, 'silly string')
