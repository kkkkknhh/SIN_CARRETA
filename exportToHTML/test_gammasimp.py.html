119  def test_issue_22606():
120      fx = Function('f')(x)
121      eq = x + gamma(y)
122      # seems like ans should be `eq`, not `(x*y + gamma(y + 1))/y`
123      ans = gammasimp(eq)
124      assert gammasimp(eq.subs(x, fx)).subs(fx, x) == ans
125      assert gammasimp(eq.subs(x, cos(x))).subs(cos(x), x) == ans
126      assert 1/gammasimp(1/eq) == ans
127      assert gammasimp(fx.subs(x, eq)).args[0] == ans
