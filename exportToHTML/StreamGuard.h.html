12   /** 
13    * A StreamGuard is an RAII class that changes the current device 
14    * to the device corresponding to some stream, and changes the 
15    * default stream on that device to be this stream. 
16    * 
17    * Use of StreamGuard is HIGHLY discouraged in operator definitions.  In 
18    * a single operator, you probably don't know enough about the global 
19    * state of the world to profitably decide how to set streams.  Let 
20    * the caller handle this appropriately, and just use the current stream 
21    * in your operator code. 
22    * 
23    * This StreamGuard does NOT have an uninitialized state; it is guaranteed 
24    * to reset the stream and device on exit.  If you are in a situation 
25    * where you *might* want to setup a stream guard, see OptionalStreamGuard. 
26    */
27   struct StreamGuard {
28     /// No default constructor, see Note [Omitted default constructor from RAII]
29     explicit StreamGuard() = delete;
30     ~StreamGuard() = default;
31   
32     /// Set the current device to the device associated with the passed stream,
33     /// and set the current  stream on that device to the passed stream.
34     explicit StreamGuard(Stream stream) : guard_(stream) {}
35   
36     /// Copy is disallowed
37     StreamGuard(const StreamGuard&) = delete;
38     StreamGuard& operator=(const StreamGuard&) = delete;
39   
40     /// Move is disallowed, as StreamGuard does not have an uninitialized state,
41     /// which is required for moves on types with nontrivial destructors.
42     StreamGuard(StreamGuard&& other) = delete;
43     StreamGuard& operator=(StreamGuard&& other) = delete;
44   
45     /// Resets the currently set stream to the original stream and
46     /// the currently set device to the original device.  Then,
47     /// set the current device to the device associated with the passed stream,
48     /// and set the current stream on that device to the passed stream.
49     ///
50     /// NOTE: this implementation may skip some stream/device setting if
51     /// it can prove that it is unnecessary.
52     ///
53     /// WARNING: reset_stream does NOT preserve previously set streams on
54     /// different devices.  If you need to set streams on multiple devices
55     /// on , use MultiStreamGuard instead.
56     void reset_stream(Stream stream) {
57       guard_.reset_stream(stream);
58     }
59   
60     /// Returns the stream that was set at the time the guard was constructed.
61     Stream original_stream() const {
62       return guard_.original_stream();
63     }
64   
65     /// Returns the most recent stream that was set using this device guard,
66     /// either from construction, or via set_stream.
67     Stream current_stream() const {
68       return guard_.current_stream();
69     }
70   
71     /// Returns the most recent device that was set using this device guard,
72     /// either from construction, or via set_device/reset_device/set_index.
73     Device current_device() const {
74       return guard_.current_device();
75     }
76   
77     /// Returns the device that was set at the most recent reset_stream(),
78     /// or otherwise the device at construction time.
79     Device original_device() const {
80       return guard_.original_device();
81     }
82   
83    private:
84     c10::impl::InlineStreamGuard<impl::VirtualGuardImpl> guard_;
85   };

87   /** 
88    * An OptionalStreamGuard is an RAII class that sets a device to some value on 
89    * initialization, and resets the device to its original value on destruction. 
90    * See OptionalDeviceGuard for more guidance on how to use this class. 
91    */
92   struct OptionalStreamGuard {
93     /// Create an uninitialized guard.
94     explicit OptionalStreamGuard() = default;
95   
96     /// Set the current device to the device associated with the passed stream,
97     /// and set the current stream on that device to the passed stream.
98     explicit OptionalStreamGuard(Stream stream) : guard_(stream) {}
99   
100    /// Set the current device to the device associated with the passed stream,
101    /// and set the current stream on that device to the passed stream,
102    /// if the passed stream is not nullopt.
103    explicit OptionalStreamGuard(std::optional<Stream> stream_opt)
104        : guard_(stream_opt) {}
105  
106    /// Copy is disallowed
107    OptionalStreamGuard(const OptionalStreamGuard&) = delete;
108    OptionalStreamGuard& operator=(const OptionalStreamGuard&) = delete;
109  
110    // See Note [Move construction for RAII guards is tricky]
111    OptionalStreamGuard(OptionalStreamGuard&& other) = delete;
112  
113    // See Note [Move assignment for RAII guards is tricky]
114    OptionalStreamGuard& operator=(OptionalStreamGuard&& other) = delete;
115    ~OptionalStreamGuard() = default;
116  
117    /// Resets the currently set stream to the original stream and
118    /// the currently set device to the original device.  Then,
119    /// set the current device to the device associated with the passed stream,
120    /// and set the current stream on that device to the passed stream.
121    /// Initializes the guard if it was not previously initialized.
122    void reset_stream(Stream stream) {
123      guard_.reset_stream(stream);
124    }
125  
126    /// Returns the stream that was set at the time the guard was most recently
127    /// initialized, or nullopt if the guard is uninitialized.
128    std::optional<Stream> original_stream() const {
129      return guard_.original_stream();
130    }
131  
132    /// Returns the most recent  stream that was set using this stream guard,
133    /// either from construction, or via reset_stream, if the guard is
134    /// initialized, or nullopt if the guard is uninitialized.
135    std::optional<Stream> current_stream() const {
136      return guard_.current_stream();
137    }
138  
139    /// Restore the original  device and stream, resetting this guard to
140    /// uninitialized state.
141    void reset() {
142      guard_.reset();
143    }
144  
145   private:
146    c10::impl::InlineOptionalStreamGuard<impl::VirtualGuardImpl> guard_{};
147  };

149  /** 
150   * A MultiStreamGuard is an RAII class that sets the current streams of a set of 
151   * devices all at once, and resets them to their original values on destruction. 
152   */
153  struct MultiStreamGuard {
154    /// Set the current streams to the passed streams on each of their respective
155    /// devices.
156    explicit MultiStreamGuard(ArrayRef<Stream> streams) : guard_(streams) {}
157  
158    /// Copy is disallowed
159    MultiStreamGuard(const MultiStreamGuard&) = delete;
160    MultiStreamGuard& operator=(const MultiStreamGuard&) = delete;
161  
162    // See Note [Move construction for RAII guards is tricky]
163    MultiStreamGuard(MultiStreamGuard&& other) = delete;
164  
165    // See Note [Move assignment for RAII guards is tricky]
166    MultiStreamGuard& operator=(MultiStreamGuard&& other) = delete;
167    ~MultiStreamGuard() = default;
168  
169   private:
170    c10::impl::InlineMultiStreamGuard<impl::VirtualGuardImpl> guard_;
171  };
