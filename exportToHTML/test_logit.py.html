83       def test_large(self):
84           for dtype in (np.float32, np.float64, np.longdouble):
85               for n in (88, 89, 709, 710, 11356, 11357):
86                   n = np.array(n, dtype=dtype)
87                   assert_allclose(expit(n), 1.0, atol=1e-20)
88                   assert_allclose(expit(-n), 0.0, atol=1e-20)
89                   assert_equal(expit(n).dtype, dtype)
90                   assert_equal(expit(-n).dtype, dtype)
