11   class Broadcast(Function):
12       @staticmethod
13       def forward(ctx, target_gpus, *inputs):
14           assert all(i.device.type != "cpu" for i in inputs), (
15               "Broadcast function not implemented for CPU tensors"
16           )
17           target_gpus = [_get_device_index(x, True) for x in target_gpus]
18           ctx.target_gpus = target_gpus
19           if len(inputs) == 0:
20               return ()
21           ctx.num_inputs = len(inputs)
22           ctx.input_device = inputs[0].get_device()
23           outputs = comm.broadcast_coalesced(inputs, ctx.target_gpus)
24           non_differentiables = []
25           for idx, input_requires_grad in enumerate(ctx.needs_input_grad[1:]):
26               if not input_requires_grad:
27                   non_differentiables.extend(output[idx] for output in outputs)
28           ctx.mark_non_differentiable(*non_differentiables)
29           return tuple(chain.from_iterable(outputs))
30   
31       @staticmethod
32       def backward(ctx, *grad_outputs):
33           return (None,) + ReduceAddCoalesced.apply(
34               ctx.input_device, ctx.num_inputs, *grad_outputs
35           )

38   class ReduceAddCoalesced(Function):
39       @staticmethod
40       def forward(ctx, destination, num_inputs, *grads):
41           ctx.target_gpus = [
42               grads[i].get_device() for i in range(0, len(grads), num_inputs)
43           ]
44   
45           grads_ = [grads[i : i + num_inputs] for i in range(0, len(grads), num_inputs)]
46           return comm.reduce_add_coalesced(grads_, destination)
47   
48       @staticmethod
49       def backward(ctx, *grad_outputs):
50           return (
51               None,
52               None,
53           ) + Broadcast.apply(ctx.target_gpus, *grad_outputs)

56   class Gather(Function):
57       @staticmethod
58       def forward(ctx, target_device, dim, *inputs):
59           assert all(i.device.type != "cpu" for i in inputs), (
60               "Gather function not implemented for CPU tensors"
61           )
62           if target_device == "cpu":
63               ctx.target_device = "cpu"
64           else:
65               target_device = _get_device_index(target_device, True)
66               ctx.target_device = target_device
67           ctx.dim = dim
68           ctx.input_gpus = tuple(i.get_device() for i in inputs)
69           if all(t.dim() == 0 for t in inputs) and dim == 0:
70               inputs = tuple(t.view(1) for t in inputs)
71               warnings.warn(
72                   "Was asked to gather along dimension 0, but all "
73                   "input tensors were scalars; will instead unsqueeze "
74                   "and return a vector."
75               )
76               ctx.unsqueezed_scalar = True
77           else:
78               ctx.unsqueezed_scalar = False
79           ctx.input_sizes = tuple(i.size(ctx.dim) for i in inputs)
80           return comm.gather(inputs, ctx.dim, ctx.target_device)
81   
82       @staticmethod
83       def backward(ctx, grad_output):
84           scattered_grads = Scatter.apply(
85               ctx.input_gpus, ctx.input_sizes, ctx.dim, grad_output
86           )
87           if ctx.unsqueezed_scalar:
88               scattered_grads = tuple(g[0] for g in scattered_grads)
89           return (None, None) + scattered_grads

92   class Scatter(Function):
93       @staticmethod
94       def forward(ctx, target_gpus, chunk_sizes, dim, input):
95           target_gpus = [_get_device_index(x, True) for x in target_gpus]
96           ctx.dim = dim
97           ctx.input_device = input.get_device() if input.device.type != "cpu" else -1
98           streams = None
99           if torch.accelerator.is_available() and ctx.input_device == -1:
100              # Perform CPU to GPU copies in a background stream
101              streams = [_get_stream(torch.device(device)) for device in target_gpus]
102          outputs = comm.scatter(input, target_gpus, chunk_sizes, ctx.dim, streams)
103          # Synchronize with the copy stream
104          if streams is not None:
105              for i, output in enumerate(outputs):
106                  with torch.accelerator.device_index(target_gpus[i]):
107                      main_stream = torch.accelerator.current_stream()
108                      main_stream.wait_stream(streams[i])
109                      output.record_stream(main_stream)
110          return outputs
111  
112      @staticmethod
113      def backward(ctx, *grad_output):
114          return None, None, None, Gather.apply(ctx.input_device, ctx.dim, *grad_output)

117  # background streams used for copying
118  _streams: Optional[list[Optional[torch.Stream]]] = None
119  
120  
121  def _get_stream(device: torch.device):
122      """Get a background stream for copying between CPU and target device."""
123      global _streams
124      if device.type == "cpu" or not torch.accelerator.is_available():
125          return None
126      assert torch.accelerator.current_accelerator().type == device.type
127      if _streams is None:
128          _streams = [None] * torch.accelerator.device_count()
129      if _streams[device.index] is None:
130          _streams[device.index] = torch.Stream(device.index)
131      return _streams[device.index]
