13           assert_(str(exc).find(strmatch) >= 0,
14                   "Did not raise floating point %s error" % strmatch)

16           assert_(False,

43       def test_half_conversions(self):
44           """Checks that all 16-bit values survive conversion 
45              to/from 32-bit and 64-bit float"""
46           # Because the underlying routines preserve the NaN bits, every
47           # value is preserved when converting to/from other floats.
48   
49           # Convert from float32 back to float16
50           with np.errstate(invalid='ignore'):
51               b = np.array(self.all_f32, dtype=float16)
52           # avoid testing NaNs due to differing bit patterns in Q/S NaNs
53           b_nn = b == b
54           assert_equal(self.all_f16[b_nn].view(dtype=uint16),
55                        b[b_nn].view(dtype=uint16))
56   
57           # Convert from float64 back to float16
58           with np.errstate(invalid='ignore'):
59               b = np.array(self.all_f64, dtype=float16)
60           b_nn = b == b
61           assert_equal(self.all_f16[b_nn].view(dtype=uint16),
62                        b[b_nn].view(dtype=uint16))
63   
64           # Convert float16 to longdouble and back
65           # This doesn't necessarily preserve the extra NaN bits,
66           # so exclude NaNs.
67           a_ld = np.array(self.nonan_f16, dtype=np.longdouble)
68           b = np.array(a_ld, dtype=float16)
69           assert_equal(self.nonan_f16.view(dtype=uint16),
70                        b.view(dtype=uint16))
71   
72           # Check the range for which all integers can be represented
73           i_int = np.arange(-2048, 2049)
74           i_f16 = np.array(i_int, dtype=float16)
75           j = np.array(i_f16, dtype=int)
76           assert_equal(i_int, j)

93       @pytest.mark.parametrize("offset", [None, "up", "down"])
94       @pytest.mark.parametrize("shift", [None, "up", "down"])
95       @pytest.mark.parametrize("float_t", [np.float32, np.float64])
96       def test_half_conversion_rounding(self, float_t, shift, offset):
97           # Assumes that round to even is used during casting.
98           max_pattern = np.float16(np.finfo(np.float16).max).view(np.uint16)
99   
100          # Test all (positive) finite numbers, denormals are most interesting
101          # however:
102          f16s_patterns = np.arange(0, max_pattern+1, dtype=np.uint16)
103          f16s_float = f16s_patterns.view(np.float16).astype(float_t)
104  
105          # Shift the values by half a bit up or a down (or do not shift),
106          if shift == "up":
107              f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[1:]
108          elif shift == "down":
109              f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[:-1]
110          else:
111              f16s_float = f16s_float[1:-1]
112  
113          # Increase the float by a minimal value:
114          if offset == "up":
115              f16s_float = np.nextafter(f16s_float, float_t(np.inf))
116          elif offset == "down":
117              f16s_float = np.nextafter(f16s_float, float_t(-np.inf))
118  
119          # Convert back to float16 and its bit pattern:
120          res_patterns = f16s_float.astype(np.float16).view(np.uint16)
121  
122          # The above calculation tries the original values, or the exact
123          # midpoints between the float16 values. It then further offsets them
124          # by as little as possible. If no offset occurs, "round to even"
125          # logic will be necessary, an arbitrarily small offset should cause
126          # normal up/down rounding always.
127  
128          # Calculate the expected pattern:
129          cmp_patterns = f16s_patterns[1:-1].copy()
130  
131          if shift == "down" and offset != "up":
132              shift_pattern = -1
133          elif shift == "up" and offset != "down":
134              shift_pattern = 1
135          else:
136              # There cannot be a shift, either shift is None, so all rounding
137              # will go back to original, or shift is reduced by offset too much.
138              shift_pattern = 0
139  
140          # If rounding occurs, is it normal rounding or round to even?
141          if offset is None:
142              # Round to even occurs, modify only non-even, cast to allow + (-1)
143              cmp_patterns[0::2].view(np.int16)[...] += shift_pattern
144          else:
145              cmp_patterns.view(np.int16)[...] += shift_pattern
146  
147          assert_equal(res_patterns, cmp_patterns)

149      @pytest.mark.parametrize(["float_t", "uint_t", "bits"],
150                               [(np.float32, np.uint32, 23),
151                                (np.float64, np.uint64, 52)])
152      def test_half_conversion_denormal_round_even(self, float_t, uint_t, bits):
153          # Test specifically that all bits are considered when deciding
154          # whether round to even should occur (i.e. no bits are lost at the
155          # end. Compare also gh-12721. The most bits can get lost for the
156          # smallest denormal:
157          smallest_value = np.uint16(1).view(np.float16).astype(float_t)
158          assert smallest_value == 2**-24
159  
160          # Will be rounded to zero based on round to even rule:
161          rounded_to_zero = smallest_value / float_t(2)
162          assert rounded_to_zero.astype(np.float16) == 0
163  
164          # The significand will be all 0 for the float_t, test that we do not
165          # lose the lower ones of these:
166          for i in range(bits):
167              # slightly increasing the value should make it round up:
168              larger_pattern = rounded_to_zero.view(uint_t) | uint_t(1 << i)
169              larger_value = larger_pattern.view(float_t)
170              assert larger_value.astype(np.float16) == smallest_value

254      def test_half_correctness(self):
255          """Take every finite float16, and check the casting functions with 
256             a manual conversion."""
257  
258          # Create an array of all finite float16s
259          a_bits = self.finite_f16.view(dtype=uint16)
260  
261          # Convert to 64-bit float manually
262          a_sgn = (-1.0)**((a_bits & 0x8000) >> 15)
263          a_exp = np.array((a_bits & 0x7c00) >> 10, dtype=np.int32) - 15
264          a_man = (a_bits & 0x03ff) * 2.0**(-10)
265          # Implicit bit of normalized floats
266          a_man[a_exp != -15] += 1
267          # Denormalized exponent is -14
268          a_exp[a_exp == -15] = -14
269  
270          a_manual = a_sgn * a_man * 2.0**a_exp
271  
272          a32_fail = np.nonzero(self.finite_f32 != a_manual)[0]
273          if len(a32_fail) != 0:
274              bad_index = a32_fail[0]
275              assert_equal(self.finite_f32, a_manual,
276                   "First non-equal is half value 0x%x -> %g != %g" %
277                              (a_bits[bad_index],
278                               self.finite_f32[bad_index],
279                               a_manual[bad_index]))
280  
281          a64_fail = np.nonzero(self.finite_f64 != a_manual)[0]
282          if len(a64_fail) != 0:
283              bad_index = a64_fail[0]
284              assert_equal(self.finite_f64, a_manual,
285                   "First non-equal is half value 0x%x -> %g != %g" %
286                              (a_bits[bad_index],
287                               self.finite_f64[bad_index],

353      def test_spacing_nextafter(self):
354          """Test np.spacing and np.nextafter"""
355          # All non-negative finite #'s
356          a = np.arange(0x7c00, dtype=uint16)
357          hinf = np.array((np.inf,), dtype=float16)
358          hnan = np.array((np.nan,), dtype=float16)
359          a_f16 = a.view(dtype=float16)
360  
361          assert_equal(np.spacing(a_f16[:-1]), a_f16[1:]-a_f16[:-1])
362  
363          assert_equal(np.nextafter(a_f16[:-1], hinf), a_f16[1:])
364          assert_equal(np.nextafter(a_f16[0], -hinf), -a_f16[1])
365          assert_equal(np.nextafter(a_f16[1:], -hinf), a_f16[:-1])
366  
367          assert_equal(np.nextafter(hinf, a_f16), a_f16[-1])
368          assert_equal(np.nextafter(-hinf, a_f16), -a_f16[-1])
369  
370          assert_equal(np.nextafter(hinf, hinf), hinf)
371          assert_equal(np.nextafter(hinf, -hinf), a_f16[-1])
372          assert_equal(np.nextafter(-hinf, hinf), -a_f16[-1])
373          assert_equal(np.nextafter(-hinf, -hinf), -hinf)
374  
375          assert_equal(np.nextafter(a_f16, hnan), hnan[0])
376          assert_equal(np.nextafter(hnan, a_f16), hnan[0])
377  
378          assert_equal(np.nextafter(hnan, hnan), hnan)
379          assert_equal(np.nextafter(hinf, hnan), hnan)
380          assert_equal(np.nextafter(hnan, hinf), hnan)
381  
382          # switch to negatives
383          a |= 0x8000
384  
385          assert_equal(np.spacing(a_f16[0]), np.spacing(a_f16[1]))
386          assert_equal(np.spacing(a_f16[1:]), a_f16[:-1]-a_f16[1:])
387

389          assert_equal(np.nextafter(a_f16[1:], hinf), a_f16[:-1])

492      def test_half_fpe(self):
493          with np.errstate(all='raise'):
494              sx16 = np.array((1e-4,), dtype=float16)
495              bx16 = np.array((1e4,), dtype=float16)
496              sy16 = float16(1e-4)
497              by16 = float16(1e4)
498  
499              # Underflow errors
500              assert_raises_fpe('underflow', lambda a, b:a*b, sx16, sx16)
501              assert_raises_fpe('underflow', lambda a, b:a*b, sx16, sy16)
502              assert_raises_fpe('underflow', lambda a, b:a*b, sy16, sx16)
503              assert_raises_fpe('underflow', lambda a, b:a*b, sy16, sy16)
504              assert_raises_fpe('underflow', lambda a, b:a/b, sx16, bx16)
505              assert_raises_fpe('underflow', lambda a, b:a/b, sx16, by16)
506              assert_raises_fpe('underflow', lambda a, b:a/b, sy16, bx16)
507              assert_raises_fpe('underflow', lambda a, b:a/b, sy16, by16)
508              assert_raises_fpe('underflow', lambda a, b:a/b,
509                                               float16(2.**-14), float16(2**11))
510              assert_raises_fpe('underflow', lambda a, b:a/b,
511                                               float16(-2.**-14), float16(2**11))
512              assert_raises_fpe('underflow', lambda a, b:a/b,
513                                               float16(2.**-14+2**-24), float16(2))
514              assert_raises_fpe('underflow', lambda a, b:a/b,
515                                               float16(-2.**-14-2**-24), float16(2))
516              assert_raises_fpe('underflow', lambda a, b:a/b,
517                                               float16(2.**-14+2**-23), float16(4))
518  
519              # Overflow errors
520              assert_raises_fpe('overflow', lambda a, b:a*b, bx16, bx16)
521              assert_raises_fpe('overflow', lambda a, b:a*b, bx16, by16)
522              assert_raises_fpe('overflow', lambda a, b:a*b, by16, bx16)
523              assert_raises_fpe('overflow', lambda a, b:a*b, by16, by16)
524              assert_raises_fpe('overflow', lambda a, b:a/b, bx16, sx16)
525              assert_raises_fpe('overflow', lambda a, b:a/b, bx16, sy16)
526              assert_raises_fpe('overflow', lambda a, b:a/b, by16, sx16)
527              assert_raises_fpe('overflow', lambda a, b:a/b, by16, sy16)
528              assert_raises_fpe('overflow', lambda a, b:a+b,
529                                               float16(65504), float16(17))
530              assert_raises_fpe('overflow', lambda a, b:a-b,
531                                               float16(-65504), float16(17))
532              assert_raises_fpe('overflow', np.nextafter, float16(65504), float16(np.inf))
533              assert_raises_fpe('overflow', np.nextafter, float16(-65504), float16(-np.inf))
534              assert_raises_fpe('overflow', np.spacing, float16(65504))
535  
536              # Invalid value errors
537              assert_raises_fpe('invalid', np.divide, float16(np.inf), float16(np.inf))
538              assert_raises_fpe('invalid', np.spacing, float16(np.inf))
539              assert_raises_fpe('invalid', np.spacing, float16(np.nan))
540  
541              # These should not raise
542              float16(65472)+float16(32)
543              float16(2**-13)/float16(2)
544              float16(2**-14)/float16(2**10)
545              np.spacing(float16(-65504))
546              np.nextafter(float16(65504), float16(-np.inf))
547              np.nextafter(float16(-65504), float16(np.inf))
548              np.nextafter(float16(np.inf), float16(0))
549              np.nextafter(float16(-np.inf), float16(0))
550              np.nextafter(float16(0), float16(np.nan))
551              np.nextafter(float16(np.nan), float16(0))
552              float16(2**-14)/float16(2**10)
553              float16(-2**-14)/float16(2**10)
554              float16(2**-14+2**-23)/float16(2)
555              float16(-2**-14-2**-23)/float16(2)
