5    /*                                                     ndtr.c 
6     * 
7     *     Normal distribution function 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, y, ndtr(); 
14    * 
15    * y = ndtr( x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Returns the area under the Gaussian probability density 
22    * function, integrated from minus infinity to x: 
23    * 
24    *                            x 
25    *                             - 
26    *                   1        | |          2 
27    *    ndtr(x)  = ---------    |    exp( - t /2 ) dt 
28    *               sqrt(2pi)  | | 
29    *                           - 
30    *                          -inf. 
31    * 
32    *             =  ( 1 + erf(z) ) / 2 
33    *             =  erfc(z) / 2 
34    * 
35    * where z = x/sqrt(2). Computation is via the functions 
36    * erf and erfc. 
37    * 
38    * 
39    * ACCURACY: 
40    * 
41    *                      Relative error: 
42    * arithmetic   domain     # trials      peak         rms 
43    *    IEEE     -13,0        30000       3.4e-14     6.7e-15 
44    * 
45    * 
46    * ERROR MESSAGES: 
47    * 
48    *   message         condition         value returned 
49    * erfc underflow    x > 37.519379347       0.0 
50    * 
51    */
52   /*                          erf.c 
53    * 
54    *  Error function 
55    * 
56    * 
57    * 
58    * SYNOPSIS: 
59    * 
60    * double x, y, erf(); 
61    * 
62    * y = erf( x ); 
63    * 
64    * 
65    * 
66    * DESCRIPTION: 
67    * 
68    * The integral is 
69    * 
70    *                           x 
71    *                            - 
72    *                 2         | |          2 
73    *   erf(x)  =  --------     |    exp( - t  ) dt. 
74    *              sqrt(pi)   | | 
75    *                          - 
76    *                           0 
77    * 
78    * For 0 <= |x| < 1, erf(x) = x * P4(x**2)/Q5(x**2); otherwise 
79    * erf(x) = 1 - erfc(x). 
80    * 
81    * 
82    * 
83    * ACCURACY: 
84    * 
85    *                      Relative error: 
86    * arithmetic   domain     # trials      peak         rms 
87    *    IEEE      0,1         30000       3.7e-16     1.0e-16 
88    * 
89    */
90   /*                          erfc.c 
91    * 
92    *  Complementary error function 
93    * 
94    * 
95    * 
96    * SYNOPSIS: 
97    * 
98    * double x, y, erfc(); 
99    * 
100   * y = erfc( x ); 
101   * 
102   * 
103   * 
104   * DESCRIPTION: 
105   * 
106   * 
107   *  1 - erf(x) = 
108   * 
109   *                           inf. 
110   *                             - 
111   *                  2         | |          2 
112   *   erfc(x)  =  --------     |    exp( - t  ) dt 
113   *               sqrt(pi)   | | 
114   *                           - 
115   *                            x 
116   * 
117   * 
118   * For small x, erfc(x) = 1 - erf(x); otherwise rational 
119   * approximations are computed. 
120   * 
121   * 
122   * 
123   * ACCURACY: 
124   * 
125   *                      Relative error: 
126   * arithmetic   domain     # trials      peak         rms 
127   *    IEEE      0,26.6417   30000       5.7e-14     1.5e-14 
128   */

178      XSF_HOST_DEVICE inline double erfc(double a) {
179          double p, q, x, y, z;
180  
181          if (std::isnan(a)) {
182              set_error("erfc", SF_ERROR_DOMAIN, NULL);
183              return std::numeric_limits<double>::quiet_NaN();
184          }
185  
186          if (a < 0.0) {
187              x = -a;
188          } else {
189              x = a;
190          }
191  
192          if (x < 1.0) {
193              return 1.0 - erf(a);
194          }
195  
196          z = -a * a;
197  
198          if (z < -detail::MAXLOG) {
199              goto under;
200          }
201  
202          z = std::exp(z);
203  
204          if (x < 8.0) {
205              p = polevl(x, detail::ndtr_P, 8);
206              q = p1evl(x, detail::ndtr_Q, 8);
207          } else {
208              p = polevl(x, detail::ndtr_R, 5);
209              q = p1evl(x, detail::ndtr_S, 6);
210          }
211          y = (z * p) / q;
212  
213          if (a < 0) {
214              y = 2.0 - y;
215          }
216  
217          if (y != 0.0) {
218              return y;
219          }
220  
221      under:
222          set_error("erfc", SF_ERROR_UNDERFLOW, NULL);
223          if (a < 0) {
224              return 2.0;
225          } else {
226              return 0.0;
227          }
228      }

230      XSF_HOST_DEVICE inline double erf(double x) {
231          double y, z;
232  
233          if (std::isnan(x)) {
234              set_error("erf", SF_ERROR_DOMAIN, NULL);
235              return std::numeric_limits<double>::quiet_NaN();
236          }
237  
238          if (x < 0.0) {
239              return -erf(-x);
240          }
241  
242          if (std::abs(x) > 1.0) {
243              return (1.0 - erfc(x));
244          }
245          z = x * x;
246  
247          y = x * polevl(z, detail::ndtr_T, 4) / p1evl(z, detail::ndtr_U, 5);
248          return y;
249      }

251      XSF_HOST_DEVICE inline double ndtr(double a) {
252          double x, y, z;
253  
254          if (std::isnan(a)) {
255              set_error("ndtr", SF_ERROR_DOMAIN, NULL);
256              return std::numeric_limits<double>::quiet_NaN();
257          }
258  
259          x = a * M_SQRT1_2;
260          z = std::abs(x);
261  
262          if (z < 1.0) {
263              y = 0.5 + 0.5 * erf(x);
264          } else {
265              y = 0.5 * erfc(z);
266              if (x > 0) {
267                  y = 1.0 - y;
268              }
269          }
270  
271          return y;
272      }
