130          matches = matcher.find_adversatives(text)

136      def test_competence_verb_extraction(matcher):
137          """Test extraction of competence-sensitive verbs."""
138          text = """ 
139          El municipio procederá a contratar docentes para las escuelas. 
140          También construirá un hospital de alta complejidad. 
141          """
142  
143          verbs = matcher.extract_competence_verbs(text)
144  
145          assert len(verbs) > 0
146          # Should find "contratar docentes" and "construir hospital"
147          verb_texts = [v[0] for v in verbs]
148          assert any("contratar" in v and "docentes" in v for v in verb_texts)
149          assert any("construi" in v and "hospital" in v for v in verb_texts)
150  
151      @staticmethod
152      def test_quantitative_pattern_detection(matcher):
153          """Test detection of quantitative patterns."""
154          text = """ 
155          La meta es alcanzar 95% de cobertura. 
156          Se invertirán $500 millones COP. 
157          Beneficiará a 10,000 familias. 
158          """
159  
160          matches = matcher.find_adversatives(text, context_window=500)
161          # Even without adversatives, we can check quantitative extraction
162  
163          # Test direct pattern matching
164          found = []
165          for pattern in matcher.quantitative:
166              for match in pattern.finditer(text):
167                  found.append(match.group())
168  
169          assert len(found) > 0
170          assert any("95%" in f for f in found)
171          assert any("500 millones" in f for f in found)
172          assert any(
173              "10,000 familias" in f or "10.000 familias" in f for f in found)

254      @pytest.fixture
255      def scorer(self):
256          """Create risk scorer instance."""
257          return RiskScorer(alpha=0.1)
258  
259      @staticmethod
260      def test_basic_risk_calculation(scorer):
261          """Test basic risk calculation."""
262          # Mock contradiction data
263          contradictions = [
264              type("Contradiction", (), {"confidence": 0.8, "type": "semantic"}),
265              type("Contradiction", (), {
266                   "confidence": 0.6, "type": "quantitative"}),
267          ]
268  
269          result = scorer.calculate_risk(contradictions, [], [])
270  
271          assert "overall_risk" in result
272          assert 0 <= result["overall_risk"] <= 1
273          assert "risk_level" in result
274          assert "confidence_intervals" in result
275  
276      @staticmethod
277      def test_confidence_intervals(scorer):
278          """Test confidence interval calculation."""
279          # Add some calibration data
280          for _ in range(30):
281              scorer.add_calibration_sample(
282                  np.random.random(),
283                  np.random.random(),
284                  np.random.random(),
285                  np.random.random(),
286              )
287  
288          result = scorer.calculate_risk([], [], [])
289  
290          intervals = result["confidence_intervals"]
291          assert "overall" in intervals
292          assert len(intervals["overall"]) == 2
293          assert intervals["overall"][0] <= intervals["overall"][1]
294  
295      @staticmethod
296      def test_risk_levels(scorer):
297          """Test risk level assignment."""
298          # Test different severity levels
299          test_cases = [
300              ([], [], [], "LOW"),  # No issues
301              ([type("C", (), {"confidence": 0.3})],
302               [], [], "LOW"),  # Low confidence
303              (
304                  [type("C", (), {"confidence": 0.9})] * 5,
305                  [],
306                  [],
307                  "HIGH",
308              ),  # Many high confidence
309          ]
310  
311          for contradictions, competences, agenda, expected_level in test_cases:
312              result = scorer.calculate_risk(contradictions, competences, agenda)
313              # Allow for some flexibility in level assignment
314              assert expected_level in result["risk_level"] or result["risk_level"] in [
315                  "LOW",
316                  "MEDIUM",
317                  "MEDIUM_HIGH",
318                  "HIGH",
319                  "CRITICAL",
320              ]

409          # Run analysis
410          analysis = detector.detect_contradictions(
411              text=doc_data["text"],

413              pdm_structure={"sections": doc_data.get("sections", {})},
414          )

416          # Verify results

418          assert (

421          assert analysis.risk_score > 0
