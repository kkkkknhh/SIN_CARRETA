2    Abstract base class for the various polynomial Classes. 
3     
4    The ABCPolyBase class provides the methods needed to implement the common API 
5    for the various polynomial classes. It operates as a mixin, but uses the 
6    abc module from the stdlib, hence it is only available for Python >= 2.6. 
7     
8    """

19   class ABCPolyBase(abc.ABC):
20       """An abstract base class for immutable series classes. 
21    
22       ABCPolyBase provides the standard Python numerical methods 
23       '+', '-', '*', '//', '%', 'divmod', '**', and '()' along with the 
24       methods listed below. 
25    
26       Parameters 
27       ---------- 
28       coef : array_like 
29           Series coefficients in order of increasing degree, i.e., 
30           ``(1, 2, 3)`` gives ``1*P_0(x) + 2*P_1(x) + 3*P_2(x)``, where 
31           ``P_i`` is the basis polynomials of degree ``i``. 
32       domain : (2,) array_like, optional 
33           Domain to use. The interval ``[domain[0], domain[1]]`` is mapped 
34           to the interval ``[window[0], window[1]]`` by shifting and scaling. 
35           The default value is the derived class domain. 
36       window : (2,) array_like, optional 
37           Window, see domain for its use. The default value is the 
38           derived class window. 
39       symbol : str, optional 
40           Symbol used to represent the independent variable in string 
41           representations of the polynomial expression, e.g. for printing. 
42           The symbol must be a valid Python identifier. Default value is 'x'. 
43    
44           .. versionadded:: 1.24 
45    
46       Attributes 
47       ---------- 
48       coef : (N,) ndarray 
49           Series coefficients in order of increasing degree. 
50       domain : (2,) ndarray 
51           Domain that is mapped to window. 
52       window : (2,) ndarray 
53           Window that domain is mapped to. 
54       symbol : str 
55           Symbol representing the independent variable. 
56    
57       Class Attributes 
58       ---------------- 
59       maxpower : int 
60           Maximum power allowed, i.e., the largest number ``n`` such that 
61           ``p(x)**n`` is allowed. This is to limit runaway polynomial size. 
62       domain : (2,) ndarray 
63           Default domain of the class. 
64       window : (2,) ndarray 
65           Default window of the class. 
66    
67       """
68

188      def has_samecoef(self, other):
189          """Check if coefficients match. 
190   
191          Parameters 
192          ---------- 
193          other : class instance 
194              The other class must have the ``coef`` attribute. 
195   
196          Returns 
197          ------- 
198          bool : boolean 
199              True if the coefficients are the same, False otherwise. 
200   
201          """
202          if len(self.coef) != len(other.coef):
203              return False
204          elif not np.all(self.coef == other.coef):
205              return False
206          else:
207              return True

257      def _get_coefficients(self, other):
258          """Interpret other as polynomial coefficients. 
259   
260          The `other` argument is checked to see if it is of the same 
261          class as self with identical domain and window. If so, 
262          return its coefficients, otherwise return `other`. 
263   
264          Parameters 
265          ---------- 
266          other : anything 
267              Object to be checked. 
268   
269          Returns 
270          ------- 
271          coef 
272              The coefficients of`other` if it is a compatible instance, 
273              of ABCPolyBase, otherwise `other`. 
274   
275          Raises 
276          ------ 
277          TypeError 
278              When `other` is an incompatible instance of ABCPolyBase. 
279   
280          """
281          if isinstance(other, ABCPolyBase):
282              if not isinstance(other, self.__class__):
283                  raise TypeError("Polynomial types differ")
284              elif not np.all(self.domain == other.domain):
285                  raise TypeError("Domains differ")
286              elif not np.all(self.window == other.window):
287                  raise TypeError("Windows differ")
288              elif self.symbol != other.symbol:
289                  raise ValueError("Polynomial symbols differ")
290              return other.coef
291          return other

293      def __init__(self, coef, domain=None, window=None, symbol='x'):
294          [coef] = pu.as_series([coef], trim=False)
295          self.coef = coef
296  
297          if domain is not None:
298              [domain] = pu.as_series([domain], trim=False)
299              if len(domain) != 2:
300                  raise ValueError("Domain has wrong number of elements.")
301              self.domain = domain
302  
303          if window is not None:
304              [window] = pu.as_series([window], trim=False)
305              if len(window) != 2:
306                  raise ValueError("Window has wrong number of elements.")
307              self.window = window
308  
309          # Validation for symbol
310          try:
311              if not symbol.isidentifier():
312                  raise ValueError(
313                      "Symbol string must be a valid Python identifier"
314                  )
315          # If a user passes in something other than a string, the above
316          # results in an AttributeError. Catch this and raise a more
317          # informative exception
318          except AttributeError:
319              raise TypeError("Symbol must be a non-empty string")
320  
321          self._symbol = symbol

331      def __format__(self, fmt_str):
332          if fmt_str == '':
333              return self.__str__()
334          if fmt_str not in ('ascii', 'unicode'):
335              raise ValueError(
336                  f"Unsupported format string '{fmt_str}' passed to "
337                  f"{self.__class__}.__format__. Valid options are "
338                  f"'ascii' and 'unicode'"
339              )
340          if fmt_str == 'ascii':
341              return self._generate_string(self._str_term_ascii)
342          return self._generate_string(self._str_term_unicode)

344      def __str__(self):
345          if self._use_unicode:
346              return self._generate_string(self._str_term_unicode)
347          return self._generate_string(self._str_term_ascii)

349      def _generate_string(self, term_method):
350          """ 
351          Generate the full string representation of the polynomial, using 
352          ``term_method`` to generate each polynomial term. 
353          """
354          # Get configuration for line breaks
355          linewidth = np.get_printoptions().get('linewidth', 75)
356          if linewidth < 1:
357              linewidth = 1
358          out = pu.format_float(self.coef[0])
359  
360          off, scale = self.mapparms()
361  
362          scaled_symbol, needs_parens = self._format_term(pu.format_float,
363                                                          off, scale)
364          if needs_parens:
365              scaled_symbol = '(' + scaled_symbol + ')'
366  
367          for i, coef in enumerate(self.coef[1:]):
368              out += " "
369              power = str(i + 1)
370              # Polynomial coefficient
371              # The coefficient array can be an object array with elements that
372              # will raise a TypeError with >= 0 (e.g. strings or Python
373              # complex). In this case, represent the coefficient as-is.
374              try:
375                  if coef >= 0:
376                      next_term = "+ " + pu.format_float(coef, parens=True)
377                  else:
378                      next_term = "- " + pu.format_float(-coef, parens=True)
379              except TypeError:
380                  next_term = f"+ {coef}"
381              # Polynomial term
382              next_term += term_method(power, scaled_symbol)
383              # Length of the current line with next term added
384              line_len = len(out.split('\n')[-1]) + len(next_term)
385              # If not the last term in the polynomial, it will be two
386              # characters longer due to the +/- with the next term
387              if i < len(self.coef[1:]) - 1:
388                  line_len += 2
389              # Handle linebreaking
390              if line_len >= linewidth:
391                  next_term = next_term.replace(" ", "\n", 1)
392              out += next_term
393          return out

395      @classmethod
396      def _str_term_unicode(cls, i, arg_str):
397          """ 
398          String representation of single polynomial term using unicode 
399          characters for superscripts and subscripts. 
400          """
401          if cls.basis_name is None:
402              raise NotImplementedError(
403                  "Subclasses must define either a basis_name, or override "
404                  "_str_term_unicode(cls, i, arg_str)"
405              )
406          return (f"·{cls.basis_name}{i.translate(cls._subscript_mapping)}"
407                  f"({arg_str})")

422      @classmethod
423      def _repr_latex_term(cls, i, arg_str, needs_parens):
424          if cls.basis_name is None:

426                  "Subclasses must define either a basis name, or override "
427                  "_repr_latex_term(i, arg_str, needs_parens)")

431      @staticmethod
432      def _repr_latex_scalar(x, parens=False):
433          # TODO: we're stuck with disabling math formatting until we handle
434          # exponents in this function
435          return r'\text{{{}}}'.format(pu.format_float(x, parens=parens))

437      def _format_term(self, scalar_format: Callable, off: float, scale: float):
438          """ Format a single term in the expansion """
439          if off == 0 and scale == 1:
440              term = self.symbol
441              needs_parens = False
442          elif scale == 1:
443              term = f"{scalar_format(off)} + {self.symbol}"
444              needs_parens = True
445          elif off == 0:
446              term = f"{scalar_format(scale)}{self.symbol}"
447              needs_parens = True
448          else:
449              term = (
450                  f"{scalar_format(off)} + "
451                  f"{scalar_format(scale)}{self.symbol}"
452              )
453              needs_parens = True
454          return term, needs_parens

456      def _repr_latex_(self):
457          # get the scaled argument string to the basis functions
458          off, scale = self.mapparms()
459          term, needs_parens = self._format_term(self._repr_latex_scalar,
460                                                 off, scale)
461  
462          mute = r"\color{{LightGray}}{{{}}}".format
463  
464          parts = []
465          for i, c in enumerate(self.coef):
466              # prevent duplication of + and - signs
467              if i == 0:
468                  coef_str = f"{self._repr_latex_scalar(c)}"
469              elif not isinstance(c, numbers.Real):
470                  coef_str = f" + ({self._repr_latex_scalar(c)})"
471              elif c >= 0:
472                  coef_str = f" + {self._repr_latex_scalar(c, parens=True)}"
473              else:
474                  coef_str = f" - {self._repr_latex_scalar(-c, parens=True)}"
475  
476              # produce the string for the term
477              term_str = self._repr_latex_term(i, term, needs_parens)
478              if term_str == '1':
479                  part = coef_str
480              else:
481                  part = rf"{coef_str}\,{term_str}"
482  
483              if c == 0:
484                  part = mute(part)
485  
486              parts.append(part)
487  
488          if parts:
489              body = ''.join(parts)
490          else:
491              # in case somehow there are no coefficients at all
492              body = '0'
493  
494          return rf"${self.symbol} \mapsto {body}$"

557      def __truediv__(self, other):
558          # there is no true divide if the rhs is not a Number, although it
559          # could return the first n elements of an infinite series.
560          # It is hard to see where n would come from, though.
561          if not isinstance(other, numbers.Number) or isinstance(other, bool):
562              raise TypeError(
563                  f"unsupported types for true division: "
564                  f"'{type(self)}', '{type(other)}'"
565              )
566          return self.__floordiv__(other)

580      def __divmod__(self, other):
581          othercoef = self._get_coefficients(other)
582          try:
583              quo, rem = self._div(self.coef, othercoef)

585              raise

677      def degree(self):
678          """The degree of the series. 
679   
680          Returns 
681          ------- 
682          degree : int 
683              Degree of the series, one less than the number of coefficients. 
684   
685          Examples 
686          -------- 
687   
688          Create a polynomial object for ``1 + 7*x + 4*x**2``: 
689   
690          >>> poly = np.polynomial.Polynomial([1, 7, 4]) 
691          >>> print(poly) 
692          1.0 + 7.0·x + 4.0·x² 
693          >>> poly.degree() 
694          2 
695   
696          Note that this method does not check for non-zero coefficients. 
697          You must trim the polynomial to remove any trailing zeroes: 
698   
699          >>> poly = np.polynomial.Polynomial([1, 7, 0]) 
700          >>> print(poly) 
701          1.0 + 7.0·x + 0.0·x² 
702          >>> poly.degree() 
703          2 
704          >>> poly.trim().degree() 
705          1 
706   
707          """
708          return len(self) - 1

710      def cutdeg(self, deg):
711          """Truncate series to the given degree. 
712   
713          Reduce the degree of the series to `deg` by discarding the 
714          high order terms. If `deg` is greater than the current degree a 
715          copy of the current series is returned. This can be useful in least 
716          squares where the coefficients of the high degree terms may be very 
717          small. 
718   
719          Parameters 
720          ---------- 
721          deg : non-negative int 
722              The series is reduced to degree `deg` by discarding the high 
723              order terms. The value of `deg` must be a non-negative integer. 
724   
725          Returns 
726          ------- 
727          new_series : series 
728              New instance of series with reduced degree. 
729   
730          """
731          return self.truncate(deg + 1)

733      def trim(self, tol=0):
734          """Remove trailing coefficients 
735   
736          Remove trailing coefficients until a coefficient is reached whose 
737          absolute value greater than `tol` or the beginning of the series is 
738          reached. If all the coefficients would be removed the series is set 
739          to ``[0]``. A new series instance is returned with the new 
740          coefficients.  The current instance remains unchanged. 
741   
742          Parameters 
743          ---------- 
744          tol : non-negative number. 
745              All trailing coefficients less than `tol` will be removed. 
746   
747          Returns 
748          ------- 
749          new_series : series 
750              New instance of series with trimmed coefficients. 
751   
752          """
753          coef = pu.trimcoef(self.coef, tol)
754          return self.__class__(coef, self.domain, self.window, self.symbol)

756      def truncate(self, size):
757          """Truncate series to length `size`. 
758   
759          Reduce the series to length `size` by discarding the high 
760          degree terms. The value of `size` must be a positive integer. This 
761          can be useful in least squares where the coefficients of the 
762          high degree terms may be very small. 
763   
764          Parameters 
765          ---------- 
766          size : positive int 
767              The series is reduced to length `size` by discarding the high 
768              degree terms. The value of `size` must be a positive integer. 
769   
770          Returns 
771          ------- 
772          new_series : series 
773              New instance of series with truncated coefficients. 
774   
775          """
776          isize = int(size)
777          if isize != size or isize < 1:
778              raise ValueError("size must be a positive integer")
779          if isize >= len(self.coef):
780              coef = self.coef
781          else:
782              coef = self.coef[:isize]
783          return self.__class__(coef, self.domain, self.window, self.symbol)

785      def convert(self, domain=None, kind=None, window=None):
786          """Convert series to a different kind and/or domain and/or window. 
787   
788          Parameters 
789          ---------- 
790          domain : array_like, optional 
791              The domain of the converted series. If the value is None, 
792              the default domain of `kind` is used. 
793          kind : class, optional 
794              The polynomial series type class to which the current instance 
795              should be converted. If kind is None, then the class of the 
796              current instance is used. 
797          window : array_like, optional 
798              The window of the converted series. If the value is None, 
799              the default window of `kind` is used. 
800   
801          Returns 
802          ------- 
803          new_series : series 
804              The returned class can be of different type than the current 
805              instance and/or have a different domain and/or different 
806              window. 
807   
808          Notes 
809          ----- 
810          Conversion between domains and class types can result in 
811          numerically ill defined series. 
812   
813          """
814          if kind is None:
815              kind = self.__class__
816          if domain is None:
817              domain = kind.domain
818          if window is None:
819              window = kind.window
820          return self(kind.identity(domain, window=window, symbol=self.symbol))

822      def mapparms(self):
823          """Return the mapping parameters. 
824   
825          The returned values define a linear map ``off + scl*x`` that is 
826          applied to the input arguments before the series is evaluated. The 
827          map depends on the ``domain`` and ``window``; if the current 
828          ``domain`` is equal to the ``window`` the resulting map is the 
829          identity.  If the coefficients of the series instance are to be 
830          used by themselves outside this class, then the linear function 
831          must be substituted for the ``x`` in the standard representation of 
832          the base polynomials. 
833   
834          Returns 
835          ------- 
836          off, scl : float or complex 
837              The mapping function is defined by ``off + scl*x``. 
838   
839          Notes 
840          ----- 
841          If the current domain is the interval ``[l1, r1]`` and the window 
842          is ``[l2, r2]``, then the linear mapping function ``L`` is 
843          defined by the equations:: 
844   
845              L(l1) = l2 
846              L(r1) = r2 
847   
848          """
849          return pu.mapparms(self.domain, self.window)

851      def integ(self, m=1, k=[], lbnd=None):
852          """Integrate. 
853   
854          Return a series instance that is the definite integral of the 
855          current series. 
856   
857          Parameters 
858          ---------- 
859          m : non-negative int 
860              The number of integrations to perform. 
861          k : array_like 
862              Integration constants. The first constant is applied to the 
863              first integration, the second to the second, and so on. The 
864              list of values must less than or equal to `m` in length and any 
865              missing values are set to zero. 
866          lbnd : Scalar 
867              The lower bound of the definite integral. 
868   
869          Returns 
870          ------- 
871          new_series : series 
872              A new series representing the integral. The domain is the same 
873              as the domain of the integrated series. 
874   
875          """
876          off, scl = self.mapparms()
877          if lbnd is None:
878              lbnd = 0
879          else:
880              lbnd = off + scl*lbnd
881          coef = self._int(self.coef, m, k, lbnd, 1./scl)
882          return self.__class__(coef, self.domain, self.window, self.symbol)

886   
887          Return a series instance of that is the derivative of the current

890          Parameters

894

921      def linspace(self, n=100, domain=None):
922          """Return x, y values at equally spaced points in domain. 
923   
924          Returns the x, y values at `n` linearly spaced points across the 
925          domain.  Here y is the value of the polynomial at the points x. By 
926          default the domain is the same as that of the series instance. 
927          This method is intended mostly as a plotting aid. 
928   
929          Parameters 
930          ---------- 
931          n : int, optional 
932              Number of point pairs to return. The default value is 100. 
933          domain : {None, array_like}, optional

937   
938          Returns 
939          ------- 
940          x, y : ndarray

942              y is the series evaluated at element of x. 
943   
944          """
945          if domain is None:
946              domain = self.domain
947          x = np.linspace(domain[0], domain[1], n)
948          y = self(x)
949          return x, y

951      @classmethod
952      def fit(cls, x, y, deg, domain=None, rcond=None, full=False, w=None,
953          window=None, symbol='x'):
954          """Least squares fit to data. 
955   
956          Return a series instance that is the least squares fit to the data 
957          `y` sampled at `x`. The domain of the returned instance can be 
958          specified and this will often result in a superior fit with less 
959          chance of ill conditioning. 
960   
961          Parameters 
962          ---------- 
963          x : array_like, shape (M,) 
964              x-coordinates of the M sample points ``(x[i], y[i])``. 
965          y : array_like, shape (M,) 
966              y-coordinates of the M sample points ``(x[i], y[i])``. 
967          deg : int or 1-D array_like 
968              Degree(s) of the fitting polynomials. If `deg` is a single integer 
969              all terms up to and including the `deg`'th term are included in the 
970              fit. For NumPy versions >= 1.11.0 a list of integers specifying the 
971              degrees of the terms to include may be used instead. 
972          domain : {None, [beg, end], []}, optional 
973              Domain to use for the returned series. If ``None``, 
974              then a minimal domain that covers the points `x` is chosen.  If 
975              ``[]`` the class domain is used. The default value was the 
976              class domain in NumPy 1.4 and ``None`` in later versions. 
977              The ``[]`` option was added in numpy 1.5.0. 
978          rcond : float, optional 
979              Relative condition number of the fit. Singular values smaller 
980              than this relative to the largest singular value will be 
981              ignored. The default value is ``len(x)*eps``, where eps is the 
982              relative precision of the float type, about 2e-16 in most 
983              cases. 
984          full : bool, optional 
985              Switch determining nature of return value. When it is False 
986              (the default) just the coefficients are returned, when True 
987              diagnostic information from the singular value decomposition is 
988              also returned. 
989          w : array_like, shape (M,), optional 
990              Weights. If not None, the weight ``w[i]`` applies to the unsquared 
991              residual ``y[i] - y_hat[i]`` at ``x[i]``. Ideally the weights are 
992              chosen so that the errors of the products ``w[i]*y[i]`` all have 
993              the same variance.  When using inverse-variance weighting, use 
994              ``w[i] = 1/sigma(y[i])``.  The default value is None. 
995          window : {[beg, end]}, optional 
996              Window to use for the returned series. The default 
997              value is the default class domain 
998          symbol : str, optional 
999              Symbol representing the independent variable. Default is 'x'. 
1000  
1001         Returns 
1002         ------- 
1003         new_series : series 
1004             A series that represents the least squares fit to the data and 
1005             has the domain and window specified in the call. If the 
1006             coefficients for the unscaled and unshifted basis polynomials are 
1007             of interest, do ``new_series.convert().coef``. 
1008  
1009         [resid, rank, sv, rcond] : list 
1010             These values are only returned if ``full == True`` 
1011  
1012             - resid -- sum of squared residuals of the least squares fit 
1013             - rank -- the numerical rank of the scaled Vandermonde matrix 
1014             - sv -- singular values of the scaled Vandermonde matrix 
1015             - rcond -- value of `rcond`. 
1016  
1017             For more details, see `linalg.lstsq`. 
1018  
1019         """
1020         if domain is None:
1021             domain = pu.getdomain(x)
1022             if domain[0] == domain[1]:
1023                 domain[0] -= 1
1024                 domain[1] += 1
1025         elif type(domain) is list and len(domain) == 0:
1026             domain = cls.domain
1027 
1028         if window is None:
1029             window = cls.window
1030 
1031         xnew = pu.mapdomain(x, domain, window)
1032         res = cls._fit(xnew, y, deg, w=w, rcond=rcond, full=full)
1033         if full:
1034             [coef, status] = res
1035             return (
1036                 cls(coef, domain=domain, window=window, symbol=symbol), status
1037             )
1038         else:
1039             coef = res
1040             return cls(coef, domain=domain, window=window, symbol=symbol)

1042     @classmethod
1043     def fromroots(cls, roots, domain=[], window=None, symbol='x'):
1044         """Return series instance that has the specified roots. 
1045  
1046         Returns a series representing the product 
1047         ``(x - r[0])*(x - r[1])*...*(x - r[n-1])``, where ``r`` is a 
1048         list of roots. 
1049  
1050         Parameters 
1051         ---------- 
1052         roots : array_like 
1053             List of roots. 
1054         domain : {[], None, array_like}, optional 
1055             Domain for the resulting series. If None the domain is the 
1056             interval from the smallest root to the largest. If [] the 
1057             domain is the class domain. The default is []. 
1058         window : {None, array_like}, optional 
1059             Window for the returned series. If None the class window is 
1060             used. The default is None. 
1061         symbol : str, optional 
1062             Symbol representing the independent variable. Default is 'x'. 
1063  
1064         Returns 
1065         ------- 
1066         new_series : series 
1067             Series with the specified roots. 
1068  
1069         """
1070         [roots] = pu.as_series([roots], trim=False)
1071         if domain is None:
1072             domain = pu.getdomain(roots)
1073         elif type(domain) is list and len(domain) == 0:
1074             domain = cls.domain
1075 
1076         if window is None:
1077             window = cls.window
1078 
1079         deg = len(roots)
1080         off, scl = pu.mapparms(domain, window)
1081         rnew = off + scl*roots
1082         coef = cls._fromroots(rnew) / scl**deg
1083         return cls(coef, domain=domain, window=window, symbol=symbol)
