193  @pytest.mark.parametrize('win_params, Nx', [(('gaussian', 2.), 9),  # in docstr
194                                              ('triang', 7),
195                                              (('kaiser', 4.0), 9),
196                                              (('exponential', None, 1.), 9),
197                                              (4.0, 9)])
198  def test_from_window(win_params, Nx: int):
199      """Verify that `from_window()` handles parameters correctly. 
200   
201      The window parameterizations are documented in the `get_window` docstring. 
202      """
203      w_sym, fs = get_window(win_params, Nx, fftbins=False), 16.
204      w_per = get_window(win_params, Nx, fftbins=True)
205      SFT0 = ShortTimeFFT(w_sym, hop=3, fs=fs, fft_mode='twosided',
206                          scale_to='psd', phase_shift=1)
207      nperseg = len(w_sym)
208      noverlap = nperseg - SFT0.hop
209      SFT1 = ShortTimeFFT.from_window(win_params, fs, nperseg, noverlap,
210                                      symmetric_win=True, fft_mode='twosided',
211                                      scale_to='psd', phase_shift=1)
212      # periodic window:
213      SFT2 = ShortTimeFFT.from_window(win_params, fs, nperseg, noverlap,
214                                      symmetric_win=False, fft_mode='twosided',
215                                      scale_to='psd', phase_shift=1)
216      # Be informative when comparing instances:
217      xp_assert_equal(SFT1.win, SFT0.win)
218      xp_assert_close(SFT2.win, w_per / np.sqrt(sum(w_per**2) * fs))
219      for n_ in ('hop', 'T', 'fft_mode', 'mfft', 'scaling', 'phase_shift'):
220          v0, v1, v2 = (getattr(SFT_, n_) for SFT_ in (SFT0, SFT1, SFT2))
221          assert v1 == v0, f"SFT1.{n_}={v1} does not equal SFT0.{n_}={v0}"
222          assert v2 == v0, f"SFT2.{n_}={v2} does not equal SFT0.{n_}={v0}"

226      """Verify the duality of `win` and `dual_win`. 
227   
228      Note that this test does not work for arbitrary windows, since dual windows 
229      are not unique. It always works for invertible STFTs if the windows do not 
230      overlap. 
231      """
232      # Non-standard values for keyword arguments (except for `scale_to`):

240  @pytest.mark.parametrize('scale_to, fac_psd, fac_mag',
241                           [(None, 0.25, 0.125),
242                            ('magnitude', 2.0, 1),
243                            ('psd', 1, 0.5)])
244  def test_scaling(scale_to: Literal['magnitude', 'psd'], fac_psd, fac_mag):
245      """Verify scaling calculations. 
246   
247      * Verify passing `scale_to`parameter  to ``__init__(). 
248      * Roundtrip while changing scaling factor. 
249      """
250      SFT = ShortTimeFFT(np.ones(4) * 2, hop=4, fs=1, scale_to=scale_to)
251      assert SFT.fac_psd == fac_psd
252      assert SFT.fac_magnitude == fac_mag
253      # increase coverage by accessing properties twice:
254      assert SFT.fac_psd == fac_psd
255      assert SFT.fac_magnitude == fac_mag
256  
257      x = np.fft.irfft([0, 0, 7, 0, 0, 0, 0])  # periodic signal
258      Sx = SFT.stft(x)
259      Sx_mag, Sx_psd = Sx * SFT.fac_magnitude, Sx * SFT.fac_psd
260  
261      SFT.scale_to('magnitude')
262      x_mag = SFT.istft(Sx_mag, k1=len(x))
263      xp_assert_close(x_mag, x)
264  
265      SFT.scale_to('psd')
266      x_psd = SFT.istft(Sx_psd, k1=len(x))
267      xp_assert_close(x_psd, x)

298      `const_ext`, `odd_ext()`, and `even_ext()` from the _array_tools module.

391  @pytest.mark.parametrize('n', [20, 21])
392  @pytest.mark.parametrize('m', [5, 6])
393  @pytest.mark.parametrize('fft_mode', ['onesided', 'centered'])
394  def test_extent(n, m, fft_mode: FFT_MODE_TYPE):
395      """Ensure that the `extent()` method is correct. """
396      SFT = ShortTimeFFT(np.ones(m), hop=m, fs=m, fft_mode=fft_mode)
397  
398      t0 = SFT.t(n)[0]  # first timestamp
399      t1 = SFT.t(n)[-1] + SFT.delta_t  # last timestamp + 1
400      t0c, t1c = t0 - SFT.delta_t / 2, t1 - SFT.delta_t / 2  # centered timestamps
401  
402      f0 = SFT.f[0]  # first frequency
403      f1 = SFT.f[-1] + SFT.delta_f  # last frequency + 1
404      f0c, f1c = f0 - SFT.delta_f / 2, f1 - SFT.delta_f / 2  # centered frequencies
405  
406      assert SFT.extent(n, 'tf', False) == (t0, t1, f0, f1)
407      assert SFT.extent(n, 'ft', False) == (f0, f1, t0, t1)
408      assert SFT.extent(n, 'tf', True) == (t0c, t1c, f0c, f1c)
409      assert SFT.extent(n, 'ft', True) == (f0c, f1c, t0c, t1c)

421  @pytest.mark.parametrize('n', [8, 9])
422  def test_fft_func_roundtrip(n: int):
423      """Test roundtrip `ifft_func(fft_func(x)) == x` for all permutations of 
424      relevant parameters. """
425      np.random.seed(2394795)
426      x0 = np.random.rand(n)
427      w, h_n = np.ones(n), 4
428  
429      pp = dict(
430          fft_mode=get_args(FFT_MODE_TYPE),
431          mfft=[None, n, n+1, n+2],
432          scaling=[None, 'magnitude', 'psd'],
433          phase_shift=[None, -n+1, 0, n // 2, n-1])
434      for f_typ, mfft, scaling, phase_shift in product(*pp.values()):
435          if f_typ == 'onesided2X' and scaling is None:
436              continue  # this combination is forbidden
437          SFT = ShortTimeFFT(w, h_n, fs=n, fft_mode=f_typ, mfft=mfft,
438                             scale_to=scaling, phase_shift=phase_shift)
439          X0 = SFT._fft_func(x0)
440          x1 = SFT._ifft_func(X0)
441          xp_assert_close(x0.astype(x1.dtype), x1,
442                          err_msg="_fft_func() roundtrip failed for " +
443                          f"{f_typ=}, {mfft=}, {scaling=}, {phase_shift=}")
444  
445      SFT = ShortTimeFFT(w, h_n, fs=1)
446      SFT._fft_mode = 'invalid_fft'  # type: ignore
447      with pytest.raises(RuntimeError):
448          SFT._fft_func(x0)
449      with pytest.raises(RuntimeError):
450          SFT._ifft_func(x0)

815  @pytest.mark.parametrize('N_x', (128, 129, 255, 256, 1337))  # signal length
816  @pytest.mark.parametrize('w_size', (128, 256))  # window length
817  @pytest.mark.parametrize('t_step', (4, 64))  # SFT time hop
818  @pytest.mark.parametrize('f_c', (7., 23.))  # frequency of input sine
819  def test_energy_conservation(N_x: int, w_size: int, t_step: int, f_c: float):
820      """Test if a `psd`-scaled STFT conserves the L2 norm. 
821   
822      This test is adapted from MNE-Python [1]_. Besides being battle-tested, 
823      this test has the benefit of using non-standard window including 
824      non-positive values and a 2d input signal. 
825   
826      Since `ShortTimeFFT` requires the signal length `N_x` to be at least the 
827      window length `w_size`, the parameter `N_x` was changed from 
828      ``(127, 128, 255, 256, 1337)`` to ``(128, 129, 255, 256, 1337)`` to be 
829      more useful. 
830   
831      .. [1] File ``test_stft.py`` of MNE-Python 
832          https://github.com/mne-tools/mne-python/blob/main/mne/time_frequency/tests/test_stft.py 
833      """
834      window = np.sin(np.arange(.5, w_size + .5) / w_size * np.pi)
835      SFT = ShortTimeFFT(window, t_step, fs=1000, fft_mode='onesided2X',
836                         scale_to='psd')
837      atol = 2*np.finfo(window.dtype).resolution
838      N_x = max(N_x, w_size)  # minimal sing
839      # Test with low frequency signal
840      t = np.arange(N_x).astype(np.float64)
841      x = np.sin(2 * np.pi * f_c * t * SFT.T)
842      x = np.array([x, x + 1.])
843      X = SFT.stft(x)
844      xp = SFT.istft(X, k1=N_x)
845  
846      max_freq = SFT.f[np.argmax(np.sum(np.abs(X[0]) ** 2, axis=1))]
847  
848      assert X.shape[1] == SFT.f_pts
849      assert np.all(SFT.f >= 0.)
850      assert np.abs(max_freq - f_c) < 1.
851      xp_assert_close(x, xp, atol=atol)
852  
853      # check L2-norm squared (i.e., energy) conservation:
854      E_x = np.sum(x**2, axis=-1) * SFT.T  # numerical integration
855      aX2 = X.real**2 + X.imag.real**2
856      E_X = np.sum(np.sum(aX2, axis=-1) * SFT.delta_t, axis=-1) * SFT.delta_f
857      xp_assert_close(E_X, E_x, atol=atol)
858  
859      # Test with random signal
860      np.random.seed(2392795)
861      x = np.random.randn(2, N_x)
862      X = SFT.stft(x)
863      xp = SFT.istft(X, k1=N_x)
864  
865      assert X.shape[1] == SFT.f_pts
866      assert np.all(SFT.f >= 0.)
867      assert np.abs(max_freq - f_c) < 1.
868      xp_assert_close(x, xp, atol=atol)
869  
870      # check L2-norm squared (i.e., energy) conservation:
871      E_x = np.sum(x**2, axis=-1) * SFT.T  # numeric integration
872      aX2 = X.real ** 2 + X.imag.real ** 2
873      E_X = np.sum(np.sum(aX2, axis=-1) * SFT.delta_t, axis=-1) * SFT.delta_f
874      xp_assert_close(E_X, E_x, atol=atol)
875  
876      # Try with empty array
877      x = np.zeros((0, N_x))
878      X = SFT.stft(x)
879      xp = SFT.istft(X, k1=N_x)
880      assert xp.shape == x.shape
881
