41   def approx(a, b, fill_value=True, rtol=1e-5, atol=1e-8):
42       """ 
43       Returns true if all components of a and b are equal to given tolerances. 
44    
45       If fill_value is True, masked values considered equal. Otherwise, 
46       masked values are considered unequal.  The relative error rtol should 
47       be positive and << 1.0 The absolute error atol comes into play for 
48       those elements of b that are very small or zero; it says how small a 
49       must be also. 
50    
51       """
52       m = mask_or(getmask(a), getmask(b))
53       d1 = filled(a)
54       d2 = filled(b)
55       if d1.dtype.char == "O" or d2.dtype.char == "O":
56           return np.equal(d1, d2).ravel()
57       x = filled(
58           masked_array(d1, copy=False, mask=m), fill_value
59       ).astype(np.float64)
60       y = filled(masked_array(d2, copy=False, mask=m), 1).astype(np.float64)
61       d = np.less_equal(umath.absolute(x - y), atol + rtol * umath.absolute(y))
62       return d.ravel()

65   def almost(a, b, decimal=6, fill_value=True):
66       """ 
67       Returns True if a and b are equal up to decimal places. 
68    
69       If fill_value is True, masked values considered equal. Otherwise, 
70       masked values are considered unequal. 
71    
72       """
73       m = mask_or(getmask(a), getmask(b))
74       d1 = filled(a)
75       d2 = filled(b)
76       if d1.dtype.char == "O" or d2.dtype.char == "O":
77           return np.equal(d1, d2).ravel()
78       x = filled(
79           masked_array(d1, copy=False, mask=m), fill_value
80       ).astype(np.float64)
81       y = filled(masked_array(d2, copy=False, mask=m), 1).astype(np.float64)
82       d = np.around(np.abs(x - y), decimal) <= 10.0 ** (-decimal)
83       return d.ravel()

86   def _assert_equal_on_sequences(actual, desired, err_msg=''):
87       """ 
88       Asserts the equality of two non-array sequences. 
89    
90       """
91       assert_equal(len(actual), len(desired), err_msg)
92       for k in range(len(desired)):
93           assert_equal(actual[k], desired[k], f'item={k!r}\n{err_msg}')
94       return

97   def assert_equal_records(a, b):
98       """ 
99       Asserts that two records are equal. 
100   
101      Pretty crude for now. 
102   
103      """
104      assert_equal(a.dtype, b.dtype)
105      for f in a.dtype.names:
106          (af, bf) = (operator.getitem(a, f), operator.getitem(b, f))
107          if not (af is masked) and not (bf is masked):
108              assert_equal(operator.getitem(a, f), operator.getitem(b, f))
109      return

112  def assert_equal(actual, desired, err_msg=''):
113      """ 
114      Asserts that two items are equal. 
115   
116      """
117      # Case #1: dictionary .....
118      if isinstance(desired, dict):
119          if not isinstance(actual, dict):
120              raise AssertionError(repr(type(actual)))
121          assert_equal(len(actual), len(desired), err_msg)
122          for k, i in desired.items():
123              if k not in actual:
124                  raise AssertionError(f"{k} not in {actual}")
125              assert_equal(actual[k], desired[k], f'key={k!r}\n{err_msg}')
126          return
127      # Case #2: lists .....
128      if isinstance(desired, (list, tuple)) and isinstance(actual, (list, tuple)):
129          return _assert_equal_on_sequences(actual, desired, err_msg='')
130      if not (isinstance(actual, ndarray) or isinstance(desired, ndarray)):
131          msg = build_err_msg([actual, desired], err_msg,)
132          if not desired == actual:
133              raise AssertionError(msg)
134          return
135      # Case #4. arrays or equivalent
136      if ((actual is masked) and not (desired is masked)) or \
137              ((desired is masked) and not (actual is masked)):
138          msg = build_err_msg([actual, desired],
139                              err_msg, header='', names=('x', 'y'))
140          raise ValueError(msg)
141      actual = np.asanyarray(actual)
142      desired = np.asanyarray(desired)
143      (actual_dtype, desired_dtype) = (actual.dtype, desired.dtype)
144      if actual_dtype.char == "S" and desired_dtype.char == "S":
145          return _assert_equal_on_sequences(actual.tolist(),
146                                            desired.tolist(),
147                                            err_msg='')
148      return assert_array_equal(actual, desired, err_msg)

151  def fail_if_equal(actual, desired, err_msg='',):
152      """ 
153      Raises an assertion error if two items are equal. 
154   
155      """
156      if isinstance(desired, dict):
157          if not isinstance(actual, dict):
158              raise AssertionError(repr(type(actual)))
159          fail_if_equal(len(actual), len(desired), err_msg)
160          for k, i in desired.items():
161              if k not in actual:
162                  raise AssertionError(repr(k))
163              fail_if_equal(actual[k], desired[k], f'key={k!r}\n{err_msg}')
164          return
165      if isinstance(desired, (list, tuple)) and isinstance(actual, (list, tuple)):
166          fail_if_equal(len(actual), len(desired), err_msg)
167          for k in range(len(desired)):
168              fail_if_equal(actual[k], desired[k], f'item={k!r}\n{err_msg}')
169          return
170      if isinstance(actual, np.ndarray) or isinstance(desired, np.ndarray):
171          return fail_if_array_equal(actual, desired, err_msg)
172      msg = build_err_msg([actual, desired], err_msg)
173      if not desired != actual:
174          raise AssertionError(msg)

180  def assert_almost_equal(actual, desired, decimal=7, err_msg='', verbose=True):
181      """ 
182      Asserts that two items are almost equal. 
183   
184      The test is equivalent to abs(desired-actual) < 0.5 * 10**(-decimal). 
185   
186      """
187      if isinstance(actual, np.ndarray) or isinstance(desired, np.ndarray):
188          return assert_array_almost_equal(actual, desired, decimal=decimal,
189                                           err_msg=err_msg, verbose=verbose)
190      msg = build_err_msg([actual, desired],
191                          err_msg=err_msg, verbose=verbose)
192      if not round(abs(desired - actual), decimal) == 0:
193          raise AssertionError(msg)

199  def assert_array_compare(comparison, x, y, err_msg='', verbose=True, header='',
200                           fill_value=True):
201      """ 
202      Asserts that comparison between two masked arrays is satisfied. 
203   
204      The comparison is elementwise. 
205   
206      """
207      # Allocate a common mask and refill
208      m = mask_or(getmask(x), getmask(y))
209      x = masked_array(x, copy=False, mask=m, keep_mask=False, subok=False)
210      y = masked_array(y, copy=False, mask=m, keep_mask=False, subok=False)
211      if ((x is masked) and not (y is masked)) or \
212              ((y is masked) and not (x is masked)):
213          msg = build_err_msg([x, y], err_msg=err_msg, verbose=verbose,
214                              header=header, names=('x', 'y'))
215          raise ValueError(msg)
216      # OK, now run the basic tests on filled versions
217      return np.testing.assert_array_compare(comparison,
218                                             x.filled(fill_value),
219                                             y.filled(fill_value),
220                                             err_msg=err_msg,
221                                             verbose=verbose, header=header)

225      """ 
226      Checks the elementwise equality of two masked arrays. 
227   
228      """
229      assert_array_compare(operator.__eq__, x, y,
230                           err_msg=err_msg, verbose=verbose,
231                           header='Arrays are not equal')

234  def fail_if_array_equal(x, y, err_msg='', verbose=True):
235      """ 
236      Raises an assertion error if two masked arrays are not equal elementwise. 
237   
238      """
239      def compare(x, y):
240          return (not np.all(approx(x, y)))
241      assert_array_compare(compare, x, y, err_msg=err_msg, verbose=verbose,
242                           header='Arrays are not equal')

245  def assert_array_approx_equal(x, y, decimal=6, err_msg='', verbose=True):
246      """ 
247      Checks the equality of two masked arrays, up to given number odecimals. 
248   
249      The equality is checked elementwise. 
250   
251      """
252      def compare(x, y):
253          "Returns the result of the loose comparison between x and y)."
254          return approx(x, y, rtol=10. ** -decimal)
255      assert_array_compare(compare, x, y, err_msg=err_msg, verbose=verbose,
256                           header='Arrays are not almost equal')

283  def assert_mask_equal(m1, m2, err_msg=''):
284      """ 
285      Asserts the equality of two masks. 
286   
287      """
288      if m1 is nomask:
289          assert_(m2 is nomask)
290      if m2 is nomask:
291          assert_(m1 is nomask)
