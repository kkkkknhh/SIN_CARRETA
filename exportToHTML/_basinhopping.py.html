18   class Storage:
19       """ 
20       Class used to store the lowest energy structure 
21       """
22       def __init__(self, minres):
23           self._add(minres)
24   
25       def _add(self, minres):
26           self.minres = minres
27           self.minres.x = np.copy(minres.x)
28   
29       def update(self, minres):
30           if minres.success and (minres.fun < self.minres.fun
31                                  or not self.minres.success):
32               self._add(minres)
33               return True
34           else:
35               return False
36   
37       def get_lowest(self):
38           return self.minres

41   class BasinHoppingRunner:
42       """This class implements the core of the basinhopping algorithm. 
43    
44       x0 : ndarray 
45           The starting coordinates. 
46       minimizer : callable 
47           The local minimizer, with signature ``result = minimizer(x)``. 
48           The return value is an `optimize.OptimizeResult` object. 
49       step_taking : callable 
50           This function displaces the coordinates randomly. Signature should 
51           be ``x_new = step_taking(x)``. Note that `x` may be modified in-place. 
52       accept_tests : list of callables 
53           Each test is passed the kwargs `f_new`, `x_new`, `f_old` and 
54           `x_old`. These tests will be used to judge whether or not to accept 
55           the step. The acceptable return values are True, False, or ``"force 
56           accept"``. If any of the tests return False then the step is rejected. 
57           If ``"force accept"``, then this will override any other tests in 
58           order to accept the step. This can be used, for example, to forcefully 
59           escape from a local minimum that ``basinhopping`` is trapped in. 
60       disp : bool, optional 
61           Display status messages. 
62    
63       """
64       def __init__(self, x0, minimizer, step_taking, accept_tests, disp=False):
65           self.x = np.copy(x0)
66           self.minimizer = minimizer
67           self.step_taking = step_taking
68           self.accept_tests = accept_tests
69           self.disp = disp
70   
71           self.nstep = 0
72   
73           # initialize return object
74           self.res = scipy.optimize.OptimizeResult()
75           self.res.minimization_failures = 0
76   
77           # do initial minimization
78           minres = minimizer(self.x)
79           if not minres.success:
80               self.res.minimization_failures += 1
81               if self.disp:
82                   print("warning: basinhopping: local minimization failure")
83           self.x = np.copy(minres.x)
84           self.energy = minres.fun
85           self.incumbent_minres = minres  # best minimize result found so far
86           if self.disp:
87               print("basinhopping step %d: f %g" % (self.nstep, self.energy))
88   
89           # initialize storage class
90           self.storage = Storage(minres)
91   
92           if hasattr(minres, "nfev"):
93               self.res.nfev = minres.nfev
94           if hasattr(minres, "njev"):
95               self.res.njev = minres.njev
96           if hasattr(minres, "nhev"):
97               self.res.nhev = minres.nhev
98   
99       def _monte_carlo_step(self):
100          """Do one Monte Carlo iteration 
101   
102          Randomly displace the coordinates, minimize, and decide whether 
103          or not to accept the new coordinates. 
104          """
105          # Take a random step.  Make a copy of x because the step_taking
106          # algorithm might change x in place
107          x_after_step = np.copy(self.x)
108          x_after_step = self.step_taking(x_after_step)
109  
110          # do a local minimization
111          minres = self.minimizer(x_after_step)
112          x_after_quench = minres.x
113          energy_after_quench = minres.fun
114          if not minres.success:
115              self.res.minimization_failures += 1
116              if self.disp:
117                  print("warning: basinhopping: local minimization failure")
118          if hasattr(minres, "nfev"):
119              self.res.nfev += minres.nfev
120          if hasattr(minres, "njev"):
121              self.res.njev += minres.njev
122          if hasattr(minres, "nhev"):
123              self.res.nhev += minres.nhev
124  
125          # accept the move based on self.accept_tests. If any test is False,
126          # then reject the step.  If any test returns the special string
127          # 'force accept', then accept the step regardless. This can be used
128          # to forcefully escape from a local minimum if normal basin hopping
129          # steps are not sufficient.
130          accept = True
131          for test in self.accept_tests:
132              if inspect.signature(test) == _new_accept_test_signature:
133                  testres = test(res_new=minres, res_old=self.incumbent_minres)
134              else:
135                  testres = test(f_new=energy_after_quench, x_new=x_after_quench,
136                                 f_old=self.energy, x_old=self.x)
137  
138              if testres == 'force accept':
139                  accept = True
140                  break
141              elif testres is None:
142                  raise ValueError("accept_tests must return True, False, or "
143                                   "'force accept'")
144              elif not testres:
145                  accept = False
146  
147          # Report the result of the acceptance test to the take step class.
148          # This is for adaptive step taking
149          if hasattr(self.step_taking, "report"):
150              self.step_taking.report(accept, f_new=energy_after_quench,
151                                      x_new=x_after_quench, f_old=self.energy,
152                                      x_old=self.x)
153  
154          return accept, minres
155  
156      def one_cycle(self):
157          """Do one cycle of the basinhopping algorithm 
158          """
159          self.nstep += 1
160          new_global_min = False
161  
162          accept, minres = self._monte_carlo_step()
163  
164          if accept:
165              self.energy = minres.fun
166              self.x = np.copy(minres.x)
167              self.incumbent_minres = minres  # best minimize result found so far
168              new_global_min = self.storage.update(minres)
169  
170          # print some information
171          if self.disp:
172              self.print_report(minres.fun, accept)
173              if new_global_min:
174                  print("found new global minimum on step %d with function"
175                        " value %g" % (self.nstep, self.energy))
176  
177          # save some variables as BasinHoppingRunner attributes
178          self.xtrial = minres.x
179          self.energy_trial = minres.fun
180          self.accept = accept
181  
182          return new_global_min
183  
184      def print_report(self, energy_trial, accept):
185          """print a status update"""
186          minres = self.storage.get_lowest()
187          print("basinhopping step %d: f %g trial_f %g accepted %d "
188                " lowest_f %g" % (self.nstep, self.energy, energy_trial,
189                                  accept, minres.fun))

192  class AdaptiveStepsize:
193      """ 
194      Class to implement adaptive stepsize. 
195   
196      This class wraps the step taking class and modifies the stepsize to 
197      ensure the true acceptance rate is as close as possible to the target. 
198   
199      Parameters 
200      ---------- 
201      takestep : callable 
202          The step taking routine.  Must contain modifiable attribute 
203          takestep.stepsize 
204      accept_rate : float, optional 
205          The target step acceptance rate 
206      interval : int, optional 
207          Interval for how often to update the stepsize 
208      factor : float, optional 
209          The step size is multiplied or divided by this factor upon each 
210          update. 
211      verbose : bool, optional 
212          Print information about each update 
213   
214      """
215      def __init__(self, takestep, accept_rate=0.5, interval=50, factor=0.9,
216                   verbose=True):
217          self.takestep = takestep
218          self.target_accept_rate = accept_rate
219          self.interval = interval
220          self.factor = factor
221          self.verbose = verbose
222  
223          self.nstep = 0
224          self.nstep_tot = 0
225          self.naccept = 0
226  
227      def __call__(self, x):
228          return self.take_step(x)
229  
230      def _adjust_step_size(self):
231          old_stepsize = self.takestep.stepsize
232          accept_rate = float(self.naccept) / self.nstep
233          if accept_rate > self.target_accept_rate:
234              # We're accepting too many steps. This generally means we're
235              # trapped in a basin. Take bigger steps.
236              self.takestep.stepsize /= self.factor
237          else:
238              # We're not accepting enough steps. Take smaller steps.
239              self.takestep.stepsize *= self.factor
240          if self.verbose:
241              print(f"adaptive stepsize: acceptance rate {accept_rate:f} target "
242                    f"{self.target_accept_rate:f} new stepsize "
243                    f"{self.takestep.stepsize:g} old stepsize {old_stepsize:g}")
244  
245      def take_step(self, x):
246          self.nstep += 1
247          self.nstep_tot += 1
248          if self.nstep % self.interval == 0:
249              self._adjust_step_size()
250          return self.takestep(x)
251  
252      def report(self, accept, **kwargs):
253          "called by basinhopping to report the result of the step"
254          if accept:
255              self.naccept += 1

258  class RandomDisplacement:
259      """Add a random displacement of maximum size `stepsize` to each coordinate. 
260   
261      Calling this updates `x` in-place. 
262   
263      Parameters 
264      ---------- 
265      stepsize : float, optional 
266          Maximum stepsize in any dimension 
267      rng : {None, int, `numpy.random.Generator`}, optional 
268          Random number generator 
269      """
270  
271      def __init__(self, stepsize=0.5, rng=None):
272          self.stepsize = stepsize
273          self.rng = check_random_state(rng)
274  
275      def __call__(self, x):
276          x += self.rng.uniform(-self.stepsize, self.stepsize,
277                                np.shape(x))
278          return x

297  class Metropolis:
298      """Metropolis acceptance criterion. 
299   
300      Parameters 
301      ---------- 
302      T : float 
303          The "temperature" parameter for the accept or reject criterion. 
304      rng : {None, int, `numpy.random.Generator`}, optional 
305          Random number generator used for acceptance test. 
306   
307      """
308  
309      def __init__(self, T, rng=None):
310          # Avoid ZeroDivisionError since "MBH can be regarded as a special case
311          # of the BH framework with the Metropolis criterion, where temperature
312          # T = 0." (Reject all steps that increase energy.)
313          self.beta = 1.0 / T if T != 0 else float('inf')
314          self.rng = check_random_state(rng)
315  
316      def accept_reject(self, res_new, res_old):
317          """ 
318          Assuming the local search underlying res_new was successful: 
319          If new energy is lower than old, it will always be accepted. 
320          If new is higher than old, there is a chance it will be accepted, 
321          less likely for larger differences. 
322          """
323          with np.errstate(invalid='ignore'):
324              # The energy values being fed to Metropolis are 1-length arrays, and if
325              # they are equal, their difference is 0, which gets multiplied by beta,
326              # which is inf, and array([0]) * float('inf') causes
327              #
328              # RuntimeWarning: invalid value encountered in multiply
329              #
330              # Ignore this warning so when the algorithm is on a flat plane, it always
331              # accepts the step, to try to move off the plane.
332              prod = -(res_new.fun - res_old.fun) * self.beta
333              w = math.exp(min(0, prod))
334  
335          rand = self.rng.uniform()
336          return w >= rand and (res_new.success or not res_old.success)
337  
338      def __call__(self, *, res_new, res_old):
339          """ 
340          f_new and f_old are mandatory in kwargs 
341          """
342          return bool(self.accept_reject(res_new, res_old))

345  @_transition_to_rng("seed", position_num=12, replace_doc=True)
346  def basinhopping(func, x0, niter=100, T=1.0, stepsize=0.5,
347                   minimizer_kwargs=None, take_step=None, accept_test=None,
348                   callback=None, interval=50, disp=False, niter_success=None,
349                   rng=None, *, target_accept_rate=0.5, stepwise_factor=0.9):
350      """Find the global minimum of a function using the basin-hopping algorithm. 
351   
352      Basin-hopping is a two-phase method that combines a global stepping 
353      algorithm with local minimization at each step. Designed to mimic 
354      the natural process of energy minimization of clusters of atoms, it works 
355      well for similar problems with "funnel-like, but rugged" energy landscapes 
356      [5]_. 
357   
358      As the step-taking, step acceptance, and minimization methods are all 
359      customizable, this function can also be used to implement other two-phase 
360      methods. 
361   
362      Parameters 
363      ---------- 
364      func : callable ``f(x, *args)`` 
365          Function to be optimized.  ``args`` can be passed as an optional item 
366          in the dict `minimizer_kwargs` 
367      x0 : array_like 
368          Initial guess. 
369      niter : integer, optional 
370          The number of basin-hopping iterations. There will be a total of 
371          ``niter + 1`` runs of the local minimizer. 
372      T : float, optional 
373          The "temperature" parameter for the acceptance or rejection criterion. 
374          Higher "temperatures" mean that larger jumps in function value will be 
375          accepted.  For best results `T` should be comparable to the 
376          separation (in function value) between local minima. 
377      stepsize : float, optional 
378          Maximum step size for use in the random displacement. 
379      minimizer_kwargs : dict, optional 
380          Extra keyword arguments to be passed to the local minimizer 
381          `scipy.optimize.minimize` Some important options could be: 
382   
383          method : str 
384              The minimization method (e.g. ``"L-BFGS-B"``) 
385          args : tuple 
386              Extra arguments passed to the objective function (`func`) and 
387              its derivatives (Jacobian, Hessian). 
388   
389      take_step : callable ``take_step(x)``, optional 
390          Replace the default step-taking routine with this routine. The default 
391          step-taking routine is a random displacement of the coordinates, but 
392          other step-taking algorithms may be better for some systems. 
393          `take_step` can optionally have the attribute ``take_step.stepsize``. 
394          If this attribute exists, then `basinhopping` will adjust 
395          ``take_step.stepsize`` in order to try to optimize the global minimum 
396          search. 
397      accept_test : callable, ``accept_test(f_new=f_new, x_new=x_new, f_old=fold, x_old=x_old)``, optional 
398          Define a test which will be used to judge whether to accept the 
399          step. This will be used in addition to the Metropolis test based on 
400          "temperature" `T`. The acceptable return values are True, 
401          False, or ``"force accept"``. If any of the tests return False 
402          then the step is rejected. If the latter, then this will override any 
403          other tests in order to accept the step. This can be used, for example, 
404          to forcefully escape from a local minimum that `basinhopping` is 
405          trapped in. 
406      callback : callable, ``callback(x, f, accept)``, optional 
407          A callback function which will be called for all minima found. ``x`` 
408          and ``f`` are the coordinates and function value of the trial minimum, 
409          and ``accept`` is whether that minimum was accepted. This can 
410          be used, for example, to save the lowest N minima found. Also, 
411          `callback` can be used to specify a user defined stop criterion by 
412          optionally returning True to stop the `basinhopping` routine. 
413      interval : integer, optional 
414          interval for how often to update the `stepsize` 
415      disp : bool, optional 
416          Set to True to print status messages 
417      niter_success : integer, optional 
418          Stop the run if the global minimum candidate remains the same for this 
419          number of iterations. 
420      rng : `numpy.random.Generator`, optional 
421          Pseudorandom number generator state. When `rng` is None, a new 
422          `numpy.random.Generator` is created using entropy from the 
423          operating system. Types other than `numpy.random.Generator` are 
424          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
425   
426          The random numbers generated only affect the default Metropolis 
427          `accept_test` and the default `take_step`. If you supply your own 
428          `take_step` and `accept_test`, and these functions use random 
429          number generation, then those functions are responsible for the state 
430          of their random number generator. 
431      target_accept_rate : float, optional 
432          The target acceptance rate that is used to adjust the `stepsize`. 
433          If the current acceptance rate is greater than the target, 
434          then the `stepsize` is increased. Otherwise, it is decreased. 
435          Range is (0, 1). Default is 0.5. 
436   
437          .. versionadded:: 1.8.0 
438   
439      stepwise_factor : float, optional 
440          The `stepsize` is multiplied or divided by this stepwise factor upon 
441          each update. Range is (0, 1). Default is 0.9. 
442   
443          .. versionadded:: 1.8.0 
444   
445      Returns 
446      ------- 
447      res : OptimizeResult 
448          The optimization result represented as a `OptimizeResult` object. 
449          Important attributes are: ``x`` the solution array, ``fun`` the value 
450          of the function at the solution, and ``message`` which describes the 
451          cause of the termination. The ``OptimizeResult`` object returned by the 
452          selected minimizer at the lowest minimum is also contained within this 
453          object and can be accessed through the ``lowest_optimization_result`` 
454          attribute.  See `OptimizeResult` for a description of other attributes. 
455   
456      See Also 
457      -------- 
458      minimize : 
459          The local minimization function called once for each basinhopping step. 
460          `minimizer_kwargs` is passed to this routine. 
461   
462      Notes 
463      ----- 
464      Basin-hopping is a stochastic algorithm which attempts to find the global 
465      minimum of a smooth scalar function of one or more variables [1]_ [2]_ [3]_ 
466      [4]_. The algorithm in its current form was described by David Wales and 
467      Jonathan Doye [2]_ http://www-wales.ch.cam.ac.uk/. 
468   
469      The algorithm is iterative with each cycle composed of the following 
470      features 
471   
472      1) random perturbation of the coordinates 
473   
474      2) local minimization 
475   
476      3) accept or reject the new coordinates based on the minimized function 
477         value 
478   
479      The acceptance test used here is the Metropolis criterion of standard Monte 
480      Carlo algorithms, although there are many other possibilities [3]_. 
481   
482      This global minimization method has been shown to be extremely efficient 
483      for a wide variety of problems in physics and chemistry. It is 
484      particularly useful when the function has many minima separated by large 
485      barriers. See the `Cambridge Cluster Database 
486      <https://www-wales.ch.cam.ac.uk/CCD.html>`_ for databases of molecular 
487      systems that have been optimized primarily using basin-hopping. This 
488      database includes minimization problems exceeding 300 degrees of freedom. 
489   
490      See the free software program `GMIN <https://www-wales.ch.cam.ac.uk/GMIN>`_ 
491      for a Fortran implementation of basin-hopping. This implementation has many 
492      variations of the procedure described above, including more 
493      advanced step taking algorithms and alternate acceptance criterion. 
494   
495      For stochastic global optimization there is no way to determine if the true 
496      global minimum has actually been found. Instead, as a consistency check, 
497      the algorithm can be run from a number of different random starting points 
498      to ensure the lowest minimum found in each example has converged to the 
499      global minimum. For this reason, `basinhopping` will by default simply 
500      run for the number of iterations `niter` and return the lowest minimum 
501      found. It is left to the user to ensure that this is in fact the global 
502      minimum. 
503   
504      Choosing `stepsize`:  This is a crucial parameter in `basinhopping` and 
505      depends on the problem being solved. The step is chosen uniformly in the 
506      region from x0-stepsize to x0+stepsize, in each dimension. Ideally, it 
507      should be comparable to the typical separation (in argument values) between 
508      local minima of the function being optimized. `basinhopping` will, by 
509      default, adjust `stepsize` to find an optimal value, but this may take 
510      many iterations. You will get quicker results if you set a sensible 
511      initial value for ``stepsize``. 
512   
513      Choosing `T`: The parameter `T` is the "temperature" used in the 
514      Metropolis criterion. Basinhopping steps are always accepted if 
515      ``func(xnew) < func(xold)``. Otherwise, they are accepted with 
516      probability:: 
517   
518          exp( -(func(xnew) - func(xold)) / T ) 
519   
520      So, for best results, `T` should to be comparable to the typical 
521      difference (in function values) between local minima. (The height of 
522      "walls" between local minima is irrelevant.) 
523   
524      If `T` is 0, the algorithm becomes Monotonic Basin-Hopping, in which all 
525      steps that increase energy are rejected. 
526   
527      .. versionadded:: 0.12.0 
528   
529      References 
530      ---------- 
531      .. [1] Wales, David J. 2003, Energy Landscapes, Cambridge University Press, 
532          Cambridge, UK. 
533      .. [2] Wales, D J, and Doye J P K, Global Optimization by Basin-Hopping and 
534          the Lowest Energy Structures of Lennard-Jones Clusters Containing up to 
535          110 Atoms.  Journal of Physical Chemistry A, 1997, 101, 5111. 
536      .. [3] Li, Z. and Scheraga, H. A., Monte Carlo-minimization approach to the 
537          multiple-minima problem in protein folding, Proc. Natl. Acad. Sci. USA, 
538          1987, 84, 6611. 
539      .. [4] Wales, D. J. and Scheraga, H. A., Global optimization of clusters, 
540          crystals, and biomolecules, Science, 1999, 285, 1368. 
541      .. [5] Olson, B., Hashmi, I., Molloy, K., and Shehu1, A., Basin Hopping as 
542          a General and Versatile Optimization Framework for the Characterization 
543          of Biological Macromolecules, Advances in Artificial Intelligence, 
544          Volume 2012 (2012), Article ID 674832, :doi:`10.1155/2012/674832` 
545   
546      Examples 
547      -------- 
548      The following example is a 1-D minimization problem, with many 
549      local minima superimposed on a parabola. 
550   
551      >>> import numpy as np 
552      >>> from scipy.optimize import basinhopping 
553      >>> func = lambda x: np.cos(14.5 * x - 0.3) + (x + 0.2) * x 
554      >>> x0 = [1.] 
555   
556      Basinhopping, internally, uses a local minimization algorithm. We will use 
557      the parameter `minimizer_kwargs` to tell basinhopping which algorithm to 
558      use and how to set up that minimizer. This parameter will be passed to 
559      `scipy.optimize.minimize`. 
560   
561      >>> minimizer_kwargs = {"method": "BFGS"} 
562      >>> ret = basinhopping(func, x0, minimizer_kwargs=minimizer_kwargs, 
563      ...                    niter=200) 
564      >>> # the global minimum is: 
565      >>> ret.x, ret.fun 
566      -0.1951, -1.0009 
567   
568      Next consider a 2-D minimization problem. Also, this time, we 
569      will use gradient information to significantly speed up the search. 
570   
571      >>> def func2d(x): 
572      ...     f = np.cos(14.5 * x[0] - 0.3) + (x[1] + 0.2) * x[1] + (x[0] + 
573      ...                                                            0.2) * x[0] 
574      ...     df = np.zeros(2) 
575      ...     df[0] = -14.5 * np.sin(14.5 * x[0] - 0.3) + 2. * x[0] + 0.2 
576      ...     df[1] = 2. * x[1] + 0.2 
577      ...     return f, df 
578   
579      We'll also use a different local minimization algorithm. Also, we must tell 
580      the minimizer that our function returns both energy and gradient (Jacobian). 
581   
582      >>> minimizer_kwargs = {"method":"L-BFGS-B", "jac":True} 
583      >>> x0 = [1.0, 1.0] 
584      >>> ret = basinhopping(func2d, x0, minimizer_kwargs=minimizer_kwargs, 
585      ...                    niter=200) 
586      >>> print("global minimum: x = [%.4f, %.4f], f(x) = %.4f" % (ret.x[0], 
587      ...                                                           ret.x[1], 
588      ...                                                           ret.fun)) 
589      global minimum: x = [-0.1951, -0.1000], f(x) = -1.0109 
590   
591      Here is an example using a custom step-taking routine. Imagine you want 
592      the first coordinate to take larger steps than the rest of the coordinates. 
593      This can be implemented like so: 
594   
595      >>> class MyTakeStep: 
596      ...    def __init__(self, stepsize=0.5): 
597      ...        self.stepsize = stepsize 
598      ...        self.rng = np.random.default_rng() 
599      ...    def __call__(self, x): 
600      ...        s = self.stepsize 
601      ...        x[0] += self.rng.uniform(-2.*s, 2.*s) 
602      ...        x[1:] += self.rng.uniform(-s, s, x[1:].shape) 
603      ...        return x 
604   
605      Since ``MyTakeStep.stepsize`` exists basinhopping will adjust the magnitude 
606      of `stepsize` to optimize the search. We'll use the same 2-D function as 
607      before 
608
