1    """Test how the ufuncs in special handle nan inputs. 
2     
3    """
4    from typing import Callable
5    
6    import numpy as np
7    from numpy.testing import assert_array_equal, assert_, suppress_warnings
8    import pytest
9    import scipy.special as sc
10   
11   
12   KNOWNFAILURES: dict[str, Callable] = {}
13   
14   POSTPROCESSING: dict[str, Callable] = {}
15   
16   
17   def _get_ufuncs():
18       ufuncs = []
19       ufunc_names = []
20       for name in sorted(sc.__dict__):
21           obj = sc.__dict__[name]
22           if not isinstance(obj, np.ufunc):
23               continue
24           msg = KNOWNFAILURES.get(obj)
25           if msg is None:
26               ufuncs.append(obj)
27               ufunc_names.append(name)
28           else:
29               fail = pytest.mark.xfail(run=False, reason=msg)
30               ufuncs.append(pytest.param(obj, marks=fail))
31               ufunc_names.append(name)
32       return ufuncs, ufunc_names
33   
34   
35   UFUNCS, UFUNC_NAMES = _get_ufuncs()
36   
37   
38   @pytest.mark.thread_unsafe
39   @pytest.mark.parametrize("func", UFUNCS, ids=UFUNC_NAMES)
40   def test_nan_inputs(func):
41       args = (np.nan,)*func.nin
42       with suppress_warnings() as sup:
43           # Ignore warnings about unsafe casts from legacy wrappers
44           sup.filter(RuntimeWarning,
45                      "floating point number truncated to an integer")
46           try:
47               with suppress_warnings() as sup:
48                   sup.filter(DeprecationWarning)
49                   res = func(*args)
50           except TypeError:
51               # One of the arguments doesn't take real inputs
52               return
53       if func in POSTPROCESSING:
54           res = POSTPROCESSING[func](*res)
55   
56       msg = f"got {res} instead of nan"
57       assert_array_equal(np.isnan(res), True, err_msg=msg)
58   
59   
60   def test_legacy_cast():
61       with suppress_warnings() as sup:
62           sup.filter(RuntimeWarning,
63                      "floating point number truncated to an integer")
64           res = sc.bdtrc(np.nan, 1, 0.5)
65           assert_(np.isnan(res))
