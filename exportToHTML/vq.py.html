vq.py 
1    """ 
2    K-means clustering and vector quantization (:mod:`scipy.cluster.vq`) 
3    ==================================================================== 
4     
5    Provides routines for k-means clustering, generating code books 
6    from k-means models and quantizing vectors by comparing them with 
7    centroids in a code book. 
8     
9    .. autosummary:: 
10      :toctree: generated/ 
11    
12      whiten -- Normalize a group of observations so each feature has unit variance 
13      vq -- Calculate code book membership of a set of observation vectors 
14      kmeans -- Perform k-means on a set of observation vectors forming k clusters 
15      kmeans2 -- A different implementation of k-means with more methods 
16              -- for initializing centroids 
17    
18   Background information 
19   ---------------------- 
20   The k-means algorithm takes as input the number of clusters to 
21   generate, k, and a set of observation vectors to cluster. It 
22   returns a set of centroids, one for each of the k clusters. An 
23   observation vector is classified with the cluster number or 
24   centroid index of the centroid closest to it. 
25    
26   A vector v belongs to cluster i if it is closer to centroid i than 
27   any other centroid. If v belongs to i, we say centroid i is the 
28   dominating centroid of v. The k-means algorithm tries to 
29   minimize distortion, which is defined as the sum of the squared distances 
30   between each observation vector and its dominating centroid. 
31   The minimization is achieved by iteratively reclassifying 
32   the observations into clusters and recalculating the centroids until 
33   a configuration is reached in which the centroids are stable. One can 
34   also define a maximum number of iterations. 
35    
36   Since vector quantization is a natural application for k-means, 
37   information theory terminology is often used. The centroid index 
38   or cluster index is also referred to as a "code" and the table 
39   mapping codes to centroids and, vice versa, is often referred to as a 
40   "code book". The result of k-means, a set of centroids, can be 
41   used to quantize vectors. Quantization aims to find an encoding of 
42   vectors that reduces the expected distortion. 
43    
44   All routines expect obs to be an M by N array, where the rows are 
45   the observation vectors. The codebook is a k by N array, where the 
46   ith row is the centroid of code word i. The observation vectors 
47   and centroids have the same feature dimension. 
48    
49   As an example, suppose we wish to compress a 24-bit color image 
50   (each pixel is represented by one byte for red, one for blue, and 
51   one for green) before sending it over the web. By using a smaller 
52   8-bit encoding, we can reduce the amount of data by two 
53   thirds. Ideally, the colors for each of the 256 possible 8-bit 
54   encoding values should be chosen to minimize distortion of the 
55   color. Running k-means with k=256 generates a code book of 256 
56   codes, which fills up all possible 8-bit sequences. Instead of 
57   sending a 3-byte value for each pixel, the 8-bit centroid index 
58   (or code word) of the dominating centroid is transmitted. The code 
59   book is also sent over the wire so each 8-bit code can be 
60   translated back to a 24-bit pixel value representation. If the 
61   image of interest was of an ocean, we would expect many 24-bit 
62   blues to be represented by 8-bit codes. If it was an image of a 
63   human face, more flesh-tone colors would be represented in the 
64   code book. 
65    
66   """

89   def whiten(obs, check_finite=True):
90       """ 
91       Normalize a group of observations on a per feature basis. 
92    
93       Before running k-means, it is beneficial to rescale each feature 
94       dimension of the observation set by its standard deviation (i.e. "whiten" 
95       it - as in "white noise" where each frequency has equal power). 
96       Each feature is divided by its standard deviation across all observations 
97       to give it unit variance. 
98    
99       Parameters 
100      ---------- 
101      obs : ndarray 
102          Each row of the array is an observation.  The 
103          columns are the features seen during each observation. 
104   
105          >>> #         f0    f1    f2 
106          >>> obs = [[  1.,   1.,   1.],  #o0 
107          ...        [  2.,   2.,   2.],  #o1 
108          ...        [  3.,   3.,   3.],  #o2 
109          ...        [  4.,   4.,   4.]]  #o3 
110   
111      check_finite : bool, optional 
112          Whether to check that the input matrices contain only finite numbers. 
113          Disabling may give a performance gain, but may result in problems 
114          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
115          Default: True 
116   
117      Returns 
118      ------- 
119      result : ndarray 
120          Contains the values in `obs` scaled by the standard deviation 
121          of each column. 
122   
123      Examples 
124      -------- 
125      >>> import numpy as np 
126      >>> from scipy.cluster.vq import whiten 
127      >>> features  = np.array([[1.9, 2.3, 1.7], 
128      ...                       [1.5, 2.5, 2.2], 
129      ...                       [0.8, 0.6, 1.7,]]) 
130      >>> whiten(features) 
131      array([[ 4.17944278,  2.69811351,  7.21248917], 
132             [ 3.29956009,  2.93273208,  9.33380951], 
133             [ 1.75976538,  0.7038557 ,  7.21248917]]) 
134   
135      """
136      xp = array_namespace(obs)
137      obs = _asarray(obs, check_finite=check_finite, xp=xp)
138      std_dev = xp.std(obs, axis=0)
139      zero_std_mask = std_dev == 0
140      if xp.any(zero_std_mask):
141          std_dev[zero_std_mask] = 1.0
142          warnings.warn("Some columns have standard deviation zero. "
143                        "The values of these columns will not change.",
144                        RuntimeWarning, stacklevel=2)
145      return obs / std_dev

148  def vq(obs, code_book, check_finite=True):
149      """ 
150      Assign codes from a code book to observations. 
151   
152      Assigns a code from a code book to each observation. Each 
153      observation vector in the 'M' by 'N' `obs` array is compared with the 
154      centroids in the code book and assigned the code of the closest 
155      centroid. 
156   
157      The features in `obs` should have unit variance, which can be 
158      achieved by passing them through the whiten function. The code 
159      book can be created with the k-means algorithm or a different 
160      encoding algorithm. 
161   
162      Parameters 
163      ---------- 
164      obs : ndarray 
165          Each row of the 'M' x 'N' array is an observation. The columns are 
166          the "features" seen during each observation. The features must be 
167          whitened first using the whiten function or something equivalent. 
168      code_book : ndarray 
169          The code book is usually generated using the k-means algorithm. 
170          Each row of the array holds a different code, and the columns are 
171          the features of the code. 
172   
173           >>> #              f0    f1    f2   f3 
174           >>> code_book = [ 
175           ...             [  1.,   2.,   3.,   4.],  #c0 
176           ...             [  1.,   2.,   3.,   4.],  #c1 
177           ...             [  1.,   2.,   3.,   4.]]  #c2 
178   
179      check_finite : bool, optional 
180          Whether to check that the input matrices contain only finite numbers. 
181          Disabling may give a performance gain, but may result in problems 
182          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
183          Default: True 
184   
185      Returns 
186      ------- 
187      code : ndarray 
188          A length M array holding the code book index for each observation. 
189      dist : ndarray 
190          The distortion (distance) between the observation and its nearest 
191          code. 
192   
193      Examples 
194      -------- 
195      >>> import numpy as np 
196      >>> from scipy.cluster.vq import vq 
197      >>> code_book = np.array([[1., 1., 1.], 
198      ...                       [2., 2., 2.]]) 
199      >>> features  = np.array([[1.9, 2.3, 1.7], 
200      ...                       [1.5, 2.5, 2.2], 
201      ...                       [0.8, 0.6, 1.7]]) 
202      >>> vq(features, code_book) 
203      (array([1, 1, 0], dtype=int32), array([0.43588989, 0.73484692, 0.83066239])) 
204   
205      """
206      xp = array_namespace(obs, code_book)
207      obs = _asarray(obs, xp=xp, check_finite=check_finite)
208      code_book = _asarray(code_book, xp=xp, check_finite=check_finite)
209      ct = xp.result_type(obs, code_book)
210  
211      c_obs = xp.astype(obs, ct, copy=False)
212      c_code_book = xp.astype(code_book, ct, copy=False)
213  
214      if xp.isdtype(ct, kind='real floating'):
215          c_obs = np.asarray(c_obs)
216          c_code_book = np.asarray(c_code_book)
217          result = _vq.vq(c_obs, c_code_book)
218          return xp.asarray(result[0]), xp.asarray(result[1])
219      return py_vq(obs, code_book, check_finite=False)

222  def py_vq(obs, code_book, check_finite=True):
223      """ Python version of vq algorithm. 
224   
225      The algorithm computes the Euclidean distance between each 
226      observation and every frame in the code_book. 
227   
228      Parameters 
229      ---------- 
230      obs : ndarray 
231          Expects a rank 2 array. Each row is one observation. 
232      code_book : ndarray 
233          Code book to use. Same format than obs. Should have same number of 
234          features (e.g., columns) than obs. 
235      check_finite : bool, optional 
236          Whether to check that the input matrices contain only finite numbers. 
237          Disabling may give a performance gain, but may result in problems 
238          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
239          Default: True 
240   
241      Returns 
242      ------- 
243      code : ndarray 
244          code[i] gives the label of the ith obversation; its code is 
245          code_book[code[i]]. 
246      mind_dist : ndarray 
247          min_dist[i] gives the distance between the ith observation and its 
248          corresponding code. 
249   
250      Notes 
251      ----- 
252      This function is slower than the C version but works for 
253      all input types. If the inputs have the wrong types for the 
254      C versions of the function, this one is called as a last resort. 
255   
256      It is about 20 times slower than the C version. 
257   
258      """
259      xp = array_namespace(obs, code_book)
260      obs = _asarray(obs, xp=xp, check_finite=check_finite)
261      code_book = _asarray(code_book, xp=xp, check_finite=check_finite)
262  
263      if obs.ndim != code_book.ndim:
264          raise ValueError("Observation and code_book should have the same rank")
265  
266      if obs.ndim == 1:
267          obs = obs[:, xp.newaxis]
268          code_book = code_book[:, xp.newaxis]
269  
270      # Once `cdist` has array API support, this `xp.asarray` call can be removed
271      dist = xp.asarray(cdist(obs, code_book))
272      code = xp.argmin(dist, axis=1)
273      min_dist = xp.min(dist, axis=1)
274      return code, min_dist

277  def _kmeans(obs, guess, thresh=1e-5, xp=None):
278      """ "raw" version of k-means. 
279   
280      Returns 
281      ------- 
282      code_book 
283          The lowest distortion codebook found. 
284      avg_dist 
285          The average distance a observation is from a code in the book. 
286          Lower means the code_book matches the data better. 
287   
288      See Also 
289      -------- 
290      kmeans : wrapper around k-means 
291   
292      Examples 
293      -------- 
294      Note: not whitened in this example. 
295   
296      >>> import numpy as np 
297      >>> from scipy.cluster.vq import _kmeans 
298      >>> features  = np.array([[ 1.9,2.3], 
299      ...                       [ 1.5,2.5], 
300      ...                       [ 0.8,0.6], 
301      ...                       [ 0.4,1.8], 
302      ...                       [ 1.0,1.0]]) 
303      >>> book = np.array((features[0],features[2])) 
304      >>> _kmeans(features,book) 
305      (array([[ 1.7       ,  2.4       ], 
306             [ 0.73333333,  1.13333333]]), 0.40563916697728591) 
307   
308      """
309      xp = np if xp is None else xp
310      code_book = guess
311      diff = xp.inf
312      prev_avg_dists = deque([diff], maxlen=2)
313      while diff > thresh:
314          # compute membership and distances between obs and code_book
315          obs_code, distort = vq(obs, code_book, check_finite=False)
316          prev_avg_dists.append(xp.mean(distort, axis=-1))
317          # recalc code_book as centroids of associated obs
318          obs = np.asarray(obs)
319          obs_code = np.asarray(obs_code)
320          code_book, has_members = _vq.update_cluster_means(obs, obs_code,
321                                                            code_book.shape[0])
322          obs = xp.asarray(obs)
323          obs_code = xp.asarray(obs_code)
324          code_book = xp.asarray(code_book)
325          has_members = xp.asarray(has_members)
326          code_book = code_book[has_members]
327          diff = xp.abs(prev_avg_dists[0] - prev_avg_dists[1])
328  
329      return code_book, prev_avg_dists[1]

332  @_transition_to_rng("seed")
333  def kmeans(obs, k_or_guess, iter=20, thresh=1e-5, check_finite=True,
334             *, rng=None):
335      """ 
336      Performs k-means on a set of observation vectors forming k clusters. 
337   
338      The k-means algorithm adjusts the classification of the observations 
339      into clusters and updates the cluster centroids until the position of 
340      the centroids is stable over successive iterations. In this 
341      implementation of the algorithm, the stability of the centroids is 
342      determined by comparing the absolute value of the change in the average 
343      Euclidean distance between the observations and their corresponding 
344      centroids against a threshold. This yields 
345      a code book mapping centroids to codes and vice versa. 
346   
347      Parameters 
348      ---------- 
349      obs : ndarray 
350         Each row of the M by N array is an observation vector. The 
351         columns are the features seen during each observation. 
352         The features must be whitened first with the `whiten` function. 
353   
354      k_or_guess : int or ndarray 
355         The number of centroids to generate. A code is assigned to 
356         each centroid, which is also the row index of the centroid 
357         in the code_book matrix generated. 
358   
359         The initial k centroids are chosen by randomly selecting 
360         observations from the observation matrix. Alternatively, 
361         passing a k by N array specifies the initial k centroids. 
362   
363      iter : int, optional 
364         The number of times to run k-means, returning the codebook 
365         with the lowest distortion. This argument is ignored if 
366         initial centroids are specified with an array for the 
367         ``k_or_guess`` parameter. This parameter does not represent the 
368         number of iterations of the k-means algorithm. 
369   
370      thresh : float, optional 
371         Terminates the k-means algorithm if the change in 
372         distortion since the last k-means iteration is less than 
373         or equal to threshold. 
374   
375      check_finite : bool, optional 
376          Whether to check that the input matrices contain only finite numbers. 
377          Disabling may give a performance gain, but may result in problems 
378          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
379          Default: True 
380      rng : `numpy.random.Generator`, optional 
381          Pseudorandom number generator state. When `rng` is None, a new 
382          `numpy.random.Generator` is created using entropy from the 
383          operating system. Types other than `numpy.random.Generator` are 
384          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
385   
386      Returns 
387      ------- 
388      codebook : ndarray 
389         A k by N array of k centroids. The ith centroid 
390         codebook[i] is represented with the code i. The centroids 
391         and codes generated represent the lowest distortion seen, 
392         not necessarily the globally minimal distortion. 
393         Note that the number of centroids is not necessarily the same as the 
394         ``k_or_guess`` parameter, because centroids assigned to no observations 
395         are removed during iterations. 
396   
397      distortion : float 
398         The mean (non-squared) Euclidean distance between the observations 
399         passed and the centroids generated. Note the difference to the standard 
400         definition of distortion in the context of the k-means algorithm, which 
401         is the sum of the squared distances. 
402   
403      See Also 
404      -------- 
405      kmeans2 : a different implementation of k-means clustering 
406         with more methods for generating initial centroids but without 
407         using a distortion change threshold as a stopping criterion. 
408   
409      whiten : must be called prior to passing an observation matrix 
410         to kmeans. 
411   
412      Notes 
413      ----- 
414      For more functionalities or optimal performance, you can use 
415      `sklearn.cluster.KMeans <https://scikit-learn.org/stable/modules/generated/sklearn.cluster.KMeans.html>`_. 
416      `This <https://hdbscan.readthedocs.io/en/latest/performance_and_scalability.html#comparison-of-high-performance-implementations>`_ 
417      is a benchmark result of several implementations. 
418   
419      Examples 
420      -------- 
421      >>> import numpy as np 
422      >>> from scipy.cluster.vq import vq, kmeans, whiten 
423      >>> import matplotlib.pyplot as plt 
424      >>> features  = np.array([[ 1.9,2.3], 
425      ...                       [ 1.5,2.5], 
426      ...                       [ 0.8,0.6], 
427      ...                       [ 0.4,1.8], 
428      ...                       [ 0.1,0.1], 
429      ...                       [ 0.2,1.8], 
430      ...                       [ 2.0,0.5], 
431      ...                       [ 0.3,1.5], 
432      ...                       [ 1.0,1.0]]) 
433      >>> whitened = whiten(features) 
434      >>> book = np.array((whitened[0],whitened[2])) 
435      >>> kmeans(whitened,book) 
436      (array([[ 2.3110306 ,  2.86287398],    # random 
437             [ 0.93218041,  1.24398691]]), 0.85684700941625547) 
438   
439      >>> codes = 3 
440      >>> kmeans(whitened,codes) 
441      (array([[ 2.3110306 ,  2.86287398],    # random 
442             [ 1.32544402,  0.65607529], 
443             [ 0.40782893,  2.02786907]]), 0.5196582527686241) 
444   
445      >>> # Create 50 datapoints in two clusters a and b 
446      >>> pts = 50 
447      >>> rng = np.random.default_rng() 
448      >>> a = rng.multivariate_normal([0, 0], [[4, 1], [1, 4]], size=pts) 
449      >>> b = rng.multivariate_normal([30, 10], 
450      ...                             [[10, 2], [2, 1]], 
451      ...                             size=pts) 
452      >>> features = np.concatenate((a, b)) 
453      >>> # Whiten data 
454      >>> whitened = whiten(features) 
455      >>> # Find 2 clusters in the data 
456      >>> codebook, distortion = kmeans(whitened, 2) 
457      >>> # Plot whitened data and cluster centers in red 
458      >>> plt.scatter(whitened[:, 0], whitened[:, 1]) 
459      >>> plt.scatter(codebook[:, 0], codebook[:, 1], c='r') 
460      >>> plt.show() 
461   
462      """
463      if isinstance(k_or_guess, int):
464          xp = array_namespace(obs)
465      else:
466          xp = array_namespace(obs, k_or_guess)
467      obs = _asarray(obs, xp=xp, check_finite=check_finite)
468      guess = _asarray(k_or_guess, xp=xp, check_finite=check_finite)
469      if iter < 1:
470          raise ValueError(f"iter must be at least 1, got {iter}")
471  
472      # Determine whether a count (scalar) or an initial guess (array) was passed.
473      if xp_size(guess) != 1:
474          if xp_size(guess) < 1:
475              raise ValueError(f"Asked for 0 clusters. Initial book was {guess}")
476          return _kmeans(obs, guess, thresh=thresh, xp=xp)
477  
478      # k_or_guess is a scalar, now verify that it's an integer
479      k = int(guess)
480      if k != guess:
481          raise ValueError("If k_or_guess is a scalar, it must be an integer.")
482      if k < 1:
483          raise ValueError("Asked for %d clusters." % k)
484  
485      rng = check_random_state(rng)
486  
487      # initialize best distance value to a large value
488      best_dist = xp.inf
489      for i in range(iter):
490          # the initial code book is randomly selected from observations
491          guess = _kpoints(obs, k, rng, xp)
492          book, dist = _kmeans(obs, guess, thresh=thresh, xp=xp)
493          if dist < best_dist:
494              best_book = book
495              best_dist = dist
496      return best_book, best_dist

499  def _kpoints(data, k, rng, xp):
500      """Pick k points at random in data (one row = one observation). 
501   
502      Parameters 
503      ---------- 
504      data : ndarray 
505          Expect a rank 1 or 2 array. Rank 1 are assumed to describe one 
506          dimensional data, rank 2 multidimensional data, in which case one 
507          row is one observation. 
508      k : int 
509          Number of samples to generate. 
510      rng : `numpy.random.Generator` or `numpy.random.RandomState` 
511          Random number generator. 
512   
513      Returns 
514      ------- 
515      x : ndarray 
516          A 'k' by 'N' containing the initial centroids 
517   
518      """
519      idx = rng.choice(data.shape[0], size=int(k), replace=False)
520      # convert to array with default integer dtype (avoids numpy#25607)
521      idx = xp.asarray(idx, dtype=xp.asarray([1]).dtype)
522      return xp.take(data, idx, axis=0)

525  def _krandinit(data, k, rng, xp):
526      """Returns k samples of a random variable whose parameters depend on data. 
527   
528      More precisely, it returns k observations sampled from a Gaussian random 
529      variable whose mean and covariances are the ones estimated from the data. 
530   
531      Parameters 
532      ---------- 
533      data : ndarray 
534          Expect a rank 1 or 2 array. Rank 1 is assumed to describe 1-D 
535          data, rank 2 multidimensional data, in which case one 
536          row is one observation. 
537      k : int 
538          Number of samples to generate. 
539      rng : `numpy.random.Generator` or `numpy.random.RandomState` 
540          Random number generator. 
541   
542      Returns 
543      ------- 
544      x : ndarray 
545          A 'k' by 'N' containing the initial centroids 
546   
547      """
548      mu = xp.mean(data, axis=0)
549      k = np.asarray(k)
550  
551      if data.ndim == 1:
552          _cov = xpx.cov(data, xp=xp)
553          x = rng.standard_normal(size=k)
554          x = xp.asarray(x)
555          x *= xp.sqrt(_cov)
556      elif data.shape[1] > data.shape[0]:
557          # initialize when the covariance matrix is rank deficient
558          _, s, vh = xp.linalg.svd(data - mu, full_matrices=False)
559          x = rng.standard_normal(size=(k, xp_size(s)))
560          x = xp.asarray(x)
561          sVh = s[:, None] * vh / xp.sqrt(data.shape[0] - xp.asarray(1.))
562          x = x @ sVh
563      else:
564          _cov = xpx.atleast_nd(xpx.cov(data.T, xp=xp), ndim=2, xp=xp)
565  
566          # k rows, d cols (one row = one obs)
567          # Generate k sample of a random variable ~ Gaussian(mu, cov)
568          x = rng.standard_normal(size=(k, xp_size(mu)))
569          x = xp.asarray(x)
570          x = x @ xp.linalg.cholesky(_cov).T
571  
572      x += mu
573      return x

576  def _kpp(data, k, rng, xp):
577      """ Picks k points in the data based on the kmeans++ method. 
578   
579      Parameters 
580      ---------- 
581      data : ndarray 
582          Expect a rank 1 or 2 array. Rank 1 is assumed to describe 1-D 
583          data, rank 2 multidimensional data, in which case one 
584          row is one observation. 
585      k : int 
586          Number of samples to generate. 
587      rng : `numpy.random.Generator` or `numpy.random.RandomState` 
588          Random number generator. 
589   
590      Returns 
591      ------- 
592      init : ndarray 
593          A 'k' by 'N' containing the initial centroids. 
594   
595      References 
596      ---------- 
597      .. [1] D. Arthur and S. Vassilvitskii, "k-means++: the advantages of 
598         careful seeding", Proceedings of the Eighteenth Annual ACM-SIAM Symposium 
599         on Discrete Algorithms, 2007. 
600      """
601  
602      ndim = len(data.shape)
603      if ndim == 1:
604          data = data[:, None]
605  
606      dims = data.shape[1]
607  
608      init = xp.empty((int(k), dims))
609  
610      for i in range(k):
611          if i == 0:
612              init[i, :] = data[rng_integers(rng, data.shape[0]), :]
613  
614          else:
615              D2 = cdist(init[:i,:], data, metric='sqeuclidean').min(axis=0)
616              probs = D2/D2.sum()
617              cumprobs = probs.cumsum()
618              r = rng.uniform()
619              cumprobs = np.asarray(cumprobs)
620              init[i, :] = data[np.searchsorted(cumprobs, r), :]
621  
622      if ndim == 1:
623          init = init[:, 0]
624      return init

646  @_transition_to_rng("seed")
647  def kmeans2(data, k, iter=10, thresh=1e-5, minit='random',
648              missing='warn', check_finite=True, *, rng=None):
649      """ 
650      Classify a set of observations into k clusters using the k-means algorithm. 
651   
652      The algorithm attempts to minimize the Euclidean distance between 
653      observations and centroids. Several initialization methods are 
654      included. 
655   
656      Parameters 
657      ---------- 
658      data : ndarray 
659          A 'M' by 'N' array of 'M' observations in 'N' dimensions or a length 
660          'M' array of 'M' 1-D observations. 
661      k : int or ndarray 
662          The number of clusters to form as well as the number of 
663          centroids to generate. If `minit` initialization string is 
664          'matrix', or if a ndarray is given instead, it is 
665          interpreted as initial cluster to use instead. 
666      iter : int, optional 
667          Number of iterations of the k-means algorithm to run. Note 
668          that this differs in meaning from the iters parameter to 
669          the kmeans function. 
670      thresh : float, optional 
671          (not used yet) 
672      minit : str, optional 
673          Method for initialization. Available methods are 'random', 
674          'points', '++' and 'matrix': 
675   
676          'random': generate k centroids from a Gaussian with mean and 
677          variance estimated from the data. 
678   
679          'points': choose k observations (rows) at random from data for 
680          the initial centroids. 
681   
682          '++': choose k observations accordingly to the kmeans++ method 
683          (careful seeding) 
684   
685          'matrix': interpret the k parameter as a k by M (or length k 
686          array for 1-D data) array of initial centroids. 
687      missing : str, optional 
688          Method to deal with empty clusters. Available methods are 
689          'warn' and 'raise': 
690   
691          'warn': give a warning and continue. 
692   
693          'raise': raise an ClusterError and terminate the algorithm. 
694      check_finite : bool, optional 
695          Whether to check that the input matrices contain only finite numbers. 
696          Disabling may give a performance gain, but may result in problems 
697          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
698          Default: True 
699      rng : `numpy.random.Generator`, optional 
700          Pseudorandom number generator state. When `rng` is None, a new 
701          `numpy.random.Generator` is created using entropy from the 
702          operating system. Types other than `numpy.random.Generator` are 
703          passed to `numpy.random.default_rng` to instantiate a ``Generator``. 
704   
705      Returns 
706      ------- 
707      centroid : ndarray 
708          A 'k' by 'N' array of centroids found at the last iteration of 
709          k-means. 
710      label : ndarray 
711          label[i] is the code or index of the centroid the 
712          ith observation is closest to. 
713   
714      See Also 
715      -------- 
716      kmeans 
717   
718      References 
719      ---------- 
720      .. [1] D. Arthur and S. Vassilvitskii, "k-means++: the advantages of 
721         careful seeding", Proceedings of the Eighteenth Annual ACM-SIAM Symposium 
722         on Discrete Algorithms, 2007. 
723   
724      Examples 
725      -------- 
726      >>> from scipy.cluster.vq import kmeans2 
727      >>> import matplotlib.pyplot as plt 
728      >>> import numpy as np 
729   
730      Create z, an array with shape (100, 2) containing a mixture of samples 
731      from three multivariate normal distributions. 
732   
733      >>> rng = np.random.default_rng() 
734      >>> a = rng.multivariate_normal([0, 6], [[2, 1], [1, 1.5]], size=45) 
735      >>> b = rng.multivariate_normal([2, 0], [[1, -1], [-1, 3]], size=30) 
736      >>> c = rng.multivariate_normal([6, 4], [[5, 0], [0, 1.2]], size=25) 
737      >>> z = np.concatenate((a, b, c)) 
738      >>> rng.shuffle(z) 
739   
740      Compute three clusters. 
741   
742      >>> centroid, label = kmeans2(z, 3, minit='points') 
743      >>> centroid 
744      array([[ 2.22274463, -0.61666946],  # may vary 
745             [ 0.54069047,  5.86541444], 
746             [ 6.73846769,  4.01991898]]) 
747   
748      How many points are in each cluster? 
749   
750      >>> counts = np.bincount(label) 
751      >>> counts 
752      array([29, 51, 20])  # may vary 
753   
754      Plot the clusters. 
755   
756      >>> w0 = z[label == 0] 
757      >>> w1 = z[label == 1] 
758      >>> w2 = z[label == 2] 
759      >>> plt.plot(w0[:, 0], w0[:, 1], 'o', alpha=0.5, label='cluster 0') 
760      >>> plt.plot(w1[:, 0], w1[:, 1], 'd', alpha=0.5, label='cluster 1') 
761      >>> plt.plot(w2[:, 0], w2[:, 1], 's', alpha=0.5, label='cluster 2') 
762      >>> plt.plot(centroid[:, 0], centroid[:, 1], 'k*', label='centroids') 
763      >>> plt.axis('equal') 
764      >>> plt.legend(shadow=True) 
765      >>> plt.show() 
766   
767      """
768      if int(iter) < 1:
769          raise ValueError(f"Invalid iter ({iter}), must be a positive integer.")
770      try:
771          miss_meth = _valid_miss_meth[missing]
772      except KeyError as e:
773          raise ValueError(f"Unknown missing method {missing!r}") from e
774  
775      if isinstance(k, int):
776          xp = array_namespace(data)
777      else:
778          xp = array_namespace(data, k)
779      data = _asarray(data, xp=xp, check_finite=check_finite)
780      code_book = xp_copy(k, xp=xp)
781      if data.ndim == 1:
782          d = 1
783      elif data.ndim == 2:
784          d = data.shape[1]
785      else:
786          raise ValueError("Input of rank > 2 is not supported.")
787  
788      if xp_size(data) < 1 or xp_size(code_book) < 1:
789          raise ValueError("Empty input is not supported.")
790  
791      # If k is not a single value, it should be compatible with data's shape
792      if minit == 'matrix' or xp_size(code_book) > 1:
793          if data.ndim != code_book.ndim:
794              raise ValueError("k array doesn't match data rank")
795          nc = code_book.shape[0]
796          if data.ndim > 1 and code_book.shape[1] != d:
797              raise ValueError("k array doesn't match data dimension")
798      else:
799          nc = int(code_book)
800  
801          if nc < 1:
802              raise ValueError("Cannot ask kmeans2 for %d clusters"
803                               " (k was %s)" % (nc, code_book))
804          elif nc != code_book:
805              warnings.warn("k was not an integer, was converted.", stacklevel=2)
806  
807          try:
808              init_meth = _valid_init_meth[minit]
809          except KeyError as e:
810              raise ValueError(f"Unknown init method {minit!r}") from e
811          else:
812              rng = check_random_state(rng)
813              code_book = init_meth(data, code_book, rng, xp)
814  
815      data = np.asarray(data)
816      code_book = np.asarray(code_book)
817      for i in range(iter):
818          # Compute the nearest neighbor for each obs using the current code book
819          label = vq(data, code_book, check_finite=check_finite)[0]
820          # Update the code book by computing centroids
821          new_code_book, has_members = _vq.update_cluster_means(data, label, nc)
822          if not has_members.all():
823              miss_meth()
824              # Set the empty clusters to their previous positions
825              new_code_book[~has_members] = code_book[~has_members]
826          code_book = new_code_book
827  
828      return xp.asarray(code_book), xp.asarray(label)
