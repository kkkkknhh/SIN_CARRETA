36   class _BarycentricRational:
37       """Base class for Barycentric representation of a rational function."""
38       def __init__(self, x, y, **kwargs):
39           # input validation
40           z = np.asarray(x)
41           f = np.asarray(y)
42   
43           self._input_validation(z, f, **kwargs)
44   
45           # Remove infinite or NaN function values and repeated entries
46           to_keep = np.logical_and.reduce(
47               ((np.isfinite(f)) & (~np.isnan(f))).reshape(f.shape[0], -1),
48               axis=-1
49           )
50           f = f[to_keep, ...]
51           z = z[to_keep]
52           z, uni = np.unique(z, return_index=True)
53           f = f[uni, ...]
54   
55           self._shape = f.shape[1:]
56           self._support_points, self._support_values, self.weights = (
57               self._compute_weights(z, f, **kwargs)
58           )
59   
60           # only compute once
61           self._poles = None
62           self._residues = None
63           self._roots = None
64   
65       def _input_validation(self, x, y, **kwargs):
66           if x.ndim != 1:
67               raise ValueError("`x` must be 1-D.")
68   
69           if not y.ndim >= 1:
70               raise ValueError("`y` must be at least 1-D.")
71   
72           if x.size != y.shape[0]:
73               raise ValueError("`x` be the same size as the first dimension of `y`.")
74   
75           if not np.all(np.isfinite(x)):
76               raise ValueError("`x` must be finite.")
77   
78       def _compute_weights(z, f, **kwargs):
79           raise NotImplementedError
80   
81       def __call__(self, z):
82           """Evaluate the rational approximation at given values. 
83    
84           Parameters 
85           ---------- 
86           z : array_like 
87               Input values. 
88           """
89           # evaluate rational function in barycentric form.
90           z = np.asarray(z)
91           zv = np.ravel(z)
92   
93           support_values = self._support_values.reshape(
94               (self._support_values.shape[0], -1)
95           )
96           weights = self.weights[..., np.newaxis]
97   
98           # Cauchy matrix
99           # Ignore errors due to inf/inf at support points, these will be fixed later
100          with np.errstate(invalid="ignore", divide="ignore"):
101              CC = 1 / np.subtract.outer(zv, self._support_points)
102              # Vector of values
103              r = CC @ (weights * support_values) / (CC @ weights)
104  
105          # Deal with input inf: `r(inf) = lim r(z) = sum(w*f) / sum(w)`
106          if np.any(np.isinf(zv)):
107              r[np.isinf(zv)] = (np.sum(weights * support_values)
108                                 / np.sum(weights))
109  
110          # Deal with NaN
111          ii = np.nonzero(np.isnan(r))[0]
112          for jj in ii:
113              if np.isnan(zv[jj]) or not np.any(zv[jj] == self._support_points):
114                  # r(NaN) = NaN is fine.
115                  # The second case may happen if `r(zv[ii]) = 0/0` at some point.
116                  pass
117              else:
118                  # Clean up values `NaN = inf/inf` at support points.
119                  # Find the corresponding node and set entry to correct value:
120                  r[jj] = support_values[zv[jj] == self._support_points].squeeze()
121  
122          return np.reshape(r, z.shape + self._shape)
123  
124      def poles(self):
125          """Compute the poles of the rational approximation. 
126   
127          Returns 
128          ------- 
129          poles : array 
130              Poles of the AAA approximation, repeated according to their multiplicity 
131              but not in any specific order. 
132          """
133          if self._poles is None:
134              # Compute poles via generalized eigenvalue problem
135              m = self.weights.size
136              B = np.eye(m + 1, dtype=self.weights.dtype)
137              B[0, 0] = 0
138  
139              E = np.zeros_like(B, dtype=np.result_type(self.weights,
140                                                        self._support_points))
141              E[0, 1:] = self.weights
142              E[1:, 0] = 1
143              np.fill_diagonal(E[1:, 1:], self._support_points)
144  
145              pol = scipy.linalg.eigvals(E, B)
146              self._poles = pol[np.isfinite(pol)]
147          return self._poles
148  
149      def residues(self):
150          """Compute the residues of the poles of the approximation. 
151   
152          Returns 
153          ------- 
154          residues : array 
155              Residues associated with the `poles` of the approximation 
156          """
157          if self._residues is None:
158              # Compute residues via formula for res of quotient of analytic functions
159              with np.errstate(divide="ignore", invalid="ignore"):
160                  N = (1/(np.subtract.outer(self.poles(), self._support_points))) @ (
161                      self._support_values * self.weights
162                  )
163                  Ddiff = (
164                      -((1/np.subtract.outer(self.poles(), self._support_points))**2)
165                      @ self.weights
166                  )
167                  self._residues = N / Ddiff
168          return self._residues
169  
170      def roots(self):
171          """Compute the zeros of the rational approximation. 
172   
173          Returns 
174          ------- 
175          zeros : array 
176              Zeros of the AAA approximation, repeated according to their multiplicity 
177              but not in any specific order. 
178          """
179          if self._roots is None:
180              # Compute zeros via generalized eigenvalue problem
181              m = self.weights.size
182              B = np.eye(m + 1, dtype=self.weights.dtype)
183              B[0, 0] = 0
184              E = np.zeros_like(B, dtype=np.result_type(self.weights,
185                                                        self._support_values,
186                                                        self._support_points))
187              E[0, 1:] = self.weights * self._support_values
188              E[1:, 0] = 1
189              np.fill_diagonal(E[1:, 1:], self._support_points)
190  
191              zer = scipy.linalg.eigvals(E, B)
192              self._roots = zer[np.isfinite(zer)]
193          return self._roots

196  class AAA(_BarycentricRational):
197      r""" 
198      AAA real or complex rational approximation. 
199   
200      As described in [1]_, the AAA algorithm is a greedy algorithm for approximation by 
201      rational functions on a real or complex set of points. The rational approximation is 
202      represented in a barycentric form from which the roots (zeros), poles, and residues 
203      can be computed. 
204   
205      Parameters 
206      ---------- 
207      x : 1D array_like, shape (n,) 
208          1-D array containing values of the independent variable. Values may be real or 
209          complex but must be finite. 
210      y : 1D array_like, shape (n,) 
211          Function values ``f(x)``. Infinite and NaN values of `values` and 
212          corresponding values of `points` will be discarded. 
213      rtol : float, optional 
214          Relative tolerance, defaults to ``eps**0.75``. If a small subset of the entries 
215          in `values` are much larger than the rest the default tolerance may be too 
216          loose. If the tolerance is too tight then the approximation may contain 
217          Froissart doublets or the algorithm may fail to converge entirely. 
218      max_terms : int, optional 
219          Maximum number of terms in the barycentric representation, defaults to ``100``. 
220          Must be greater than or equal to one. 
221      clean_up : bool, optional 
222          Automatic removal of Froissart doublets, defaults to ``True``. See notes for 
223          more details. 
224      clean_up_tol : float, optional 
225          Poles with residues less than this number times the geometric mean 
226          of `values` times the minimum distance to `points` are deemed spurious by the 
227          cleanup procedure, defaults to 1e-13. See notes for more details. 
228   
229      Attributes 
230      ---------- 
231      support_points : array 
232          Support points of the approximation. These are a subset of the provided `x` at 
233          which the approximation strictly interpolates `y`. 
234          See notes for more details. 
235      support_values : array 
236          Value of the approximation at the `support_points`. 
237      weights : array 
238          Weights of the barycentric approximation. 
239      errors : array 
240          Error :math:`|f(z) - r(z)|_\infty` over `points` in the successive iterations 
241          of AAA. 
242   
243      Warns 
244      ----- 
245      RuntimeWarning 
246          If `rtol` is not achieved in `max_terms` iterations. 
247   
248      See Also 
249      -------- 
250      FloaterHormannInterpolator : Floater-Hormann barycentric rational interpolation. 
251      pade : Pad√© approximation. 
252   
253      Notes 
254      ----- 
255      At iteration :math:`m` (at which point there are :math:`m` terms in the both the 
256      numerator and denominator of the approximation), the 
257      rational approximation in the AAA algorithm takes the barycentric form 
258   
259      .. math:: 
260   
261          r(z) = n(z)/d(z) = 
262          \frac{\sum_{j=1}^m\ w_j f_j / (z - z_j)}{\sum_{j=1}^m w_j / (z - z_j)}, 
263   
264      where :math:`z_1,\dots,z_m` are real or complex support points selected from 
265      `x`, :math:`f_1,\dots,f_m` are the corresponding real or complex data values 
266      from `y`, and :math:`w_1,\dots,w_m` are real or complex weights. 
267   
268      Each iteration of the algorithm has two parts: the greedy selection the next support 
269      point and the computation of the weights. The first part of each iteration is to 
270      select the next support point to be added :math:`z_{m+1}` from the remaining 
271      unselected `x`, such that the nonlinear residual 
272      :math:`|f(z_{m+1}) - n(z_{m+1})/d(z_{m+1})|` is maximised. The algorithm terminates 
273      when this maximum is less than ``rtol * np.linalg.norm(f, ord=np.inf)``. This means 
274      the interpolation property is only satisfied up to a tolerance, except at the 
275      support points where approximation exactly interpolates the supplied data. 
276   
277      In the second part of each iteration, the weights :math:`w_j` are selected to solve 
278      the least-squares problem 
279   
280      .. math:: 
281   
282          \text{minimise}_{w_j}|fd - n| \quad \text{subject to} \quad 
283          \sum_{j=1}^{m+1} w_j = 1, 
284   
285      over the unselected elements of `x`. 
286   
287      One of the challenges with working with rational approximations is the presence of 
288      Froissart doublets, which are either poles with vanishingly small residues or 
289      pole-zero pairs that are close enough together to nearly cancel, see [2]_. The 
290      greedy nature of the AAA algorithm means Froissart doublets are rare. However, if 
291      `rtol` is set too tight then the approximation will stagnate and many Froissart 
292      doublets will appear. Froissart doublets can usually be removed by removing support 
293      points and then resolving the least squares problem. The support point :math:`z_j`, 
294      which is the closest support point to the pole :math:`a` with residue 
295      :math:`\alpha`, is removed if the following is satisfied 
296   
297      .. math:: 
298   
299          |\alpha| / |z_j - a| < \verb|clean_up_tol| \cdot \tilde{f}, 
300   
301      where :math:`\tilde{f}` is the geometric mean of `support_values`. 
302   
303   
304      References 
305      ---------- 
306      .. [1] Y. Nakatsukasa, O. Sete, and L. N. Trefethen, "The AAA algorithm for 
307              rational approximation", SIAM J. Sci. Comp. 40 (2018), A1494-A1522. 
308              :doi:`10.1137/16M1106122` 
309      .. [2] J. Gilewicz and M. Pindor, Pade approximants and noise: rational functions, 
310             J. Comp. Appl. Math. 105 (1999), pp. 285-297. 
311             :doi:`10.1016/S0377-0427(02)00674-X` 
312   
313      Examples 
314      -------- 
315   
316      Here we reproduce a number of the numerical examples from [1]_ as a demonstration 
317      of the functionality offered by this method. 
318   
319      >>> import numpy as np 
320      >>> import matplotlib.pyplot as plt 
321      >>> from scipy.interpolate import AAA 
322      >>> import warnings 
323   
324      For the first example we approximate the gamma function on ``[-3.5, 4.5]`` by 
325      extrapolating from 100 samples in ``[-1.5, 1.5]``. 
326   
327      >>> from scipy.special import gamma 
328      >>> sample_points = np.linspace(-1.5, 1.5, num=100) 
329      >>> r = AAA(sample_points, gamma(sample_points)) 
330      >>> z = np.linspace(-3.5, 4.5, num=1000) 
331      >>> fig, ax = plt.subplots() 
332      >>> ax.plot(z, gamma(z), label="Gamma") 
333      >>> ax.plot(sample_points, gamma(sample_points), label="Sample points") 
334      >>> ax.plot(z, r(z).real, '--', label="AAA approximation") 
335      >>> ax.set(xlabel="z", ylabel="r(z)", ylim=[-8, 8], xlim=[-3.5, 4.5]) 
336      >>> ax.legend() 
337      >>> plt.show() 
338   
339      We can also view the poles of the rational approximation and their residues: 
340   
341      >>> order = np.argsort(r.poles()) 
342      >>> r.poles()[order] 
343      array([-3.81591039e+00+0.j        , -3.00269049e+00+0.j        , 
344             -1.99999988e+00+0.j        , -1.00000000e+00+0.j        , 
345              5.85842812e-17+0.j        ,  4.77485458e+00-3.06919376j, 
346              4.77485458e+00+3.06919376j,  5.29095868e+00-0.97373072j, 
347              5.29095868e+00+0.97373072j]) 
348      >>> r.residues()[order]

352             -0.81132013 +2.30193429j,  0.87326839+10.70148546j, 
353              0.87326839-10.70148546j]) 
354   
355      For the second example, we call `AAA` with a spiral of 1000 points that wind 7.5 
356      times around the origin in the complex plane. 
357   
358      >>> z = np.exp(np.linspace(-0.5, 0.5 + 15j*np.pi, 1000)) 
359      >>> r = AAA(z, np.tan(np.pi*z/2), rtol=1e-13) 
360   
361      We see that AAA takes 12 steps to converge with the following errors: 
362   
363      >>> r.errors.size 
364      12 
365      >>> r.errors 
366      array([2.49261500e+01, 4.28045609e+01, 1.71346935e+01, 8.65055336e-02, 
367             1.27106444e-02, 9.90889874e-04, 5.86910543e-05, 1.28735561e-06, 
368             3.57007424e-08, 6.37007837e-10, 1.67103357e-11, 1.17112299e-13]) 
369   
370      We can also plot the computed poles: 
371   
372      >>> fig, ax = plt.subplots() 
373      >>> ax.plot(z.real, z.imag, '.', markersize=2, label="Sample points") 
374      >>> ax.plot(r.poles().real, r.poles().imag, '.', markersize=5, 
375      ...         label="Computed poles") 
376      >>> ax.set(xlim=[-3.5, 3.5], ylim=[-3.5, 3.5], aspect="equal") 
377      >>> ax.legend() 
378      >>> plt.show() 
379   
380      We now demonstrate the removal of Froissart doublets using the `clean_up` method 
381      using an example from [1]_. Here we approximate the function 
382      :math:`f(z)=\log(2 + z^4)/(1 + 16z^4)` by sampling it at 1000 roots of unity. The 
383      algorithm is run with ``rtol=0`` and ``clean_up=False`` to deliberately cause 
384      Froissart doublets to appear. 
385   
386      >>> z = np.exp(1j*2*np.pi*np.linspace(0,1, num=1000)) 
387      >>> def f(z): 
388      ...     return np.log(2 + z**4)/(1 - 16*z**4) 
389      >>> with warnings.catch_warnings():  # filter convergence warning due to rtol=0 
390      ...     warnings.simplefilter('ignore', RuntimeWarning) 
391      ...     r = AAA(z, f(z), rtol=0, max_terms=50, clean_up=False) 
392      >>> mask = np.abs(r.residues()) < 1e-13 
393      >>> fig, axs = plt.subplots(ncols=2) 
394      >>> axs[0].plot(r.poles().real[~mask], r.poles().imag[~mask], '.') 
395      >>> axs[0].plot(r.poles().real[mask], r.poles().imag[mask], 'r.') 
396   
397      Now we call the `clean_up` method to remove Froissart doublets. 
398   
399      >>> with warnings.catch_warnings(): 
400      ...     warnings.simplefilter('ignore', RuntimeWarning) 
401      ...     r.clean_up() 
402      4 
403      >>> mask = np.abs(r.residues()) < 1e-13 
404      >>> axs[1].plot(r.poles().real[~mask], r.poles().imag[~mask], '.') 
405      >>> axs[1].plot(r.poles().real[mask], r.poles().imag[mask], 'r.') 
406      >>> plt.show() 
407   
408      The left image shows the poles prior of the approximation ``clean_up=False`` with 
409      poles with residue less than ``10^-13`` in absolute value shown in red. The right 
410      image then shows the poles after the `clean_up` method has been called. 
411      """
412      def __init__(self, x, y, *, rtol=None, max_terms=100, clean_up=True,
413                   clean_up_tol=1e-13):
414          super().__init__(x, y, rtol=rtol, max_terms=max_terms)
415  
416          if clean_up:
417              self.clean_up(clean_up_tol)

419      def _input_validation(self, x, y, rtol=None, max_terms=100, clean_up=True,
420                            clean_up_tol=1e-13):
421          max_terms = operator.index(max_terms)
422          if max_terms < 1:
423              raise ValueError("`max_terms` must be an integer value greater than or "
424                               "equal to one.")
425  
426          if y.ndim != 1:
427              raise ValueError("`y` must be 1-D.")
428  
429          super()._input_validation(x, y)

431      @property
432      def support_points(self):
433          return self._support_points
434  
435      @property
436      def support_values(self):
437          return self._support_values

439      def _compute_weights(self, z, f, rtol, max_terms):
440          # Initialization for AAA iteration
441          M = np.size(z)
442          mask = np.ones(M, dtype=np.bool_)
443          dtype = np.result_type(z, f, 1.0)
444          rtol = np.finfo(dtype).eps**0.75 if rtol is None else rtol
445          atol = rtol * np.linalg.norm(f, ord=np.inf)
446          zj = np.empty(max_terms, dtype=dtype)
447          fj = np.empty(max_terms, dtype=dtype)
448          # Cauchy matrix
449          C = np.empty((M, max_terms), dtype=dtype)
450          # Loewner matrix
451          A = np.empty((M, max_terms), dtype=dtype)
452          errors = np.empty(max_terms, dtype=A.real.dtype)
453          R = np.repeat(np.mean(f), M)
454  
455          # AAA iteration
456          for m in range(max_terms):
457              # Introduce next support point
458              # Select next support point
459              jj = np.argmax(np.abs(f[mask] - R[mask]))
460              # Update support points
461              zj[m] = z[mask][jj]
462              # Update data values
463              fj[m] = f[mask][jj]
464              # Next column of Cauchy matrix
465              # Ignore errors as we manually interpolate at support points
466              with np.errstate(divide="ignore", invalid="ignore"):
467                  C[:, m] = 1 / (z - z[mask][jj])
468              # Update mask
469              mask[np.nonzero(mask)[0][jj]] = False
470              # Update Loewner matrix
471              # Ignore errors as inf values will be masked out in SVD call
472              with np.errstate(invalid="ignore"):
473                  A[:, m] = (f - fj[m]) * C[:, m]
474  
475              # Compute weights
476              rows = mask.sum()
477              if rows >= m + 1:
478                  # The usual tall-skinny case
479                  _, s, V = scipy.linalg.svd(
480                      A[mask, : m + 1], full_matrices=False, check_finite=False,
481                  )
482                  # Treat case of multiple min singular values
483                  mm = s == np.min(s)
484                  # Aim for non-sparse weight vector
485                  wj = (V.conj()[mm, :].sum(axis=0) / np.sqrt(mm.sum())).astype(dtype)
486              else:
487                  # Fewer rows than columns
488                  V = scipy.linalg.null_space(A[mask, : m + 1], check_finite=False)
489                  nm = V.shape[-1]
490                  # Aim for non-sparse wt vector
491                  wj = V.sum(axis=-1) / np.sqrt(nm)
492  
493              # Compute rational approximant
494              # Omit columns with `wj == 0`
495              i0 = wj != 0
496              # Ignore errors as we manually interpolate at support points
497              with np.errstate(invalid="ignore"):
498                  # Numerator
499                  N = C[:, : m + 1][:, i0] @ (wj[i0] * fj[: m + 1][i0])
500                  # Denominator
501                  D = C[:, : m + 1][:, i0] @ wj[i0]
502              # Interpolate at support points with `wj !=0`
503              D_inf = np.isinf(D) | np.isnan(D)
504              D[D_inf] = 1
505              N[D_inf] = f[D_inf]
506              R = N / D
507  
508              # Check if converged
509              max_error = np.linalg.norm(f - R, ord=np.inf)
510              errors[m] = max_error
511              if max_error <= atol:
512                  break
513  
514          if m == max_terms - 1:
515              warnings.warn(f"AAA failed to converge within {max_terms} iterations.",
516                            RuntimeWarning, stacklevel=2)
517  
518          # Trim off unused array allocation
519          zj = zj[: m + 1]
520          fj = fj[: m + 1]
521  
522          # Remove support points with zero weight
523          i_non_zero = wj != 0
524          self.errors = errors[: m + 1]
525          self._points = z
526          self._values = f
527          return zj[i_non_zero], fj[i_non_zero], wj[i_non_zero]
528  
529      def clean_up(self, cleanup_tol=1e-13):
530          """Automatic removal of Froissart doublets. 
531   
532          Parameters 
533          ---------- 
534          cleanup_tol : float, optional 
535              Poles with residues less than this number times the geometric mean 
536              of `values` times the minimum distance to `points` are deemed spurious by 
537              the cleanup procedure, defaults to 1e-13. 
538   
539          Returns 
540          ------- 
541          int 
542              Number of Froissart doublets detected 
543          """
544          # Find negligible residues
545          geom_mean_abs_f = scipy.stats.gmean(np.abs(self._values))
546  
547          Z_distances = np.min(
548              np.abs(np.subtract.outer(self.poles(), self._points)), axis=1
549          )
550  
551          with np.errstate(divide="ignore", invalid="ignore"):
552              ii = np.nonzero(
553                  np.abs(self.residues()) / Z_distances < cleanup_tol * geom_mean_abs_f
554              )
555  
556          ni = ii[0].size
557          if ni == 0:
558              return ni
559  
560          warnings.warn(f"{ni} Froissart doublets detected.", RuntimeWarning,
561                          stacklevel=2)
562  
563          # For each spurious pole find and remove closest support point
564          closest_spt_point = np.argmin(
565              np.abs(np.subtract.outer(self._support_points, self.poles()[ii])), axis=0
566          )
567          self._support_points = np.delete(self._support_points, closest_spt_point)
568          self._support_values = np.delete(self._support_values, closest_spt_point)
569  
570          # Remove support points z from sample set
571          mask = np.logical_and.reduce(
572              np.not_equal.outer(self._points, self._support_points), axis=1
573          )
574          f = self._values[mask]
575          z = self._points[mask]
576  
577          # recompute weights, we resolve the least squares problem for the remaining
578          # support points
579  
580          m = self._support_points.size
581  
582          # Cauchy matrix
583          C = 1 / np.subtract.outer(z, self._support_points)
584          # Loewner matrix
585          A = f[:, np.newaxis] * C - C * self._support_values
586  
587          # Solve least-squares problem to obtain weights
588          _, _, V = scipy.linalg.svd(A, check_finite=False)
589          self.weights = np.conj(V[m - 1,:])
590  
591          # reset roots, poles, residues as cached values will be wrong with new weights
592          self._poles = None
593          self._residues = None
594          self._roots = None
595  
596          return ni
597

599  class FloaterHormannInterpolator(_BarycentricRational):
600      r""" 
601      Floater-Hormann barycentric rational interpolation. 
602   
603      As described in [1]_, the method of Floater and Hormann computes weights for a 
604      Barycentric rational interpolant with no poles on the real axis. 
605   
606      Parameters 
607      ---------- 
608      x : 1D array_like, shape (n,) 
609          1-D array containing values of the independent variable. Values may be real or 
610          complex but must be finite. 
611      y : array_like, shape (n, ...) 
612          Array containing values of the dependent variable. Infinite and NaN values 
613          of `values` and corresponding values of `x` will be discarded. 
614      d : int, optional 
615          Blends ``n - d`` degree `d` polynomials together. For ``d = n - 1`` it is 
616          equivalent to polynomial interpolation. Must satisfy ``0 <= d < n``, 
617          defaults to 3. 
618   
619      Attributes 
620      ---------- 
621      weights : array 
622          Weights of the barycentric approximation. 
623   
624      See Also 
625      -------- 
626      AAA : Barycentric rational approximation of real and complex functions. 
627      pade : Pad√© approximation. 
628   
629      Notes 
630      ----- 
631      The Floater-Hormann interpolant is a rational function that interpolates the data 
632      with approximation order :math:`O(h^{d+1})`. The rational function blends ``n - d`` 
633      polynomials of degree `d` together to produce a rational interpolant that contains 
634      no poles on the real axis, unlike `AAA`. The interpolant is given 
635      by 
636   
637      .. math:: 
638   
639          r(x) = \frac{\sum_{i=0}^{n-d} \lambda_i(x) p_i(x)} 
640          {\sum_{i=0}^{n-d} \lambda_i(x)}, 
641   
642      where :math:`p_i(x)` is an interpolating polynomials of at most degree `d` through 
643      the points :math:`(x_i,y_i),\dots,(x_{i+d},y_{i+d}), and :math:`\lambda_i(z)` are 
644      blending functions defined by 
645   
646      .. math:: 
647   
648          \lambda_i(x) = \frac{(-1)^i}{(x - x_i)\cdots(x - x_{i+d})}. 
649   
650      When ``d = n - 1`` this reduces to polynomial interpolation. 
651   
652      Due to its stability following barycentric representation of the above equation 
653      is used instead for computation 
654   
655      .. math:: 
656   
657          r(z) = \frac{\sum_{k=1}^m\ w_k f_k / (x - x_k)}{\sum_{k=1}^m w_k / (x - x_k)}, 
658   
659      where the weights :math:`w_j` are computed as 
660   
661      .. math:: 
662   
663          w_k &= (-1)^{k - d} \sum_{i \in J_k} \prod_{j = i, j \neq k}^{i + d} 
664          1/|x_k - x_j|, \\ 
665          J_k &= \{ i \in I: k - d \leq i \leq k\},\\ 
666          I &= \{0, 1, \dots, n - d\}. 
667   
668      References 
669      ---------- 
670      .. [1] M.S. Floater and K. Hormann, "Barycentric rational interpolation with no 
671             poles and high rates of approximation", Numer. Math. 107, 315 (2007). 
672             :doi:`10.1007/s00211-007-0093-y` 
673   
674      Examples 
675      -------- 
676   
677      Here we compare the method against polynomial interpolation for an example where 
678      the polynomial interpolation fails due to Runge's phenomenon. 
679   
680      >>> import numpy as np 
681      >>> from scipy.interpolate import (FloaterHormannInterpolator, 
682      ...                                BarycentricInterpolator) 
683      >>> def f(z): 
684      ...     return 1/(1 + z**2) 
685      >>> z = np.linspace(-5, 5, num=15) 
686      >>> r = FloaterHormannInterpolator(z, f(z)) 
687      >>> p = BarycentricInterpolator(z, f(z)) 
688      >>> zz = np.linspace(-5, 5, num=1000) 
689      >>> import matplotlib.pyplot as plt 
690      >>> fig, ax = plt.subplots() 
691      >>> ax.plot(zz, r(zz), label="Floater=Hormann") 
692      >>> ax.plot(zz, p(zz), label="Polynomial") 
693      >>> ax.legend() 
694      >>> plt.show() 
695      """
696      def __init__(self, points, values, *, d=3):
697          super().__init__(points, values, d=d)
698  
699      def _input_validation(self, x, y, d):
700          d = operator.index(d)
701          if not (0 <= d < len(x)):
702              raise ValueError("`d` must satisfy 0 <= d < n")
703  
704          super()._input_validation(x, y)
705  
706      def _compute_weights(self, z, f, d):
707          # Floater and Hormann 2007 Eqn. (18) 3 equations later
708          w = np.zeros_like(z, dtype=np.result_type(z, 1.0))
709          n = w.size
710          for k in range(n):
711              for i in range(max(k-d, 0), min(k+1, n-d)):
712                  w[k] += 1/np.prod(np.abs(np.delete(z[k] - z[i : i + d + 1], k - i)))
713          w *= (-1.)**(np.arange(n) - d)
714  
715          return z, f, w
