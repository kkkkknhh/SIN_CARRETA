17   def test_implicit_multiplication():
18       cases = {
19           '5x': '5*x',
20           'abc': 'a*b*c',
21           '3sin(x)': '3*sin(x)',
22           '(x+1)(x+2)': '(x+1)*(x+2)',
23           '(5 x**2)sin(x)': '(5*x**2)*sin(x)',
24           '2 sin(x) cos(x)': '2*sin(x)*cos(x)',
25           'pi x': 'pi*x',
26           'x pi': 'x*pi',
27           'E x': 'E*x',
28           'EulerGamma y': 'EulerGamma*y',
29           'E pi': 'E*pi',
30           'pi (x + 2)': 'pi*(x+2)',
31           '(x + 2) pi': '(x+2)*pi',
32           'pi sin(x)': 'pi*sin(x)',
33       }
34       transformations = standard_transformations + (convert_xor,)
35       transformations2 = transformations + (split_symbols,
36                                             implicit_multiplication)
37       for case in cases:
38           implicit = parse_expr(case, transformations=transformations2)
39           normal = parse_expr(cases[case], transformations=transformations)
40           assert(implicit == normal)
41   
42       application = ['sin x', 'cos 2*x', 'sin cos x']
43       for case in application:
44           raises(SyntaxError,
45                  lambda: parse_expr(case, transformations=transformations2))
46       raises(TypeError,
47              lambda: parse_expr('sin**2(x)', transformations=transformations2))

50   def test_implicit_application():
51       cases = {
52           'factorial': 'factorial',
53           'sin x': 'sin(x)',
54           'tan y**3': 'tan(y**3)',
55           'cos 2*x': 'cos(2*x)',
56           '(cot)': 'cot',
57           'sin cos tan x': 'sin(cos(tan(x)))'
58       }
59       transformations = standard_transformations + (convert_xor,)
60       transformations2 = transformations + (implicit_application,)
61       for case in cases:
62           implicit = parse_expr(case, transformations=transformations2)
63           normal = parse_expr(cases[case], transformations=transformations)
64           assert(implicit == normal), (implicit, normal)
65   
66       multiplication = ['x y', 'x sin x', '2x']
67       for case in multiplication:
68           raises(SyntaxError,
69                  lambda: parse_expr(case, transformations=transformations2))
70       raises(TypeError,
71              lambda: parse_expr('sin**2(x)', transformations=transformations2))

75       cases = {

85           assert(implicit == normal)

89       for case in other_implicit:

91                  lambda: parse_expr(case, transformations=transformations2))
92   
93       assert parse_expr('x**2', local_dict={ 'x': sympy.Symbol('x') },
94                         transformations=transformations2) == parse_expr('x**2')

97   def test_symbol_splitting():
98       # By default Greek letter names should not be split (lambda is a keyword
99       # so skip it)
100      transformations = standard_transformations + (split_symbols,)
101      greek_letters = ('alpha', 'beta', 'gamma', 'delta', 'epsilon', 'zeta',
102                       'eta', 'theta', 'iota', 'kappa', 'mu', 'nu', 'xi',
103                       'omicron', 'pi', 'rho', 'sigma', 'tau', 'upsilon',
104                       'phi', 'chi', 'psi', 'omega')
105  
106      for letter in greek_letters:
107          assert(parse_expr(letter, transformations=transformations) ==
108                 parse_expr(letter))
109  
110      # Make sure symbol splitting resolves names
111      transformations += (implicit_multiplication,)
112      local_dict = { 'e': sympy.E }
113      cases = {
114          'xe': 'E*x',
115          'Iy': 'I*y',
116          'ee': 'E*E',
117      }
118      for case, expected in cases.items():
119          assert(parse_expr(case, local_dict=local_dict,
120                            transformations=transformations) ==
121                 parse_expr(expected))
122  
123      # Make sure custom splitting works
124      def can_split(symbol):
125          if symbol not in ('unsplittable', 'names'):
126              return _token_splittable(symbol)
127          return False
128      transformations = standard_transformations
129      transformations += (split_symbols_custom(can_split),
130                          implicit_multiplication)
131  
132      assert(parse_expr('unsplittable', transformations=transformations) ==
133             parse_expr('unsplittable'))
134      assert(parse_expr('names', transformations=transformations) ==
135             parse_expr('names'))
136      assert(parse_expr('xy', transformations=transformations) ==
137             parse_expr('x*y'))
138      for letter in greek_letters:
139          assert(parse_expr(letter, transformations=transformations) ==
140                 parse_expr(letter))

143  def test_all_implicit_steps():
144      cases = {
145          '2x': '2*x',  # implicit multiplication
146          'x y': 'x*y',
147          'xy': 'x*y',
148          'sin x': 'sin(x)',  # add parentheses
149          '2sin x': '2*sin(x)',
150          'x y z': 'x*y*z',
151          'sin(2 * 3x)': 'sin(2 * 3 * x)',
152          'sin(x) (1 + cos(x))': 'sin(x) * (1 + cos(x))',
153          '(x + 2) sin(x)': '(x + 2) * sin(x)',
154          '(x + 2) sin x': '(x + 2) * sin(x)',
155          'sin(sin x)': 'sin(sin(x))',
156          'sin x!': 'sin(factorial(x))',
157          'sin x!!': 'sin(factorial2(x))',
158          'factorial': 'factorial',  # don't apply a bare function
159          'x sin x': 'x * sin(x)',  # both application and multiplication
160          'xy sin x': 'x * y * sin(x)',
161          '(x+2)(x+3)': '(x + 2) * (x+3)',
162          'x**2 + 2xy + y**2': 'x**2 + 2 * x * y + y**2',  # split the xy
163          'pi': 'pi',  # don't mess with constants
164          'None': 'None',
165          'ln sin x': 'ln(sin(x))',  # multiple implicit function applications
166          'sin x**2': 'sin(x**2)',  # implicit application to an exponential
167          'alpha': 'Symbol("alpha")',  # don't split Greek letters/subscripts
168          'x_2': 'Symbol("x_2")',
169          'sin^2 x**2': 'sin(x**2)**2',  # function raised to a power
170          'sin**3(x)': 'sin(x)**3',
171          '(factorial)': 'factorial',
172          'tan 3x': 'tan(3*x)',
173          'sin^2(3*E^(x))': 'sin(3*E**(x))**2',
174          'sin**2(E^(3x))': 'sin(E**(3*x))**2',
175          'sin^2 (3x*E^(x))': 'sin(3*x*E^x)**2',
176          'pi sin x': 'pi*sin(x)',
177      }
178      transformations = standard_transformations + (convert_xor,)
179      transformations2 = transformations + (implicit_multiplication_application,)
180      for case in cases:
181          implicit = parse_expr(case, transformations=transformations2)
182          normal = parse_expr(cases[case], transformations=transformations)
183          assert(implicit == normal)

186  def test_no_methods_implicit_multiplication():
187      # Issue 21020
188      u = sympy.Symbol('u')
189      transformations = standard_transformations + \
190                        (implicit_multiplication,)
191      expr = parse_expr('x.is_polynomial(x)', transformations=transformations)
192      assert expr == True
193      expr = parse_expr('(exp(x) / (1 + exp(2x))).subs(exp(x), u)',
194                        transformations=transformations)
195      assert expr == u/(u**2 + 1)
