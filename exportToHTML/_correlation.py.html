14   def _xi_statistic(x, y, y_continuous):
15       # Compute xi correlation statistic
16   
17       # `axis=-1` is guaranteed by _axis_nan_policy decorator
18       n = x.shape[-1]
19   
20       # "Rearrange the data as (X(1), Y(1)), . . . ,(X(n), Y(n))
21       # such that X(1) ≤ ··· ≤ X(n)"
22       j = np.argsort(x, axis=-1)
23       j, y = np.broadcast_arrays(j, y)
24       y = np.take_along_axis(y, j, axis=-1)
25   
26       # "Let ri be the rank of Y(i), that is, the number of j such that Y(j) ≤ Y(i)"
27       r = stats.rankdata(y, method='max', axis=-1)
28       # " additionally define li to be the number of j such that Y(j) ≥ Y(i)"
29       # Could probably compute this from r, but that can be an enhancement
30       l = stats.rankdata(-y, method='max', axis=-1)
31   
32       num = np.sum(np.abs(np.diff(r, axis=-1)), axis=-1)
33       if y_continuous:  # [1] Eq. 1.1
34           statistic = 1 - 3 * num / (n ** 2 - 1)
35       else:  # [1] Eq. 1.2
36           den = 2 * np.sum((n - l) * l, axis=-1)
37           statistic = 1 - n * num / den
38   
39       return statistic, r, l

42   def _xi_std(r, l, y_continuous):
43       # Compute asymptotic standard deviation of xi under null hypothesis of independence
44   
45       # `axis=-1` is guaranteed by _axis_nan_policy decorator
46       n = np.float64(r.shape[-1])
47   
48       # "Suppose that X and Y are independent and Y is continuous. Then
49       # √n·ξn(X, Y) → N(0, 2/5) in distribution as n → ∞"
50       if y_continuous:  # [1] Theorem 2.1
51           return np.sqrt(2 / 5) / np.sqrt(n)
52   
53       # "Suppose that X and Y are independent. Then √n·ξn(X, Y)
54       # converges to N(0, τ²) in distribution as n → ∞
55       # [1] Eq. 2.2 and surrounding math
56       i = np.arange(1, n + 1)
57       u = np.sort(r, axis=-1)
58       v = np.cumsum(u, axis=-1)
59       an = 1 / n**4 * np.sum((2*n - 2*i + 1) * u**2, axis=-1)
60       bn = 1 / n**5 * np.sum((v + (n - i)*u)**2, axis=-1)
61       cn = 1 / n**3 * np.sum((2*n - 2*i + 1) * u, axis=-1)
62       dn = 1 / n**3 * np.sum((l * (n - l)), axis=-1)
63       tau2 = (an - 2*bn + cn**2) / dn**2
64   
65       return np.sqrt(tau2) / np.sqrt(n)

68   def _chatterjeexi_iv(y_continuous, method):
69       # Input validation for `chatterjeexi`
70       # x, y, `axis` input validation taken care of by decorator
71   
72       if y_continuous not in {True, False}:
73           raise ValueError('`y_continuous` must be boolean.')
74   
75       if not isinstance(method, stats.PermutationMethod):
76           method = method.lower()
77           message = "`method` must be 'asymptotic' or a `PermutationMethod` instance."
78           if method != 'asymptotic':
79               raise ValueError(message)
80   
81       return y_continuous, method

88   @_axis_nan_policy_factory(SignificanceResult, paired=True, n_samples=2,
89                             result_to_tuple=_unpack, n_outputs=2, too_small=1)
90   def chatterjeexi(x, y, *, axis=0, y_continuous=False, method='asymptotic'):
91       r"""Compute the xi correlation and perform a test of independence 
92    
93       The xi correlation coefficient is a measure of association between two 
94       variables; the value tends to be close to zero when the variables are 
95       independent and close to 1 when there is a strong association. Unlike 
96       other correlation coefficients, the xi correlation is effective even 
97       when the association is not monotonic. 
98    
99       Parameters 
100      ---------- 
101      x, y : array-like 
102          The samples: corresponding observations of the independent and 
103          dependent variable. The (N-d) arrays must be broadcastable. 
104      axis : int, default: 0 
105          Axis along which to perform the test. 
106      method : 'asymptotic' or `PermutationMethod` instance, optional 
107          Selects the method used to calculate the *p*-value. 
108          Default is 'asymptotic'. The following options are available. 
109   
110          * ``'asymptotic'``: compares the standardized test statistic 
111            against the normal distribution. 
112          * `PermutationMethod` instance. In this case, the p-value 
113            is computed using `permutation_test` with the provided 
114            configuration options and other appropriate settings. 
115   
116      y_continuous : bool, default: False 
117          Whether `y` is assumed to be drawn from a continuous distribution. 
118          If `y` is drawn from a continuous distribution, results are valid 
119          whether this is assumed or not, but enabling this assumption will 
120          result in faster computation and typically produce similar results. 
121   
122      Returns 
123      ------- 
124      res : SignificanceResult 
125          An object containing attributes: 
126   
127          statistic : float 
128              The xi correlation statistic. 
129          pvalue : float 
130              The associated *p*-value: the probability of a statistic at least as 
131              high as the observed value under the null hypothesis of independence. 
132   
133      See Also 
134      -------- 
135      scipy.stats.pearsonr, scipy.stats.spearmanr, scipy.stats.kendalltau 
136   
137      Notes 
138      ----- 
139      There is currently no special handling of ties in `x`; they are broken arbitrarily 
140      by the implementation. 
141   
142      [1]_ notes that the statistic is not symmetric in `x` and `y` *by design*: 
143      "...we may want to understand if :math:`Y` is a function :math:`X`, and not just 
144      if one of the variables is a function of the other." See [1]_ Remark 1. 
145   
146      References 
147      ---------- 
148      .. [1] Chatterjee, Sourav. "A new coefficient of correlation." Journal of 
149             the American Statistical Association 116.536 (2021): 2009-2022. 
150             :doi:`10.1080/01621459.2020.1758115`. 
151   
152      Examples 
153      -------- 
154      Generate perfectly correlated data, and observe that the xi correlation is 
155      nearly 1.0. 
156   
157      >>> import numpy as np 
158      >>> from scipy import stats 
159      >>> rng = np.random.default_rng(348932549825235) 
160      >>> x = rng.uniform(0, 10, size=100) 
161      >>> y = np.sin(x) 
162      >>> res = stats.chatterjeexi(x, y) 
163      >>> res.statistic 
164      np.float64(0.9012901290129013) 
165   
166      The probability of observing such a high value of the statistic under the 
167      null hypothesis of independence is very low. 
168   
169      >>> res.pvalue 
170      np.float64(2.2206974648177804e-46) 
171   
172      As noise is introduced, the correlation coefficient decreases. 
173   
174      >>> noise = rng.normal(scale=[[0.1], [0.5], [1]], size=(3, 100)) 
175      >>> res = stats.chatterjeexi(x, y + noise, axis=-1) 
176      >>> res.statistic 
177      array([0.79507951, 0.41824182, 0.16651665]) 
178   
179      Because the distribution of `y` is continuous, it is valid to pass 
180      ``y_continuous=True``. The statistic is identical, and the p-value 
181      (not shown) is only slightly different. 
182   
183      >>> stats.chatterjeexi(x, y + noise, y_continuous=True, axis=-1).statistic 
184      array([0.79507951, 0.41824182, 0.16651665]) 
185   
186      """
187      # x, y, `axis` input validation taken care of by decorator
188      # In fact, `axis` is guaranteed to be -1
189      y_continuous, method = _chatterjeexi_iv(y_continuous, method)
190  
191      # A highly negative statistic is possible, e.g.
192      # x = np.arange(100.), y = (x % 2 == 0)
193      # Unclear whether we should expose `alternative`, though.
194      alternative = 'greater'
195  
196      if method == 'asymptotic':
197          xi, r, l = _xi_statistic(x, y, y_continuous)
198          std = _xi_std(r, l, y_continuous)
199          norm = _SimpleNormal()
200          pvalue = _get_pvalue(xi / std, norm, alternative=alternative)
201      elif isinstance(method, stats.PermutationMethod):
202          res = stats.permutation_test(
203              # Could be faster if we just permuted the ranks; for now, keep it simple.
204              data=(y,), statistic=lambda y, axis: _xi_statistic(x, y, y_continuous)[0],
205              alternative=alternative, permutation_type='pairings', **method._asdict(),
206              axis=-1)  # `axis=-1` is guaranteed by _axis_nan_policy decorator
207  
208          xi, pvalue = res.statistic, res.pvalue
209  
210      return SignificanceResult(xi, pvalue)
