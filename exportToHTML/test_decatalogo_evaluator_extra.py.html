74       def test_evaluator_has_detector_and_uses_it(self):
75           self.assertTrue(hasattr(self.evaluator, "contradiction_detector"))
76           cd = self.evaluator.contradiction_detector
77           self.assertIsNotNone(cd)
78           # Run a quick detection
79           analysis = cd.detect_contradictions(
80               "La meta es 95%, sin embargo no hay presupuesto"
81           )
82           self.assertIsInstance(analysis, self.module.ContradictionAnalysis)

84       def test_evaluar_punto_completo_returns_expected_tuple(self):
85           evidencia = self._sample_evidence()
86           result = self.evaluator.evaluar_punto_completo(evidencia, punto_id=1)
87           # Expect a tuple of (EvaluacionPuntoCompleto, AnalisisEvidenciaDecalogo, ResultadoDimensionIndustrial|None)
88           self.assertIsInstance(result, tuple)
89           self.assertGreaterEqual(len(result), 2)
90           evaluacion_punto, analisis = result[0], result[1]
91           self.assertTrue(hasattr(evaluacion_punto, "puntaje_agregado_punto"))
92           self.assertTrue(hasattr(analisis, "contradicciones"))
