5    /*                                                     jv.c 
6     * 
7     *     Bessel function of noninteger order 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double v, x, y, jv(); 
14    * 
15    * y = jv( v, x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Returns Bessel function of order v of the argument, 
22    * where v is real.  Negative x is allowed if v is an integer. 
23    * 
24    * Several expansions are included: the ascending power 
25    * series, the Hankel expansion, and two transitional 
26    * expansions for large v.  If v is not too large, it 
27    * is reduced by recurrence to a region of best accuracy. 
28    * The transitional expansions give 12D accuracy for v > 500. 
29    * 
30    * 
31    * 
32    * ACCURACY: 
33    * Results for integer v are indicated by *, where x and v 
34    * both vary from -125 to +125.  Otherwise, 
35    * x ranges from 0 to 125, v ranges as indicated by "domain." 
36    * Error criterion is absolute, except relative when |jv()| > 1. 
37    * 
38    * arithmetic  v domain  x domain    # trials      peak       rms 
39    *    IEEE      0,125     0,125      100000      4.6e-15    2.2e-16 
40    *    IEEE   -125,0       0,125       40000      5.4e-11    3.7e-13 
41    *    IEEE      0,500     0,500       20000      4.4e-15    4.0e-16 
42    * Integer v: 
43    *    IEEE   -125,125   -125,125      50000      3.5e-15*   1.9e-16* 
44    * 
45    */

68           constexpr double jv_BIG = 1.44115188075855872E+17;

70           /* Reduce the order by backward recurrence. 
71            * AMS55 #9.1.27 and 9.1.73. 
72            */
73   
74           XSF_HOST_DEVICE inline double jv_recur(double *n, double x, double *newn, int cancel) {
75               double pkm2, pkm1, pk, qkm2, qkm1;
76   
77               /* double pkp1; */
78               double k, ans, qk, xk, yk, r, t, kf;
79               constexpr double big = jv_BIG;
80               int nflag, ctr;
81               int miniter, maxiter;
82   
83               /* Continued fraction for Jn(x)/Jn-1(x) 
84                * AMS 9.1.73 
85                * 
86                *    x       -x^2      -x^2 
87                * ------  ---------  ---------   ... 
88                * 2 n +   2(n+1) +   2(n+2) + 
89                * 
90                * Compute it with the simplest possible algorithm. 
91                * 
92                * This continued fraction starts to converge when (|n| + m) > |x|. 
93                * Hence, at least |x|-|n| iterations are necessary before convergence is 
94                * achieved. There is a hard limit set below, m <= 30000, which is chosen 
95                * so that no branch in `jv` requires more iterations to converge. 
96                * The exact maximum number is (500/3.6)^2 - 500 ~ 19000 
97                */
98   
99               maxiter = 22000;
100              miniter = std::abs(x) - std::abs(*n);
101              if (miniter < 1) {
102                  miniter = 1;
103              }
104  
105              if (*n < 0.0) {
106                  nflag = 1;
107              } else {
108                  nflag = 0;
109              }
110  
111          fstart:
112              pkm2 = 0.0;
113              qkm2 = 1.0;
114              pkm1 = x;
115              qkm1 = *n + *n;
116              xk = -x * x;
117              yk = qkm1;
118              ans = 0.0; /* ans=0.0 ensures that t=1.0 in the first iteration */
119              ctr = 0;
120              do {
121                  yk += 2.0;
122                  pk = pkm1 * yk + pkm2 * xk;
123                  qk = qkm1 * yk + qkm2 * xk;
124                  pkm2 = pkm1;
125                  pkm1 = pk;
126                  qkm2 = qkm1;
127                  qkm1 = qk;
128  
129                  /* check convergence */
130                  if (qk != 0 && ctr > miniter)
131                      r = pk / qk;
132                  else
133                      r = 0.0;
134  
135                  if (r != 0) {
136                      t = std::abs((ans - r) / r);
137                      ans = r;
138                  } else {
139                      t = 1.0;
140                  }
141  
142                  if (++ctr > maxiter) {
143                      set_error("jv", SF_ERROR_UNDERFLOW, NULL);
144                      goto done;
145                  }
146                  if (t < MACHEP) {
147                      goto done;
148                  }
149  
150                  /* renormalize coefficients */
151                  if (std::abs(pk) > big) {
152                      pkm2 /= big;
153                      pkm1 /= big;
154                      qkm2 /= big;
155                      qkm1 /= big;
156                  }
157              } while (t > MACHEP);
158  
159          done:
160              if (ans == 0)
161                  ans = 1.0;
162  
163              /* Change n to n-1 if n < 0 and the continued fraction is small */
164              if (nflag > 0) {
165                  if (std::abs(ans) < 0.125) {
166                      nflag = -1;
167                      *n = *n - 1.0;
168                      goto fstart;
169                  }
170              }
171  
172              kf = *newn;
173  
174              /* backward recurrence 
175               *              2k 
176               *  J   (x)  =  --- J (x)  -  J   (x) 
177               *   k-1         x   k         k+1 
178               */
179  
180              pk = 1.0;
181              pkm1 = 1.0 / ans;
182              k = *n - 1.0;
183              r = 2 * k;
184              do {
185                  pkm2 = (pkm1 * r - pk * x) / x;
186                  /*      pkp1 = pk; */
187                  pk = pkm1;
188                  pkm1 = pkm2;
189                  r -= 2.0;
190                  /* 
191                   * t = fabs(pkp1) + fabs(pk); 
192                   * if( (k > (kf + 2.5)) && (fabs(pkm1) < 0.25*t) ) 
193                   * { 
194                   * k -= 1.0; 
195                   * t = x*x; 
196                   * pkm2 = ( (r*(r+2.0)-t)*pk - r*x*pkp1 )/t; 
197                   * pkp1 = pk; 
198                   * pk = pkm1; 
199                   * pkm1 = pkm2; 
200                   * r -= 2.0; 
201                   * } 
202                   */
203                  k -= 1.0;
204              } while (k > (kf + 0.5));
205  
206              /* Take the larger of the last two iterates 
207               * on the theory that it may have less cancellation error. 
208               */
209  
210              if (cancel) {
211                  if ((kf >= 0.0) && (std::abs(pk) > std::abs(pkm1))) {
212                      k += 1.0;
213                      pkm2 = pk;
214                  }
215              }
216              *newn = k;
217              return (pkm2);
218          }

220          /* Ascending power series for Jv(x). 
221           * AMS55 #9.1.10. 
222           */
223  
224          XSF_HOST_DEVICE inline double jv_jvs(double n, double x) {
225              double t, u, y, z, k;
226              int ex, sgngam;
227  
228              z = -x * x / 4.0;
229              u = 1.0;
230              y = u;
231              k = 1.0;
232              t = 1.0;
233  
234              while (t > MACHEP) {
235                  u *= z / (k * (n + k));
236                  y += u;
237                  k += 1.0;
238                  if (y != 0)
239                      t = std::abs(u / y);
240              }
241              t = std::frexp(0.5 * x, &ex);
242              ex = ex * n;
243              if ((ex > -1023) && (ex < 1023) && (n > 0.0) && (n < (MAXGAM - 1.0))) {
244                  t = std::pow(0.5 * x, n) * xsf::cephes::rgamma(n + 1.0);
245                  y *= t;
246              } else {
247                  t = n * std::log(0.5 * x) - lgam_sgn(n + 1.0, &sgngam);
248                  if (y < 0) {
249                      sgngam = -sgngam;
250                      y = -y;
251                  }
252                  t += std::log(y);
253                  if (t < -MAXLOG) {
254                      return (0.0);
255                  }
256                  if (t > MAXLOG) {
257                      set_error("Jv", SF_ERROR_OVERFLOW, NULL);
258                      return (std::numeric_limits<double>::infinity());
259                  }
260                  y = sgngam * std::exp(t);
261              }
262              return (y);
263          }

265          /* Hankel's asymptotic expansion 
266           * for large x. 
267           * AMS55 #9.2.5. 
268           */

270          XSF_HOST_DEVICE inline double jv_hankel(double n, double x) {
271              double t, u, z, k, sign, conv;
272              double p, q, j, m, pp, qq;
273              int flag;
274  
275              m = 4.0 * n * n;
276              j = 1.0;
277              z = 8.0 * x;
278              k = 1.0;
279              p = 1.0;
280              u = (m - 1.0) / z;
281              q = u;
282              sign = 1.0;
283              conv = 1.0;
284              flag = 0;
285              t = 1.0;
286              pp = 1.0e38;
287              qq = 1.0e38;
288  
289              while (t > MACHEP) {
290                  k += 2.0;
291                  j += 1.0;
292                  sign = -sign;
293                  u *= (m - k * k) / (j * z);
294                  p += sign * u;
295                  k += 2.0;
296                  j += 1.0;
297                  u *= (m - k * k) / (j * z);
298                  q += sign * u;
299                  t = std::abs(u / p);
300                  if (t < conv) {
301                      conv = t;
302                      qq = q;
303                      pp = p;
304                      flag = 1;
305                  }
306                  /* stop if the terms start getting larger */
307                  if ((flag != 0) && (t > conv)) {
308                      goto hank1;
309                  }
310              }
311  
312          hank1:
313              u = x - (0.5 * n + 0.25) * M_PI;
314              t = std::sqrt(2.0 / (M_PI * x)) * (pp * std::cos(u) - qq * std::sin(u));
315              return (t);
316          }

336          XSF_HOST_DEVICE inline double jv_jnt(double n, double x) {
337              double z, zz, z3;
338              double cbn, n23, cbtwo;
339              double ai, aip, bi, bip; /* Airy functions */
340              double nk, fk, gk, pp, qq;
341              double F[5], G[4];
342              int k;
343  
344              cbn = cbrt(n);
345              z = (x - n) / cbn;
346              cbtwo = cbrt(2.0);
347  
348              /* Airy function */
349              zz = -cbtwo * z;
350              xsf::cephes::airy(zz, &ai, &aip, &bi, &bip);
351  
352              /* polynomials in expansion */
353              zz = z * z;
354              z3 = zz * z;
355              F[0] = 1.0;
356              F[1] = -z / 5.0;
357              F[2] = xsf::cephes::polevl(z3, jv_PF2, 1) * zz;
358              F[3] = xsf::cephes::polevl(z3, jv_PF3, 2);
359              F[4] = xsf::cephes::polevl(z3, jv_PF4, 3) * z;
360              G[0] = 0.3 * zz;
361              G[1] = xsf::cephes::polevl(z3, jv_PG1, 1);
362              G[2] = xsf::cephes::polevl(z3, jv_PG2, 2) * z;
363              G[3] = xsf::cephes::polevl(z3, jv_PG3, 2) * zz;
364  
365              pp = 0.0;
366              qq = 0.0;
367              nk = 1.0;
368              n23 = cbrt(n * n);
369  
370              for (k = 0; k <= 4; k++) {
371                  fk = F[k] * nk;
372                  pp += fk;
373                  if (k != 4) {
374                      gk = G[k] * nk;
375                      qq += gk;
376                  }
377                  nk /= n23;
378              }
379  
380              fk = cbtwo * ai * pp / cbn + cbrt(4.0) * aip * qq / n;
381              return (fk);
382          }

435          XSF_HOST_DEVICE inline double jv_jnx(double n, double x) {
436              double zeta, sqz, zz, zp, np;
437              double cbn, n23, t, z, sz;
438              double pp, qq, z32i, zzi;
439              double ak, bk, akl, bkl;
440              int sign, doa, dob, nflg, k, s, tk, tkp1, m;
441              double u[8];
442              double ai, aip, bi, bip;
443  
444              /* Test for x very close to n. Use expansion for transition region if so. */
445              cbn = cbrt(n);
446              z = (x - n) / cbn;
447              if (std::abs(z) <= 0.7) {
448                  return (jv_jnt(n, x));
449              }
450  
451              z = x / n;
452              zz = 1.0 - z * z;
453              if (zz == 0.0) {
454                  return (0.0);
455              }
456  
457              if (zz > 0.0) {
458                  sz = std::sqrt(zz);
459                  t = 1.5 * (std::log((1.0 + sz) / z) - sz); /* zeta ** 3/2          */
460                  zeta = cbrt(t * t);
461                  nflg = 1;
462              } else {
463                  sz = std::sqrt(-zz);
464                  t = 1.5 * (sz - std::acos(1.0 / z));
465                  zeta = -cbrt(t * t);
466                  nflg = -1;
467              }
468              z32i = std::abs(1.0 / t);
469              sqz = cbrt(t);
470  
471              /* Airy function */
472              n23 = cbrt(n * n);
473              t = n23 * zeta;
474  
475              xsf::cephes::airy(t, &ai, &aip, &bi, &bip);
476  
477              /* polynomials in expansion */
478              u[0] = 1.0;
479              zzi = 1.0 / zz;
480              u[1] = xsf::cephes::polevl(zzi, jv_P1, 1) / sz;
481              u[2] = xsf::cephes::polevl(zzi, jv_P2, 2) / zz;
482              u[3] = xsf::cephes::polevl(zzi, jv_P3, 3) / (sz * zz);
483              pp = zz * zz;
484              u[4] = xsf::cephes::polevl(zzi, jv_P4, 4) / pp;
485              u[5] = xsf::cephes::polevl(zzi, jv_P5, 5) / (pp * sz);
486              pp *= zz;
487              u[6] = xsf::cephes::polevl(zzi, jv_P6, 6) / pp;
488              u[7] = xsf::cephes::polevl(zzi, jv_P7, 7) / (pp * sz);
489  
490              pp = 0.0;
491              qq = 0.0;
492              np = 1.0;
493              /* flags to stop when terms get larger */
494              doa = 1;
495              dob = 1;
496              akl = std::numeric_limits<double>::infinity();
497              bkl = std::numeric_limits<double>::infinity();
498  
499              for (k = 0; k <= 3; k++) {
500                  tk = 2 * k;
501                  tkp1 = tk + 1;
502                  zp = 1.0;
503                  ak = 0.0;
504                  bk = 0.0;
505                  for (s = 0; s <= tk; s++) {
506                      if (doa) {
507                          if ((s & 3) > 1)
508                              sign = nflg;
509                          else
510                              sign = 1;
511                          ak += sign * jv_mu[s] * zp * u[tk - s];
512                      }
513  
514                      if (dob) {
515                          m = tkp1 - s;
516                          if (((m + 1) & 3) > 1)
517                              sign = nflg;
518                          else
519                              sign = 1;
520                          bk += sign * jv_lambda[s] * zp * u[m];
521                      }
522                      zp *= z32i;
523                  }
524  
525                  if (doa) {
526                      ak *= np;
527                      t = std::abs(ak);
528                      if (t < akl) {
529                          akl = t;
530                          pp += ak;
531                      } else
532                          doa = 0;
533                  }
534  
535                  if (dob) {
536                      bk += jv_lambda[tkp1] * zp * u[0];
537                      bk *= -np / sqz;
538                      t = std::abs(bk);
539                      if (t < bkl) {
540                          bkl = t;
541                          qq += bk;
542                      } else
543                          dob = 0;
544                  }
545                  if (np < MACHEP)
546                      break;
547                  np /= n * n;
548              }
549  
550              /* normalizing factor ( 4*zeta/(1 - z**2) )**1/4    */
551              t = 4.0 * zeta / zz;
552              t = sqrt(sqrt(t));
553  
554              t *= ai * pp / cbrt(n) + aip * qq / (n23 * n);
555              return (t);
556          }

560      XSF_HOST_DEVICE inline double jv(double n, double x) {
561          double k, q, t, y, an;
562          int i, sign, nint;
563  
564          nint = 0; /* Flag for integer n */
565          sign = 1; /* Flag for sign inversion */
566          an = std::abs(n);
567          y = std::floor(an);
568          if (y == an) {
569              nint = 1;
570              i = an - 16384.0 * std::floor(an / 16384.0);
571              if (n < 0.0) {
572                  if (i & 1)
573                      sign = -sign;
574                  n = an;
575              }
576              if (x < 0.0) {
577                  if (i & 1)
578                      sign = -sign;
579                  x = -x;
580              }
581              if (n == 0.0)
582                  return (j0(x));
583              if (n == 1.0)
584                  return (sign * j1(x));
585          }
586  
587          if ((x < 0.0) && (y != an)) {
588              set_error("Jv", SF_ERROR_DOMAIN, NULL);
589              y = std::numeric_limits<double>::quiet_NaN();
590              goto done;
591          }
592  
593          if (x == 0 && n < 0 && !nint) {
594              set_error("Jv", SF_ERROR_OVERFLOW, NULL);
595              return std::numeric_limits<double>::infinity() * rgamma(n + 1);
596          }
597  
598          y = std::abs(x);
599  
600          if (y * y < std::abs(n + 1) * detail::MACHEP) {
601              return std::pow(0.5 * x, n) * rgamma(n + 1);
602          }
603  
604          k = 3.6 * std::sqrt(y);
605          t = 3.6 * std::sqrt(an);
606          if ((y < t) && (an > 21.0)) {
607              return (sign * detail::jv_jvs(n, x));
608          }
609          if ((an < k) && (y > 21.0))
610              return (sign * detail::jv_hankel(n, x));
611  
612          if (an < 500.0) {
613              /* Note: if x is too large, the continued fraction will fail; but then the 
614               * Hankel expansion can be used. */
615              if (nint != 0) {
616                  k = 0.0;
617                  q = detail::jv_recur(&n, x, &k, 1);
618                  if (k == 0.0) {
619                      y = j0(x) / q;
620                      goto done;
621                  }
622                  if (k == 1.0) {
623                      y = j1(x) / q;
624                      goto done;
625                  }
626              }
627  
628              if (an > 2.0 * y)
629                  goto rlarger;
630  
631              if ((n >= 0.0) && (n < 20.0) && (y > 6.0) && (y < 20.0)) {
632                  /* Recur backwards from a larger value of n */
633              rlarger:
634                  k = n;
635  
636                  y = y + an + 1.0;
637                  if (y < 30.0)
638                      y = 30.0;
639                  y = n + std::floor(y - n);
640                  q = detail::jv_recur(&y, x, &k, 0);
641                  y = detail::jv_jvs(y, x) * q;
642                  goto done;
643              }
644  
645              if (k <= 30.0) {
646                  k = 2.0;
647              } else if (k < 90.0) {
648                  k = (3 * k) / 4;
649              }
650              if (an > (k + 3.0)) {
651                  if (n < 0.0) {
652                      k = -k;
653                  }
654                  q = n - std::floor(n);
655                  k = std::floor(k) + q;
656                  if (n > 0.0) {
657                      q = detail::jv_recur(&n, x, &k, 1);
658                  } else {
659                      t = k;
660                      k = n;
661                      q = detail::jv_recur(&t, x, &k, 1);
662                      k = t;
663                  }
664                  if (q == 0.0) {
665                      y = 0.0;
666                      goto done;
667                  }
668              } else {
669                  k = n;
670                  q = 1.0;
671              }
672  
673              /* boundary between convergence of 
674               * power series and Hankel expansion 
675               */
676              y = std::abs(k);
677              if (y < 26.0)
678                  t = (0.0083 * y + 0.09) * y + 12.9;
679              else
680                  t = 0.9 * y;
681  
682              if (x > t)
683                  y = detail::jv_hankel(k, x);
684              else
685                  y = detail::jv_jvs(k, x);
686              if (n > 0.0)
687                  y /= q;
688              else
689                  y *= q;
690          }
691  
692          else {
693              /* For large n, use the uniform expansion or the transitional expansion. 
694               * But if x is of the order of n**2, these may blow up, whereas the 
695               * Hankel expansion will then work. 
696               */
697              if (n < 0.0) {
698                  set_error("jv", SF_ERROR_LOSS, NULL);
699                  y = std::numeric_limits<double>::quiet_NaN();
700                  goto done;
701              }
702              t = x / n;
703              t /= n;
704              if (t > 0.3)
705                  y = detail::jv_hankel(n, x);
706              else
707                  y = detail::jv_jnx(n, x);
708          }
709  
710      done:
711          return (sign * y);
712      }
