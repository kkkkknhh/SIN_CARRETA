109  # In most cases we will use the "as_name" field to store arguments which are
110  # SymInts.
111  # The "as_int" field is used in the case where we have a list containing a mix
112  # of SymInt and ints (ex. [1, s0, ...]). We will serialize this type of list to
113  # be List[SymIntArgument] and map the SymInts to the "as_name" field, and ints
114  # to the "as_int" field.
115  @dataclass(repr=False)
116  class SymIntArgument(_Union):
117      as_name: Annotated[str, 10]
118      as_int: Annotated[int, 20]
119

121  # In most cases we will use the "as_name" field to store arguments which are
122  # SymFloats.
123  # The "as_float" field is used in the case where we have a list containing a mix
124  # of SymFloat and float (ex. [1.0, s0, ...]). We will serialize this type of list to
125  # be List[SymFloatArgument] and map the SymFloats to the "as_name" field, and ints
126  # to the "as_float" field.
127  @dataclass(repr=False)
128  class SymFloatArgument(_Union):
