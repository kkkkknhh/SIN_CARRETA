1    ! Check that parameters are correct intercepted.
2    ! Constants with comma separations are commonly
3    ! used, for instance Pi = 3._dp
4    subroutine foo_compound_int(x)
5      implicit none
6      integer, parameter :: ii = selected_int_kind(9)
7      integer(ii), intent(inout) :: x
8      dimension x(3)
9      integer(ii), parameter :: three = 3_ii
10     integer(ii), parameter :: two = 2_ii
11     integer(ii), parameter :: six = three * 1_ii * two
12   
13     x(1) = x(1) + x(2) + x(3) * six
14     return
15   end subroutine
