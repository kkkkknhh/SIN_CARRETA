56     template <uint64_t mask>
57     static Vectorized<float> blend(
58         const Vectorized<float>& a,
59         const Vectorized<float>& b) {
60       // Build an array of flags: each element is 1 if the corresponding bit in
61       // 'mask' is set, 0 otherwise.
62       __at_align__ int32_t flag_arr[size()];
63       for (int i = 0; i < size(); i++) {
64         flag_arr[i] = (mask & (1ULL << i)) ? 1 : 0;
65       }
66       // Load the flag array into an SVE int32 vector.
67       svint32_t int_mask = svld1_s32(svptrue_b32(), flag_arr);
68       // Compare each lane of int_mask to 0; returns an svbool_t predicate where
69       // true indicates a nonzero flag.
70       svbool_t blend_mask = svcmpne_n_s32(svptrue_b32(), int_mask, 0);
71       // Use svsel to select elements from b where the predicate is true, else
72       // from a.
73       svfloat32_t result = svsel_f32(blend_mask, b.values, a.values);
74       return Vectorized<float>(result);
75     }
76     static Vectorized<float> blendv(

105    // Implementation is picked from
106    // https://github.com/ARM-software/ComputeLibrary/blob/v25.01/src/core/NEON/SVEMath.inl#L105
107    inline svfloat32_t svexp_f32_z(svbool_t pg, svfloat32_t x) const {
108      const auto c1 =
109          svreinterpret_f32_u32(svdup_n_u32(0x3f7ffff6)); // x^1: 0x1.ffffecp-1f
110      const auto c2 =
111          svreinterpret_f32_u32(svdup_n_u32(0x3efffedb)); // x^2: 0x1.fffdb6p-2f
112      const auto c3 =
113          svreinterpret_f32_u32(svdup_n_u32(0x3e2aaf33)); // x^3: 0x1.555e66p-3f
114      const auto c4 =
115          svreinterpret_f32_u32(svdup_n_u32(0x3d2b9f17)); // x^4: 0x1.573e2ep-5f
116      const auto c5 =
117          svreinterpret_f32_u32(svdup_n_u32(0x3c072010)); // x^5: 0x1.0e4020p-7f
118      const auto shift = svreinterpret_f32_u32(
119          svdup_n_u32(0x4b00007f)); // 2^23 + 127 = 0x1.0000fep23f
120      const auto inv_ln2 = svreinterpret_f32_u32(
121          svdup_n_u32(0x3fb8aa3b)); // 1 / ln(2) = 0x1.715476p+0f
122      const auto neg_ln2_hi = svreinterpret_f32_u32(svdup_n_u32(
123          0xbf317200)); // -ln(2) from bits  -1 to -19: -0x1.62e400p-1f
124      const auto neg_ln2_lo = svreinterpret_f32_u32(svdup_n_u32(
125          0xb5bfbe8e)); // -ln(2) from bits -20 to -42: -0x1.7f7d1cp-20f
126      const auto inf = svdup_n_f32(std::numeric_limits<float>::infinity());
127      const auto max_input = svdup_n_f32(88.37f); // Approximately ln(2^127.5)
128      const auto zero = svdup_n_f32(0.f);
129      const auto min_input = svdup_n_f32(-86.64f); // Approximately ln(2^-125)
130      // Range reduction:
131      //   e^x = 2^n * e^r
132      // where:
133      //   n = floor(x / ln(2))
134      //   r = x - n * ln(2)
135      //
136      // By adding x / ln(2) with 2^23 + 127 (shift):
137      //   * As FP32 fraction part only has 23-bits, the addition of 2^23 + 127
138      //   forces decimal part
139      //     of x / ln(2) out of the result. The integer part of x / ln(2) (i.e.
140      //     n) + 127 will occupy the whole fraction part of z in FP32 format.
141      //     Subtracting 2^23 + 127 (shift) from z will result in the integer part
142      //     of x / ln(2) (i.e. n) because the decimal part has been pushed out
143      //     and lost.
144      //   * The addition of 127 makes the FP32 fraction part of z ready to be
145      //   used as the exponent
146      //     in FP32 format. Left shifting z by 23 bits will result in 2^n.
147      const auto z = svmla_f32_z(pg, shift, x, inv_ln2);
148      const auto n = svsub_f32_z(pg, z, shift);
149      const auto scale = svreinterpret_f32_u32(
150          svlsl_n_u32_z(pg, svreinterpret_u32_f32(z), 23)); // 2^n
151      // The calculation of n * ln(2) is done using 2 steps to achieve accuracy
152      // beyond FP32. This outperforms longer Taylor series (3-4 tabs) both in
153      // term of accuracy and performance.
154      const auto r_hi = svmla_f32_z(pg, x, n, neg_ln2_hi);
155      const auto r = svmla_f32_z(pg, r_hi, n, neg_ln2_lo);
156      // Compute the truncated Taylor series of e^r.
157      //   poly = scale * (1 + c1 * r + c2 * r^2 + c3 * r^3 + c4 * r^4 + c5 * r^5)
158      const auto r2 = svmul_f32_z(pg, r, r);
159      const auto p1 = svmul_f32_z(pg, c1, r);
160      const auto p23 = svmla_f32_z(pg, c2, c3, r);
161      const auto p45 = svmla_f32_z(pg, c4, c5, r);
162      const auto p2345 = svmla_f32_z(pg, p23, p45, r2);
163      const auto p12345 = svmla_f32_z(pg, p1, p2345, r2);
164      auto poly = svmla_f32_z(pg, scale, p12345, scale);
165      // Handle underflow and overflow.
166      poly = svsel_f32(svcmplt_f32(pg, x, min_input), zero, poly);
167      poly = svsel_f32(svcmpgt_f32(pg, x, max_input), inf, poly);
168      return poly;
169    }
170    static Vectorized<float> loadu(const void* ptr, int64_t count = size()) {
171      if (count == size())
172        return svld1_f32(ptrue, reinterpret_cast<const float*>(ptr));
173      svbool_t pg = svwhilelt_b32(0ull, count);
174      return svld1_f32(pg, reinterpret_cast<const float*>(ptr));
175    }
176    void store(void* ptr, int64_t count = size()) const {
177      if (count == size()) {
178        svst1_f32(ptrue, reinterpret_cast<float*>(ptr), values);
179      } else {
180        svbool_t pg = svwhilelt_b32(0ull, count);
181        svst1_f32(pg, reinterpret_cast<float*>(ptr), values);
182      }
183    }

186    int64_t zero_mask() const {
187      // returns an integer mask where all zero elements are translated to 1-bit
188      // and others are translated to 0-bit
189      int64_t mask = 0;
190      __at_align__ int32_t mask_array[size()];
191  
192      svbool_t svbool_mask = svcmpeq_f32(ptrue, values, ZERO_F32);
193      svst1_s32(
194          ptrue,
195          mask_array,
196          svsel_s32(svbool_mask, ALL_S32_TRUE_MASK, ALL_S32_FALSE_MASK));
197      for (int64_t i = 0; i < size(); ++i) {
198        if (mask_array[i])
199          mask |= (1ull << i);
200      }
201      return mask;
202    }

429    // Implementation is picked from
430    // https://github.com/ARM-software/ComputeLibrary/blob/v25.01/src/core/NEON/SVEMath.inl#L179
431    Vectorized<float> tanh() const {
432      // Constants used for the tanh calculation.
433      const svfloat32_t CONST_1 =
434          svdup_n_f32(1.f); // Constant 1.0f for the tanh formula.
435      const svfloat32_t CONST_2 = svdup_n_f32(
436          2.f); // Constant 2.0f for the tanh formula (used in exp(2x)).
437      const svfloat32_t CONST_MIN_TANH = svdup_n_f32(
438          -10.f); // Minimum threshold for input values to prevent overflow.
439      const svfloat32_t CONST_MAX_TANH = svdup_n_f32(
440          10.f); // Maximum threshold for input values to prevent overflow.
441  
442      // Step 1: Clamp the values within the range [-10, 10] to prevent overflow
443      // during exponentiation. The tanh function approaches 1 rapidly as the
444      // input grows large, so we limit the input range to avoid numerical
445      // instability. svmax_f32_z ensures values are greater than -10, and
446      // svmin_f32_z ensures they are less than 10.
447      svfloat32_t x = svmin_f32_z(
448          ptrue, svmax_f32_z(ptrue, values, CONST_MIN_TANH), CONST_MAX_TANH);
449  
450      // Step 2: Calculate exp(2 * x), where x is the clamped value.
451      // svmul_f32_z computes 2 * x, and svexp_f32_z computes the exponential of
452      // the result.
453      svfloat32_t exp2x = svexp_f32_z(ptrue, svmul_f32_z(ptrue, CONST_2, x));
454  
455      // Step 3: Calculate the numerator of the tanh function, which is exp(2x)
456      // - 1.
457      svfloat32_t num = svsub_f32_z(ptrue, exp2x, CONST_1);
458  
459      // Step 4: Calculate the denominator of the tanh function, which is exp(2x)
460      // + 1.
461      svfloat32_t den = svadd_f32_z(ptrue, exp2x, CONST_1);
462  
463      // Step 5: Calculate the tanh function as the ratio of the numerator and
464      // denominator: num / den.
465      svfloat32_t tanh = svdiv_f32_z(ptrue, num, den);
466  
467      // Return the calculated tanh values.
468      return tanh;
469    }

666  inline void convert(const float* src, float* dst, int64_t n) {
667    const int64_t fraction = n % Vectorized<float>::size();
668  #pragma unroll
669    for (int64_t i = 0; i < n - fraction; i += Vectorized<float>::size()) {
670      svst1_f32(ptrue, dst + i, svldnt1_f32(ptrue, src + i));
671    }
672  #pragma unroll
673    for (int64_t i = n - fraction; i < n; i += Vectorized<float>::size()) {
674      svbool_t pg = svwhilelt_b32(i, n);
675      svst1_f32(pg, dst + i, svldnt1_f32(pg, src + i));
676    }
677  }

681    const int64_t fraction = n % Vectorized<float>::size();
682    svbool_t pg_16 = svwhilelt_b16(0ull, Vectorized<float>::size());
683    svbool_t pg_32 = svwhilelt_b32(0ull, Vectorized<float>::size());
684  #pragma unroll
685    for (int64_t i = 0; i < n - fraction; i += Vectorized<float>::size()) {
686      svfloat16_t src_vec = svuzp1_f16(
687          svcvt_f16_f32_x(ptrue, svldnt1_f32(pg_32, src + i)), ZERO_F16);
688      svst1_f16(pg_16, reinterpret_cast<float16_t*>(dst) + i, src_vec);
689    }
690  #pragma unroll
691    for (int64_t i = n - fraction; i < n; i += Vectorized<float>::size()) {
692      pg_16 = svwhilelt_b16(i, n);
693      pg_32 = svwhilelt_b32(i, n);
694      svfloat16_t src_vec = svuzp1_f16(
695          svcvt_f16_f32_x(ptrue, svldnt1_f32(pg_32, src + i)), ZERO_F16);
696      svst1_f16(pg_16, reinterpret_cast<float16_t*>(dst) + i, src_vec);
697    }

724  inline void convert(const bool* src, float* dst, int64_t n) {
725    const int64_t fraction = n % Vectorized<float>::size();
726    svbool_t pg_8 = svwhilelt_b8(0ull, Vectorized<float>::size());
727    svbool_t pg_32 = svwhilelt_b32(0ull, Vectorized<float>::size());
728  #pragma unroll
729    for (int64_t i = 0; i < n - fraction; i += Vectorized<float>::size()) {
730      svuint8_t src_vec_u8 =
731          svldnt1_u8(pg_8, reinterpret_cast<const uint8_t*>(src) + i);
732      svuint32_t src_vec_u32 = svunpklo_u32(svunpklo_u16(src_vec_u8));
733      svbool_t mask = svcmpne_u32(pg_32, src_vec_u32, ZERO_U32);
734      svst1_f32(pg_32, dst + i, svsel_f32(mask, ONE_F32, ZERO_F32));
735    }
736  #pragma unroll
737    for (int64_t i = n - fraction; i < n; i += Vectorized<float>::size()) {
738      pg_8 = svwhilelt_b8(i, n);
739      pg_32 = svwhilelt_b32(i, n);
740      svuint8_t src_vec_u8 =
741          svldnt1_u8(pg_8, reinterpret_cast<const uint8_t*>(src) + i);
742      svuint32_t src_vec_u32 = svunpklo_u32(svunpklo_u16(src_vec_u8));
743      svbool_t mask = svcmpne_u32(pg_32, src_vec_u32, ZERO_U32);
744      svst1_f32(pg_32, dst + i, svsel_f32(mask, ONE_F32, ZERO_F32));
745    }
746  }
