4    subroutine foo(x)
5      implicit none
6      integer, parameter :: sp = selected_real_kind(6)
7      integer, parameter :: dp = selected_real_kind(15)

9      integer, parameter :: il = selected_int_kind(18)
10     real(dp), intent(inout) :: x
11     dimension x(3)
12     real(sp), parameter :: three_s = 3._sp
13     real(dp), parameter :: three_d = 3._dp
14     integer(ii), parameter :: three_i = 3_ii
15     integer(il), parameter :: three_l = 3_il
16     x(1) = x(1) + x(2) * three_s * three_i + x(3) * three_d * three_l
17     x(2) = x(2) * three_s
18     x(3) = x(3) * three_l
