82   # NB!  This treats "skip" kwarg specially!!
83   def compile_time_strobelight_meta(
84       phase_name: str,
85   ) -> Callable[[Callable[_P, _T]], Callable[_P, _T]]:
86       def compile_time_strobelight_meta_inner(
87           function: Callable[_P, _T],
88       ) -> Callable[_P, _T]:
89           @functools.wraps(function)
90           def wrapper_function(*args: _P.args, **kwargs: _P.kwargs) -> _T:
91               if "skip" in kwargs and isinstance(skip := kwargs["skip"], int):
92                   kwargs["skip"] = skip + 1
93   
94               # This is not needed but we have it here to avoid having profile_compile_time
95               # in stack traces when profiling is not enabled.
96               if not StrobelightCompileTimeProfiler.enabled:
97                   return function(*args, **kwargs)
98   
99               return StrobelightCompileTimeProfiler.profile_compile_time(
100                  function, phase_name, *args, **kwargs
101              )
102  
103          return wrapper_function
104  
105      return compile_time_strobelight_meta_inner

176  def justknobs_check(name: str, default: bool = True) -> bool:
177      """ 
178      This function can be used to killswitch functionality in FB prod, 
179      where you can toggle this value to False in JK without having to 
180      do a code push.  In OSS, we always have everything turned on all 
181      the time, because downstream users can simply choose to not update 
182      PyTorch.  (If more fine-grained enable/disable is needed, we could 
183      potentially have a map we lookup name in to toggle behavior.  But 
184      the point is that it's all tied to source code in OSS, since there's 
185      no live server to query.) 
186   
187      This is the bare minimum functionality I needed to do some killswitches. 
188      We have a more detailed plan at 
189      https://docs.google.com/document/d/1Ukerh9_42SeGh89J-tGtecpHBPwGlkQ043pddkKb3PU/edit 
190      In particular, in some circumstances it may be necessary to read in 
191      a knob once at process start, and then use it consistently for the 
192      rest of the process.  Future functionality will codify these patterns 
193      into a better high level API. 
194   
195      WARNING: Do NOT call this function at module import time, JK is not 
196      fork safe and you will break anyone who forks the process and then 
197      hits JK again. 
198      """
199      return default

214  def max_clock_rate():
215      if not torch.version.hip:
216          from triton.testing import nvsmi
217  
218          return nvsmi(["clocks.max.sm"])[0]
219      else:
220          # Manually set max-clock speeds on ROCm until equivalent nvmsi
221          # functionality in triton.testing or via pyamdsmi enablement. Required
222          # for test_snode_runtime unit tests.
223          gcn_arch = str(torch.cuda.get_device_properties(0).gcnArchName.split(":", 1)[0])
224          if "gfx94" in gcn_arch:
225              return 1700
226          elif "gfx90a" in gcn_arch:
227              return 1700
228          elif "gfx908" in gcn_arch:
229              return 1502
230          elif "gfx12" in gcn_arch:
231              return 1700
232          elif "gfx11" in gcn_arch:
233              return 1700
234          elif "gfx103" in gcn_arch:
235              return 1967
236          elif "gfx101" in gcn_arch:
237              return 1144
238          elif "gfx95" in gcn_arch:
239              return 1700  # TODO: placeholder, get actual value
240          else:
241              return 1100
