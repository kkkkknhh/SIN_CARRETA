232  def types_compatible(var1, var2):
233      """Check if types are same or compatible. 
234   
235      0-D numpy scalars are compatible with bare python scalars. 
236      """
237      type1 = type(var1)
238      type2 = type(var2)
239      if type1 is type2:
240          return True
241      if type1 is np.ndarray and var1.shape == ():
242          return type(var1.item()) is type2
243      if type2 is np.ndarray and var2.shape == ():
244          return type(var2.item()) is type1
245      return False

248  def _check_level(label, expected, actual):
249      """ Check one level of a potentially nested array """
250      if issparse(expected):  # allow different types of sparse matrices
251          assert_(issparse(actual))
252          assert_array_almost_equal(actual.toarray(),
253                                    expected.toarray(),
254                                    err_msg=label,
255                                    decimal=5)
256          return
257      # Check types are as expected
258      assert_(types_compatible(expected, actual),
259              f"Expected type {type(expected)}, got {type(actual)} at {label}")
260      # A field in a record array may not be an ndarray
261      # A scalar from a record array will be type np.void
262      if not isinstance(expected, np.void | np.ndarray | MatlabObject):
263          assert_equal(expected, actual)
264          return
265      # This is an ndarray-like thing
266      assert_(expected.shape == actual.shape,
267              msg=f'Expected shape {expected.shape}, got {actual.shape} at {label}')
268      ex_dtype = expected.dtype
269      if ex_dtype.hasobject:  # array of objects
270          if isinstance(expected, MatlabObject):
271              assert_equal(expected.classname, actual.classname)
272          for i, ev in enumerate(expected):
273              level_label = "%s, [%d], " % (label, i)
274              _check_level(level_label, ev, actual[i])
275          return
276      if ex_dtype.fields:  # probably recarray
277          for fn in ex_dtype.fields:
278              level_label = f"{label}, field {fn}, "
279              _check_level(level_label,
280                           expected[fn], actual[fn])
281          return
282      if ex_dtype.type in (str,  # string or bool
283                           np.str_,
284                           np.bool_):
285          assert_equal(actual, expected, err_msg=label)
286          return
287      # Something numeric
288      assert_array_almost_equal(actual, expected, err_msg=label, decimal=5)

292      for file_name in files:
293          matdict = loadmat(file_name, struct_as_record=True, spmatrix=False)
294          label = f"test {name}; file {file_name}"

297              assert_(k in matdict, f"Missing key at {k_label}")
298              _check_level(k_label, expected, matdict[k])

304  
305          whos = whosmat(file_name)

706  def test_to_writeable():
707      # Test to_writeable function
708      res = to_writeable(np.array([1]))  # pass through ndarrays
709      assert_equal(res.shape, (1,))
710      assert_array_equal(res, 1)
711      # Dict fields can be written in any order
712      expected1 = np.array([(1, 2)], dtype=[('a', '|O8'), ('b', '|O8')])
713      expected2 = np.array([(2, 1)], dtype=[('b', '|O8'), ('a', '|O8')])
714      alternatives = (expected1, expected2)
715      assert_any_equal(to_writeable({'a':1,'b':2}), alternatives)
716      # Fields with underscores discarded
717      assert_any_equal(to_writeable({'a':1,'b':2, '_c':3}), alternatives)
718      # Not-string fields discarded
719      assert_any_equal(to_writeable({'a':1,'b':2, 100:3}), alternatives)
720      # String fields that are valid Python identifiers discarded
721      assert_any_equal(to_writeable({'a':1,'b':2, '99':3}), alternatives)
722      # Object with field names is equivalent
723  
724      class klass:
725          pass
726  
727      c = klass
728      c.a = 1
729      c.b = 2
730      assert_any_equal(to_writeable(c), alternatives)
731      # empty list and tuple go to empty array
732      res = to_writeable([])
733      assert_equal(res.shape, (0,))
734      assert_equal(res.dtype.type, np.float64)
735      res = to_writeable(())
736      assert_equal(res.shape, (0,))
737      assert_equal(res.dtype.type, np.float64)
738      # None -> None
739      assert_(to_writeable(None) is None)
740      # String to strings
741      assert_equal(to_writeable('a string').dtype.type, np.str_)
742      # Scalars to numpy to NumPy scalars
743      res = to_writeable(1)
744      assert_equal(res.shape, ())
745      assert_equal(res.dtype.type, np.array(1).dtype.type)
746      assert_array_equal(res, 1)
747      # Empty dict returns EmptyStructMarker
748      assert_(to_writeable({}) is EmptyStructMarker)
749      # Object does not have (even empty) __dict__
750      assert_(to_writeable(object()) is None)
751      # Custom object does have empty __dict__, returns EmptyStructMarker
752  
753      class C:
754          pass
755  
756      assert_(to_writeable(c()) is EmptyStructMarker)
757      # dict keys with legal characters are convertible
758      res = to_writeable({'a': 1})['a']
759      assert_equal(res.shape, (1,))
760      assert_equal(res.dtype.type, np.object_)
761      # Only fields with illegal characters, falls back to EmptyStruct
762      assert_(to_writeable({'1':1}) is EmptyStructMarker)
763      assert_(to_writeable({'_a':1}) is EmptyStructMarker)
764      # Unless there are valid fields, in which case structured array
765      assert_equal(to_writeable({'1':1, 'f': 2}),
766                   np.array([(2,)], dtype=[('f', '|O8')]))
