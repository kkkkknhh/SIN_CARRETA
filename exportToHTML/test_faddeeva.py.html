57       def test_symmetry(self):
58           x = np.linspace(0, 10, 20)
59           assert_allclose(
60               sc.voigt_profile(x, 1, 1),
61               sc.voigt_profile(-x, 1, 1),
62               rtol=1e-15,
63               atol=0
64           )

66       @pytest.mark.parametrize('x, sigma, gamma, desired', [
