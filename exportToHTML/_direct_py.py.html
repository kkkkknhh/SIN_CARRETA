13       "Number of function evaluations done is larger than maxfun={}",
14       "Number of iterations is larger than maxiter={}",

16       "maxfun is too large",

36   def direct(
37       func: Callable[
38           [np.ndarray[tuple[int], np.dtype[np.float64]]],
39           float | np.floating[Any] | np.integer[Any] | np.bool_,
40       ],
41       bounds: Iterable | Bounds,
42       *,
43       args: tuple = (),
44       eps: float = 1e-4,
45       maxfun: int | None = None,
46       maxiter: int = 1000,
47       locally_biased: bool = True,
48       f_min: float = -np.inf,
49       f_min_rtol: float = 1e-4,
50       vol_tol: float = 1e-16,
51       len_tol: float = 1e-6,
52       callback: Callable[
53           [np.ndarray[tuple[int], np.dtype[np.float64]]],
54           object,
55       ] | None = None,
56   ) -> OptimizeResult:
57       """ 
58       Finds the global minimum of a function using the 
59       DIRECT algorithm. 
60    
61       Parameters 
62       ---------- 
63       func : callable 
64           The objective function to be minimized. 
65           ``func(x, *args) -> float`` 
66           where ``x`` is an 1-D array with shape (n,) and ``args`` is a tuple of 
67           the fixed parameters needed to completely specify the function. 
68       bounds : sequence or `Bounds` 
69           Bounds for variables. There are two ways to specify the bounds: 
70    
71           1. Instance of `Bounds` class. 
72           2. ``(min, max)`` pairs for each element in ``x``. 
73    
74       args : tuple, optional 
75           Any additional fixed parameters needed to 
76           completely specify the objective function. 
77       eps : float, optional 
78           Minimal required difference of the objective function values 
79           between the current best hyperrectangle and the next potentially 
80           optimal hyperrectangle to be divided. In consequence, `eps` serves as a 
81           tradeoff between local and global search: the smaller, the more local 
82           the search becomes. Default is 1e-4. 
83       maxfun : int or None, optional 
84           Approximate upper bound on objective function evaluations. 
85           If `None`, will be automatically set to ``1000 * N`` where ``N`` 
86           represents the number of dimensions. Will be capped if necessary to 
87           limit DIRECT's RAM usage to app. 1GiB. This will only occur for very 
88           high dimensional problems and excessive `max_fun`. Default is `None`. 
89       maxiter : int, optional 
90           Maximum number of iterations. Default is 1000. 
91       locally_biased : bool, optional 
92           If `True` (default), use the locally biased variant of the 
93           algorithm known as DIRECT_L. If `False`, use the original unbiased 
94           DIRECT algorithm. For hard problems with many local minima, 
95           `False` is recommended. 
96       f_min : float, optional 
97           Function value of the global optimum. Set this value only if the 
98           global optimum is known. Default is ``-np.inf``, so that this 
99           termination criterion is deactivated. 
100      f_min_rtol : float, optional 
101          Terminate the optimization once the relative error between the 
102          current best minimum `f` and the supplied global minimum `f_min` 
103          is smaller than `f_min_rtol`. This parameter is only used if 
104          `f_min` is also set. Must lie between 0 and 1. Default is 1e-4. 
105      vol_tol : float, optional 
106          Terminate the optimization once the volume of the hyperrectangle 
107          containing the lowest function value is smaller than `vol_tol` 
108          of the complete search space. Must lie between 0 and 1. 
109          Default is 1e-16. 
110      len_tol : float, optional 
111          If ``locally_biased=True``, terminate the optimization once half of 
112          the normalized maximal side length of the hyperrectangle containing 
113          the lowest function value is smaller than `len_tol`. 
114          If ``locally_biased=False``, terminate the optimization once half of 
115          the normalized diagonal of the hyperrectangle containing the lowest 
116          function value is smaller than `len_tol`. Must lie between 0 and 1. 
117          Default is 1e-6. 
118      callback : callable, optional 
119          A callback function with signature ``callback(xk)`` where ``xk`` 
120          represents the best function value found so far. 
121   
122      Returns 
123      ------- 
124      res : OptimizeResult 
125          The optimization result represented as a ``OptimizeResult`` object. 
126          Important attributes are: ``x`` the solution array, ``success`` a 
127          Boolean flag indicating if the optimizer exited successfully and 
128          ``message`` which describes the cause of the termination. See 
129          `OptimizeResult` for a description of other attributes. 
130   
131      Notes 
132      ----- 
133      DIviding RECTangles (DIRECT) is a deterministic global 
134      optimization algorithm capable of minimizing a black box function with 
135      its variables subject to lower and upper bound constraints by sampling 
136      potential solutions in the search space [1]_. The algorithm starts by 
137      normalising the search space to an n-dimensional unit hypercube. 
138      It samples the function at the center of this hypercube and at 2n 
139      (n is the number of variables) more points, 2 in each coordinate 
140      direction. Using these function values, DIRECT then divides the 
141      domain into hyperrectangles, each having exactly one of the sampling 
142      points as its center. In each iteration, DIRECT chooses, using the `eps` 
143      parameter which defaults to 1e-4, some of the existing hyperrectangles 
144      to be further divided. This division process continues until either the 
145      maximum number of iterations or maximum function evaluations allowed 
146      are exceeded, or the hyperrectangle containing the minimal value found 
147      so far becomes small enough. If `f_min` is specified, the optimization 
148      will stop once this function value is reached within a relative tolerance. 
149      The locally biased variant of DIRECT (originally called DIRECT_L) [2]_ is 
150      used by default. It makes the search more locally biased and more 
151      efficient for cases with only a few local minima. 
152   
153      A note about termination criteria: `vol_tol` refers to the volume of the 
154      hyperrectangle containing the lowest function value found so far. This 
155      volume decreases exponentially with increasing dimensionality of the 
156      problem. Therefore `vol_tol` should be decreased to avoid premature 
157      termination of the algorithm for higher dimensions. This does not hold 
158      for `len_tol`: it refers either to half of the maximal side length 
159      (for ``locally_biased=True``) or half of the diagonal of the 
160      hyperrectangle (for ``locally_biased=False``). 
161   
162      This code is based on the DIRECT 2.0.4 Fortran code by Gablonsky et al. at 
163      https://ctk.math.ncsu.edu/SOFTWARE/DIRECTv204.tar.gz . 
164      This original version was initially converted via f2c and then cleaned up 
165      and reorganized by Steven G. Johnson, August 2007, for the NLopt project. 
166      The `direct` function wraps the C implementation. 
167

170      References 
171      ---------- 
172      .. [1] Jones, D.R., Perttunen, C.D. & Stuckman, B.E. Lipschitzian 
173          optimization without the Lipschitz constant. J Optim Theory Appl 
174          79, 157-181 (1993). 
175      .. [2] Gablonsky, J., Kelley, C. A Locally-Biased form of the DIRECT 
176          Algorithm. Journal of Global Optimization 21, 27-37 (2001). 
177   
178      Examples 
179      -------- 
180      The following example is a 2-D problem with four local minima: minimizing 
181      the Styblinski-Tang function 
182      (https://en.wikipedia.org/wiki/Test_functions_for_optimization). 
183   
184      >>> from scipy.optimize import direct, Bounds 
185      >>> def styblinski_tang(pos): 
186      ...     x, y = pos 
187      ...     return 0.5 * (x**4 - 16*x**2 + 5*x + y**4 - 16*y**2 + 5*y) 
188      >>> bounds = Bounds([-4., -4.], [4., 4.]) 
189      >>> result = direct(styblinski_tang, bounds) 
190      >>> result.x, result.fun, result.nfev 
191      array([-2.90321597, -2.90321597]), -78.3323279095383, 2011 
192   
193      The correct global minimum was found but with a huge number of function 
194      evaluations (2011). Loosening the termination tolerances `vol_tol` and 
195      `len_tol` can be used to stop DIRECT earlier. 
196   
197      >>> result = direct(styblinski_tang, bounds, len_tol=1e-3) 
198      >>> result.x, result.fun, result.nfev 
199      array([-2.9044353, -2.9044353]), -78.33230330754142, 207 
200   
201      """
202      # convert bounds to new Bounds class if necessary
203      if not isinstance(bounds, Bounds):
204          if isinstance(bounds, list) or isinstance(bounds, tuple):
205              lb, ub = old_bound_to_new(bounds)
206              bounds = Bounds(lb, ub)
207          else:
208              message = ("bounds must be a sequence or "
209                         "instance of Bounds class")
210              raise ValueError(message)
211  
212      lb = np.ascontiguousarray(bounds.lb, dtype=np.float64)
213      ub = np.ascontiguousarray(bounds.ub, dtype=np.float64)
214  
215      # validate bounds
216      # check that lower bounds are smaller than upper bounds
217      if not np.all(lb < ub):
218          raise ValueError('Bounds are not consistent min < max')
219      # check for infs
220      if (np.any(np.isinf(lb)) or np.any(np.isinf(ub))):
221          raise ValueError("Bounds must not be inf.")
222  
223      # validate tolerances
224      if (vol_tol < 0 or vol_tol > 1):
225          raise ValueError("vol_tol must be between 0 and 1.")
226      if (len_tol < 0 or len_tol > 1):
227          raise ValueError("len_tol must be between 0 and 1.")
228      if (f_min_rtol < 0 or f_min_rtol > 1):
229          raise ValueError("f_min_rtol must be between 0 and 1.")
230  
231      # validate maxfun and maxiter
232      if maxfun is None:
233          maxfun = 1000 * lb.shape[0]
234      if not isinstance(maxfun, int):
235          raise ValueError("maxfun must be of type int.")
236      if maxfun < 0:
237          raise ValueError("maxfun must be > 0.")
238      if not isinstance(maxiter, int):
239          raise ValueError("maxiter must be of type int.")
240      if maxiter < 0:
241          raise ValueError("maxiter must be > 0.")
242  
243      # validate boolean parameters
244      if not isinstance(locally_biased, bool):
245          raise ValueError("locally_biased must be True or False.")
246  
247      def _func_wrap(x, args=None):
248          x = np.asarray(x)
249          if args is None:
250              f = func(x)
251          else:
252              f = func(x, *args)
253          # always return a float
254          return np.asarray(f).item()
255  
256      # TODO: fix disp argument
257      x, fun, ret_code, nfev, nit = _direct(
258          _func_wrap,
259          np.asarray(lb), np.asarray(ub),
260          args,
261          False, eps, maxfun, maxiter,
262          locally_biased,
263          f_min, f_min_rtol,
264          vol_tol, len_tol, callback
265      )
266  
267      format_val = (maxfun, maxiter, f_min_rtol, vol_tol, len_tol)
268      if ret_code > 2:
269          message = SUCCESS_MESSAGES[ret_code - 3].format(
270                      format_val[ret_code - 1])
271      elif 0 < ret_code <= 2:
272          message = ERROR_MESSAGES[ret_code - 1].format(format_val[ret_code - 1])
273      elif 0 > ret_code > -100:
274          message = ERROR_MESSAGES[abs(ret_code) + 1]
275      else:
276          message = ERROR_MESSAGES[ret_code + 99]
277  
278      return OptimizeResult(x=np.asarray(x), fun=fun, status=ret_code,
279                            success=ret_code > 2, message=message,
280                            nfev=nfev, nit=nit)
