1    import os
2    import subprocess
3    import sysconfig

5    import pytest
6    import numpy as np

8    from numpy.testing import IS_WASM
9    
10   
11   is_editable = not bool(np.__path__)
12   numpy_in_sitepackages = sysconfig.get_path('platlib') in np.__file__
13   # We only expect to have a `numpy-config` available if NumPy was installed via
14   # a build frontend (and not `spin` for example)
15   if not (numpy_in_sitepackages or is_editable):
16       pytest.skip("`numpy-config` not expected to be installed",
17                   allow_module_level=True)
18   
19   
20   def check_numpyconfig(arg):
21       p = subprocess.run(['numpy-config', arg], capture_output=True, text=True)
22       p.check_returncode()
23       return p.stdout.strip()
24   
25   @pytest.mark.skipif(IS_WASM, reason="wasm interpreter cannot start subprocess")
26   def test_configtool_version():
27       stdout = check_numpyconfig('--version')
28       assert stdout == np.__version__
29   
30   @pytest.mark.skipif(IS_WASM, reason="wasm interpreter cannot start subprocess")
31   def test_configtool_cflags():
32       stdout = check_numpyconfig('--cflags')
33       assert stdout.endswith(os.path.join('numpy', '_core', 'include'))
34   
35   @pytest.mark.skipif(IS_WASM, reason="wasm interpreter cannot start subprocess")
36   def test_configtool_pkgconfigdir():
37       stdout = check_numpyconfig('--pkgconfigdir')
38       assert stdout.endswith(os.path.join('numpy', '_core', 'lib', 'pkgconfig'))
39   
40       if not is_editable:
41           # Also check that the .pc file actually exists (unless we're using an
42           # editable install, then it'll be hiding in the build dir)
43           assert os.path.exists(os.path.join(stdout, 'numpy.pc'))
