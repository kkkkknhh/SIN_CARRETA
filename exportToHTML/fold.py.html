11   class Fold(Module):
12       (
13           r"""Combines an array of sliding local blocks into a large containing tensor. 
14    
15       Consider a batched :attr:`input` tensor containing sliding local blocks, 
16       e.g., patches of images, of shape :math:`(N, C \times  \prod(\text{kernel\_size}), L)`, 
17       where :math:`N` is batch dimension, :math:`C \times \prod(\text{kernel\_size})` 
18       is the number of values within a block (a block has :math:`\prod(\text{kernel\_size})` 
19       spatial locations each containing a :math:`C`-channeled vector), and 
20       :math:`L` is the total number of blocks. (This is exactly the 
21       same specification as the output shape of :class:`~torch.nn.Unfold`.) This 
22       operation combines these local blocks into the large :attr:`output` tensor 
23       of shape :math:`(N, C, \text{output\_size}[0], \text{output\_size}[1], \dots)` 
24       by summing the overlapping values. Similar to :class:`~torch.nn.Unfold`, the 
25       arguments must satisfy 
26    
27       .. math:: 
28           L = \prod_d \left\lfloor\frac{\text{output\_size}[d] + 2 \times \text{padding}[d] % 
29               - \text{dilation}[d] \times (\text{kernel\_size}[d] - 1) - 1}{\text{stride}[d]} + 1\right\rfloor, 
30    
31       where :math:`d` is over all spatial dimensions. 
32    
33       * :attr:`output_size` describes the spatial shape of the large containing 
34         tensor of the sliding local blocks. It is useful to resolve the ambiguity 
35         when multiple input shapes map to same number of sliding blocks, e.g., 
36         with ``stride > 0``. 
37    
38       The :attr:`padding`, :attr:`stride` and :attr:`dilation` arguments specify 
39       how the sliding blocks are retrieved. 
40    
41       * :attr:`stride` controls the stride for the sliding blocks. 
42    
43       * :attr:`padding` controls the amount of implicit zero-paddings on both 
44         sides for :attr:`padding` number of points for each dimension before 
45         reshaping. 
46   """
47           """ 
48       * :attr:`dilation` controls the spacing between the kernel points; also known as the \u00e0 trous algorithm. 
49         It is harder to describe, but this `link`_ has a nice visualization of what :attr:`dilation` does. 
50   """
51           r""" 
52       Args: 
53           output_size (int or tuple): the shape of the spatial dimensions of the 
54                                       output (i.e., ``output.sizes()[2:]``) 
55           kernel_size (int or tuple): the size of the sliding blocks 
56           dilation (int or tuple, optional): a parameter that controls the 
57                                              stride of elements within the 
58                                              neighborhood. Default: 1 
59           padding (int or tuple, optional): implicit zero padding to be added on 
60                                             both sides of input. Default: 0 
61           stride (int or tuple): the stride of the sliding blocks in the input 
62                                  spatial dimensions. Default: 1 
63    
64       * If :attr:`output_size`, :attr:`kernel_size`, :attr:`dilation`, 
65         :attr:`padding` or :attr:`stride` is an int or a tuple of length 1 then 
66         their values will be replicated across all spatial dimensions. 
67    
68       * For the case of two output spatial dimensions this operation is sometimes 
69         called ``col2im``. 
70    
71       .. note:: 
72           :class:`~torch.nn.Fold` calculates each combined value in the resulting 
73           large tensor by summing all values from all containing blocks. 
74           :class:`~torch.nn.Unfold` extracts the values in the local blocks by 
75           copying from the large tensor. So, if the blocks overlap, they are not 
76           inverses of each other. 
77    
78           In general, folding and unfolding operations are related as 
79           follows. Consider :class:`~torch.nn.Fold` and 
80           :class:`~torch.nn.Unfold` instances created with the same 
81           parameters: 
82    
83           >>> fold_params = dict(kernel_size=..., dilation=..., padding=..., stride=...) 
84           >>> fold = nn.Fold(output_size=..., **fold_params) 
85           >>> unfold = nn.Unfold(**fold_params) 
86    
87           Then for any (supported) ``input`` tensor the following 
88           equality holds: 
89    
90           :: 
91    
92               fold(unfold(input)) == divisor * input 
93    
94           where ``divisor`` is a tensor that depends only on the shape 
95           and dtype of the ``input``: 
96    
97           >>> # xdoctest: +SKIP 
98           >>> input_ones = torch.ones(input.shape, dtype=input.dtype) 
99           >>> divisor = fold(unfold(input_ones)) 
100   
101          When the ``divisor`` tensor contains no zero elements, then 
102          ``fold`` and ``unfold`` operations are inverses of each 
103          other (up to constant divisor). 
104   
105      .. warning:: 
106          Currently, only unbatched (3D) or batched (4D) image-like output tensors are supported. 
107   
108      Shape: 
109          - Input: :math:`(N, C \times \prod(\text{kernel\_size}), L)` or :math:`(C \times \prod(\text{kernel\_size}), L)` 
110          - Output: :math:`(N, C, \text{output\_size}[0], \text{output\_size}[1], \dots)` 
111            or :math:`(C, \text{output\_size}[0], \text{output\_size}[1], \dots)` as described above 
112   
113      Examples:: 
114   
115          >>> fold = nn.Fold(output_size=(4, 5), kernel_size=(2, 2)) 
116          >>> input = torch.randn(1, 3 * 2 * 2, 12) 
117          >>> output = fold(input) 
118          >>> output.size() 
119          torch.Size([1, 3, 4, 5]) 
120   
121      .. _link: 
122          https://github.com/vdumoulin/conv_arithmetic/blob/master/README.md 
123   
124      """
125      )

145          self.dilation = dilation

147          self.stride = stride

156              self.stride,
157          )

163                  **self.__dict__
164              )

168  class Unfold(Module):
169      (
170          r"""Extracts sliding local blocks from a batched input tensor. 
171   
172      Consider a batched :attr:`input` tensor of shape :math:`(N, C, *)`, 
173      where :math:`N` is the batch dimension, :math:`C` is the channel dimension, 
174      and :math:`*` represent arbitrary spatial dimensions. This operation flattens 
175      each sliding :attr:`kernel_size`-sized block within the spatial dimensions 
176      of :attr:`input` into a column (i.e., last dimension) of a 3-D :attr:`output` 
177      tensor of shape :math:`(N, C \times \prod(\text{kernel\_size}), L)`, where 
178      :math:`C \times \prod(\text{kernel\_size})` is the total number of values 
179      within each block (a block has :math:`\prod(\text{kernel\_size})` spatial 
180      locations each containing a :math:`C`-channeled vector), and :math:`L` is 
181      the total number of such blocks: 
182   
183      .. math:: 
184          L = \prod_d \left\lfloor\frac{\text{spatial\_size}[d] + 2 \times \text{padding}[d] % 
185              - \text{dilation}[d] \times (\text{kernel\_size}[d] - 1) - 1}{\text{stride}[d]} + 1\right\rfloor, 
186   
187      where :math:`\text{spatial\_size}` is formed by the spatial dimensions 
188      of :attr:`input` (:math:`*` above), and :math:`d` is over all spatial 
189      dimensions. 
190   
191      Therefore, indexing :attr:`output` at the last dimension (column dimension) 
192      gives all values within a certain block. 
193   
194      The :attr:`padding`, :attr:`stride` and :attr:`dilation` arguments specify 
195      how the sliding blocks are retrieved. 
196   
197      * :attr:`stride` controls the stride for the sliding blocks. 
198   
199      * :attr:`padding` controls the amount of implicit zero-paddings on both 
200        sides for :attr:`padding` number of points for each dimension before 
201        reshaping. 
202  """
203          """ 
204      * :attr:`dilation` controls the spacing between the kernel points; also known as the \u00e0 trous algorithm. 
205        It is harder to describe, but this `link`_ has a nice visualization of what :attr:`dilation` does. 
206  """
207          r""" 
208      Args: 
209          kernel_size (int or tuple): the size of the sliding blocks 
210          dilation (int or tuple, optional): a parameter that controls the 
211                                             stride of elements within the 
212                                             neighborhood. Default: 1 
213          padding (int or tuple, optional): implicit zero padding to be added on 
214                                            both sides of input. Default: 0 
215          stride (int or tuple, optional): the stride of the sliding blocks in the input 
216                                           spatial dimensions. Default: 1 
217   
218      * If :attr:`kernel_size`, :attr:`dilation`, :attr:`padding` or 
219        :attr:`stride` is an int or a tuple of length 1, their values will be 
220        replicated across all spatial dimensions. 
221   
222      * For the case of two input spatial dimensions this operation is sometimes 
223        called ``im2col``. 
224   
225      .. note:: 
226          :class:`~torch.nn.Fold` calculates each combined value in the resulting 
227          large tensor by summing all values from all containing blocks. 
228          :class:`~torch.nn.Unfold` extracts the values in the local blocks by 
229          copying from the large tensor. So, if the blocks overlap, they are not 
230          inverses of each other. 
231   
232          In general, folding and unfolding operations are related as 
233          follows. Consider :class:`~torch.nn.Fold` and 
234          :class:`~torch.nn.Unfold` instances created with the same 
235          parameters: 
236   
237          >>> fold_params = dict(kernel_size=..., dilation=..., padding=..., stride=...) 
238          >>> fold = nn.Fold(output_size=..., **fold_params) 
239          >>> unfold = nn.Unfold(**fold_params) 
240   
241          Then for any (supported) ``input`` tensor the following 
242          equality holds: 
243   
244          :: 
245   
246              fold(unfold(input)) == divisor * input 
247   
248          where ``divisor`` is a tensor that depends only on the shape 
249          and dtype of the ``input``: 
250   
251          >>> # xdoctest: +SKIP 
252          >>> input_ones = torch.ones(input.shape, dtype=input.dtype) 
253          >>> divisor = fold(unfold(input_ones)) 
254   
255          When the ``divisor`` tensor contains no zero elements, then 
256          ``fold`` and ``unfold`` operations are inverses of each 
257          other (up to constant divisor). 
258   
259      .. warning:: 
260          Currently, only 4-D input tensors (batched image-like tensors) are 
261          supported. 
262   
263      Shape: 
264          - Input: :math:`(N, C, *)` 
265          - Output: :math:`(N, C \times \prod(\text{kernel\_size}), L)` as described above 
266   
267      Examples:: 
268   
269          >>> unfold = nn.Unfold(kernel_size=(2, 3)) 
270          >>> input = torch.randn(2, 5, 3, 4) 
271          >>> output = unfold(input) 
272          >>> # each patch contains 30 values (2x3=6 vectors, each of 5 channels) 
273          >>> # 4 blocks (2x3 kernels) in total in the 3x4 input 
274          >>> output.size() 
275          torch.Size([2, 30, 4]) 
276   
277          >>> # xdoctest: +IGNORE_WANT 
278          >>> # Convolution is equivalent with Unfold + Matrix Multiplication + Fold (or view to output shape) 
279          >>> inp = torch.randn(1, 3, 10, 12) 
280          >>> w = torch.randn(2, 3, 4, 5) 
281          >>> inp_unf = torch.nn.functional.unfold(inp, (4, 5)) 
282          >>> out_unf = inp_unf.transpose(1, 2).matmul(w.view(w.size(0), -1).t()).transpose(1, 2) 
283          >>> out = torch.nn.functional.fold(out_unf, (7, 8), (1, 1)) 
284          >>> # or equivalently (and avoiding a copy), 
285          >>> # out = out_unf.view(1, 2, 7, 8) 
286          >>> (torch.nn.functional.conv2d(inp, w) - out).abs().max() 
287          tensor(1.9073e-06) 
288   
289      .. _link: 
290          https://github.com/vdumoulin/conv_arithmetic/blob/master/README.md 
291   
292      """
293      )
