46   def _attribute_type_compatible_with_arg(
47       attr: _schemas.AttributeParameter,
48       value: ir.Value | int | float | bool | Sequence[int] | Sequence[float] | None,
49   ) -> bool:
50       """Check if the attribute type is compatible with the argument."""
51       if isinstance(value, bool):
52           return attr.type is ir.AttributeType.INT
53       if isinstance(value, str):
54           return attr.type is ir.AttributeType.STRING
55       if isinstance(value, int):
56           return attr.type in {ir.AttributeType.INT, ir.AttributeType.FLOAT}
57       if isinstance(value, float):
58           return attr.type is ir.AttributeType.FLOAT
59       if isinstance(value, complex):
60           return False
61       if isinstance(value, Sequence):
62           if attr.type is ir.AttributeType.INTS:
63               return all(isinstance(i, int) for i in value)
64           if attr.type is ir.AttributeType.FLOATS:
65               return all(isinstance(i, (int, float)) for i in value)
66       if isinstance(value, torch.dtype):
67           return attr.type is ir.AttributeType.INT
68       if isinstance(value, (torch.device, torch.memory_format, torch.layout)):
69           return attr.type is ir.AttributeType.STRING
70       if value is None and not attr.required:
71           # An optional attribute is not supplied
72           return True
73       return False

76   def _param_type_compatible_with_arg(
77       param: _schemas.Parameter,
78       value: ir.TypeProtocol
79       | str
80       | int
81       | float
82       | complex
83       | Sequence[int]
84       | Sequence[float]
85       | None,
86       assigned_types: dict[str, ir.TypeProtocol],
87   ) -> bool:
88       # Handle Python types first
89       if isinstance(value, bool):  # noqa: SIM102
90           if param.type_constraint.allowed_types & {ir.TensorType(ir.DataType.BOOL)}:
91               return True
92       if isinstance(value, int) and param.type_constraint.allowed_types & {
93           ir.TensorType(ir.DataType.INT4),
94           ir.TensorType(ir.DataType.INT8),
95           ir.TensorType(ir.DataType.INT16),
96           ir.TensorType(ir.DataType.INT32),
97           ir.TensorType(ir.DataType.INT64),
98           # Int inputs can be casted to a float too
99           ir.TensorType(ir.DataType.FLOAT4E2M1),
100          ir.TensorType(ir.DataType.FLOAT8E4M3FN),
101          ir.TensorType(ir.DataType.FLOAT8E4M3FNUZ),
102          ir.TensorType(ir.DataType.FLOAT8E5M2),
103          ir.TensorType(ir.DataType.FLOAT8E5M2FNUZ),
104          ir.TensorType(ir.DataType.FLOAT16),
105          ir.TensorType(ir.DataType.FLOAT),
106          ir.TensorType(ir.DataType.DOUBLE),
107      }:
108          return True
109      if isinstance(value, float) and param.type_constraint.allowed_types & {
110          ir.TensorType(ir.DataType.FLOAT4E2M1),
111          ir.TensorType(ir.DataType.FLOAT8E4M3FN),
112          ir.TensorType(ir.DataType.FLOAT8E4M3FNUZ),
113          ir.TensorType(ir.DataType.FLOAT8E5M2),
114          ir.TensorType(ir.DataType.FLOAT8E5M2FNUZ),
115          ir.TensorType(ir.DataType.FLOAT16),
116          ir.TensorType(ir.DataType.FLOAT),
117          ir.TensorType(ir.DataType.DOUBLE),
118      }:
119          return True
120      if isinstance(value, complex) and param.type_constraint.allowed_types & {
121          ir.TensorType(ir.DataType.FLOAT),
122          ir.TensorType(ir.DataType.DOUBLE),
123          ir.TensorType(ir.DataType.COMPLEX64),
124          ir.TensorType(ir.DataType.COMPLEX128),
125      }:
126          return True
127      if isinstance(value, str):  # noqa: SIM102
128          if param.type_constraint.allowed_types & {ir.TensorType(ir.DataType.STRING)}:
129              return True
130      if isinstance(value, (list, tuple)):
131          if param.type_constraint.allowed_types & {
132              ir.TensorType(ir.DataType.INT32),
133              ir.TensorType(ir.DataType.INT64),
134              ir.TensorType(ir.DataType.FLOAT),
135              ir.TensorType(ir.DataType.DOUBLE),
136              ir.SequenceType(ir.TensorType(ir.DataType.INT32)),
137              ir.SequenceType(ir.TensorType(ir.DataType.INT64)),
138              ir.SequenceType(ir.TensorType(ir.DataType.FLOAT)),
139              ir.SequenceType(ir.TensorType(ir.DataType.DOUBLE)),
140          } and all(isinstance(i, (int)) for i in value):
141              # We will just allow any fx node and trust that the overload handles it
142              return True
143          if param.type_constraint.allowed_types & {
144              ir.TensorType(ir.DataType.FLOAT),
145              ir.TensorType(ir.DataType.DOUBLE),
146              ir.SequenceType(ir.TensorType(ir.DataType.FLOAT)),
147              ir.SequenceType(ir.TensorType(ir.DataType.DOUBLE)),
148          } and all(isinstance(i, (int, float)) for i in value):
149              # We will just allow any fx node and trust that the overload handles it
150              return True
151      if value is None and not param.required:
152          # An optional parameter is not supplied
153          return True
154  
155      if not isinstance(value, ir.TypeProtocol):
156          return False
157  
158      # Then check tensor types
159      if param.type_constraint.name in assigned_types:
160          # If a typevar is already bound, check if the value has the same type
161          assigned_type = assigned_types[param.type_constraint.name]
162          return assigned_type == value
163      # If the typevar is not bound, bind it to the value type
164      if value in param.type_constraint.allowed_types:
165          # TODO: Maybe just check dtype? Being more strict here for now
166          assigned_types[param.type_constraint.name] = value
167          return True
168      return False

171  def _get_type_from_tensor(
172      tensor: torch.Tensor
173      | torch.SymBool
174      | torch.SymInt
175      | torch.SymFloat
176      | Sequence[torch.Tensor],
177  ) -> ir.TypeProtocol:
178      if isinstance(tensor, torch.Tensor):
179          return ir.TensorType(_torch_dtype_to_onnx_compatible_dtype(tensor.dtype))
180      if isinstance(tensor, torch.SymBool):
181          return ir.TensorType(ir.DataType.BOOL)
182      if isinstance(tensor, torch.SymInt):
183          return ir.TensorType(ir.DataType.INT64)
184      if isinstance(tensor, torch.SymFloat):
185          return ir.TensorType(ir.DataType.FLOAT)
186  
187      # Handle sequences
188      first_tensor = next((item for item in tensor if item is not None), None)
189      if first_tensor is None:
190          return ir.SequenceType(ir.TensorType(ir.DataType.UNDEFINED))
191      return ir.SequenceType(
192          ir.TensorType(_torch_dtype_to_onnx_compatible_dtype(first_tensor.dtype))
193      )

196  def _get_first_tensor_in_node_list(
197      nodes: Sequence[torch.fx.Node | Any],
198  ) -> torch.Tensor | None:
199      for node in nodes:
200          if (
201              isinstance(node, torch.fx.Node)
202              and "val" in node.meta
203              and isinstance(node.meta["val"], torch.Tensor)
204          ):
205              return node.meta["val"]
206      return None

209  def _get_named_fx_node_args(node: torch.fx.Node) -> dict[str, torch.fx.node.Argument]:
210      assert hasattr(node.target, "_schema")
211      torch_schema: torch.FunctionSchema = node.target._schema  # type: ignore[union-attr]
212      node_args = {}
213      for arg, schema_arg in zip(node.args, torch_schema.arguments):
214          node_args[schema_arg.name] = arg
215  
216      node_args.update(node.kwargs)
217      return node_args

220  def get_matching_overload(
221      node: torch.fx.Node,
222      overloads: Sequence[_registration.OnnxDecompMeta],
223  ) -> tuple[Callable | None, str]:
224      """Get the overload that matches the node's arguments. 
225   
226      Args: 
227          node: The node to match. 
228          overloads: The OnnxDecompMeta with overloads and their signatures to match against. 
229   
230      Returns: 
231          A tuple containing the matched overload and a string describing the reason for failure or success. 
232      """
233      if not hasattr(node.target, "_schema"):
234          # FIXME(justinchuby): When the target is a builtin, we should instead
235          # Match only the inputs positionally. Figure out how to do that as right
236          # now we assume all inputs are named.
237          return overloads[
238              0
239          ].onnx_function, "The node target does not have a schema. Return the first one."
240      named_args = _get_named_fx_node_args(node)
241      # FIXME: Handle when we don't know the names of the arguments
242      schema_args: dict[str, torch.Argument] = {
243          arg.name: arg
244          for arg in node.target._schema.arguments  # type: ignore[union-attr]
245      }
246      failure_messages: list[str] = []
247      for overload in overloads:
248          assigned_types: dict[str, ir.TypeProtocol] = {}
249          fail_reason = ""
250          if overload.signature is None:
251              # When an overload does not have a signature, we assume it is a custom op and should be matched
252              return (
253                  overload.onnx_function,
254                  "The overload does not have a signature. Assuming it is a custom op and matching it.",
255              )
256          for param in overload.signature:
257              if param.name not in schema_args and param.required:
258                  # We don't need to handle variadic inputs as there is none.
259                  # A required parameter is not supplied.
260                  fail_reason = "Required parameter not supplied"
261                  break
262  
263              # Get the argument
264              if param.name in named_args:
265                  # Provided in Node args
266                  arg = named_args[param.name]
267              elif (
268                  param.name in schema_args
269                  and schema_args[param.name].has_default_value()
270              ):
271                  # Provided in schema args
272                  arg = schema_args[param.name].default_value
273              elif param.has_default():
274                  # Provided in the ONNX op definition
275                  arg = param.default  # type: ignore[assignment]
276              else:
277                  fail_reason = "Parameter not provided"
278                  break
279  
280              if isinstance(param, _schemas.Parameter):
281                  if isinstance(arg, torch.Tensor):
282                      arg = _get_type_from_tensor(arg)  # type: ignore[assignment]
283                  if isinstance(arg, (list, tuple)) and any(
284                      isinstance(t, torch.fx.Node) for t in arg
285                  ):
286                      first_tensor = _get_first_tensor_in_node_list(arg)  # type: ignore[arg-type]
287                      assert first_tensor is not None
288                      # FIXME: Handle symfloat here
289                      arg = ir.SequenceType(_get_type_from_tensor(first_tensor))  # type: ignore[assignment]
290                  elif isinstance(arg, torch.fx.Node):
291                      meta_val = arg.meta["val"]
292                      arg = _get_type_from_tensor(meta_val)  # type: ignore[assignment]
293                  # TODO: Handle None attributes
294                  # FIXME: Handle symfloat etc.
295                  # Handle tensors and Python values
296                  if not _param_type_compatible_with_arg(param, arg, assigned_types):  # type: ignore[arg-type]
297                      fail_reason = (
298                          f"Parameter type not compatible with argument: param=`{param}`, "
299                          f"assigned_types=`{assigned_types}`, arg=`{arg}`"
300                      )
301                      break
302              elif isinstance(param, _schemas.AttributeParameter):
303                  if not _attribute_type_compatible_with_arg(param, arg):  # type: ignore[arg-type]
304                      fail_reason = f"Attribute type not compatible with argument: param=`{param}`, arg=`{arg}`"
305                      break
306              else:
307                  raise TypeError(f"Unknown parameter type: {type(param)}")
308          if not fail_reason:
309              return overload.onnx_function, "Successfully matched overload"
310          else:
311              failure_messages.append(
312                  f"- Failed to match overload `{overload}`: {fail_reason}"
313              )
314      return (
315          None,
316          f"All overloads did not match the node `{node.format_node()}`.\n"
317          + "\n".join(failure_messages),
318      )

321  def _arg_has_complex_dtype(arg) -> bool:
322      """Check if the node has complex dtype recursively."""
323      if (
324          isinstance(arg, torch.fx.Node)
325          and "val" in arg.meta
326          and isinstance(arg.meta["val"], torch.Tensor)
327          and torch.is_complex(arg.meta["val"])
328      ):
329          return True
330      elif isinstance(arg, list):
331          return any(_arg_has_complex_dtype(item) for item in arg)
332      return False

335  def dispatch(
336      node: torch.fx.Node, registry: _registration.ONNXRegistry
337  ) -> tuple[Callable | None, str]:
338      """Dispatch a node to an ONNX function based on the node's target and the ONNX registry. 
339   
340      Args: 
341          node: The node to dispatch. 
342          registry: The ONNX registry to use for dispatching. 
343   
344      Returns: 
345          A tuple containing the matched ONNX function and a string describing the reason for failure or success. 
346      """
347      # TODO: Handle when node does not have a target
348      decomp_metas = registry.get_decomps(node.target)  # type: ignore[arg-type]
349      # Determine if the node has complex inputs.
350      is_complex = any(_arg_has_complex_dtype(arg) for arg in node.args) or any(
351          _arg_has_complex_dtype(arg) for arg in node.kwargs.values()
352      )
353      if is_complex:
354          decomp_metas = [decomp for decomp in decomp_metas if decomp.is_complex]
355          if not decomp_metas:
356              return None, "No decompositions registered for the complex-valued input"
357      else:
358          decomp_metas = [decomp for decomp in decomp_metas if not decomp.is_complex]
359          if not decomp_metas:
360              return None, "No decompositions registered for the real-valued input"
361  
362      if len(decomp_metas) == 1:
363          return (
364              decomp_metas[0].onnx_function,
365              "Fast path: Only one decomposition is defined",
366          )
367  
368      overload, message = get_matching_overload(node, decomp_metas)
369      return overload, message
