93       def _print_Function(self, expr):
94   
95           op = self.mapping.get(type(expr), None)
96           if op is None:
97               return super()._print_Basic(expr)
98           children = [self._print(arg) for arg in expr.args]
99           if len(children) == 1:
100              return "%s(%s)" % (
101                  self._module_format(op),
102                  children[0]
103              )
104          else:
105              return self._expand_fold_binary_op(op, children)

123      def _print_Transpose(self, expr):
124          if expr.arg.is_Matrix and expr.arg.shape[0] == expr.arg.shape[1]:
125              # For square matrices, we can use the .t() method
126              return "{}({}).t()".format("torch.transpose", self._print(expr.arg))
127          else:
128              # For non-square matrices or more general cases
129              # transpose first and second dimensions (typical matrix transpose)
130              return "{}.permute({})".format(
131                  self._print(expr.arg),
132                  ", ".join([str(i) for i in range(len(expr.arg.shape))])[::-1]
133              )

141      def _print_Derivative(self, expr):
142          # this version handles multi-variable and mixed partial derivatives. The tensorflow version does not.
143          variables = expr.variables
144          expr_arg = expr.expr
145  
146          # Handle multi-variable or repeated derivatives
147          if len(variables) > 1 or (
148              len(variables) == 1 and not isinstance(variables[0], tuple) and variables.count(variables[0]) > 1):
149              result = self._print(expr_arg)
150              var_groups = {}
151  
152              # Group variables by base symbol
153              for var in variables:
154                  if isinstance(var, tuple):
155                      base_var, order = var
156                      var_groups[base_var] = var_groups.get(base_var, 0) + order
157                  else:
158                      var_groups[var] = var_groups.get(var, 0) + 1
159  
160              # Apply gradients in sequence
161              for var, order in var_groups.items():
162                  for _ in range(order):
163                      result = "torch.autograd.grad({}, {}, create_graph=True)[0]".format(result, self._print(var))
164              return result
165  
166          # Handle single variable case
167          if len(variables) == 1:
168              variable = variables[0]
169              if isinstance(variable, tuple) and len(variable) == 2:
170                  base_var, order = variable
171                  if not isinstance(order, Integer): raise NotImplementedError("Only integer orders are supported")
172                  result = self._print(expr_arg)
173                  for _ in range(order):
174                      result = "torch.autograd.grad({}, {}, create_graph=True)[0]".format(result, self._print(base_var))
175                  return result
176              return "torch.autograd.grad({}, {})[0]".format(self._print(expr_arg), self._print(variable))
177  
178          return self._print(expr_arg)  # Empty variables case

180      def _print_Piecewise(self, expr):
181          from sympy import Piecewise
182          e, cond = expr.args[0].args
183          if len(expr.args) == 1:
184              return '{}({}, {}, {})'.format(
185                  self._module_format("torch.where"),
186                  self._print(cond),
187                  self._print(e),
188                  0)
189  
190          return '{}({}, {}, {})'.format(
191              self._module_format("torch.where"),
192              self._print(cond),
193              self._print(e),
194              self._print(Piecewise(*expr.args[1:])))

196      def _print_Pow(self, expr):
197          # XXX May raise error for
198          # int**float or int**complex or float**complex
199          base, exp = expr.args
200          if expr.exp == S.Half:
201              return "{}({})".format(
202                  self._module_format("torch.sqrt"), self._print(base))
203          return "{}({}, {})".format(
204              self._module_format("torch.pow"),
205              self._print(base), self._print(exp))

207      def _print_MatMul(self, expr):
208          # Separate matrix and scalar arguments
209          mat_args = [arg for arg in expr.args if isinstance(arg, MatrixExpr)]
210          args = [arg for arg in expr.args if arg not in mat_args]
211          # Handle scalar multipliers if present
212          if args:
213              return "%s*%s" % (
214                  self.parenthesize(Mul.fromiter(args), PRECEDENCE["Mul"]),
215                  self._expand_fold_binary_op("torch.matmul", mat_args)
216              )
217          else:
218              return self._expand_fold_binary_op("torch.matmul", mat_args)
219  
220      def _print_MatPow(self, expr):
221          return self._expand_fold_binary_op("torch.mm", [expr.base]*expr.exp)

223      def _print_MatrixBase(self, expr):
224          data = "[" + ", ".join(["[" + ", ".join([self._print(j) for j in i]) + "]" for i in expr.tolist()]) + "]"
225          params = [str(data)]
226          params.append(f"dtype={self.dtype}")
227          if self.requires_grad:
228              params.append("requires_grad=True")
229  
230          return "{}({})".format(
231              self._module_format("torch.tensor"),
232              ", ".join(params)
233          )

241      def _print_Identity(self, expr):
242          if all(dim.is_Integer for dim in expr.shape):
243              return "{}({})".format(
244                  self._module_format("torch.eye"),
245                  self._print(expr.shape[0])
246              )
247          else:
248              # For symbolic dimensions, fall back to a more general approach
249              return "{}({}, {})".format(
250                  self._module_format("torch.eye"),
251                  self._print(expr.shape[0]),
252                  self._print(expr.shape[1])
253              )

295  def torch_code(expr, requires_grad=False, dtype="torch.float64", **settings):
296      printer = TorchPrinter(settings={'requires_grad': requires_grad, 'dtype': dtype})
297      return printer.doprint(expr, **settings)
