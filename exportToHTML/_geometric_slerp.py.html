13   def _geometric_slerp(start, end, t):
14       # create an orthogonal basis using QR decomposition
15       basis = np.vstack([start, end])
16       Q, R = np.linalg.qr(basis.T)
17       signs = 2 * (np.diag(R) >= 0) - 1
18       Q = Q.T * signs.T[:, np.newaxis]
19       R = R.T * signs.T[:, np.newaxis]
20   
21       # calculate the angle between `start` and `end`
22       c = np.dot(start, end)
23       s = np.linalg.det(R)
24       omega = np.arctan2(s, c)
25   
26       # interpolate
27       start, end = Q
28       s = np.sin(t * omega)
29       c = np.cos(t * omega)
30       return start * c[:, np.newaxis] + end * s[:, np.newaxis]

33   def geometric_slerp(
34       start: "npt.ArrayLike",
35       end: "npt.ArrayLike",
36       t: "npt.ArrayLike",
37       tol: float = 1e-7,
38   ) -> np.ndarray:
39       """ 
40       Geometric spherical linear interpolation. 
41    
42       The interpolation occurs along a unit-radius 
43       great circle arc in arbitrary dimensional space. 
44    
45       Parameters 
46       ---------- 
47       start : (n_dimensions, ) array-like 
48           Single n-dimensional input coordinate in a 1-D array-like 
49           object. `n` must be greater than 1. 
50       end : (n_dimensions, ) array-like 
51           Single n-dimensional input coordinate in a 1-D array-like 
52           object. `n` must be greater than 1. 
53       t : float or (n_points,) 1D array-like 
54           A float or 1D array-like of doubles representing interpolation 
55           parameters, with values required in the inclusive interval 
56           between 0 and 1. A common approach is to generate the array 
57           with ``np.linspace(0, 1, n_pts)`` for linearly spaced points. 
58           Ascending, descending, and scrambled orders are permitted. 
59       tol : float 
60           The absolute tolerance for determining if the start and end 
61           coordinates are antipodes. 
62    
63       Returns 
64       ------- 
65       result : (t.size, D) 
66           An array of doubles containing the interpolated 
67           spherical path and including start and 
68           end when 0 and 1 t are used. The 
69           interpolated values should correspond to the 
70           same sort order provided in the t array. The result 
71           may be 1-dimensional if ``t`` is a float. 
72    
73       Raises 
74       ------ 
75       ValueError 
76           If ``start`` and ``end`` are antipodes, not on the 
77           unit n-sphere, or for a variety of degenerate conditions. 
78    
79       See Also 
80       -------- 
81       scipy.spatial.transform.Slerp : 3-D Slerp that works with quaternions 
82    
83       Notes 
84       ----- 
85       The implementation is based on the mathematical formula provided in [1]_, 
86       and the first known presentation of this algorithm, derived from study of 
87       4-D geometry, is credited to Glenn Davis in a footnote of the original 
88       quaternion Slerp publication by Ken Shoemake [2]_. 
89    
90       .. versionadded:: 1.5.0 
91    
92       References 
93       ---------- 
94       .. [1] https://en.wikipedia.org/wiki/Slerp#Geometric_Slerp 
95       .. [2] Ken Shoemake (1985) Animating rotation with quaternion curves. 
96              ACM SIGGRAPH Computer Graphics, 19(3): 245-254. 
97    
98       Examples 
99       -------- 
100      Interpolate four linearly-spaced values on the circumference of 
101      a circle spanning 90 degrees: 
102   
103      >>> import numpy as np 
104      >>> from scipy.spatial import geometric_slerp 
105      >>> import matplotlib.pyplot as plt 
106      >>> fig = plt.figure() 
107      >>> ax = fig.add_subplot(111) 
108      >>> start = np.array([1, 0]) 
109      >>> end = np.array([0, 1]) 
110      >>> t_vals = np.linspace(0, 1, 4) 
111      >>> result = geometric_slerp(start, 
112      ...                          end, 
113      ...                          t_vals) 
114   
115      The interpolated results should be at 30 degree intervals 
116      recognizable on the unit circle: 
117   
118      >>> ax.scatter(result[...,0], result[...,1], c='k') 
119      >>> circle = plt.Circle((0, 0), 1, color='grey') 
120      >>> ax.add_artist(circle) 
121      >>> ax.set_aspect('equal') 
122      >>> plt.show() 
123   
124      Attempting to interpolate between antipodes on a circle is 
125      ambiguous because there are two possible paths, and on a 
126      sphere there are infinite possible paths on the geodesic surface. 
127      Nonetheless, one of the ambiguous paths is returned along 
128      with a warning: 
129   
130      >>> opposite_pole = np.array([-1, 0]) 
131      >>> with np.testing.suppress_warnings() as sup: 
132      ...     sup.filter(UserWarning) 
133      ...     geometric_slerp(start, 
134      ...                     opposite_pole, 
135      ...                     t_vals) 
136      array([[ 1.00000000e+00,  0.00000000e+00], 
137             [ 5.00000000e-01,  8.66025404e-01], 
138             [-5.00000000e-01,  8.66025404e-01], 
139             [-1.00000000e+00,  1.22464680e-16]]) 
140   
141      Extend the original example to a sphere and plot interpolation 
142      points in 3D: 
143   
144      >>> from mpl_toolkits.mplot3d import proj3d 
145      >>> fig = plt.figure() 
146      >>> ax = fig.add_subplot(111, projection='3d') 
147   
148      Plot the unit sphere for reference (optional): 
149   
150      >>> u = np.linspace(0, 2 * np.pi, 100) 
151      >>> v = np.linspace(0, np.pi, 100) 
152      >>> x = np.outer(np.cos(u), np.sin(v)) 
153      >>> y = np.outer(np.sin(u), np.sin(v)) 
154      >>> z = np.outer(np.ones(np.size(u)), np.cos(v)) 
155      >>> ax.plot_surface(x, y, z, color='y', alpha=0.1) 
156   
157      Interpolating over a larger number of points 
158      may provide the appearance of a smooth curve on 
159      the surface of the sphere, which is also useful 
160      for discretized integration calculations on a 
161      sphere surface: 
162   
163      >>> start = np.array([1, 0, 0]) 
164      >>> end = np.array([0, 0, 1]) 
165      >>> t_vals = np.linspace(0, 1, 200) 
166      >>> result = geometric_slerp(start, 
167      ...                          end, 
168      ...                          t_vals) 
169      >>> ax.plot(result[...,0], 
170      ...         result[...,1], 
171      ...         result[...,2], 
172      ...         c='k') 
173      >>> plt.show() 
174      """
175  
176      start = np.asarray(start, dtype=np.float64)
177      end = np.asarray(end, dtype=np.float64)
178      t = np.asarray(t)
179  
180      if t.ndim > 1:
181          raise ValueError("The interpolation parameter "
182                           "value must be one dimensional.")
183  
184      if start.ndim != 1 or end.ndim != 1:
185          raise ValueError("Start and end coordinates "
186                           "must be one-dimensional")
187  
188      if start.size != end.size:
189          raise ValueError("The dimensions of start and "
190                           "end must match (have same size)")
191  
192      if start.size < 2 or end.size < 2:
193          raise ValueError("The start and end coordinates must "
194                           "both be in at least two-dimensional "
195                           "space")
196  
197      if np.array_equal(start, end):
198          return np.linspace(start, start, t.size)
199  
200      # for points that violate equation for n-sphere
201      for coord in [start, end]:
202          if not np.allclose(np.linalg.norm(coord), 1.0,
203                             rtol=1e-9,
204                             atol=0):
205              raise ValueError("start and end are not"
206                               " on a unit n-sphere")
207  
208      if not isinstance(tol, float):
209          raise ValueError("tol must be a float")
210      else:
211          tol = np.fabs(tol)
212  
213      coord_dist = euclidean(start, end)
214  
215      # diameter of 2 within tolerance means antipodes, which is a problem
216      # for all unit n-spheres (even the 0-sphere would have an ambiguous path)
217      if np.allclose(coord_dist, 2.0, rtol=0, atol=tol):
218          warnings.warn("start and end are antipodes "
219                        "using the specified tolerance; "
220                        "this may cause ambiguous slerp paths",
221                        stacklevel=2)
222  
223      t = np.asarray(t, dtype=np.float64)
224  
225      if t.size == 0:
226          return np.empty((0, start.size))
227  
228      if t.min() < 0 or t.max() > 1:
229          raise ValueError("interpolation parameter must be in [0, 1]")
230  
231      if t.ndim == 0:
232          return _geometric_slerp(start,
233                                  end,
234                                  np.atleast_1d(t)).ravel()
235      else:
236          return _geometric_slerp(start,
237                                  end,
238                                  t)
