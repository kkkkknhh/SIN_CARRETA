11   def test_heap_global_selection():
12       """Test de la funcionalidad de selecci√≥n global con heap"""
13       print("üß™ Probando funcionalidad de heap para selecci√≥n global top-k")
14       
15       # Simular datos de segmentos con scores
16       segmentos_simulados = [
17           {"texto": "Segmento 1", "score": 0.9, "pagina": 1},
18           {"texto": "Segmento 2", "score": 0.8, "pagina": 2}, 
19           {"texto": "Segmento 3", "score": 0.95, "pagina": 1},
20           {"texto": "Segmento 4", "score": 0.7, "pagina": 3},
21           {"texto": "Segmento 5", "score": 0.85, "pagina": 2},
22           {"texto": "Segmento 6", "score": 0.6, "pagina": 4},
23           {"texto": "Segmento 7", "score": 0.92, "pagina": 1},
24           {"texto": "Segmento 8", "score": 0.75, "pagina": 3},
25       ]
26       
27       # Configurar par√°metros
28       max_segmentos = 5
29       heap_global = []
30       
31       print(f"üìä Procesando {len(segmentos_simulados)} segmentos para seleccionar top-{max_segmentos}")
32       
33       # Simular el procesamiento con heap (usando min-heap)
34       for idx, seg in enumerate(segmentos_simulados):
35           score = seg["score"]
36           
37           datos_segmento = {
38               "texto": seg["texto"],
39               "pagina": seg["pagina"],
40               "similitud_semantica": score,
41               "score_final": score,
42               "hash_segmento": hashlib.md5(seg["texto"].encode('utf-8')).hexdigest()[:8],
43               "timestamp_extraccion": datetime.now().isoformat()
44           }
45           
46           # Usar heap para mantener top-k global eficientemente (min-heap con scores directos)
47           if len(heap_global) < max_segmentos:
48               # Heap no est√° lleno, agregar directamente
49               heapq.heappush(heap_global, (score, idx, datos_segmento))
50               print(f"  ‚ûï Agregado: {seg['texto']} (score: {score})")
51           elif score > heap_global[0][0]:  # score > min_score_in_heap
52               # Reemplazar el peor elemento con este mejor elemento
53               old_score, old_idx, old_datos = heapq.heappushpop(heap_global, (score, idx, datos_segmento))
54               print(f"  üîÑ Reemplazado: {old_datos['texto']} (score: {old_score:.2f}) por {seg['texto']} (score: {score})")
55           else:
56               print(f"  ‚ùå Rechazado: {seg['texto']} (score: {score}) - score demasiado bajo (min en heap: {heap_global[0][0]:.2f})")
57       
58       # Extraer los resultados finales del heap y ordenar por score descendente
59       resultados_finales = []
60       while heap_global:
61           score, _, datos_segmento = heapq.heappop(heap_global)
62           resultados_finales.append(datos_segmento)
63       
64       # Ordenar por score final descendente para tener los mejores primero
65       resultados_finales.sort(key=lambda x: x['score_final'], reverse=True)
66       
67       print(f"\n‚úÖ Selecci√≥n completada: {len(resultados_finales)} segmentos seleccionados")
68       print("\nüèÜ Top segmentos seleccionados (ordenados por score):")
69       for i, seg in enumerate(resultados_finales, 1):
70           print(f"  {i}. {seg['texto']} (score: {seg['score_final']:.2f}, p√°g: {seg['pagina']})")
71       
72       # Verificar que efectivamente tenemos los top-k
73       scores_originales = sorted([s["score"] for s in segmentos_simulados], reverse=True)
74       top_k_esperados = scores_originales[:max_segmentos]
75       scores_obtenidos = [s["score_final"] for s in resultados_finales]
76       
77       print(f"\nüîç Verificaci√≥n:")
78       print(f"  Top-{max_segmentos} esperados: {top_k_esperados}")
79       print(f"  Top-{max_segmentos} obtenidos: {scores_obtenidos}")
80       
81       if set(top_k_esperados) == set(scores_obtenidos):
82           print("  ‚úÖ ¬°Selecci√≥n correcta!")
83           return True
84       else:
85           print("  ‚ùå Error en la selecci√≥n")
86           return False

88   def test_batch_processing():

91       
92       # Simular queries

100          batch_queries = queries[batch_start:batch_start + batch_size]
101          print(f"  üì¶ Batch {batches_procesados + 1}: procesando queries {batch_start + 1}-{min(batch_start + batch_size, len(queries))}")
102          print(f"     Queries en este batch: {batch_queries}")
103          batches_procesados += 1
104      
105      print(f"‚úÖ Procesados {batches_procesados} batches en total")
106      
107      # Verificar que se procesaron todas las queries
108      queries_procesadas = batches_procesados * batch_size - max(0, batches_procesados * batch_size - len(queries))
109      if queries_procesadas == len(queries):
110          print("  ‚úÖ Todas las queries fueron procesadas correctamente")
111          return True
112      else:
113          print(f"  ‚ùå Error: se procesaron {queries_procesadas} queries de {len(queries)} esperadas")
