37   def roots_linear(f):
38       """Returns a list of roots of a linear polynomial."""
39       r = -f.nth(0)/f.nth(1)
40       dom = f.get_domain()
41   
42       if not dom.is_Numerical:
43           if dom.is_Composite:
44               r = factor(r)
45           else:
46               from sympy.simplify.simplify import simplify
47               r = simplify(r)
48   
49       return [r]

52   def roots_quadratic(f):
53       """Returns a list of roots of a quadratic polynomial. If the domain is ZZ 
54       then the roots will be sorted with negatives coming before positives. 
55       The ordering will be the same for any numerical coefficients as long as 
56       the assumptions tested are correct, otherwise the ordering will not be 
57       sorted (but will be canonical). 
58       """
59   
60       a, b, c = f.all_coeffs()
61       dom = f.get_domain()
62   
63       def _sqrt(d):
64           # remove squares from square root since both will be represented
65           # in the results; a similar thing is happening in roots() but
66           # must be duplicated here because not all quadratics are binomials
67           co = []
68           other = []
69           for di in Mul.make_args(d):
70               if di.is_Pow and di.exp.is_Integer and di.exp % 2 == 0:
71                   co.append(Pow(di.base, di.exp//2))
72               else:
73                   other.append(di)
74           if co:
75               d = Mul(*other)
76               co = Mul(*co)
77               return co*sqrt(d)
78           return sqrt(d)
79   
80       def _simplify(expr):
81           if dom.is_Composite:
82               return factor(expr)
83           else:
84               from sympy.simplify.simplify import simplify
85               return simplify(expr)
86   
87       if c is S.Zero:
88           r0, r1 = S.Zero, -b/a
89   
90           if not dom.is_Numerical:
91               r1 = _simplify(r1)
92           elif r1.is_negative:
93               r0, r1 = r1, r0
94       elif b is S.Zero:
95           r = -c/a
96           if not dom.is_Numerical:
97               r = _simplify(r)
98   
99           R = _sqrt(r)
100          r0 = -R
101          r1 = R
102      else:
103          d = b**2 - 4*a*c
104          A = 2*a
105          B = -b/A
106  
107          if not dom.is_Numerical:
108              d = _simplify(d)
109              B = _simplify(B)
110  
111          D = factor_terms(_sqrt(d)/A)
112          r0 = B - D
113          r1 = B + D
114          if a.is_negative:
115              r0, r1 = r1, r0
116          elif not dom.is_Numerical:
117              r0, r1 = [expand_2arg(i) for i in (r0, r1)]
118  
119      return [r0, r1]

122  def roots_cubic(f, trig=False):
123      """Returns a list of roots of a cubic polynomial. 
124   
125      References 
126      ========== 
127      [1] https://en.wikipedia.org/wiki/Cubic_function, General formula for roots, 
128      (accessed November 17, 2014). 
129      """
130      if trig:
131          a, b, c, d = f.all_coeffs()
132          p = (3*a*c - b**2)/(3*a**2)
133          q = (2*b**3 - 9*a*b*c + 27*a**2*d)/(27*a**3)
134          D = 18*a*b*c*d - 4*b**3*d + b**2*c**2 - 4*a*c**3 - 27*a**2*d**2
135          if (D > 0) == True:
136              rv = []
137              for k in range(3):
138                  rv.append(2*sqrt(-p/3)*cos(acos(q/p*sqrt(-3/p)*Rational(3, 2))/3 - k*pi*Rational(2, 3)))
139              return [i - b/3/a for i in rv]
140  
141      # a*x**3 + b*x**2 + c*x + d -> x**3 + a*x**2 + b*x + c
142      _, a, b, c = f.monic().all_coeffs()
143  
144      if c is S.Zero:
145          x1, x2 = roots([1, a, b], multiple=True)
146          return [x1, S.Zero, x2]
147  
148      # x**3 + a*x**2 + b*x + c -> u**3 + p*u + q
149      p = b - a**2/3
150      q = c - a*b/3 + 2*a**3/27
151  
152      pon3 = p/3
153      aon3 = a/3
154  
155      u1 = None
156      if p is S.Zero:
157          if q is S.Zero:
158              return [-aon3]*3
159          u1 = -root(q, 3) if q.is_positive else root(-q, 3)
160      elif q is S.Zero:
161          y1, y2 = roots([1, 0, p], multiple=True)
162          return [tmp - aon3 for tmp in [y1, S.Zero, y2]]
163      elif q.is_real and q.is_negative:
164          u1 = -root(-q/2 + sqrt(q**2/4 + pon3**3), 3)
165  
166      coeff = I*sqrt(3)/2
167      if u1 is None:
168          u1 = S.One
169          u2 = Rational(-1, 2) + coeff
170          u3 = Rational(-1, 2) - coeff
171          b, c, d = a, b, c  # a, b, c, d = S.One, a, b, c
172          D0 = b**2 - 3*c  # b**2 - 3*a*c
173          D1 = 2*b**3 - 9*b*c + 27*d  # 2*b**3 - 9*a*b*c + 27*a**2*d
174          C = root((D1 + sqrt(D1**2 - 4*D0**3))/2, 3)
175          return [-(b + uk*C + D0/C/uk)/3 for uk in [u1, u2, u3]]  # -(b + uk*C + D0/C/uk)/3/a
176  
177      u2 = u1*(Rational(-1, 2) + coeff)
178      u3 = u1*(Rational(-1, 2) - coeff)
179  
180      if p is S.Zero:
181          return [u1 - aon3, u2 - aon3, u3 - aon3]
182  
183      soln = [
184          -u1 + pon3/u1 - aon3,
185          -u2 + pon3/u2 - aon3,
186          -u3 + pon3/u3 - aon3
187      ]
188  
189      return soln

191  def _roots_quartic_euler(p, q, r, a):
192      """ 
193      Descartes-Euler solution of the quartic equation 
194   
195      Parameters 
196      ========== 
197   
198      p, q, r: coefficients of ``x**4 + p*x**2 + q*x + r`` 
199      a: shift of the roots 
200   
201      Notes 
202      ===== 
203   
204      This is a helper function for ``roots_quartic``. 
205   
206      Look for solutions of the form :: 
207   
208        ``x1 = sqrt(R) - sqrt(A + B*sqrt(R))`` 
209        ``x2 = -sqrt(R) - sqrt(A - B*sqrt(R))`` 
210        ``x3 = -sqrt(R) + sqrt(A - B*sqrt(R))`` 
211        ``x4 = sqrt(R) + sqrt(A + B*sqrt(R))`` 
212   
213      To satisfy the quartic equation one must have 
214      ``p = -2*(R + A); q = -4*B*R; r = (R - A)**2 - B**2*R`` 
215      so that ``R`` must satisfy the Descartes-Euler resolvent equation 
216      ``64*R**3 + 32*p*R**2 + (4*p**2 - 16*r)*R - q**2 = 0`` 
217   
218      If the resolvent does not have a rational solution, return None; 
219      in that case it is likely that the Ferrari method gives a simpler 
220      solution. 
221   
222      Examples 
223      ======== 
224   
225      >>> from sympy import S 
226      >>> from sympy.polys.polyroots import _roots_quartic_euler 
227      >>> p, q, r = -S(64)/5, -S(512)/125, -S(1024)/3125 
228      >>> _roots_quartic_euler(p, q, r, S(0))[0] 
229      -sqrt(32*sqrt(5)/125 + 16/5) + 4*sqrt(5)/5 
230      """
231      # solve the resolvent equation
232      x = Dummy('x')
233      eq = 64*x**3 + 32*p*x**2 + (4*p**2 - 16*r)*x - q**2
234      xsols = list(roots(Poly(eq, x), cubics=False).keys())
235      xsols = [sol for sol in xsols if sol.is_rational and sol.is_nonzero]
236      if not xsols:
237          return None
238      R = max(xsols)
239      c1 = sqrt(R)
240      B = -q*c1/(4*R)
241      A = -R - p/2
242      c2 = sqrt(A + B)
243      c3 = sqrt(A - B)
244      return [c1 - c2 - a, -c1 - c3 - a, -c1 + c3 - a, c1 + c2 - a]

247  def roots_quartic(f):
248      r""" 
249      Returns a list of roots of a quartic polynomial. 
250   
251      There are many references for solving quartic expressions available [1-5]. 
252      This reviewer has found that many of them require one to select from among 
253      2 or more possible sets of solutions and that some solutions work when one 
254      is searching for real roots but do not work when searching for complex roots 
255      (though this is not always stated clearly). The following routine has been 
256      tested and found to be correct for 0, 2 or 4 complex roots. 
257   
258      The quasisymmetric case solution [6] looks for quartics that have the form 
259      `x**4 + A*x**3 + B*x**2 + C*x + D = 0` where `(C/A)**2 = D`. 
260   
261      Although no general solution that is always applicable for all 
262      coefficients is known to this reviewer, certain conditions are tested 
263      to determine the simplest 4 expressions that can be returned: 
264   
265        1) `f = c + a*(a**2/8 - b/2) == 0` 
266        2) `g = d - a*(a*(3*a**2/256 - b/16) + c/4) = 0` 
267        3) if `f != 0` and `g != 0` and `p = -d + a*c/4 - b**2/12` then 
268          a) `p == 0` 
269          b) `p != 0` 
270   
271      Examples 
272      ======== 
273   
274          >>> from sympy import Poly 
275          >>> from sympy.polys.polyroots import roots_quartic 
276   
277          >>> r = roots_quartic(Poly('x**4-6*x**3+17*x**2-26*x+20')) 
278   
279          >>> # 4 complex roots: 1+-I*sqrt(3), 2+-I 
280          >>> sorted(str(tmp.evalf(n=2)) for tmp in r) 
281          ['1.0 + 1.7*I', '1.0 - 1.7*I', '2.0 + 1.0*I', '2.0 - 1.0*I'] 
282   
283      References 
284      ========== 
285   
286      1. http://mathforum.org/dr.math/faq/faq.cubic.equations.html 
287      2. https://en.wikipedia.org/wiki/Quartic_function#Summary_of_Ferrari.27s_method 
288      3. https://planetmath.org/encyclopedia/GaloisTheoreticDerivationOfTheQuarticFormula.html 
289      4. https://people.bath.ac.uk/masjhd/JHD-CA.pdf 
290      5. http://www.albmath.org/files/Math_5713.pdf 
291      6. https://web.archive.org/web/20171002081448/http://www.statemaster.com/encyclopedia/Quartic-equation 
292      7. https://eqworld.ipmnet.ru/en/solutions/ae/ae0108.pdf 
293      """
294      _, a, b, c, d = f.monic().all_coeffs()
295  
296      if not d:
297          return [S.Zero] + roots([1, a, b, c], multiple=True)
298      elif (c/a)**2 == d:
299          x, m = f.gen, c/a
300  
301          g = Poly(x**2 + a*x + b - 2*m, x)
302  
303          z1, z2 = roots_quadratic(g)
304  
305          h1 = Poly(x**2 - z1*x + m, x)
306          h2 = Poly(x**2 - z2*x + m, x)
307  
308          r1 = roots_quadratic(h1)
309          r2 = roots_quadratic(h2)
310  
311          return r1 + r2
312      else:
313          a2 = a**2
314          e = b - 3*a2/8
315          f = _mexpand(c + a*(a2/8 - b/2))
316          aon4 = a/4
317          g = _mexpand(d - aon4*(a*(3*a2/64 - b/4) + c))
318  
319          if f.is_zero:
320              y1, y2 = [sqrt(tmp) for tmp in
321                        roots([1, e, g], multiple=True)]
322              return [tmp - aon4 for tmp in [-y1, -y2, y1, y2]]
323          if g.is_zero:
324              y = [S.Zero] + roots([1, 0, e, f], multiple=True)
325              return [tmp - aon4 for tmp in y]
326          else:
327              # Descartes-Euler method, see [7]
328              sols = _roots_quartic_euler(e, f, g, aon4)
329              if sols:
330                  return sols
331              # Ferrari method, see [1, 2]
332              p = -e**2/12 - g
333              q = -e**3/108 + e*g/3 - f**2/8
334              TH = Rational(1, 3)
335  
336              def _ans(y):
337                  w = sqrt(e + 2*y)
338                  arg1 = 3*e + 2*y
339                  arg2 = 2*f/w
340                  ans = []
341                  for s in [-1, 1]:
342                      root = sqrt(-(arg1 + s*arg2))
343                      for t in [-1, 1]:
344                          ans.append((s*w - t*root)/2 - aon4)
345                  return ans
346  
347              # whether a Piecewise is returned or not
348              # depends on knowing p, so try to put
349              # in a simple form
350              p = _mexpand(p)
351  
352  
353              # p == 0 case
354              y1 = e*Rational(-5, 6) - q**TH
355              if p.is_zero:
356                  return _ans(y1)
357  
358              # if p != 0 then u below is not 0
359              root = sqrt(q**2/4 + p**3/27)
360              r = -q/2 + root  # or -q/2 - root
361              u = r**TH  # primary root of solve(x**3 - r, x)
362              y2 = e*Rational(-5, 6) + u - p/u/3
363              if fuzzy_not(p.is_zero):
364                  return _ans(y2)
365  
366              # sort it out once they know the values of the coefficients
367              return [Piecewise((a1, Eq(p, 0)), (a2, True))
368                      for a1, a2 in zip(_ans(y1), _ans(y2))]

371  def roots_binomial(f):
372      """Returns a list of roots of a binomial polynomial. If the domain is ZZ 
373      then the roots will be sorted with negatives coming before positives. 
374      The ordering will be the same for any numerical coefficients as long as 
375      the assumptions tested are correct, otherwise the ordering will not be 
376      sorted (but will be canonical). 
377      """
378      n = f.degree()
379  
380      a, b = f.nth(n), f.nth(0)
381      base = -cancel(b/a)
382      alpha = root(base, n)
383  
384      if alpha.is_number:
385          alpha = alpha.expand(complex=True)
386  
387      # define some parameters that will allow us to order the roots.
388      # If the domain is ZZ this is guaranteed to return roots sorted
389      # with reals before non-real roots and non-real sorted according
390      # to real part and imaginary part, e.g. -1, 1, -1 + I, 2 - I
391      neg = base.is_negative
392      even = n % 2 == 0
393      if neg:
394          if even == True and (base + 1).is_positive:
395              big = True
396          else:
397              big = False
398  
399      # get the indices in the right order so the computed
400      # roots will be sorted when the domain is ZZ
401      ks = []
402      imax = n//2
403      if even:
404          ks.append(imax)
405          imax -= 1
406      if not neg:
407          ks.append(0)
408      for i in range(imax, 0, -1):
409          if neg:
410              ks.extend([i, -i])
411          else:
412              ks.extend([-i, i])
413      if neg:
414          ks.append(0)
415          if big:
416              for i in range(0, len(ks), 2):
417                  pair = ks[i: i + 2]
418                  pair = list(reversed(pair))
419  
420      # compute the roots
421      roots, d = [], 2*I*pi/n
422      for k in ks:
423          zeta = exp(k*d).expand(complex=True)
424          roots.append((alpha*zeta).expand(power_base=False))
425  
426      return roots

429  def _inv_totient_estimate(m):
430      """ 
431      Find ``(L, U)`` such that ``L <= phi^-1(m) <= U``. 
432   
433      Examples 
434      ======== 
435   
436      >>> from sympy.polys.polyroots import _inv_totient_estimate 
437   
438      >>> _inv_totient_estimate(192) 
439      (192, 840) 
440      >>> _inv_totient_estimate(400) 
441      (400, 1750) 
442   
443      """
444      primes = [ d + 1 for d in divisors(m) if isprime(d + 1) ]
445  
446      a, b = 1, 1
447  
448      for p in primes:
449          a *= p
450          b *= p - 1
451  
452      L = m
453      U = int(math.ceil(m*(float(a)/b)))
454  
455      P = p = 2
456      primes = []
457  
458      while P <= U:
459          p = nextprime(p)
460          primes.append(p)
461          P *= p
462  
463      P //= p
464      b = 1
465  
466      for p in primes[:-1]:
467          b *= p - 1
468  
469      U = int(math.ceil(m*(float(P)/b)))
470  
471      return L, U

474  def roots_cyclotomic(f, factor=False):
475      """Compute roots of cyclotomic polynomials. """
476      L, U = _inv_totient_estimate(f.degree())
477  
478      for n in range(L, U + 1):
479          g = cyclotomic_poly(n, f.gen, polys=True)
480  
481          if f.expr == g.expr:
482              break
483      else:  # pragma: no cover
484          raise RuntimeError("failed to find index of a cyclotomic polynomial")
485  
486      roots = []
487  
488      if not factor:
489          # get the indices in the right order so the computed
490          # roots will be sorted
491          h = n//2
492          ks = [i for i in range(1, n + 1) if igcd(i, n) == 1]
493          ks.sort(key=lambda x: (x, -1) if x <= h else (abs(x - n), 1))
494          d = 2*I*pi/n
495          for k in reversed(ks):
496              roots.append(exp(k*d).expand(complex=True))
497      else:
498          g = Poly(f, extension=root(-1, n))
499  
500          for h, _ in ordered(g.factor_list()[1]):
501              roots.append(-h.TC())
502  
503      return roots

506  def roots_quintic(f):
507      """ 
508      Calculate exact roots of a solvable irreducible quintic with rational coefficients. 
509      Return an empty list if the quintic is reducible or not solvable. 
510      """
511      result = []
512  
513      coeff_5, coeff_4, p_, q_, r_, s_ = f.all_coeffs()
514  
515      if not all(coeff.is_Rational for coeff in (coeff_5, coeff_4, p_, q_, r_, s_)):
516          return result
517  
518      if coeff_5 != 1:
519          f = Poly(f / coeff_5)
520          _, coeff_4, p_, q_, r_, s_ = f.all_coeffs()
521  
522      # Cancel coeff_4 to form x^5 + px^3 + qx^2 + rx + s
523      if coeff_4:
524          p = p_ - 2*coeff_4*coeff_4/5
525          q = q_ - 3*coeff_4*p_/5 + 4*coeff_4**3/25
526          r = r_ - 2*coeff_4*q_/5 + 3*coeff_4**2*p_/25 - 3*coeff_4**4/125
527          s = s_ - coeff_4*r_/5 + coeff_4**2*q_/25 - coeff_4**3*p_/125 + 4*coeff_4**5/3125
528          x = f.gen
529          f = Poly(x**5 + p*x**3 + q*x**2 + r*x + s)
530      else:
531          p, q, r, s = p_, q_, r_, s_
532  
533      quintic = PolyQuintic(f)
534  
535      # Eqn standardized. Algo for solving starts here
536      if not f.is_irreducible:
537          return result
538      f20 = quintic.f20
539      # Check if f20 has linear factors over domain Z
540      if f20.is_irreducible:
541          return result
542      # Now, we know that f is solvable
543      for _factor in f20.factor_list()[1]:
544          if _factor[0].is_linear:
545              theta = _factor[0].root(0)
546              break
547      d = discriminant(f)
548      delta = sqrt(d)
549      # zeta = a fifth root of unity
550      zeta1, zeta2, zeta3, zeta4 = quintic.zeta
551      T = quintic.T(theta, d)
552      tol = S(1e-10)
553      alpha = T[1] + T[2]*delta
554      alpha_bar = T[1] - T[2]*delta
555      beta = T[3] + T[4]*delta
556      beta_bar = T[3] - T[4]*delta
557  
558      disc = alpha**2 - 4*beta
559      disc_bar = alpha_bar**2 - 4*beta_bar
560  
561      l0 = quintic.l0(theta)
562      Stwo = S(2)
563      l1 = _quintic_simplify((-alpha + sqrt(disc)) / Stwo)
564      l4 = _quintic_simplify((-alpha - sqrt(disc)) / Stwo)
565  
566      l2 = _quintic_simplify((-alpha_bar + sqrt(disc_bar)) / Stwo)
567      l3 = _quintic_simplify((-alpha_bar - sqrt(disc_bar)) / Stwo)
568  
569      order = quintic.order(theta, d)
570      test = (order*delta.n()) - ( (l1.n() - l4.n())*(l2.n() - l3.n()) )
571      # Comparing floats
572      if not comp(test, 0, tol):
573          l2, l3 = l3, l2
574  
575      # Now we have correct order of l's
576      R1 = l0 + l1*zeta1 + l2*zeta2 + l3*zeta3 + l4*zeta4
577      R2 = l0 + l3*zeta1 + l1*zeta2 + l4*zeta3 + l2*zeta4
578      R3 = l0 + l2*zeta1 + l4*zeta2 + l1*zeta3 + l3*zeta4
579      R4 = l0 + l4*zeta1 + l3*zeta2 + l2*zeta3 + l1*zeta4
580  
581      Res = [None, [None]*5, [None]*5, [None]*5, [None]*5]
582      Res_n = [None, [None]*5, [None]*5, [None]*5, [None]*5]
583  
584      # Simplifying improves performance a lot for exact expressions
585      R1 = _quintic_simplify(R1)
586      R2 = _quintic_simplify(R2)
587      R3 = _quintic_simplify(R3)
588      R4 = _quintic_simplify(R4)
589  
590      # hard-coded results for [factor(i) for i in _vsolve(x**5 - a - I*b, x)]
591      x0 = z**(S(1)/5)
592      x1 = sqrt(2)
593      x2 = sqrt(5)
594      x3 = sqrt(5 - x2)
595      x4 = I*x2
596      x5 = x4 + I
597      x6 = I*x0/4
598      x7 = x1*sqrt(x2 + 5)
599      sol = [x0, -x6*(x1*x3 - x5), x6*(x1*x3 + x5), -x6*(x4 + x7 - I), x6*(-x4 + x7 + I)]
600  
601      R1 = R1.as_real_imag()
602      R2 = R2.as_real_imag()
603      R3 = R3.as_real_imag()
604      R4 = R4.as_real_imag()
605  
606      for i, s in enumerate(sol):
607          Res[1][i] = _quintic_simplify(s.xreplace({z: R1[0] + I*R1[1]}))
608          Res[2][i] = _quintic_simplify(s.xreplace({z: R2[0] + I*R2[1]}))
609          Res[3][i] = _quintic_simplify(s.xreplace({z: R3[0] + I*R3[1]}))
610          Res[4][i] = _quintic_simplify(s.xreplace({z: R4[0] + I*R4[1]}))
611  
612      for i in range(1, 5):
613          for j in range(5):
614              Res_n[i][j] = Res[i][j].n()
615              Res[i][j] = _quintic_simplify(Res[i][j])
616      r1 = Res[1][0]
617      r1_n = Res_n[1][0]
618  
619      for i in range(5):
620          if comp(im(r1_n*Res_n[4][i]), 0, tol):
621              r4 = Res[4][i]
622              break
623  
624      # Now we have various Res values. Each will be a list of five
625      # values. We have to pick one r value from those five for each Res
626      u, v = quintic.uv(theta, d)
627      testplus = (u + v*delta*sqrt(5)).n()
628      testminus = (u - v*delta*sqrt(5)).n()
629  
630      # Evaluated numbers suffixed with _n
631      # We will use evaluated numbers for calculation. Much faster.
632      r4_n = r4.n()
633      r2 = r3 = None
634  
635      for i in range(5):
636          r2temp_n = Res_n[2][i]
637          for j in range(5):
638              # Again storing away the exact number and using
639              # evaluated numbers in computations
640              r3temp_n = Res_n[3][j]
641              if (comp((r1_n*r2temp_n**2 + r4_n*r3temp_n**2 - testplus).n(), 0, tol) and
642                  comp((r3temp_n*r1_n**2 + r2temp_n*r4_n**2 - testminus).n(), 0, tol)):
643                  r2 = Res[2][i]
644                  r3 = Res[3][j]
645                  break
646          if r2 is not None:
647              break
648      else:
649          return []  # fall back to normal solve
650  
651      # Now, we have r's so we can get roots
652      x1 = (r1 + r2 + r3 + r4)/5
653      x2 = (r1*zeta4 + r2*zeta3 + r3*zeta2 + r4*zeta1)/5
654      x3 = (r1*zeta3 + r2*zeta1 + r3*zeta4 + r4*zeta2)/5
655      x4 = (r1*zeta2 + r2*zeta4 + r3*zeta1 + r4*zeta3)/5
656      x5 = (r1*zeta1 + r2*zeta2 + r3*zeta3 + r4*zeta4)/5
657      result = [x1, x2, x3, x4, x5]
658  
659      # Now check if solutions are distinct
660  
661      saw = set()
662      for r in result:
663          r = r.n(2)
664          if r in saw:
665              # Roots were identical. Abort, return []
666              # and fall back to usual solve
667              return []
668          saw.add(r)
669  
670      # Restore to original equation where coeff_4 is nonzero
671      if coeff_4:
672          result = [x - coeff_4 / 5 for x in result]
673      return result

679      expr = cancel(expr)

683  def _integer_basis(poly):
684      """Compute coefficient basis for a polynomial over integers. 
685   
686      Returns the integer ``div`` such that substituting ``x = div*y`` 
687      ``p(x) = m*q(y)`` where the coefficients of ``q`` are smaller 
688      than those of ``p``. 
689   
690      For example ``x**5 + 512*x + 1024 = 0`` 
691      with ``div = 4`` becomes ``y**5 + 2*y + 1 = 0`` 
692   
693      Returns the integer ``div`` or ``None`` if there is no possible scaling. 
694   
695      Examples 
696      ======== 
697   
698      >>> from sympy.polys import Poly 
699      >>> from sympy.abc import x 
700      >>> from sympy.polys.polyroots import _integer_basis 
701      >>> p = Poly(x**5 + 512*x + 1024, x, domain='ZZ') 
702      >>> _integer_basis(p) 
703      4 
704      """
705      monoms, coeffs = list(zip(*poly.terms()))
706  
707      monoms, = list(zip(*monoms))
708      coeffs = list(map(abs, coeffs))
709  
710      if coeffs[0] < coeffs[-1]:
711          coeffs = list(reversed(coeffs))
712          n = monoms[0]
713          monoms = [n - i for i in reversed(monoms)]
714      else:
715          return None
716  
717      monoms = monoms[:-1]
718      coeffs = coeffs[:-1]
719  
720      # Special case for two-term polynominals
721      if len(monoms) == 1:
722          r = Pow(coeffs[0], S.One/monoms[0])
723          if r.is_Integer:
724              return int(r)
725          else:
726              return None
727  
728      divs = reversed(divisors(gcd_list(coeffs))[1:])
729  
730      try:
731          div = next(divs)
732      except StopIteration:
733          return None
734  
735      while True:
736          for monom, coeff in zip(monoms, coeffs):
737              if coeff % div**monom != 0:
738                  try:
739                      div = next(divs)
740                  except StopIteration:
741                      return None
742                  else:
743                      break
744          else:
745              return div

748  def preprocess_roots(poly):
749      """Try to get rid of symbolic coefficients from ``poly``. """
750      coeff = S.One
751  
752      poly_func = poly.func
753      try:
754          _, poly = poly.clear_denoms(convert=True)
755      except DomainError:
756          return coeff, poly
757  
758      poly = poly.primitive()[1]
759      poly = poly.retract()
760  
761      # TODO: This is fragile. Figure out how to make this independent of construct_domain().
762      if poly.get_domain().is_Poly and all(c.is_term for c in poly.rep.coeffs()):
763          poly = poly.inject()
764  
765          strips = list(zip(*poly.monoms()))
766          gens = list(poly.gens[1:])
767  
768          base, strips = strips[0], strips[1:]
769  
770          for gen, strip in zip(list(gens), strips):
771              reverse = False
772  
773              if strip[0] < strip[-1]:
774                  strip = reversed(strip)
775                  reverse = True
776  
777              ratio = None
778  
779              for a, b in zip(base, strip):
780                  if not a and not b:
781                      continue
782                  elif not a or not b:
783                      break
784                  elif b % a != 0:
785                      break
786                  else:
787                      _ratio = b // a
788  
789                      if ratio is None:
790                          ratio = _ratio
791                      elif ratio != _ratio:
792                          break
793              else:
794                  if reverse:
795                      ratio = -ratio
796  
797                  poly = poly.eval(gen, 1)
798                  coeff *= gen**(-ratio)
799                  gens.remove(gen)
800  
801          if gens:
802              poly = poly.eject(*gens)
803  
804      if poly.is_univariate and poly.get_domain().is_ZZ:
805          basis = _integer_basis(poly)
806  
807          if basis is not None:
808              n = poly.degree()
809  
810              def func(k, coeff):
811                  return coeff//basis**(n - k[0])
812  
813              poly = poly.termwise(func)
814              coeff *= basis
815  
816      if not isinstance(poly, poly_func):
817          poly = poly_func(poly)
818      return coeff, poly

822  def roots(f, *gens,
823          auto=True,
824          cubics=True,
825          trig=False,
826          quartics=True,
827          quintics=False,
828          multiple=False,
829          filter=None,
830          predicate=None,
831          strict=False,
832          **flags):
833      """ 
834      Computes symbolic roots of a univariate polynomial. 
835   
836      Given a univariate polynomial f with symbolic coefficients (or 
837      a list of the polynomial's coefficients), returns a dictionary 
838      with its roots and their multiplicities. 
839   
840      Only roots expressible via radicals will be returned.  To get 
841      a complete set of roots use RootOf class or numerical methods 
842      instead. By default cubic and quartic formulas are used in 
843      the algorithm. To disable them because of unreadable output 
844      set ``cubics=False`` or ``quartics=False`` respectively. If cubic 
845      roots are real but are expressed in terms of complex numbers 
846      (casus irreducibilis [1]) the ``trig`` flag can be set to True to 
847      have the solutions returned in terms of cosine and inverse cosine 
848      functions. 
849   
850      To get roots from a specific domain set the ``filter`` flag with 
851      one of the following specifiers: Z, Q, R, I, C. By default all 
852      roots are returned (this is equivalent to setting ``filter='C'``). 
853   
854      By default a dictionary is returned giving a compact result in 
855      case of multiple roots.  However to get a list containing all 
856      those roots set the ``multiple`` flag to True; the list will 
857      have identical roots appearing next to each other in the result. 
858      (For a given Poly, the all_roots method will give the roots in 
859      sorted numerical order.) 
860   
861      If the ``strict`` flag is True, ``UnsolvableFactorError`` will be 
862      raised if the roots found are known to be incomplete (because 
863      some roots are not expressible in radicals). 
864   
865      Examples 
866      ======== 
867   
868      >>> from sympy import Poly, roots, degree 
869      >>> from sympy.abc import x, y 
870   
871      >>> roots(x**2 - 1, x) 
872      {-1: 1, 1: 1} 
873   
874      >>> p = Poly(x**2-1, x) 
875      >>> roots(p) 
876      {-1: 1, 1: 1} 
877   
878      >>> p = Poly(x**2-y, x, y) 
879   
880      >>> roots(Poly(p, x)) 
881      {-sqrt(y): 1, sqrt(y): 1} 
882   
883      >>> roots(x**2 - y, x) 
884      {-sqrt(y): 1, sqrt(y): 1} 
885   
886      >>> roots([1, 0, -1]) 
887      {-1: 1, 1: 1} 
888   
889      ``roots`` will only return roots expressible in radicals. If 
890      the given polynomial has some or all of its roots inexpressible in 
891      radicals, the result of ``roots`` will be incomplete or empty 
892      respectively. 
893   
894      Example where result is incomplete: 
895   
896      >>> roots((x-1)*(x**5-x+1), x) 
897      {1: 1} 
898   
899      In this case, the polynomial has an unsolvable quintic factor 
900      whose roots cannot be expressed by radicals. The polynomial has a 
901      rational root (due to the factor `(x-1)`), which is returned since 
902      ``roots`` always finds all rational roots. 
903   
904      Example where result is empty: 
905   
906      >>> roots(x**7-3*x**2+1, x) 
907      {} 
908   
909      Here, the polynomial has no roots expressible in radicals, so 
910      ``roots`` returns an empty dictionary. 
911   
912      The result produced by ``roots`` is complete if and only if the 
913      sum of the multiplicity of each root is equal to the degree of 
914      the polynomial. If strict=True, UnsolvableFactorError will be 
915      raised if the result is incomplete. 
916   
917      The result can be be checked for completeness as follows: 
918   
919      >>> f = x**3-2*x**2+1 
920      >>> sum(roots(f, x).values()) == degree(f, x) 
921      True 
922      >>> f = (x-1)*(x**5-x+1) 
923      >>> sum(roots(f, x).values()) == degree(f, x) 
924      False 
925   
926   
927      References 
928      ========== 
929   
930      .. [1] https://en.wikipedia.org/wiki/Cubic_equation#Trigonometric_and_hyperbolic_solutions 
931   
932      """
933      from sympy.polys.polytools import to_rational_coeffs

937          if gens:

952                  raise PolynomialError("generator must be a Symbol")

1171             '''))

1184 def root_factors(f, *gens, filter=None, **args):
1185     """ 
1186     Returns all factors of a univariate polynomial. 
1187  
1188     Examples 
1189     ======== 
1190  
1191     >>> from sympy.abc import x, y 
1192     >>> from sympy.polys.polyroots import root_factors 
1193  
1194     >>> root_factors(x**2 - y, x) 
1195     [x - sqrt(y), x + sqrt(y)] 
1196  
1197     """
1198     args = dict(args)
1199 
1200     F = Poly(f, *gens, **args)
1201 
1202     if not F.is_Poly:
1203         return [f]
1204 
1205     if F.is_multivariate:
1206         raise ValueError('multivariate polynomials are not supported')
1207 
1208     x = F.gens[0]
1209 
1210     zeros = roots(F, filter=filter)
1211

1213         factors = [F]
1214     else:

1216

1218             factors, N = factors + [Poly(x - r, x)]*n, N + n

1222             factors.append(F.quo(G))
