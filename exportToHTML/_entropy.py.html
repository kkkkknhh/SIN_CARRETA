16   @_axis_nan_policy_factory(
17       lambda x: x,
18       n_samples=lambda kwgs: (
19           2 if ("qk" in kwgs and kwgs["qk"] is not None)
20           else 1

22       n_outputs=1, result_to_tuple=lambda x: (x,), paired=True,
23       too_small=-1  # entropy doesn't have too small inputs
24   )
25   def entropy(pk: np.typing.ArrayLike,
26               qk: np.typing.ArrayLike | None = None,
27               base: float | None = None,
28               axis: int = 0
29               ) -> np.number | np.ndarray:
30       """ 
31       Calculate the Shannon entropy/relative entropy of given distribution(s). 
32    
33       If only probabilities `pk` are given, the Shannon entropy is calculated as 
34       ``H = -sum(pk * log(pk))``. 
35    
36       If `qk` is not None, then compute the relative entropy 
37       ``D = sum(pk * log(pk / qk))``. This quantity is also known 
38       as the Kullback-Leibler divergence. 
39    
40       This routine will normalize `pk` and `qk` if they don't sum to 1. 
41    
42       Parameters 
43       ---------- 
44       pk : array_like 
45           Defines the (discrete) distribution. Along each axis-slice of ``pk``, 
46           element ``i`` is the  (possibly unnormalized) probability of event 
47           ``i``. 
48       qk : array_like, optional 
49           Sequence against which the relative entropy is computed. Should be in 
50           the same format as `pk`. 
51       base : float, optional 
52           The logarithmic base to use, defaults to ``e`` (natural logarithm). 
53       axis : int, optional 
54           The axis along which the entropy is calculated. Default is 0. 
55    
56       Returns 
57       ------- 
58       S : {float, array_like} 
59           The calculated entropy. 
60    
61       Notes 
62       ----- 
63       Informally, the Shannon entropy quantifies the expected uncertainty 
64       inherent in the possible outcomes of a discrete random variable. 
65       For example, 
66       if messages consisting of sequences of symbols from a set are to be 
67       encoded and transmitted over a noiseless channel, then the Shannon entropy 
68       ``H(pk)`` gives a tight lower bound for the average number of units of 
69       information needed per symbol if the symbols occur with frequencies 
70       governed by the discrete distribution `pk` [1]_. The choice of base 
71       determines the choice of units; e.g., ``e`` for nats, ``2`` for bits, etc. 
72    
73       The relative entropy, ``D(pk|qk)``, quantifies the increase in the average 
74       number of units of information needed per symbol if the encoding is 
75       optimized for the probability distribution `qk` instead of the true 
76       distribution `pk`. Informally, the relative entropy quantifies the expected 
77       excess in surprise experienced if one believes the true distribution is 
78       `qk` when it is actually `pk`. 
79    
80       A related quantity, the cross entropy ``CE(pk, qk)``, satisfies the 
81       equation ``CE(pk, qk) = H(pk) + D(pk|qk)`` and can also be calculated with 
82       the formula ``CE = -sum(pk * log(qk))``. It gives the average 
83       number of units of information needed per symbol if an encoding is 
84       optimized for the probability distribution `qk` when the true distribution 
85       is `pk`. It is not computed directly by `entropy`, but it can be computed 
86       using two calls to the function (see Examples). 
87    
88       See [2]_ for more information. 
89    
90       References 
91       ---------- 
92       .. [1] Shannon, C.E. (1948), A Mathematical Theory of Communication. 
93              Bell System Technical Journal, 27: 379-423. 
94              https://doi.org/10.1002/j.1538-7305.1948.tb01338.x 
95       .. [2] Thomas M. Cover and Joy A. Thomas. 2006. Elements of Information 
96              Theory (Wiley Series in Telecommunications and Signal Processing). 
97              Wiley-Interscience, USA. 
98    
99    
100      Examples 
101      -------- 
102      The outcome of a fair coin is the most uncertain: 
103   
104      >>> import numpy as np 
105      >>> from scipy.stats import entropy 
106      >>> base = 2  # work in units of bits 
107      >>> pk = np.array([1/2, 1/2])  # fair coin 
108      >>> H = entropy(pk, base=base) 
109      >>> H 
110      1.0 
111      >>> H == -np.sum(pk * np.log(pk)) / np.log(base) 
112      True 
113   
114      The outcome of a biased coin is less uncertain: 
115   
116      >>> qk = np.array([9/10, 1/10])  # biased coin 
117      >>> entropy(qk, base=base) 
118      0.46899559358928117 
119   
120      The relative entropy between the fair coin and biased coin is calculated 
121      as: 
122   
123      >>> D = entropy(pk, qk, base=base) 
124      >>> D 
125      0.7369655941662062 
126      >>> np.isclose(D, np.sum(pk * np.log(pk/qk)) / np.log(base), rtol=4e-16, atol=0) 
127      True 
128   
129      The cross entropy can be calculated as the sum of the entropy and 
130      relative entropy`: 
131   
132      >>> CE = entropy(pk, base=base) + entropy(pk, qk, base=base) 
133      >>> CE 
134      1.736965594166206 
135      >>> CE == -np.sum(pk * np.log(qk)) / np.log(base) 
136      True 
137   
138      """
139      if base is not None and base <= 0:
140          raise ValueError("`base` must be a positive number or `None`.")
141  
142      xp = array_namespace(pk) if qk is None else array_namespace(pk, qk)
143  
144      pk = xp.asarray(pk)
145      with np.errstate(invalid='ignore'):
146          pk = 1.0*pk / xp.sum(pk, axis=axis, keepdims=True)  # type: ignore[operator]
147      if qk is None:
148          vec = special.entr(pk)
149      else:
150          qk = xp.asarray(qk)
151          pk, qk = _broadcast_arrays((pk, qk), axis=None, xp=xp)  # don't ignore any axes
152          sum_kwargs = dict(axis=axis, keepdims=True)
153          qk = 1.0*qk / xp.sum(qk, **sum_kwargs)  # type: ignore[operator, call-overload]
154          vec = special.rel_entr(pk, qk)
155      S = xp.sum(vec, axis=axis)
156      if base is not None:
157          S /= math.log(base)
158      return S

161  def _differential_entropy_is_too_small(samples, kwargs, axis=-1):
162      values = samples[0]
163      n = values.shape[axis]
164      window_length = kwargs.get("window_length",
165                                 math.floor(math.sqrt(n) + 0.5))
166      if not 2 <= 2 * window_length < n:
167          return True
168      return False

171  @_axis_nan_policy_factory(
172      lambda x: x, n_outputs=1, result_to_tuple=lambda x: (x,),
173      too_small=_differential_entropy_is_too_small
174  )
175  def differential_entropy(
176      values: np.typing.ArrayLike,
177      *,
178      window_length: int | None = None,
179      base: float | None = None,
180      axis: int = 0,
181      method: str = "auto",
182  ) -> np.number | np.ndarray:
183      r"""Given a sample of a distribution, estimate the differential entropy. 
184   
185      Several estimation methods are available using the `method` parameter. By 
186      default, a method is selected based the size of the sample. 
187   
188      Parameters 
189      ---------- 
190      values : sequence 
191          Sample from a continuous distribution. 
192      window_length : int, optional 
193          Window length for computing Vasicek estimate. Must be an integer 
194          between 1 and half of the sample size. If ``None`` (the default), it 
195          uses the heuristic value 
196   
197          .. math:: 
198              \left \lfloor \sqrt{n} + 0.5 \right \rfloor 
199   
200          where :math:`n` is the sample size. This heuristic was originally 
201          proposed in [2]_ and has become common in the literature. 
202      base : float, optional 
203          The logarithmic base to use, defaults to ``e`` (natural logarithm). 
204      axis : int, optional 
205          The axis along which the differential entropy is calculated. 
206          Default is 0. 
207      method : {'vasicek', 'van es', 'ebrahimi', 'correa', 'auto'}, optional 
208          The method used to estimate the differential entropy from the sample. 
209          Default is ``'auto'``.  See Notes for more information. 
210   
211      Returns 
212      ------- 
213      entropy : float 
214          The calculated differential entropy. 
215   
216      Notes 
217      ----- 
218      This function will converge to the true differential entropy in the limit 
219   
220      .. math:: 
221          n \to \infty, \quad m \to \infty, \quad \frac{m}{n} \to 0 
222   
223      The optimal choice of ``window_length`` for a given sample size depends on 
224      the (unknown) distribution. Typically, the smoother the density of the 
225      distribution, the larger the optimal value of ``window_length`` [1]_. 
226   
227      The following options are available for the `method` parameter. 
228   
229      * ``'vasicek'`` uses the estimator presented in [1]_. This is 
230        one of the first and most influential estimators of differential entropy. 
231      * ``'van es'`` uses the bias-corrected estimator presented in [3]_, which 
232        is not only consistent but, under some conditions, asymptotically normal. 
233      * ``'ebrahimi'`` uses an estimator presented in [4]_, which was shown 
234        in simulation to have smaller bias and mean squared error than 
235        the Vasicek estimator. 
236      * ``'correa'`` uses the estimator presented in [5]_ based on local linear 
237        regression. In a simulation study, it had consistently smaller mean 
238        square error than the Vasiceck estimator, but it is more expensive to 
239        compute. 
240      * ``'auto'`` selects the method automatically (default). Currently, 
241        this selects ``'van es'`` for very small samples (<10), ``'ebrahimi'`` 
242        for moderate sample sizes (11-1000), and ``'vasicek'`` for larger 
243        samples, but this behavior is subject to change in future versions. 
244   
245      All estimators are implemented as described in [6]_. 
246   
247      References 
248      ---------- 
249      .. [1] Vasicek, O. (1976). A test for normality based on sample entropy. 
250             Journal of the Royal Statistical Society: 
251             Series B (Methodological), 38(1), 54-59. 
252      .. [2] Crzcgorzewski, P., & Wirczorkowski, R. (1999). Entropy-based 
253             goodness-of-fit test for exponentiality. Communications in 
254             Statistics-Theory and Methods, 28(5), 1183-1202. 
255      .. [3] Van Es, B. (1992). Estimating functionals related to a density by a 
256             class of statistics based on spacings. Scandinavian Journal of 
257             Statistics, 61-72. 
258      .. [4] Ebrahimi, N., Pflughoeft, K., & Soofi, E. S. (1994). Two measures 
259             of sample entropy. Statistics & Probability Letters, 20(3), 225-234. 
260      .. [5] Correa, J. C. (1995). A new estimator of entropy. Communications 
261             in Statistics-Theory and Methods, 24(10), 2439-2449. 
262      .. [6] Noughabi, H. A. (2015). Entropy Estimation Using Numerical Methods. 
263             Annals of Data Science, 2(2), 231-241. 
264             https://link.springer.com/article/10.1007/s40745-015-0045-9 
265   
266      Examples 
267      -------- 
268      >>> import numpy as np 
269      >>> from scipy.stats import differential_entropy, norm 
270   
271      Entropy of a standard normal distribution: 
272   
273      >>> rng = np.random.default_rng() 
274      >>> values = rng.standard_normal(100) 
275      >>> differential_entropy(values) 
276      1.3407817436640392 
277   
278      Compare with the true entropy: 
279   
280      >>> float(norm.entropy()) 
281      1.4189385332046727 
282   
283      For several sample sizes between 5 and 1000, compare the accuracy of 
284      the ``'vasicek'``, ``'van es'``, and ``'ebrahimi'`` methods. Specifically, 
285      compare the root mean squared error (over 1000 trials) between the estimate 
286      and the true differential entropy of the distribution. 
287   
288      >>> from scipy import stats 
289      >>> import matplotlib.pyplot as plt 
290      >>> 
291      >>> 
292      >>> def rmse(res, expected): 
293      ...     '''Root mean squared error''' 
294      ...     return np.sqrt(np.mean((res - expected)**2)) 
295      >>> 
296      >>> 
297      >>> a, b = np.log10(5), np.log10(1000) 
298      >>> ns = np.round(np.logspace(a, b, 10)).astype(int) 
299      >>> reps = 1000  # number of repetitions for each sample size 
300      >>> expected = stats.expon.entropy() 
301      >>> 
302      >>> method_errors = {'vasicek': [], 'van es': [], 'ebrahimi': []} 
303      >>> for method in method_errors: 
304      ...     for n in ns: 
305      ...        rvs = stats.expon.rvs(size=(reps, n), random_state=rng) 
306      ...        res = stats.differential_entropy(rvs, method=method, axis=-1) 
307      ...        error = rmse(res, expected) 
308      ...        method_errors[method].append(error) 
309      >>> 
310      >>> for method, errors in method_errors.items(): 
311      ...     plt.loglog(ns, errors, label=method) 
312      >>> 
313      >>> plt.legend() 
314      >>> plt.xlabel('sample size') 
315      >>> plt.ylabel('RMSE (1000 trials)') 
316      >>> plt.title('Entropy Estimator Error (Exponential Distribution)') 
317   
318      """
319      xp = array_namespace(values)
320      values = xp.asarray(values)
321      if xp.isdtype(values.dtype, "integral"):  # type: ignore[union-attr]
322          values = xp.astype(values, xp.asarray(1.).dtype)
323      values = xp_moveaxis_to_end(values, axis, xp=xp)
324      n = values.shape[-1]  # type: ignore[union-attr]
325  
326      if window_length is None:
327          window_length = math.floor(math.sqrt(n) + 0.5)
328  
329      if not 2 <= 2 * window_length < n:
330          raise ValueError(
331              f"Window length ({window_length}) must be positive and less "
332              f"than half the sample size ({n}).",
333          )
334  
335      if base is not None and base <= 0:
336          raise ValueError("`base` must be a positive number or `None`.")
337  
338      sorted_data = xp.sort(values, axis=-1)
339  
340      methods = {"vasicek": _vasicek_entropy,
341                 "van es": _van_es_entropy,
342                 "correa": _correa_entropy,
343                 "ebrahimi": _ebrahimi_entropy,
344                 "auto": _vasicek_entropy}
345      method = method.lower()
346      if method not in methods:
347          message = f"`method` must be one of {set(methods)}"
348          raise ValueError(message)
349  
350      if method == "auto":
351          if n <= 10:
352              method = 'van es'
353          elif n <= 1000:
354              method = 'ebrahimi'
355          else:
356              method = 'vasicek'
357  
358      res = methods[method](sorted_data, window_length, xp=xp)
359  
360      if base is not None:
361          res /= math.log(base)
362  
363      # avoid dtype changes due to data-apis/array-api-compat#152
364      # can be removed when data-apis/array-api-compat#152 is resolved
365      return xp.astype(res, values.dtype)  # type: ignore[union-attr]

368  def _pad_along_last_axis(X, m, *, xp):
369      """Pad the data for computing the rolling window difference."""
370      # scales a  bit better than method in _vasicek_like_entropy
371      shape = X.shape[:-1] + (m,)
372      Xl = xp.broadcast_to(X[..., :1], shape)  # :1 vs 0 to maintain shape
373      Xr = xp.broadcast_to(X[..., -1:], shape)
374      return xp.concat((Xl, X, Xr), axis=-1)

377  def _vasicek_entropy(X, m, *, xp):
378      """Compute the Vasicek estimator as described in [6] Eq. 1.3."""
379      n = X.shape[-1]
380      X = _pad_along_last_axis(X, m, xp=xp)
381      differences = X[..., 2 * m:] - X[..., : -2 * m:]
382      logs = xp.log(n/(2*m) * differences)
383      return xp.mean(logs, axis=-1)

386  def _van_es_entropy(X, m, *, xp):
387      """Compute the van Es estimator as described in [6]."""
388      # No equation number, but referred to as HVE_mn.
389      # Typo: there should be a log within the summation.
390      n = X.shape[-1]
391      difference = X[..., m:] - X[..., :-m]
392      term1 = 1/(n-m) * xp.sum(xp.log((n+1)/m * difference), axis=-1)
393      k = xp.arange(m, n+1, dtype=term1.dtype)
394      return term1 + xp.sum(1/k) + math.log(m) - math.log(n+1)

397  def _ebrahimi_entropy(X, m, *, xp):
398      """Compute the Ebrahimi estimator as described in [6]."""
399      # No equation number, but referred to as HE_mn
400      n = X.shape[-1]
401      X = _pad_along_last_axis(X, m, xp=xp)
402  
403      differences = X[..., 2 * m:] - X[..., : -2 * m:]
404  
405      i = xp.arange(1, n+1, dtype=X.dtype)
406      ci = xp.ones_like(i)*2
407      ci[i <= m] = 1 + (i[i <= m] - 1)/m
408      ci[i >= n - m + 1] = 1 + (n - i[i >= n-m+1])/m
409  
410      logs = xp.log(n * differences / (ci * m))
411      return xp.mean(logs, axis=-1)

414  def _correa_entropy(X, m, *, xp):
415      """Compute the Correa estimator as described in [6]."""
416      # No equation number, but referred to as HC_mn
417      n = X.shape[-1]
418      X = _pad_along_last_axis(X, m, xp=xp)
419  
420      i = xp.arange(1, n+1)
421      dj = xp.arange(-m, m+1)[:, None]
422      j = i + dj
423      j0 = j + m - 1  # 0-indexed version of j
424  
425      Xibar = xp.mean(X[..., j0], axis=-2, keepdims=True)
426      difference = X[..., j0] - Xibar
427      num = xp.sum(difference*dj, axis=-2)  # dj is d-i
428      den = n*xp.sum(difference**2, axis=-2)
429      return -xp.mean(xp.log(num/den), axis=-1)
