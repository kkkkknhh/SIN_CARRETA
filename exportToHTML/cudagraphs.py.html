3    """ 
4    This module implements CUDA graphs support for TorchDynamo backends. 
5     
6    CUDA graphs allow for capturing and replaying GPU operations, which can significantly 
7    reduce CPU overhead in GPU-accelerated PyTorch models. This module provides: 
8     
9    - CUDA graph creation and management for both forward and backward passes 
10   - Input mutation detection and handling 
11   - Device compatibility checking 
12   - Stack trace management for debugging 
13   - Integration with TorchInductor's cudagraph trees 
14    
15   The backend supports two main modes: 
16   1. cudagraphs: Full CUDA graph support with both forward and backward pass optimization 
17   2. cudagraphs_inner: Lower-level CUDA graph implementation used for benchmarking 
18    
19   Key components: 
20   - CudagraphsBackend: Main backend class for CUDA graph integration 
21   - Mutation detection utilities to ensure graph safety 
22   - Device mapping and compatibility checks 
23   - Stack trace collection for debugging 
24   """

54   def find_input_mutations(g):
55       def meta_fk(meta):
56           return meta["val"] if "val" in meta else meta["fake_result"]
57   
58       inputs = defaultdict(set)
59       input_idx = 0
60       mutated_inputs = set()
61       for n in g.nodes:
62           if n.op == "placeholder":
63               if isinstance(meta_fk(n.meta), torch.Tensor):
64                   inputs[StorageWeakRef(meta_fk(n.meta)._typed_storage())].add(input_idx)
65               input_idx += 1
66           elif n.op == "call_function":
67               if not hasattr(n.target, "_schema"):
68                   continue
69   
70               schema = n.target._schema
71               for i, arg in enumerate(schema.arguments):
72                   if i < len(n.args):
73                       argument = n.args[i]
74                   else:
75                       if arg.name not in n.kwargs:
76                           continue
77                       argument = n.kwargs[arg.name]
78                   mut_arg = False
79                   if arg.alias_info:
80                       if arg.alias_info.is_write:
81                           mut_arg = True
82                   if mut_arg:
83                       # TODO: not correct for args that contain tensors in a struct
84                       # like list
85                       mutated_inputs |= inputs[
86                           StorageWeakRef(meta_fk(argument.meta)._typed_storage())
87                       ]
88   
89           # TODO: error on unrecognized nodes
90       return mutated_inputs

93   def get_device_node_mapping(gm: torch.fx.GraphModule):
94       device_node_mapping: dict[torch.device, torch.fx.Node] = {}
95       for n in gm.graph.nodes:
96           t = n.meta.get("val", None)
97           if isinstance(t, torch.Tensor) and t.device not in device_node_mapping:
98               device_node_mapping[t.device] = n
99       return device_node_mapping

102  def check_for_mutation_ignore_cuda_graph_managed_tensor(
103      aot_model: torch.fx.GraphModule, num_fixed
104  ) -> Optional[str]:
105      mutation_indices = find_input_mutations(aot_model.graph) - set(range(num_fixed))
106      if not mutation_indices:
107          return None
108  
109      placeholders = get_placeholder_info(aot_model.graph)
110      return get_mutation_stack_trace(placeholders, mutation_indices)
111  
112  
113  def check_for_skip(aot_model: torch.fx.GraphModule, num_fixed) -> Optional[str]:
114      if not config.cudagraph_backend_support_input_mutation:
115          if mut_skip := check_for_mutation_ignore_cuda_graph_managed_tensor(
116              aot_model, num_fixed
117          ):
118              return mut_skip
119  
120      if skip := check_multiple_devices_or_any_cpu_nodes(
121          get_device_node_mapping(aot_model)
122      ):
123          return skip
124  
125      if node := get_first_incompatible_cudagraph_node(aot_model):
126          return format_default_skip_message(f"incompatible op ({node.name})")
127  
128      return None

131  def get_device_index(gm) -> int:
132      device = next(iter(get_device_node_mapping(gm)))
133      assert device.type == "cuda"
134      return device.index

137  def get_stack_traces(gm) -> list[Optional[str]]:
138      output = output_node(gm)
139      assert len(output.args) == 1
140      return [
141          (arg.stack_trace if isinstance(arg, torch.fx.node.Node) else None)
142          for arg in output.args[0]
143      ]

146  def cudagraphs(dynamo_model, dynamo_inputs):
147      from torch._inductor.cudagraph_trees import cudagraphify_impl
148  
149      do_cudagraphs = BoxedBool(True)
150      boxed_device_index = BoxedDeviceIndex(None)
151  
152      def forward_cudagraphs(aot_model, aot_inputs, is_inference=False):
153          interp = boxed_nop(aot_model, aot_inputs)
154          fixed = num_fw_fixed_arguments(len(dynamo_inputs), len(aot_inputs))
155          if skip_msg := check_for_skip(aot_model, fixed):
156              BoxedBool.disable(do_cudagraphs)
157              log_cudagraph_skip_and_bump_counter(
158                  f"skipping cudagraphs due to {skip_msg}"
159              )
160              return interp
161  
162          boxed_device_index.set(get_device_index(aot_model))
163          out = cudagraphify_impl(
164              interp,
165              aot_inputs,
166              range(fixed),
167              device_index=boxed_device_index.value,
168              is_backward=False,
169              is_inference=False,
170              stack_traces=get_stack_traces(aot_model),
171              placeholders=get_placeholder_info(aot_model.graph),
172              mutated_input_idxs=find_input_mutations(aot_model.graph),
173          )
174          out._boxed_call = True
175          return out
176  
177      def backward_cudagraphs(aot_model, aot_inputs):
178          interp = boxed_nop(aot_model, aot_inputs)
179          if not do_cudagraphs:
180              return aot_model
181  
182          fixed = count_tangents(aot_model)
183          if skip_msg := check_for_skip(aot_model, fixed):
184              log_cudagraph_skip_and_bump_counter(
185                  "skipping cudagraphs due to %s", skip_msg
186              )
187  
188              # See [Backward Generation Handling]
189              manager = torch._inductor.cudagraph_trees.get_manager(
190                  boxed_device_index.value, create_if_none_exists=False
191              )
192              assert manager is not None
193  
194              def fn(inputs):
195                  manager.set_to_running_backward()
196                  return aot_model(inputs)
197  
198              fn._boxed_call = True
199              return fn
200  
201          out = cudagraphify_impl(
202              interp,
203              aot_inputs,
204              range(fixed),
205              device_index=get_device_index(aot_model),
206              is_backward=True,
207              is_inference=False,
208              stack_traces=get_stack_traces(aot_model),
209              placeholders=get_placeholder_info(aot_model.graph),
210              mutated_input_idxs=find_input_mutations(aot_model.graph),
211          )
212          out._boxed_call = True
213          return out
214  
215      aot_cudagraphs = aot_autograd(
216          fw_compiler=forward_cudagraphs,
217          bw_compiler=backward_cudagraphs,
218          inference_compiler=functools.partial(forward_cudagraphs, is_inference=True),
219          keep_inference_input_mutations=torch._dynamo.config.cudagraph_backend_keep_input_mutation,
220      )
221      return aot_cudagraphs(dynamo_model, dynamo_inputs)

224  class CudagraphsBackend:
225      compiler_name = "cudagraphs"
226  
227      @staticmethod
228      def reset():
229          from torch._inductor.cudagraph_trees import reset_cudagraph_trees
230  
231          reset_cudagraph_trees()
232  
233      @staticmethod
234      def __call__(model, inputs):
235          return cudagraphs(model, inputs)

238  # aot_cudagraphs only applies CUDA graphs to the graph.  It is also helpful
239  # for debugging and can serve as a perf baseline.
240  register_backend(name="cudagraphs", compiler_fn=CudagraphsBackend())

243  def cudagraphs_inner(model, inputs, copy_outputs=True, copy_inputs=True):
244      """This isn't registered as a backend, but is used in some benchmarks"""
245      assert isinstance(inputs, (list, tuple))
246      if copy_inputs:
247          static_inputs = [torch.zeros_like(x) for x in inputs]
248      else:
249          static_inputs = list(inputs)
250  
251      # warmup
252      torch.cuda.synchronize()
253      stream = torch.cuda.Stream()
254      stream.wait_stream(torch.cuda.current_stream())
255      with torch.cuda.stream(stream):
256          model(*inputs)
257      stream.synchronize()
258      torch.cuda.current_stream().wait_stream(stream)
259      torch.cuda.synchronize()
260  
261      # record
262      graph = torch.cuda.CUDAGraph()
263      with torch.cuda.graph(graph, stream=stream):
264          static_outputs = model(*static_inputs)
265      if not isinstance(static_outputs, (list, tuple)):
266          static_outputs = (static_outputs,)
267  
268      def run(*new_inputs):
269          assert len(static_inputs) == len(new_inputs)
270          if copy_inputs:
271              for dst, src in zip(static_inputs, new_inputs):
272                  dst.copy_(src)
273          graph.replay()
274          if copy_outputs:
275              return [x.clone() for x in static_outputs]
276          else:
277              return static_outputs
278  
279      return run
