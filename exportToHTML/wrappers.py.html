26   class ExportTracepoint(HigherOrderOperator):
27       def __init__(self):
28           super().__init__("_export_tracepoint")
29   
30       def __call__(self, *args, **kwargs):
31           return super().__call__(*args, **kwargs)

37   @_export_tracepoint.py_impl(ProxyTorchDispatchMode)
38   def export_tracepoint_dispatch_mode(mode, *args, **kwargs):
39       p_args, p_kwargs = pytree.tree_map(mode.tracer.unwrap_proxy, (args, kwargs))
40       proxy = mode.tracer.create_proxy(
41           "call_function", _export_tracepoint, p_args, p_kwargs
42       )
43       return track_tensor_tree(args, proxy, constant=None, tracer=mode.tracer)

72   def _wrap_submodule(mod, path, module_call_specs):
73       assert isinstance(mod, torch.nn.Module)
74       assert path != ""
75       submodule = torch.fx.graph_module._get_attr(mod, path)
76   
77       def update_module_call_signatures(path, in_spec, out_spec):
78           if path in module_call_specs:
79               assert module_call_specs[path]["in_spec"] == in_spec
80               assert module_call_specs[path]["out_spec"] == out_spec
81           module_call_specs[path] = {"in_spec": in_spec, "out_spec": out_spec}
82   
83       def check_flattened(flat_args):
84           for a in flat_args:
85               if not (isinstance(a, (torch.Tensor, str, int, float, bool)) or a is None):
86                   raise AssertionError(
87                       f"Only Tensors or scalars are supported as pytree flattened inputs, got: {a}"
88                   )
89   
90       def pre_hook(module, args, kwargs):
91           flat_args, in_spec = pytree.tree_flatten((args, kwargs))
92           check_flattened(flat_args)
93           flat_args = _export_tracepoint(*flat_args, kind="module_call_inputs", path=path)
94           args, kwargs = pytree.tree_unflatten(flat_args, in_spec)
95           return args, kwargs
96   
97       def post_hook(module, args, kwargs, res):
98           _, in_spec = pytree.tree_flatten((args, kwargs))
99           flat_res, out_spec = pytree.tree_flatten(res)
100          check_flattened(flat_res)
101          flat_res = _export_tracepoint(*flat_res, kind="module_call_outputs", path=path)
102          update_module_call_signatures(path, in_spec, out_spec)
103          return pytree.tree_unflatten(flat_res, out_spec)
104  
105      pre_handle = submodule.register_forward_pre_hook(pre_hook, with_kwargs=True)
106      post_handle = submodule.register_forward_hook(post_hook, with_kwargs=True)
107      return pre_handle, post_handle

110  @contextmanager
111  def _wrap_submodules(f, preserve_signature, module_call_signatures):
112      handles = []
113  
114      try:
115          for path in preserve_signature:
116              handles.extend(_wrap_submodule(f, path, module_call_signatures))
117          yield
118      finally:
119          for handle in handles:
120              handle.remove()

131  def _register_subclass_spec_proxy_in_tracer(tracer, name, spec):
132      """ 
133      This is a wrapper utility method on top of tracer to cache the 
134      already registered subclass spec attribute. This is useful because 
135      Subclass.__init__ will be same for each subclass. By default, fx will 
136      create multiple attributes/proxies for given attribute. 
137      """
138      fx_name = name + "0"
139      if hasattr(tracer.root, fx_name):
140          assert getattr(tracer.root, fx_name) == spec
141          return tracer.create_proxy("get_attr", fx_name, (), {})
142  
143      qualname = tracer.get_fresh_qualname(name)
144      setattr(tracer.root, qualname, spec)
145      return tracer.create_proxy("get_attr", qualname, (), {})

148  def mark_subclass_constructor_exportable_experimental(constructor_subclass):
149      """ 
150      Experimental decorator that makes subclass to be traceable in export 
151      with pre-dispatch IR. To make your subclass traceble in export, you need to: 
152          1. Implement __init__ method for your subclass (Look at DTensor implementation) 
153          2. Decorate your __init__ method with _mark_constructor_exportable_experimental 
154          3. Put torch._dynamo_disable decorator to prevent dynamo from peeking into its' impl 
155   
156      Example: 
157   
158      class FooTensor(torch.Tensor): 
159          @staticmethod 
160          def __new__(cls, elem, *, requires_grad=False): 
161              # ... 
162              return torch.Tensor._make_subclass(cls, elem, requires_grad=requires_grad) 
163   
164          @torch._dynamo_disable 
165          @mark_subclass_constructor_exportable_experimental 
166          def __init__(self, elem, ...): 
167              # ... 
168      """
169  
170      def _is_init(fn):
171          return callable(fn) and fn.__name__ == "__init__"
172  
173      if not _is_init(constructor_subclass):
174          raise RuntimeError(
175              f"torch._export.wrappers.mark_constructor_exportable_experimental can only be applied on subclass tensor.__init__"
176              f"But, you are adding it on {constructor_subclass.__name__} which is not supported. "
177              f"If __init__ doesn't exist on your subclass, please add it. Look at DTensor.__init__ implementation for example"
178          )
179  
180      def wrapper(*args, **kwargs):
181          if not is_traceable_wrapper_subclass_type(type(args[0])):
182              assert constructor_subclass.__qualname__.endswith("__init__")
183              obj_name = constructor_subclass.__qualname__[: -len("__init__")]
184              raise RuntimeError(
185                  f"Applying mark_constructor_exportable_experimental on {obj_name} is not valid as it is not a traceable "
186                  f"tensor subclass. Please look at DTensor.__init__ implementation as an example of proper usage of this API."
187              )
188          constructor_subclass(*args, **kwargs)
189          if not torch._C._is_torch_function_mode_enabled():
190              return
191          torch_function_mode_stack = torch.overrides._get_current_function_mode_stack()
192  
193          pre_dispatch_tf_modes = [
194              mode
195              for mode in torch_function_mode_stack
196              if isinstance(mode, PreDispatchTorchFunctionMode)
197          ]
198          assert (
199              len(pre_dispatch_tf_modes) <= 1
200          ), f"Expected only one PreDispatchTorchFunctionMode, found {len(pre_dispatch_tf_modes)}"
201  
202          if len(pre_dispatch_tf_modes) == 0:
203              return
204  
205          mode = pre_dispatch_tf_modes[0]
206  
207          tracer = mode.tracer
208          subclass = args[0]
209  
210          flat_args, in_spec = to_graphable((tuple(args[1:]), kwargs))
211  
212          constructor_spec_name = "_".join(
213              constructor_subclass.__qualname__.lower().split(".")
214          )
215          qualname = tracer.get_fresh_qualname(constructor_spec_name)  # type: ignore[union-attr]
216          setattr(tracer.root, qualname, in_spec)  # type: ignore[union-attr]
217          spec_proxy = tracer.create_proxy("get_attr", qualname, (), {})
218          flat_proxy_args = pytree.tree_map_only(
219              torch.Tensor, lambda x: get_proxy_slot(x, tracer).proxy, flat_args
220          )
221  
222          _, func_spec = torch.utils._pytree.tree_flatten(
223              _ConstantFunction(type(subclass))
224          )
225  
226          # We actually don't want to create a new spec for each instance
227          # In fx graph, it will look like dtensor_const_func_spec
228          # We can't directly shove DTensor.__init__ into fx as it is not
229          # allowed type.
230          fxable_constructor_call_spec_name = (
231              type(subclass).__name__.lower() + "_const_func_spec"
232          )
233  
234          # We should try to reuse the constructor call spec as it is guaranteed to be same
235          # for each subclass type. This is different from proxy-ing the init arguments which
236          # can't be reused because for example, DTensor can receive different DeviceMesh etc
237          # as it's arguments
238          func_spec_proxy = _register_subclass_spec_proxy_in_tracer(
239              tracer, fxable_constructor_call_spec_name, func_spec
240          )
241  
242          inner_proxy = tracer.create_proxy(
243              "call_function",
244              flat_apply,
245              (func_spec_proxy, spec_proxy, *flat_proxy_args),
246              {},
247          )
248          track_tensor_tree(subclass, inner_proxy, constant=None, tracer=tracer)
249          return
250  
251      return wrapper
