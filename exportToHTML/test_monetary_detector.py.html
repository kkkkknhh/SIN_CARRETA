13       def test_number_parsing_spanish_conventions(self):
14           """Test number parsing with Spanish decimal conventions."""
15           # Decimal comma
16           assert self.detector._parse_number("1,5") == 1.5
17           assert self.detector._parse_number("123,45") == 123.45
18   
19           # Thousands separator (period)
20           assert self.detector._parse_number("1.234") == 1234.0
21           assert self.detector._parse_number("1.234.567") == 1234567.0
22   
23           # Mixed: thousands separator + decimal comma
24           assert self.detector._parse_number("1.234.567,89") == 1234567.89
25           assert self.detector._parse_number("12.345,6") == 12345.6
26   
27           # English-style (period as decimal)
28           assert self.detector._parse_number("123.45") == 123.45
29   
30           # No separators
31           assert self.detector._parse_number("1234") == 1234.0
