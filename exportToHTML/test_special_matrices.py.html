19   
20       def test_basic(self):
21           y = toeplitz([1, 2, 3])
22           assert_array_equal(y, [[1, 2, 3], [2, 1, 2], [3, 2, 1]])
23           y = toeplitz([1, 2, 3], [1, 4, 5])
24           assert_array_equal(y, [[1, 4, 5], [2, 1, 4], [3, 2, 1]])
25   
26       def test_complex_01(self):
27           data = (1.0 + arange(3.0)) * (1.0 + 1.0j)
28           x = copy(data)
29           t = toeplitz(x)
30           # Calling toeplitz should not change x.
31           assert_array_equal(x, data)
32           # According to the docstring, x should be the first column of t.
33           col0 = t[:, 0]
34           assert_array_equal(col0, data)
35           assert_array_equal(t[0, 1:], data[1:].conj())
36

38           """Scalar arguments still produce a 2D array."""
39           t = toeplitz(10)

243  class TestHelmert:
244  
245      def test_orthogonality(self):
246          for n in range(1, 7):
247              H = helmert(n, full=True)
248              Id = np.eye(n)
249              assert_allclose(H.dot(H.T), Id, atol=1e-12)
250              assert_allclose(H.T.dot(H), Id, atol=1e-12)
251  
252      def test_subspace(self):
253          for n in range(2, 7):
254              H_full = helmert(n, full=True)
255              H_partial = helmert(n)
256              for U in H_full[1:, :].T, H_partial.T:
257                  C = np.eye(n) - np.full((n, n), 1 / n)
258                  assert_allclose(U.dot(U.T), C)
259                  assert_allclose(U.T.dot(U), np.eye(n-1), atol=1e-12)

444  class TestPascal:
445  
446      cases = [
447          (1, array([[1]]), array([[1]])),
448          (2, array([[1, 1],
449                     [1, 2]]),
450              array([[1, 0],
451                     [1, 1]])),
452          (3, array([[1, 1, 1],
453                     [1, 2, 3],
454                     [1, 3, 6]]),
455              array([[1, 0, 0],
456                     [1, 1, 0],
457                     [1, 2, 1]])),
458          (4, array([[1, 1, 1, 1],
459                     [1, 2, 3, 4],
460                     [1, 3, 6, 10],
461                     [1, 4, 10, 20]]),
462              array([[1, 0, 0, 0],
463                     [1, 1, 0, 0],
464                     [1, 2, 1, 0],
465                     [1, 3, 3, 1]])),
466      ]
467  
468      def check_case(self, n, sym, low):
469          assert_array_equal(pascal(n), sym)
470          assert_array_equal(pascal(n, kind='lower'), low)
471          assert_array_equal(pascal(n, kind='upper'), low.T)

495  def test_invpascal():
496  
497      def check_invpascal(n, kind, exact):
498          ip = invpascal(n, kind=kind, exact=exact)
499          p = pascal(n, kind=kind, exact=exact)
500          # Matrix-multiply ip and p, and check that we get the identity matrix.
501          # We can't use the simple expression e = ip.dot(p), because when
502          # n < 35 and exact is True, p.dtype is np.uint64 and ip.dtype is
503          # np.int64. The product of those dtypes is np.float64, which loses
504          # precision when n is greater than 18.  Instead we'll cast both to
505          # object arrays, and then multiply.
506          e = ip.astype(object).dot(p.astype(object))
507          assert_array_equal(e, eye(n), err_msg="n=%d  kind=%r exact=%r" %
508                                                (n, kind, exact))
509  
510      kinds = ['symmetric', 'lower', 'upper']
511  
512      ns = [1, 2, 5, 18]
513      for n in ns:
514          for kind in kinds:
515              for exact in [True, False]:
516                  check_invpascal(n, kind, exact)
517  
518      ns = [19, 34, 35, 50]
519      for n in ns:
520          for kind in kinds:
521              check_invpascal(n, kind, True)
