29   @contextmanager
30   def make_simple(*args, **kwargs):
31       f = netcdf_file(*args, **kwargs)
32       f.history = 'Created for a test'
33       f.createDimension('time', N_EG_ELS)
34       time = f.createVariable('time', VARTYPE_EG, ('time',))
35       time[:] = np.arange(N_EG_ELS)
36       time.units = 'days since 2008-01-01'
37       f.flush()
38       yield f
39       f.close()

50   def assert_mask_matches(arr, expected_mask):
51       ''' 
52       Asserts that the mask of arr is effectively the same as expected_mask. 
53    
54       In contrast to numpy.ma.testutils.assert_mask_equal, this function allows 
55       testing the 'mask' of a standard numpy array (the mask in this case is treated 
56       as all False). 
57    
58       Parameters 
59       ---------- 
60       arr : ndarray or MaskedArray 
61           Array to test. 
62       expected_mask : array_like of booleans 
63           A list giving the expected mask. 
64       '''
65   
66       mask = np.ma.getmaskarray(arr)
67       assert_equal(mask, expected_mask)

70   def test_read_write_files():
71       # test round trip for example file
72       cwd = os.getcwd()
73       try:
74           tmpdir = tempfile.mkdtemp()
75           os.chdir(tmpdir)
76           with make_simple('simple.nc', 'w') as f:
77               pass
78           # read the file we just created in 'a' mode
79           with netcdf_file('simple.nc', 'a') as f:
80               check_simple(f)
81               # add something
82               f._attributes['appendRan'] = 1
83   
84           # To read the NetCDF file we just created::
85           with netcdf_file('simple.nc') as f:
86               # Using mmap is the default (but not on pypy)
87               assert_equal(f.use_mmap, not IS_PYPY)
88               check_simple(f)
89               assert_equal(f._attributes['appendRan'], 1)
90   
91           # Read it in append (and check mmap is off)
92           with netcdf_file('simple.nc', 'a') as f:
93               assert_(not f.use_mmap)
94               check_simple(f)
95               assert_equal(f._attributes['appendRan'], 1)
96   
97           # Now without mmap
98           with netcdf_file('simple.nc', mmap=False) as f:
99               # Using mmap is the default
100              assert_(not f.use_mmap)
101              check_simple(f)
102  
103          # To read the NetCDF file we just created, as file object, no
104          # mmap.  When n * n_bytes(var_type) is not divisible by 4, this
105          # raised an error in pupynere 1.0.12 and scipy rev 5893, because
106          # calculated vsize was rounding up in units of 4 - see
107          # https://www.unidata.ucar.edu/software/netcdf/guide_toc.html
108          with open('simple.nc', 'rb') as fobj:
109              with netcdf_file(fobj) as f:
110                  # by default, don't use mmap for file-like
111                  assert_(not f.use_mmap)
112                  check_simple(f)
113  
114          # Read file from fileobj, with mmap
115          with suppress_warnings() as sup:
116              if IS_PYPY:
117                  sup.filter(RuntimeWarning,
118                             "Cannot close a netcdf_file opened with mmap=True.*")
119              with open('simple.nc', 'rb') as fobj:
120                  with netcdf_file(fobj, mmap=True) as f:
121                      assert_(f.use_mmap)
122                      check_simple(f)
123  
124          # Again read it in append mode (adding another att)
125          with open('simple.nc', 'r+b') as fobj:
126              with netcdf_file(fobj, 'a') as f:
127                  assert_(not f.use_mmap)
128                  check_simple(f)
129                  f.createDimension('app_dim', 1)
130                  var = f.createVariable('app_var', 'i', ('app_dim',))
131                  var[:] = 42
132  
133          # And... check that app_var made it in...
134          with netcdf_file('simple.nc') as f:
135              check_simple(f)
136              assert_equal(f.variables['app_var'][:], 42)
137  
138      finally:
139          if IS_PYPY:
140              # windows cannot remove a dead file held by a mmap
141              # that has not been collected in PyPy
142              break_cycles()
143              break_cycles()
144          os.chdir(cwd)
145          shutil.rmtree(tmpdir)

407  def test_append_recordDimension():
408      dataSize = 100
409  
410      with in_tempdir():
411          # Create file with record time dimension
412          with netcdf_file('withRecordDimension.nc', 'w') as f:
413              f.createDimension('time', None)
414              f.createVariable('time', 'd', ('time',))
415              f.createDimension('x', dataSize)
416              x = f.createVariable('x', 'd', ('x',))
417              x[:] = np.array(range(dataSize))
418              f.createDimension('y', dataSize)
419              y = f.createVariable('y', 'd', ('y',))
420              y[:] = np.array(range(dataSize))
421              f.createVariable('testData', 'i', ('time', 'x', 'y'))
422              f.flush()
423              f.close()
424  
425          for i in range(2):
426              # Open the file in append mode and add data
427              with netcdf_file('withRecordDimension.nc', 'a') as f:
428                  f.variables['time'].data = np.append(f.variables["time"].data, i)
429                  f.variables['testData'][i, :, :] = np.full((dataSize, dataSize), i)
430                  f.flush()
431  
432              # Read the file and check that append worked
433              with netcdf_file('withRecordDimension.nc') as f:
434                  assert_equal(f.variables['time'][-1], i)
435                  assert_equal(f.variables['testData'][-1, :, :].copy(),
436                               np.full((dataSize, dataSize), i))
437                  assert_equal(f.variables['time'].data.shape[0], i+1)
438                  assert_equal(f.variables['testData'].data.shape[0], i+1)
439  
440          # Read the file and check that 'data' was not saved as user defined
441          # attribute of testData variable during append operation
442          with netcdf_file('withRecordDimension.nc') as f:
443              with assert_raises(KeyError) as ar:
444                  f.variables['testData']._attributes['data']
445              ex = ar.value
446              assert_equal(ex.args[0], 'data')

448  def test_maskandscale():
449      t = np.linspace(20, 30, 15)
450      t[3] = 100
451      tm = np.ma.masked_greater(t, 99)
452      fname = pjoin(TEST_DATA_PATH, 'example_2.nc')
453      with netcdf_file(fname, maskandscale=True) as f:
454          Temp = f.variables['Temperature']
455          assert_equal(Temp.missing_value, 9999)
456          assert_equal(Temp.add_offset, 20)
457          assert_equal(Temp.scale_factor, np.float32(0.01))
458          found = Temp[:].compressed()
459          del Temp  # Remove ref to mmap, so file can be closed.
460          expected = np.round(tm.compressed(), 2)
461          assert_allclose(found, expected)
462  
463      with in_tempdir():
464          newfname = 'ms.nc'
465          f = netcdf_file(newfname, 'w', maskandscale=True)
466          f.createDimension('Temperature', len(tm))
467          temp = f.createVariable('Temperature', 'i', ('Temperature',))
468          temp.missing_value = 9999
469          temp.scale_factor = 0.01
470          temp.add_offset = 20
471          temp[:] = tm
472          f.close()
473  
474          with netcdf_file(newfname, maskandscale=True) as f:
475              Temp = f.variables['Temperature']
476              assert_equal(Temp.missing_value, 9999)
477              assert_equal(Temp.add_offset, 20)
478              assert_equal(Temp.scale_factor, np.float32(0.01))
479              expected = np.round(tm.compressed(), 2)
480              found = Temp[:].compressed()
481              del Temp
482              assert_allclose(found, expected)
