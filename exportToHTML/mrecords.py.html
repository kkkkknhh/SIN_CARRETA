36   def _checknames(descr, names=None):
37       """ 
38       Checks that field names ``descr`` are not reserved keywords. 
39    
40       If this is the case, a default 'f%i' is substituted.  If the argument 
41       `names` is not None, updates the field names to valid names. 
42    
43       """
44       ndescr = len(descr)
45       default_names = ['f%i' % i for i in range(ndescr)]
46       if names is None:
47           new_names = default_names
48       else:
49           if isinstance(names, (tuple, list)):
50               new_names = names
51           elif isinstance(names, str):
52               new_names = names.split(',')
53           else:
54               raise NameError(f'illegal input names {names!r}')
55           nnames = len(new_names)
56           if nnames < ndescr:
57               new_names += default_names[nnames:]
58       ndescr = []
59       for (n, d, t) in zip(new_names, default_names, descr.descr):
60           if n in reserved_fields:
61               if t[0] in reserved_fields:
62                   ndescr.append((d, t[1]))
63               else:
64                   ndescr.append(t)
65           else:
66               ndescr.append((n, t[1]))
67       return np.dtype(ndescr)

70   def _get_fieldmask(self):
71       mdescr = [(n, '|b1') for n in self.dtype.names]
72       fdmask = np.empty(self.shape, dtype=mdescr)
73       fdmask.flat = tuple([False] * len(mdescr))
74       return fdmask

77   class MaskedRecords(ma.MaskedArray):
78       """ 
79    
80       Attributes 
81       ---------- 
82       _data : recarray 
83           Underlying data, as a record array. 
84       _mask : boolean array 
85           Mask of the records. A record is masked when all its fields are 
86           masked. 
87       _fieldmask : boolean recarray 
88           Record array of booleans, setting the mask of each individual field 
89           of each record. 
90       _fill_value : record 
91           Filling values for each field. 
92

95       def __new__(cls, shape, dtype=None, buf=None, offset=0, strides=None,
96                   formats=None, names=None, titles=None,
97                   byteorder=None, aligned=False,
98                   mask=ma.nomask, hard_mask=False, fill_value=None, keep_mask=True,
99                   copy=False,
100                  **options):
101  
102          self = np.recarray.__new__(cls, shape, dtype=dtype, buf=buf, offset=offset,
103                                     strides=strides, formats=formats, names=names,
104                                     titles=titles, byteorder=byteorder,
105                                     aligned=aligned,)
106  
107          mdtype = ma.make_mask_descr(self.dtype)
108          if mask is ma.nomask or not np.size(mask):
109              if not keep_mask:
110                  self._mask = tuple([False] * len(mdtype))
111          else:
112              mask = np.array(mask, copy=copy)
113              if mask.shape != self.shape:
114                  (nd, nm) = (self.size, mask.size)
115                  if nm == 1:
116                      mask = np.resize(mask, self.shape)
117                  elif nm == nd:
118                      mask = np.reshape(mask, self.shape)
119                  else:
120                      msg = "Mask and data not compatible: data size is %i, " + \
121                            "mask size is %i."
122                      raise ma.MAError(msg % (nd, nm))
123              if not keep_mask:
124                  self.__setmask__(mask)
125                  self._sharedmask = True
126              else:
127                  if mask.dtype == mdtype:
128                      _mask = mask
129                  else:
130                      _mask = np.array([tuple([m] * len(mdtype)) for m in mask],
131                                       dtype=mdtype)
132                  self._mask = _mask
133          return self

135      def __array_finalize__(self, obj):
136          # Make sure we have a _fieldmask by default
137          _mask = getattr(obj, '_mask', None)
138          if _mask is None:
139              objmask = getattr(obj, '_mask', ma.nomask)
140              _dtype = np.ndarray.__getattribute__(self, 'dtype')
141              if objmask is ma.nomask:
142                  _mask = ma.make_mask_none(self.shape, dtype=_dtype)
143              else:
144                  mdescr = ma.make_mask_descr(_dtype)
145                  _mask = np.array([tuple([m] * len(mdescr)) for m in objmask],
146                                 dtype=mdescr).view(np.recarray)
147          # Update some of the attributes
148          _dict = self.__dict__
149          _dict.update(_mask=_mask)
150          self._update_from(obj)
151          if _dict['_baseclass'] == np.ndarray:
152              _dict['_baseclass'] = np.recarray
153          return

172          """ 
173          Returns the length 
174   
175          """
176          # We have more than one record
177          if self.ndim:

182      def __getattribute__(self, attr):
183          try:
184              return object.__getattribute__(self, attr)
185          except AttributeError:
186              # attr must be a fieldname
187              pass
188          fielddict = np.ndarray.__getattribute__(self, 'dtype').fields
189          try:
190              res = fielddict[attr][:2]
191          except (TypeError, KeyError) as e:
192              raise AttributeError(
193                  f'record array has no attribute {attr}') from e
194          # So far, so good
195          _localdict = np.ndarray.__getattribute__(self, '__dict__')
196          _data = np.ndarray.view(self, _localdict['_baseclass'])
197          obj = _data.getfield(*res)
198          if obj.dtype.names is not None:
199              raise NotImplementedError("MaskedRecords is currently limited to"
200                                        "simple records.")
201          # Get some special attributes
202          # Reset the object's mask
203          hasmasked = False
204          _mask = _localdict.get('_mask', None)
205          if _mask is not None:
206              try:
207                  _mask = _mask[attr]
208              except IndexError:
209                  # Couldn't find a mask: use the default (nomask)
210                  pass
211              tp_len = len(_mask.dtype)
212              hasmasked = _mask.view((bool, ((tp_len,) if tp_len else ()))).any()
213          if (obj.shape or hasmasked):
214              obj = obj.view(ma.MaskedArray)
215              obj._baseclass = np.ndarray
216              obj._isfield = True
217              obj._mask = _mask
218              # Reset the field values
219              _fill_value = _localdict.get('_fill_value', None)
220              if _fill_value is not None:
221                  try:
222                      obj._fill_value = _fill_value[attr]
223                  except ValueError:
224                      obj._fill_value = None
225          else:
226              obj = obj.item()
227          return obj

229      def __setattr__(self, attr, val):
230          """ 
231          Sets the attribute attr to the value val. 
232   
233          """
234          # Should we call __setmask__ first ?
235          if attr in ['mask', 'fieldmask']:
236              self.__setmask__(val)
237              return
238          # Create a shortcut (so that we don't have to call getattr all the time)
239          _localdict = object.__getattribute__(self, '__dict__')
240          # Check whether we're creating a new field
241          newattr = attr not in _localdict
242          try:
243              # Is attr a generic attribute ?
244              ret = object.__setattr__(self, attr, val)
245          except Exception:
246              # Not a generic attribute: exit if it's not a valid field
247              fielddict = np.ndarray.__getattribute__(self, 'dtype').fields or {}
248              optinfo = np.ndarray.__getattribute__(self, '_optinfo') or {}
249              if not (attr in fielddict or attr in optinfo):
250                  raise
251          else:
252              # Get the list of names
253              fielddict = np.ndarray.__getattribute__(self, 'dtype').fields or {}
254              # Check the attribute
255              if attr not in fielddict:
256                  return ret
257              if newattr:
258                  # We just added this one or this setattr worked on an
259                  # internal attribute.
260                  try:
261                      object.__delattr__(self, attr)
262                  except Exception:
263                      return ret
264          # Let's try to set the field
265          try:
266              res = fielddict[attr][:2]
267          except (TypeError, KeyError) as e:
268              raise AttributeError(
269                  f'record array has no attribute {attr}') from e
270  
271          if val is ma.masked:
272              _fill_value = _localdict['_fill_value']
273              if _fill_value is not None:
274                  dval = _localdict['_fill_value'][attr]
275              else:
276                  dval = val
277              mval = True
278          else:
279              dval = ma.filled(val)
280              mval = ma.getmaskarray(val)
281          obj = np.ndarray.__getattribute__(self, '_data').setfield(dval, *res)
282          _localdict['_mask'].__setitem__(attr, mval)
283          return obj

285      def __getitem__(self, indx):
286          """ 
287          Returns all the fields sharing the same fieldname base. 
288   
289          The fieldname base is either `_data` or `_mask`. 
290   
291          """
292          _localdict = self.__dict__
293          _mask = np.ndarray.__getattribute__(self, '_mask')
294          _data = np.ndarray.view(self, _localdict['_baseclass'])
295          # We want a field
296          if isinstance(indx, str):
297              # Make sure _sharedmask is True to propagate back to _fieldmask
298              # Don't use _set_mask, there are some copies being made that
299              # break propagation Don't force the mask to nomask, that wreaks
300              # easy masking
301              obj = _data[indx].view(ma.MaskedArray)
302              obj._mask = _mask[indx]
303              obj._sharedmask = True
304              fval = _localdict['_fill_value']
305              if fval is not None:
306                  obj._fill_value = fval[indx]
307              # Force to masked if the mask is True
308              if not obj.ndim and obj._mask:
309                  return ma.masked
310              return obj
311          # We want some elements.
312          # First, the data.
313          obj = np.asarray(_data[indx]).view(mrecarray)
314          obj._mask = np.asarray(_mask[indx]).view(np.recarray)
315          return obj

318          """ 
319          Sets the given record to value. 
320   
321          """
322          ma.MaskedArray.__setitem__(self, indx, value)
323          if isinstance(indx, str):
324              self._mask[indx] = ma.getmaskarray(value)

326      def __str__(self):
327          """ 
328          Calculates the string representation. 
329   
330          """
331          if self.size > 1:
332              mstr = [f"({','.join([str(i) for i in s])})"
333                      for s in zip(*[getattr(self, f) for f in self.dtype.names])]
334              return f"[{', '.join(mstr)}]"
335          else:
336              mstr = [f"{','.join([str(i) for i in s])}"
337                      for s in zip([getattr(self, f) for f in self.dtype.names])]
338              return f"({', '.join(mstr)})"

340      def __repr__(self):
341          """ 
342          Calculates the repr representation. 
343   
344          """
345          _names = self.dtype.names
346          fmt = "%%%is : %%s" % (max([len(n) for n in _names]) + 4,)
347          reprstr = [fmt % (f, getattr(self, f)) for f in self.dtype.names]
348          reprstr.insert(0, 'masked_records(')
349          reprstr.extend([fmt % ('    fill_value', self.fill_value),
350                          '              )'])
351          return str("\n".join(reprstr))

353      def view(self, dtype=None, type=None):
354          """ 
355          Returns a view of the mrecarray. 
356   
357          """
358          # OK, basic copy-paste from MaskedArray.view.
359          if dtype is None:
360              if type is None:
361                  output = np.ndarray.view(self)
362              else:
363                  output = np.ndarray.view(self, type)
364          # Here again.
365          elif type is None:
366              try:
367                  if issubclass(dtype, np.ndarray):
368                      output = np.ndarray.view(self, dtype)
369                  else:
370                      output = np.ndarray.view(self, dtype)
371              # OK, there's the change
372              except TypeError:
373                  dtype = np.dtype(dtype)
374                  # we need to revert to MaskedArray, but keeping the possibility
375                  # of subclasses (eg, TimeSeriesRecords), so we'll force a type
376                  # set to the first parent
377                  if dtype.fields is None:
378                      basetype = self.__class__.__bases__[0]
379                      output = self.__array__().view(dtype, basetype)
380                      output._update_from(self)
381                  else:
382                      output = np.ndarray.view(self, dtype)
383                  output._fill_value = None
384          else:
385              output = np.ndarray.view(self, dtype, type)
386          # Update the mask, just like in MaskedArray.view
387          if (getattr(output, '_mask', ma.nomask) is not ma.nomask):
388              mdtype = ma.make_mask_descr(output.dtype)
389              output._mask = self._mask.view(mdtype, np.ndarray)
390              output._mask.shape = output.shape
391          return output

416      def tolist(self, fill_value=None):
417          """ 
418          Return the data portion of the array as a list. 
419   
420          Data items are converted to the nearest compatible Python type. 
421          Masked values are converted to fill_value. If fill_value is None, 
422          the corresponding entries in the output list will be ``None``. 
423   
424          """
425          if fill_value is not None:
426              return self.filled(fill_value).tolist()
427          result = np.array(self.filled().tolist(), dtype=object)
428          mask = np.array(self._mask.tolist())
429          result[mask] = None
430          return result.tolist()

432      def __getstate__(self):
433          """Return the internal state of the masked array. 
434   
435          This is for pickling. 
436   
437          """
438          state = (1,
439                   self.shape,
440                   self.dtype,
441                   self.flags.fnc,
442                   self._data.tobytes(),
443                   self._mask.tobytes(),
444                   self._fill_value,
445                   )
446          return state

448      def __setstate__(self, state):
449          """ 
450          Restore the internal state of the masked array. 
451   
452          This is for pickling.  ``state`` is typically the output of the 
453          ``__getstate__`` output, and is a 5-tuple: 
454   
455          - class name 
456          - a tuple giving the shape of the data 
457          - a typecode for the data 
458          - a binary string for the data 
459          - a binary string for the mask. 
460   
461          """
462          (ver, shp, typ, isf, raw, msk, flv) = state
463          np.ndarray.__setstate__(self, (shp, typ, isf, raw))
464          mdtype = np.dtype([(k, np.bool) for (k, _) in self.dtype.descr])
465          self.__dict__['_mask'].__setstate__((shp, mdtype, isf, msk))
466          self.fill_value = flv

468      def __reduce__(self):
469          """ 
470          Return a 3-tuple for pickling a MaskedArray. 
471   
472          """
473          return (_mrreconstruct,
474                  (self.__class__, self._baseclass, (0,), 'b',),
475                  self.__getstate__())

478  def _mrreconstruct(subtype, baseclass, baseshape, basetype,):
479      """ 
480      Build a new MaskedArray from the information stored in a pickle. 
481   
482      """
483      _data = np.ndarray.__new__(baseclass, baseshape, basetype).view(subtype)
484      _mask = np.ndarray.__new__(np.ndarray, baseshape, 'b1')
485      return subtype.__new__(subtype, _data, mask=_mask, dtype=basetype,)

495  def fromarrays(arraylist, dtype=None, shape=None, formats=None,
496                 names=None, titles=None, aligned=False, byteorder=None,
497                 fill_value=None):
498      """ 
499      Creates a mrecarray from a (flat) list of masked arrays. 
500   
501      Parameters 
502      ---------- 
503      arraylist : sequence 
504          A list of (masked) arrays. Each element of the sequence is first converted 
505          to a masked array if needed. If a 2D array is passed as argument, it is 
506          processed line by line 
507      dtype : {None, dtype}, optional 
508          Data type descriptor. 
509      shape : {None, integer}, optional 
510          Number of records. If None, shape is defined from the shape of the 
511          first array in the list. 
512      formats : {None, sequence}, optional 
513          Sequence of formats for each individual field. If None, the formats will 
514          be autodetected by inspecting the fields and selecting the highest dtype 
515          possible. 
516      names : {None, sequence}, optional 
517          Sequence of the names of each field. 
518      fill_value : {None, sequence}, optional 
519          Sequence of data to be used as filling values. 
520   
521      Notes 
522      ----- 
523      Lists of tuples should be preferred over lists of lists for faster processing. 
524   
525      """
526      datalist = [ma.getdata(x) for x in arraylist]
527      masklist = [np.atleast_1d(ma.getmaskarray(x)) for x in arraylist]
528      _array = np.rec.fromarrays(datalist,
529                                 dtype=dtype, shape=shape, formats=formats,
530                                 names=names, titles=titles, aligned=aligned,
531                                 byteorder=byteorder).view(mrecarray)
532      _array._mask.flat = list(zip(*masklist))
533      if fill_value is not None:
534          _array.fill_value = fill_value
535      return _array

538  def fromrecords(reclist, dtype=None, shape=None, formats=None, names=None,
539                  titles=None, aligned=False, byteorder=None,
540                  fill_value=None, mask=ma.nomask):
541      """ 
542      Creates a MaskedRecords from a list of records. 
543   
544      Parameters 
545      ---------- 
546      reclist : sequence 
547          A list of records. Each element of the sequence is first converted 
548          to a masked array if needed. If a 2D array is passed as argument, it is 
549          processed line by line 
550      dtype : {None, dtype}, optional 
551          Data type descriptor. 
552      shape : {None,int}, optional 
553          Number of records. If None, ``shape`` is defined from the shape of the 
554          first array in the list. 
555      formats : {None, sequence}, optional 
556          Sequence of formats for each individual field. If None, the formats will 
557          be autodetected by inspecting the fields and selecting the highest dtype 
558          possible. 
559      names : {None, sequence}, optional 
560          Sequence of the names of each field. 
561      fill_value : {None, sequence}, optional 
562          Sequence of data to be used as filling values. 
563      mask : {nomask, sequence}, optional. 
564          External mask to apply on the data. 
565   
566      Notes 
567      ----- 
568      Lists of tuples should be preferred over lists of lists for faster processing. 
569   
570      """
571      # Grab the initial _fieldmask, if needed:
572      _mask = getattr(reclist, '_mask', None)
573      # Get the list of records.
574      if isinstance(reclist, np.ndarray):
575          # Make sure we don't have some hidden mask
576          if isinstance(reclist, ma.MaskedArray):
577              reclist = reclist.filled().view(np.ndarray)
578          # Grab the initial dtype, just in case
579          if dtype is None:
580              dtype = reclist.dtype
581          reclist = reclist.tolist()
582      mrec = np.rec.fromrecords(reclist, dtype=dtype, shape=shape, formats=formats,
583                            names=names, titles=titles,
584                            aligned=aligned, byteorder=byteorder).view(mrecarray)
585      # Set the fill_value if needed
586      if fill_value is not None:
587          mrec.fill_value = fill_value
588      # Now, let's deal w/ the mask
589      if mask is not ma.nomask:
590          mask = np.asarray(mask)
591          maskrecordlength = len(mask.dtype)
592          if maskrecordlength:
593              mrec._mask.flat = mask
594          elif mask.ndim == 2:
595              mrec._mask.flat = [tuple(m) for m in mask]
596          else:
597              mrec.__setmask__(mask)
598      if _mask is not None:
599          mrec._mask[:] = _mask
600      return mrec

603  def _guessvartypes(arr):
604      """ 
605      Tries to guess the dtypes of the str_ ndarray `arr`. 
606   
607      Guesses by testing element-wise conversion. Returns a list of dtypes. 
608      The array is first converted to ndarray. If the array is 2D, the test 
609      is performed on the first line. An exception is raised if the file is 
610      3D or more. 
611   
612      """
613      vartypes = []
614      arr = np.asarray(arr)
615      if arr.ndim == 2:
616          arr = arr[0]
617      elif arr.ndim > 2:
618          raise ValueError("The array should be 2D at most!")
619      # Start the conversion loop.
620      for f in arr:
621          try:
622              int(f)
623          except (ValueError, TypeError):
624              try:
625                  float(f)
626              except (ValueError, TypeError):
627                  try:
628                      complex(f)
629                  except (ValueError, TypeError):
630                      vartypes.append(arr.dtype)
631                  else:
632                      vartypes.append(np.dtype(complex))
633              else:
634                  vartypes.append(np.dtype(float))
635          else:
636              vartypes.append(np.dtype(int))
637      return vartypes

640  def openfile(fname):
641      """ 
642      Opens the file handle of file `fname`. 
643   
644      """
645      # A file handle
646      if hasattr(fname, 'readline'):
647          return fname
648      # Try to open the file and guess its type
649      try:
650          f = open(fname)
651      except FileNotFoundError as e:
652          raise FileNotFoundError(f"No such file: '{fname}'") from e
653      if f.readline()[:2] != "\\x":
654          f.seek(0, 0)
655          return f
656      f.close()
657      raise NotImplementedError("Wow, binary file")

660  def fromtextfile(fname, delimiter=None, commentchar='#', missingchar='',
661                   varnames=None, vartypes=None,
662                   *, delimitor=np._NoValue):  # backwards compatibility
663      """ 
664      Creates a mrecarray from data stored in the file `filename`. 
665   
666      Parameters 
667      ---------- 
668      fname : {file name/handle} 
669          Handle of an opened file. 
670      delimiter : {None, string}, optional 
671          Alphanumeric character used to separate columns in the file. 
672          If None, any (group of) white spacestring(s) will be used. 
673      commentchar : {'#', string}, optional 
674          Alphanumeric character used to mark the start of a comment. 
675      missingchar : {'', string}, optional 
676          String indicating missing data, and used to create the masks. 
677      varnames : {None, sequence}, optional 
678          Sequence of the variable names. If None, a list will be created from 
679          the first non empty line of the file. 
680      vartypes : {None, sequence}, optional 
681          Sequence of the variables dtypes. If None, it will be estimated from 
682          the first non-commented line. 
683   
684   
685      Ultra simple: the varnames are in the header, one line"""
686      if delimitor is not np._NoValue:
687          if delimiter is not None:
688              raise TypeError("fromtextfile() got multiple values for argument "
689                              "'delimiter'")
690          # NumPy 1.22.0, 2021-09-23
691          warnings.warn("The 'delimitor' keyword argument of "
692                        "numpy.ma.mrecords.fromtextfile() is deprecated "
693                        "since NumPy 1.22.0, use 'delimiter' instead.",
694                        DeprecationWarning, stacklevel=2)
695          delimiter = delimitor
696  
697      # Try to open the file.
698      ftext = openfile(fname)
699  
700      # Get the first non-empty line as the varnames
701      while True:
702          line = ftext.readline()
703          firstline = line[:line.find(commentchar)].strip()
704          _varnames = firstline.split(delimiter)
705          if len(_varnames) > 1:
706              break
707      if varnames is None:
708          varnames = _varnames
709  
710      # Get the data.
711      _variables = ma.masked_array([line.strip().split(delimiter) for line in ftext
712                                   if line[0] != commentchar and len(line) > 1])
713      (_, nfields) = _variables.shape
714      ftext.close()
715  
716      # Try to guess the dtype.
717      if vartypes is None:
718          vartypes = _guessvartypes(_variables[0])
719      else:
720          vartypes = [np.dtype(v) for v in vartypes]
721          if len(vartypes) != nfields:
722              msg = "Attempting to %i dtypes for %i fields!"
723              msg += " Reverting to default."
724              warnings.warn(msg % (len(vartypes), nfields), stacklevel=2)
725              vartypes = _guessvartypes(_variables[0])
726  
727      # Construct the descriptor.
728      mdescr = list(zip(varnames, vartypes))
729      mfillv = [ma.default_fill_value(f) for f in vartypes]
730  
731      # Get the data and the mask.
732      # We just need a list of masked_arrays. It's easier to create it like that:
733      _mask = (_variables.T == missingchar)
734      _datalist = [ma.masked_array(a, mask=m, dtype=t, fill_value=f)
735                   for (a, m, t, f) in zip(_variables.T, _mask, vartypes, mfillv)]
736  
737      return fromarrays(_datalist, dtype=mdescr)
738

740  def addfield(mrecord, newfield, newfieldname=None):
741      """Adds a new field to the masked record array 
742   
743      Uses `newfield` as data and `newfieldname` as name. If `newfieldname` 
744      is None, the new field name is set to 'fi', where `i` is the number of 
745      existing fields. 
746   
747      """
748      _data = mrecord._data
749      _mask = mrecord._mask
750      if newfieldname is None or newfieldname in reserved_fields:
751          newfieldname = 'f%i' % len(_data.dtype)
752      newfield = ma.array(newfield)
753      # Get the new data.
754      # Create a new empty recarray
755      newdtype = np.dtype(_data.dtype.descr + [(newfieldname, newfield.dtype)])
756      newdata = np.recarray(_data.shape, newdtype)
757      # Add the existing field
758      [newdata.setfield(_data.getfield(*f), *f)
759       for f in _data.dtype.fields.values()]
760      # Add the new field
761      newdata.setfield(newfield._data, *newdata.dtype.fields[newfieldname])
762      newdata = newdata.view(MaskedRecords)
763      # Get the new mask
764      # Create a new empty recarray
765      newmdtype = np.dtype([(n, np.bool) for n in newdtype.names])
766      newmask = np.recarray(_data.shape, newmdtype)
767      # Add the old masks
768      [newmask.setfield(_mask.getfield(*f), *f)
769       for f in _mask.dtype.fields.values()]
770      # Add the mask of the new field
771      newmask.setfield(ma.getmaskarray(newfield),
772                       *newmask.dtype.fields[newfieldname])
773      newdata._mask = newmask
774      return newdata
