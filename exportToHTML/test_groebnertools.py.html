21   def _do_test_groebner():
22       R, x,y = ring("x,y", QQ, lex)
23       f = x**2 + 2*x*y**2
24       g = x*y + 2*y**3 - 1
25   
26       assert groebner([f, g], R) == [x, y**3 - QQ(1,2)]
27   
28       R, y,x = ring("y,x", QQ, lex)
29       f = 2*x**2*y + y**2
30       g = 2*x**3 + x*y - 1
31   
32       assert groebner([f, g], R) == [y, x**3 - QQ(1,2)]
33   
34       R, x,y,z = ring("x,y,z", QQ, lex)
35       f = x - z**2
36       g = y - z**3
37   
38       assert groebner([f, g], R) == [f, g]
39   
40       R, x,y = ring("x,y", QQ, grlex)
41       f = x**3 - 2*x*y
42       g = x**2*y + x - 2*y**2
43   
44       assert groebner([f, g], R) == [x**2, x*y, -QQ(1,2)*x + y**2]
45   
46       R, x,y,z = ring("x,y,z", QQ, lex)
47       f = -x**2 + y
48       g = -x**3 + z
49   
50       assert groebner([f, g], R) == [x**2 - y, x*y - z, x*z - y**2, y**3 - z**2]
51   
52       R, x,y,z = ring("x,y,z", QQ, grlex)
53       f = -x**2 + y
54       g = -x**3 + z
55   
56       assert groebner([f, g], R) == [y**3 - z**2, x**2 - y, x*y - z, x*z - y**2]

156      E = [(1, 2), (2, 3), (1, 4), (1, 6), (1, 12), (2, 5), (2, 7), (3, 8), (3, 10),

158           (11, 12), (5, 12), (5, 9), (6, 10), (7, 11), (8, 12), (3, 4)]
159  
160      R, V = xring([ "x%d" % v for v in V ], QQ, lex)
161      E = [(V[i - 1], V[j - 1]) for i, j in E]
162  
163      x1, x2, x3, x4, x5, x6, x7, x8, x9, x10, x11, x12 = V
164  
165      I3 = [x**3 - 1 for x in V]

168      I = I3 + Ig
169  
170      assert groebner(I[:-1], R) == [
171          x1 + x11 + x12,
172          x2 - x11,
173          x3 - x12,
174          x4 - x12,
175          x5 + x11 + x12,
176          x6 - x11,
177          x7 - x12,
178          x8 + x11 + x12,
179          x9 - x11,
180          x10 + x11 + x12,
181          x11**2 + x11*x12 + x12**2,
182          x12**3 - 1,
183      ]
184  
185      assert groebner(I, R) == [1]
