226  class TestShJacobi:
227      def test_sh_jacobi(self):
228          # G^(p,q)_n(x) = n! gamma(n+p)/gamma(2*n+p) * P^(p-q,q-1)_n(2*x-1)
229          def conv(n, p):
230              return gamma(n + 1) * gamma(n + p) / gamma(2 * n + p)
231          psub = np.poly1d([2,-1])
232          q = 4 * np.random.random()
233          p = q-1 + 2*np.random.random()
234          # print("shifted jacobi p,q = ", p, q)
235          G0 = orth.sh_jacobi(0,p,q)
236          G1 = orth.sh_jacobi(1,p,q)
237          G2 = orth.sh_jacobi(2,p,q)
238          G3 = orth.sh_jacobi(3,p,q)
239          G4 = orth.sh_jacobi(4,p,q)
240          G5 = orth.sh_jacobi(5,p,q)
241          ge0 = orth.jacobi(0,p-q,q-1)(psub) * conv(0,p)
242          ge1 = orth.jacobi(1,p-q,q-1)(psub) * conv(1,p)
243          ge2 = orth.jacobi(2,p-q,q-1)(psub) * conv(2,p)
244          ge3 = orth.jacobi(3,p-q,q-1)(psub) * conv(3,p)
245          ge4 = orth.jacobi(4,p-q,q-1)(psub) * conv(4,p)
246          ge5 = orth.jacobi(5,p-q,q-1)(psub) * conv(5,p)
247  
248          assert_array_almost_equal(G0.c,ge0.c,13)
249          assert_array_almost_equal(G1.c,ge1.c,13)
250          assert_array_almost_equal(G2.c,ge2.c,13)
251          assert_array_almost_equal(G3.c,ge3.c,13)
252          assert_array_almost_equal(G4.c,ge4.c,13)
253          assert_array_almost_equal(G5.c,ge5.c,13)

294  def verify_gauss_quad(root_func, eval_func, weight_func, a, b, N,
295                        rtol=1e-15, atol=5e-14):
296      # this test is copied from numpy's TestGauss in test_hermite.py
297      x, w, mu = root_func(N, True)
298  
299      n = np.arange(N, dtype=np.dtype("long"))
300      v = eval_func(n[:,np.newaxis], x)
301      vv = np.dot(v*w, v.T)
302      vd = 1 / np.sqrt(vv.diagonal())
303      vv = vd[:, np.newaxis] * vv * vd
304      assert_allclose(vv, np.eye(N), rtol, atol)
305  
306      # check that the integral of 1 is correct
307      assert_allclose(w.sum(), mu, rtol, atol)
308  
309      # compare the results of integrating a function with quad.
310      def f(x):
311          return x ** 3 - 3 * x ** 2 + x - 2
312      resI = integrate.quad(lambda x: f(x)*weight_func(x), a, b)
313      resG = np.vdot(f(x), w)
314      rtol = 1e-6 if 1e-6 < resI[1] else resI[1] * 10
315      assert_allclose(resI[0], resG, rtol=rtol)

496  def test_roots_hermite_asy():
497      # Recursion for Hermite functions
498      def hermite_recursion(n, nodes):
499          H = np.zeros((n, nodes.size))
500          H[0,:] = np.pi**(-0.25) * np.exp(-0.5*nodes**2)
501          if n > 1:
502              H[1,:] = sqrt(2.0) * nodes * H[0,:]
503              for k in range(2, n):
504                  H[k,:] = sqrt(2.0/k) * nodes * H[k-1,:] - sqrt((k-1.0)/k) * H[k-2,:]
505          return H
506  
507      # This tests only the nodes
508      def test(N, rtol=1e-15, atol=1e-14):
509          x, w = orth._roots_hermite_asy(N)
510          H = hermite_recursion(N+1, x)
511          assert_allclose(H[-1,:], np.zeros(N), rtol, atol)
512          assert_allclose(sum(w), sqrt(np.pi), rtol, atol)
513  
514      test(150, atol=1e-12)
