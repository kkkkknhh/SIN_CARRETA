40   def call_minpack(fun, x0, jac, ftol, xtol, gtol, max_nfev, x_scale, diff_step):
41       n = x0.size
42   
43       if diff_step is None:
44           epsfcn = EPS
45       else:
46           epsfcn = diff_step**2
47   
48       # Compute MINPACK's `diag`, which is inverse of our `x_scale` and
49       # ``x_scale='jac'`` corresponds to ``diag=None``.
50       if isinstance(x_scale, str) and x_scale == 'jac':
51           diag = None
52       else:
53           diag = 1 / x_scale
54   
55       full_output = True
56       col_deriv = False
57       factor = 100.0
58   
59       if jac is None:
60           if max_nfev is None:
61               # n squared to account for Jacobian evaluations.
62               max_nfev = 100 * n * (n + 1)
63           x, info, status = _minpack._lmdif(
64               fun, x0, (), full_output, ftol, xtol, gtol,
65               max_nfev, epsfcn, factor, diag)
66       else:
67           if max_nfev is None:
68               max_nfev = 100 * n
69           x, info, status = _minpack._lmder(
70               fun, jac, x0, (), full_output, col_deriv,
71               ftol, xtol, gtol, max_nfev, factor, diag)
72   
73       f = info['fvec']
74   
75       if callable(jac):
76           J = jac(x)
77       else:
78           J = np.atleast_2d(approx_derivative(fun, x))
79   
80       cost = 0.5 * np.dot(f, f)
81       g = J.T.dot(f)
82       g_norm = norm(g, ord=np.inf)
83   
84       nfev = info['nfev']
85       njev = info.get('njev', None)
86   
87       status = FROM_MINPACK_TO_COMMON[status]
88       active_mask = np.zeros_like(x0, dtype=int)
89   
90       return OptimizeResult(
91           x=x, cost=cost, fun=f, jac=J, grad=g, optimality=g_norm,
92           active_mask=active_mask, nfev=nfev, njev=njev, status=status)

95   def prepare_bounds(bounds, n):
96       lb, ub = (np.asarray(b, dtype=float) for b in bounds)
97       if lb.ndim == 0:
98           lb = np.resize(lb, n)
99   
100      if ub.ndim == 0:
101          ub = np.resize(ub, n)
102  
103      return lb, ub

106  def check_tolerance(ftol, xtol, gtol, method):
107      def check(tol, name):
108          if tol is None:
109              tol = 0
110          elif tol < EPS:
111              warn(f"Setting `{name}` below the machine epsilon ({EPS:.2e}) effectively "
112                   f"disables the corresponding termination condition.",
113                   stacklevel=3)
114          return tol
115  
116      ftol = check(ftol, "ftol")
117      xtol = check(xtol, "xtol")
118      gtol = check(gtol, "gtol")
119  
120      if method == "lm" and (ftol < EPS or xtol < EPS or gtol < EPS):
121          raise ValueError("All tolerances must be higher than machine epsilon "
122                           f"({EPS:.2e}) for method 'lm'.")
123      elif ftol < EPS and xtol < EPS and gtol < EPS:
124          raise ValueError("At least one of the tolerances must be higher than "
125                           f"machine epsilon ({EPS:.2e}).")
126  
127      return ftol, xtol, gtol

130  def check_x_scale(x_scale, x0):
131      if isinstance(x_scale, str) and x_scale == 'jac':
132          return x_scale
133  
134      try:
135          x_scale = np.asarray(x_scale, dtype=float)
136          valid = np.all(np.isfinite(x_scale)) and np.all(x_scale > 0)
137      except (ValueError, TypeError):
138          valid = False
139  
140      if not valid:
141          raise ValueError("`x_scale` must be 'jac' or array_like with "
142                           "positive numbers.")
143  
144      if x_scale.ndim == 0:
145          x_scale = np.resize(x_scale, x0.shape)
146  
147      if x_scale.shape != x0.shape:
148          raise ValueError("Inconsistent shapes between `x_scale` and `x0`.")
149  
150      return x_scale

153  def check_jac_sparsity(jac_sparsity, m, n):
154      if jac_sparsity is None:
155          return None
156  
157      if not issparse(jac_sparsity):
158          jac_sparsity = np.atleast_2d(jac_sparsity)
159  
160      if jac_sparsity.shape != (m, n):
161          raise ValueError("`jac_sparsity` has wrong shape.")
162  
163      return jac_sparsity, group_columns(jac_sparsity)

169  def huber(z, rho, cost_only):
170      mask = z <= 1
171      rho[0, mask] = z[mask]
172      rho[0, ~mask] = 2 * z[~mask]**0.5 - 1
173      if cost_only:
174          return
175      rho[1, mask] = 1
176      rho[1, ~mask] = z[~mask]**-0.5
177      rho[2, mask] = 0
178      rho[2, ~mask] = -0.5 * z[~mask]**-1.5
179  
180  
181  def soft_l1(z, rho, cost_only):
182      t = 1 + z
183      rho[0] = 2 * (t**0.5 - 1)
184      if cost_only:
185          return
186      rho[1] = t**-0.5
187      rho[2] = -0.5 * t**-1.5
188  
189  
190  def cauchy(z, rho, cost_only):
191      rho[0] = np.log1p(z)
192      if cost_only:
193          return
194      t = 1 + z
195      rho[1] = 1 / t
196      rho[2] = -1 / t**2
197  
198  
199  def arctan(z, rho, cost_only):
200      rho[0] = np.arctan(z)
201      if cost_only:
202          return
203      t = 1 + z**2
204      rho[1] = 1 / t
205      rho[2] = -2 * z / t**2
206

209                            cauchy=cauchy, arctan=arctan)

212  def construct_loss_function(m, loss, f_scale):
213      if loss == 'linear':
214          return None
215  
216      if not callable(loss):
217          loss = IMPLEMENTED_LOSSES[loss]
218          rho = np.empty((3, m))
219  
220          def loss_function(f, cost_only=False):
221              z = (f / f_scale) ** 2
222              loss(z, rho, cost_only=cost_only)
223              if cost_only:
224                  return 0.5 * f_scale ** 2 * np.sum(rho[0])
225              rho[0] *= f_scale ** 2
226              rho[2] /= f_scale ** 2
227              return rho
228      else:
229          def loss_function(f, cost_only=False):
230              z = (f / f_scale) ** 2
231              rho = loss(z)
232              if cost_only:
233                  return 0.5 * f_scale ** 2 * np.sum(rho[0])
234              rho[0] *= f_scale ** 2
235              rho[2] /= f_scale ** 2
236              return rho
237  
238      return loss_function

241  def least_squares(
242          fun, x0, jac='2-point', bounds=(-np.inf, np.inf), method='trf',
243          ftol=1e-8, xtol=1e-8, gtol=1e-8, x_scale=1.0, loss='linear',
244          f_scale=1.0, diff_step=None, tr_solver=None, tr_options=None,
245          jac_sparsity=None, max_nfev=None, verbose=0, args=(), kwargs=None):
246      """Solve a nonlinear least-squares problem with bounds on the variables. 
247   
248      Given the residuals f(x) (an m-D real function of n real 
249      variables) and the loss function rho(s) (a scalar function), `least_squares` 
250      finds a local minimum of the cost function F(x):: 
251   
252          minimize F(x) = 0.5 * sum(rho(f_i(x)**2), i = 0, ..., m - 1) 
253          subject to lb <= x <= ub 
254   
255      The purpose of the loss function rho(s) is to reduce the influence of 
256      outliers on the solution. 
257   
258      Parameters 
259      ---------- 
260      fun : callable 
261          Function which computes the vector of residuals, with the signature 
262          ``fun(x, *args, **kwargs)``, i.e., the minimization proceeds with 
263          respect to its first argument. The argument ``x`` passed to this 
264          function is an ndarray of shape (n,) (never a scalar, even for n=1). 
265          It must allocate and return a 1-D array_like of shape (m,) or a scalar. 
266          If the argument ``x`` is complex or the function ``fun`` returns 
267          complex residuals, it must be wrapped in a real function of real 
268          arguments, as shown at the end of the Examples section. 
269      x0 : array_like with shape (n,) or float 
270          Initial guess on independent variables. If float, it will be treated 
271          as a 1-D array with one element. When `method` is 'trf', the initial 
272          guess might be slightly adjusted to lie sufficiently within the given 
273          `bounds`. 
274      jac : {'2-point', '3-point', 'cs', callable}, optional 
275          Method of computing the Jacobian matrix (an m-by-n matrix, where 
276          element (i, j) is the partial derivative of f[i] with respect to 
277          x[j]). The keywords select a finite difference scheme for numerical 
278          estimation. The scheme '3-point' is more accurate, but requires 
279          twice as many operations as '2-point' (default). The scheme 'cs' 
280          uses complex steps, and while potentially the most accurate, it is 
281          applicable only when `fun` correctly handles complex inputs and 
282          can be analytically continued to the complex plane. Method 'lm' 
283          always uses the '2-point' scheme. If callable, it is used as 
284          ``jac(x, *args, **kwargs)`` and should return a good approximation 
285          (or the exact value) for the Jacobian as an array_like (np.atleast_2d 
286          is applied), a sparse matrix (csr_matrix preferred for performance) or 
287          a `scipy.sparse.linalg.LinearOperator`. 
288      bounds : 2-tuple of array_like or `Bounds`, optional 
289          There are two ways to specify bounds: 
290   
291          1. Instance of `Bounds` class 
292          2. Lower and upper bounds on independent variables. Defaults to no 
293             bounds. Each array must match the size of `x0` or be a scalar, 
294             in the latter case a bound will be the same for all variables. 
295             Use ``np.inf`` with an appropriate sign to disable bounds on all 
296             or some variables. 
297   
298      method : {'trf', 'dogbox', 'lm'}, optional 
299          Algorithm to perform minimization. 
300   
301          * 'trf' : Trust Region Reflective algorithm, particularly suitable 
302            for large sparse problems with bounds. Generally robust method. 
303          * 'dogbox' : dogleg algorithm with rectangular trust regions, 
304            typical use case is small problems with bounds. Not recommended 
305            for problems with rank-deficient Jacobian. 
306          * 'lm' : Levenberg-Marquardt algorithm as implemented in MINPACK. 
307            Doesn't handle bounds and sparse Jacobians. Usually the most 
308            efficient method for small unconstrained problems. 
309   
310          Default is 'trf'. See Notes for more information. 
311      ftol : float or None, optional 
312          Tolerance for termination by the change of the cost function. Default 
313          is 1e-8. The optimization process is stopped when ``dF < ftol * F``, 
314          and there was an adequate agreement between a local quadratic model and 
315          the true model in the last step. 
316   
317          If None and 'method' is not 'lm', the termination by this condition is 
318          disabled. If 'method' is 'lm', this tolerance must be higher than 
319          machine epsilon. 
320      xtol : float or None, optional 
321          Tolerance for termination by the change of the independent variables. 
322          Default is 1e-8. The exact condition depends on the `method` used: 
323   
324          * For 'trf' and 'dogbox' : ``norm(dx) < xtol * (xtol + norm(x))``. 
325          * For 'lm' : ``Delta < xtol * norm(xs)``, where ``Delta`` is 
326            a trust-region radius and ``xs`` is the value of ``x`` 
327            scaled according to `x_scale` parameter (see below). 
328   
329          If None and 'method' is not 'lm', the termination by this condition is 
330          disabled. If 'method' is 'lm', this tolerance must be higher than 
331          machine epsilon. 
332      gtol : float or None, optional 
333          Tolerance for termination by the norm of the gradient. Default is 1e-8. 
334          The exact condition depends on a `method` used: 
335   
336          * For 'trf' : ``norm(g_scaled, ord=np.inf) < gtol``, where 
337            ``g_scaled`` is the value of the gradient scaled to account for 
338            the presence of the bounds [STIR]_. 
339          * For 'dogbox' : ``norm(g_free, ord=np.inf) < gtol``, where 
340            ``g_free`` is the gradient with respect to the variables which 
341            are not in the optimal state on the boundary. 
342          * For 'lm' : the maximum absolute value of the cosine of angles 
343            between columns of the Jacobian and the residual vector is less 
344            than `gtol`, or the residual vector is zero. 
345   
346          If None and 'method' is not 'lm', the termination by this condition is 
347          disabled. If 'method' is 'lm', this tolerance must be higher than 
348          machine epsilon. 
349      x_scale : array_like or 'jac', optional 
350          Characteristic scale of each variable. Setting `x_scale` is equivalent 
351          to reformulating the problem in scaled variables ``xs = x / x_scale``. 
352          An alternative view is that the size of a trust region along jth 
353          dimension is proportional to ``x_scale[j]``. Improved convergence may 
354          be achieved by setting `x_scale` such that a step of a given size 
355          along any of the scaled variables has a similar effect on the cost 
356          function. If set to 'jac', the scale is iteratively updated using the 
357          inverse norms of the columns of the Jacobian matrix (as described in 
358          [JJMore]_). 
359      loss : str or callable, optional 
360          Determines the loss function. The following keyword values are allowed: 
361   
362          * 'linear' (default) : ``rho(z) = z``. Gives a standard 
363            least-squares problem. 
364          * 'soft_l1' : ``rho(z) = 2 * ((1 + z)**0.5 - 1)``. The smooth 
365            approximation of l1 (absolute value) loss. Usually a good 
366            choice for robust least squares. 
367          * 'huber' : ``rho(z) = z if z <= 1 else 2*z**0.5 - 1``. Works 
368            similarly to 'soft_l1'. 
369          * 'cauchy' : ``rho(z) = ln(1 + z)``. Severely weakens outliers 
370            influence, but may cause difficulties in optimization process. 
371          * 'arctan' : ``rho(z) = arctan(z)``. Limits a maximum loss on 
372            a single residual, has properties similar to 'cauchy'. 
373   
374          If callable, it must take a 1-D ndarray ``z=f**2`` and return an 
375          array_like with shape (3, m) where row 0 contains function values, 
376          row 1 contains first derivatives and row 2 contains second 
377          derivatives. Method 'lm' supports only 'linear' loss. 
378      f_scale : float, optional 
379          Value of soft margin between inlier and outlier residuals, default 
380          is 1.0. The loss function is evaluated as follows 
381          ``rho_(f**2) = C**2 * rho(f**2 / C**2)``, where ``C`` is `f_scale`, 
382          and ``rho`` is determined by `loss` parameter. This parameter has 
383          no effect with ``loss='linear'``, but for other `loss` values it is 
384          of crucial importance. 
385      max_nfev : None or int, optional 
386          Maximum number of function evaluations before the termination. 
387          If None (default), the value is chosen automatically: 
388   
389          * For 'trf' and 'dogbox' : 100 * n. 
390          * For 'lm' :  100 * n if `jac` is callable and 100 * n * (n + 1) 
391            otherwise (because 'lm' counts function calls in Jacobian 
392            estimation). 
393   
394      diff_step : None or array_like, optional 
395          Determines the relative step size for the finite difference 
396          approximation of the Jacobian. The actual step is computed as 
397          ``x * diff_step``. If None (default), then `diff_step` is taken to be 
398          a conventional "optimal" power of machine epsilon for the finite 
399          difference scheme used [NR]_. 
400      tr_solver : {None, 'exact', 'lsmr'}, optional 
401          Method for solving trust-region subproblems, relevant only for 'trf' 
402          and 'dogbox' methods. 
403   
404          * 'exact' is suitable for not very large problems with dense 
405            Jacobian matrices. The computational complexity per iteration is 
406            comparable to a singular value decomposition of the Jacobian 
407            matrix. 
408          * 'lsmr' is suitable for problems with sparse and large Jacobian 
409            matrices. It uses the iterative procedure 
410            `scipy.sparse.linalg.lsmr` for finding a solution of a linear 
411            least-squares problem and only requires matrix-vector product 
412            evaluations. 
413   
414          If None (default), the solver is chosen based on the type of Jacobian 
415          returned on the first iteration. 
416      tr_options : dict, optional 
417          Keyword options passed to trust-region solver. 
418   
419          * ``tr_solver='exact'``: `tr_options` are ignored. 
420          * ``tr_solver='lsmr'``: options for `scipy.sparse.linalg.lsmr`. 
421            Additionally,  ``method='trf'`` supports  'regularize' option 
422            (bool, default is True), which adds a regularization term to the 
423            normal equation, which improves convergence if the Jacobian is 
424            rank-deficient [Byrd]_ (eq. 3.4). 
425   
426      jac_sparsity : {None, array_like, sparse matrix}, optional 
427          Defines the sparsity structure of the Jacobian matrix for finite 
428          difference estimation, its shape must be (m, n). If the Jacobian has 
429          only few non-zero elements in *each* row, providing the sparsity 
430          structure will greatly speed up the computations [Curtis]_. A zero 
431          entry means that a corresponding element in the Jacobian is identically 
432          zero. If provided, forces the use of 'lsmr' trust-region solver. 
433          If None (default), then dense differencing will be used. Has no effect 
434          for 'lm' method. 
435      verbose : {0, 1, 2}, optional 
436          Level of algorithm's verbosity: 
437   
438          * 0 (default) : work silently. 
439          * 1 : display a termination report. 
440          * 2 : display progress during iterations (not supported by 'lm' 
441            method). 
442   
443      args, kwargs : tuple and dict, optional 
444          Additional arguments passed to `fun` and `jac`. Both empty by default. 
445          The calling signature is ``fun(x, *args, **kwargs)`` and the same for 
446          `jac`. 
447   
448      Returns 
449      ------- 
450      result : OptimizeResult 
451          `OptimizeResult` with the following fields defined: 
452   
453          x : ndarray, shape (n,) 
454              Solution found. 
455          cost : float 
456              Value of the cost function at the solution. 
457          fun : ndarray, shape (m,) 
458              Vector of residuals at the solution. 
459          jac : ndarray, sparse matrix or LinearOperator, shape (m, n) 
460              Modified Jacobian matrix at the solution, in the sense that J^T J 
461              is a Gauss-Newton approximation of the Hessian of the cost function. 
462              The type is the same as the one used by the algorithm. 
463          grad : ndarray, shape (m,) 
464              Gradient of the cost function at the solution. 
465          optimality : float 
466              First-order optimality measure. In unconstrained problems, it is 
467              always the uniform norm of the gradient. In constrained problems, 
468              it is the quantity which was compared with `gtol` during iterations. 
469          active_mask : ndarray of int, shape (n,) 
470              Each component shows whether a corresponding constraint is active 
471              (that is, whether a variable is at the bound): 
472   
473              *  0 : a constraint is not active. 
474              * -1 : a lower bound is active. 
475              *  1 : an upper bound is active. 
476   
477              Might be somewhat arbitrary for 'trf' method as it generates a 
478              sequence of strictly feasible iterates and `active_mask` is 
479              determined within a tolerance threshold. 
480          nfev : int 
481              Number of function evaluations done. Methods 'trf' and 'dogbox' do 
482              not count function calls for numerical Jacobian approximation, as 
483              opposed to 'lm' method. 
484          njev : int or None 
485              Number of Jacobian evaluations done. If numerical Jacobian 
486              approximation is used in 'lm' method, it is set to None. 
487          status : int 
488              The reason for algorithm termination: 
489   
490              * -1 : improper input parameters status returned from MINPACK. 
491              *  0 : the maximum number of function evaluations is exceeded. 
492              *  1 : `gtol` termination condition is satisfied. 
493              *  2 : `ftol` termination condition is satisfied. 
494              *  3 : `xtol` termination condition is satisfied. 
495              *  4 : Both `ftol` and `xtol` termination conditions are satisfied. 
496   
497          message : str 
498              Verbal description of the termination reason. 
499          success : bool 
500              True if one of the convergence criteria is satisfied (`status` > 0). 
501

508      Notes 
509      ----- 
510      Method 'lm' (Levenberg-Marquardt) calls a wrapper over least-squares 
511      algorithms implemented in MINPACK (lmder, lmdif). It runs the 
512      Levenberg-Marquardt algorithm formulated as a trust-region type algorithm. 
513      The implementation is based on paper [JJMore]_, it is very robust and 
514      efficient with a lot of smart tricks. It should be your first choice 
515      for unconstrained problems. Note that it doesn't support bounds. Also, 
516      it doesn't work when m < n. 
517   
518      Method 'trf' (Trust Region Reflective) is motivated by the process of 
519      solving a system of equations, which constitute the first-order optimality 
520      condition for a bound-constrained minimization problem as formulated in 
521      [STIR]_. The algorithm iteratively solves trust-region subproblems 
522      augmented by a special diagonal quadratic term and with trust-region shape 
523      determined by the distance from the bounds and the direction of the 
524      gradient. This enhancements help to avoid making steps directly into bounds 
525      and efficiently explore the whole space of variables. To further improve 
526      convergence, the algorithm considers search directions reflected from the 
527      bounds. To obey theoretical requirements, the algorithm keeps iterates 
528      strictly feasible. With dense Jacobians trust-region subproblems are 
529      solved by an exact method very similar to the one described in [JJMore]_ 
530      (and implemented in MINPACK). The difference from the MINPACK 
531      implementation is that a singular value decomposition of a Jacobian 
532      matrix is done once per iteration, instead of a QR decomposition and series 
533      of Givens rotation eliminations. For large sparse Jacobians a 2-D subspace 
534      approach of solving trust-region subproblems is used [STIR]_, [Byrd]_. 
535      The subspace is spanned by a scaled gradient and an approximate 
536      Gauss-Newton solution delivered by `scipy.sparse.linalg.lsmr`. When no 
537      constraints are imposed the algorithm is very similar to MINPACK and has 
538      generally comparable performance. The algorithm works quite robust in 
539      unbounded and bounded problems, thus it is chosen as a default algorithm. 
540   
541      Method 'dogbox' operates in a trust-region framework, but considers 
542      rectangular trust regions as opposed to conventional ellipsoids [Voglis]_. 
543      The intersection of a current trust region and initial bounds is again 
544      rectangular, so on each iteration a quadratic minimization problem subject 
545      to bound constraints is solved approximately by Powell's dogleg method 
546      [NumOpt]_. The required Gauss-Newton step can be computed exactly for 
547      dense Jacobians or approximately by `scipy.sparse.linalg.lsmr` for large 
548      sparse Jacobians. The algorithm is likely to exhibit slow convergence when 
549      the rank of Jacobian is less than the number of variables. The algorithm 
550      often outperforms 'trf' in bounded problems with a small number of 
551      variables. 
552   
553      Robust loss functions are implemented as described in [BA]_. The idea 
554      is to modify a residual vector and a Jacobian matrix on each iteration 
555      such that computed gradient and Gauss-Newton Hessian approximation match 
556      the true gradient and Hessian approximation of the cost function. Then 
557      the algorithm proceeds in a normal way, i.e., robust loss functions are 
558      implemented as a simple wrapper over standard least-squares algorithms.

590      Examples 
591      -------- 
592      In this example we find a minimum of the Rosenbrock function without bounds 
593      on independent variables. 
594   
595      >>> import numpy as np 
596      >>> def fun_rosenbrock(x): 
597      ...     return np.array([10 * (x[1] - x[0]**2), (1 - x[0])]) 
598   
599      Notice that we only provide the vector of the residuals. The algorithm 
600      constructs the cost function as a sum of squares of the residuals, which 
601      gives the Rosenbrock function. The exact minimum is at ``x = [1.0, 1.0]``. 
602   
603      >>> from scipy.optimize import least_squares 
604      >>> x0_rosenbrock = np.array([2, 2]) 
605      >>> res_1 = least_squares(fun_rosenbrock, x0_rosenbrock) 
606      >>> res_1.x 
607      array([ 1.,  1.]) 
608      >>> res_1.cost 
609      9.8669242910846867e-30 
610      >>> res_1.optimality 
611      8.8928864934219529e-14 
612   
613      We now constrain the variables, in such a way that the previous solution 
614      becomes infeasible. Specifically, we require that ``x[1] >= 1.5``, and 
615      ``x[0]`` left unconstrained. To this end, we specify the `bounds` parameter 
616      to `least_squares` in the form ``bounds=([-np.inf, 1.5], np.inf)``. 
617   
618      We also provide the analytic Jacobian: 
619   
620      >>> def jac_rosenbrock(x): 
621      ...     return np.array([ 
622      ...         [-20 * x[0], 10], 
623      ...         [-1, 0]]) 
624   
625      Putting this all together, we see that the new solution lies on the bound: 
626   
627      >>> res_2 = least_squares(fun_rosenbrock, x0_rosenbrock, jac_rosenbrock, 
628      ...                       bounds=([-np.inf, 1.5], np.inf)) 
629      >>> res_2.x 
630      array([ 1.22437075,  1.5       ]) 
631      >>> res_2.cost 
632      0.025213093946805685 
633      >>> res_2.optimality 
634      1.5885401433157753e-07 
635   
636      Now we solve a system of equations (i.e., the cost function should be zero 
637      at a minimum) for a Broyden tridiagonal vector-valued function of 100000 
638      variables: 
639   
640      >>> def fun_broyden(x): 
641      ...     f = (3 - x) * x + 1 
642      ...     f[1:] -= x[:-1] 
643      ...     f[:-1] -= 2 * x[1:] 
644      ...     return f 
645   
646      The corresponding Jacobian matrix is sparse. We tell the algorithm to 
647      estimate it by finite differences and provide the sparsity structure of 
648      Jacobian to significantly speed up this process. 
649   
650      >>> from scipy.sparse import lil_matrix 
651      >>> def sparsity_broyden(n): 
652      ...     sparsity = lil_matrix((n, n), dtype=int) 
653      ...     i = np.arange(n) 
654      ...     sparsity[i, i] = 1 
655      ...     i = np.arange(1, n) 
656      ...     sparsity[i, i - 1] = 1 
657      ...     i = np.arange(n - 1) 
658      ...     sparsity[i, i + 1] = 1 
659      ...     return sparsity 
660      ... 
661      >>> n = 100000 
662      >>> x0_broyden = -np.ones(n) 
663      ... 
664      >>> res_3 = least_squares(fun_broyden, x0_broyden, 
665      ...                       jac_sparsity=sparsity_broyden(n)) 
666      >>> res_3.cost 
667      4.5687069299604613e-23 
668      >>> res_3.optimality 
669      1.1650454296851518e-11 
670   
671      Let's also solve a curve fitting problem using robust loss function to 
672      take care of outliers in the data. Define the model function as 
673      ``y = a + b * exp(c * t)``, where t is a predictor variable, y is an 
674      observation and a, b, c are parameters to estimate. 
675   
676      First, define the function which generates the data with noise and 
677      outliers, define the model parameters, and generate data: 
678   
679      >>> from numpy.random import default_rng 
680      >>> rng = default_rng() 
681      >>> def gen_data(t, a, b, c, noise=0., n_outliers=0, seed=None): 
682      ...     rng = default_rng(seed) 
683      ... 
684      ...     y = a + b * np.exp(t * c) 
685      ... 
686      ...     error = noise * rng.standard_normal(t.size) 
687      ...     outliers = rng.integers(0, t.size, n_outliers) 
688      ...     error[outliers] *= 10 
689      ... 
690      ...     return y + error 
691      ... 
692      >>> a = 0.5 
693      >>> b = 2.0 
694      >>> c = -1 
695      >>> t_min = 0 
696      >>> t_max = 10 
697      >>> n_points = 15 
698      ... 
699      >>> t_train = np.linspace(t_min, t_max, n_points) 
700      >>> y_train = gen_data(t_train, a, b, c, noise=0.1, n_outliers=3) 
701   
702      Define function for computing residuals and initial estimate of 
703      parameters. 
704   
705      >>> def fun(x, t, y): 
706      ...     return x[0] + x[1] * np.exp(x[2] * t) - y 
707      ... 
708      >>> x0 = np.array([1.0, 1.0, 0.0]) 
709   
710      Compute a standard least-squares solution: 
711   
712      >>> res_lsq = least_squares(fun, x0, args=(t_train, y_train)) 
713   
714      Now compute two solutions with two different robust loss functions. The 
715      parameter `f_scale` is set to 0.1, meaning that inlier residuals should 
716      not significantly exceed 0.1 (the noise level used). 
717   
718      >>> res_soft_l1 = least_squares(fun, x0, loss='soft_l1', f_scale=0.1, 
719      ...                             args=(t_train, y_train)) 
720      >>> res_log = least_squares(fun, x0, loss='cauchy', f_scale=0.1, 
721      ...                         args=(t_train, y_train)) 
722   
723      And, finally, plot all the curves. We see that by selecting an appropriate 
724      `loss`  we can get estimates close to optimal even in the presence of 
725      strong outliers. But keep in mind that generally it is recommended to try 
726      'soft_l1' or 'huber' losses first (if at all necessary) as the other two 
727      options may cause difficulties in optimization process. 
728   
729      >>> t_test = np.linspace(t_min, t_max, n_points * 10) 
730      >>> y_true = gen_data(t_test, a, b, c) 
731      >>> y_lsq = gen_data(t_test, *res_lsq.x) 
732      >>> y_soft_l1 = gen_data(t_test, *res_soft_l1.x) 
733      >>> y_log = gen_data(t_test, *res_log.x) 
734      ... 
735      >>> import matplotlib.pyplot as plt 
736      >>> plt.plot(t_train, y_train, 'o') 
737      >>> plt.plot(t_test, y_true, 'k', linewidth=2, label='true') 
738      >>> plt.plot(t_test, y_lsq, label='linear loss') 
739      >>> plt.plot(t_test, y_soft_l1, label='soft_l1 loss') 
740      >>> plt.plot(t_test, y_log, label='cauchy loss') 
741      >>> plt.xlabel("t") 
742      >>> plt.ylabel("y") 
743      >>> plt.legend() 
744      >>> plt.show() 
745   
746      In the next example, we show how complex-valued residual functions of 
747      complex variables can be optimized with ``least_squares()``. Consider the 
748      following function: 
749   
750      >>> def f(z): 
751      ...     return z - (0.5 + 0.5j) 
752   
753      We wrap it into a function of real variables that returns real residuals 
754      by simply handling the real and imaginary parts as independent variables: 
755   
756      >>> def f_wrap(x): 
757      ...     fx = f(x[0] + 1j*x[1]) 
758      ...     return np.array([fx.real, fx.imag]) 
759   
760      Thus, instead of the original m-D complex function of n complex 
761      variables we optimize a 2m-D real function of 2n real variables: 
762   
763      >>> from scipy.optimize import least_squares 
764      >>> res_wrapped = least_squares(f_wrap, (0.1, 0.1), bounds=([0, 0], [1, 1])) 
765      >>> z = res_wrapped.x[0] + res_wrapped.x[1]*1j 
766      >>> z 
767      (0.49999999999925893+0.49999999999925893j) 
768   
769      """
770      if method not in ['trf', 'dogbox', 'lm']:
771          raise ValueError("`method` must be 'trf', 'dogbox' or 'lm'.")
772  
773      if jac not in ['2-point', '3-point', 'cs'] and not callable(jac):
774          raise ValueError("`jac` must be '2-point', '3-point', 'cs' or "
775                           "callable.")
776  
777      if tr_solver not in [None, 'exact', 'lsmr']:
778          raise ValueError("`tr_solver` must be None, 'exact' or 'lsmr'.")
779  
780      if loss not in IMPLEMENTED_LOSSES and not callable(loss):
781          raise ValueError(f"`loss` must be one of {IMPLEMENTED_LOSSES.keys()}"
782                           " or a callable.")
783  
784      if method == 'lm' and loss != 'linear':
785          raise ValueError("method='lm' supports only 'linear' loss function.")
786  
787      if verbose not in [0, 1, 2]:
788          raise ValueError("`verbose` must be in [0, 1, 2].")
789  
790      if max_nfev is not None and max_nfev <= 0:
791          raise ValueError("`max_nfev` must be None or positive integer.")
792  
793      if np.iscomplexobj(x0):
794          raise ValueError("`x0` must be real.")
795  
796      x0 = np.atleast_1d(x0).astype(float)
797  
798      if x0.ndim > 1:
799          raise ValueError("`x0` must have at most 1 dimension.")
800  
801      if isinstance(bounds, Bounds):
802          lb, ub = bounds.lb, bounds.ub
803          bounds = (lb, ub)
804      else:
805          if len(bounds) == 2:
806              lb, ub = prepare_bounds(bounds, x0.shape[0])
807          else:
808              raise ValueError("`bounds` must contain 2 elements.")
809  
810      if method == 'lm' and not np.all((lb == -np.inf) & (ub == np.inf)):
811          raise ValueError("Method 'lm' doesn't support bounds.")
812  
813      if lb.shape != x0.shape or ub.shape != x0.shape:
814          raise ValueError("Inconsistent shapes between bounds and `x0`.")
815  
816      if np.any(lb >= ub):
817          raise ValueError("Each lower bound must be strictly less than each "
818                           "upper bound.")
819  
820      if not in_bounds(x0, lb, ub):
821          raise ValueError("Initial guess is outside of provided bounds")
822  
823      x_scale = check_x_scale(x_scale, x0)
824  
825      ftol, xtol, gtol = check_tolerance(ftol, xtol, gtol, method)
826  
827      if method == 'trf':
828          x0 = make_strictly_feasible(x0, lb, ub)
829  
830      if kwargs is None:
831          kwargs = {}
832      if tr_options is None:
833          tr_options = {}
834  
835      def fun_wrapped(x):
836          return np.atleast_1d(fun(x, *args, **kwargs))
837  
838      f0 = fun_wrapped(x0)
839  
840      if f0.ndim != 1:
841          raise ValueError("`fun` must return at most 1-d array_like. "
842                           f"f0.shape: {f0.shape}")
843  
844      if not np.all(np.isfinite(f0)):
845          raise ValueError("Residuals are not finite in the initial point.")
846  
847      n = x0.size
848      m = f0.size
849  
850      if method == 'lm' and m < n:
851          raise ValueError("Method 'lm' doesn't work when the number of "
852                           "residuals is less than the number of variables.")
853  
854      loss_function = construct_loss_function(m, loss, f_scale)
855      if callable(loss):
856          rho = loss_function(f0)
857          if rho.shape != (3, m):
858              raise ValueError("The return value of `loss` callable has wrong "
859                               "shape.")
860          initial_cost = 0.5 * np.sum(rho[0])
861      elif loss_function is not None:
862          initial_cost = loss_function(f0, cost_only=True)
863      else:
864          initial_cost = 0.5 * np.dot(f0, f0)
865  
866      if callable(jac):
867          J0 = jac(x0, *args, **kwargs)
868  
869          if issparse(J0):
870              J0 = J0.tocsr()
871  
872              def jac_wrapped(x, _=None):
873                  return jac(x, *args, **kwargs).tocsr()
874  
875          elif isinstance(J0, LinearOperator):
876              def jac_wrapped(x, _=None):
877                  return jac(x, *args, **kwargs)
878  
879          else:
880              J0 = np.atleast_2d(J0)
881  
882              def jac_wrapped(x, _=None):
883                  return np.atleast_2d(jac(x, *args, **kwargs))
884  
885      else:  # Estimate Jacobian by finite differences.
886          if method == 'lm':
887              if jac_sparsity is not None:
888                  raise ValueError("method='lm' does not support "
889                                   "`jac_sparsity`.")
890  
891              if jac != '2-point':
892                  warn(f"jac='{jac}' works equivalently to '2-point' for method='lm'.",
893                       stacklevel=2)
894  
895              J0 = jac_wrapped = None
896          else:
897              if jac_sparsity is not None and tr_solver == 'exact':
898                  raise ValueError("tr_solver='exact' is incompatible "
899                                   "with `jac_sparsity`.")
900  
901              jac_sparsity = check_jac_sparsity(jac_sparsity, m, n)
902  
903              def jac_wrapped(x, f):
904                  J = approx_derivative(fun, x, rel_step=diff_step, method=jac,
905                                        f0=f, bounds=bounds, args=args,
906                                        kwargs=kwargs, sparsity=jac_sparsity)
907                  if J.ndim != 2:  # J is guaranteed not sparse.
908                      J = np.atleast_2d(J)
909  
910                  return J
911  
912              J0 = jac_wrapped(x0, f0)
913  
914      if J0 is not None:
915          if J0.shape != (m, n):
916              raise ValueError(
917                  f"The return value of `jac` has wrong shape: expected {(m, n)}, "
918                  f"actual {J0.shape}."
919              )
920  
921          if not isinstance(J0, np.ndarray):
922              if method == 'lm':
923                  raise ValueError("method='lm' works only with dense "
924                                   "Jacobian matrices.")
925  
926              if tr_solver == 'exact':
927                  raise ValueError(
928                      "tr_solver='exact' works only with dense "
929                      "Jacobian matrices.")
930  
931          jac_scale = isinstance(x_scale, str) and x_scale == 'jac'
932          if isinstance(J0, LinearOperator) and jac_scale:
933              raise ValueError("x_scale='jac' can't be used when `jac` "
934                               "returns LinearOperator.")
935  
936          if tr_solver is None:
937              if isinstance(J0, np.ndarray):
938                  tr_solver = 'exact'
939              else:
940                  tr_solver = 'lsmr'
941  
942      if method == 'lm':
943          result = call_minpack(fun_wrapped, x0, jac_wrapped, ftol, xtol, gtol,
944                                max_nfev, x_scale, diff_step)
945  
946      elif method == 'trf':
947          result = trf(fun_wrapped, jac_wrapped, x0, f0, J0, lb, ub, ftol, xtol,
948                       gtol, max_nfev, x_scale, loss_function, tr_solver,
949                       tr_options.copy(), verbose)
950  
951      elif method == 'dogbox':
952          if tr_solver == 'lsmr' and 'regularize' in tr_options:
953              warn("The keyword 'regularize' in `tr_options` is not relevant "
954                   "for 'dogbox' method.",
955                   stacklevel=2)
956              tr_options = tr_options.copy()
957              del tr_options['regularize']
958  
959          result = dogbox(fun_wrapped, jac_wrapped, x0, f0, J0, lb, ub, ftol,
960                          xtol, gtol, max_nfev, x_scale, loss_function,
961                          tr_solver, tr_options, verbose)
962  
963      result.message = TERMINATION_MESSAGES[result.status]
964      result.success = result.status > 0
965  
966      if verbose >= 1:
967          print(result.message)
968          print(f"Function evaluations {result.nfev}, initial cost {initial_cost:.4e}, "
969                f"final cost {result.cost:.4e}, "
970                f"first-order optimality {result.optimality:.2e}.")
971  
972      return result
