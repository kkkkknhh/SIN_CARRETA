4    from numpy.testing import (assert_allclose,
5                               assert_array_less)
6    import pytest
7    import numpy as np
8    from scipy.optimize import direct, Bounds
9    import threading

14       def setup_method(self):
15           self.fun_calls = threading.local()
16           self.bounds_sphere = 4*[(-2, 3)]
17           self.optimum_sphere_pos = np.zeros((4, ))
18           self.optimum_sphere = 0.0
19           self.bounds_stylinski_tang = Bounds([-4., -4.], [4., 4.])
20           self.maxiter = 1000
21   
22       # test functions
23       def sphere(self, x):
24           if not hasattr(self.fun_calls, 'c'):
25               self.fun_calls.c = 0
26           self.fun_calls.c += 1
27           return np.square(x).sum()
28   
29       def inv(self, x):
30           if np.sum(x) == 0:
31               raise ZeroDivisionError()
32           return 1/np.sum(x)
33   
34       def nan_fun(self, x):
35           return np.nan
36   
37       def inf_fun(self, x):
38           return np.inf
39   
40       def styblinski_tang(self, pos):
41           x, y = pos
42           return 0.5 * (x**4 - 16 * x**2 + 5 * x + y**4 - 16 * y**2 + 5 * y)
43   
44       @pytest.mark.parametrize("locally_biased", [True, False])
45       def test_direct(self, locally_biased):
46           res = direct(self.sphere, self.bounds_sphere,
47                        locally_biased=locally_biased)
48   
49           # test accuracy
50           assert_allclose(res.x, self.optimum_sphere_pos,
51                           rtol=1e-3, atol=1e-3)
52           assert_allclose(res.fun, self.optimum_sphere, atol=1e-5, rtol=1e-5)
53   
54           # test that result lies within bounds
55           _bounds = np.asarray(self.bounds_sphere)
56           assert_array_less(_bounds[:, 0], res.x)
57           assert_array_less(res.x, _bounds[:, 1])
58   
59           # test number of function evaluations. Original DIRECT overshoots by
60           # up to 500 evaluations in last iteration
61           assert res.nfev <= 1000 * (len(self.bounds_sphere) + 1)
62           # test that number of function evaluations is correct
63           assert res.nfev == self.fun_calls.c
64   
65           # test that number of iterations is below supplied maximum
66           assert res.nit <= self.maxiter

68       @pytest.mark.parametrize("locally_biased", [True, False])
69       def test_direct_callback(self, locally_biased):
70           # test that callback does not change the result
71           res = direct(self.sphere, self.bounds_sphere,
72                        locally_biased=locally_biased)
73   
74           def callback(x):
75               x = 2*x
76               dummy = np.square(x)
77               print("DIRECT minimization algorithm callback test")
78               return dummy
79   
80           res_callback = direct(self.sphere, self.bounds_sphere,
81                                 locally_biased=locally_biased,
82                                 callback=callback)
83   
84           assert_allclose(res.x, res_callback.x)
85   
86           assert res.nit == res_callback.nit
87           assert res.nfev == res_callback.nfev
88           assert res.status == res_callback.status
89           assert res.success == res_callback.success
90           assert res.fun == res_callback.fun
91           assert_allclose(res.x, res_callback.x)
92           assert res.message == res_callback.message
93   
94           # test accuracy
95           assert_allclose(res_callback.x, self.optimum_sphere_pos,
96                           rtol=1e-3, atol=1e-3)
97           assert_allclose(res_callback.fun, self.optimum_sphere,
98                           atol=1e-5, rtol=1e-5)

140      @pytest.mark.parametrize("f_min_rtol", [1e-3, 1e-5, 1e-7])
141      @pytest.mark.parametrize("locally_biased", [True, False])
142      def test_f_min(self, f_min_rtol, locally_biased):
143          # test that desired function value is reached within
144          # relative tolerance of f_min_rtol
145          f_min = 1.
146          bounds = 4*[(-2., 10.)]
147          res = direct(self.sphere, bounds=bounds, f_min=f_min,
148                       f_min_rtol=f_min_rtol,
149                       locally_biased=locally_biased)
150          assert res.status == 3
151          assert res.success
152          assert res.fun < f_min * (1. + f_min_rtol)
153          message = ("The best function value found is within a relative "
154                     f"error={f_min_rtol} of the (known) global optimum f_min")
155          assert res.message == message

197      @pytest.mark.parametrize("locally_biased", [True, False])
198      def test_bounds_variants(self, locally_biased):
199          # test that new and old bounds yield same result
200  
201          lb = [-6., 1., -5.]
202          ub = [-1., 3., 5.]
203          x_opt = np.array([-1., 1., 0.])
204          bounds_old = list(zip(lb, ub))
205          bounds_new = Bounds(lb, ub)
206  
207          res_old_bounds = direct(self.sphere, bounds_old,
208                                  locally_biased=locally_biased)
209          res_new_bounds = direct(self.sphere, bounds_new,
210                                  locally_biased=locally_biased)
211  
212          assert res_new_bounds.nfev == res_old_bounds.nfev
213          assert res_new_bounds.message == res_old_bounds.message
214          assert res_new_bounds.success == res_old_bounds.success
215          assert res_new_bounds.nit == res_old_bounds.nit
216          assert_allclose(res_new_bounds.x, res_old_bounds.x)
217          assert_allclose(res_new_bounds.x, x_opt, rtol=1e-2)
