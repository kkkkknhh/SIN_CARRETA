16   def prepare_input(x, y, axis, dydx=None):
17       """Prepare input for cubic spline interpolators. 
18    
19       All data are converted to numpy arrays and checked for correctness. 
20       Axes equal to `axis` of arrays `y` and `dydx` are moved to be the 0th 
21       axis. The value of `axis` is converted to lie in 
22       [0, number of dimensions of `y`). 
23       """
24   
25       x, y = map(np.asarray, (x, y))
26       if np.issubdtype(x.dtype, np.complexfloating):
27           raise ValueError("`x` must contain real values.")
28       x = x.astype(float)
29   
30       if np.issubdtype(y.dtype, np.complexfloating):
31           dtype = complex
32       else:
33           dtype = float
34   
35       if dydx is not None:
36           dydx = np.asarray(dydx)
37           if y.shape != dydx.shape:
38               raise ValueError("The shapes of `y` and `dydx` must be identical.")
39           if np.issubdtype(dydx.dtype, np.complexfloating):
40               dtype = complex
41           dydx = dydx.astype(dtype, copy=False)
42   
43       y = y.astype(dtype, copy=False)
44       axis = axis % y.ndim
45       if x.ndim != 1:
46           raise ValueError("`x` must be 1-dimensional.")
47       if x.shape[0] < 2:
48           raise ValueError("`x` must contain at least 2 elements.")
49       if x.shape[0] != y.shape[axis]:
50           raise ValueError(f"The length of `y` along `axis`={axis} doesn't "
51                            "match the length of `x`")
52   
53       if not np.all(np.isfinite(x)):
54           raise ValueError("`x` must contain only finite values.")
55       if not np.all(np.isfinite(y)):
56           raise ValueError("`y` must contain only finite values.")
57   
58       if dydx is not None and not np.all(np.isfinite(dydx)):
59           raise ValueError("`dydx` must contain only finite values.")
60   
61       dx = np.diff(x)
62       if np.any(dx <= 0):
63           raise ValueError("`x` must be strictly increasing sequence.")
64   
65       y = np.moveaxis(y, axis, 0)
66       if dydx is not None:
67           dydx = np.moveaxis(dydx, axis, 0)
68   
69       return x, dx, y, axis, dydx

72   class CubicHermiteSpline(PPoly):
73       """Piecewise-cubic interpolator matching values and first derivatives. 
74    
75       The result is represented as a `PPoly` instance. 
76    
77       Parameters 
78       ---------- 
79       x : array_like, shape (n,) 
80           1-D array containing values of the independent variable. 
81           Values must be real, finite and in strictly increasing order. 
82       y : array_like 
83           Array containing values of the dependent variable. It can have 
84           arbitrary number of dimensions, but the length along ``axis`` 
85           (see below) must match the length of ``x``. Values must be finite. 
86       dydx : array_like 
87           Array containing derivatives of the dependent variable. It can have 
88           arbitrary number of dimensions, but the length along ``axis`` 
89           (see below) must match the length of ``x``. Values must be finite. 
90       axis : int, optional 
91           Axis along which `y` is assumed to be varying. Meaning that for 
92           ``x[i]`` the corresponding values are ``np.take(y, i, axis=axis)``. 
93           Default is 0. 
94       extrapolate : {bool, 'periodic', None}, optional 
95           If bool, determines whether to extrapolate to out-of-bounds points 
96           based on first and last intervals, or to return NaNs. If 'periodic', 
97           periodic extrapolation is used. If None (default), it is set to True. 
98    
99       Attributes 
100      ---------- 
101      x : ndarray, shape (n,) 
102          Breakpoints. The same ``x`` which was passed to the constructor. 
103      c : ndarray, shape (4, n-1, ...) 
104          Coefficients of the polynomials on each segment. The trailing 
105          dimensions match the dimensions of `y`, excluding ``axis``. 
106          For example, if `y` is 1-D, then ``c[k, i]`` is a coefficient for 
107          ``(x-x[i])**(3-k)`` on the segment between ``x[i]`` and ``x[i+1]``. 
108      axis : int 
109          Interpolation axis. The same axis which was passed to the 
110          constructor. 
111   
112      Methods 
113      ------- 
114      __call__ 
115      derivative 
116      antiderivative 
117      integrate 
118      roots 
119   
120      See Also 
121      -------- 
122      Akima1DInterpolator : Akima 1D interpolator. 
123      PchipInterpolator : PCHIP 1-D monotonic cubic interpolator. 
124      CubicSpline : Cubic spline data interpolator. 
125      PPoly : Piecewise polynomial in terms of coefficients and breakpoints 
126   
127      Notes 
128      ----- 
129      If you want to create a higher-order spline matching higher-order 
130      derivatives, use `BPoly.from_derivatives`. 
131   
132      References 
133      ---------- 
134      .. [1] `Cubic Hermite spline 
135              <https://en.wikipedia.org/wiki/Cubic_Hermite_spline>`_ 
136              on Wikipedia. 
137      """
138  
139      def __init__(self, x, y, dydx, axis=0, extrapolate=None):
140          if extrapolate is None:
141              extrapolate = True
142  
143          x, dx, y, axis, dydx = prepare_input(x, y, axis, dydx)
144  
145          dxr = dx.reshape([dx.shape[0]] + [1] * (y.ndim - 1))
146          slope = np.diff(y, axis=0) / dxr
147          t = (dydx[:-1] + dydx[1:] - 2 * slope) / dxr
148  
149          c = np.empty((4, len(x) - 1) + y.shape[1:], dtype=t.dtype)
150          c[0] = t / dxr
151          c[1] = (slope - dydx[:-1]) / dxr - t
152          c[2] = dydx[:-1]
153          c[3] = y[:-1]
154  
155          super().__init__(c, x, extrapolate=extrapolate)
156          self.axis = axis

159  class PchipInterpolator(CubicHermiteSpline):
160      r"""PCHIP 1-D monotonic cubic interpolation. 
161   
162      ``x`` and ``y`` are arrays of values used to approximate some function f, 
163      with ``y = f(x)``. The interpolant uses monotonic cubic splines 
164      to find the value of new points. (PCHIP stands for Piecewise Cubic 
165      Hermite Interpolating Polynomial). 
166   
167      Parameters 
168      ---------- 
169      x : ndarray, shape (npoints, ) 
170          A 1-D array of monotonically increasing real values. ``x`` cannot 
171          include duplicate values (otherwise f is overspecified) 
172      y : ndarray, shape (..., npoints, ...) 
173          A N-D array of real values. ``y``'s length along the interpolation 
174          axis must be equal to the length of ``x``. Use the ``axis`` 
175          parameter to select the interpolation axis. 
176      axis : int, optional 
177          Axis in the ``y`` array corresponding to the x-coordinate values. Defaults 
178          to ``axis=0``. 
179      extrapolate : bool, optional 
180          Whether to extrapolate to out-of-bounds points based on first 
181          and last intervals, or to return NaNs. 
182   
183      Methods 
184      ------- 
185      __call__ 
186      derivative 
187      antiderivative 
188      roots 
189   
190      See Also 
191      -------- 
192      CubicHermiteSpline : Piecewise-cubic interpolator. 
193      Akima1DInterpolator : Akima 1D interpolator. 
194      CubicSpline : Cubic spline data interpolator. 
195      PPoly : Piecewise polynomial in terms of coefficients and breakpoints. 
196   
197      Notes 
198      ----- 
199      The interpolator preserves monotonicity in the interpolation data and does 
200      not overshoot if the data is not smooth. 
201   
202      The first derivatives are guaranteed to be continuous, but the second 
203      derivatives may jump at :math:`x_k`. 
204   
205      Determines the derivatives at the points :math:`x_k`, :math:`f'_k`, 
206      by using PCHIP algorithm [1]_. 
207   
208      Let :math:`h_k = x_{k+1} - x_k`, and  :math:`d_k = (y_{k+1} - y_k) / h_k` 
209      are the slopes at internal points :math:`x_k`. 
210      If the signs of :math:`d_k` and :math:`d_{k-1}` are different or either of 
211      them equals zero, then :math:`f'_k = 0`. Otherwise, it is given by the 
212      weighted harmonic mean 
213   
214      .. math:: 
215   
216          \frac{w_1 + w_2}{f'_k} = \frac{w_1}{d_{k-1}} + \frac{w_2}{d_k} 
217   
218      where :math:`w_1 = 2 h_k + h_{k-1}` and :math:`w_2 = h_k + 2 h_{k-1}`. 
219   
220      The end slopes are set using a one-sided scheme [2]_. 
221   
222   
223      References 
224      ---------- 
225      .. [1] F. N. Fritsch and J. Butland, 
226             A method for constructing local 
227             monotone piecewise cubic interpolants, 
228             SIAM J. Sci. Comput., 5(2), 300-304 (1984). 
229             :doi:`10.1137/0905021`. 
230      .. [2] see, e.g., C. Moler, Numerical Computing with Matlab, 2004. 
231             :doi:`10.1137/1.9780898717952` 
232   
233      """
234  
235      def __init__(self, x, y, axis=0, extrapolate=None):
236          x, _, y, axis, _ = prepare_input(x, y, axis)
237          if np.iscomplexobj(y):
238              msg = ("`PchipInterpolator` only works with real values for `y`. "
239                     "If you are trying to use the real components of the passed array, "
240                     "use `np.real` on the array before passing to `PchipInterpolator`.")
241              raise ValueError(msg)
242          xp = x.reshape((x.shape[0],) + (1,)*(y.ndim-1))
243          dk = self._find_derivatives(xp, y)
244          super().__init__(x, y, dk, axis=0, extrapolate=extrapolate)
245          self.axis = axis
246  
247      @staticmethod
248      def _edge_case(h0, h1, m0, m1):
249          # one-sided three-point estimate for the derivative
250          d = ((2*h0 + h1)*m0 - h0*m1) / (h0 + h1)
251  
252          # try to preserve shape
253          mask = np.sign(d) != np.sign(m0)
254          mask2 = (np.sign(m0) != np.sign(m1)) & (np.abs(d) > 3.*np.abs(m0))
255          mmm = (~mask) & mask2
256  
257          d[mask] = 0.
258          d[mmm] = 3.*m0[mmm]
259  
260          return d
261  
262      @staticmethod
263      def _find_derivatives(x, y):
264          # Determine the derivatives at the points y_k, d_k, by using
265          #  PCHIP algorithm is:
266          # We choose the derivatives at the point x_k by
267          # Let m_k be the slope of the kth segment (between k and k+1)
268          # If m_k=0 or m_{k-1}=0 or sgn(m_k) != sgn(m_{k-1}) then d_k == 0
269          # else use weighted harmonic mean:
270          #   w_1 = 2h_k + h_{k-1}, w_2 = h_k + 2h_{k-1}
271          #   1/d_k = 1/(w_1 + w_2)*(w_1 / m_k + w_2 / m_{k-1})
272          #   where h_k is the spacing between x_k and x_{k+1}
273          y_shape = y.shape
274          if y.ndim == 1:
275              # So that _edge_case doesn't end up assigning to scalars
276              x = x[:, None]
277              y = y[:, None]
278  
279          hk = x[1:] - x[:-1]
280          mk = (y[1:] - y[:-1]) / hk
281  
282          if y.shape[0] == 2:
283              # edge case: only have two points, use linear interpolation
284              dk = np.zeros_like(y)
285              dk[0] = mk
286              dk[1] = mk
287              return dk.reshape(y_shape)
288  
289          smk = np.sign(mk)
290          condition = (smk[1:] != smk[:-1]) | (mk[1:] == 0) | (mk[:-1] == 0)
291  
292          w1 = 2*hk[1:] + hk[:-1]
293          w2 = hk[1:] + 2*hk[:-1]
294  
295          # values where division by zero occurs will be excluded
296          # by 'condition' afterwards
297          with np.errstate(divide='ignore', invalid='ignore'):
298              whmean = (w1/mk[:-1] + w2/mk[1:]) / (w1 + w2)
299  
300          dk = np.zeros_like(y)
301          dk[1:-1][condition] = 0.0
302          dk[1:-1][~condition] = 1.0 / whmean[~condition]
303  
304          # special case endpoints, as suggested in
305          # Cleve Moler, Numerical Computing with MATLAB, Chap 3.6 (pchiptx.m)
306          dk[0] = PchipInterpolator._edge_case(hk[0], hk[1], mk[0], mk[1])
307          dk[-1] = PchipInterpolator._edge_case(hk[-1], hk[-2], mk[-1], mk[-2])
308  
309          return dk.reshape(y_shape)
310

312  def pchip_interpolate(xi, yi, x, der=0, axis=0):
313      """ 
314      Convenience function for pchip interpolation. 
315   
316      xi and yi are arrays of values used to approximate some function f, 
317      with ``yi = f(xi)``. The interpolant uses monotonic cubic splines 
318      to find the value of new points x and the derivatives there. 
319   
320      See `scipy.interpolate.PchipInterpolator` for details. 
321   
322      Parameters 
323      ---------- 
324      xi : array_like 
325          A sorted list of x-coordinates, of length N. 
326      yi : array_like 
327          A 1-D array of real values. `yi`'s length along the interpolation 
328          axis must be equal to the length of `xi`. If N-D array, use axis 
329          parameter to select correct axis. 
330   
331          .. deprecated:: 1.13.0 
332              Complex data is deprecated and will raise an error in 
333              SciPy 1.15.0. If you are trying to use the real components of 
334              the passed array, use ``np.real`` on `yi`. 
335   
336      x : scalar or array_like 
337          Of length M. 
338      der : int or list, optional 
339          Derivatives to extract. The 0th derivative can be included to 
340          return the function value. 
341      axis : int, optional 
342          Axis in the yi array corresponding to the x-coordinate values. 
343   
344      Returns 
345      ------- 
346      y : scalar or array_like 
347          The result, of length R or length M or M by R. 
348   
349      See Also 
350      -------- 
351      PchipInterpolator : PCHIP 1-D monotonic cubic interpolator. 
352   
353      Examples 
354      -------- 
355      We can interpolate 2D observed data using pchip interpolation: 
356   
357      >>> import numpy as np 
358      >>> import matplotlib.pyplot as plt 
359      >>> from scipy.interpolate import pchip_interpolate 
360      >>> x_observed = np.linspace(0.0, 10.0, 11) 
361      >>> y_observed = np.sin(x_observed) 
362      >>> x = np.linspace(min(x_observed), max(x_observed), num=100) 
363      >>> y = pchip_interpolate(x_observed, y_observed, x) 
364      >>> plt.plot(x_observed, y_observed, "o", label="observation") 
365      >>> plt.plot(x, y, label="pchip interpolation") 
366      >>> plt.legend() 
367      >>> plt.show() 
368   
369      """
370      P = PchipInterpolator(xi, yi, axis=axis)
371  
372      if der == 0:
373          return P(x)
374      elif _isscalar(der):
375          return P.derivative(der)(x)
376      else:
377          return [P.derivative(nu)(x) for nu in der]

380  class Akima1DInterpolator(CubicHermiteSpline):
381      r""" 
382      Akima interpolator 
383   
384      Fit piecewise cubic polynomials, given vectors x and y. The interpolation 
385      method by Akima uses a continuously differentiable sub-spline built from 
386      piecewise cubic polynomials. The resultant curve passes through the given 
387      data points and will appear smooth and natural. 
388   
389      Parameters 
390      ---------- 
391      x : ndarray, shape (npoints, ) 
392          1-D array of monotonically increasing real values. 
393      y : ndarray, shape (..., npoints, ...) 
394          N-D array of real values. The length of ``y`` along the interpolation axis 
395          must be equal to the length of ``x``. Use the ``axis`` parameter to 
396          select the interpolation axis. 
397      axis : int, optional 
398          Axis in the ``y`` array corresponding to the x-coordinate values. Defaults 
399          to ``axis=0``. 
400      method : {'akima', 'makima'}, optional 
401          If ``"makima"``, use the modified Akima interpolation [2]_. 
402          Defaults to ``"akima"``, use the Akima interpolation [1]_. 
403   
404          .. versionadded:: 1.13.0 
405   
406      extrapolate : {bool, None}, optional 
407          If bool, determines whether to extrapolate to out-of-bounds points  
408          based on first and last intervals, or to return NaNs. If None,  
409          ``extrapolate`` is set to False. 
410           
411      Methods 
412      ------- 
413      __call__ 
414      derivative 
415      antiderivative 
416      roots 
417   
418      See Also 
419      -------- 
420      PchipInterpolator : PCHIP 1-D monotonic cubic interpolator. 
421      CubicSpline : Cubic spline data interpolator. 
422      PPoly : Piecewise polynomial in terms of coefficients and breakpoints 
423   
424      Notes 
425      ----- 
426      .. versionadded:: 0.14 
427   
428      Use only for precise data, as the fitted curve passes through the given 
429      points exactly. This routine is useful for plotting a pleasingly smooth 
430      curve through a few given points for purposes of plotting. 
431   
432      Let :math:`\delta_i = (y_{i+1} - y_i) / (x_{i+1} - x_i)` be the slopes of 
433      the interval :math:`\left[x_i, x_{i+1}\right)`. Akima's derivative at 
434      :math:`x_i` is defined as: 
435   
436      .. math:: 
437   
438          d_i = \frac{w_1}{w_1 + w_2}\delta_{i-1} + \frac{w_2}{w_1 + w_2}\delta_i 
439   
440      In the Akima interpolation [1]_ (``method="akima"``), the weights are: 
441   
442      .. math:: 
443   
444          \begin{aligned} 
445          w_1 &= |\delta_{i+1} - \delta_i| \\ 
446          w_2 &= |\delta_{i-1} - \delta_{i-2}| 
447          \end{aligned} 
448   
449      In the modified Akima interpolation [2]_ (``method="makima"``), 
450      to eliminate overshoot and avoid edge cases of both numerator and 
451      denominator being equal to 0, the weights are modified as follows: 
452   
453      .. math:: 
454   
455          \begin{align*} 
456          w_1 &= |\delta_{i+1} - \delta_i| + |\delta_{i+1} + \delta_i| / 2 \\ 
457          w_2 &= |\delta_{i-1} - \delta_{i-2}| + |\delta_{i-1} + \delta_{i-2}| / 2 
458          \end{align*} 
459   
460      Examples 
461      -------- 
462      Comparison of ``method="akima"`` and ``method="makima"``: 
463   
464      >>> import numpy as np 
465      >>> from scipy.interpolate import Akima1DInterpolator 
466      >>> import matplotlib.pyplot as plt 
467      >>> x = np.linspace(1, 7, 7) 
468      >>> y = np.array([-1, -1, -1, 0, 1, 1, 1]) 
469      >>> xs = np.linspace(min(x), max(x), num=100) 
470      >>> y_akima = Akima1DInterpolator(x, y, method="akima")(xs) 
471      >>> y_makima = Akima1DInterpolator(x, y, method="makima")(xs) 
472   
473      >>> fig, ax = plt.subplots() 
474      >>> ax.plot(x, y, "o", label="data") 
475      >>> ax.plot(xs, y_akima, label="akima") 
476      >>> ax.plot(xs, y_makima, label="makima") 
477      >>> ax.legend() 
478      >>> fig.show() 
479   
480      The overshoot that occurred in ``"akima"`` has been avoided in ``"makima"``. 
481   
482      References 
483      ---------- 
484      .. [1] A new method of interpolation and smooth curve fitting based 
485             on local procedures. Hiroshi Akima, J. ACM, October 1970, 17(4), 
486             589-602. :doi:`10.1145/321607.321609` 
487      .. [2] Makima Piecewise Cubic Interpolation. Cleve Moler and Cosmin Ionita, 2019. 
488             https://blogs.mathworks.com/cleve/2019/04/29/makima-piecewise-cubic-interpolation/ 
489   
490      """
491  
492      def __init__(self, x, y, axis=0, *, method: Literal["akima", "makima"]="akima", 
493                   extrapolate:bool | None = None):
494          if method not in {"akima", "makima"}:
495              raise NotImplementedError(f"`method`={method} is unsupported.")
496          # Original implementation in MATLAB by N. Shamsundar (BSD licensed), see
497          # https://www.mathworks.com/matlabcentral/fileexchange/1814-akima-interpolation
498          x, dx, y, axis, _ = prepare_input(x, y, axis)
499  
500          if np.iscomplexobj(y):
501              msg = ("`Akima1DInterpolator` only works with real values for `y`. "
502                     "If you are trying to use the real components of the passed array, "
503                     "use `np.real` on the array before passing to "
504                     "`Akima1DInterpolator`.")
505              raise ValueError(msg)
506  
507          # Akima extrapolation historically False; parent class defaults to True.
508          extrapolate = False if extrapolate is None else extrapolate
509  
510          # determine slopes between breakpoints
511          m = np.empty((x.size + 3, ) + y.shape[1:])
512          dx = dx[(slice(None), ) + (None, ) * (y.ndim - 1)]
513          m[2:-2] = np.diff(y, axis=0) / dx
514  
515          # add two additional points on the left ...
516          m[1] = 2. * m[2] - m[3]
517          m[0] = 2. * m[1] - m[2]
518          # ... and on the right
519          m[-2] = 2. * m[-3] - m[-4]
520          m[-1] = 2. * m[-2] - m[-3]
521  
522          # if m1 == m2 != m3 == m4, the slope at the breakpoint is not
523          # defined. This is the fill value:
524          t = .5 * (m[3:] + m[:-3])
525          # get the denominator of the slope t
526          dm = np.abs(np.diff(m, axis=0))
527          if method == "makima":
528              pm = np.abs(m[1:] + m[:-1])
529              f1 = dm[2:] + 0.5 * pm[2:]
530              f2 = dm[:-2] + 0.5 * pm[:-2]
531          else:
532              f1 = dm[2:]
533              f2 = dm[:-2]
534          f12 = f1 + f2
535          # These are the mask of where the slope at breakpoint is defined:
536          ind = np.nonzero(f12 > 1e-9 * np.max(f12, initial=-np.inf))
537          x_ind, y_ind = ind[0], ind[1:]
538          # Set the slope at breakpoint
539          t[ind] = (f1[ind] * m[(x_ind + 1,) + y_ind] +
540                    f2[ind] * m[(x_ind + 2,) + y_ind]) / f12[ind]
541  
542          super().__init__(x, y, t, axis=0, extrapolate=extrapolate)
543          self.axis = axis
544  
545      def extend(self, c, x, right=True):
546          raise NotImplementedError("Extending a 1-D Akima interpolator is not "
547                                    "yet implemented")
548  
549      # These are inherited from PPoly, but they do not produce an Akima
550      # interpolator. Hence stub them out.
551      @classmethod
552      def from_spline(cls, tck, extrapolate=None):
553          raise NotImplementedError("This method does not make sense for "
554                                    "an Akima interpolator.")
555  
556      @classmethod
557      def from_bernstein_basis(cls, bp, extrapolate=None):
558          raise NotImplementedError("This method does not make sense for "
559                                    "an Akima interpolator.")

562  class CubicSpline(CubicHermiteSpline):
563      """Cubic spline data interpolator. 
564   
565      Interpolate data with a piecewise cubic polynomial which is twice 
566      continuously differentiable [1]_. The result is represented as a `PPoly` 
567      instance with breakpoints matching the given data. 
568   
569      Parameters 
570      ---------- 
571      x : array_like, shape (n,) 
572          1-D array containing values of the independent variable. 
573          Values must be real, finite and in strictly increasing order. 
574      y : array_like 
575          Array containing values of the dependent variable. It can have 
576          arbitrary number of dimensions, but the length along ``axis`` 
577          (see below) must match the length of ``x``. Values must be finite. 
578      axis : int, optional 
579          Axis along which `y` is assumed to be varying. Meaning that for 
580          ``x[i]`` the corresponding values are ``np.take(y, i, axis=axis)``. 
581          Default is 0. 
582      bc_type : string or 2-tuple, optional 
583          Boundary condition type. Two additional equations, given by the 
584          boundary conditions, are required to determine all coefficients of 
585          polynomials on each segment [2]_. 
586   
587          If `bc_type` is a string, then the specified condition will be applied 
588          at both ends of a spline. Available conditions are: 
589   
590          * 'not-a-knot' (default): The first and second segment at a curve end 
591            are the same polynomial. It is a good default when there is no 
592            information on boundary conditions. 
593          * 'periodic': The interpolated functions is assumed to be periodic 
594            of period ``x[-1] - x[0]``. The first and last value of `y` must be 
595            identical: ``y[0] == y[-1]``. This boundary condition will result in 
596            ``y'[0] == y'[-1]`` and ``y''[0] == y''[-1]``. 
597          * 'clamped': The first derivative at curves ends are zero. Assuming 
598            a 1D `y`, ``bc_type=((1, 0.0), (1, 0.0))`` is the same condition. 
599          * 'natural': The second derivative at curve ends are zero. Assuming 
600            a 1D `y`, ``bc_type=((2, 0.0), (2, 0.0))`` is the same condition. 
601   
602          If `bc_type` is a 2-tuple, the first and the second value will be 
603          applied at the curve start and end respectively. The tuple values can 
604          be one of the previously mentioned strings (except 'periodic') or a 
605          tuple ``(order, deriv_values)`` allowing to specify arbitrary 
606          derivatives at curve ends: 
607   
608          * `order`: the derivative order, 1 or 2. 
609          * `deriv_value`: array_like containing derivative values, shape must 
610            be the same as `y`, excluding ``axis`` dimension. For example, if 
611            `y` is 1-D, then `deriv_value` must be a scalar. If `y` is 3-D with 
612            the shape (n0, n1, n2) and axis=2, then `deriv_value` must be 2-D 
613            and have the shape (n0, n1). 
614      extrapolate : {bool, 'periodic', None}, optional 
615          If bool, determines whether to extrapolate to out-of-bounds points 
616          based on first and last intervals, or to return NaNs. If 'periodic', 
617          periodic extrapolation is used. If None (default), ``extrapolate`` is 
618          set to 'periodic' for ``bc_type='periodic'`` and to True otherwise. 
619   
620      Attributes 
621      ---------- 
622      x : ndarray, shape (n,) 
623          Breakpoints. The same ``x`` which was passed to the constructor. 
624      c : ndarray, shape (4, n-1, ...) 
625          Coefficients of the polynomials on each segment. The trailing 
626          dimensions match the dimensions of `y`, excluding ``axis``. 
627          For example, if `y` is 1-d, then ``c[k, i]`` is a coefficient for 
628          ``(x-x[i])**(3-k)`` on the segment between ``x[i]`` and ``x[i+1]``. 
629      axis : int 
630          Interpolation axis. The same axis which was passed to the 
631          constructor. 
632

638      integrate

645      PPoly : Piecewise polynomial in terms of coefficients and breakpoints.

647      Notes 
648      ----- 
649      Parameters `bc_type` and ``extrapolate`` work independently, i.e. the 
650      former controls only construction of a spline, and the latter only 
651      evaluation. 
652   
653      When a boundary condition is 'not-a-knot' and n = 2, it is replaced by 
654      a condition that the first derivative is equal to the linear interpolant 
655      slope. When both boundary conditions are 'not-a-knot' and n = 3, the 
656      solution is sought as a parabola passing through given points. 
657   
658      When 'not-a-knot' boundary conditions is applied to both ends, the 
659      resulting spline will be the same as returned by `splrep` (with ``s=0``) 
660      and `InterpolatedUnivariateSpline`, but these two methods use a 
661      representation in B-spline basis.

665      Examples 
666      -------- 
667      In this example the cubic spline is used to interpolate a sampled sinusoid. 
668      You can see that the spline continuity property holds for the first and 
669      second derivatives and violates only for the third derivative. 
670   
671      >>> import numpy as np 
672      >>> from scipy.interpolate import CubicSpline 
673      >>> import matplotlib.pyplot as plt 
674      >>> x = np.arange(10) 
675      >>> y = np.sin(x) 
676      >>> cs = CubicSpline(x, y) 
677      >>> xs = np.arange(-0.5, 9.6, 0.1) 
678      >>> fig, ax = plt.subplots(figsize=(6.5, 4)) 
679      >>> ax.plot(x, y, 'o', label='data') 
680      >>> ax.plot(xs, np.sin(xs), label='true') 
681      >>> ax.plot(xs, cs(xs), label="S") 
682      >>> ax.plot(xs, cs(xs, 1), label="S'") 
683      >>> ax.plot(xs, cs(xs, 2), label="S''") 
684      >>> ax.plot(xs, cs(xs, 3), label="S'''") 
685      >>> ax.set_xlim(-0.5, 9.5) 
686      >>> ax.legend(loc='lower left', ncol=2) 
687      >>> plt.show() 
688   
689      In the second example, the unit circle is interpolated with a spline. A 
690      periodic boundary condition is used. You can see that the first derivative 
691      values, ds/dx=0, ds/dy=1 at the periodic point (1, 0) are correctly 
692      computed. Note that a circle cannot be exactly represented by a cubic 
693      spline. To increase precision, more breakpoints would be required. 
694   
695      >>> theta = 2 * np.pi * np.linspace(0, 1, 5) 
696      >>> y = np.c_[np.cos(theta), np.sin(theta)] 
697      >>> cs = CubicSpline(theta, y, bc_type='periodic') 
698      >>> print("ds/dx={:.1f} ds/dy={:.1f}".format(cs(0, 1)[0], cs(0, 1)[1])) 
699      ds/dx=0.0 ds/dy=1.0 
700      >>> xs = 2 * np.pi * np.linspace(0, 1, 100) 
701      >>> fig, ax = plt.subplots(figsize=(6.5, 4)) 
702      >>> ax.plot(y[:, 0], y[:, 1], 'o', label='data') 
703      >>> ax.plot(np.cos(xs), np.sin(xs), label='true') 
704      >>> ax.plot(cs(xs)[:, 0], cs(xs)[:, 1], label='spline') 
705      >>> ax.axes.set_aspect('equal') 
706      >>> ax.legend(loc='center') 
707      >>> plt.show() 
708   
709      The third example is the interpolation of a polynomial y = x**3 on the 
710      interval 0 <= x<= 1. A cubic spline can represent this function exactly. 
711      To achieve that we need to specify values and first derivatives at 
712      endpoints of the interval. Note that y' = 3 * x**2 and thus y'(0) = 0 and 
713      y'(1) = 3. 
714   
715      >>> cs = CubicSpline([0, 1], [0, 1], bc_type=((1, 0), (1, 3))) 
716      >>> x = np.linspace(0, 1) 
717      >>> np.allclose(x**3, cs(x)) 
718      True 
719   
720      References 
721      ---------- 
722      .. [1] `Cubic Spline Interpolation 
723              <https://en.wikiversity.org/wiki/Cubic_Spline_Interpolation>`_ 
724              on Wikiversity. 
725      .. [2] Carl de Boor, "A Practical Guide to Splines", Springer-Verlag, 1978. 
726      """

728      def __init__(self, x, y, axis=0, bc_type='not-a-knot', extrapolate=None):
729          x, dx, y, axis, _ = prepare_input(x, y, axis)
730          n = len(x)
731  
732          bc, y = self._validate_bc(bc_type, y, y.shape[1:], axis)
733  
734          if extrapolate is None:
735              if bc[0] == 'periodic':
736                  extrapolate = 'periodic'
737              else:
738                  extrapolate = True
739  
740          if y.size == 0:
741              # bail out early for zero-sized arrays
742              s = np.zeros_like(y)
743          else:
744              dxr = dx.reshape([dx.shape[0]] + [1] * (y.ndim - 1))
745              slope = np.diff(y, axis=0) / dxr
746  
747              # If bc is 'not-a-knot' this change is just a convention.
748              # If bc is 'periodic' then we already checked that y[0] == y[-1],
749              # and the spline is just a constant, we handle this case in the
750              # same way by setting the first derivatives to slope, which is 0.
751              if n == 2:
752                  if bc[0] in ['not-a-knot', 'periodic']:
753                      bc[0] = (1, slope[0])
754                  if bc[1] in ['not-a-knot', 'periodic']:
755                      bc[1] = (1, slope[0])
756  
757              # This is a special case, when both conditions are 'not-a-knot'
758              # and n == 3. In this case 'not-a-knot' can't be handled regularly
759              # as the both conditions are identical. We handle this case by
760              # constructing a parabola passing through given points.
761              if n == 3 and bc[0] == 'not-a-knot' and bc[1] == 'not-a-knot':
762                  A = np.zeros((3, 3))  # This is a standard matrix.
763                  b = np.empty((3,) + y.shape[1:], dtype=y.dtype)
764  
765                  A[0, 0] = 1
766                  A[0, 1] = 1
767                  A[1, 0] = dx[1]
768                  A[1, 1] = 2 * (dx[0] + dx[1])
769                  A[1, 2] = dx[0]
770                  A[2, 1] = 1
771                  A[2, 2] = 1
772  
773                  b[0] = 2 * slope[0]
774                  b[1] = 3 * (dxr[0] * slope[1] + dxr[1] * slope[0])
775                  b[2] = 2 * slope[1]
776  
777                  s = solve(A, b, overwrite_a=True, overwrite_b=True,
778                            check_finite=False)
779              elif n == 3 and bc[0] == 'periodic':
780                  # In case when number of points is 3 we compute the derivatives
781                  # manually
782                  t = (slope / dxr).sum(0) / (1. / dxr).sum(0)
783                  s = np.broadcast_to(t, (n,) + y.shape[1:])
784              else:
785                  # Find derivative values at each x[i] by solving a tridiagonal
786                  # system.
787                  A = np.zeros((3, n))  # This is a banded matrix representation.
788                  b = np.empty((n,) + y.shape[1:], dtype=y.dtype)
789  
790                  # Filling the system for i=1..n-2
791                  #                         (x[i-1] - x[i]) * s[i-1] +\
792                  # 2 * ((x[i] - x[i-1]) + (x[i+1] - x[i])) * s[i]   +\
793                  #                         (x[i] - x[i-1]) * s[i+1] =\
794                  #       3 * ((x[i+1] - x[i])*(y[i] - y[i-1])/(x[i] - x[i-1]) +\
795                  #           (x[i] - x[i-1])*(y[i+1] - y[i])/(x[i+1] - x[i]))
796  
797                  A[1, 1:-1] = 2 * (dx[:-1] + dx[1:])  # The diagonal
798                  A[0, 2:] = dx[:-1]                   # The upper diagonal
799                  A[-1, :-2] = dx[1:]                  # The lower diagonal
800  
801                  b[1:-1] = 3 * (dxr[1:] * slope[:-1] + dxr[:-1] * slope[1:])
802  
803                  bc_start, bc_end = bc
804  
805                  if bc_start == 'periodic':
806                      # Due to the periodicity, and because y[-1] = y[0], the
807                      # linear system has (n-1) unknowns/equations instead of n:
808                      A = A[:, 0:-1]
809                      A[1, 0] = 2 * (dx[-1] + dx[0])
810                      A[0, 1] = dx[-1]
811  
812                      b = b[:-1]
813  
814                      # Also, due to the periodicity, the system is not tri-diagonal.
815                      # We need to compute a "condensed" matrix of shape (n-2, n-2).
816                      # See https://web.archive.org/web/20151220180652/http://www.cfm.brown.edu/people/gk/chap6/node14.html
817                      # for more explanations.
818                      # The condensed matrix is obtained by removing the last column
819                      # and last row of the (n-1, n-1) system matrix. The removed
820                      # values are saved in scalar variables with the (n-1, n-1)
821                      # system matrix indices forming their names:
822                      a_m1_0 = dx[-2]  # lower left corner value: A[-1, 0]
823                      a_m1_m2 = dx[-1]
824                      a_m1_m1 = 2 * (dx[-1] + dx[-2])
825                      a_m2_m1 = dx[-3]
826                      a_0_m1 = dx[0]
827  
828                      b[0] = 3 * (dxr[0] * slope[-1] + dxr[-1] * slope[0])
829                      b[-1] = 3 * (dxr[-1] * slope[-2] + dxr[-2] * slope[-1])
830  
831                      Ac = A[:, :-1]
832                      b1 = b[:-1]
833                      b2 = np.zeros_like(b1)
834                      b2[0] = -a_0_m1
835                      b2[-1] = -a_m2_m1
836  
837                      # s1 and s2 are the solutions of (n-2, n-2) system
838                      s1 = solve_banded((1, 1), Ac, b1, overwrite_ab=False,
839                                        overwrite_b=False, check_finite=False)
840  
841                      s2 = solve_banded((1, 1), Ac, b2, overwrite_ab=False,
842                                        overwrite_b=False, check_finite=False)
843  
844                      # computing the s[n-2] solution:
845                      s_m1 = ((b[-1] - a_m1_0 * s1[0] - a_m1_m2 * s1[-1]) /
846                              (a_m1_m1 + a_m1_0 * s2[0] + a_m1_m2 * s2[-1]))
847  
848                      # s is the solution of the (n, n) system:
849                      s = np.empty((n,) + y.shape[1:], dtype=y.dtype)
850                      s[:-2] = s1 + s_m1 * s2
851                      s[-2] = s_m1
852                      s[-1] = s[0]
853                  else:
854                      if bc_start == 'not-a-knot':
855                          A[1, 0] = dx[1]
856                          A[0, 1] = x[2] - x[0]
857                          d = x[2] - x[0]
858                          b[0] = ((dxr[0] + 2*d) * dxr[1] * slope[0] +
859                                  dxr[0]**2 * slope[1]) / d
860                      elif bc_start[0] == 1:
861                          A[1, 0] = 1
862                          A[0, 1] = 0
863                          b[0] = bc_start[1]
864                      elif bc_start[0] == 2:
865                          A[1, 0] = 2 * dx[0]
866                          A[0, 1] = dx[0]
867                          b[0] = -0.5 * bc_start[1] * dx[0]**2 + 3 * (y[1] - y[0])
868  
869                      if bc_end == 'not-a-knot':
870                          A[1, -1] = dx[-2]
871                          A[-1, -2] = x[-1] - x[-3]
872                          d = x[-1] - x[-3]
873                          b[-1] = ((dxr[-1]**2*slope[-2] +
874                                   (2*d + dxr[-1])*dxr[-2]*slope[-1]) / d)
875                      elif bc_end[0] == 1:
876                          A[1, -1] = 1
877                          A[-1, -2] = 0
878                          b[-1] = bc_end[1]
879                      elif bc_end[0] == 2:
880                          A[1, -1] = 2 * dx[-1]
881                          A[-1, -2] = dx[-1]
882                          b[-1] = 0.5 * bc_end[1] * dx[-1]**2 + 3 * (y[-1] - y[-2])
883  
884                      s = solve_banded((1, 1), A, b, overwrite_ab=True,
885                                       overwrite_b=True, check_finite=False)
886  
887          super().__init__(x, y, s, axis=0, extrapolate=extrapolate)
888          self.axis = axis
889  
890      @staticmethod
891      def _validate_bc(bc_type, y, expected_deriv_shape, axis):
892          """Validate and prepare boundary conditions. 
893   
894          Returns 
895          ------- 
896          validated_bc : 2-tuple 
897              Boundary conditions for a curve start and end. 
898          y : ndarray 
899              y casted to complex dtype if one of the boundary conditions has 
900              complex dtype. 
901          """
902          if isinstance(bc_type, str):
903              if bc_type == 'periodic':
904                  if not np.allclose(y[0], y[-1], rtol=1e-15, atol=1e-15):
905                      raise ValueError(
906                          f"The first and last `y` point along axis {axis} must "
907                          "be identical (within machine precision) when "
908                          "bc_type='periodic'.")
909  
910              bc_type = (bc_type, bc_type)
911  
912          else:
913              if len(bc_type) != 2:
914                  raise ValueError("`bc_type` must contain 2 elements to "
915                                   "specify start and end conditions.")
916  
917              if 'periodic' in bc_type:
918                  raise ValueError("'periodic' `bc_type` is defined for both "
919                                   "curve ends and cannot be used with other "
920                                   "boundary conditions.")
921  
922          validated_bc = []
923          for bc in bc_type:
924              if isinstance(bc, str):
925                  if bc == 'clamped':
926                      validated_bc.append((1, np.zeros(expected_deriv_shape)))
927                  elif bc == 'natural':
928                      validated_bc.append((2, np.zeros(expected_deriv_shape)))
929                  elif bc in ['not-a-knot', 'periodic']:
930                      validated_bc.append(bc)
931                  else:
932                      raise ValueError(f"bc_type={bc} is not allowed.")
933              else:
934                  try:
935                      deriv_order, deriv_value = bc
936                  except Exception as e:
937                      raise ValueError(
938                          "A specified derivative value must be "
939                          "given in the form (order, value)."
940                      ) from e
941  
942                  if deriv_order not in [1, 2]:
943                      raise ValueError("The specified derivative order must "
944                                       "be 1 or 2.")
945  
946                  deriv_value = np.asarray(deriv_value)
947                  if deriv_value.shape != expected_deriv_shape:
948                      raise ValueError(
949                          f"`deriv_value` shape {deriv_value.shape} is not "
950                          f"the expected one {expected_deriv_shape}."
951                      )
952  
953                  if np.issubdtype(deriv_value.dtype, np.complexfloating):
954                      y = y.astype(complex, copy=False)
955  
956                  validated_bc.append((deriv_order, deriv_value))
957  
958          return validated_bc, y
