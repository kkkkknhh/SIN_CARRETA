9    def mpmath_wrightomega(x):
10       return mpmath.lambertw(mpmath.exp(x), mpmath.mpf('-0.5'))

13   def wrightomega_series_error(x):
14       series = x
15       desired = mpmath_wrightomega(x)
16       return abs(series - desired) / desired
17   
18   
19   def wrightomega_exp_error(x):
20       exponential_approx = mpmath.exp(x)
21       desired = mpmath_wrightomega(x)
22       return abs(exponential_approx - desired) / desired
