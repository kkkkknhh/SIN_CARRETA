19   def test_smith_normal():
20       m = Matrix([[12,6,4,8],[3,9,6,12],[2,16,14,28],[20,10,10,20]])
21       smf = Matrix([[1, 0, 0, 0], [0, 10, 0, 0], [0, 0, 30, 0], [0, 0, 0, 0]])
22       assert smith_normal_form(m) == smf
23   
24       a, s, t = smith_normal_decomp(m)
25       assert a == s * m * t
26   
27       x = Symbol('x')
28       with warns_deprecated_sympy():
29           m = Matrix([[Poly(x-1), Poly(1, x),Poly(-1,x)],
30                       [0, Poly(x), Poly(-1,x)],
31                       [Poly(0,x),Poly(-1,x),Poly(x)]])
32       invs = 1, x - 1, x**2 - 1
33       assert invariant_factors(m, domain=QQ[x]) == invs
34   
35       m = Matrix([[2, 4]])
36       smf = Matrix([[2, 0]])
37       assert smith_normal_form(m) == smf
38   
39       prng = random.Random(0)
40       for i in range(6):
41           for j in range(6):
42               for _ in range(10 if i*j else 1):
43                   m = randMatrix(i, j, max=5, percent=50, prng=prng)
44                   a, s, t = smith_normal_decomp(m)
45                   assert a == s * m * t
46                   assert is_smith_normal_form(a)
47                   s.inv().to_DM(ZZ)
48                   t.inv().to_DM(ZZ)
49   
50                   a, s, t = smith_normal_decomp(m, QQ)
51                   assert a == s * m * t
52                   assert is_smith_normal_form(a)
53                   s.inv()
54                   t.inv()

57   def test_smith_normal_deprecated():
58       from sympy.polys.solvers import RawMatrix as Matrix
59   
60       with warns_deprecated_sympy():
61           m = Matrix([[12, 6, 4,8],[3,9,6,12],[2,16,14,28],[20,10,10,20]])
62       setattr(m, 'ring', ZZ)
63       with warns_deprecated_sympy():
64           smf = Matrix([[1, 0, 0, 0], [0, 10, 0, 0], [0, 0, 30, 0], [0, 0, 0, 0]])
65       assert smith_normal_form(m) == smf
66   
67       x = Symbol('x')
68       with warns_deprecated_sympy():
69           m = Matrix([[Poly(x-1), Poly(1, x),Poly(-1,x)],
70                       [0, Poly(x), Poly(-1,x)],
71                       [Poly(0,x),Poly(-1,x),Poly(x)]])
72       setattr(m, 'ring', QQ[x])
73       invs = (Poly(1, x, domain='QQ'), Poly(x - 1, domain='QQ'), Poly(x**2 - 1, domain='QQ'))
74       assert invariant_factors(m) == invs
75   
76       with warns_deprecated_sympy():
77           m = Matrix([[2, 4]])
78       setattr(m, 'ring', ZZ)
79       with warns_deprecated_sympy():
80           smf = Matrix([[2, 0]])
81       assert smith_normal_form(m) == smf

84   def test_hermite_normal():
85       m = Matrix([[2, 7, 17, 29, 41], [3, 11, 19, 31, 43], [5, 13, 23, 37, 47]])
86       hnf = Matrix([[1, 0, 0], [0, 2, 1], [0, 0, 1]])
87       assert hermite_normal_form(m) == hnf
88   
89       tr_hnf = Matrix([[37, 0, 19], [222, -6, 113], [48, 0, 25], [0, 2, 1], [0, 0, 1]])
90       assert hermite_normal_form(m.transpose()) == tr_hnf
91   
92       m = Matrix([[8, 28, 68, 116, 164], [3, 11, 19, 31, 43], [5, 13, 23, 37, 47]])
93       hnf = Matrix([[4, 0, 0], [0, 2, 1], [0, 0, 1]])
94       assert hermite_normal_form(m) == hnf
95       assert hermite_normal_form(m, D=8) == hnf
96       assert hermite_normal_form(m, D=ZZ(8)) == hnf
97       assert hermite_normal_form(m, D=Integer(8)) == hnf
98   
99       m = Matrix([[10, 8, 6, 30, 2], [45, 36, 27, 18, 9], [5, 4, 3, 2, 1]])
100      hnf = Matrix([[26, 2], [0, 9], [0, 1]])
101      assert hermite_normal_form(m) == hnf
102  
103      m = Matrix([[2, 7], [0, 0], [0, 0]])
104      hnf = Matrix([[1], [0], [0]])
105      assert hermite_normal_form(m) == hnf
