31   def _nsort(roots, separated=False):
32       """Sort the numerical roots putting the real roots first, then sorting 
33       according to real and imaginary parts. If ``separated`` is True, then 
34       the real and imaginary roots will be returned in two lists, respectively. 
35    
36       This routine tries to avoid issue 6137 by separating the roots into real 
37       and imaginary parts before evaluation. In addition, the sorting will raise 
38       an error if any computation cannot be done with precision. 
39       """
40       if not all(r.is_number for r in roots):
41           raise NotImplementedError
42       if not len(roots):
43           return [] if not separated else ([], [])
44       # see issue 6137:
45       # get the real part of the evaluated real and imaginary parts of each root
46       key = [[i.n(2).as_real_imag()[0] for i in r.as_real_imag()] for r in roots]
47       # make sure the parts were computed with precision
48       if len(roots) > 1 and any(i._prec == 1 for k in key for i in k):
49           raise NotImplementedError("could not compute root with precision")
50       # insert a key to indicate if the root has an imaginary part
51       key = [(1 if i else 0, r, i) for r, i in key]
52       key = sorted(zip(key, roots))
53       # return the real and imaginary roots separately if desired
54       if separated:
55           r = []
56           i = []
57           for (im, _, _), v in key:
58               if im:
59                   i.append(v)
60               else:
61                   r.append(v)
62           return r, i
63       _, roots = zip(*key)
64       return list(roots)

67   def _sort_gens(gens, **args):
68       """Sort generators in a reasonably intelligent way. """
69       opt = build_options(args)
70   
71       gens_order, wrt = {}, None
72   
73       if opt is not None:
74           gens_order, wrt = {}, opt.wrt
75   
76           for i, gen in enumerate(opt.sort):
77               gens_order[gen] = i + 1
78   
79       def order_key(gen):
80           gen = str(gen)
81   
82           if wrt is not None:
83               try:
84                   return (-len(wrt) + wrt.index(gen), gen, 0)
85               except ValueError:
86                   pass
87   
88           name, index = _re_gen.match(gen).groups()
89   
90           if index:
91               index = int(index)
92           else:
93               index = 0
94   
95           try:
96               return ( gens_order[name], name, index)
97           except KeyError:
98               pass
99   
100          try:
101              return (_gens_order[name], name, index)
102          except KeyError:
103              pass
104  
105          return (_max_order, name, index)
106  
107      try:
108          gens = sorted(gens, key=order_key)
109      except TypeError:  # pragma: no cover
110          pass
111  
112      return tuple(gens)

115  def _unify_gens(f_gens, g_gens):
116      """Unify generators in a reasonably intelligent way. """
117      f_gens = list(f_gens)
118      g_gens = list(g_gens)
119  
120      if f_gens == g_gens:
121          return tuple(f_gens)
122  
123      gens, common, k = [], [], 0
124  
125      for gen in f_gens:
126          if gen in g_gens:
127              common.append(gen)
128  
129      for i, gen in enumerate(g_gens):
130          if gen in common:
131              g_gens[i], k = common[k], k + 1
132  
133      for gen in common:
134          i = f_gens.index(gen)
135  
136          gens.extend(f_gens[:i])
137          f_gens = f_gens[i + 1:]
138  
139          i = g_gens.index(gen)
140  
141          gens.extend(g_gens[:i])
142          g_gens = g_gens[i + 1:]
143  
144          gens.append(gen)
145  
146      gens.extend(f_gens)
147      gens.extend(g_gens)
148  
149      return tuple(gens)

160  def _sort_factors(factors, **args):
161      """Sort low-level factors in increasing 'complexity' order. """
162  
163      # XXX: GF(p) does not support comparisons so we need a key function to sort
164      # the factors if python-flint is being used. A better solution might be to
165      # add a sort key method to each domain.
166      def order_key(factor):
167          if isinstance(factor, _GF_types):
168              return int(factor)
169          elif isinstance(factor, list):
170              return [order_key(f) for f in factor]
171          else:
172              return factor
173  
174      def order_if_multiple_key(factor):
175          (f, n) = factor
176          return (len(f), n, order_key(f))
177  
178      def order_no_multiple_key(f):
179          return (len(f), order_key(f))
180  
181      if args.get('multiple', True):
182          return sorted(factors, key=order_if_multiple_key)
183      else:
184          return sorted(factors, key=order_no_multiple_key)

200  def _parallel_dict_from_expr_if_gens(exprs, opt):
201      """Transform expressions into a multinomial form given generators. """
202      k, indices = len(opt.gens), {}
203  
204      for i, g in enumerate(opt.gens):
205          indices[g] = i
206  
207      polys = []
208  
209      for expr in exprs:
210          poly = {}
211  
212          if expr.is_Equality:
213              expr = expr.lhs - expr.rhs
214  
215          for term in Add.make_args(expr):
216              coeff, monom = [], [0]*k
217  
218              for factor in Mul.make_args(term):
219                  if not _not_a_coeff(factor) and factor.is_Number:
220                      coeff.append(factor)
221                  else:
222                      try:
223                          if opt.series is False:
224                              base, exp = decompose_power(factor)
225  
226                              if exp < 0:
227                                  exp, base = -exp, Pow(base, -S.One)
228                          else:
229                              base, exp = decompose_power_rat(factor)
230  
231                          monom[indices[base]] = exp
232                      except KeyError:
233                          if not factor.has_free(*opt.gens):
234                              coeff.append(factor)
235                          else:
236                              raise PolynomialError("%s contains an element of "
237                                                    "the set of generators." % factor)
238  
239              monom = tuple(monom)
240  
241              if monom in poly:
242                  poly[monom] += Mul(*coeff)
243              else:
244                  poly[monom] = Mul(*coeff)
245  
246          polys.append(poly)
247  
248      return polys, opt.gens

251  def _parallel_dict_from_expr_no_gens(exprs, opt):
252      """Transform expressions into a multinomial form and figure out generators. """
253      if opt.domain is not None:
254          def _is_coeff(factor):
255              return factor in opt.domain
256      elif opt.extension is True:
257          def _is_coeff(factor):
258              return factor.is_algebraic
259      elif opt.greedy is not False:
260          def _is_coeff(factor):
261              return factor is S.ImaginaryUnit
262      else:
263          def _is_coeff(factor):
264              return factor.is_number
265  
266      gens, reprs = set(), []
267  
268      for expr in exprs:
269          terms = []
270  
271          if expr.is_Equality:
272              expr = expr.lhs - expr.rhs
273  
274          for term in Add.make_args(expr):
275              coeff, elements = [], {}
276  
277              for factor in Mul.make_args(term):
278                  if not _not_a_coeff(factor) and (factor.is_Number or _is_coeff(factor)):
279                      coeff.append(factor)
280                  else:
281                      if opt.series is False:
282                          base, exp = decompose_power(factor)
283  
284                          if exp < 0:
285                              exp, base = -exp, Pow(base, -S.One)
286                      else:
287                          base, exp = decompose_power_rat(factor)
288  
289                      elements[base] = elements.setdefault(base, 0) + exp
290                      gens.add(base)
291  
292              terms.append((coeff, elements))
293  
294          reprs.append(terms)
295  
296      gens = _sort_gens(gens, opt=opt)
297      k, indices = len(gens), {}
298  
299      for i, g in enumerate(gens):
300          indices[g] = i
301  
302      polys = []
303  
304      for terms in reprs:
305          poly = {}
306  
307          for coeff, term in terms:
308              monom = [0]*k
309  
310              for base, exp in term.items():
311                  monom[indices[base]] = exp
312  
313              monom = tuple(monom)
314  
315              if monom in poly:
316                  poly[monom] += Mul(*coeff)
317              else:
318                  poly[monom] = Mul(*coeff)
319  
320          polys.append(poly)
321  
322      return polys, tuple(gens)

343  def _parallel_dict_from_expr(exprs, opt):
344      """Transform expressions into a multinomial form. """
345      if opt.expand is not False:
346          exprs = [ expr.expand() for expr in exprs ]
347  
348      if any(expr.is_commutative is False for expr in exprs):
349          raise PolynomialError('non-commutative expressions are not supported')
350  
351      if opt.gens:
352          reps, gens = _parallel_dict_from_expr_if_gens(exprs, opt)
353      else:
354          reps, gens = _parallel_dict_from_expr_no_gens(exprs, opt)
355  
356      return reps, opt.clone({'gens': gens})

365  def _dict_from_expr(expr, opt):
366      """Transform an expression into a multinomial form. """
367      if expr.is_commutative is False:
368          raise PolynomialError('non-commutative expressions are not supported')
369  
370      def _is_expandable_pow(expr):
371          return (expr.is_Pow and expr.exp.is_positive and expr.exp.is_Integer
372                  and expr.base.is_Add)
373  
374      if opt.expand is not False:
375          if not isinstance(expr, (Expr, Eq)):
376              raise PolynomialError('expression must be of type Expr')
377          expr = expr.expand()
378          # TODO: Integrate this into expand() itself
379          while any(_is_expandable_pow(i) or i.is_Mul and
380              any(_is_expandable_pow(j) for j in i.args) for i in
381                  Add.make_args(expr)):
382  
383              expr = expand_multinomial(expr)
384          while any(i.is_Mul and any(j.is_Add for j in i.args) for i in Add.make_args(expr)):
385              expr = expand_mul(expr)
386  
387      if opt.gens:
388          rep, gens = _dict_from_expr_if_gens(expr, opt)
389      else:
390          rep, gens = _dict_from_expr_no_gens(expr, opt)
391  
392      return rep, opt.clone({'gens': gens})

395  def expr_from_dict(rep, *gens):
396      """Convert a multinomial form into an expression. """
397      result = []
398  
399      for monom, coeff in rep.items():
400          term = [coeff]
401          for g, m in zip(gens, monom):
402              if m:
403                  term.append(Pow(g, m))
404  
405          result.append(Mul(*term))
406  
407      return Add(*result)

414  def _dict_reorder(rep, gens, new_gens):
415      """Reorder levels using dict representation. """
416      gens = list(gens)
417  
418      monoms = rep.keys()
419      coeffs = rep.values()
420  
421      new_monoms = [ [] for _ in range(len(rep)) ]
422      used_indices = set()
423  
424      for gen in new_gens:
425          try:
426              j = gens.index(gen)
427              used_indices.add(j)
428  
429              for M, new_M in zip(monoms, new_monoms):
430                  new_M.append(M[j])
431          except ValueError:
432              for new_M in new_monoms:
433                  new_M.append(0)
434  
435      for i, _ in enumerate(gens):
436          if i not in used_indices:
437              for monom in monoms:
438                  if monom[i]:
439                      raise GeneratorsError("unable to drop generators")
440  
441      return map(tuple, new_monoms), coeffs

444  class PicklableWithSlots:
445      """ 
446      Mixin class that allows to pickle objects with ``__slots__``. 
447   
448      Examples 
449      ======== 
450   
451      First define a class that mixes :class:`PicklableWithSlots` in:: 
452   
453          >>> from sympy.polys.polyutils import PicklableWithSlots 
454          >>> class Some(PicklableWithSlots): 
455          ...     __slots__ = ('foo', 'bar') 
456          ... 
457          ...     def __init__(self, foo, bar): 
458          ...         self.foo = foo 
459          ...         self.bar = bar 
460   
461      To make :mod:`pickle` happy in doctest we have to use these hacks:: 
462   
463          >>> import builtins 
464          >>> builtins.Some = Some 
465          >>> from sympy.polys import polyutils 
466          >>> polyutils.Some = Some 
467   
468      Next lets see if we can create an instance, pickle it and unpickle:: 
469   
470          >>> some = Some('abc', 10) 
471          >>> some.foo, some.bar 
472          ('abc', 10) 
473   
474          >>> from pickle import dumps, loads 
475          >>> some2 = loads(dumps(some)) 
476   
477          >>> some2.foo, some2.bar 
478          ('abc', 10) 
479   
480      """
481  
482      __slots__ = ()
483  
484      def __getstate__(self, cls=None):
485          if cls is None:
486              # This is the case for the instance that gets pickled
487              cls = self.__class__
488  
489          d = {}
490  
491          # Get all data that should be stored from super classes
492          for c in cls.__bases__:
493              # XXX: Python 3.11 defines object.__getstate__ and it does not
494              # accept any arguments so we need to make sure not to call it with
495              # an argument here. To be compatible with Python < 3.11 we need to
496              # be careful not to assume that c or object has a __getstate__
497              # method though.
498              getstate = getattr(c, "__getstate__", None)
499              objstate = getattr(object, "__getstate__", None)
500              if getstate is not None and getstate is not objstate:
501                  d.update(getstate(self, c))
502  
503          # Get all information that should be stored from cls and return the dict
504          for name in cls.__slots__:
505              if hasattr(self, name):
506                  d[name] = getattr(self, name)
507  
508          return d
509  
510      def __setstate__(self, d):
511          # All values that were pickled are now assigned to a fresh instance
512          for name, value in d.items():
513              setattr(self, name, value)

516  class IntegerPowerable:
517      r""" 
518      Mixin class for classes that define a `__mul__` method, and want to be 
519      raised to integer powers in the natural way that follows. Implements 
520      powering via binary expansion, for efficiency. 
521   
522      By default, only integer powers $\geq 2$ are supported. To support the 
523      first, zeroth, or negative powers, override the corresponding methods, 
524      `_first_power`, `_zeroth_power`, `_negative_power`, below. 
525      """
526  
527      def __pow__(self, e, modulo=None):
528          if e < 2:
529              try:
530                  if e == 1:
531                      return self._first_power()
532                  elif e == 0:
533                      return self._zeroth_power()
534                  else:
535                      return self._negative_power(e, modulo=modulo)
536              except NotImplementedError:
537                  return NotImplemented
538          else:
539              bits = [int(d) for d in reversed(bin(e)[2:])]
540              n = len(bits)
541              p = self
542              first = True
543              for i in range(n):
544                  if bits[i]:
545                      if first:
546                          r = p
547                          first = False
548                      else:
549                          r *= p
550                          if modulo is not None:
551                              r %= modulo
552                  if i < n - 1:
553                      p *= p
554                      if modulo is not None:
555                          p %= modulo
556              return r
557  
558      def _negative_power(self, e, modulo=None):
559          """ 
560          Compute inverse of self, then raise that to the abs(e) power. 
561          For example, if the class has an `inv()` method, 
562              return self.inv() ** abs(e) % modulo 
563          """
564          raise NotImplementedError
565  
566      def _zeroth_power(self):
567          """Return unity element of algebraic struct to which self belongs."""
568          raise NotImplementedError
569  
570      def _first_power(self):
571          """Return a copy of self."""
572          raise NotImplementedError
