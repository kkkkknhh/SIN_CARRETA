194  def _mixed_data_generator(n_samples, n_repetitions, axis, rng,
195                            paired=False):
196      # generate random samples to check the response of hypothesis tests to
197      # samples with different (but broadcastable) shapes and various
198      # nan patterns (e.g. all nans, some nans, no nans) along axis-slices
199  
200      data = []
201      for i in range(n_samples):
202          n_patterns = 6  # number of distinct nan patterns
203          n_obs = 20 if paired else 20 + i  # observations per axis-slice
204          x = np.ones((n_repetitions, n_patterns, n_obs)) * np.nan
205  
206          for j in range(n_repetitions):
207              samples = x[j, :, :]
208  
209              # case 0: axis-slice with all nans (0 reals)
210              # cases 1-3: axis-slice with 1-3 reals (the rest nans)
211              # case 4: axis-slice with mostly (all but two) reals
212              # case 5: axis slice with all reals
213              for k, n_reals in enumerate([0, 1, 2, 3, n_obs-2, n_obs]):
214                  # for cases 1-3, need paired nansw  to be in the same place
215                  indices = rng.permutation(n_obs)[:n_reals]
216                  samples[k, indices] = rng.random(size=n_reals)
217  
218              # permute the axis-slices just to show that order doesn't matter
219              samples[:] = rng.permutation(samples, axis=0)
220  
221          # For multi-sample tests, we want to test broadcasting and check
222          # that nan policy works correctly for each nan pattern for each input.
223          # This takes care of both simultaneously.
224          new_shape = [n_repetitions] + [1]*n_samples + [n_obs]
225          new_shape[1 + i] = 6
226          x = x.reshape(new_shape)
227  
228          x = np.moveaxis(x, -1, axis)
229          data.append(x)
230      return data

233  def _homogeneous_data_generator(n_samples, n_repetitions, axis, rng,
234                                  paired=False, all_nans=True):
235      # generate random samples to check the response of hypothesis tests to
236      # samples with different (but broadcastable) shapes and homogeneous
237      # data (all nans or all finite)
238      data = []
239      for i in range(n_samples):
240          n_obs = 20 if paired else 20 + i  # observations per axis-slice
241          shape = [n_repetitions] + [1]*n_samples + [n_obs]
242          shape[1 + i] = 2
243          x = np.ones(shape) * np.nan if all_nans else rng.random(shape)
244          x = np.moveaxis(x, -1, axis)
245          data.append(x)
246      return data

249  def nan_policy_1d(hypotest, data1d, unpacker, *args, n_outputs=2,
250                    nan_policy='raise', paired=False, _no_deco=True, **kwds):
251      # Reference implementation for how `nan_policy` should work for 1d samples
252  
253      if nan_policy == 'raise':
254          for sample in data1d:
255              if np.any(np.isnan(sample)):
256                  raise ValueError("The input contains nan values")
257  
258      elif (nan_policy == 'propagate'
259            and hypotest not in override_propagate_funcs):
260          # For all hypothesis tests tested, returning nans is the right thing.
261          # But many hypothesis tests don't propagate correctly (e.g. they treat
262          # np.nan the same as np.inf, which doesn't make sense when ranks are
263          # involved) so override that behavior here.
264          for sample in data1d:
265              if np.any(np.isnan(sample)):
266                  return np.full(n_outputs, np.nan)
267  
268      elif nan_policy == 'omit':
269          # manually omit nans (or pairs in which at least one element is nan)
270          if not paired:
271              data1d = [sample[~np.isnan(sample)] for sample in data1d]
272          else:
273              nan_mask = np.isnan(data1d[0])
274              for sample in data1d[1:]:
275                  nan_mask = np.logical_or(nan_mask, np.isnan(sample))
276              data1d = [sample[~nan_mask] for sample in data1d]
277  
278      return unpacker(hypotest(*data1d, *args, _no_deco=_no_deco, **kwds))

349  def _axis_nan_policy_test(hypotest, args, kwds, n_samples, n_outputs, paired,
350                            unpacker, nan_policy, axis, data_generator):
351      # Tests the 1D and vectorized behavior of hypothesis tests against a
352      # reference implementation (nan_policy_1d with np.ndenumerate)
353  
354      # Some hypothesis tests return a non-iterable that needs an `unpacker` to
355      # extract the statistic and p-value. For those that don't:
356      if not unpacker:
357          def unpacker(res):
358              return res
359  
360      rng = np.random.default_rng(0)
361  
362      # Generate multi-dimensional test data with all important combinations
363      # of patterns of nans along `axis`
364      n_repetitions = 3  # number of repetitions of each pattern
365      data_gen_kwds = {'n_samples': n_samples, 'n_repetitions': n_repetitions,
366                       'axis': axis, 'rng': rng, 'paired': paired}
367      if data_generator == 'mixed':
368          inherent_size = 6  # number of distinct types of patterns
369          data = _mixed_data_generator(**data_gen_kwds)
370      elif data_generator == 'all_nans':
371          inherent_size = 2  # hard-coded in _homogeneous_data_generator
372          data_gen_kwds['all_nans'] = True
373          data = _homogeneous_data_generator(**data_gen_kwds)
374      elif data_generator == 'all_finite':
375          inherent_size = 2  # hard-coded in _homogeneous_data_generator
376          data_gen_kwds['all_nans'] = False
377          data = _homogeneous_data_generator(**data_gen_kwds)
378  
379      output_shape = [n_repetitions] + [inherent_size]*n_samples
380  
381      # To generate reference behavior to compare against, loop over the axis-
382      # slices in data. Make indexing easier by moving `axis` to the end and
383      # broadcasting all samples to the same shape.
384      data_b = [np.moveaxis(sample, axis, -1) for sample in data]
385      data_b = [np.broadcast_to(sample, output_shape + [sample.shape[-1]])
386                for sample in data_b]
387      res_1d = np.zeros(output_shape + [n_outputs])
388  
389      for i, _ in np.ndenumerate(np.zeros(output_shape)):
390          data1d = [sample[i] for sample in data_b]
391          contains_nan = any([np.isnan(sample).any() for sample in data1d])
392  
393          # Take care of `nan_policy='raise'`.
394          # Afterward, the 1D part of the test is over
395          message = "The input contains nan values"
396          if nan_policy == 'raise' and contains_nan:
397              with pytest.raises(ValueError, match=message):
398                  nan_policy_1d(hypotest, data1d, unpacker, *args,
399                                n_outputs=n_outputs,
400                                nan_policy=nan_policy,
401                                paired=paired, _no_deco=True, **kwds)
402  
403              with pytest.raises(ValueError, match=message):
404                  hypotest(*data1d, *args, nan_policy=nan_policy, **kwds)
405  
406              continue
407  
408          # Take care of `nan_policy='propagate'` and `nan_policy='omit'`
409  
410          # Get results of simple reference implementation
411          try:
412              res_1da = nan_policy_1d(hypotest, data1d, unpacker, *args,
413                                      n_outputs=n_outputs,
414                                      nan_policy=nan_policy,
415                                      paired=paired, _no_deco=True, **kwds)
416          except (ValueError, RuntimeWarning, ZeroDivisionError) as ea:
417              ea_str = str(ea)
418              if any([str(ea_str).startswith(msg) for msg in too_small_messages]):
419                  res_1da = np.full(n_outputs, np.nan)
420              else:
421                  raise
422  
423          # Get results of public function with 1D slices
424          # Should warn for all slices
425          if (nan_policy == 'omit' and data_generator == "all_nans"
426                and hypotest not in too_small_special_case_funcs):
427              with pytest.warns(SmallSampleWarning, match=too_small_1d_omit):
428                  res = hypotest(*data1d, *args, nan_policy=nan_policy, **kwds)
429          # warning depends on slice
430          elif (nan_policy == 'omit' and data_generator == "mixed"
431                and hypotest not in too_small_special_case_funcs):
432              with np.testing.suppress_warnings() as sup:
433                  sup.filter(SmallSampleWarning, too_small_1d_omit)
434                  res = hypotest(*data1d, *args, nan_policy=nan_policy, **kwds)
435          # shouldn't complain if there are no NaNs
436          else:
437              res = hypotest(*data1d, *args, nan_policy=nan_policy, **kwds)
438          res_1db = unpacker(res)
439  
440          assert_allclose(res_1db, res_1da, rtol=1e-15)
441          res_1d[i] = res_1db
442  
443      res_1d = np.moveaxis(res_1d, -1, 0)
444  
445      # Perform a vectorized call to the hypothesis test.
446  
447      # If `nan_policy == 'raise'`, check that it raises the appropriate error.
448      # Test is done, so return
449      if nan_policy == 'raise' and not data_generator == "all_finite":
450          message = 'The input contains nan values'
451          with pytest.raises(ValueError, match=message):
452              hypotest(*data, axis=axis, nan_policy=nan_policy, *args, **kwds)
453          return
454  
455      # If `nan_policy == 'omit', we might be left with a small sample.
456      # Check for the appropriate warning.
457      if (nan_policy == 'omit' and data_generator in {"all_nans", "mixed"}
458            and hypotest not in too_small_special_case_funcs):
459          with pytest.warns(SmallSampleWarning, match=too_small_nd_omit):
460              res = hypotest(*data, axis=axis, nan_policy=nan_policy, *args, **kwds)
461      else:  # otherwise, there should be no warning

754  def _check_arrays_broadcastable(arrays, axis):
755      # https://numpy.org/doc/stable/user/basics.broadcasting.html
756      # "When operating on two arrays, NumPy compares their shapes element-wise.
757      # It starts with the trailing (i.e. rightmost) dimensions and works its
758      # way left.
759      # Two dimensions are compatible when
760      # 1. they are equal, or
761      # 2. one of them is 1
762      # ...
763      # Arrays do not need to have the same number of dimensions."
764      # (Clarification: if the arrays are compatible according to the criteria
765      #  above and an array runs out of dimensions, it is still compatible.)
766      # Below, we follow the rules above except ignoring `axis`
767  
768      n_dims = max([arr.ndim for arr in arrays])
769      if axis is not None:
770          # convert to negative axis
771          axis = (-n_dims + axis) if axis >= 0 else axis
772  
773      for dim in range(1, n_dims+1):  # we'll index from -1 to -n_dims, inclusive
774          if -dim == axis:
775              continue  # ignore lengths along `axis`
776  
777          dim_lengths = set()
778          for arr in arrays:
779              if dim <= arr.ndim and arr.shape[-dim] != 1:
780                  dim_lengths.add(arr.shape[-dim])
781  
782          if len(dim_lengths) > 1:
783              return False
784      return True
