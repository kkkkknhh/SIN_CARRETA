1    # Many scipy.stats functions support `axis` and `nan_policy` parameters.
2    # When the two are combined, it can be tricky to get all the behavior just
3    # right. This file contains utility functions useful for scipy.stats functions
4    # that support `axis` and `nan_policy`, including a decorator that
5    # automatically adds `axis` and `nan_policy` arguments to a function.

41   def _broadcast_arrays(arrays, axis=None, xp=None):
42       """ 
43       Broadcast shapes of arrays, ignoring incompatibility of specified axes 
44       """
45       if not arrays:
46           return arrays
47       xp = array_namespace(*arrays) if xp is None else xp
48       arrays = [xp.asarray(arr) for arr in arrays]
49       shapes = [arr.shape for arr in arrays]
50       new_shapes = _broadcast_shapes(shapes, axis)
51       if axis is None:
52           new_shapes = [new_shapes]*len(arrays)
53       return [xp.broadcast_to(array, new_shape)
54               for array, new_shape in zip(arrays, new_shapes)]

57   def _broadcast_shapes(shapes, axis=None):
58       """ 
59       Broadcast shapes, ignoring incompatibility of specified axes 
60       """
61       if not shapes:
62           return shapes
63   
64       # input validation
65       if axis is not None:
66           axis = np.atleast_1d(axis)
67           message = '`axis` must be an integer, a tuple of integers, or `None`.'
68           try:
69               with np.errstate(invalid='ignore'):
70                   axis_int = axis.astype(int)
71           except ValueError as e:
72               raise AxisError(message) from e
73           if not np.array_equal(axis_int, axis):
74               raise AxisError(message)
75           axis = axis_int
76   
77       # First, ensure all shapes have same number of dimensions by prepending 1s.
78       n_dims = max([len(shape) for shape in shapes])
79       new_shapes = np.ones((len(shapes), n_dims), dtype=int)
80       for row, shape in zip(new_shapes, shapes):
81           row[len(row)-len(shape):] = shape  # can't use negative indices (-0:)
82   
83       # Remove the shape elements of the axes to be ignored, but remember them.
84       if axis is not None:
85           axis[axis < 0] = n_dims + axis[axis < 0]
86           axis = np.sort(axis)
87           if axis[-1] >= n_dims or axis[0] < 0:
88               message = (f"`axis` is out of bounds "
89                          f"for array of dimension {n_dims}")
90               raise AxisError(message)
91   
92           if len(np.unique(axis)) != len(axis):
93               raise AxisError("`axis` must contain only distinct elements")
94   
95           removed_shapes = new_shapes[:, axis]
96           new_shapes = np.delete(new_shapes, axis, axis=1)
97   
98       # If arrays are broadcastable, shape elements that are 1 may be replaced
99       # with a corresponding non-1 shape element. Assuming arrays are
100      # broadcastable, that final shape element can be found with:
101      new_shape = np.max(new_shapes, axis=0)
102      # except in case of an empty array:
103      new_shape *= new_shapes.all(axis=0)
104  
105      # Among all arrays, there can only be one unique non-1 shape element.
106      # Therefore, if any non-1 shape element does not match what we found
107      # above, the arrays must not be broadcastable after all.
108      if np.any(~((new_shapes == 1) | (new_shapes == new_shape))):
109          raise ValueError("Array shapes are incompatible for broadcasting.")
110  
111      if axis is not None:
112          # Add back the shape elements that were ignored
113          new_axis = axis - np.arange(len(axis))
114          new_shapes = [tuple(np.insert(new_shape, new_axis, removed_shape))
115                        for removed_shape in removed_shapes]
116          return new_shapes
117      else:
118          return tuple(new_shape)

121  def _broadcast_array_shapes_remove_axis(arrays, axis=None):
122      """ 
123      Broadcast shapes of arrays, dropping specified axes 
124   
125      Given a sequence of arrays `arrays` and an integer or tuple `axis`, find 
126      the shape of the broadcast result after consuming/dropping `axis`. 
127      In other words, return output shape of a typical hypothesis test on 
128      `arrays` vectorized along `axis`. 
129   
130      Examples 
131      -------- 
132      >>> import numpy as np 
133      >>> from scipy.stats._axis_nan_policy import _broadcast_array_shapes_remove_axis 
134      >>> a = np.zeros((5, 2, 1)) 
135      >>> b = np.zeros((9, 3)) 
136      >>> _broadcast_array_shapes_remove_axis((a, b), 1) 
137      (5, 3) 
138      """
139      # Note that here, `axis=None` means do not consume/drop any axes - _not_
140      # ravel arrays before broadcasting.
141      shapes = [arr.shape for arr in arrays]
142      return _broadcast_shapes_remove_axis(shapes, axis)

145  def _broadcast_shapes_remove_axis(shapes, axis=None):
146      """ 
147      Broadcast shapes, dropping specified axes 
148   
149      Same as _broadcast_array_shapes_remove_axis, but given a sequence 
150      of array shapes `shapes` instead of the arrays themselves. 
151      """
152      shapes = _broadcast_shapes(shapes, axis)
153      shape = shapes[0]
154      if axis is not None:
155          shape = np.delete(shape, axis)
156      return tuple(shape)

159  def _broadcast_concatenate(arrays, axis, paired=False):
160      """Concatenate arrays along an axis with broadcasting."""
161      arrays = _broadcast_arrays(arrays, axis if not paired else None)
162      res = np.concatenate(arrays, axis=axis)
163      return res

167  def _remove_nans(samples, paired):
168      "Remove nans from paired or unpaired 1D samples"
169      # potential optimization: don't copy arrays that don't contain nans
170      if not paired:
171          return [sample[~np.isnan(sample)] for sample in samples]
172  
173      # for paired samples, we need to remove the whole pair when any part
174      # has a nan
175      nans = np.isnan(samples[0])
176      for sample in samples[1:]:
177          nans = nans | np.isnan(sample)
178      not_nans = ~nans
179      return [sample[not_nans] for sample in samples]

182  def _remove_sentinel(samples, paired, sentinel):
183      "Remove sentinel values from paired or unpaired 1D samples"
184      # could consolidate with `_remove_nans`, but it's not quite as simple as
185      # passing `sentinel=np.nan` because `(np.nan == np.nan) is False`
186  
187      # potential optimization: don't copy arrays that don't contain sentinel
188      if not paired:
189          return [sample[sample != sentinel] for sample in samples]
190  
191      # for paired samples, we need to remove the whole pair when any part
192      # has a nan
193      sentinels = (samples[0] == sentinel)
194      for sample in samples[1:]:
195          sentinels = sentinels | (sample == sentinel)
196      not_sentinels = ~sentinels
197      return [sample[not_sentinels] for sample in samples]

200  def _masked_arrays_2_sentinel_arrays(samples):
201      # masked arrays in `samples` are converted to regular arrays, and values
202      # corresponding with masked elements are replaced with a sentinel value
203  
204      # return without modifying arrays if none have a mask
205      has_mask = False
206      for sample in samples:
207          mask = getattr(sample, 'mask', False)
208          has_mask = has_mask or np.any(mask)
209      if not has_mask:
210          return samples, None  # None means there is no sentinel value
211  
212      # Choose a sentinel value. We can't use `np.nan`, because sentinel (masked)
213      # values are always omitted, but there are different nan policies.
214      dtype = np.result_type(*samples)
215      dtype = dtype if np.issubdtype(dtype, np.number) else np.float64
216      for i in range(len(samples)):
217          # Things get more complicated if the arrays are of different types.
218          # We could have different sentinel values for each array, but
219          # the purpose of this code is convenience, not efficiency.
220          samples[i] = samples[i].astype(dtype, copy=False)
221  
222      inexact = np.issubdtype(dtype, np.inexact)
223      info = np.finfo if inexact else np.iinfo
224      max_possible, min_possible = info(dtype).max, info(dtype).min
225      nextafter = np.nextafter if inexact else (lambda x, _: x - 1)
226  
227      sentinel = max_possible
228      # For simplicity, min_possible/np.infs are not candidate sentinel values
229      while sentinel > min_possible:
230          for sample in samples:
231              if np.any(sample == sentinel):  # choose a new sentinel value
232                  sentinel = nextafter(sentinel, -np.inf)
233                  break
234          else:  # when sentinel value is OK, break the while loop
235              break
236      else:
237          message = ("This function replaces masked elements with sentinel "
238                     "values, but the data contains all distinct values of this "
239                     "data type. Consider promoting the dtype to `np.float64`.")
240          raise ValueError(message)
241  
242      # replace masked elements with sentinel value
243      out_samples = []
244      for sample in samples:
245          mask = getattr(sample, 'mask', None)
246          if mask is not None:  # turn all masked arrays into sentinel arrays
247              mask = np.broadcast_to(mask, sample.shape)
248              sample = sample.data.copy() if np.any(mask) else sample.data
249              sample = np.asarray(sample)  # `sample.data` could be a memoryview?
250              sample[mask] = sentinel
251          out_samples.append(sample)
252  
253      return out_samples, sentinel

256  def _check_empty_inputs(samples, axis):
257      """ 
258      Check for empty sample; return appropriate output for a vectorized hypotest 
259      """
260      # if none of the samples are empty, we need to perform the test
261      if not any(sample.size == 0 for sample in samples):
262          return None
263      # otherwise, the statistic and p-value will be either empty arrays or
264      # arrays with NaNs. Produce the appropriate array and return it.
265      output_shape = _broadcast_array_shapes_remove_axis(samples, axis)
266      output = np.ones(output_shape) * _get_nan(*samples)
267      return output

271      """ 
272      Add reduced axes back to all the arrays in the result object

339  def _axis_nan_policy_factory(tuple_to_result, default_axis=0,
340                               n_samples=1, paired=False,
341                               result_to_tuple=None, too_small=0,
342                               n_outputs=2, kwd_samples=(), override=None):
343      """Factory for a wrapper that adds axis/nan_policy params to a function. 
344   
345      Parameters 
346      ---------- 
347      tuple_to_result : callable 
348          Callable that returns an object of the type returned by the function 
349          being wrapped (e.g. the namedtuple or dataclass returned by a 
350          statistical test) provided the separate components (e.g. statistic, 
351          pvalue). 
352      default_axis : int, default: 0 
353          The default value of the axis argument. Standard is 0 except when 
354          backwards compatibility demands otherwise (e.g. `None`). 
355      n_samples : int or callable, default: 1 
356          The number of data samples accepted by the function 
357          (e.g. `mannwhitneyu`), a callable that accepts a dictionary of 
358          parameters passed into the function and returns the number of data 
359          samples (e.g. `wilcoxon`), or `None` to indicate an arbitrary number 
360          of samples (e.g. `kruskal`). 
361      paired : {False, True} 
362          Whether the function being wrapped treats the samples as paired (i.e. 
363          corresponding elements of each sample should be considered as different 
364          components of the same sample.) 
365      result_to_tuple : callable, optional 
366          Function that unpacks the results of the function being wrapped into 
367          a tuple. This is essentially the inverse of `tuple_to_result`. Default 
368          is `None`, which is appropriate for statistical tests that return a 
369          statistic, pvalue tuple (rather than, e.g., a non-iterable datalass). 
370      too_small : int or callable, default: 0 
371          The largest unnacceptably small sample for the function being wrapped. 
372          For example, some functions require samples of size two or more or they 
373          raise an error. This argument prevents the error from being raised when 
374          input is not 1D and instead places a NaN in the corresponding element 
375          of the result. If callable, it must accept a list of samples, axis, 
376          and a dictionary of keyword arguments passed to the wrapper function as 
377          arguments and return a bool indicating weather the samples passed are 
378          too small. 
379      n_outputs : int or callable, default: 2 
380          The number of outputs produced by the function given 1d sample(s). For 
381          example, hypothesis tests that return a namedtuple or result object 
382          with attributes ``statistic`` and ``pvalue`` use the default 
383          ``n_outputs=2``; summary statistics with scalar output use 
384          ``n_outputs=1``. Alternatively, may be a callable that accepts a 
385          dictionary of arguments passed into the wrapped function and returns 
386          the number of outputs corresponding with those arguments. 
387      kwd_samples : sequence, default: () 
388          The names of keyword parameters that should be treated as samples. For 
389          example, `gmean` accepts as its first argument a sample `a` but 
390          also `weights` as a fourth, optional keyword argument. In this case, we 
391          use `n_samples=1` and kwd_samples=['weights']. 
392      override : dict, default: {'vectorization': False, 'nan_propagation': True} 
393          Pass a dictionary with ``'vectorization': True`` to ensure that the 
394          decorator overrides the function's behavior for multimensional input. 
395          Use ``'nan_propagation': False`` to ensure that the decorator does not 
396          override the function's behavior for ``nan_policy='propagate'``. 
397      """
398      # Specify which existing behaviors the decorator must override
399      temp = override or {}
400      override = {'vectorization': False,
401                  'nan_propagation': True}
402      override.update(temp)
403  
404      if result_to_tuple is None:
405          def result_to_tuple(res):
406              return res
407  
408      # The only `result_to_tuple` that needs the second argument (number of
409      # outputs) is the one for `moment`, and this was realized very late.
410      # Rather than changing all `result_to_tuple` definitions, we wrap them
411      # here to accept a second argument if they don't already.
412      if len(inspect.signature(result_to_tuple).parameters) == 1:
413          def result_to_tuple(res, _, f=result_to_tuple):
414              return f(res)
415  
416      if not callable(too_small):
417          def is_too_small(samples, *ts_args, axis=-1, **ts_kwargs):
418              for sample in samples:
419                  if sample.shape[axis] <= too_small:
420                      return True
421              return False
422      else:
423          is_too_small = too_small
424  
425      def axis_nan_policy_decorator(hypotest_fun_in):
426          @wraps(hypotest_fun_in)
427          def axis_nan_policy_wrapper(*args, _no_deco=False, **kwds):
428  
429              if _no_deco:  # for testing, decorator does nothing
430                  return hypotest_fun_in(*args, **kwds)
431  
432              # For now, skip the decorator entirely if using array API. In the future,
433              # we'll probably want to use it for `keepdims`, `axis` tuples, etc.
434              if len(args) == 0:  # extract sample from `kwds` if there are no `args`
435                  used_kwd_samples = list(set(kwds).intersection(set(kwd_samples)))
436                  temp = used_kwd_samples[:1]
437              else:
438                  temp = args[0]
439  
440              if not is_numpy(array_namespace(temp)):
441                  msg = ("Use of `nan_policy` and `keepdims` "
442                         "is incompatible with non-NumPy arrays.")
443                  if 'nan_policy' in kwds or 'keepdims' in kwds:
444                      raise NotImplementedError(msg)
445                  return hypotest_fun_in(*args, **kwds)
446  
447              # We need to be flexible about whether position or keyword
448              # arguments are used, but we need to make sure users don't pass
449              # both for the same parameter. To complicate matters, some
450              # functions accept samples with *args, and some functions already
451              # accept `axis` and `nan_policy` as positional arguments.
452              # The strategy is to make sure that there is no duplication
453              # between `args` and `kwds`, combine the two into `kwds`, then
454              # the samples, `nan_policy`, and `axis` from `kwds`, as they are
455              # dealt with separately.
456  
457              # Check for intersection between positional and keyword args
458              params = list(inspect.signature(hypotest_fun_in).parameters)
459              if n_samples is None:
460                  # Give unique names to each positional sample argument
461                  # Note that *args can't be provided as a keyword argument
462                  params = [f"arg{i}" for i in range(len(args))] + params[1:]
463  
464              # raise if there are too many positional args
465              maxarg = (np.inf if inspect.getfullargspec(hypotest_fun_in).varargs
466                        else len(inspect.getfullargspec(hypotest_fun_in).args))
467              if len(args) > maxarg:  # let the function raise the right error
468                  hypotest_fun_in(*args, **kwds)
469  
470              # raise if multiple values passed for same parameter
471              d_args = dict(zip(params, args))
472              intersection = set(d_args) & set(kwds)
473              if intersection:  # let the function raise the right error
474                  hypotest_fun_in(*args, **kwds)
475  
476              # Consolidate other positional and keyword args into `kwds`
477              kwds.update(d_args)
478  
479              # rename avoids UnboundLocalError
480              if callable(n_samples):
481                  # Future refactoring idea: no need for callable n_samples.
482                  # Just replace `n_samples` and `kwd_samples` with a single
483                  # list of the names of all samples, and treat all of them
484                  # as `kwd_samples` are treated below.
485                  n_samp = n_samples(kwds)
486              else:
487                  n_samp = n_samples or len(args)
488  
489              # get the number of outputs
490              n_out = n_outputs  # rename to avoid UnboundLocalError
491              if callable(n_out):
492                  n_out = n_out(kwds)
493  
494              # If necessary, rearrange function signature: accept other samples
495              # as positional args right after the first n_samp args
496              kwd_samp = [name for name in kwd_samples
497                          if kwds.get(name, None) is not None]
498              n_kwd_samp = len(kwd_samp)
499              if not kwd_samp:
500                  hypotest_fun_out = hypotest_fun_in
501              else:
502                  def hypotest_fun_out(*samples, **kwds):
503                      new_kwds = dict(zip(kwd_samp, samples[n_samp:]))
504                      kwds.update(new_kwds)
505                      return hypotest_fun_in(*samples[:n_samp], **kwds)
506  
507              # Extract the things we need here
508              try:  # if something is missing
509                  samples = [np.atleast_1d(kwds.pop(param))
510                             for param in (params[:n_samp] + kwd_samp)]
511              except KeyError:  # let the function raise the right error
512                  # might need to revisit this if required arg is not a "sample"
513                  hypotest_fun_in(*args, **kwds)
514              vectorized = True if 'axis' in params else False
515              vectorized = vectorized and not override['vectorization']
516              axis = kwds.pop('axis', default_axis)
517              nan_policy = kwds.pop('nan_policy', 'propagate')
518              keepdims = kwds.pop("keepdims", False)
519              del args  # avoid the possibility of passing both `args` and `kwds`
520  
521              # convert masked arrays to regular arrays with sentinel values
522              samples, sentinel = _masked_arrays_2_sentinel_arrays(samples)
523  
524              # standardize to always work along last axis
525              reduced_axes = axis
526              if axis is None:
527                  if samples:
528                      # when axis=None, take the maximum of all dimensions since
529                      # all the dimensions are reduced.
530                      n_dims = np.max([sample.ndim for sample in samples])
531                      reduced_axes = tuple(range(n_dims))
532                  samples = [np.asarray(sample.ravel()) for sample in samples]
533              else:
534                  # don't ignore any axes when broadcasting if paired
535                  samples = _broadcast_arrays(samples, axis=axis if not paired else None)
536                  axis = np.atleast_1d(axis)
537                  n_axes = len(axis)
538                  # move all axes in `axis` to the end to be raveled
539                  samples = [np.moveaxis(sample, axis, range(-len(axis), 0))
540                             for sample in samples]
541                  shapes = [sample.shape for sample in samples]
542                  # New shape is unchanged for all axes _not_ in `axis`
543                  # At the end, we append the product of the shapes of the axes
544                  # in `axis`. Appending -1 doesn't work for zero-size arrays!
545                  new_shapes = [shape[:-n_axes] + (np.prod(shape[-n_axes:]),)
546                                for shape in shapes]
547                  samples = [sample.reshape(new_shape)
548                             for sample, new_shape in zip(samples, new_shapes)]
549              axis = -1  # work over the last axis
550              NaN = _get_nan(*samples) if samples else np.nan
551  
552              # if axis is not needed, just handle nan_policy and return
553              ndims = np.array([sample.ndim for sample in samples])
554              if np.all(ndims <= 1):
555                  # Addresses nan_policy == "raise"
556                  if nan_policy != 'propagate' or override['nan_propagation']:
557                      contains_nan = [_contains_nan(sample, nan_policy)[0]
558                                      for sample in samples]
559                  else:
560                      # Behave as though there are no NaNs (even if there are)
561                      contains_nan = [False]*len(samples)
562  
563                  # Addresses nan_policy == "propagate"
564                  if any(contains_nan) and (nan_policy == 'propagate'
565                                            and override['nan_propagation']):
566                      res = np.full(n_out, NaN)
567                      res = _add_reduced_axes(res, reduced_axes, keepdims)

595                  empty_output is not None

628                      if sentinel:
629                          samples = _remove_sentinel(samples, paired, sentinel)
630                      if is_too_small(samples, kwds):
631                          warnings.warn(too_small_nd_omit, SmallSampleWarning,
632                                        stacklevel=4)
633                          return np.full(n_out, NaN)
634                      return result_to_tuple(hypotest_fun_out(*samples, **kwds), n_out)
635  
636              # Addresses nan_policy == "propagate"
637              elif (contains_nan and nan_policy == 'propagate'
638                    and override['nan_propagation']):
639                  def hypotest_fun(x):
640                      if np.isnan(x).any():
641                          return np.full(n_out, NaN)

643                      samples = np.split(x, split_indices)[:n_samp+n_kwd_samp]
644                      if sentinel:
645                          samples = _remove_sentinel(samples, paired, sentinel)
646                      if is_too_small(samples, kwds):
647                          return np.full(n_out, NaN)
648                      return result_to_tuple(hypotest_fun_out(*samples, **kwds), n_out)

652                      samples = np.split(x, split_indices)[:n_samp+n_kwd_samp]

654                          samples = _remove_sentinel(samples, paired, sentinel)

656                          return np.full(n_out, NaN)
657                      return result_to_tuple(hypotest_fun_out(*samples, **kwds), n_out)

659              x = np.moveaxis(x, axis, 0)
660              res = np.apply_along_axis(hypotest_fun, axis=0, arr=x)
661              res = _add_reduced_axes(res, reduced_axes, keepdims)
662              return tuple_to_result(*res)
