54   def _compile_and_register_class(obj, rcb, qualified_name):
55       script_class = _get_script_class(obj)
56   
57       if not script_class:
58           ast = get_jit_class_def(obj, obj.__name__)
59           defaults = torch.jit.frontend.get_default_args_for_class(obj)
60           script_class = torch._C._jit_script_class_compile(
61               qualified_name, ast, defaults, rcb
62           )
63           _add_script_class(obj, script_class)
64   
65       return script_class

74   def make_stub_from_method(nn_module, method_name):
75       func = getattr(nn_module, method_name)
76       if isinstance(func, ScriptMethodStub):
77           return func
78       # Make sure the name present in the resulting AST will match the name
79       # requested here. The only time they don't match is if you do something
80       # like:
81       #   def _forward(self):
82       #       pass
83       #   forward = _forward
84       # In this case, the actual function object will have the name `_forward`,
85       # even though we requested a stub for `forward`.
86       return make_stub(func, method_name)

90       stubs = []
91       for name in dir(mod):
92           item = getattr(mod, name, None)
93           if (
94               _jit_internal.get_torchscript_modifier(item)
95               is _jit_internal.FunctionModifiers.EXPORT
96           ):
97               stubs.append(make_stub_from_method(mod, name))

99       return stubs

102  def jit_ignored_properties(module):
103      user_annotated_ignored_attributes = getattr(
104          module, "__jit_ignored_attributes__", []
105      )
106  
107      def get_properties_names(module):
108          return {k for k, v in vars(module).items() if isinstance(v, property)}
109  
110      properties = get_properties_names(type(module))
111      user_annoted_ignored_properties = set()
112  
113      for ignored_attr in user_annotated_ignored_attributes:
114          if ignored_attr in properties:
115              user_annoted_ignored_properties.add(ignored_attr)
116      return user_annoted_ignored_properties

136  def _get_valid_constant(attr, v, owner_type):
137      if isinstance(v, _constant_types):
138          return v
139      elif isinstance(v, (tuple, list)):
140          return tuple(_get_valid_constant(attr, x, owner_type) for x in v)
141      constants = ", ".join(torch.typename(typ) for typ in _constant_types)
142      raise TypeError(
143          textwrap.dedent(
144              f"""
145          '{torch.typename(type(v))}' object in attribute '{owner_type}.{attr}' is not a valid constant.
146          Valid constants are:
147          1. a nn.ModuleList
148          2. a value of type {{{constants}}}
149          3. a list or tuple of (2)
150          """
151          )

160  def get_annotations(obj):
161      if sys.version_info < (3, 10):
162          return getattr(obj, "__annotations__", {})
163      # In Python-3.10+ it is recommended to use inspect.get_annotations
164      # See https://docs.python.org/3.10/howto/annotations.html
165      # But also, in 3.10 annotations from base class are not inherited
166      # by unannotated derived one, so they must be manually extracted
167      annotations = inspect.get_annotations(obj)
168      if annotations:
169          return annotations
170  
171      def get_cls_annotations(cls):
172          cls_annotations = inspect.get_annotations(cls)
173          if cls_annotations:
174              return cls_annotations
175          for base in cls.__bases__:
176              cls_annotations = get_cls_annotations(base)
177              if cls_annotations:
178                  return cls_annotations
179          return {}
180  
181      cls = obj if isinstance(obj, type) else type(obj)
182      return get_cls_annotations(cls)

185  def infer_concrete_type_builder(nn_module, share_types=True):
186      """ 
187      Build a ConcreteModuleTypeBuilder from an nn.Module. 
188   
189      This ConcreteModuleType doesn't have a JIT type associated with it yet, it 
190      must be filled in by the caller. 
191      """
192      concrete_type_builder = torch._C.ConcreteModuleTypeBuilder(type(nn_module))
193      if isinstance(nn_module, (torch.nn.ModuleDict)):
194          concrete_type_builder.set_module_dict()
195      if isinstance(nn_module, (torch.nn.ModuleList, torch.nn.Sequential)):
196          concrete_type_builder.set_module_list()
197      if isinstance(nn_module, (torch.nn.ParameterList)):
198          concrete_type_builder.set_parameter_list()
199      if isinstance(nn_module, (torch.nn.ParameterDict)):
200          concrete_type_builder.set_parameter_dict()
201  
202      class_annotations = get_annotations(nn_module)
203      if isinstance(nn_module, (torch.ao.quantization.QuantWrapper)):
204          class_annotations = {}
205  
206      # Get user-annotated ignored attributes.
207      user_annotated_ignored_attributes = getattr(
208          nn_module, "__jit_ignored_attributes__", []
209      )
210      concrete_type_builder.add_ignored_attributes(user_annotated_ignored_attributes)
211      ignored_properties = jit_ignored_properties(nn_module)
212  
213      # try to infer the type from type annotation or from the object itself
214      def infer_type(name, item):
215          # The forward function from Module is special; never use this annotations; we
216          # need to infer type directly using JIT.  I originally wanted to write
217          # this test as isinstance(class_annotations[name], Callable) but
218          # isinstance on typing things doesn't seem to work: isinstance(list, Callable)
219          # is also true!
220          inferred = False
221          try:
222              if (
223                  name in class_annotations
224                  and class_annotations[name]
225                  != torch.nn.Module.__annotations__["forward"]
226              ):
227                  ann_to_type = torch.jit.annotations.ann_to_type(
228                      class_annotations[name], fake_range()
229                  )
230                  attr_type = torch._C.InferredType(ann_to_type)
231              elif isinstance(item, torch.jit.Attribute):
232                  ann_to_type = torch.jit.annotations.ann_to_type(item.type, fake_range())
233                  attr_type = torch._C.InferredType(ann_to_type)
234              else:
235                  attr_type = torch._C._jit_try_infer_type(item)
236                  inferred = True
237          except RuntimeError as re:
238              raise RuntimeError(f"Error inferring type for {name}: {item}: {re}") from re
239  
240          return attr_type, inferred
241  
242      added_names = set()
243  
244      for name, item in nn_module._parameters.items():
245          if name in user_annotated_ignored_attributes:
246              continue
247  
248          assert item is None or isinstance(item, torch.Tensor)
249          attr_type, _ = infer_type(name, item)
250          # We currently have the invariant in various places in our code
251          # that parameters must be Tensors. However, the nn.Module API also
252          # allows NoneType parameters. These parameters are not returned as
253          # part of `parameters()` and its variants, but are available
254          # through direct attribute access.
255          concrete_type_builder.add_attribute(name, attr_type.type(), True, False)
256          added_names.add(name)
257  
258      for name, item in nn_module._buffers.items():
259          if name in user_annotated_ignored_attributes:
260              continue
261  
262          assert item is None or isinstance(item, torch.Tensor)
263          attr_type, _ = infer_type(name, item)
264          concrete_type_builder.add_attribute(name, attr_type.type(), False, True)
265          added_names.add(name)
266  
267      for name, item in nn_module._modules.items():
268          if name in user_annotated_ignored_attributes:
269              continue
270  
271          attr_type, _ = infer_type(name, item)
272          if item is None:
273              # Modules can be None. We don't have direct support for optional
274              # Modules, so the register it as an NoneType attribute instead.
275              concrete_type_builder.add_attribute(name, attr_type.type(), False, False)
276              continue
277          if attr_type.success():
278              assert attr_type.type().is_interface_type()
279              # if the type can be inferred, it should be a module interface type
280              sub_concrete_type = torch._C.ConcreteModuleType.from_jit_type(
281                  attr_type.type()
282              )
283          else:
284              # otherwise we get the concrete module type for item and add it to concrete_type
285              sub_concrete_type = get_module_concrete_type(item, share_types)
286          concrete_type_builder.add_module(name, sub_concrete_type)
287  
288          added_names.add(name)
289  
290      # populate constants_set
291      constants_set = set(getattr(nn_module, "__constants__", ()))
292  
293      # Constants annotated via `Final[T]` rather than being added to `__constants__`
294      for name, ann in class_annotations.items():
295          if torch._jit_internal.is_final(ann):
296              constants_set.add(name)
297  
298      for name in constants_set:
299          if name in added_names:
300              # TODO: We should really error in this case, but its bc-breaking so
301              # we need to warn for at least one release
302              if name in nn_module._modules:
303                  hint = "submodule"
304              elif name in nn_module._buffers:
305                  hint = "buffer"
306              elif name in nn_module._parameters:
307                  hint = "parameter"
308              else:
309                  raise AssertionError(
310                      "added_names must be submodule, parameter, or buffer"
311                  )
312  
313              warnings.warn(
314                  f"'{name}' was found in ScriptModule constants, "
315                  f" but it is a non-constant {hint}. Consider removing it."
316              )
317              continue
318          if not hasattr(nn_module, name):
319              # TODO: We should really error in this case, but its bc-breaking so
320              # we need to warn for at least one release
321              warnings.warn(
322                  f"'{name}' was found in ScriptModule constants, "
323                  "but was not actually set in __init__. "
324                  "Consider removing it."
325              )
326              continue
327          value = getattr(nn_module, name)
328          concrete_type_builder.add_constant(
329              name, _get_valid_constant(name, value, type(nn_module).__name__)
330          )
331          added_names.add(name)
332  
333      # populate overloads
334      overloads = getattr(nn_module, "__overloads__", {})
335      # update with any annotated overloads
336      overloads.update(
337          get_overload_name_mapping(
338              get_overload_annotations(nn_module, ignored_properties)
339          )
340      )
341      for name, overloaded_names in overloads.items():
342          concrete_type_builder.add_overload(name, overloaded_names)
343  
344      for name, value in nn_module.__dict__.items():
345          if name in ignored_attributes or name.startswith("__"):
346              # Python objects have lots of random attributes attached to them;
347              # PyTorch adds a few more. Prevent these from getting compiled.
348              continue
349  
350          if name in user_annotated_ignored_attributes:
351              continue
352  
353          if name in added_names:
354              # Don't re-add anything we already added
355              continue
356  
357          isoverloadpacket = isinstance(value, torch._ops.OpOverloadPacket)
358          if isoverloadpacket:
359              value = value.op
360          # Handle Python function attributes
361          if inspect.isfunction(value):
362              try:
363                  scripted_fn = torch.jit.script(value)
364                  concrete_type_builder.add_function_attribute(
365                      name, torch._C._jit_try_infer_type(scripted_fn).type(), value
366                  )
367              except Exception as e:
368                  # If we fail to script the function, it isn't a hard error.
369                  # Instead, we will add it to the list of attributes we failed
370                  # to convert, with the compilation error.
371                  hint = (
372                      "(This function exists as an attribute on the Python module, "
373                      "but we failed to compile it to a TorchScript function. "
374                      f"\nThe error stack is reproduced here:\n{e}"
375                  )
376                  concrete_type_builder.add_failed_attribute(name, hint)
377  
378              continue
379  
380          # Handle calls to builtin functions (either bespoke builtins from torch.jit._builtins or
381          # a call to an aten function like torch.add)
382          builtin_symbol_name = _find_builtin(value)
383          if builtin_symbol_name:
384              concrete_type_builder.add_builtin_function(name, builtin_symbol_name)
385              continue
386  
387          # Handle Script function attributes
388          if isinstance(value, torch.jit.ScriptFunction):
389              concrete_type_builder.add_function_attribute(
390                  name, torch._C._jit_try_infer_type(value).type(), value
391              )
392              continue
393  
394          # If we got here, this is a regular "data" attribute, add it to the concrete type
395          attr_type, inferred = infer_type(name, value)
396          if attr_type.success():
397              concrete_type_builder.add_attribute(name, attr_type.type(), False, False)
398          else:
399              # TODO: could add more detail here. For example, what the user should do
400              # when the pytype is `list` or `NoneType`
401              inferred_msg = (
402                  "Its type was inferred; try adding a type annotation for the attribute."
403                  if inferred
404                  else ""
405              )
406              additional_info = f"{attr_type.reason()}. {inferred_msg}"
407              hint = (
408                  "(This attribute exists on the Python module, "
409                  f"but we failed to convert Python type: '{torch.typename(type(value))}' "
410                  f"to a TorchScript type. {additional_info})"
411              )
412              concrete_type_builder.add_failed_attribute(name, hint)
413  
414      # add hooks to concrete type
415      for hook in nn_module._forward_hooks.values():
416          concrete_type_builder.add_forward_hook(hook)
417      for pre_hook in nn_module._forward_pre_hooks.values():
418          concrete_type_builder.add_forward_pre_hook(pre_hook)
419  
420      return concrete_type_builder

423  class ConcreteTypeStore:
424      type_store: dict[type[Module], list[torch._C.ConcreteModuleType]]
425      methods_compiled: set[torch._C.ConcreteModuleType]
426  
427      def __init__(self) -> None:
428          # Python module type => List[ConcreteModuleType)]
429          self.type_store = {}
430          # ConcreteTypes that have had their methods already compiled
431          self.methods_compiled = set()
432  
433      def get_or_create_concrete_type(self, nn_module):
434          """Infer a ConcreteType from this `nn.Module` instance. Underlying JIT types are re-used if possible."""
435          concrete_type_builder = infer_concrete_type_builder(nn_module)
436  
437          nn_module_type = type(nn_module)
438          if nn_module_type not in self.type_store:
439              self.type_store[nn_module_type] = []
440  
441          # Search the type store for an already-available JIT type
442          known_types = self.type_store[nn_module_type]
443          for known_type in known_types:
444              if known_type.equals(concrete_type_builder):
445                  return known_type
446  
447          # We didn't find anything; generate a new JIT type from this concrete type
448          concrete_type = concrete_type_builder.build()
449          self.type_store[nn_module_type].append(concrete_type)
450          return concrete_type

481  def get_module_concrete_type(nn_module, share_types=True):
482      """ 
483      Get a concrete type for nn_modules. 
484   
485      If share_types is True, the concrete type is fetched from concrete_type_store. 
486      If it is False, a new concrete type is created without first searching concrete_type_store. 
487   
488      Args: 
489          nn_module:  The original Python nn.Module that we are creating a ScriptModule for. 
490          share_types = Whether to share underlying JIT types between modules (if possible). 
491   
492      Returns: 
493          A concrete type for nn_module. 
494      """
495      assert isinstance(nn_module, Module)
496      if isinstance(nn_module, torch.jit.ScriptModule) and hasattr(
497          nn_module, "_concrete_type"
498      ):
499          return nn_module._concrete_type
500  
501      if share_types:
502          # Look into the store of cached JIT types
503          concrete_type = concrete_type_store.get_or_create_concrete_type(nn_module)
504      else:
505          # Get a concrete type directly, without trying to re-use an existing JIT
506          # type from the type store.
507          concrete_type_builder = infer_concrete_type_builder(nn_module, share_types)
508          concrete_type_builder.set_poisoned()
509          concrete_type = concrete_type_builder.build()
510  
511      return concrete_type

514  def create_script_class(obj):
515      """ 
516      Create and return a RecursiveScriptClass instance from a Python object. 
517   
518      Arguments: 
519          obj: A Python object. 
520      """
521      qualified_class_name = _jit_internal._qualified_name(type(obj))
522      rcb = _jit_internal.createResolutionCallbackForClassMethods(type(obj))
523      # Script the type of obj if it hasn't already been scripted.
524      _compile_and_register_class(type(obj), rcb, qualified_class_name)
525      class_ty = _python_cu.get_class(qualified_class_name)
526      # Create an empty torch._C.ScriptObject with the scripted type.
527      cpp_object = torch._C._create_object_with_type(class_ty)
528      # Copy all of the attributes over to the torch._C.ScriptObject.
529      for name, value in obj.__dict__.items():
530          cpp_object.setattr(name, value)
531  
532      # Wrap the torch._C.ScriptObject in a RecursiveScriptClass instance.
533      return wrap_cpp_class(cpp_object)

536  def create_script_module(nn_module, stubs_fn, share_types=True, is_tracing=False):
537      """ 
538      Create a new ScriptModule from an nn.Module. 
539   
540      Args: 
541          nn_module:  The original Python nn.Module that we are creating a ScriptModule for. 
542          stubs_fn:  Lambda that takes an nn.Module and generates a list of ScriptMethodStubs to compile. 
543          share_types:  Whether to share underlying JIT types between modules (if possible). 
544              NOTE: Only set to False this when we cannot guarantee type sharing will work 
545                  correctly. This only happens today for traced modules, where the same 
546                  module can produce different traced methods depending on the inputs. 
547          is_tracing: Whether this function is called during tracing or scripting. If tracing, 
548                  we don't need to do AttributeTypeIsSupportedChecker because all the unsupported 
549                  attributes will be baked as constant in the tracing graph. In addition, 
550                  this check significantly slows down the traced modules when the module size is big. 
551      """
552      assert not isinstance(nn_module, torch.jit.RecursiveScriptModule)
553      check_module_initialized(nn_module)
554      concrete_type = get_module_concrete_type(nn_module, share_types)
555      if not is_tracing:
556          AttributeTypeIsSupportedChecker().check(nn_module)
557      return create_script_module_impl(nn_module, concrete_type, stubs_fn)
558

560  def create_script_module_impl(nn_module, concrete_type, stubs_fn):
561      """ 
562      Convert an nn.Module to a RecursiveScriptModule. 
563   
564      Args: 
565          nn_module:  The original Python nn.Module that we are creating a ScriptModule for. 
566          concrete_type:  The fully initialized ConcreteType of the module. 
567          stubs_fn:  Lambda that takes an nn.Module and generates a list of ScriptMethodStubs to compile. 
568      """
569      cpp_module = torch._C._create_module_with_type(concrete_type.jit_type)
570      method_stubs = stubs_fn(nn_module)
571      property_stubs = get_property_stubs(nn_module)
572      hook_stubs, pre_hook_stubs = get_hook_stubs(nn_module)
573      ignored_properties = jit_ignored_properties(nn_module)
574  
575      def init_fn(script_module):
576          # Initialize the ScriptModule:
577          # 1. Copy the attributes/parameters/buffers from the original `nn_module` to the new ScriptModule.
578          for name in concrete_type.get_attributes().keys():
579              orig_value = getattr(nn_module, name)
580              orig_value = (
581                  orig_value.value
582                  if isinstance(orig_value, torch.jit.Attribute)
583                  else orig_value
584              )
585              cpp_module.setattr(name, orig_value)
586  
587          # 2. Copy the submodules from the original `nn_module` to the new ScriptModule,
588          #    recursively scripting them.
589          for name, sub_concrete_type in concrete_type.get_modules():
590              orig_value = getattr(nn_module, name)
591              assert isinstance(orig_value, Module), (
592                  f"Expected Module but got {type(orig_value)}"
593              )
594              module_type = sub_concrete_type.jit_type
595              if isinstance(module_type, torch._C.InterfaceType):
596                  # use the interface inference rule to compile the module
597                  scripted = interface_script(module_type, orig_value)
598              elif isinstance(orig_value, torch.jit.ScriptModule):
599                  scripted = orig_value
600              else:
601                  # always reuse the provided stubs_fn to infer the methods to compile
602                  scripted = create_script_module_impl(
603                      orig_value, sub_concrete_type, stubs_fn
604                  )
605  
606              cpp_module.setattr(name, scripted)
607              script_module._modules[name] = scripted
608  
609          # 3. Copy @ignored/@unused methods and attrs from the original `nn_module` to the new ScriptModule.
610          #    This ensures we can access these Python methods on the ScriptModule.
611          for name in dir(nn_module):
612              if name in ignored_properties:
613                  continue
614              item = getattr(nn_module, name, None)
615              if inspect.ismethod(item) and _jit_internal.is_ignored_fn(item):
616                  unbound_function = getattr(nn_module, name).__func__
617                  bound_method = unbound_function.__get__(script_module)
618                  setattr(script_module, name, bound_method)
619              elif concrete_type.is_ignored_attribute(name):
620                  setattr(script_module, name, item)
621  
622          # For convenience, attach the concrete type to the new ScriptModule
623          script_module._concrete_type = concrete_type
624  
625      # Actually create the ScriptModule, initializing it with the function we just defined
626      script_module = torch.jit.RecursiveScriptModule._construct(cpp_module, init_fn)
627  
628      # Compile methods if necessary
629      if concrete_type not in concrete_type_store.methods_compiled:
630          create_methods_and_properties_from_stubs(
631              concrete_type, method_stubs, property_stubs
632          )
633          # Create hooks after methods to ensure no name collisions between hooks and methods.
634          # If done before, hooks can overshadow methods that aren't exported.
635          create_hooks_from_stubs(concrete_type, hook_stubs, pre_hook_stubs)
636          torch._C._run_emit_module_hook(cpp_module)
637          concrete_type_store.methods_compiled.add(concrete_type)
638  
639      # Copy the forward hooks and pre-hooks to the new ScriptModule
640      # to allow the hooks to be run from eager as ScriptFunctions
641      for idx, fn in enumerate(script_module._c._get_forward_pre_hooks()):
642          script_module._forward_pre_hooks[idx] = fn
643      for idx, fn in enumerate(script_module._c._get_forward_hooks()):
644          script_module._forward_hooks[idx] = fn
645  
646      # Special handling so methods like __len__ work in script methods on classes derived from containers
647      if (
648          isinstance(
649              nn_module, (torch.nn.ModuleList, torch.nn.Sequential, torch.nn.ModuleDict)
650          )
651          and "__len__" not in cpp_module._method_names()
652      ):
653          script_module.define(f"def __len__(self):\n   return {len(nn_module)}\n")
654      if (
655          isinstance(nn_module, torch.nn.ModuleDict)
656          and "__contains__" not in cpp_module._method_names()
657      ):
658          if len(nn_module.keys()):
659              keys = repr(list(nn_module.keys()))
660              script_module.define(
661                  f"def __contains__(self, key: str):\n   return key in {keys}\n"
662              )
663          else:
664              script_module.define("def __contains__(self, key: str):\n   return False\n")
665  
666      # Make the compiled methods available to the Python ScriptModule class.
667      for method_stub in method_stubs:
668          if method_stub.original_method is None:
669              # define()'d methods don't have an Python original_method, so we
670              # don't need to do any Python re-wrapping stuff
671              continue
672  
673          name = method_stub.original_method.__name__
674          if name != method_stub.def_.name().name:
675              # TODO: Why skip this? Because @torch.jit._overload_method will
676              # mangle the name of the function.
677              continue
678          script_method = cpp_module._get_method(name)
679  
680          # Wrap the original to propagate docstrings and such.
681          # TODO: we don't currently do this functions that are recursively
682          # compiled, we should.
683          wrapped_script_method = functools.wraps(method_stub.original_method)(
684              script_method
685          )
686  
687          # Add the methods to the script_module directly. This ensures they will
688          # be found first when `name` is looked up (as opposed to the stubs or
689          # nn.Module.forward)
690          script_module.__dict__[name] = wrapped_script_method
691  
692      # Make module properties available on the Python ScriptModule class.
693      for property_stub in property_stubs:
694          property_name = property_stub.def_.name().name
695          fget = cpp_module._get_method(property_stub.def_.getter_name().name)
696          # Setter is optional, so it may not exist.
697          setter_name = property_stub.def_.setter_name()
698          fset = cpp_module._get_method(setter_name.name) if setter_name else None
699          script_module.__dict__[property_name] = property(property_name, fget, fset)  # type: ignore[arg-type]
700  
701      # copy over python methods to script module if they aren't defined on the script module
702      # this is currently an internal api used only on module containers
703      for name in dir(nn_module):
704          if name in ignored_properties:
705              continue
706          item = getattr(nn_module, name, None)
707          if (
708              _jit_internal.get_torchscript_modifier(item)
709              is _jit_internal.FunctionModifiers.COPY_TO_SCRIPT_WRAPPER
710          ):
711              add_python_attr_to_scripted_model(script_module, nn_module, name)
712  
713      return script_module
714

716  # We define shims of certain attributes on the RecursiveScriptModule to support
717  # magic methods. To check if a script model defines an attribute we need
718  # to also check that the attribute is not the shim
719  def script_model_defines_attr(script_model, attr):
720      script_attr = getattr(script_model, attr, None)
721      if script_attr is None:
722          return False
723      default_attr = getattr(torch.jit.RecursiveScriptModule, attr, None)
724      if default_attr is None:
725          return False
726      return script_attr != default_attr

734  def get_overload_annotations(mod, jit_ignored_properties):
735      # original function => [(mangled overload name, overload function)]
736      overloads = {}
737  
738      for name in dir(type(mod)):
739          if name in jit_ignored_properties:
740              continue
741          item = getattr(mod, name, None)
742          if not callable(item):
743              continue
744  
745          # builtin functions like repr() in python 2 do not have __module__ defined
746          if hasattr(item, "__module__") and item.__module__ is not None:
747              method_overloads = _jit_internal._get_overloaded_methods(
748                  item, mod.__class__
749              )
750              if method_overloads is None:
751                  continue
752  
753              if item.__func__ in method_overloads:
754                  raise RuntimeError(
755                      _jit_internal.get_overload_no_implementation_error_message(
756                          "method", item.__func__
757                      )
758                  )
759  
760              names = [name + "__" + str(i) for i in range(len(method_overloads))]
761              overloads[item] = list(zip(names, method_overloads))
762  
763      return overloads

766  def get_overload_name_mapping(overload_info):
767      # Same format as __overloads__
768      # original function => [overload names]
769      overload_name_mappings: dict[str, list[str]] = {}
770      for orig_fn, overloads in overload_info.items():
771          original_name = orig_fn.__name__
772          if original_name not in overload_name_mappings:
773              overload_name_mappings[original_name] = []
774  
775          for overload_name, _ in overloads:
776              overload_name_mappings[original_name].append(overload_name)
777      return overload_name_mappings

791  def make_stubs_for_overloads(overload_info):
792      overload_stubs = []
793      for orig_fn, overloads in overload_info.items():
794          orig_ast = get_jit_def(
795              orig_fn, orig_fn.__name__, self_name="RecursiveScriptModule"
796          )
797          for overload_name, overload_fn in overloads:
798              _check_no_signature(overload_fn)
799              over_ast = get_jit_def(
800                  overload_fn, overload_fn.__name__, self_name="RecursiveScriptModule"
801              )
802              new_ast = torch._C._replace_overloaded_method_decl(
803                  over_ast.decl(), orig_ast, overload_name
804              )
805              _rcb = _jit_internal.createResolutionCallbackFromClosure(orig_fn)
806              overload_stubs.append(ScriptMethodStub(_rcb, new_ast, overload_fn))
807      return overload_stubs

810  def check_module_initialized(mod):
811      assert isinstance(mod, torch.nn.Module)
812      if not hasattr(mod, "_parameters"):
813          raise RuntimeError(
814              f"'{torch.typename(type(mod))}' has not been initialized, did you forget to call 'super()'?"
815          )
816  
817      # This is to avoid importing torch.distributed.nn
818      if not hasattr(mod, "remote_parameters"):
819          for name, param in mod._parameters.items():
820              if param is not None and torch.nn.parameter.is_lazy(param):
821                  raise RuntimeError(
822                      f"'{torch.typename(type(mod))}' has uninitialized parameters {name}. Did you forget to run a forward pass?"
823                  )
824          for name, buf in mod._buffers.items():
825              if buf is not None and torch.nn.parameter.is_lazy(buf):
826                  raise RuntimeError(
827                      f"'{torch.typename(type(mod))}' has uninitialized buffers {name}. Did you forget to run a forward pass?"
828                  )

831  def infer_methods_to_compile(nn_module):
832      """Implement the default rules for which methods should act as starting points for compilation. 
833   
834      (TODO add a link when the rules are published). 
835      """
836      check_module_initialized(nn_module)
837      ignored_properties = jit_ignored_properties(nn_module)
838  
839      methods: list[str] = []
840      if hasattr(nn_module, "forward") and not _jit_internal.is_ignored_fn(
841          nn_module.forward
842      ):
843          forward_func = getattr(nn_module.forward, "__func__", None)
844          module_forward = getattr(torch.nn.Module, "forward", None)
845          if forward_func != module_forward:
846              methods = ["forward"]
847  
848      exported = []
849      for name in dir(nn_module):
850          if name in ignored_properties:
851              continue
852          item = getattr(nn_module, name, None)
853          if (
854              _jit_internal.get_torchscript_modifier(item)
855              is _jit_internal.FunctionModifiers.EXPORT
856          ):
857              exported.append(name)
858  
859      methods = methods + exported
860  
861      overload_name_mappings = dict(getattr(nn_module, "__overloads__", {}))
862      overload_info = get_overload_annotations(nn_module, ignored_properties)
863      overload_name_mappings.update(get_overload_name_mapping(overload_info))
864      overload_stubs = make_stubs_for_overloads(overload_info)
865  
866      nn_module.__overloads__ = overload_name_mappings
867  
868      # we shouldn't directly compile overloaded methods, just its overloads
869      def ignore_overloaded(method_name):
870          return method_name not in overload_name_mappings
871  
872      filtered_methods = filter(ignore_overloaded, methods)
873  
874      # Unique the methods. We don't want to use a set to store the methods because it
875      # introduces non-determinism to compile order.
876      uniquer: set[str] = set()
877      uniqued_methods = []
878      for name in filtered_methods:
879          if name in uniquer:
880              continue
881          uniqued_methods.append(name)
882          uniquer.add(name)
883  
884      stubs = [make_stub_from_method(nn_module, method) for method in uniqued_methods]
885      return overload_stubs + stubs
886

888  def get_hook_stubs(nn_module):
889      """Return forward hook and pre_hook ScriptModuleStubs."""
890      check_module_initialized(nn_module)
891      hook_map: dict = {}
892  
893      hook_stubs = []
894      for hook in nn_module._forward_hooks.values():
895          if hook.__name__ in hook_map:
896              if id(hook) != id(hook_map[hook.__name__]):
897                  raise RuntimeError(
898                      f"Hook '{hook.__name__}' on {type(nn_module).__name__} "
899                      "has at least two different python definitions."
900                      " Please use unique names for all hooks."
901                  )
902          else:
903              hook_map[hook.__name__] = hook
904          hook_stubs.append(make_stub(hook, hook.__name__))
905  
906      pre_hook_stubs = []
907      for pre_hook in nn_module._forward_pre_hooks.values():
908          if pre_hook.__name__ in hook_map:
909              if id(pre_hook) != id(hook_map[pre_hook.__name__]):
910                  raise RuntimeError(
911                      f"Pre-hook '{pre_hook.__name__}' on {type(nn_module).__name__} "
912                      "has at least two different python definitions."
913                      " Please use unique names for all hooks."
914                  )
915          else:
916              hook_map[pre_hook.__name__] = pre_hook
917          pre_hook_stubs.append(make_stub(pre_hook, pre_hook.__name__))
918  
919      return hook_stubs, pre_hook_stubs

922  def get_property_stubs(nn_module):
923      """Create property stubs for the properties of the module by creating method stubs for the getter and setter."""
924      module_ty = type(nn_module)
925      properties_asts = get_class_properties(module_ty, self_name="RecursiveScriptModule")
926      rcbs = {}
927  
928      for name in dir(module_ty):
929          item = getattr(module_ty, name, None)
930          if isinstance(item, property):
931              if not item.fget:
932                  raise RuntimeError(
933                      f"Property {name} of {nn_module.__name__} must have a getter"
934                  )
935  
936              rcbs[name] = _jit_internal.createResolutionCallbackFromClosure(item.fget)
937  
938      stubs = [PropertyStub(rcbs[ast.name().name], ast) for ast in properties_asts]
939      return stubs

942  def interface_script(mod_interface, nn_module):
943      """ 
944      Make a ScriptModule from an nn.Module, using the interface methods rule for determining which methods to compile. 
945   
946      Args: 
947          mod_interface: the interface type that the module have 
948          nn_module:  The original Python nn.Module that we are creating a ScriptModule for. 
949      """
950      if isinstance(nn_module, torch.jit.ScriptModule):
951          return nn_module
952  
953      check_module_initialized(nn_module)
954  
955      def infer_interface_methods_to_compile(nn_module):
956          """Rule to infer the methods from the interface type. 
957   
958          It is used to know which methods need to act as starting points for compilation. 
959          """
960          stubs = [
961              make_stub_from_method(nn_module, method)
962              for method in mod_interface.getMethodNames()
963          ]
964          return stubs
965  
966      return create_script_module(nn_module, infer_interface_methods_to_compile)

969  def try_compile_fn(fn, loc):
970      if _jit_internal.is_ignored_fn(fn):
971          # Don't do anything for @ignore'd functions
972          return None
973  
974      if isinstance(fn, torch.nn.Module):
975          # Since modules are callable pybind recognizes them as functions, but
976          # don't do anything for them
977          return None
978  
979      if not inspect.isfunction(fn) and not inspect.ismethod(fn):
980          raise RuntimeError(
981              f"`{fn}` is not a function. Recursive scripting only supports "
982              "Python functions or methods currently.\n"
983              f"Consider manually annotating `{fn}` with @torch.jit.script."
984          )
985  
986      # The object returned by __prepare_scriptable__ might have a different closure.
987      # Resolve it here to get the right resolution callback.
988      fn = fn.__prepare_scriptable__() if hasattr(fn, "__prepare_scriptable__") else fn  # type: ignore[operator]
989  
990      # We don't have the actual scope where the function was defined, but we can
991      # extract the necessary info from the closed over variables on the function
992      # object
993      rcb = _jit_internal.createResolutionCallbackFromClosure(fn)
994      return torch.jit.script(fn, _rcb=rcb)

1002 def wrap_cpp_module(cpp_module):
1003     """Wrap this torch._C.ScriptModule in a Python ScriptModule, recursively for all submodules."""
1004 
1005     def init_fn(script_module):
1006         for name, cpp_module in torch._C.ModuleDict(script_module._c).items():
1007             setattr(script_module, name, wrap_cpp_module(cpp_module))
1008         script_module._concrete_type = torch._C.ConcreteModuleType.from_jit_type(
1009             script_module._c._type()
1010         )
1011 
1012         for idx, fn in enumerate(script_module._c._get_forward_pre_hooks()):
1013             script_module._forward_pre_hooks[idx] = fn
1014         for idx, fn in enumerate(script_module._c._get_forward_hooks()):
1015             script_module._forward_hooks[idx] = fn
1016 
1017     return torch.jit.RecursiveScriptModule._construct(cpp_module, init_fn)

1020 def compile_unbound_method(concrete_type, fn):
1021     if _jit_internal.is_ignored_fn(fn):
1022         return None
1023     stub = make_stub(fn, fn.__name__)
1024     with torch._jit_internal._disable_emit_hooks():
1025         # We don't want to call the hooks here since the graph that is calling
1026         # this function is not yet complete
1027         create_methods_and_properties_from_stubs(concrete_type, (stub,), ())
1028     return stub

1031 def lazy_bind(concrete_type, unbound_method):
1032     """ 
1033     Return a function that lazily binds `unbound_method` to a provided Module IValue, then invokes the method. 
1034  
1035     We do this so that any Python shenanigans that 
1036     will poison type sharing are impossible at compile time. 
1037     """
1038 
1039     def lazy_binding_method(cpp_module, *args):
1040         def init_fn(script_module):
1041             orig_class = concrete_type.py_class
1042 
1043             # Copy @ignored/@unused methods from the original module to the new one.
1044             # This ensures they are available during execution.
1045             for name in dir(orig_class):
1046                 item = getattr(orig_class, name, None)
1047                 if _jit_internal.is_ignored_fn(item):
1048                     setattr(script_module, name, item)
1049 
1050             # Copy constants over so they are available during execution.
1051             for name, value in concrete_type.get_constants().items():
1052                 setattr(script_module, name, value)
1053 
1054         script_module = torch.jit.RecursiveScriptModule._construct(cpp_module, init_fn)
1055         method = types.MethodType(unbound_method, script_module)
1056         return method(*args)
1057 
1058     # make the lazy binding method "look like" the original method
1059     lazy_binding_method.original_fn = unbound_method  # type: ignore[attr-defined]
1060     lazy_binding_method.__name__ = unbound_method.__name__
1061     torch._jit_internal.copy_torchscript_modifier(unbound_method, lazy_binding_method)
1062 
1063     return lazy_binding_method
