12           class M(torch.nn.Module):

14                   return torch.add(x, y)

18           self.__origin__ = TensorType

22           return f"TensorType[{self.__args__}]"

25           if isinstance(other, self.__class__):
26               return list(self.__args__) == list(other.__args__)

28               return False

59   def is_consistent(t1, t2):
60       """ 
61       A binary relation denoted by ~ that determines if t1 is consistent with t2. 
62       The relation is reflexive, symmetric but not transitive. 
63       returns True if t1 and t2 are consistent and False otherwise. 
64       Example: 
65           Dyn ~ TensorType((1,2,3)) 
66           int ~ Dyn 
67           int ~ int 
68           TensorType((1,Dyn,3)) ~ TensorType((1,2,3)) 
69       """
70   
71       if t1 == t2:
72           return True
73   
74       if t1 == Dyn or t2 == Dyn or isinstance(t1, Var) or isinstance(t2, Var):
75           return True
76   
77       if isinstance(t1, TensorType) and isinstance(t2, TensorType):
78           return len(t1.__args__) == len(t2.__args__) and all(
79               is_consistent(elem1, elem2)
80               for elem1, elem2 in zip(t1.__args__, t2.__args__)
81           )
82       else:
83           return False
