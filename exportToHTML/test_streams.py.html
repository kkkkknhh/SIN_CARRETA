24   @contextmanager
25   def setup_test_file():
26       val = b'a\x00string'
27       fd, fname = mkstemp()
28   
29       with os.fdopen(fd, 'wb') as fs:
30           fs.write(val)
31       with open(fname, 'rb') as fs:
32           gs = BytesIO(val)
33           cs = BytesIO(val)
34           yield fs, gs, cs
35       os.unlink(fname)

38   def test_make_stream():
39       with setup_test_file() as (fs, gs, cs):
40           # test stream initialization
41           assert_(isinstance(make_stream(gs), GenericStream))

44   def test_tell_seek():
45       with setup_test_file() as (fs, gs, cs):
46           for s in (fs, gs, cs):
47               st = make_stream(s)
48               res = st.seek(0)
49               assert_equal(res, 0)
50               assert_equal(st.tell(), 0)
51               res = st.seek(5)
52               assert_equal(res, 0)
53               assert_equal(st.tell(), 5)
54               res = st.seek(2, 1)
55               assert_equal(res, 0)
56               assert_equal(st.tell(), 7)
57               res = st.seek(-2, 2)
58               assert_equal(res, 0)
59               assert_equal(st.tell(), 6)

62   def test_read():
63       with setup_test_file() as (fs, gs, cs):
64           for s in (fs, gs, cs):
65               st = make_stream(s)
66               st.seek(0)
67               res = st.read(-1)
68               assert_equal(res, b'a\x00string')
69               st.seek(0)
70               res = st.read(4)
71               assert_equal(res, b'a\x00st')
72               # read into
73               st.seek(0)
74               res = _read_into(st, 4)
75               assert_equal(res, b'a\x00st')
76               res = _read_into(st, 4)
77               assert_equal(res, b'ring')
78               assert_raises(OSError, _read_into, st, 2)
79               # read alloc
80               st.seek(0)
81               res = _read_string(st, 4)
82               assert_equal(res, b'a\x00st')
83               res = _read_string(st, 4)
84               assert_equal(res, b'ring')
85               assert_raises(OSError, _read_string, st, 2)

88   class TestZlibInputStream:
89       def _get_data(self, size):
90           data = np.random.randint(0, 256, size).astype(np.uint8).tobytes()
91           compressed_data = zlib.compress(data)
92           stream = BytesIO(compressed_data)
93           return stream, len(compressed_data), data
94   
95       def test_read(self):
96           SIZES = [0, 1, 10, BLOCK_SIZE//2, BLOCK_SIZE-1,
97                    BLOCK_SIZE, BLOCK_SIZE+1, 2*BLOCK_SIZE-1]
98   
99           READ_SIZES = [BLOCK_SIZE//2, BLOCK_SIZE-1,
100                        BLOCK_SIZE, BLOCK_SIZE+1]
101  
102          def check(size, read_size):
103              compressed_stream, compressed_data_len, data = self._get_data(size)
104              stream = ZlibInputStream(compressed_stream, compressed_data_len)
105              data2 = b''
106              so_far = 0
107              while True:
108                  block = stream.read(min(read_size,
109                                          size - so_far))
110                  if not block:
111                      break
112                  so_far += len(block)
113                  data2 += block
114              assert_equal(data, data2)
115  
116          for size in SIZES:
117              for read_size in READ_SIZES:
118                  check(size, read_size)

120      def test_read_max_length(self):
121          size = 1234
122          data = np.random.randint(0, 256, size).astype(np.uint8).tobytes()
123          compressed_data = zlib.compress(data)
124          compressed_stream = BytesIO(compressed_data + b"abbacaca")
125          stream = ZlibInputStream(compressed_stream, len(compressed_data))
126  
127          stream.read(len(data))
128          assert_equal(compressed_stream.tell(), len(compressed_data))
129  
130          assert_raises(OSError, stream.read, 1)

134          compressed_data = zlib.compress(data)

145      def test_seek(self):
146          compressed_stream, compressed_data_len, data = self._get_data(1024)
147  
148          stream = ZlibInputStream(compressed_stream, compressed_data_len)
149  
150          stream.seek(123)
151          p = 123
152          assert_equal(stream.tell(), p)
153          d1 = stream.read(11)
154          assert_equal(d1, data[p:p+11])
155  
156          stream.seek(321, 1)
157          p = 123+11+321
158          assert_equal(stream.tell(), p)
159          d2 = stream.read(21)
160          assert_equal(d2, data[p:p+21])
161  
162          stream.seek(641, 0)
163          p = 641
164          assert_equal(stream.tell(), p)
165          d3 = stream.read(11)
166          assert_equal(d3, data[p:p+11])
167  
168          assert_raises(OSError, stream.seek, 10, 2)
169          assert_raises(OSError, stream.seek, -1, 1)
170          assert_raises(ValueError, stream.seek, 1, 123)
171  
172          stream.seek(10000, 1)
173          assert_raises(OSError, stream.read, 12)

188      def test_all_data_read(self):
189          compressed_stream, compressed_data_len, data = self._get_data(1024)
190          stream = ZlibInputStream(compressed_stream, compressed_data_len)
191          assert_(not stream.all_data_read())
192          stream.seek(512)
193          assert_(not stream.all_data_read())
194          stream.seek(1024)
195          assert_(stream.all_data_read())

197      def test_all_data_read_overlap(self):
198          COMPRESSION_LEVEL = 6
199  
200          data = np.arange(33707000).astype(np.uint8).tobytes()
201          compressed_data = zlib.compress(data, COMPRESSION_LEVEL)
202          compressed_data_len = len(compressed_data)
203  
204          # check that part of the checksum overlaps
205          assert_(compressed_data_len == BLOCK_SIZE + 2)
206  
207          compressed_stream = BytesIO(compressed_data)
208          stream = ZlibInputStream(compressed_stream, compressed_data_len)
209          assert_(not stream.all_data_read())
210          stream.seek(len(data))
211          assert_(stream.all_data_read())
