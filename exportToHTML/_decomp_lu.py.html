20   def lu_factor(a, overwrite_a=False, check_finite=True):
21       """ 
22       Compute pivoted LU decomposition of a matrix. 
23    
24       The decomposition is:: 
25    
26           A = P L U 
27    
28       where P is a permutation matrix, L lower triangular with unit 
29       diagonal elements, and U upper triangular. 
30    
31       Parameters 
32       ---------- 
33       a : (M, N) array_like 
34           Matrix to decompose 
35       overwrite_a : bool, optional 
36           Whether to overwrite data in A (may increase performance) 
37       check_finite : bool, optional 
38           Whether to check that the input matrix contains only finite numbers. 
39           Disabling may give a performance gain, but may result in problems 
40           (crashes, non-termination) if the inputs do contain infinities or NaNs. 
41    
42       Returns 
43       ------- 
44       lu : (M, N) ndarray 
45           Matrix containing U in its upper triangle, and L in its lower triangle. 
46           The unit diagonal elements of L are not stored. 
47       piv : (K,) ndarray 
48           Pivot indices representing the permutation matrix P: 
49           row i of matrix was interchanged with row piv[i]. 
50           Of shape ``(K,)``, with ``K = min(M, N)``. 
51    
52       See Also 
53       -------- 
54       lu : gives lu factorization in more user-friendly format 
55       lu_solve : solve an equation system using the LU factorization of a matrix 
56    
57       Notes 
58       ----- 
59       This is a wrapper to the ``*GETRF`` routines from LAPACK. Unlike 
60       :func:`lu`, it outputs the L and U factors into a single array 
61       and returns pivot indices instead of a permutation matrix. 
62    
63       While the underlying ``*GETRF`` routines return 1-based pivot indices, the 
64       ``piv`` array returned by ``lu_factor`` contains 0-based indices. 
65    
66       Examples 
67       -------- 
68       >>> import numpy as np 
69       >>> from scipy.linalg import lu_factor 
70       >>> A = np.array([[2, 5, 8, 7], [5, 2, 2, 8], [7, 5, 6, 6], [5, 4, 4, 8]]) 
71       >>> lu, piv = lu_factor(A) 
72       >>> piv 
73       array([2, 2, 3, 3], dtype=int32) 
74    
75       Convert LAPACK's ``piv`` array to NumPy index and test the permutation 
76    
77       >>> def pivot_to_permutation(piv): 
78       ...     perm = np.arange(len(piv)) 
79       ...     for i in range(len(piv)): 
80       ...         perm[i], perm[piv[i]] = perm[piv[i]], perm[i] 
81       ...     return perm 
82       ... 
83       >>> p_inv = pivot_to_permutation(piv) 
84       >>> p_inv 
85       array([2, 0, 3, 1]) 
86       >>> L, U = np.tril(lu, k=-1) + np.eye(4), np.triu(lu) 
87       >>> np.allclose(A[p_inv] - L @ U, np.zeros((4, 4))) 
88       True 
89    
90       The P matrix in P L U is defined by the inverse permutation and 
91       can be recovered using argsort: 
92    
93       >>> p = np.argsort(p_inv) 
94       >>> p 
95       array([1, 3, 0, 2]) 
96       >>> np.allclose(A - L[p] @ U, np.zeros((4, 4))) 
97       True 
98    
99       or alternatively: 
100   
101      >>> P = np.eye(4)[p] 
102      >>> np.allclose(A - P @ L @ U, np.zeros((4, 4))) 
103      True 
104      """
105      if check_finite:
106          a1 = asarray_chkfinite(a)
107      else:
108          a1 = asarray(a)
109  
110      # accommodate empty arrays
111      if a1.size == 0:
112          lu = np.empty_like(a1)
113          piv = np.arange(0, dtype=np.int32)
114          return lu, piv
115  
116      overwrite_a = overwrite_a or (_datacopied(a1, a))
117  
118      getrf, = get_lapack_funcs(('getrf',), (a1,))
119      lu, piv, info = getrf(a1, overwrite_a=overwrite_a)
120      if info < 0:
121          raise ValueError('illegal value in %dth argument of '
122                           'internal getrf (lu_factor)' % -info)
123      if info > 0:
124          warn("Diagonal number %d is exactly zero. Singular matrix." % info,
125               LinAlgWarning, stacklevel=2)
126      return lu, piv

129  def lu_solve(lu_and_piv, b, trans=0, overwrite_b=False, check_finite=True):
130      """Solve an equation system, a x = b, given the LU factorization of a 
131   
132      Parameters 
133      ---------- 
134      (lu, piv) 
135          Factorization of the coefficient matrix a, as given by lu_factor. 
136          In particular piv are 0-indexed pivot indices. 
137      b : array 
138          Right-hand side 
139      trans : {0, 1, 2}, optional 
140          Type of system to solve: 
141   
142          =====  ========= 
143          trans  system 
144          =====  ========= 
145          0      a x   = b 
146          1      a^T x = b 
147          2      a^H x = b 
148          =====  ========= 
149      overwrite_b : bool, optional 
150          Whether to overwrite data in b (may increase performance) 
151      check_finite : bool, optional 
152          Whether to check that the input matrices contain only finite numbers. 
153          Disabling may give a performance gain, but may result in problems 
154          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
155   
156      Returns 
157      ------- 
158      x : array 
159          Solution to the system 
160   
161      See Also 
162      -------- 
163      lu_factor : LU factorize a matrix 
164   
165      Examples 
166      -------- 
167      >>> import numpy as np 
168      >>> from scipy.linalg import lu_factor, lu_solve 
169      >>> A = np.array([[2, 5, 8, 7], [5, 2, 2, 8], [7, 5, 6, 6], [5, 4, 4, 8]]) 
170      >>> b = np.array([1, 1, 1, 1]) 
171      >>> lu, piv = lu_factor(A) 
172      >>> x = lu_solve((lu, piv), b) 
173      >>> np.allclose(A @ x - b, np.zeros((4,))) 
174      True 
175   
176      """
177      (lu, piv) = lu_and_piv
178      if check_finite:
179          b1 = asarray_chkfinite(b)
180      else:
181          b1 = asarray(b)
182  
183      overwrite_b = overwrite_b or _datacopied(b1, b)
184  
185      if lu.shape[0] != b1.shape[0]:
186          raise ValueError(f"Shapes of lu {lu.shape} and b {b1.shape} are incompatible")
187  
188      # accommodate empty arrays
189      if b1.size == 0:
190          m = lu_solve((np.eye(2, dtype=lu.dtype), [0, 1]), np.ones(2, dtype=b.dtype))
191          return np.empty_like(b1, dtype=m.dtype)
192  
193      getrs, = get_lapack_funcs(('getrs',), (lu, b1))
194      x, info = getrs(lu, piv, b1, trans=trans, overwrite_b=overwrite_b)
195      if info == 0:
196          return x
197      raise ValueError('illegal value in %dth argument of internal gesv|posv'
198                       % -info)

201  def lu(a, permute_l=False, overwrite_a=False, check_finite=True,
202         p_indices=False):
203      """ 
204      Compute LU decomposition of a matrix with partial pivoting. 
205   
206      The decomposition satisfies:: 
207   
208          A = P @ L @ U 
209   
210      where ``P`` is a permutation matrix, ``L`` lower triangular with unit 
211      diagonal elements, and ``U`` upper triangular. If `permute_l` is set to 
212      ``True`` then ``L`` is returned already permuted and hence satisfying 
213      ``A = L @ U``. 
214   
215      Parameters 
216      ---------- 
217      a : (M, N) array_like 
218          Array to decompose 
219      permute_l : bool, optional 
220          Perform the multiplication P*L (Default: do not permute) 
221      overwrite_a : bool, optional 
222          Whether to overwrite data in a (may improve performance) 
223      check_finite : bool, optional 
224          Whether to check that the input matrix contains only finite numbers. 
225          Disabling may give a performance gain, but may result in problems 
226          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
227      p_indices : bool, optional 
228          If ``True`` the permutation information is returned as row indices. 
229          The default is ``False`` for backwards-compatibility reasons. 
230   
231      Returns 
232      ------- 
233      **(If `permute_l` is ``False``)** 
234   
235      p : (..., M, M) ndarray 
236          Permutation arrays or vectors depending on `p_indices` 
237      l : (..., M, K) ndarray 
238          Lower triangular or trapezoidal array with unit diagonal. 
239          ``K = min(M, N)`` 
240      u : (..., K, N) ndarray 
241          Upper triangular or trapezoidal array 
242   
243      **(If `permute_l` is ``True``)** 
244   
245      pl : (..., M, K) ndarray 
246          Permuted L matrix. 
247          ``K = min(M, N)`` 
248      u : (..., K, N) ndarray 
249          Upper triangular or trapezoidal array 
250   
251      Notes 
252      ----- 
253      Permutation matrices are costly since they are nothing but row reorder of 
254      ``L`` and hence indices are strongly recommended to be used instead if the 
255      permutation is required. The relation in the 2D case then becomes simply 
256      ``A = L[P, :] @ U``. In higher dimensions, it is better to use `permute_l` 
257      to avoid complicated indexing tricks. 
258   
259      In 2D case, if one has the indices however, for some reason, the 
260      permutation matrix is still needed then it can be constructed by 
261      ``np.eye(M)[P, :]``. 
262   
263      Examples 
264      -------- 
265   
266      >>> import numpy as np 
267      >>> from scipy.linalg import lu 
268      >>> A = np.array([[2, 5, 8, 7], [5, 2, 2, 8], [7, 5, 6, 6], [5, 4, 4, 8]]) 
269      >>> p, l, u = lu(A) 
270      >>> np.allclose(A, p @ l @ u) 
271      True 
272      >>> p  # Permutation matrix 
273      array([[0., 1., 0., 0.],  # Row index 1 
274             [0., 0., 0., 1.],  # Row index 3 
275             [1., 0., 0., 0.],  # Row index 0 
276             [0., 0., 1., 0.]]) # Row index 2 
277      >>> p, _, _ = lu(A, p_indices=True) 
278      >>> p 
279      array([1, 3, 0, 2], dtype=int32)  # as given by row indices above 
280      >>> np.allclose(A, l[p, :] @ u) 
281      True 
282   
283      We can also use nd-arrays, for example, a demonstration with 4D array: 
284   
285      >>> rng = np.random.default_rng() 
286      >>> A = rng.uniform(low=-4, high=4, size=[3, 2, 4, 8]) 
287      >>> p, l, u = lu(A) 
288      >>> p.shape, l.shape, u.shape 
289      ((3, 2, 4, 4), (3, 2, 4, 4), (3, 2, 4, 8)) 
290      >>> np.allclose(A, p @ l @ u) 
291      True 
292      >>> PL, U = lu(A, permute_l=True) 
293      >>> np.allclose(A, PL @ U) 
294      True 
295   
296      """
297      a1 = np.asarray_chkfinite(a) if check_finite else np.asarray(a)
298      if a1.ndim < 2:
299          raise ValueError('The input array must be at least two-dimensional.')
300  
301      # Also check if dtype is LAPACK compatible
302      if a1.dtype.char not in 'fdFD':
303          dtype_char = lapack_cast_dict[a1.dtype.char]
304          if not dtype_char:  # No casting possible
305              raise TypeError(f'The dtype {a1.dtype} cannot be cast '
306                              'to float(32, 64) or complex(64, 128).')
307  
308          a1 = a1.astype(dtype_char[0])  # makes a copy, free to scratch
309          overwrite_a = True
310  
311      *nd, m, n = a1.shape
312      k = min(m, n)
313      real_dchar = 'f' if a1.dtype.char in 'fF' else 'd'
314  
315      # Empty input
316      if min(*a1.shape) == 0:
317          if permute_l:
318              PL = np.empty(shape=[*nd, m, k], dtype=a1.dtype)
319              U = np.empty(shape=[*nd, k, n], dtype=a1.dtype)
320              return PL, U
321          else:
322              P = (np.empty([*nd, 0], dtype=np.int32) if p_indices else
323                   np.empty([*nd, 0, 0], dtype=real_dchar))
324              L = np.empty(shape=[*nd, m, k], dtype=a1.dtype)
325              U = np.empty(shape=[*nd, k, n], dtype=a1.dtype)
326              return P, L, U
327  
328      # Scalar case
329      if a1.shape[-2:] == (1, 1):
330          if permute_l:
331              return np.ones_like(a1), (a1 if overwrite_a else a1.copy())
332          else:
333              P = (np.zeros(shape=[*nd, m], dtype=int) if p_indices
334                   else np.ones_like(a1))
335              return P, np.ones_like(a1), (a1 if overwrite_a else a1.copy())
336  
337      # Then check overwrite permission
338      if not _datacopied(a1, a):  # "a"  still alive through "a1"
339          if not overwrite_a:
340              # Data belongs to "a" so make a copy
341              a1 = a1.copy(order='C')
342          #  else: Do nothing we'll use "a" if possible
343      # else:  a1 has its own data thus free to scratch
344  
345      # Then layout checks, might happen that overwrite is allowed but original
346      # array was read-only or non-contiguous.
347  
348      if not (a1.flags['C_CONTIGUOUS'] and a1.flags['WRITEABLE']):
349          a1 = a1.copy(order='C')
350  
351      if not nd:  # 2D array
352  
353          p = np.empty(m, dtype=np.int32)
354          u = np.zeros([k, k], dtype=a1.dtype)
355          lu_dispatcher(a1, u, p, permute_l)
356          P, L, U = (p, a1, u) if m > n else (p, u, a1)
357  
358      else:  # Stacked array
359  
360          # Prepare the contiguous data holders
361          P = np.empty([*nd, m], dtype=np.int32)  # perm vecs
362  
363          if m > n:  # Tall arrays, U will be created
364              U = np.zeros([*nd, k, k], dtype=a1.dtype)
365              for ind in product(*[range(x) for x in a1.shape[:-2]]):
366                  lu_dispatcher(a1[ind], U[ind], P[ind], permute_l)
367              L = a1
368  
369          else:  # Fat arrays, L will be created
370              L = np.zeros([*nd, k, k], dtype=a1.dtype)
371              for ind in product(*[range(x) for x in a1.shape[:-2]]):
372                  lu_dispatcher(a1[ind], L[ind], P[ind], permute_l)
373              U = a1
374  
375      # Convert permutation vecs to permutation arrays
376      # permute_l=False needed to enter here to avoid wasted efforts
377      if (not p_indices) and (not permute_l):
378          if nd:
379              Pa = np.zeros([*nd, m, m], dtype=real_dchar)
380              # An unreadable index hack - One-hot encoding for perm matrices
381              nd_ix = np.ix_(*([np.arange(x) for x in nd]+[np.arange(m)]))
382              Pa[(*nd_ix, P)] = 1
383              P = Pa
384          else:  # 2D case
385              Pa = np.zeros([m, m], dtype=real_dchar)
386              Pa[np.arange(m), P] = 1
387              P = Pa
388  
389      return (L, U) if permute_l else (P, L, U)
