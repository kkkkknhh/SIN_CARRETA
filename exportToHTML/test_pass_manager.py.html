36       def test_two_pass_managers(self) -> None:
37           """Make sure we can construct the PassManager twice and not share any 
38           state between them"""
39   
40           passes = [lambda x: 2 * x for _ in range(3)]
41           constraint = these_before_those_pass_constraint(passes[0], passes[1])
42           pm1 = PassManager()
43           for p in passes:
44               pm1.add_pass(p)
45           pm1.add_constraint(constraint)
46           output1 = pm1(1)
47           self.assertEqual(output1, 2**3)
48   
49           passes = [lambda x: 3 * x for _ in range(3)]

51           pm2 = PassManager()
