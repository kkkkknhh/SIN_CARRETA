24   def empty(shape, dtype=None, order='C'):
25       """Return a new matrix of given shape and type, without initializing entries. 
26    
27       Parameters 
28       ---------- 
29       shape : int or tuple of int 
30           Shape of the empty matrix. 
31       dtype : data-type, optional 
32           Desired output data-type. 
33       order : {'C', 'F'}, optional 
34           Whether to store multi-dimensional data in row-major 
35           (C-style) or column-major (Fortran-style) order in 
36           memory. 
37    
38       See Also 
39       -------- 
40       numpy.empty : Equivalent array function. 
41       matlib.zeros : Return a matrix of zeros. 
42       matlib.ones : Return a matrix of ones. 
43    
44       Notes 
45       ----- 
46       Unlike other matrix creation functions (e.g. `matlib.zeros`, 
47       `matlib.ones`), `matlib.empty` does not initialize the values of the 
48       matrix, and may therefore be marginally faster. However, the values 
49       stored in the newly allocated matrix are arbitrary. For reproducible 
50       behavior, be sure to set each element of the matrix before reading. 
51    
52       Examples 
53       -------- 
54       >>> import numpy.matlib 
55       >>> np.matlib.empty((2, 2))    # filled with random data 
56       matrix([[  6.76425276e-320,   9.79033856e-307], # random 
57               [  7.39337286e-309,   3.22135945e-309]]) 
58       >>> np.matlib.empty((2, 2), dtype=int) 
59       matrix([[ 6600475,        0], # random 
60               [ 6586976, 22740995]]) 
61    
62       """
63       return ndarray.__new__(matrix, shape, dtype, order=order)

65   def ones(shape, dtype=None, order='C'):
66       """ 
67       Matrix of ones. 
68    
69       Return a matrix of given shape and type, filled with ones. 
70    
71       Parameters 
72       ---------- 
73       shape : {sequence of ints, int} 
74           Shape of the matrix 
75       dtype : data-type, optional 
76           The desired data-type for the matrix, default is np.float64. 
77       order : {'C', 'F'}, optional 
78           Whether to store matrix in C- or Fortran-contiguous order, 
79           default is 'C'. 
80    
81       Returns 
82       ------- 
83       out : matrix 
84           Matrix of ones of given shape, dtype, and order. 
85    
86       See Also 
87       -------- 
88       ones : Array of ones. 
89       matlib.zeros : Zero matrix. 
90    
91       Notes 
92       ----- 
93       If `shape` has length one i.e. ``(N,)``, or is a scalar ``N``, 
94       `out` becomes a single row matrix of shape ``(1,N)``. 
95    
96       Examples 
97       -------- 
98       >>> np.matlib.ones((2,3)) 
99       matrix([[1.,  1.,  1.], 
100              [1.,  1.,  1.]]) 
101   
102      >>> np.matlib.ones(2) 
103      matrix([[1.,  1.]]) 
104   
105      """
106      a = ndarray.__new__(matrix, shape, dtype, order=order)
107      a.fill(1)
108      return a

154  def identity(n,dtype=None):
155      """ 
156      Returns the square identity matrix of given size. 
157   
158      Parameters 
159      ---------- 
160      n : int 
161          Size of the returned identity matrix. 
162      dtype : data-type, optional 
163          Data-type of the output. Defaults to ``float``. 
164   
165      Returns 
166      ------- 
167      out : matrix 
168          `n` x `n` matrix with its main diagonal set to one, 
169          and all other elements zero. 
170   
171      See Also 
172      -------- 
173      numpy.identity : Equivalent array function. 
174      matlib.eye : More general matrix identity function. 
175   
176      Examples 
177      -------- 
178      >>> import numpy.matlib 
179      >>> np.matlib.identity(3, dtype=int) 
180      matrix([[1, 0, 0], 
181              [0, 1, 0], 
182              [0, 0, 1]]) 
183   
184      """
185      a = array([1]+n*[0], dtype=dtype)
186      b = empty((n, n), dtype=dtype)
187      b.flat = a
188      return b

190  def eye(n,M=None, k=0, dtype=float, order='C'):
191      """ 
192      Return a matrix with ones on the diagonal and zeros elsewhere. 
193   
194      Parameters 
195      ---------- 
196      n : int 
197          Number of rows in the output. 
198      M : int, optional 
199          Number of columns in the output, defaults to `n`. 
200      k : int, optional 
201          Index of the diagonal: 0 refers to the main diagonal, 
202          a positive value refers to an upper diagonal, 
203          and a negative value to a lower diagonal. 
204      dtype : dtype, optional 
205          Data-type of the returned matrix. 
206      order : {'C', 'F'}, optional 
207          Whether the output should be stored in row-major (C-style) or 
208          column-major (Fortran-style) order in memory. 
209   
210      Returns 
211      ------- 
212      I : matrix 
213          A `n` x `M` matrix where all elements are equal to zero, 
214          except for the `k`-th diagonal, whose values are equal to one. 
215   
216      See Also 
217      -------- 
218      numpy.eye : Equivalent array function. 
219      identity : Square identity matrix. 
220   
221      Examples 
222      -------- 
223      >>> import numpy.matlib 
224      >>> np.matlib.eye(3, k=1, dtype=float) 
225      matrix([[0.,  1.,  0.], 
226              [0.,  0.,  1.], 
227              [0.,  0.,  0.]]) 
228   
229      """
230      return asmatrix(np.eye(n, M=M, k=k, dtype=dtype, order=order))

232  def rand(*args):
233      """ 
234      Return a matrix of random values with given shape. 
235   
236      Create a matrix of the given shape and propagate it with 
237      random samples from a uniform distribution over ``[0, 1)``. 
238   
239      Parameters 
240      ---------- 
241      \\*args : Arguments 
242          Shape of the output. 
243          If given as N integers, each integer specifies the size of one 
244          dimension. 
245          If given as a tuple, this tuple gives the complete shape. 
246   
247      Returns 
248      ------- 
249      out : ndarray 
250          The matrix of random values with shape given by `\\*args`. 
251   
252      See Also 
253      -------- 
254      randn, numpy.random.RandomState.rand 
255   
256      Examples 
257      -------- 
258      >>> np.random.seed(123) 
259      >>> import numpy.matlib 
260      >>> np.matlib.rand(2, 3) 
261      matrix([[0.69646919, 0.28613933, 0.22685145], 
262              [0.55131477, 0.71946897, 0.42310646]]) 
263      >>> np.matlib.rand((2, 3)) 
264      matrix([[0.9807642 , 0.68482974, 0.4809319 ], 
265              [0.39211752, 0.34317802, 0.72904971]]) 
266   
267      If the first argument is a tuple, other arguments are ignored: 
268   
269      >>> np.matlib.rand((2, 3), 4) 
270      matrix([[0.43857224, 0.0596779 , 0.39804426], 
271              [0.73799541, 0.18249173, 0.17545176]]) 
272   
273      """
274      if isinstance(args[0], tuple):
275          args = args[0]
276      return asmatrix(np.random.rand(*args))

278  def randn(*args):
279      """ 
280      Return a random matrix with data from the "standard normal" distribution. 
281   
282      `randn` generates a matrix filled with random floats sampled from a 
283      univariate "normal" (Gaussian) distribution of mean 0 and variance 1. 
284   
285      Parameters 
286      ---------- 
287      \\*args : Arguments 
288          Shape of the output. 
289          If given as N integers, each integer specifies the size of one 
290          dimension. If given as a tuple, this tuple gives the complete shape. 
291   
292      Returns 
293      ------- 
294      Z : matrix of floats 
295          A matrix of floating-point samples drawn from the standard normal 
296          distribution. 
297   
298      See Also 
299      -------- 
300      rand, numpy.random.RandomState.randn 
301   
302      Notes 
303      ----- 
304      For random samples from the normal distribution with mean ``mu`` and 
305      standard deviation ``sigma``, use:: 
306   
307          sigma * np.matlib.randn(...) + mu 
308   
309      Examples 
310      -------- 
311      >>> np.random.seed(123) 
312      >>> import numpy.matlib 
313      >>> np.matlib.randn(1) 
314      matrix([[-1.0856306]]) 
315      >>> np.matlib.randn(1, 2, 3) 
316      matrix([[ 0.99734545,  0.2829785 , -1.50629471], 
317              [-0.57860025,  1.65143654, -2.42667924]]) 
318   
319      Two-by-four matrix of samples from the normal distribution with 
320      mean 3 and standard deviation 2.5: 
321   
322      >>> 2.5 * np.matlib.randn((2, 4)) + 3 
323      matrix([[1.92771843, 6.16484065, 0.83314899, 1.30278462], 
324              [2.76322758, 6.72847407, 1.40274501, 1.8900451 ]]) 
325   
326      """
327      if isinstance(args[0], tuple):
328          args = args[0]
329      return asmatrix(np.random.randn(*args))

331  def repmat(a, m, n):
332      """ 
333      Repeat a 0-D to 2-D array or matrix MxN times. 
334   
335      Parameters 
336      ---------- 
337      a : array_like 
338          The array or matrix to be repeated. 
339      m, n : int 
340          The number of times `a` is repeated along the first and second axes. 
341   
342      Returns 
343      ------- 
344      out : ndarray 
345          The result of repeating `a`. 
346   
347      Examples 
348      -------- 
349      >>> import numpy.matlib 
350      >>> a0 = np.array(1) 
351      >>> np.matlib.repmat(a0, 2, 3) 
352      array([[1, 1, 1], 
353             [1, 1, 1]]) 
354   
355      >>> a1 = np.arange(4) 
356      >>> np.matlib.repmat(a1, 2, 2) 
357      array([[0, 1, 2, 3, 0, 1, 2, 3], 
358             [0, 1, 2, 3, 0, 1, 2, 3]]) 
359   
360      >>> a2 = np.asmatrix(np.arange(6).reshape(2, 3)) 
361      >>> np.matlib.repmat(a2, 2, 3) 
362      matrix([[0, 1, 2, 0, 1, 2, 0, 1, 2], 
363              [3, 4, 5, 3, 4, 5, 3, 4, 5], 
364              [0, 1, 2, 0, 1, 2, 0, 1, 2], 
365              [3, 4, 5, 3, 4, 5, 3, 4, 5]]) 
366   
367      """
368      a = asanyarray(a)
369      ndim = a.ndim
370      if ndim == 0:
371          origrows, origcols = (1, 1)
372      elif ndim == 1:
373          origrows, origcols = (1, a.shape[0])
374      else:
375          origrows, origcols = a.shape
376      rows = origrows * m
377      cols = origcols * n
378      c = a.reshape(1, a.size).repeat(m, 0).reshape(rows, origcols).repeat(n, 0)
379      return c.reshape(rows, cols)
