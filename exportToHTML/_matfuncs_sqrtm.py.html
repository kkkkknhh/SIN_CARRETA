28   def _sqrtm_triu(T, blocksize=64):
29       """ 
30       Matrix square root of an upper triangular matrix. 
31    
32       This is a helper function for `sqrtm` and `logm`. 
33    
34       Parameters 
35       ---------- 
36       T : (N, N) array_like upper triangular 
37           Matrix whose square root to evaluate 
38       blocksize : int, optional 
39           If the blocksize is not degenerate with respect to the 
40           size of the input array, then use a blocked algorithm. (Default: 64) 
41    
42       Returns 
43       ------- 
44       sqrtm : (N, N) ndarray 
45           Value of the sqrt function at `T` 
46    
47       References 
48       ---------- 
49       .. [1] Edvin Deadman, Nicholas J. Higham, Rui Ralha (2013) 
50              "Blocked Schur Algorithms for Computing the Matrix Square Root, 
51              Lecture Notes in Computer Science, 7782. pp. 171-182. 
52    
53       """
54       T_diag = np.diag(T)
55       keep_it_real = np.isrealobj(T) and np.min(T_diag, initial=0.) >= 0
56   
57       # Cast to complex as necessary + ensure double precision
58       if not keep_it_real:
59           T = np.asarray(T, dtype=np.complex128, order="C")
60           T_diag = np.asarray(T_diag, dtype=np.complex128)
61       else:
62           T = np.asarray(T, dtype=np.float64, order="C")
63           T_diag = np.asarray(T_diag, dtype=np.float64)
64   
65       R = np.diag(np.sqrt(T_diag))
66   
67       # Compute the number of blocks to use; use at least one block.
68       n, n = T.shape
69       nblocks = max(n // blocksize, 1)
70   
71       # Compute the smaller of the two sizes of blocks that
72       # we will actually use, and compute the number of large blocks.
73       bsmall, nlarge = divmod(n, nblocks)
74       blarge = bsmall + 1
75       nsmall = nblocks - nlarge
76       if nsmall * bsmall + nlarge * blarge != n:
77           raise Exception('internal inconsistency')
78   
79       # Define the index range covered by each block.
80       start_stop_pairs = []
81       start = 0
82       for count, size in ((nsmall, bsmall), (nlarge, blarge)):
83           for i in range(count):
84               start_stop_pairs.append((start, start + size))
85               start += size
86   
87       # Within-block interactions (Cythonized)
88       try:
89           within_block_loop(R, T, start_stop_pairs, nblocks)
90       except RuntimeError as e:
91           raise SqrtmError(*e.args) from e
92   
93       # Between-block interactions (Cython would give no significant speedup)
94       for j in range(nblocks):
95           jstart, jstop = start_stop_pairs[j]
96           for i in range(j-1, -1, -1):
97               istart, istop = start_stop_pairs[i]
98               S = T[istart:istop, jstart:jstop]
99               if j - i > 1:
100                  S = S - R[istart:istop, istop:jstart].dot(R[istop:jstart,
101                                                              jstart:jstop])
102  
103              # Invoke LAPACK.
104              # For more details, see the solve_sylvester implementation
105              # and the fortran dtrsyl and ztrsyl docs.
106              Rii = R[istart:istop, istart:istop]
107              Rjj = R[jstart:jstop, jstart:jstop]
108              if keep_it_real:
109                  x, scale, info = dtrsyl(Rii, Rjj, S)
110              else:
111                  x, scale, info = ztrsyl(Rii, Rjj, S)
112              R[istart:istop, jstart:jstop] = x * scale
113  
114      # Return the matrix square root.
115      return R

118  def sqrtm(A, disp=True, blocksize=64):
119      """ 
120      Matrix square root. 
121   
122      Parameters 
123      ---------- 
124      A : (N, N) array_like 
125          Matrix whose square root to evaluate 
126      disp : bool, optional 
127          Print warning if error in the result is estimated large 
128          instead of returning estimated error. (Default: True) 
129      blocksize : integer, optional 
130          If the blocksize is not degenerate with respect to the 
131          size of the input array, then use a blocked algorithm. (Default: 64) 
132   
133      Returns 
134      ------- 
135      sqrtm : (N, N) ndarray 
136          Value of the sqrt function at `A`. The dtype is float or complex. 
137          The precision (data size) is determined based on the precision of 
138          input `A`. 
139   
140      errest : float 
141          (if disp == False) 
142   
143          Frobenius norm of the estimated error, ||err||_F / ||A||_F 
144   
145      References 
146      ---------- 
147      .. [1] Edvin Deadman, Nicholas J. Higham, Rui Ralha (2013) 
148             "Blocked Schur Algorithms for Computing the Matrix Square Root, 
149             Lecture Notes in Computer Science, 7782. pp. 171-182. 
150   
151      Examples 
152      -------- 
153      >>> import numpy as np 
154      >>> from scipy.linalg import sqrtm 
155      >>> a = np.array([[1.0, 3.0], [1.0, 4.0]]) 
156      >>> r = sqrtm(a) 
157      >>> r 
158      array([[ 0.75592895,  1.13389342], 
159             [ 0.37796447,  1.88982237]]) 
160      >>> r.dot(r) 
161      array([[ 1.,  3.], 
162             [ 1.,  4.]]) 
163   
164      """
165      A = _asarray_validated(A, check_finite=True, as_inexact=True)
166      if len(A.shape) != 2:
167          raise ValueError("Non-matrix input to matrix function.")
168      if blocksize < 1:
169          raise ValueError("The blocksize should be at least 1.")
170      A, = _ensure_dtype_cdsz(A)
171      keep_it_real = np.isrealobj(A)
172      if keep_it_real:
173          T, Z = schur(A)
174          d0 = np.diagonal(T)
175          d1 = np.diagonal(T, -1)
176          eps = np.finfo(T.dtype).eps
177          needs_conversion = abs(d1) > eps * (abs(d0[1:]) + abs(d0[:-1]))
178          if needs_conversion.any():
179              T, Z = rsf2csf(T, Z)
180      else:
181          T, Z = schur(A, output='complex')
182      failflag = False
183      try:
184          R = _sqrtm_triu(T, blocksize=blocksize)
185          ZH = np.conjugate(Z).T
186          X = Z.dot(R).dot(ZH)
187          dtype = np.result_type(A.dtype, 1j if np.iscomplexobj(X) else 1)
188          X = X.astype(dtype, copy=False)
189      except SqrtmError:
190          failflag = True
191          X = np.empty_like(A)
192          X.fill(np.nan)
193  
194      if disp:
195          if failflag:
196              print("Failed to find a square root.")
197          return X
198      else:
199          try:
200              arg2 = norm(X.dot(X) - A, 'fro')**2 / norm(A, 'fro')
201          except ValueError:
202              # NaNs in matrix
203              arg2 = np.inf
204  
205          return X, arg2
