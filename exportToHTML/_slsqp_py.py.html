36   def approx_jacobian(x, func, epsilon, *args):
37       """ 
38       Approximate the Jacobian matrix of a callable function. 
39    
40       Parameters 
41       ---------- 
42       x : array_like 
43           The state vector at which to compute the Jacobian matrix. 
44       func : callable f(x,*args) 
45           The vector-valued function. 
46       epsilon : float 
47           The perturbation used to determine the partial derivatives. 
48       args : sequence 
49           Additional arguments passed to func. 
50    
51       Returns 
52       ------- 
53       An array of dimensions ``(lenf, lenx)`` where ``lenf`` is the length 
54       of the outputs of `func`, and ``lenx`` is the number of elements in 
55       `x`. 
56    
57       Notes 
58       ----- 
59       The approximation is done using forward differences. 
60    
61       """
62       # approx_derivative returns (m, n) == (lenf, lenx)
63       jac = approx_derivative(func, x, method='2-point', abs_step=epsilon,
64                               args=args)
65       # if func returns a scalar jac.shape will be (lenx,). Make sure
66       # it's at least a 2D array.
67       return np.atleast_2d(jac)

70   def fmin_slsqp(func, x0, eqcons=(), f_eqcons=None, ieqcons=(), f_ieqcons=None,
71                  bounds=(), fprime=None, fprime_eqcons=None,
72                  fprime_ieqcons=None, args=(), iter=100, acc=1.0E-6,
73                  iprint=1, disp=None, full_output=0, epsilon=_epsilon,
74                  callback=None):
75       """ 
76       Minimize a function using Sequential Least Squares Programming 
77    
78       Python interface function for the SLSQP Optimization subroutine 
79       originally implemented by Dieter Kraft. 
80    
81       Parameters 
82       ---------- 
83       func : callable f(x,*args) 
84           Objective function.  Must return a scalar. 
85       x0 : 1-D ndarray of float 
86           Initial guess for the independent variable(s). 
87       eqcons : list, optional 
88           A list of functions of length n such that 
89           eqcons[j](x,*args) == 0.0 in a successfully optimized 
90           problem. 
91       f_eqcons : callable f(x,*args), optional 
92           Returns a 1-D array in which each element must equal 0.0 in a 
93           successfully optimized problem. If f_eqcons is specified, 
94           eqcons is ignored. 
95       ieqcons : list, optional 
96           A list of functions of length n such that 
97           ieqcons[j](x,*args) >= 0.0 in a successfully optimized 
98           problem. 
99       f_ieqcons : callable f(x,*args), optional 
100          Returns a 1-D ndarray in which each element must be greater or 
101          equal to 0.0 in a successfully optimized problem. If 
102          f_ieqcons is specified, ieqcons is ignored. 
103      bounds : list, optional 
104          A list of tuples specifying the lower and upper bound 
105          for each independent variable [(xl0, xu0),(xl1, xu1),...] 
106          Infinite values will be interpreted as large floating values. 
107      fprime : callable ``f(x,*args)``, optional 
108          A function that evaluates the partial derivatives of func. 
109      fprime_eqcons : callable ``f(x,*args)``, optional 
110          A function of the form ``f(x, *args)`` that returns the m by n 
111          array of equality constraint normals. If not provided, 
112          the normals will be approximated. The array returned by 
113          fprime_eqcons should be sized as ( len(eqcons), len(x0) ). 
114      fprime_ieqcons : callable ``f(x,*args)``, optional 
115          A function of the form ``f(x, *args)`` that returns the m by n 
116          array of inequality constraint normals. If not provided, 
117          the normals will be approximated. The array returned by 
118          fprime_ieqcons should be sized as ( len(ieqcons), len(x0) ). 
119      args : sequence, optional 
120          Additional arguments passed to func and fprime. 
121      iter : int, optional 
122          The maximum number of iterations. 
123      acc : float, optional 
124          Requested accuracy. 
125      iprint : int, optional 
126          The verbosity of fmin_slsqp : 
127   
128          * iprint <= 0 : Silent operation 
129          * iprint == 1 : Print summary upon completion (default) 
130          * iprint >= 2 : Print status of each iterate and summary 
131      disp : int, optional 
132          Overrides the iprint interface (preferred). 
133      full_output : bool, optional 
134          If False, return only the minimizer of func (default). 
135          Otherwise, output final objective function and summary 
136          information. 
137      epsilon : float, optional 
138          The step size for finite-difference derivative estimates. 
139      callback : callable, optional 
140          Called after each iteration, as ``callback(x)``, where ``x`` is the 
141          current parameter vector. 
142   
143      Returns 
144      ------- 
145      out : ndarray of float 
146          The final minimizer of func. 
147      fx : ndarray of float, if full_output is true 
148          The final value of the objective function. 
149      its : int, if full_output is true 
150          The number of iterations. 
151      imode : int, if full_output is true 
152          The exit mode from the optimizer (see below). 
153      smode : string, if full_output is true 
154          Message describing the exit mode from the optimizer. 
155   
156      See also 
157      -------- 
158      minimize: Interface to minimization algorithms for multivariate 
159          functions. See the 'SLSQP' `method` in particular. 
160   
161      Notes 
162      ----- 
163      Exit modes are defined as follows: 
164   
165      - ``-1`` : Gradient evaluation required (g & a) 
166      - ``0`` : Optimization terminated successfully 
167      - ``1`` : Function evaluation required (f & c) 
168      - ``2`` : More equality constraints than independent variables 
169      - ``3`` : More than 3*n iterations in LSQ subproblem 
170      - ``4`` : Inequality constraints incompatible 
171      - ``5`` : Singular matrix E in LSQ subproblem 
172      - ``6`` : Singular matrix C in LSQ subproblem 
173      - ``7`` : Rank-deficient equality constraint subproblem HFTI 
174      - ``8`` : Positive directional derivative for linesearch 
175      - ``9`` : Iteration limit reached 
176   
177      Examples 
178      -------- 
179      Examples are given :ref:`in the tutorial <tutorial-sqlsp>`. 
180   
181      """
182      if disp is not None:
183          iprint = disp
184  
185      opts = {'maxiter': iter,
186              'ftol': acc,
187              'iprint': iprint,
188              'disp': iprint != 0,
189              'eps': epsilon,
190              'callback': callback}
191  
192      # Build the constraints as a tuple of dictionaries
193      cons = ()
194      # 1. constraints of the 1st kind (eqcons, ieqcons); no Jacobian; take
195      #    the same extra arguments as the objective function.
196      cons += tuple({'type': 'eq', 'fun': c, 'args': args} for c in eqcons)
197      cons += tuple({'type': 'ineq', 'fun': c, 'args': args} for c in ieqcons)
198      # 2. constraints of the 2nd kind (f_eqcons, f_ieqcons) and their Jacobian
199      #    (fprime_eqcons, fprime_ieqcons); also take the same extra arguments
200      #    as the objective function.
201      if f_eqcons:
202          cons += ({'type': 'eq', 'fun': f_eqcons, 'jac': fprime_eqcons,
203                    'args': args}, )
204      if f_ieqcons:
205          cons += ({'type': 'ineq', 'fun': f_ieqcons, 'jac': fprime_ieqcons,
206                    'args': args}, )
207  
208      res = _minimize_slsqp(func, x0, args, jac=fprime, bounds=bounds,
209                            constraints=cons, **opts)
210      if full_output:
211          return res['x'], res['fun'], res['nit'], res['status'], res['message']
212      else:
213          return res['x']

216  def _minimize_slsqp(func, x0, args=(), jac=None, bounds=None,
217                      constraints=(),
218                      maxiter=100, ftol=1.0E-6, iprint=1, disp=False,
219                      eps=_epsilon, callback=None, finite_diff_rel_step=None,
220                      **unknown_options):
221      """ 
222      Minimize a scalar function of one or more variables using Sequential 
223      Least Squares Programming (SLSQP). 
224   
225      Options 
226      ------- 
227      ftol : float 
228          Precision goal for the value of f in the stopping criterion. 
229      eps : float 
230          Step size used for numerical approximation of the Jacobian. 
231      disp : bool 
232          Set to True to print convergence messages. If False, 
233          `verbosity` is ignored and set to 0. 
234      maxiter : int 
235          Maximum number of iterations. 
236      finite_diff_rel_step : None or array_like, optional 
237          If ``jac in ['2-point', '3-point', 'cs']`` the relative step size to 
238          use for numerical approximation of `jac`. The absolute step 
239          size is computed as ``h = rel_step * sign(x) * max(1, abs(x))``, 
240          possibly adjusted to fit into the bounds. For ``method='3-point'`` 
241          the sign of `h` is ignored. If None (default) then step is selected 
242          automatically. 
243      """
244      _check_unknown_options(unknown_options)
245      iter = maxiter - 1
246      acc = ftol
247      epsilon = eps
248  
249      if not disp:
250          iprint = 0
251  
252      # Transform x0 into an array.
253      xp = array_namespace(x0)
254      x0 = xpx.atleast_nd(xp.asarray(x0), ndim=1, xp=xp)
255      dtype = xp.float64
256      if xp.isdtype(x0.dtype, "real floating"):
257          dtype = x0.dtype
258      x = xp.reshape(xp.astype(x0, dtype), -1)
259  
260      # SLSQP is sent 'old-style' bounds, 'new-style' bounds are required by
261      # ScalarFunction
262      if bounds is None or len(bounds) == 0:
263          new_bounds = (-np.inf, np.inf)
264      else:
265          new_bounds = old_bound_to_new(bounds)
266  
267      # clip the initial guess to bounds, otherwise ScalarFunction doesn't work
268      x = np.clip(x, new_bounds[0], new_bounds[1])
269  
270      # Constraints are triaged per type into a dictionary of tuples
271      if isinstance(constraints, dict):
272          constraints = (constraints, )
273  
274      cons = {'eq': (), 'ineq': ()}
275      for ic, con in enumerate(constraints):
276          # check type
277          try:
278              ctype = con['type'].lower()
279          except KeyError as e:
280              raise KeyError('Constraint %d has no type defined.' % ic) from e
281          except TypeError as e:
282              raise TypeError('Constraints must be defined using a '
283                              'dictionary.') from e
284          except AttributeError as e:
285              raise TypeError("Constraint's type must be a string.") from e
286          else:
287              if ctype not in ['eq', 'ineq']:
288                  raise ValueError(f"Unknown constraint type '{con['type']}'.")
289  
290          # check function
291          if 'fun' not in con:
292              raise ValueError('Constraint %d has no function defined.' % ic)
293  
294          # check Jacobian
295          cjac = con.get('jac')
296          if cjac is None:
297              # approximate Jacobian function. The factory function is needed
298              # to keep a reference to `fun`, see gh-4240.
299              def cjac_factory(fun):
300                  def cjac(x, *args):
301                      x = _check_clip_x(x, new_bounds)
302  
303                      if jac in ['2-point', '3-point', 'cs']:
304                          return approx_derivative(fun, x, method=jac, args=args,
305                                                   rel_step=finite_diff_rel_step,
306                                                   bounds=new_bounds)
307                      else:
308                          return approx_derivative(fun, x, method='2-point',
309                                                   abs_step=epsilon, args=args,
310                                                   bounds=new_bounds)
311  
312                  return cjac
313              cjac = cjac_factory(con['fun'])
314  
315          # update constraints' dictionary
316          cons[ctype] += ({'fun': con['fun'],
317                           'jac': cjac,
318                           'args': con.get('args', ())}, )
319  
320      exit_modes = {-1: "Gradient evaluation required (g & a)",
321                     0: "Optimization terminated successfully",
322                     1: "Function evaluation required (f & c)",
323                     2: "More equality constraints than independent variables",
324                     3: "More than 3*n iterations in LSQ subproblem",
325                     4: "Inequality constraints incompatible",
326                     5: "Singular matrix E in LSQ subproblem",
327                     6: "Singular matrix C in LSQ subproblem",
328                     7: "Rank-deficient equality constraint subproblem HFTI",
329                     8: "Positive directional derivative for linesearch",
330                     9: "Iteration limit reached"}
331  
332      # Set the parameters that SLSQP will need
333      # meq, mieq: number of equality and inequality constraints
334      meq = sum(map(len, [atleast_1d(c['fun'](x, *c['args']))
335                for c in cons['eq']]))
336      mieq = sum(map(len, [atleast_1d(c['fun'](x, *c['args']))
337                 for c in cons['ineq']]))
338      # m = The total number of constraints
339      m = meq + mieq
340      # la = The number of constraints, or 1 if there are no constraints
341      la = array([1, m]).max()
342      # n = The number of independent variables
343      n = len(x)
344  
345      # Define the workspaces for SLSQP
346      n1 = n + 1
347      mineq = m - meq + n1 + n1
348      len_w = (3*n1+m)*(n1+1)+(n1-meq+1)*(mineq+2) + 2*mineq+(n1+mineq)*(n1-meq) \
349              + 2*meq + n1 + ((n+1)*n)//2 + 2*m + 3*n + 3*n1 + 1
350      len_jw = mineq
351      w = zeros(len_w)
352      jw = zeros(len_jw)
353  
354      # Decompose bounds into xl and xu
355      if bounds is None or len(bounds) == 0:
356          xl = np.empty(n, dtype=float)
357          xu = np.empty(n, dtype=float)
358          xl.fill(np.nan)
359          xu.fill(np.nan)
360      else:
361          bnds = array([(_arr_to_scalar(l), _arr_to_scalar(u))
362                        for (l, u) in bounds], float)
363          if bnds.shape[0] != n:
364              raise IndexError('SLSQP Error: the length of bounds is not '
365                               'compatible with that of x0.')
366  
367          with np.errstate(invalid='ignore'):
368              bnderr = bnds[:, 0] > bnds[:, 1]
369  
370          if bnderr.any():
371              raise ValueError("SLSQP Error: lb > ub in bounds "
372                               f"{', '.join(str(b) for b in bnderr)}.")
373          xl, xu = bnds[:, 0], bnds[:, 1]
374  
375          # Mark infinite bounds with nans; the Fortran code understands this
376          infbnd = ~isfinite(bnds)
377          xl[infbnd[:, 0]] = np.nan
378          xu[infbnd[:, 1]] = np.nan
379  
380      # ScalarFunction provides function and gradient evaluation
381      sf = _prepare_scalar_function(func, x, jac=jac, args=args, epsilon=eps,
382                                    finite_diff_rel_step=finite_diff_rel_step,
383                                    bounds=new_bounds)
384      # gh11403 SLSQP sometimes exceeds bounds by 1 or 2 ULP, make sure this
385      # doesn't get sent to the func/grad evaluator.
386      wrapped_fun = _clip_x_for_func(sf.fun, new_bounds)
387      wrapped_grad = _clip_x_for_func(sf.grad, new_bounds)
388  
389      # Initialize the iteration counter and the mode value
390      mode = array(0, int)
391      acc = array(acc, float)
392      majiter = array(iter, int)
393      majiter_prev = 0
394  
395      # Initialize internal SLSQP state variables
396      alpha = array(0, float)
397      f0 = array(0, float)
398      gs = array(0, float)
399      h1 = array(0, float)
400      h2 = array(0, float)
401      h3 = array(0, float)
402      h4 = array(0, float)
403      t = array(0, float)
404      t0 = array(0, float)
405      tol = array(0, float)
406      iexact = array(0, int)
407      incons = array(0, int)
408      ireset = array(0, int)
409      itermx = array(0, int)
410      line = array(0, int)
411      n1 = array(0, int)
412      n2 = array(0, int)
413      n3 = array(0, int)
414  
415      # Print the header if iprint >= 2
416      if iprint >= 2:
417          print("%5s %5s %16s %16s" % ("NIT", "FC", "OBJFUN", "GNORM"))
418  
419      # mode is zero on entry, so call objective, constraints and gradients
420      # there should be no func evaluations here because it's cached from
421      # ScalarFunction
422      fx = wrapped_fun(x)
423      g = append(wrapped_grad(x), 0.0)
424      c = _eval_constraint(x, cons)
425      a = _eval_con_normals(x, cons, la, n, m, meq, mieq)
426  
427      while 1:
428          # Call SLSQP
429          slsqp(m, meq, x, xl, xu, fx, c, g, a, acc, majiter, mode, w, jw,
430                alpha, f0, gs, h1, h2, h3, h4, t, t0, tol,
431                iexact, incons, ireset, itermx, line,
432                n1, n2, n3)
433  
434          if mode == 1:  # objective and constraint evaluation required
435              fx = wrapped_fun(x)
436              c = _eval_constraint(x, cons)
437  
438          if mode == -1:  # gradient evaluation required
439              g = append(wrapped_grad(x), 0.0)
440              a = _eval_con_normals(x, cons, la, n, m, meq, mieq)
441  
442          if majiter > majiter_prev:
443              # call callback if major iteration has incremented
444              if callback is not None:
445                  callback(np.copy(x))
446  
447              # Print the status of the current iterate if iprint > 2
448              if iprint >= 2:
449                  print("%5i %5i % 16.6E % 16.6E" % (majiter, sf.nfev,
450                                                     fx, linalg.norm(g)))
451  
452          # If exit mode is not -1 or 1, slsqp has completed
453          if abs(mode) != 1:
454              break
455  
456          majiter_prev = int(majiter)
457  
458      # Optimization loop complete. Print status if requested
459      if iprint >= 1:
460          print(exit_modes[int(mode)] + "    (Exit mode " + str(mode) + ')')
461          print("            Current function value:", fx)
462          print("            Iterations:", majiter)
463          print("            Function evaluations:", sf.nfev)
464          print("            Gradient evaluations:", sf.ngev)
465  
466      return OptimizeResult(x=x, fun=fx, jac=g[:-1], nit=int(majiter),
467                            nfev=sf.nfev, njev=sf.ngev, status=int(mode),
468                            message=exit_modes[int(mode)], success=(mode == 0))
469

471  def _eval_constraint(x, cons):
472      # Compute constraints
473      if cons['eq']:
474          c_eq = concatenate([atleast_1d(con['fun'](x, *con['args']))
475                              for con in cons['eq']])
476      else:
477          c_eq = zeros(0)
478  
479      if cons['ineq']:
480          c_ieq = concatenate([atleast_1d(con['fun'](x, *con['args']))
481                               for con in cons['ineq']])
482      else:
483          c_ieq = zeros(0)
484  
485      # Now combine c_eq and c_ieq into a single matrix
486      c = concatenate((c_eq, c_ieq))
487      return c
488

490  def _eval_con_normals(x, cons, la, n, m, meq, mieq):
491      # Compute the normals of the constraints
492      if cons['eq']:
493          a_eq = vstack([con['jac'](x, *con['args'])
494                         for con in cons['eq']])
495      else:  # no equality constraint
496          a_eq = zeros((meq, n))
497  
498      if cons['ineq']:
499          a_ieq = vstack([con['jac'](x, *con['args'])
500                          for con in cons['ineq']])
501      else:  # no inequality constraint
502          a_ieq = zeros((mieq, n))
503  
504      # Now combine a_eq and a_ieq into a single a matrix
505      if m == 0:  # no constraints
506          a = zeros((la, n))
507      else:
508          a = vstack((a_eq, a_ieq))
509      a = concatenate((a, zeros([la, 1])), 1)
510  
511      return a
