310  def _raise_bad_format(format_tag):
311      try:
312          format_name = WAVE_FORMAT(format_tag).name
313      except ValueError:
314          format_name = f'{format_tag:#06x}'
315      raise ValueError(f"Unknown wave file format: {format_name}. Supported "
316                       "formats: " +
317                       ', '.join(x.name for x in KNOWN_WAVE_FORMATS))

320  def _read_fmt_chunk(fid, is_big_endian):
321      """ 
322      Returns 
323      ------- 
324      size : int 
325          size of format subchunk in bytes (minus 8 for "fmt " and itself) 
326      format_tag : int 
327          PCM, float, or compressed format 
328      channels : int 
329          number of channels 
330      fs : int 
331          sampling frequency in samples per second 
332      bytes_per_second : int 
333          overall byte rate for the file 
334      block_align : int 
335          bytes per sample, including all channels 
336      bit_depth : int 
337          bits per sample 
338   
339      Notes 
340      ----- 
341      Assumes file pointer is immediately after the 'fmt ' id 
342      """
343      if is_big_endian:
344          fmt = '>'
345      else:
346          fmt = '<'
347  
348      size = struct.unpack(fmt+'I', fid.read(4))[0]
349  
350      if size < 16:
351          raise ValueError("Binary structure of wave file is not compliant")
352  
353      res = struct.unpack(fmt+'HHIIHH', fid.read(16))
354      bytes_read = 16
355  
356      format_tag, channels, fs, bytes_per_second, block_align, bit_depth = res
357  
358      if format_tag == WAVE_FORMAT.EXTENSIBLE and size >= (16+2):
359          ext_chunk_size = struct.unpack(fmt+'H', fid.read(2))[0]
360          bytes_read += 2
361          if ext_chunk_size >= 22:
362              extensible_chunk_data = fid.read(22)
363              bytes_read += 22
364              raw_guid = extensible_chunk_data[2+4:2+4+16]
365              # GUID template {XXXXXXXX-0000-0010-8000-00AA00389B71} (RFC-2361)
366              # MS GUID byte order: first three groups are native byte order,
367              # rest is Big Endian
368              if is_big_endian:
369                  tail = b'\x00\x00\x00\x10\x80\x00\x00\xAA\x00\x38\x9B\x71'
370              else:
371                  tail = b'\x00\x00\x10\x00\x80\x00\x00\xAA\x00\x38\x9B\x71'
372              if raw_guid.endswith(tail):
373                  format_tag = struct.unpack(fmt+'I', raw_guid[:4])[0]
374          else:
375              raise ValueError("Binary structure of wave file is not compliant")
376  
377      if format_tag not in KNOWN_WAVE_FORMATS:
378          _raise_bad_format(format_tag)
379  
380      # move file pointer to next chunk
381      if size > bytes_read:
382          fid.read(size - bytes_read)
383  
384      # fmt should always be 16, 18 or 40, but handle it just in case
385      _handle_pad_byte(fid, size)
386  
387      if format_tag == WAVE_FORMAT.PCM:
388          if bytes_per_second != fs * block_align:
389              raise ValueError("WAV header is invalid: nAvgBytesPerSec must"
390                               " equal product of nSamplesPerSec and"
391                               " nBlockAlign, but file has nSamplesPerSec ="
392                               f" {fs}, nBlockAlign = {block_align}, and"
393                               f" nAvgBytesPerSec = {bytes_per_second}")
394  
395      return (size, format_tag, channels, fs, bytes_per_second, block_align,
396              bit_depth)

399  def _read_data_chunk(fid, format_tag, channels, bit_depth, is_big_endian, is_rf64,
400                       block_align, mmap=False):
401      """ 
402      Notes 
403      ----- 
404      Assumes file pointer is immediately after the 'data' id 
405   
406      It's possible to not use all available bits in a container, or to store 
407      samples in a container bigger than necessary, so bytes_per_sample uses 
408      the actual reported container size (nBlockAlign / nChannels).  Real-world 
409      examples: 
410   
411      Adobe Audition's "24-bit packed int (type 1, 20-bit)" 
412   
413          nChannels = 2, nBlockAlign = 6, wBitsPerSample = 20 
414   
415      http://www-mmsp.ece.mcgill.ca/Documents/AudioFormats/WAVE/Samples/AFsp/M1F1-int12-AFsp.wav 
416      is: 
417   
418          nChannels = 2, nBlockAlign = 4, wBitsPerSample = 12 
419   
420      http://www-mmsp.ece.mcgill.ca/Documents/AudioFormats/WAVE/Docs/multichaudP.pdf 
421      gives an example of: 
422   
423          nChannels = 2, nBlockAlign = 8, wBitsPerSample = 20 
424      """
425      if is_big_endian:
426          fmt = '>'
427      else:
428          fmt = '<'
429  
430      # Size of the data subchunk in bytes
431      if not is_rf64:
432          size = struct.unpack(fmt+'I', fid.read(4))[0]
433      else:
434          pos = fid.tell()
435          # chunk size is stored in global file header for RF64
436          fid.seek(28)
437          size = struct.unpack('<Q', fid.read(8))[0]
438          fid.seek(pos)
439          # skip data chunk size as it is 0xFFFFFFF
440          fid.read(4)
441  
442      # Number of bytes per sample (sample container size)
443      bytes_per_sample = block_align // channels
444      n_samples = size // bytes_per_sample
445  
446      if format_tag == WAVE_FORMAT.PCM:
447          if 1 <= bit_depth <= 8:
448              dtype = 'u1'  # WAV of 8-bit integer or less are unsigned
449          elif bytes_per_sample in {3, 5, 6, 7}:
450              # No compatible dtype.  Load as raw bytes for reshaping later.
451              dtype = 'V1'
452          elif bit_depth <= 64:
453              # Remaining bit depths can map directly to signed numpy dtypes
454              dtype = f'{fmt}i{bytes_per_sample}'
455          else:
456              raise ValueError("Unsupported bit depth: the WAV file "
457                               f"has {bit_depth}-bit integer data.")
458      elif format_tag == WAVE_FORMAT.IEEE_FLOAT:
459          if bit_depth in {32, 64}:
460              dtype = f'{fmt}f{bytes_per_sample}'
461          else:
462              raise ValueError("Unsupported bit depth: the WAV file "
463                               f"has {bit_depth}-bit floating-point data.")
464      else:
465          _raise_bad_format(format_tag)
466  
467      start = fid.tell()
468      if not mmap:
469          try:
470              count = size if dtype == 'V1' else n_samples
471              data = np.fromfile(fid, dtype=dtype, count=count)
472          except io.UnsupportedOperation:  # not a C-like file
473              fid.seek(start, 0)  # just in case it seeked, though it shouldn't
474              data = np.frombuffer(fid.read(size), dtype=dtype)
475  
476          if dtype == 'V1':
477              # Rearrange raw bytes into smallest compatible numpy dtype
478              dt = f'{fmt}i4' if bytes_per_sample == 3 else f'{fmt}i8'
479              a = np.zeros((len(data) // bytes_per_sample, np.dtype(dt).itemsize),
480                              dtype='V1')
481              if is_big_endian:
482                  a[:, :bytes_per_sample] = data.reshape((-1, bytes_per_sample))
483              else:
484                  a[:, -bytes_per_sample:] = data.reshape((-1, bytes_per_sample))
485              data = a.view(dt).reshape(a.shape[:-1])
486      else:
487          if bytes_per_sample in {1, 2, 4, 8}:
488              start = fid.tell()
489              data = np.memmap(fid, dtype=dtype, mode='c', offset=start,
490                                  shape=(n_samples,))
491              fid.seek(start + size)
492          else:
493              raise ValueError("mmap=True not compatible with "
494                               f"{bytes_per_sample}-byte container size.")
495  
496      _handle_pad_byte(fid, size)
497  
498      if channels > 1:
499          data = data.reshape(-1, channels)
500      return data

503  def _skip_unknown_chunk(fid, is_big_endian):
504      if is_big_endian:
505          fmt = '>I'
506      else:
507          fmt = '<I'
508  
509      data = fid.read(4)
510      # call unpack() and seek() only if we have really read data from file
511      # otherwise empty read at the end of the file would trigger
512      # unnecessary exception at unpack() call
513      # in case data equals somehow to 0, there is no need for seek() anyway
514      if data:
515          size = struct.unpack(fmt, data)[0]
516          fid.seek(size, 1)
517          _handle_pad_byte(fid, size)

520  def _read_riff_chunk(fid):
521      str1 = fid.read(4)  # File signature
522      if str1 == b'RIFF':
523          is_rf64 = False
524          is_big_endian = False
525          fmt = '<I'
526      elif str1 == b'RIFX':
527          is_rf64 = False
528          is_big_endian = True
529          fmt = '>I'
530      elif str1 == b'RF64':
531          is_rf64 = True
532          is_big_endian = False
533          fmt = '<Q'
534      else:
535          # There are also .wav files with "FFIR" or "XFIR" signatures?
536          raise ValueError(f"File format {repr(str1)} not understood. Only "
537                           "'RIFF', 'RIFX', and 'RF64' supported.")
538      # Size of entire file
539      if not is_rf64:
540          file_size = struct.unpack(fmt, fid.read(4))[0] + 8
541          str2 = fid.read(4)
542      else:
543          # Skip 0xFFFFFFFF (-1) bytes
544          fid.read(4)
545          str2 = fid.read(4)
546          str3 = fid.read(4)
547          if str3 != b'ds64':
548              raise ValueError("Invalid RF64 file: ds64 chunk not found.")
549          ds64_size = struct.unpack("<I", fid.read(4))[0]
550          file_size = struct.unpack(fmt, fid.read(8))[0] + 8
551          # Ignore additional attributes of ds64 chunk like sample count, tables, etc.
552          # and just skip to the next chunk
553          fid.seek(ds64_size - 8, 1)
554  
555      if str2 != b'WAVE':
556          raise ValueError(f"Not a WAV file. RIFF form type is {repr(str2)}.")
557  
558      return file_size, is_big_endian, is_rf64

561  def _handle_pad_byte(fid, size):
562      # "If the chunk size is an odd number of bytes, a pad byte with value zero
563      # is written after ckData." So we need to seek past this after each chunk.
564      if size % 2:
565          fid.seek(1, 1)

568  def read(filename, mmap=False):
569      """ 
570      Open a WAV file. 
571   
572      Return the sample rate (in samples/sec) and data from an LPCM WAV file. 
573   
574      Parameters 
575      ---------- 
576      filename : string or open file handle 
577          Input WAV file. 
578      mmap : bool, optional 
579          Whether to read data as memory-mapped (default: False).  Not compatible 
580          with some bit depths; see Notes.  Only to be used on real files. 
581   
582          .. versionadded:: 0.12.0 
583   
584      Returns 
585      ------- 
586      rate : int 
587          Sample rate of WAV file. 
588      data : numpy array 
589          Data read from WAV file. Data-type is determined from the file; 
590          see Notes.  Data is 1-D for 1-channel WAV, or 2-D of shape 
591          (Nsamples, Nchannels) otherwise. If a file-like input without a 
592          C-like file descriptor (e.g., :class:`python:io.BytesIO`) is 
593          passed, this will not be writeable. 
594   
595      Notes 
596      ----- 
597      Common data types: [1]_ 
598   
599      =====================  ===========  ===========  ============= 
600           WAV format            Min          Max       NumPy dtype 
601      =====================  ===========  ===========  ============= 
602      32-bit floating-point  -1.0         +1.0         float32 
603      32-bit integer PCM     -2147483648  +2147483647  int32 
604      24-bit integer PCM     -2147483648  +2147483392  int32 
605      16-bit integer PCM     -32768       +32767       int16 
606      8-bit integer PCM      0            255          uint8 
607      =====================  ===========  ===========  ============= 
608   
609      WAV files can specify arbitrary bit depth, and this function supports 
610      reading any integer PCM depth from 1 to 64 bits.  Data is returned in the 
611      smallest compatible numpy int type, in left-justified format.  8-bit and 
612      lower is unsigned, while 9-bit and higher is signed. 
613   
614      For example, 24-bit data will be stored as int32, with the MSB of the 
615      24-bit data stored at the MSB of the int32, and typically the least 
616      significant byte is 0x00.  (However, if a file actually contains data past 
617      its specified bit depth, those bits will be read and output, too. [2]_) 
618   
619      This bit justification and sign matches WAV's native internal format, which 
620      allows memory mapping of WAV files that use 1, 2, 4, or 8 bytes per sample 
621      (so 24-bit files cannot be memory-mapped, but 32-bit can). 
622   
623      IEEE float PCM in 32- or 64-bit format is supported, with or without mmap. 
624      Values exceeding [-1, +1] are not clipped. 
625   
626      Non-linear PCM (mu-law, A-law) is not supported. 
627   
628      References 
629      ---------- 
630      .. [1] IBM Corporation and Microsoft Corporation, "Multimedia Programming 
631         Interface and Data Specifications 1.0", section "Data Format of the 
632         Samples", August 1991 
633         http://www.tactilemedia.com/info/MCI_Control_Info.html 
634      .. [2] Adobe Systems Incorporated, "Adobe Audition 3 User Guide", section 
635         "Audio file formats: 24-bit Packed Int (type 1, 20-bit)", 2007 
636   
637      Examples 
638      -------- 
639      >>> from os.path import dirname, join as pjoin 
640      >>> from scipy.io import wavfile 
641      >>> import scipy.io 
642   
643      Get the filename for an example .wav file from the tests/data directory. 
644   
645      >>> data_dir = pjoin(dirname(scipy.io.__file__), 'tests', 'data') 
646      >>> wav_fname = pjoin(data_dir, 'test-44100Hz-2ch-32bit-float-be.wav') 
647   
648      Load the .wav file contents. 
649   
650      >>> samplerate, data = wavfile.read(wav_fname) 
651      >>> print(f"number of channels = {data.shape[1]}") 
652      number of channels = 2 
653      >>> length = data.shape[0] / samplerate 
654      >>> print(f"length = {length}s") 
655      length = 0.01s 
656   
657      Plot the waveform. 
658   
659      >>> import matplotlib.pyplot as plt 
660      >>> import numpy as np 
661      >>> time = np.linspace(0., length, data.shape[0]) 
662      >>> plt.plot(time, data[:, 0], label="Left channel") 
663      >>> plt.plot(time, data[:, 1], label="Right channel") 
664      >>> plt.legend() 
665      >>> plt.xlabel("Time [s]") 
666      >>> plt.ylabel("Amplitude") 
667      >>> plt.show() 
668   
669      """
670      if hasattr(filename, 'read'):
671          fid = filename
672          mmap = False
673      else:
674          fid = open(filename, 'rb')
675  
676      try:
677          file_size, is_big_endian, is_rf64 = _read_riff_chunk(fid)
678          fmt_chunk_received = False
679          data_chunk_received = False
680          while fid.tell() < file_size:
681              # read the next chunk
682              chunk_id = fid.read(4)
683  
684              if not chunk_id:
685                  if data_chunk_received:
686                      # End of file but data successfully read
687                      warnings.warn(
688                          f"Reached EOF prematurely; finished at {fid.tell():d} bytes, "
689                          f"expected {file_size:d} bytes from header.",
690                          WavFileWarning, stacklevel=2)
691                      break
692                  else:
693                      raise ValueError("Unexpected end of file.")
694              elif len(chunk_id) < 4:
695                  msg = f"Incomplete chunk ID: {repr(chunk_id)}"
696                  # If we have the data, ignore the broken chunk
697                  if fmt_chunk_received and data_chunk_received:
698                      warnings.warn(msg + ", ignoring it.", WavFileWarning,
699                                    stacklevel=2)
700                  else:
701                      raise ValueError(msg)
702  
703              if chunk_id == b'fmt ':
704                  fmt_chunk_received = True
705                  fmt_chunk = _read_fmt_chunk(fid, is_big_endian)
706                  format_tag, channels, fs = fmt_chunk[1:4]
707                  bit_depth = fmt_chunk[6]
708                  block_align = fmt_chunk[5]
709              elif chunk_id == b'fact':
710                  _skip_unknown_chunk(fid, is_big_endian)
711              elif chunk_id == b'data':
712                  data_chunk_received = True
713                  if not fmt_chunk_received:
714                      raise ValueError("No fmt chunk before data")
715                  data = _read_data_chunk(fid, format_tag, channels, bit_depth,
716                                          is_big_endian, is_rf64, block_align, mmap)
717              elif chunk_id == b'LIST':
718                  # Someday this could be handled properly but for now skip it
719                  _skip_unknown_chunk(fid, is_big_endian)
720              elif chunk_id in {b'JUNK', b'Fake'}:
721                  # Skip alignment chunks without warning
722                  _skip_unknown_chunk(fid, is_big_endian)
723              else:
724                  warnings.warn("Chunk (non-data) not understood, skipping it.",
725                                WavFileWarning, stacklevel=2)
726                  _skip_unknown_chunk(fid, is_big_endian)
727      finally:
728          if not hasattr(filename, 'read'):
729              fid.close()
730          else:
731              fid.seek(0)
732  
733      return fs, data

736  def write(filename, rate, data):
737      """ 
738      Write a NumPy array as a WAV file. 
739   
740      Parameters 
741      ---------- 
742      filename : string or open file handle 
743          Output wav file. 
744      rate : int 
745          The sample rate (in samples/sec). 
746      data : ndarray 
747          A 1-D or 2-D NumPy array of either integer or float data-type. 
748   
749      Notes 
750      ----- 
751      * Writes a simple uncompressed WAV file. 
752      * To write multiple-channels, use a 2-D array of shape 
753        (Nsamples, Nchannels). 
754      * The bits-per-sample and PCM/float will be determined by the data-type. 
755   
756      Common data types: [1]_ 
757   
758      =====================  ===========  ===========  ============= 
759           WAV format            Min          Max       NumPy dtype 
760      =====================  ===========  ===========  ============= 
761      32-bit floating-point  -1.0         +1.0         float32 
762      32-bit PCM             -2147483648  +2147483647  int32 
763      16-bit PCM             -32768       +32767       int16 
764      8-bit PCM              0            255          uint8 
765      =====================  ===========  ===========  ============= 
766   
767      Note that 8-bit PCM is unsigned. 
768   
769      References 
770      ---------- 
771      .. [1] IBM Corporation and Microsoft Corporation, "Multimedia Programming 
772         Interface and Data Specifications 1.0", section "Data Format of the 
773         Samples", August 1991 
774         http://www.tactilemedia.com/info/MCI_Control_Info.html 
775   
776      Examples 
777      -------- 
778      Create a 100Hz sine wave, sampled at 44100Hz. 
779      Write to 16-bit PCM, Mono. 
780   
781      >>> from scipy.io.wavfile import write 
782      >>> import numpy as np 
783      >>> samplerate = 44100; fs = 100 
784      >>> t = np.linspace(0., 1., samplerate) 
785      >>> amplitude = np.iinfo(np.int16).max 
786      >>> data = amplitude * np.sin(2. * np.pi * fs * t) 
787      >>> write("example.wav", samplerate, data.astype(np.int16)) 
788   
789      """
790      if hasattr(filename, 'write'):
791          fid = filename
792      else:
793          fid = open(filename, 'wb')
794  
795      fs = rate
796  
797      try:
798          dkind = data.dtype.kind
799          allowed_dtypes = ['float32', 'float64',
800                            'uint8', 'int16', 'int32', 'int64']
801          if data.dtype.name not in allowed_dtypes:
802              raise ValueError(f"Unsupported data type '{data.dtype}'")
803  
804          header_data = b''
805  
806          header_data += b'RIFF'
807          header_data += b'\x00\x00\x00\x00'
808          header_data += b'WAVE'
809  
810          # fmt chunk
811          header_data += b'fmt '
812          if dkind == 'f':
813              format_tag = WAVE_FORMAT.IEEE_FLOAT
814          else:
815              format_tag = WAVE_FORMAT.PCM
816          if data.ndim == 1:
817              channels = 1
818          else:
819              channels = data.shape[1]
820          bit_depth = data.dtype.itemsize * 8
821          bytes_per_second = fs*(bit_depth // 8)*channels
822          block_align = channels * (bit_depth // 8)
823  
824          fmt_chunk_data = struct.pack('<HHIIHH', format_tag, channels, fs,
825                                       bytes_per_second, block_align, bit_depth)
826          if not (dkind == 'i' or dkind == 'u'):
827              # add cbSize field for non-PCM files
828              fmt_chunk_data += b'\x00\x00'
829  
830          header_data += struct.pack('<I', len(fmt_chunk_data))
831          header_data += fmt_chunk_data
832  
833          # check data size (needs to be immediately before the data chunk)
834          # if too large for standard RIFF, use RF64 instead
835          resulting_file_size = len(header_data) + 4 + 4 + data.nbytes
836          is_rf64 = (resulting_file_size - 8) > 0xFFFFFFFF
837          if is_rf64:
838              header_data = b''
839              header_data += b'RF64'
840              header_data += b'\xFF\xFF\xFF\xFF'
841              header_data += b'WAVE'
842              header_data += b'ds64'
843              # size of ds64 chunk
844              header_data += struct.pack('<I', 28)
845              # will be filled later with real file size
846              header_data += struct.pack('<Q', 0)
847              header_data += struct.pack('<Q', data.nbytes)
848              header_data += struct.pack('<Q', data.shape[0])
849              # ignore 'table' field for now
850              header_data += struct.pack('<I', 0)
851              header_data += b'fmt '
852              header_data += struct.pack('<I', len(fmt_chunk_data))
853              header_data += fmt_chunk_data
854  
855          # fact chunk (non-PCM files)
856          if not (dkind == 'i' or dkind == 'u'):
857              header_data += b'fact'
858              header_data += struct.pack('<II', 4, data.shape[0])
859  
860          fid.write(header_data)
861  
862          # data chunk
863          fid.write(b'data')
864          # write data chunk size, unless its too big in which case 0xFFFFFFFF is written
865          fid.write(struct.pack('<I', min(data.nbytes, 4294967295)))
866  
867          if data.dtype.byteorder == '>' or (data.dtype.byteorder == '=' and
868                                             sys.byteorder == 'big'):
869              data = data.byteswap()
870          _array_tofile(fid, data)
871  
872          # Determine file size and place it in correct
873          # position at start of the file or the data chunk.
874          size = fid.tell()
875          if not is_rf64:
876              fid.seek(4)
877              fid.write(struct.pack('<I', size-8))
878          else:
879              fid.seek(20)
880              fid.write(struct.pack('<Q', size-8))
881  
882      finally:
883          if not hasattr(filename, 'write'):
884              fid.close()
885          else:
886              fid.seek(0)
887

889  def _array_tofile(fid, data):
890      # ravel gives a c-contiguous buffer
891      fid.write(data.ravel().view('b').data)
