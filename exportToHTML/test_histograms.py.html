522      def test_outlier(self):
523          """ 
524          Check the FD, Scott and Doane with outliers. 
525   
526          The FD estimates a smaller binwidth since it's less affected by 
527          outliers. Since the range is so (artificially) large, this means more 
528          bins, most of which will be empty, but the data of interest usually is 
529          unaffected. The Scott estimator is more affected and returns fewer bins, 
530          despite most of the variance being in one area of the data. The Doane 
531          estimator lies somewhere between the other two. 
532          """
533          xcenter = np.linspace(-10, 10, 50)
534          outlier_dataset = np.hstack((np.linspace(-110, -100, 5), xcenter))
535  
536          outlier_resultdict = {'fd': 21, 'scott': 5, 'doane': 11, 'stone': 6}

542      def test_scott_vs_stone(self):
543          """Verify that Scott's rule and Stone's rule converges for normally distributed data"""
544  
545          def nbins_ratio(seed, size):
546              rng = np.random.RandomState(seed)
547              x = rng.normal(loc=0, scale=2, size=size)
548              a, b = len(np.histogram(x, 'stone')[0]), len(np.histogram(x, 'scott')[0])
549              return a / (a + b)
550  
551          ll = [[nbins_ratio(seed, size) for size in np.geomspace(start=10, stop=100, num=4).round().astype(int)]
552                for seed in range(10)]
553  
554          # the average difference between the two methods decreases as the dataset size increases.
555          avg = abs(np.mean(ll, axis=0) - 0.5)
556          assert_almost_equal(avg, [0.15, 0.09, 0.08, 0.03], decimal=2)
