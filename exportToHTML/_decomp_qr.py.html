11   def safecall(f, name, *args, **kwargs):
12       """Call a LAPACK routine, determining lwork automatically and handling 
13       error return values"""
14       lwork = kwargs.get("lwork", None)
15       if lwork in (None, -1):
16           kwargs['lwork'] = -1
17           ret = f(*args, **kwargs)
18           kwargs['lwork'] = ret[-2][0].real.astype(np.int_)
19       ret = f(*args, **kwargs)
20       if ret[-1] < 0:
21           raise ValueError("illegal value in %dth argument of internal %s"
22                            % (-ret[-1], name))
23       return ret[:-2]

26   def qr(a, overwrite_a=False, lwork=None, mode='full', pivoting=False,
27          check_finite=True):
28       """ 
29       Compute QR decomposition of a matrix. 
30    
31       Calculate the decomposition ``A = Q R`` where Q is unitary/orthogonal 
32       and R upper triangular. 
33    
34       Parameters 
35       ---------- 
36       a : (M, N) array_like 
37           Matrix to be decomposed 
38       overwrite_a : bool, optional 
39           Whether data in `a` is overwritten (may improve performance if 
40           `overwrite_a` is set to True by reusing the existing input data 
41           structure rather than creating a new one.) 
42       lwork : int, optional 
43           Work array size, lwork >= a.shape[1]. If None or -1, an optimal size 
44           is computed. 
45       mode : {'full', 'r', 'economic', 'raw'}, optional 
46           Determines what information is to be returned: either both Q and R 
47           ('full', default), only R ('r') or both Q and R but computed in 
48           economy-size ('economic', see Notes). The final option 'raw' 
49           (added in SciPy 0.11) makes the function return two matrices 
50           (Q, TAU) in the internal format used by LAPACK. 
51       pivoting : bool, optional 
52           Whether or not factorization should include pivoting for rank-revealing 
53           qr decomposition. If pivoting, compute the decomposition 
54           ``A[:, P] = Q @ R`` as above, but where P is chosen such that the 
55           diagonal of R is non-increasing. Equivalently, albeit less efficiently, 
56           an explicit P matrix may be formed explicitly by permuting the rows or columns 
57           (depending on the side of the equation on which it is to be used) of 
58           an identity matrix. See Examples. 
59       check_finite : bool, optional 
60           Whether to check that the input matrix contains only finite numbers. 
61           Disabling may give a performance gain, but may result in problems 
62           (crashes, non-termination) if the inputs do contain infinities or NaNs. 
63    
64       Returns 
65       ------- 
66       Q : float or complex ndarray 
67           Of shape (M, M), or (M, K) for ``mode='economic'``. Not returned 
68           if ``mode='r'``. Replaced by tuple ``(Q, TAU)`` if ``mode='raw'``. 
69       R : float or complex ndarray 
70           Of shape (M, N), or (K, N) for ``mode in ['economic', 'raw']``. 
71           ``K = min(M, N)``. 
72       P : int ndarray 
73           Of shape (N,) for ``pivoting=True``. Not returned if 
74           ``pivoting=False``. 
75    
76       Raises 
77       ------ 
78       LinAlgError 
79           Raised if decomposition fails 
80    
81       Notes 
82       ----- 
83       This is an interface to the LAPACK routines dgeqrf, zgeqrf, 
84       dorgqr, zungqr, dgeqp3, and zgeqp3. 
85    
86       If ``mode=economic``, the shapes of Q and R are (M, K) and (K, N) instead 
87       of (M,M) and (M,N), with ``K=min(M,N)``. 
88    
89       Examples 
90       -------- 
91       >>> import numpy as np 
92       >>> from scipy import linalg 
93       >>> rng = np.random.default_rng() 
94       >>> a = rng.standard_normal((9, 6)) 
95    
96       >>> q, r = linalg.qr(a) 
97       >>> np.allclose(a, np.dot(q, r)) 
98       True 
99       >>> q.shape, r.shape 
100      ((9, 9), (9, 6)) 
101   
102      >>> r2 = linalg.qr(a, mode='r') 
103      >>> np.allclose(r, r2) 
104      True 
105   
106      >>> q3, r3 = linalg.qr(a, mode='economic') 
107      >>> q3.shape, r3.shape 
108      ((9, 6), (6, 6)) 
109   
110      >>> q4, r4, p4 = linalg.qr(a, pivoting=True) 
111      >>> d = np.abs(np.diag(r4)) 
112      >>> np.all(d[1:] <= d[:-1]) 
113      True 
114      >>> np.allclose(a[:, p4], np.dot(q4, r4)) 
115      True 
116      >>> P = np.eye(p4.size)[p4] 
117      >>> np.allclose(a, np.dot(q4, r4) @ P) 
118      True 
119      >>> np.allclose(a @ P.T, np.dot(q4, r4)) 
120      True 
121      >>> q4.shape, r4.shape, p4.shape 
122      ((9, 9), (9, 6), (6,)) 
123   
124      >>> q5, r5, p5 = linalg.qr(a, mode='economic', pivoting=True) 
125      >>> q5.shape, r5.shape, p5.shape 
126      ((9, 6), (6, 6), (6,)) 
127      >>> P = np.eye(6)[:, p5] 
128      >>> np.allclose(a @ P, np.dot(q5, r5)) 
129      True 
130   
131      """
132      # 'qr' was the old default, equivalent to 'full'. Neither 'full' nor
133      # 'qr' are used below.
134      # 'raw' is used internally by qr_multiply
135      if mode not in ['full', 'qr', 'r', 'economic', 'raw']:
136          raise ValueError("Mode argument should be one of ['full', 'r', "
137                           "'economic', 'raw']")
138  
139      if check_finite:
140          a1 = np.asarray_chkfinite(a)
141      else:
142          a1 = np.asarray(a)
143      if len(a1.shape) != 2:
144          raise ValueError("expected a 2-D array")
145  
146      M, N = a1.shape
147  
148      # accommodate empty arrays
149      if a1.size == 0:
150          K = min(M, N)
151  
152          if mode not in ['economic', 'raw']:
153              Q = np.empty_like(a1, shape=(M, M))
154              Q[...] = np.identity(M)
155              R = np.empty_like(a1)
156          else:
157              Q = np.empty_like(a1, shape=(M, K))
158              R = np.empty_like(a1, shape=(K, N))
159  
160          if pivoting:
161              Rj = R, np.arange(N, dtype=np.int32)
162          else:
163              Rj = R,
164  
165          if mode == 'r':
166              return Rj
167          elif mode == 'raw':
168              qr = np.empty_like(a1, shape=(M, N))
169              tau = np.zeros_like(a1, shape=(K,))
170              return ((qr, tau),) + Rj
171          return (Q,) + Rj
172  
173      overwrite_a = overwrite_a or (_datacopied(a1, a))
174  
175      if pivoting:
176          geqp3, = get_lapack_funcs(('geqp3',), (a1,))
177          qr, jpvt, tau = safecall(geqp3, "geqp3", a1, overwrite_a=overwrite_a)
178          jpvt -= 1  # geqp3 returns a 1-based index array, so subtract 1
179      else:
180          geqrf, = get_lapack_funcs(('geqrf',), (a1,))
181          qr, tau = safecall(geqrf, "geqrf", a1, lwork=lwork,
182                             overwrite_a=overwrite_a)
183  
184      if mode not in ['economic', 'raw'] or M < N:
185          R = np.triu(qr)
186      else:
187          R = np.triu(qr[:N, :])
188  
189      if pivoting:
190          Rj = R, jpvt
191      else:
192          Rj = R,
193  
194      if mode == 'r':
195          return Rj
196      elif mode == 'raw':
197          return ((qr, tau),) + Rj
198  
199      gor_un_gqr, = get_lapack_funcs(('orgqr',), (qr,))
200  
201      if M < N:
202          Q, = safecall(gor_un_gqr, "gorgqr/gungqr", qr[:, :M], tau,
203                        lwork=lwork, overwrite_a=1)
204      elif mode == 'economic':
205          Q, = safecall(gor_un_gqr, "gorgqr/gungqr", qr, tau, lwork=lwork,
206                        overwrite_a=1)
207      else:
208          t = qr.dtype.char
209          qqr = np.empty((M, M), dtype=t)
210          qqr[:, :N] = qr
211          Q, = safecall(gor_un_gqr, "gorgqr/gungqr", qqr, tau, lwork=lwork,
212                        overwrite_a=1)
213  
214      return (Q,) + Rj

217  def qr_multiply(a, c, mode='right', pivoting=False, conjugate=False,
218                  overwrite_a=False, overwrite_c=False):
219      """ 
220      Calculate the QR decomposition and multiply Q with a matrix. 
221   
222      Calculate the decomposition ``A = Q R`` where Q is unitary/orthogonal 
223      and R upper triangular. Multiply Q with a vector or a matrix c. 
224   
225      Parameters 
226      ---------- 
227      a : (M, N), array_like 
228          Input array 
229      c : array_like 
230          Input array to be multiplied by ``q``. 
231      mode : {'left', 'right'}, optional 
232          ``Q @ c`` is returned if mode is 'left', ``c @ Q`` is returned if 
233          mode is 'right'. 
234          The shape of c must be appropriate for the matrix multiplications, 
235          if mode is 'left', ``min(a.shape) == c.shape[0]``, 
236          if mode is 'right', ``a.shape[0] == c.shape[1]``. 
237      pivoting : bool, optional 
238          Whether or not factorization should include pivoting for rank-revealing 
239          qr decomposition, see the documentation of qr. 
240      conjugate : bool, optional 
241          Whether Q should be complex-conjugated. This might be faster 
242          than explicit conjugation. 
243      overwrite_a : bool, optional 
244          Whether data in a is overwritten (may improve performance) 
245      overwrite_c : bool, optional 
246          Whether data in c is overwritten (may improve performance). 
247          If this is used, c must be big enough to keep the result, 
248          i.e. ``c.shape[0]`` = ``a.shape[0]`` if mode is 'left'. 
249   
250      Returns 
251      ------- 
252      CQ : ndarray 
253          The product of ``Q`` and ``c``. 
254      R : (K, N), ndarray 
255          R array of the resulting QR factorization where ``K = min(M, N)``. 
256      P : (N,) ndarray 
257          Integer pivot array. Only returned when ``pivoting=True``. 
258   
259      Raises 
260      ------ 
261      LinAlgError 
262          Raised if QR decomposition fails. 
263   
264      Notes 
265      ----- 
266      This is an interface to the LAPACK routines ``?GEQRF``, ``?ORMQR``, 
267      ``?UNMQR``, and ``?GEQP3``. 
268   
269      .. versionadded:: 0.11.0 
270   
271      Examples 
272      -------- 
273      >>> import numpy as np 
274      >>> from scipy.linalg import qr_multiply, qr 
275      >>> A = np.array([[1, 3, 3], [2, 3, 2], [2, 3, 3], [1, 3, 2]]) 
276      >>> qc, r1, piv1 = qr_multiply(A, 2*np.eye(4), pivoting=1) 
277      >>> qc 
278      array([[-1.,  1., -1.], 
279             [-1., -1.,  1.], 
280             [-1., -1., -1.], 
281             [-1.,  1.,  1.]]) 
282      >>> r1 
283      array([[-6., -3., -5.            ], 
284             [ 0., -1., -1.11022302e-16], 
285             [ 0.,  0., -1.            ]]) 
286      >>> piv1 
287      array([1, 0, 2], dtype=int32) 
288      >>> q2, r2, piv2 = qr(A, mode='economic', pivoting=1) 
289      >>> np.allclose(2*q2 - qc, np.zeros((4, 3))) 
290      True 
291   
292      """
293      if mode not in ['left', 'right']:
294          raise ValueError("Mode argument can only be 'left' or 'right' but "
295                           f"not '{mode}'")
296      c = np.asarray_chkfinite(c)
297      if c.ndim < 2:
298          onedim = True
299          c = np.atleast_2d(c)
300          if mode == "left":
301              c = c.T
302      else:
303          onedim = False
304  
305      a = np.atleast_2d(np.asarray(a))  # chkfinite done in qr
306      M, N = a.shape
307  
308      if mode == 'left':
309          if c.shape[0] != min(M, N + overwrite_c*(M-N)):
310              raise ValueError('Array shapes are not compatible for Q @ c'
311                               f' operation: {a.shape} vs {c.shape}')
312      else:
313          if M != c.shape[1]:
314              raise ValueError('Array shapes are not compatible for c @ Q'
315                               f' operation: {c.shape} vs {a.shape}')
316  
317      raw = qr(a, overwrite_a, None, "raw", pivoting)
318      Q, tau = raw[0]
319  
320      # accommodate empty arrays
321      if c.size == 0:
322          return (np.empty_like(c),) + raw[1:]
323  
324      gor_un_mqr, = get_lapack_funcs(('ormqr',), (Q,))
325      if gor_un_mqr.typecode in ('s', 'd'):
326          trans = "T"
327      else:
328          trans = "C"
329  
330      Q = Q[:, :min(M, N)]
331      if M > N and mode == "left" and not overwrite_c:
332          if conjugate:
333              cc = np.zeros((c.shape[1], M), dtype=c.dtype, order="F")
334              cc[:, :N] = c.T
335          else:
336              cc = np.zeros((M, c.shape[1]), dtype=c.dtype, order="F")
337              cc[:N, :] = c
338              trans = "N"
339          if conjugate:
340              lr = "R"
341          else:
342              lr = "L"
343          overwrite_c = True
344      elif c.flags["C_CONTIGUOUS"] and trans == "T" or conjugate:
345          cc = c.T
346          if mode == "left":
347              lr = "R"
348          else:
349              lr = "L"
350      else:
351          trans = "N"
352          cc = c
353          if mode == "left":
354              lr = "L"
355          else:
356              lr = "R"
357      cQ, = safecall(gor_un_mqr, "gormqr/gunmqr", lr, trans, Q, tau, cc,
358                     overwrite_c=overwrite_c)
359      if trans != "N":
360          cQ = cQ.T
361      if mode == "right":
362          cQ = cQ[:, :min(M, N)]
363      if onedim:
364          cQ = cQ.ravel()
365  
366      return (cQ,) + raw[1:]

369  def rq(a, overwrite_a=False, lwork=None, mode='full', check_finite=True):
370      """ 
371      Compute RQ decomposition of a matrix. 
372   
373      Calculate the decomposition ``A = R Q`` where Q is unitary/orthogonal 
374      and R upper triangular. 
375   
376      Parameters 
377      ---------- 
378      a : (M, N) array_like 
379          Matrix to be decomposed 
380      overwrite_a : bool, optional 
381          Whether data in a is overwritten (may improve performance) 
382      lwork : int, optional 
383          Work array size, lwork >= a.shape[1]. If None or -1, an optimal size 
384          is computed. 
385      mode : {'full', 'r', 'economic'}, optional 
386          Determines what information is to be returned: either both Q and R 
387          ('full', default), only R ('r') or both Q and R but computed in 
388          economy-size ('economic', see Notes). 
389      check_finite : bool, optional 
390          Whether to check that the input matrix contains only finite numbers. 
391          Disabling may give a performance gain, but may result in problems 
392          (crashes, non-termination) if the inputs do contain infinities or NaNs. 
393   
394      Returns 
395      ------- 
396      R : float or complex ndarray 
397          Of shape (M, N) or (M, K) for ``mode='economic'``. ``K = min(M, N)``. 
398      Q : float or complex ndarray 
399          Of shape (N, N) or (K, N) for ``mode='economic'``. Not returned 
400          if ``mode='r'``. 
401   
402      Raises 
403      ------ 
404      LinAlgError 
405          If decomposition fails. 
406   
407      Notes 
408      ----- 
409      This is an interface to the LAPACK routines sgerqf, dgerqf, cgerqf, zgerqf, 
410      sorgrq, dorgrq, cungrq and zungrq. 
411   
412      If ``mode=economic``, the shapes of Q and R are (K, N) and (M, K) instead 
413      of (N,N) and (M,N), with ``K=min(M,N)``. 
414   
415      Examples 
416      -------- 
417      >>> import numpy as np 
418      >>> from scipy import linalg 
419      >>> rng = np.random.default_rng() 
420      >>> a = rng.standard_normal((6, 9)) 
421      >>> r, q = linalg.rq(a) 
422      >>> np.allclose(a, r @ q) 
423      True 
424      >>> r.shape, q.shape 
425      ((6, 9), (9, 9)) 
426      >>> r2 = linalg.rq(a, mode='r') 
427      >>> np.allclose(r, r2) 
428      True 
429      >>> r3, q3 = linalg.rq(a, mode='economic') 
430      >>> r3.shape, q3.shape 
431      ((6, 6), (6, 9)) 
432   
433      """
434      if mode not in ['full', 'r', 'economic']:
435          raise ValueError(
436                   "Mode argument should be one of ['full', 'r', 'economic']")
437  
438      if check_finite:
439          a1 = np.asarray_chkfinite(a)
440      else:
441          a1 = np.asarray(a)
442      if len(a1.shape) != 2:
443          raise ValueError('expected matrix')
444  
445      M, N = a1.shape
446  
447      # accommodate empty arrays
448      if a1.size == 0:
449          K = min(M, N)
450  
451          if not mode == 'economic':
452              R = np.empty_like(a1)
453              Q = np.empty_like(a1, shape=(N, N))
454              Q[...] = np.identity(N)
455          else:
456              R = np.empty_like(a1, shape=(M, K))
457              Q = np.empty_like(a1, shape=(K, N))
458  
459          if mode == 'r':
460              return R
461          return R, Q
462  
463      overwrite_a = overwrite_a or (_datacopied(a1, a))
464  
465      gerqf, = get_lapack_funcs(('gerqf',), (a1,))
466      rq, tau = safecall(gerqf, 'gerqf', a1, lwork=lwork,
467                         overwrite_a=overwrite_a)
468      if not mode == 'economic' or N < M:
469          R = np.triu(rq, N-M)
470      else:
471          R = np.triu(rq[-M:, -M:])
472  
473      if mode == 'r':
474          return R
475  
476      gor_un_grq, = get_lapack_funcs(('orgrq',), (rq,))
477  
478      if N < M:
479          Q, = safecall(gor_un_grq, "gorgrq/gungrq", rq[-N:], tau, lwork=lwork,
480                        overwrite_a=1)
481      elif mode == 'economic':
482          Q, = safecall(gor_un_grq, "gorgrq/gungrq", rq, tau, lwork=lwork,
483                        overwrite_a=1)
484      else:
485          rq1 = np.empty((N, N), dtype=rq.dtype)
486          rq1[-M:] = rq
487          Q, = safecall(gor_un_grq, "gorgrq/gungrq", rq1, tau, lwork=lwork,
488                        overwrite_a=1)
489  
490      return R, Q
