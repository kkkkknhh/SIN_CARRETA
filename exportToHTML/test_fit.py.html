109  def test_cont_fit(distname, arg, method):
110      run_xfail = int(os.getenv('SCIPY_XFAIL', default=False))
111      run_xslow = int(os.getenv('SCIPY_XSLOW', default=False))
112  
113      if distname in failing_fits[method] and not run_xfail:
114          # The generic `fit` method can't be expected to work perfectly for all
115          # distributions, data, and guesses. Some failures are expected.
116          msg = "Failure expected; set environment variable SCIPY_XFAIL=1 to run."
117          pytest.xfail(msg)
118  
119      if distname in xslow_fits[method] and not run_xslow:
120          msg = "Very slow; set environment variable SCIPY_XSLOW=1 to run."
121          pytest.skip(msg)
122  
123      distfn = getattr(stats, distname)
124  
125      truearg = np.hstack([arg, [0.0, 1.0]])
126      diffthreshold = np.max(np.vstack([truearg*thresh_percent,
127                                        np.full(distfn.numargs+2, thresh_min)]),
128                             0)
129  
130      for fit_size in fit_sizes:
131          # Note that if a fit succeeds, the other fit_sizes are skipped
132          np.random.seed(1234)
133  
134          with np.errstate(all='ignore'):
135              rvs = distfn.rvs(size=fit_size, *arg)
136              if method == 'MLE' and distfn.name in mle_use_floc0:
137                  kwds = {'floc': 0}
138              else:
139                  kwds = {}
140              # start with default values
141              est = distfn.fit(rvs, method=method, **kwds)
142              if method == 'MLE':
143                  # Trivial test of the use of CensoredData.  The fit() method
144                  # will check that data contains no actual censored data, and
145                  # do a regular uncensored fit.
146                  data1 = stats.CensoredData(rvs)
147                  est1 = distfn.fit(data1, **kwds)
148                  msg = ('Different results fitting uncensored data wrapped as'
149                         f' CensoredData: {distfn.name}: est={est} est1={est1}')
150                  assert_allclose(est1, est, rtol=1e-10, err_msg=msg)
151              if method == 'MLE' and distname not in fail_interval_censored:
152                  # Convert the first `nic` values in rvs to interval-censored
153                  # values. The interval is small, so est2 should be close to
154                  # est.
155                  nic = 15
156                  interval = np.column_stack((rvs, rvs))
157                  interval[:nic, 0] *= 0.99
158                  interval[:nic, 1] *= 1.01
159                  interval.sort(axis=1)
160                  data2 = stats.CensoredData(interval=interval)
161                  est2 = distfn.fit(data2, **kwds)
162                  msg = ('Different results fitting interval-censored'
163                         f' data: {distfn.name}: est={est} est2={est2}')
164                  assert_allclose(est2, est, rtol=0.05, err_msg=msg)
165  
166          diff = est - truearg
167  
168          # threshold for location
169          diffthreshold[-2] = np.max([np.abs(rvs.mean())*thresh_percent,
170                                      thresh_min])
171  
172          if np.any(np.isnan(est)):
173              raise AssertionError('nan returned in fit')
174          else:
175              if np.all(np.abs(diff) <= diffthreshold):
176                  break
177      else:
178          txt = f'parameter: {str(truearg)}\n'
179          txt += f'estimated: {str(est)}\n'
180          txt += f'diff     : {str(diff)}\n'
181          raise AssertionError(f'fit not very good in {distfn.name}\n' + txt)

1039     def test_custom_statistic(self):
1040         # Test support for custom statistic function.
1041 
1042         # References:
1043         # [1] Pyke, R. (1965).  "Spacings".  Journal of the Royal Statistical
1044         #     Society: Series B (Methodological), 27(3): 395-436.
1045         # [2] Burrows, P. M. (1979).  "Selected Percentage Points of
1046         #     Greenwood's Statistics".  Journal of the Royal Statistical
1047         #     Society. Series A (General), 142(2): 256-258.
1048 
1049         # Use the Greenwood statistic for illustration; see [1, p.402].
1050         def greenwood(dist, data, *, axis):
1051             x = np.sort(data, axis=axis)
1052             y = dist.cdf(x)
1053             d = np.diff(y, axis=axis, prepend=0, append=1)
1054             return np.sum(d ** 2, axis=axis)
1055 
1056         # Run the Monte Carlo test with sample size = 5 on a fully specified
1057         # null distribution, and compare the simulated quantiles to the exact
1058         # ones given in [2, Table 1, column (n = 5)].
1059         rng = np.random.default_rng(9121950977643805391)
1060         data = stats.expon.rvs(size=5, random_state=rng)
1061         result = goodness_of_fit(stats.expon, data,
1062                                  known_params={'loc': 0, 'scale': 1},
1063                                  statistic=greenwood, rng=rng)
1064         p = [.01, .05, .1, .2, .3, .4, .5, .6, .7, .8, .9, .95, .99]
1065         exact_quantiles = [
1066             .183863, .199403, .210088, .226040, .239947, .253677, .268422,
1067             .285293, .306002, .334447, .382972, .432049, .547468]
1068         simulated_quantiles = np.quantile(result.null_distribution, p)
1069         assert_allclose(simulated_quantiles, exact_quantiles, atol=0.005)
