67   class Intent:
68       def __init__(self, intent_list=[]):
69           self.intent_list = intent_list[:]
70           flags = 0
71           for i in intent_list:
72               if i == "optional":
73                   flags |= wrap.F2PY_OPTIONAL
74               else:
75                   flags |= getattr(wrap, "F2PY_INTENT_" + i.upper())
76           self.flags = flags
77   
78       def __getattr__(self, name):
79           name = name.lower()
80           if name == "in_":
81               name = "in"
82           return self.__class__(self.intent_list + [name])
83   
84       def __str__(self):
85           return "intent(%s)" % (",".join(self.intent_list))
86   
87       def __repr__(self):
88           return "Intent(%r)" % (self.intent_list)
89   
90       def is_intent(self, *names):
91           return all(name in self.intent_list for name in names)
92   
93       def is_intent_exact(self, *names):
94           return len(self.intent_list) == len(names) and self.is_intent(*names)

168  class Type:
169      _type_cache = {}
170  
171      def __new__(cls, name):
172          if isinstance(name, np.dtype):
173              dtype0 = name
174              name = None
175              for n, i in c_names_dict.items():
176                  if not isinstance(i, type) and dtype0.type is i.type:
177                      name = n
178                      break
179          obj = cls._type_cache.get(name.upper(), None)
180          if obj is not None:
181              return obj
182          obj = object.__new__(cls)
183          obj._init(name)
184          cls._type_cache[name.upper()] = obj
185          return obj
186  
187      def _init(self, name):
188          self.NAME = name.upper()
189  
190          if self.NAME == 'CHARACTER':
191              info = c_names_dict[self.NAME]
192              self.type_num = wrap.NPY_STRING
193              self.elsize = 1
194              self.dtype = np.dtype('c')
195          elif self.NAME.startswith('STRING'):
196              info = c_names_dict[self.NAME[:6]]
197              self.type_num = wrap.NPY_STRING
198              self.elsize = int(self.NAME[6:] or 0)
199              self.dtype = np.dtype(f'S{self.elsize}')
200          else:
201              info = c_names_dict[self.NAME]
202              self.type_num = getattr(wrap, 'NPY_' + self.NAME)
203              self.elsize = info.itemsize
204              self.dtype = np.dtype(info.type)
205  
206          assert self.type_num == info.num
207          self.type = info.type
208          self.dtypechar = info.char
209  
210      def __repr__(self):
211          return (f"Type({self.NAME})|type_num={self.type_num},"
212                  f" dtype={self.dtype},"
213                  f" type={self.type}, elsize={self.elsize},"
214                  f" dtypechar={self.dtypechar}")
215  
216      def cast_types(self):
217          return [self.__class__(_m) for _m in _cast_dict[self.NAME]]
218  
219      def all_types(self):
220          return [self.__class__(_m) for _m in _type_names]
221  
222      def smaller_types(self):
223          bits = c_names_dict[self.NAME].alignment
224          types = []
225          for name in _type_names:
226              if c_names_dict[name].alignment < bits:
227                  types.append(Type(name))
228          return types
229  
230      def equal_types(self):
231          bits = c_names_dict[self.NAME].alignment
232          types = []
233          for name in _type_names:
234              if name == self.NAME:
235                  continue
236              if c_names_dict[name].alignment == bits:
237                  types.append(Type(name))
238          return types
239  
240      def larger_types(self):
241          bits = c_names_dict[self.NAME].alignment
242          types = []
243          for name in _type_names:
244              if c_names_dict[name].alignment > bits:
245                  types.append(Type(name))
246          return types

249  class Array:
250  
251      def __repr__(self):
252          return (f'Array({self.type}, {self.dims}, {self.intent},'
253                  f' {self.obj})|arr={self.arr}')
254  
255      def __init__(self, typ, dims, intent, obj):
256          self.type = typ
257          self.dims = dims
258          self.intent = intent
259          self.obj_copy = copy.deepcopy(obj)
260          self.obj = obj
261  
262          # arr.dtypechar may be different from typ.dtypechar
263          self.arr = wrap.call(typ.type_num,
264                               typ.elsize,
265                               dims, intent.flags, obj)
266  
267          assert isinstance(self.arr, np.ndarray)
268  
269          self.arr_attr = wrap.array_attrs(self.arr)
270  
271          if len(dims) > 1:
272              if self.intent.is_intent("c"):
273                  assert (intent.flags & wrap.F2PY_INTENT_C)
274                  assert not self.arr.flags["FORTRAN"]
275                  assert self.arr.flags["CONTIGUOUS"]
276                  assert (not self.arr_attr[6] & wrap.FORTRAN)
277              else:
278                  assert (not intent.flags & wrap.F2PY_INTENT_C)
279                  assert self.arr.flags["FORTRAN"]
280                  assert not self.arr.flags["CONTIGUOUS"]
281                  assert (self.arr_attr[6] & wrap.FORTRAN)
282  
283          if obj is None:
284              self.pyarr = None
285              self.pyarr_attr = None
286              return
287  
288          if intent.is_intent("cache"):
289              assert isinstance(obj, np.ndarray), repr(type(obj))
290              self.pyarr = np.array(obj).reshape(*dims).copy()
291          else:
292              self.pyarr = np.array(
293                  np.array(obj, dtype=typ.dtypechar).reshape(*dims),
294                  order=self.intent.is_intent("c") and "C" or "F",
295              )
296              assert self.pyarr.dtype == typ
297          self.pyarr.setflags(write=self.arr.flags["WRITEABLE"])
298          assert self.pyarr.flags["OWNDATA"], (obj, intent)
299          self.pyarr_attr = wrap.array_attrs(self.pyarr)
300  
301          if len(dims) > 1:
302              if self.intent.is_intent("c"):
303                  assert not self.pyarr.flags["FORTRAN"]
304                  assert self.pyarr.flags["CONTIGUOUS"]
305                  assert (not self.pyarr_attr[6] & wrap.FORTRAN)
306              else:
307                  assert self.pyarr.flags["FORTRAN"]
308                  assert not self.pyarr.flags["CONTIGUOUS"]
309                  assert (self.pyarr_attr[6] & wrap.FORTRAN)
310  
311          assert self.arr_attr[1] == self.pyarr_attr[1]  # nd
312          assert self.arr_attr[2] == self.pyarr_attr[2]  # dimensions
313          if self.arr_attr[1] <= 1:
314              assert self.arr_attr[3] == self.pyarr_attr[3], repr((
315                  self.arr_attr[3],
316                  self.pyarr_attr[3],
317                  self.arr.tobytes(),
318                  self.pyarr.tobytes(),
319              ))  # strides
320          assert self.arr_attr[5][-2:] == self.pyarr_attr[5][-2:], repr((
321              self.arr_attr[5], self.pyarr_attr[5]
322              ))  # descr
323          assert self.arr_attr[6] == self.pyarr_attr[6], repr((
324              self.arr_attr[6],
325              self.pyarr_attr[6],
326              flags2names(0 * self.arr_attr[6] - self.pyarr_attr[6]),
327              flags2names(self.arr_attr[6]),
328              intent,
329          ))  # flags
330  
331          if intent.is_intent("cache"):
332              assert self.arr_attr[5][3] >= self.type.elsize
333          else:
334              assert self.arr_attr[5][3] == self.type.elsize
335              assert (self.arr_equal(self.pyarr, self.arr))
336  
337          if isinstance(self.obj, np.ndarray):
338              if typ.elsize == Type(obj.dtype).elsize:
339                  if not intent.is_intent("copy") and self.arr_attr[1] <= 1:
340                      assert self.has_shared_memory()
341  
342      def arr_equal(self, arr1, arr2):
343          if arr1.shape != arr2.shape:
344              return False
345          return (arr1 == arr2).all()
346  
347      def __str__(self):
348          return str(self.arr)
349  
350      def has_shared_memory(self):
351          """Check that created array shares data with input array."""
352          if self.obj is self.arr:
353              return True
354          if not isinstance(self.obj, np.ndarray):
355              return False
356          obj_attr = wrap.array_attrs(self.obj)
357          return obj_attr[0] == self.arr_attr[0]
