53       def forward(self, input, extra_input):  # type: ignore[override]
54           # Temporarily using len(shape) instead of ndim due to JIT issue
55           # https://github.com/pytorch/pytorch/issues/23890
56           if len(input.shape) != 4:
57               raise ValueError("Input shape must be `(N, C, H, W)`!")
58           if self.padding_mode != "zeros":
59               _reversed_padding_repeated_twice = _reverse_repeat_padding(self.padding)
60               input = F.pad(

62               )
63           return torch.ops.quantized.conv2d_add(
64               input, extra_input, self._packed_params, self.scale, self.zero_point
65           )
