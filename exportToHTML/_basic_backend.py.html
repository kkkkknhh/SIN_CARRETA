8    def _validate_fft_args(workers, plan, norm):
9        if workers is not None:
10           raise ValueError(xp_unsupported_param_msg("workers"))
11       if plan is not None:
12           raise ValueError(xp_unsupported_param_msg("plan"))
13       if norm is None:
14           norm = 'backward'
15       return norm

18   # these functions expect complex input in the fft standard extension
19   complex_funcs = {'fft', 'ifft', 'fftn', 'ifftn', 'hfft', 'irfft', 'irfftn'}
20   
21   # pocketfft is used whenever SCIPY_ARRAY_API is not set,
22   # or x is a NumPy array or array-like.
23   # When SCIPY_ARRAY_API is set, we try to use xp.fft for CuPy arrays,
24   # PyTorch arrays and other array API standard supporting objects.
25   # If xp.fft does not exist, we attempt to convert to np and back to use pocketfft.
26   
27   def _execute_1D(func_str, pocketfft_func, x, n, axis, norm, overwrite_x, workers, plan):
28       xp = array_namespace(x)
29   
30       if is_numpy(xp):
31           x = np.asarray(x)
32           return pocketfft_func(x, n=n, axis=axis, norm=norm,
33                                 overwrite_x=overwrite_x, workers=workers, plan=plan)
34   
35       norm = _validate_fft_args(workers, plan, norm)
36       if hasattr(xp, 'fft'):
37           xp_func = getattr(xp.fft, func_str)
38           if func_str in complex_funcs:
39               try:
40                   res = xp_func(x, n=n, axis=axis, norm=norm)
41               except: # backends may require complex input  # noqa: E722
42                   x = xp_float_to_complex(x, xp)
43                   res = xp_func(x, n=n, axis=axis, norm=norm)
44               return res
45           return xp_func(x, n=n, axis=axis, norm=norm)
46   
47       x = np.asarray(x)
48       y = pocketfft_func(x, n=n, axis=axis, norm=norm)
49       return xp.asarray(y)

52   def _execute_nD(func_str, pocketfft_func, x, s, axes, norm, overwrite_x, workers, plan):
53       xp = array_namespace(x)
54       
55       if is_numpy(xp):
56           x = np.asarray(x)
57           return pocketfft_func(x, s=s, axes=axes, norm=norm,
58                                 overwrite_x=overwrite_x, workers=workers, plan=plan)
59   
60       norm = _validate_fft_args(workers, plan, norm)
61       if hasattr(xp, 'fft'):
62           xp_func = getattr(xp.fft, func_str)
63           if func_str in complex_funcs:
64               try:
65                   res = xp_func(x, s=s, axes=axes, norm=norm)
66               except: # backends may require complex input  # noqa: E722
67                   x = xp_float_to_complex(x, xp)
68                   res = xp_func(x, s=s, axes=axes, norm=norm)
69               return res
70           return xp_func(x, s=s, axes=axes, norm=norm)
71   
72       x = np.asarray(x)
73       y = pocketfft_func(x, s=s, axes=axes, norm=norm)
74       return xp.asarray(y)

158  def _swap_direction(norm):
159      if norm in (None, 'backward'):
160          norm = 'forward'
161      elif norm == 'forward':
162          norm = 'backward'
163      elif norm != 'ortho':
164          raise ValueError(f'Invalid norm value {norm}; should be "backward", '
165                           '"ortho", or "forward".')
166      return norm

169  def hfftn(x, s=None, axes=None, norm=None,
170            overwrite_x=False, workers=None, *, plan=None):
171      xp = array_namespace(x)
172      if is_numpy(xp):
173          x = np.asarray(x)
174          return _pocketfft.hfftn(x, s, axes, norm, overwrite_x, workers, plan=plan)
175      if is_complex(x, xp):
176          x = xp.conj(x)
177      return irfftn(x, s, axes, _swap_direction(norm),
178                    overwrite_x, workers, plan=plan)

186  def ihfftn(x, s=None, axes=None, norm=None,
187             overwrite_x=False, workers=None, *, plan=None):
188      xp = array_namespace(x)
189      if is_numpy(xp):
190          x = np.asarray(x)
191          return _pocketfft.ihfftn(x, s, axes, norm, overwrite_x, workers, plan=plan)
192      return xp.conj(rfftn(x, s, axes, _swap_direction(norm),
193                           overwrite_x, workers, plan=plan))
