1    """Precompute the polynomials for the asymptotic expansion of the 
2    generalized exponential integral. 
3     
4    Sources 
5    ------- 
6    [1] NIST, Digital Library of Mathematical Functions, 
7        https://dlmf.nist.gov/8.20#ii 
8     
9    """
10   import os
11   
12   try:
13       import sympy
14       from sympy import Poly
15       x = sympy.symbols('x')
16   except ImportError:
17       pass
18   
19   
20   def generate_A(K):
21       A = [Poly(1, x)]
22       for k in range(K):
23           A.append(Poly(1 - 2*k*x, x)*A[k] + Poly(x*(x + 1))*A[k].diff())
24       return A

35       print(__doc__)
36       fn = os.path.join('..', 'cephes', 'expn.h')

38       K = 12
39       A = generate_A(K)

41           f.write(WARNING)

43           for k, Ak in enumerate(A):
44               ', '.join([str(x.evalf(18)) for x in Ak.coeffs()])
45               f.write(f"static const double A{k}[] = {{tmp}};\n")
46           ", ".join([f"A{k}" for k in range(K + 1)])
47           f.write("static const double *A[] = {{tmp}};\n")
48           ", ".join([str(Ak.degree()) for Ak in A])
49           f.write("static const int Adegs[] = {{tmp}};\n")
