7    from torch._jit_internal import boolean_dispatched
8    from torch.fx import Transformer
9    from torch.fx.node import Argument, Target
10   from torch.fx.operator_schemas import _torchscript_type_to_python_type

13   class AnnotateTypesWithSchema(Transformer):
14       """ 
15       Use Python function signatures to annotate types for `Nodes` within an FX graph. 
16       This pulls out Python function signatures for: 
17    
18           1. Standard `torch.nn` Module calls 
19           2. `torch.nn.functional` calls 
20           3. Attribute fetches via `get_attr` 
21    
22       Example usage: 
23    
24           m = torchvision.models.resnet18() 
25    
26           traced = torch.fx.symbolic_trace(m) 
27    
28           traced = AnnotateTypesWithSchema(traced).transform() 
29    
30       """
31   
32       def __init__(
33           self,
34           module: torch.nn.Module,
35           annotate_functionals: bool = True,
36           annotate_modules: bool = True,
37           annotate_get_attrs: bool = True,
38       ):
39           super().__init__(module)
40           self.annotate_functionals = annotate_functionals
41           self.annotate_modules = annotate_modules
42           self.annotate_get_attrs = annotate_get_attrs
43   
44       def call_function(
45           self, target: Target, args: tuple[Argument, ...], kwargs: dict[str, Any]
46       ):
47           python_ret_type = None
48           if self.annotate_functionals and target.__module__ == "torch.nn.functional":
49               target_for_analysis = target
50               if target in boolean_dispatched:
51                   # HACK: `boolean_dispatch` as used in `torch.nn.functional` makes it so that we have
52                   # a 2-way dispatch based on a boolean value. Here we check that the `true` and `false`
53                   # branches of the dispatch have exactly the same signature. If they do, use the `true`
54                   # branch signature for analysis. Otherwise, leave this un-normalized
55                   assert not isinstance(target, str)
56                   dispatched = boolean_dispatched[target]
57                   if_true, if_false = dispatched["if_true"], dispatched["if_false"]
58                   # TODO: can we emit the union of these? What are the implications on TorchScript
59                   # compilation?
60                   if (
61                       inspect.signature(if_true).return_annotation
62                       != inspect.signature(if_false).return_annotation
63                   ):
64                       return super().call_function(target, args, kwargs)
65                   target_for_analysis = if_true
66   
67               python_ret_type = self._extract_python_return_type(target_for_analysis)
68   
69           return_proxy = super().call_function(target, args, kwargs)
70           return_proxy.node.type = (
71               return_proxy.node.type if return_proxy.node.type else python_ret_type
72           )
73           return return_proxy
74   
75       def call_module(
76           self, target: Target, args: tuple[Argument, ...], kwargs: dict[str, Any]
77       ):
78           python_ret_type = None
79           assert isinstance(target, str)
80           submod = self.fetch_attr(target)
81           if self.annotate_modules and hasattr(submod.__class__, "__name__"):
82               classname = submod.__class__.__name__
83               if getattr(torch.nn, classname, None) == submod.__class__:
84                   python_ret_type = self._extract_python_return_type(submod.forward)
85           return_proxy = super().call_module(target, args, kwargs)
86           return_proxy.node.type = (
87               return_proxy.node.type if return_proxy.node.type else python_ret_type
88           )
89           return return_proxy
90   
91       def get_attr(
92           self,
93           target: torch.fx.node.Target,
94           args: tuple[Argument, ...],
95           kwargs: dict[str, Any],
96       ):
97           attr_proxy = super().get_attr(target, args, kwargs)
98   
99           if self.annotate_get_attrs:
100              module_itr = self.module
101              assert isinstance(target, str)
102              atoms = target.split(".")
103              for i, atom in enumerate(atoms):
104                  if not hasattr(module_itr, atom):
105                      raise RuntimeError(
106                          f"Node referenced nonextent target {'.'.join(atoms[:i])}!"
107                      )
108                  module_itr = getattr(module_itr, atom)
109  
110              maybe_inferred_ts_type = torch._C._jit_try_infer_type(module_itr)
111              if maybe_inferred_ts_type.success():
112                  python_type = _torchscript_type_to_python_type(
113                      maybe_inferred_ts_type.type()
114                  )
115                  attr_proxy.node.type = (
116                      python_type if not attr_proxy.node.type else attr_proxy.node.type
117                  )
118  
119          return attr_proxy
120  
121      def _extract_python_return_type(self, target: Target) -> Optional[Any]:
122          """ 
123          Given a Python call target, try to extract the Python return annotation 
124          if it is available, otherwise return None 
125   
126          Args: 
127   
128              target (Callable): Python callable to get return annotation for 
129   
130          Returns: 
131   
132              Optional[Any]: Return annotation from the `target`, or None if it was 
133                  not available. 
134          """
135          assert callable(target)
136          try:
137              sig = inspect.signature(target)
138          except (ValueError, TypeError):
139              return None
140  
141          return (
142              sig.return_annotation
143              if sig.return_annotation is not inspect.Signature.empty
144              else None
145          )
