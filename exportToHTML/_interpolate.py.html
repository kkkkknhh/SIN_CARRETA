19   def lagrange(x, w):
20       r""" 
21       Return a Lagrange interpolating polynomial. 
22    
23       Given two 1-D arrays `x` and `w,` returns the Lagrange interpolating 
24       polynomial through the points ``(x, w)``. 
25    
26       Warning: This implementation is numerically unstable. Do not expect to 
27       be able to use more than about 20 points even if they are chosen optimally. 
28    
29       Parameters 
30       ---------- 
31       x : array_like 
32           `x` represents the x-coordinates of a set of datapoints. 
33       w : array_like 
34           `w` represents the y-coordinates of a set of datapoints, i.e., f(`x`). 
35    
36       Returns 
37       ------- 
38       lagrange : `numpy.poly1d` instance 
39           The Lagrange interpolating polynomial. 
40    
41       Examples 
42       -------- 
43       Interpolate :math:`f(x) = x^3` by 3 points. 
44    
45       >>> import numpy as np 
46       >>> from scipy.interpolate import lagrange 
47       >>> x = np.array([0, 1, 2]) 
48       >>> y = x**3 
49       >>> poly = lagrange(x, y) 
50    
51       Since there are only 3 points, Lagrange polynomial has degree 2. Explicitly, 
52       it is given by 
53    
54       .. math:: 
55    
56           \begin{aligned} 
57               L(x) &= 1\times \frac{x (x - 2)}{-1} + 8\times \frac{x (x-1)}{2} \\ 
58                    &= x (-2 + 3x) 
59           \end{aligned} 
60    
61       >>> from numpy.polynomial.polynomial import Polynomial 
62       >>> Polynomial(poly.coef[::-1]).coef 
63       array([ 0., -2.,  3.]) 
64    
65       >>> import matplotlib.pyplot as plt 
66       >>> x_new = np.arange(0, 2.1, 0.1) 
67       >>> plt.scatter(x, y, label='data') 
68       >>> plt.plot(x_new, Polynomial(poly.coef[::-1])(x_new), label='Polynomial') 
69       >>> plt.plot(x_new, 3*x_new**2 - 2*x_new + 0*x_new, 
70       ...          label=r"$3 x^2 - 2 x$", linestyle='-.') 
71       >>> plt.legend() 
72       >>> plt.show() 
73    
74       """
75   
76       M = len(x)
77       p = poly1d(0.0)
78       for j in range(M):
79           pt = poly1d(w[j])
80           for k in range(M):
81               if k == j:
82                   continue
83               fac = x[j]-x[k]
84               pt *= poly1d([1.0, -x[k]])/fac
85           p += pt
86       return p

132  def _check_broadcast_up_to(arr_from, shape_to, name):
133      """Helper to check that arr_from broadcasts up to shape_to"""
134      shape_from = arr_from.shape
135      if len(shape_to) >= len(shape_from):
136          for t, f in zip(shape_to[::-1], shape_from[::-1]):
137              if f != 1 and f != t:
138                  break
139          else:  # all checks pass, do the upcasting that we need later
140              if arr_from.size != 1 and arr_from.shape != shape_to:
141                  arr_from = np.ones(shape_to, arr_from.dtype) * arr_from
142              return arr_from.ravel()
143      # at least one check failed
144      raise ValueError(f'{name} argument must be able to broadcast up '
145                       f'to shape {shape_to} but had shape {shape_from}')

154  class interp1d(_Interpolator1D):
155      """ 
156      Interpolate a 1-D function. 
157   
158      .. legacy:: class 
159   
160          For a guide to the intended replacements for `interp1d` see 
161          :ref:`tutorial-interpolate_1Dsection`. 
162   
163      `x` and `y` are arrays of values used to approximate some function f: 
164      ``y = f(x)``. This class returns a function whose call method uses 
165      interpolation to find the value of new points. 
166   
167      Parameters 
168      ---------- 
169      x : (npoints, ) array_like 
170          A 1-D array of real values. 
171      y : (..., npoints, ...) array_like 
172          A N-D array of real values. The length of `y` along the interpolation 
173          axis must be equal to the length of `x`. Use the ``axis`` parameter 
174          to select correct axis. Unlike other interpolators, the default 
175          interpolation axis is the last axis of `y`. 
176      kind : str or int, optional 
177          Specifies the kind of interpolation as a string or as an integer 
178          specifying the order of the spline interpolator to use. 
179          The string has to be one of 'linear', 'nearest', 'nearest-up', 'zero', 
180          'slinear', 'quadratic', 'cubic', 'previous', or 'next'. 'zero', 
181          'slinear', 'quadratic' and 'cubic' refer to a spline interpolation of 
182          zeroth, first, second or third order; 'previous' and 'next' simply 
183          return the previous or next value of the point; 'nearest-up' and 
184          'nearest' differ when interpolating half-integers (e.g. 0.5, 1.5) 
185          in that 'nearest-up' rounds up and 'nearest' rounds down. Default 
186          is 'linear'. 
187      axis : int, optional 
188          Axis in the ``y`` array corresponding to the x-coordinate values. Unlike 
189          other interpolators, defaults to ``axis=-1``. 
190      copy : bool, optional 
191          If ``True``, the class makes internal copies of x and y. If ``False``, 
192          references to ``x`` and ``y`` are used if possible. The default is to copy. 
193      bounds_error : bool, optional 
194          If True, a ValueError is raised any time interpolation is attempted on 
195          a value outside of the range of x (where extrapolation is 
196          necessary). If False, out of bounds values are assigned `fill_value`. 
197          By default, an error is raised unless ``fill_value="extrapolate"``. 
198      fill_value : array-like or (array-like, array_like) or "extrapolate", optional 
199          - if a ndarray (or float), this value will be used to fill in for 
200            requested points outside of the data range. If not provided, then 
201            the default is NaN. The array-like must broadcast properly to the 
202            dimensions of the non-interpolation axes. 
203          - If a two-element tuple, then the first element is used as a 
204            fill value for ``x_new < x[0]`` and the second element is used for 
205            ``x_new > x[-1]``. Anything that is not a 2-element tuple (e.g., 
206            list or ndarray, regardless of shape) is taken to be a single 
207            array-like argument meant to be used for both bounds as 
208            ``below, above = fill_value, fill_value``. Using a two-element tuple 
209            or ndarray requires ``bounds_error=False``. 
210   
211            .. versionadded:: 0.17.0 
212          - If "extrapolate", then points outside the data range will be 
213            extrapolated. 
214   
215            .. versionadded:: 0.17.0 
216      assume_sorted : bool, optional 
217          If False, values of `x` can be in any order and they are sorted first. 
218          If True, `x` has to be an array of monotonically increasing values. 
219   
220      Attributes 
221      ---------- 
222      fill_value 
223   
224      Methods 
225      ------- 
226      __call__ 
227   
228      See Also 
229      -------- 
230      splrep, splev 
231          Spline interpolation/smoothing based on FITPACK. 
232      UnivariateSpline : An object-oriented wrapper of the FITPACK routines. 
233      interp2d : 2-D interpolation 
234   
235      Notes 
236      ----- 
237      Calling `interp1d` with NaNs present in input values results in 
238      undefined behaviour. 
239   
240      Input values `x` and `y` must be convertible to `float` values like 
241      `int` or `float`. 
242   
243      If the values in `x` are not unique, the resulting behavior is 
244      undefined and specific to the choice of `kind`, i.e., changing 
245      `kind` will change the behavior for duplicates. 
246   
247   
248      Examples 
249      -------- 
250      >>> import numpy as np 
251      >>> import matplotlib.pyplot as plt 
252      >>> from scipy import interpolate 
253      >>> x = np.arange(0, 10) 
254      >>> y = np.exp(-x/3.0) 
255      >>> f = interpolate.interp1d(x, y) 
256   
257      >>> xnew = np.arange(0, 9, 0.1) 
258      >>> ynew = f(xnew)   # use interpolation function returned by `interp1d` 
259      >>> plt.plot(x, y, 'o', xnew, ynew, '-') 
260      >>> plt.show()

263      def __init__(self, x, y, kind='linear', axis=-1,
264                   copy=True, bounds_error=None, fill_value=np.nan,
265                   assume_sorted=False):
266          """ Initialize a 1-D linear interpolation class."""
267          _Interpolator1D.__init__(self, x, y, axis=axis)
268  
269          self.bounds_error = bounds_error  # used by fill_value setter
270  
271          # `copy` keyword semantics changed in NumPy 2.0, once that is
272          # the minimum version this can use `copy=None`.
273          self.copy = copy
274          if not copy:
275              self.copy = copy_if_needed
276  
277          if kind in ['zero', 'slinear', 'quadratic', 'cubic']:
278              order = {'zero': 0, 'slinear': 1,
279                       'quadratic': 2, 'cubic': 3}[kind]
280              kind = 'spline'
281          elif isinstance(kind, int):
282              order = kind
283              kind = 'spline'
284          elif kind not in ('linear', 'nearest', 'nearest-up', 'previous',
285                            'next'):
286              raise NotImplementedError(f"{kind} is unsupported: Use fitpack "
287                                        "routines for other types.")
288          x = array(x, copy=self.copy)
289          y = array(y, copy=self.copy)
290  
291          if not assume_sorted:
292              ind = np.argsort(x, kind="mergesort")
293              x = x[ind]
294              y = np.take(y, ind, axis=axis)
295  
296          if x.ndim != 1:
297              raise ValueError("the x array must have exactly one dimension.")
298          if y.ndim == 0:
299              raise ValueError("the y array must have at least one dimension.")
300  
301          # Force-cast y to a floating-point type, if it's not yet one
302          if not issubclass(y.dtype.type, np.inexact):
303              y = y.astype(np.float64)
304  
305          # Backward compatibility
306          self.axis = axis % y.ndim
307  
308          # Interpolation goes internally along the first axis
309          self.y = y
310          self._y = self._reshape_yi(self.y)
311          self.x = x
312          del y, x  # clean up namespace to prevent misuse; use attributes
313          self._kind = kind
314  
315          # Adjust to interpolation kind; store reference to *unbound*
316          # interpolation methods, in order to avoid circular references to self
317          # stored in the bound instance methods, and therefore delayed garbage
318          # collection.  See: https://docs.python.org/reference/datamodel.html
319          if kind in ('linear', 'nearest', 'nearest-up', 'previous', 'next'):
320              # Make a "view" of the y array that is rotated to the interpolation
321              # axis.
322              minval = 1
323              if kind == 'nearest':
324                  # Do division before addition to prevent possible integer
325                  # overflow
326                  self._side = 'left'
327                  self.x_bds = self.x / 2.0
328                  self.x_bds = self.x_bds[1:] + self.x_bds[:-1]
329  
330                  self._call = self.__class__._call_nearest
331              elif kind == 'nearest-up':
332                  # Do division before addition to prevent possible integer
333                  # overflow
334                  self._side = 'right'
335                  self.x_bds = self.x / 2.0
336                  self.x_bds = self.x_bds[1:] + self.x_bds[:-1]
337  
338                  self._call = self.__class__._call_nearest
339              elif kind == 'previous':
340                  # Side for np.searchsorted and index for clipping
341                  self._side = 'left'
342                  self._ind = 0
343                  # Move x by one floating point value to the left
344                  self._x_shift = np.nextafter(self.x, -np.inf)
345                  self._call = self.__class__._call_previousnext
346                  if _do_extrapolate(fill_value):
347                      self._check_and_update_bounds_error_for_extrapolation()
348                      # assume y is sorted by x ascending order here.
349                      fill_value = (np.nan, np.take(self.y, -1, axis))
350              elif kind == 'next':
351                  self._side = 'right'
352                  self._ind = 1
353                  # Move x by one floating point value to the right
354                  self._x_shift = np.nextafter(self.x, np.inf)
355                  self._call = self.__class__._call_previousnext
356                  if _do_extrapolate(fill_value):
357                      self._check_and_update_bounds_error_for_extrapolation()
358                      # assume y is sorted by x ascending order here.
359                      fill_value = (np.take(self.y, 0, axis), np.nan)
360              else:
361                  # Check if we can delegate to numpy.interp (2x-10x faster).
362                  np_dtypes = (np.dtype(np.float64), np.dtype(int))
363                  cond = self.x.dtype in np_dtypes and self.y.dtype in np_dtypes
364                  cond = cond and self.y.ndim == 1
365                  cond = cond and not _do_extrapolate(fill_value)
366  
367                  if cond:
368                      self._call = self.__class__._call_linear_np
369                  else:
370                      self._call = self.__class__._call_linear
371          else:
372              minval = order + 1
373  
374              rewrite_nan = False
375              xx, yy = self.x, self._y
376              if order > 1:
377                  # Quadratic or cubic spline. If input contains even a single
378                  # nan, then the output is all nans. We cannot just feed data
379                  # with nans to make_interp_spline because it calls LAPACK.
380                  # So, we make up a bogus x and y with no nans and use it
381                  # to get the correct shape of the output, which we then fill
382                  # with nans.
383                  # For slinear or zero order spline, we just pass nans through.
384                  mask = np.isnan(self.x)
385                  if mask.any():
386                      sx = self.x[~mask]
387                      if sx.size == 0:
388                          raise ValueError("`x` array is all-nan")
389                      xx = np.linspace(np.nanmin(self.x),
390                                       np.nanmax(self.x),
391                                       len(self.x))
392                      rewrite_nan = True
393                  if np.isnan(self._y).any():
394                      yy = np.ones_like(self._y)
395                      rewrite_nan = True
396  
397              self._spline = make_interp_spline(xx, yy, k=order,
398                                                check_finite=False)
399              if rewrite_nan:
400                  self._call = self.__class__._call_nan_spline
401              else:
402                  self._call = self.__class__._call_spline
403  
404          if len(self.x) < minval:
405              raise ValueError("x and y arrays must have at "
406                               "least %d entries" % minval)
407  
408          self.fill_value = fill_value  # calls the setter, can modify bounds_err

416      @fill_value.setter
417      def fill_value(self, fill_value):
418          # extrapolation only works for nearest neighbor and linear methods
419          if _do_extrapolate(fill_value):
420              self._check_and_update_bounds_error_for_extrapolation()
421              self._extrapolate = True
422          else:
423              broadcast_shape = (self.y.shape[:self.axis] +
424                                 self.y.shape[self.axis + 1:])
425              if len(broadcast_shape) == 0:
426                  broadcast_shape = (1,)
427              # it's either a pair (_below_range, _above_range) or a single value
428              # for both above and below range
429              if isinstance(fill_value, tuple) and len(fill_value) == 2:
430                  below_above = [np.asarray(fill_value[0]),
431                                 np.asarray(fill_value[1])]
432                  names = ('fill_value (below)', 'fill_value (above)')
433                  for ii in range(2):
434                      below_above[ii] = _check_broadcast_up_to(
435                          below_above[ii], broadcast_shape, names[ii])
436              else:
437                  fill_value = np.asarray(fill_value)
438                  below_above = [_check_broadcast_up_to(
439                      fill_value, broadcast_shape, 'fill_value')] * 2
440              self._fill_value_below, self._fill_value_above = below_above
441              self._extrapolate = False
442              if self.bounds_error is None:
443                  self.bounds_error = True
444          # backwards compat: fill_value was a public attr; make it writeable
445          self._fill_value_orig = fill_value

457      def _call_linear(self, x_new):
458          # 2. Find where in the original data, the values to interpolate
459          #    would be inserted.
460          #    Note: If x_new[n] == x[m], then m is returned by searchsorted.
461          x_new_indices = searchsorted(self.x, x_new)
462  
463          # 3. Clip x_new_indices so that they are within the range of
464          #    self.x indices and at least 1. Removes mis-interpolation
465          #    of x_new[n] = x[0]
466          x_new_indices = x_new_indices.clip(1, len(self.x)-1).astype(int)
467  
468          # 4. Calculate the slope of regions that each x_new value falls in.
469          lo = x_new_indices - 1
470          hi = x_new_indices
471  
472          x_lo = self.x[lo]
473          x_hi = self.x[hi]
474          y_lo = self._y[lo]
475          y_hi = self._y[hi]
476  
477          # Note that the following two expressions rely on the specifics of the
478          # broadcasting semantics.
479          slope = (y_hi - y_lo) / (x_hi - x_lo)[:, None]
480  
481          # 5. Calculate the actual value for each entry in x_new.
482          y_new = slope*(x_new - x_lo)[:, None] + y_lo
483  
484          return y_new

486      def _call_nearest(self, x_new):
487          """ Find nearest neighbor interpolated y_new = f(x_new)."""
488  
489          # 2. Find where in the averaged data the values to interpolate
490          #    would be inserted.
491          #    Note: use side='left' (right) to searchsorted() to define the
492          #    halfway point to be nearest to the left (right) neighbor
493          x_new_indices = searchsorted(self.x_bds, x_new, side=self._side)
494  
495          # 3. Clip x_new_indices so that they are within the range of x indices.
496          x_new_indices = x_new_indices.clip(0, len(self.x)-1).astype(intp)
497  
498          # 4. Calculate the actual value for each entry in x_new.
499          y_new = self._y[x_new_indices]
500  
501          return y_new

503      def _call_previousnext(self, x_new):
504          """Use previous/next neighbor of x_new, y_new = f(x_new)."""
505  
506          # 1. Get index of left/right value
507          x_new_indices = searchsorted(self._x_shift, x_new, side=self._side)
508  
509          # 2. Clip x_new_indices so that they are within the range of x indices.
510          x_new_indices = x_new_indices.clip(1-self._ind,
511                                             len(self.x)-self._ind).astype(intp)
512  
513          # 3. Calculate the actual value for each entry in x_new.
514          y_new = self._y[x_new_indices+self._ind-1]
515  
516          return y_new

518      def _call_spline(self, x_new):
519          return self._spline(x_new)

521      def _call_nan_spline(self, x_new):

526      def _evaluate(self, x_new):
527          # 1. Handle values in x_new that are outside of x. Throw error,
528          #    or return a list of mask array indicating the outofbounds values.
529          #    The behavior is set by the bounds_error variable.
530          x_new = asarray(x_new)
531          y_new = self._call(self, x_new)
532          if not self._extrapolate:
533              below_bounds, above_bounds = self._check_bounds(x_new)
534              if len(y_new) > 0:
535                  # Note fill_value must be broadcast up to the proper size
536                  # and flattened to work here
537                  y_new[below_bounds] = self._fill_value_below
538                  y_new[above_bounds] = self._fill_value_above
539          return y_new

541      def _check_bounds(self, x_new):
542          """Check the inputs for being in the bounds of the interpolated data. 
543   
544          Parameters 
545          ---------- 
546          x_new : array 
547   
548          Returns 
549          ------- 
550          out_of_bounds : bool array 
551              The mask on x_new of values that are out of the bounds. 
552          """
553  
554          # If self.bounds_error is True, we raise an error if any x_new values
555          # fall outside the range of x. Otherwise, we return an array indicating
556          # which values are outside the boundary region.
557          below_bounds = x_new < self.x[0]
558          above_bounds = x_new > self.x[-1]
559  
560          if self.bounds_error and below_bounds.any():
561              below_bounds_value = x_new[np.argmax(below_bounds)]
562              raise ValueError(f"A value ({below_bounds_value}) in x_new is below "
563                               f"the interpolation range's minimum value ({self.x[0]}).")
564          if self.bounds_error and above_bounds.any():
565              above_bounds_value = x_new[np.argmax(above_bounds)]
566              raise ValueError(f"A value ({above_bounds_value}) in x_new is above "
567                               f"the interpolation range's maximum value ({self.x[-1]}).")
568  
569          # !! Should we emit a warning if some values are out of bounds?
570          # !! matlab does not.
571          return below_bounds, above_bounds

590                               "2-dimensional.")

592          if not (0 <= axis < self.c.ndim - 1):
593              raise ValueError(f"axis={axis} must be between 0 and {self.c.ndim-1}")
594  
595          self.axis = axis
596          if axis != 0:
597              # move the interpolation axis to be the first one in self.c
598              # More specifically, the target shape for self.c is (k, m, ...),
599              # and axis !=0 means that we have c.shape (..., k, m, ...)
600              #                                               ^
601              #                                              axis
602              # So we roll two of them.
603              self.c = np.moveaxis(self.c, axis+1, 0)
604              self.c = np.moveaxis(self.c, axis+1, 0)
605  
606          if self.x.ndim != 1:
607              raise ValueError("x must be 1-dimensional")
608          if self.x.size < 2:
609              raise ValueError("at least 2 breakpoints are needed")
610          if self.c.ndim < 2:
611              raise ValueError("c must have at least 2 dimensions")
612          if self.c.shape[0] == 0:
613              raise ValueError("polynomial must be at least of order 0")
614          if self.c.shape[1] != self.x.size-1:
615              raise ValueError("number of coefficients != len(x)-1")
616          dx = np.diff(self.x)
617          if not (np.all(dx >= 0) or np.all(dx <= 0)):
618              raise ValueError("`x` must be strictly increasing or decreasing.")
619  
620          dtype = self._get_dtype(self.c.dtype)
621          self.c = np.ascontiguousarray(self.c, dtype=dtype)
622  
623      def _get_dtype(self, dtype):
624          if np.issubdtype(dtype, np.complexfloating) \
625                 or np.issubdtype(self.c.dtype, np.complexfloating):
626              return np.complex128
627          else:
628              return np.float64

630      @classmethod
631      def construct_fast(cls, c, x, extrapolate=None, axis=0):
632          """ 
633          Construct the piecewise polynomial without making checks. 
634   
635          Takes the same parameters as the constructor. Input arguments 
636          ``c`` and ``x`` must be arrays of the correct shape and type. The 
637          ``c`` array can only be of dtypes float and complex, and ``x`` 
638          array must have dtype float. 
639          """
640          self = object.__new__(cls)
641          self.c = c
642          self.x = x
643          self.axis = axis
644          if extrapolate is None:
645              extrapolate = True
646          self.extrapolate = extrapolate
647          return self
648  
649      def _ensure_c_contiguous(self):
650          """ 
651          c and x may be modified by the user. The Cython code expects 
652          that they are C contiguous. 
653          """
654          if not self.x.flags.c_contiguous:
655              self.x = self.x.copy()
656          if not self.c.flags.c_contiguous:
657              self.c = self.c.copy()

659      def extend(self, c, x):
660          """ 
661          Add additional breakpoints and coefficients to the polynomial. 
662   
663          Parameters 
664          ---------- 
665          c : ndarray, size (k, m, ...) 
666              Additional coefficients for polynomials in intervals. Note that 
667              the first additional interval will be formed using one of the 
668              ``self.x`` end points. 
669          x : ndarray, size (m,) 
670              Additional breakpoints. Must be sorted in the same order as 
671              ``self.x`` and either to the right or to the left of the current 
672              breakpoints. 
673   
674          Notes 
675          ----- 
676          This method is not thread safe and must not be executed concurrently 
677          with other methods available in this class. Doing so may cause 
678          unexpected errors or numerical output mismatches. 
679          """
680  
681          c = np.asarray(c)
682          x = np.asarray(x)
683  
684          if c.ndim < 2:
685              raise ValueError("invalid dimensions for c")
686          if x.ndim != 1:
687              raise ValueError("invalid dimensions for x")
688          if x.shape[0] != c.shape[1]:
689              raise ValueError(f"Shapes of x {x.shape} and c {c.shape} are incompatible")
690          if c.shape[2:] != self.c.shape[2:] or c.ndim != self.c.ndim:
691              raise ValueError(
692                  f"Shapes of c {c.shape} and self.c {self.c.shape} are incompatible"
693              )
694  
695          if c.size == 0:
696              return
697  
698          dx = np.diff(x)
699          if not (np.all(dx >= 0) or np.all(dx <= 0)):
700              raise ValueError("`x` is not sorted.")
701  
702          if self.x[-1] >= self.x[0]:
703              if not x[-1] >= x[0]:
704                  raise ValueError("`x` is in the different order "
705                                  "than `self.x`.")
706  
707              if x[0] >= self.x[-1]:
708                  action = 'append'
709              elif x[-1] <= self.x[0]:
710                  action = 'prepend'
711              else:
712                  raise ValueError("`x` is neither on the left or on the right "
713                                  "from `self.x`.")
714          else:
715              if not x[-1] <= x[0]:
716                  raise ValueError("`x` is in the different order "
717                                  "than `self.x`.")
718  
719              if x[0] <= self.x[-1]:
720                  action = 'append'
721              elif x[-1] >= self.x[0]:
722                  action = 'prepend'
723              else:
724                  raise ValueError("`x` is neither on the left or on the right "
725                                  "from `self.x`.")
726  
727          dtype = self._get_dtype(c.dtype)
728  
729          k2 = max(c.shape[0], self.c.shape[0])
730          c2 = np.zeros((k2, self.c.shape[1] + c.shape[1]) + self.c.shape[2:],
731                      dtype=dtype)
732  
733          if action == 'append':
734              c2[k2-self.c.shape[0]:, :self.c.shape[1]] = self.c
735              c2[k2-c.shape[0]:, self.c.shape[1]:] = c
736              self.x = np.r_[self.x, x]
737          elif action == 'prepend':
738              c2[k2-self.c.shape[0]:, :c.shape[1]] = c
739              c2[k2-c.shape[0]:, c.shape[1]:] = self.c
740              self.x = np.r_[x, self.x]
741  
742          self.c = c2
743  
744      def __call__(self, x, nu=0, extrapolate=None):
745          """ 
746          Evaluate the piecewise polynomial or its derivative. 
747   
748          Parameters 
749          ---------- 
750          x : array_like 
751              Points to evaluate the interpolant at. 
752          nu : int, optional 
753              Order of derivative to evaluate. Must be non-negative. 
754          extrapolate : {bool, 'periodic', None}, optional 
755              If bool, determines whether to extrapolate to out-of-bounds points 
756              based on first and last intervals, or to return NaNs. 
757              If 'periodic', periodic extrapolation is used. 
758              If None (default), use `self.extrapolate`. 
759   
760          Returns 
761          ------- 
762          y : array_like 
763              Interpolated values. Shape is determined by replacing 
764              the interpolation axis in the original array with the shape of x. 
765   
766          Notes 
767          ----- 
768          Derivatives are evaluated piecewise for each polynomial 
769          segment, even if the polynomial is not differentiable at the 
770          breakpoints. The polynomial intervals are considered half-open, 
771          ``[a, b)``, except for the last interval which is closed 
772          ``[a, b]``. 
773          """
774          if extrapolate is None:
775              extrapolate = self.extrapolate
776          x = np.asarray(x)
777          x_shape, x_ndim = x.shape, x.ndim
778          x = np.ascontiguousarray(x.ravel(), dtype=np.float64)
779  
780          # With periodic extrapolation we map x to the segment
781          # [self.x[0], self.x[-1]].
782          if extrapolate == 'periodic':
783              x = self.x[0] + (x - self.x[0]) % (self.x[-1] - self.x[0])
784              extrapolate = False
785  
786          out = np.empty((len(x), prod(self.c.shape[2:])), dtype=self.c.dtype)
787          self._ensure_c_contiguous()
788          self._evaluate(x, nu, extrapolate, out)
789          out = out.reshape(x_shape + self.c.shape[2:])
790          if self.axis != 0:
791              # transpose to move the calculated values to the interpolation axis
792              l = list(range(out.ndim))
793              l = l[x_ndim:x_ndim+self.axis] + l[:x_ndim] + l[x_ndim+self.axis:]
794              out = out.transpose(l)
795          return out

798  class PPoly(_PPolyBase):
799      """ 
800      Piecewise polynomial in terms of coefficients and breakpoints 
801   
802      The polynomial between ``x[i]`` and ``x[i + 1]`` is written in the 
803      local power basis:: 
804   
805          S = sum(c[m, i] * (xp - x[i])**(k-m) for m in range(k+1)) 
806   
807      where ``k`` is the degree of the polynomial. 
808   
809      Parameters 
810      ---------- 
811      c : ndarray, shape (k, m, ...) 
812          Polynomial coefficients, order `k` and `m` intervals. 
813      x : ndarray, shape (m+1,) 
814          Polynomial breakpoints. Must be sorted in either increasing or 
815          decreasing order. 
816      extrapolate : bool or 'periodic', optional 
817          If bool, determines whether to extrapolate to out-of-bounds points 
818          based on first and last intervals, or to return NaNs. If 'periodic', 
819          periodic extrapolation is used. Default is True. 
820      axis : int, optional 
821          Interpolation axis. Default is zero. 
822   
823      Attributes 
824      ---------- 
825      x : ndarray 
826          Breakpoints. 
827      c : ndarray 
828          Coefficients of the polynomials. They are reshaped 
829          to a 3-D array with the last dimension representing 
830          the trailing dimensions of the original coefficient array. 
831      axis : int 
832          Interpolation axis. 
833   
834      Methods 
835      ------- 
836      __call__ 
837      derivative 
838      antiderivative 
839      integrate 
840      solve 
841      roots 
842      extend 
843      from_spline 
844      from_bernstein_basis 
845      construct_fast 
846   
847      See also 
848      -------- 
849      BPoly : piecewise polynomials in the Bernstein basis 
850   
851      Notes 
852      ----- 
853      High-order polynomials in the power basis can be numerically 
854      unstable. Precision problems can start to appear for orders 
855      larger than 20-30. 
856      """
857  
858      def _evaluate(self, x, nu, extrapolate, out):
859          _ppoly.evaluate(self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
860                          self.x, x, nu, bool(extrapolate), out)
861  
862      def derivative(self, nu=1):
863          """ 
864          Construct a new piecewise polynomial representing the derivative. 
865   
866          Parameters 
867          ---------- 
868          nu : int, optional 
869              Order of derivative to evaluate. Default is 1, i.e., compute the 
870              first derivative. If negative, the antiderivative is returned. 
871   
872          Returns 
873          ------- 
874          pp : PPoly 
875              Piecewise polynomial of order k2 = k - n representing the derivative 
876              of this polynomial. 
877   
878          Notes 
879          ----- 
880          Derivatives are evaluated piecewise for each polynomial 
881          segment, even if the polynomial is not differentiable at the 
882          breakpoints. The polynomial intervals are considered half-open, 
883          ``[a, b)``, except for the last interval which is closed 
884          ``[a, b]``. 
885          """
886          if nu < 0:
887              return self.antiderivative(-nu)
888  
889          # reduce order
890          if nu == 0:
891              c2 = self.c.copy()
892          else:
893              c2 = self.c[:-nu, :].copy()
894  
895          if c2.shape[0] == 0:
896              # derivative of order 0 is zero
897              c2 = np.zeros((1,) + c2.shape[1:], dtype=c2.dtype)
898  
899          # multiply by the correct rising factorials
900          factor = spec.poch(np.arange(c2.shape[0], 0, -1), nu)
901          c2 *= factor[(slice(None),) + (None,)*(c2.ndim-1)]
902  
903          # construct a compatible polynomial
904          return self.construct_fast(c2, self.x, self.extrapolate, self.axis)
905  
906      def antiderivative(self, nu=1):
907          """ 
908          Construct a new piecewise polynomial representing the antiderivative. 
909   
910          Antiderivative is also the indefinite integral of the function, 
911          and derivative is its inverse operation. 
912   
913          Parameters 
914          ---------- 
915          nu : int, optional 
916              Order of antiderivative to evaluate. Default is 1, i.e., compute 
917              the first integral. If negative, the derivative is returned. 
918   
919          Returns 
920          ------- 
921          pp : PPoly 
922              Piecewise polynomial of order k2 = k + n representing 
923              the antiderivative of this polynomial. 
924   
925          Notes 
926          ----- 
927          The antiderivative returned by this function is continuous and 
928          continuously differentiable to order n-1, up to floating point 
929          rounding error. 
930   
931          If antiderivative is computed and ``self.extrapolate='periodic'``, 
932          it will be set to False for the returned instance. This is done because 
933          the antiderivative is no longer periodic and its correct evaluation 
934          outside of the initially given x interval is difficult. 
935          """
936          if nu <= 0:
937              return self.derivative(-nu)
938  
939          c = np.zeros((self.c.shape[0] + nu, self.c.shape[1]) + self.c.shape[2:],
940                       dtype=self.c.dtype)
941          c[:-nu] = self.c
942  
943          # divide by the correct rising factorials
944          factor = spec.poch(np.arange(self.c.shape[0], 0, -1), nu)
945          c[:-nu] /= factor[(slice(None),) + (None,)*(c.ndim-1)]
946  
947          # fix continuity of added degrees of freedom
948          self._ensure_c_contiguous()
949          _ppoly.fix_continuity(c.reshape(c.shape[0], c.shape[1], -1),
950                                self.x, nu - 1)
951  
952          if self.extrapolate == 'periodic':
953              extrapolate = False
954          else:
955              extrapolate = self.extrapolate
956  
957          # construct a compatible polynomial
958          return self.construct_fast(c, self.x, extrapolate, self.axis)

960      def integrate(self, a, b, extrapolate=None):
961          """ 
962          Compute a definite integral over a piecewise polynomial. 
963   
964          Parameters 
965          ---------- 
966          a : float 
967              Lower integration bound 
968          b : float 
969              Upper integration bound 
970          extrapolate : {bool, 'periodic', None}, optional 
971              If bool, determines whether to extrapolate to out-of-bounds points 
972              based on first and last intervals, or to return NaNs. 
973              If 'periodic', periodic extrapolation is used. 
974              If None (default), use `self.extrapolate`. 
975   
976          Returns 
977          ------- 
978          ig : array_like 
979              Definite integral of the piecewise polynomial over [a, b] 
980          """
981          if extrapolate is None:
982              extrapolate = self.extrapolate
983  
984          # Swap integration bounds if needed
985          sign = 1
986          if b < a:
987              a, b = b, a
988              sign = -1
989  
990          range_int = np.empty((prod(self.c.shape[2:]),), dtype=self.c.dtype)
991          self._ensure_c_contiguous()
992  
993          # Compute the integral.
994          if extrapolate == 'periodic':
995              # Split the integral into the part over period (can be several
996              # of them) and the remaining part.
997  
998              xs, xe = self.x[0], self.x[-1]
999              period = xe - xs
1000             interval = b - a
1001             n_periods, left = divmod(interval, period)
1002 
1003             if n_periods > 0:
1004                 _ppoly.integrate(
1005                     self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1006                     self.x, xs, xe, False, out=range_int)
1007                 range_int *= n_periods
1008             else:
1009                 range_int.fill(0)
1010 
1011             # Map a to [xs, xe], b is always a + left.
1012             a = xs + (a - xs) % period
1013             b = a + left
1014 
1015             # If b <= xe then we need to integrate over [a, b], otherwise
1016             # over [a, xe] and from xs to what is remained.
1017             remainder_int = np.empty_like(range_int)
1018             if b <= xe:
1019                 _ppoly.integrate(
1020                     self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1021                     self.x, a, b, False, out=remainder_int)
1022                 range_int += remainder_int
1023             else:
1024                 _ppoly.integrate(
1025                     self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1026                     self.x, a, xe, False, out=remainder_int)
1027                 range_int += remainder_int
1028 
1029                 _ppoly.integrate(
1030                     self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1031                     self.x, xs, xs + left + a - xe, False, out=remainder_int)
1032                 range_int += remainder_int
1033         else:
1034             _ppoly.integrate(
1035                 self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1036                 self.x, a, b, bool(extrapolate), out=range_int)
1037 
1038         # Return
1039         range_int *= sign
1040         return range_int.reshape(self.c.shape[2:])
1041 
1042     def solve(self, y=0., discontinuity=True, extrapolate=None):
1043         """ 
1044         Find real solutions of the equation ``pp(x) == y``. 
1045  
1046         Parameters 
1047         ---------- 
1048         y : float, optional 
1049             Right-hand side. Default is zero. 
1050         discontinuity : bool, optional 
1051             Whether to report sign changes across discontinuities at 
1052             breakpoints as roots. 
1053         extrapolate : {bool, 'periodic', None}, optional 
1054             If bool, determines whether to return roots from the polynomial 
1055             extrapolated based on first and last intervals, 'periodic' works 
1056             the same as False. If None (default), use `self.extrapolate`. 
1057  
1058         Returns 
1059         ------- 
1060         roots : ndarray 
1061             Roots of the polynomial(s). 
1062  
1063             If the PPoly object describes multiple polynomials, the 
1064             return value is an object array whose each element is an 
1065             ndarray containing the roots. 
1066  
1067         Notes 
1068         ----- 
1069         This routine works only on real-valued polynomials. 
1070  
1071         If the piecewise polynomial contains sections that are 
1072         identically zero, the root list will contain the start point 
1073         of the corresponding interval, followed by a ``nan`` value. 
1074  
1075         If the polynomial is discontinuous across a breakpoint, and 
1076         there is a sign change across the breakpoint, this is reported 
1077         if the `discont` parameter is True. 
1078  
1079         Examples 
1080         -------- 
1081  
1082         Finding roots of ``[x**2 - 1, (x - 1)**2]`` defined on intervals 
1083         ``[-2, 1], [1, 2]``: 
1084  
1085         >>> import numpy as np 
1086         >>> from scipy.interpolate import PPoly 
1087         >>> pp = PPoly(np.array([[1, -4, 3], [1, 0, 0]]).T, [-2, 1, 2]) 
1088         >>> pp.solve() 
1089         array([-1.,  1.]) 
1090         """
1091         if extrapolate is None:
1092             extrapolate = self.extrapolate
1093 
1094         self._ensure_c_contiguous()
1095 
1096         if np.issubdtype(self.c.dtype, np.complexfloating):
1097             raise ValueError("Root finding is only for "
1098                              "real-valued polynomials")
1099 
1100         y = float(y)
1101         r = _ppoly.real_roots(self.c.reshape(self.c.shape[0], self.c.shape[1], -1),
1102                               self.x, y, bool(discontinuity),
1103                               bool(extrapolate))
1104         if self.c.ndim == 2:
1105             return r[0]
1106         else:
1107             r2 = np.empty(prod(self.c.shape[2:]), dtype=object)
1108             # this for-loop is equivalent to ``r2[...] = r``, but that's broken
1109             # in NumPy 1.6.0
1110             for ii, root in enumerate(r):
1111                 r2[ii] = root
1112 
1113             return r2.reshape(self.c.shape[2:])

1145     def from_spline(cls, tck, extrapolate=None):
1146         """ 
1147         Construct a piecewise polynomial from a spline 
1148  
1149         Parameters 
1150         ---------- 
1151         tck 
1152             A spline, as returned by `splrep` or a BSpline object. 
1153         extrapolate : bool or 'periodic', optional 
1154             If bool, determines whether to extrapolate to out-of-bounds points 
1155             based on first and last intervals, or to return NaNs. 
1156             If 'periodic', periodic extrapolation is used. Default is True. 
1157  
1158         Examples 
1159         -------- 
1160         Construct an interpolating spline and convert it to a `PPoly` instance 
1161  
1162         >>> import numpy as np 
1163         >>> from scipy.interpolate import splrep, PPoly 
1164         >>> x = np.linspace(0, 1, 11) 
1165         >>> y = np.sin(2*np.pi*x) 
1166         >>> tck = splrep(x, y, s=0) 
1167         >>> p = PPoly.from_spline(tck) 
1168         >>> isinstance(p, PPoly) 
1169         True 
1170  
1171         Note that this function only supports 1D splines out of the box. 
1172  
1173         If the ``tck`` object represents a parametric spline (e.g. constructed 
1174         by `splprep` or a `BSpline` with ``c.ndim > 1``), you will need to loop 
1175         over the dimensions manually. 
1176  
1177         >>> from scipy.interpolate import splprep, splev 
1178         >>> t = np.linspace(0, 1, 11) 
1179         >>> x = np.sin(2*np.pi*t) 
1180         >>> y = np.cos(2*np.pi*t) 
1181         >>> (t, c, k), u = splprep([x, y], s=0) 
1182  
1183         Note that ``c`` is a list of two arrays of length 11. 
1184  
1185         >>> unew = np.arange(0, 1.01, 0.01) 
1186         >>> out = splev(unew, (t, c, k)) 
1187  
1188         To convert this spline to the power basis, we convert each 
1189         component of the list of b-spline coefficients, ``c``, into the 
1190         corresponding cubic polynomial. 
1191  
1192         >>> polys = [PPoly.from_spline((t, cj, k)) for cj in c] 
1193         >>> polys[0].c.shape 
1194         (4, 14) 
1195  
1196         Note that the coefficients of the polynomials `polys` are in the 
1197         power basis and their dimensions reflect just that: here 4 is the order 
1198         (degree+1), and 14 is the number of intervals---which is nothing but 
1199         the length of the knot array of the original `tck` minus one. 
1200  
1201         Optionally, we can stack the components into a single `PPoly` along 
1202         the third dimension: 
1203  
1204         >>> cc = np.dstack([p.c for p in polys])    # has shape = (4, 14, 2) 
1205         >>> poly = PPoly(cc, polys[0].x) 
1206         >>> np.allclose(poly(unew).T,     # note the transpose to match `splev` 
1207         ...             out, atol=1e-15) 
1208         True 
1209  
1210         """
1211         if isinstance(tck, BSpline):
1212             t, c, k = tck.tck
1213             if extrapolate is None:
1214                 extrapolate = tck.extrapolate
1215         else:
1216             t, c, k = tck
1217 
1218         cvals = np.empty((k + 1, len(t)-1), dtype=c.dtype)
1219         for m in range(k, -1, -1):
1220             y = _fitpack_py.splev(t[:-1], tck, der=m)
1221             cvals[k - m, :] = y/spec.gamma(m+1)
1222 
1223         return cls.construct_fast(cvals, t, extrapolate)

1226     def from_bernstein_basis(cls, bp, extrapolate=None):
1227         """ 
1228         Construct a piecewise polynomial in the power basis 
1229         from a polynomial in Bernstein basis. 
1230  
1231         Parameters 
1232         ---------- 
1233         bp : BPoly 
1234             A Bernstein basis polynomial, as created by BPoly 
1235         extrapolate : bool or 'periodic', optional 
1236             If bool, determines whether to extrapolate to out-of-bounds points 
1237             based on first and last intervals, or to return NaNs. 
1238             If 'periodic', periodic extrapolation is used. Default is True. 
1239         """
1240         if not isinstance(bp, BPoly):
1241             raise TypeError(f".from_bernstein_basis only accepts BPoly instances. "
1242                             f"Got {type(bp)} instead.")
1243 
1244         dx = np.diff(bp.x)
1245         k = bp.c.shape[0] - 1  # polynomial order
1246 
1247         rest = (None,)*(bp.c.ndim-2)
1248 
1249         c = np.zeros_like(bp.c)
1250         for a in range(k+1):
1251             factor = (-1)**a * comb(k, a) * bp.c[a]
1252             for s in range(a, k+1):
1253                 val = comb(k-a, s-a) * (-1)**s
1254                 c[k-s] += factor * val / dx[(slice(None),)+rest]**s
1255 
1256         if extrapolate is None:
1257             extrapolate = bp.extrapolate
1258 
1259         return cls.construct_fast(c, bp.x, extrapolate, bp.axis)

1264

1287         periodic extrapolation is used. Default is True.

1354     def derivative(self, nu=1):
1355         """ 
1356         Construct a new piecewise polynomial representing the derivative. 
1357  
1358         Parameters 
1359         ---------- 
1360         nu : int, optional 
1361             Order of derivative to evaluate. Default is 1, i.e., compute the 
1362             first derivative. If negative, the antiderivative is returned. 
1363  
1364         Returns 
1365         ------- 
1366         bp : BPoly 
1367             Piecewise polynomial of order k - nu representing the derivative of 
1368             this polynomial. 
1369  
1370         """
1371         if nu < 0:
1372             return self.antiderivative(-nu)
1373 
1374         if nu > 1:
1375             bp = self
1376             for k in range(nu):
1377                 bp = bp.derivative()
1378             return bp
1379 
1380         # reduce order
1381         if nu == 0:
1382             c2 = self.c.copy()
1383         else:
1384             # For a polynomial
1385             #    B(x) = \sum_{a=0}^{k} c_a b_{a, k}(x),
1386             # we use the fact that
1387             #   b'_{a, k} = k ( b_{a-1, k-1} - b_{a, k-1} ),
1388             # which leads to
1389             #   B'(x) = \sum_{a=0}^{k-1} (c_{a+1} - c_a) b_{a, k-1}
1390             #
1391             # finally, for an interval [y, y + dy] with dy != 1,
1392             # we need to correct for an extra power of dy
1393 
1394             rest = (None,)*(self.c.ndim-2)
1395 
1396             k = self.c.shape[0] - 1
1397             dx = np.diff(self.x)[(None, slice(None))+rest]
1398             c2 = k * np.diff(self.c, axis=0) / dx
1399 
1400         if c2.shape[0] == 0:
1401             # derivative of order 0 is zero
1402             c2 = np.zeros((1,) + c2.shape[1:], dtype=c2.dtype)
1403 
1404         # construct a compatible polynomial
1405         return self.construct_fast(c2, self.x, self.extrapolate, self.axis)

1407     def antiderivative(self, nu=1):
1408         """ 
1409         Construct a new piecewise polynomial representing the antiderivative. 
1410  
1411         Parameters 
1412         ---------- 
1413         nu : int, optional 
1414             Order of antiderivative to evaluate. Default is 1, i.e., compute 
1415             the first integral. If negative, the derivative is returned. 
1416  
1417         Returns 
1418         ------- 
1419         bp : BPoly 
1420             Piecewise polynomial of order k + nu representing the 
1421             antiderivative of this polynomial. 
1422  
1423         Notes 
1424         ----- 
1425         If antiderivative is computed and ``self.extrapolate='periodic'``, 
1426         it will be set to False for the returned instance. This is done because 
1427         the antiderivative is no longer periodic and its correct evaluation 
1428         outside of the initially given x interval is difficult. 
1429         """
1430         if nu <= 0:
1431             return self.derivative(-nu)
1432 
1433         if nu > 1:
1434             bp = self
1435             for k in range(nu):
1436                 bp = bp.antiderivative()
1437             return bp
1438 
1439         # Construct the indefinite integrals on individual intervals
1440         c, x = self.c, self.x
1441         k = c.shape[0]
1442         c2 = np.zeros((k+1,) + c.shape[1:], dtype=c.dtype)
1443 
1444         c2[1:, ...] = np.cumsum(c, axis=0) / k
1445         delta = x[1:] - x[:-1]
1446         c2 *= delta[(None, slice(None)) + (None,)*(c.ndim-2)]
1447 
1448         # Now fix continuity: on the very first interval, take the integration
1449         # constant to be zero; on an interval [x_j, x_{j+1}) with j>0,
1450         # the integration constant is then equal to the jump of the `bp` at x_j.
1451         # The latter is given by the coefficient of B_{n+1, n+1}
1452         # *on the previous interval* (other B. polynomials are zero at the
1453         # breakpoint). Finally, use the fact that BPs form a partition of unity.
1454         c2[:,1:] += np.cumsum(c2[k, :], axis=0)[:-1]
1455 
1456         if self.extrapolate == 'periodic':
1457             extrapolate = False
1458         else:
1459             extrapolate = self.extrapolate
1460 
1461         return self.construct_fast(c2, x, extrapolate, axis=self.axis)

1463     def integrate(self, a, b, extrapolate=None):
1464         """ 
1465         Compute a definite integral over a piecewise polynomial. 
1466  
1467         Parameters 
1468         ---------- 
1469         a : float 
1470             Lower integration bound 
1471         b : float 
1472             Upper integration bound 
1473         extrapolate : {bool, 'periodic', None}, optional 
1474             Whether to extrapolate to out-of-bounds points based on first 
1475             and last intervals, or to return NaNs. If 'periodic', periodic 
1476             extrapolation is used. If None (default), use `self.extrapolate`. 
1477  
1478         Returns 
1479         ------- 
1480         array_like 
1481             Definite integral of the piecewise polynomial over [a, b] 
1482  
1483         """
1484         # XXX: can probably use instead the fact that
1485         # \int_0^{1} B_{j, n}(x) \dx = 1/(n+1)
1486         ib = self.antiderivative()
1487         if extrapolate is None:
1488             extrapolate = self.extrapolate
1489 
1490         # ib.extrapolate shouldn't be 'periodic', it is converted to
1491         # False for 'periodic. in antiderivative() call.
1492         if extrapolate != 'periodic':
1493             ib.extrapolate = extrapolate
1494 
1495         if extrapolate == 'periodic':
1496             # Split the integral into the part over period (can be several
1497             # of them) and the remaining part.
1498 
1499             # For simplicity and clarity convert to a <= b case.
1500             if a <= b:
1501                 sign = 1
1502             else:
1503                 a, b = b, a
1504                 sign = -1
1505 
1506             xs, xe = self.x[0], self.x[-1]
1507             period = xe - xs
1508             interval = b - a
1509             n_periods, left = divmod(interval, period)
1510             res = n_periods * (ib(xe) - ib(xs))
1511 
1512             # Map a and b to [xs, xe].
1513             a = xs + (a - xs) % period
1514             b = a + left
1515 
1516             # If b <= xe then we need to integrate over [a, b], otherwise
1517             # over [a, xe] and from xs to what is remained.
1518             if b <= xe:
1519                 res += ib(b) - ib(a)
1520             else:
1521                 res += ib(xe) - ib(a) + ib(xs + left + a - xe) - ib(xs)
1522 
1523             return sign * res
1524         else:
1525             return ib(b) - ib(a)

1569     @classmethod
1570     def from_derivatives(cls, xi, yi, orders=None, extrapolate=None):
1571         """Construct a piecewise polynomial in the Bernstein basis, 
1572         compatible with the specified values and derivatives at breakpoints. 
1573  
1574         Parameters 
1575         ---------- 
1576         xi : array_like 
1577             sorted 1-D array of x-coordinates 
1578         yi : array_like or list of array_likes 
1579             ``yi[i][j]`` is the ``j``\\ th derivative known at ``xi[i]`` 
1580         orders : None or int or array_like of ints. Default: None. 
1581             Specifies the degree of local polynomials. If not None, some 
1582             derivatives are ignored. 
1583         extrapolate : bool or 'periodic', optional 
1584             If bool, determines whether to extrapolate to out-of-bounds points 
1585             based on first and last intervals, or to return NaNs. 
1586             If 'periodic', periodic extrapolation is used. Default is True. 
1587  
1588         Notes 
1589         ----- 
1590         If ``k`` derivatives are specified at a breakpoint ``x``, the 
1591         constructed polynomial is exactly ``k`` times continuously 
1592         differentiable at ``x``, unless the ``order`` is provided explicitly. 
1593         In the latter case, the smoothness of the polynomial at 
1594         the breakpoint is controlled by the ``order``. 
1595  
1596         Deduces the number of derivatives to match at each end 
1597         from ``order`` and the number of derivatives available. If 
1598         possible it uses the same number of derivatives from 
1599         each end; if the number is odd it tries to take the 
1600         extra one from y2. In any case if not enough derivatives 
1601         are available at one end or another it draws enough to 
1602         make up the total from the other end. 
1603  
1604         If the order is too high and not enough derivatives are available, 
1605         an exception is raised. 
1606  
1607         Examples 
1608         -------- 
1609  
1610         >>> from scipy.interpolate import BPoly 
1611         >>> BPoly.from_derivatives([0, 1], [[1, 2], [3, 4]]) 
1612  
1613         Creates a polynomial `f(x)` of degree 3, defined on ``[0, 1]`` 
1614         such that `f(0) = 1, df/dx(0) = 2, f(1) = 3, df/dx(1) = 4` 
1615  
1616         >>> BPoly.from_derivatives([0, 1, 2], [[0, 1], [0], [2]]) 
1617  
1618         Creates a piecewise polynomial `f(x)`, such that 
1619         `f(0) = f(1) = 0`, `f(2) = 2`, and `df/dx(0) = 1`. 
1620         Based on the number of derivatives provided, the order of the 
1621         local polynomials is 2 on ``[0, 1]`` and 1 on ``[1, 2]``. 
1622         Notice that no restriction is imposed on the derivatives at 
1623         ``x = 1`` and ``x = 2``. 
1624  
1625         Indeed, the explicit form of the polynomial is:: 
1626  
1627             f(x) = | x * (1 - x),  0 <= x < 1 
1628                    | 2 * (x - 1),  1 <= x <= 2 
1629  
1630         So that f'(1-0) = -1 and f'(1+0) = 2 
1631  
1632         """
1633         xi = np.asarray(xi)
1634         if len(xi) != len(yi):
1635             raise ValueError("xi and yi need to have the same length")
1636         if np.any(xi[1:] - xi[:1] <= 0):
1637             raise ValueError("x coordinates are not in increasing order")
1638 
1639         # number of intervals
1640         m = len(xi) - 1
1641 
1642         # global poly order is k-1, local orders are <=k and can vary
1643         try:
1644             k = max(len(yi[i]) + len(yi[i+1]) for i in range(m))
1645         except TypeError as e:
1646             raise ValueError(
1647                 "Using a 1-D array for y? Please .reshape(-1, 1)."
1648             ) from e
1649 
1650         if orders is None:
1651             orders = [None] * m
1652         else:
1653             if isinstance(orders, (int, np.integer)):
1654                 orders = [orders] * m
1655             k = max(k, max(orders))
1656 
1657             if any(o <= 0 for o in orders):
1658                 raise ValueError("Orders must be positive.")
1659 
1660         c = []
1661         for i in range(m):
1662             y1, y2 = yi[i], yi[i+1]
1663             if orders[i] is None:
1664                 n1, n2 = len(y1), len(y2)
1665             else:
1666                 n = orders[i]+1
1667                 n1 = min(n//2, len(y1))
1668                 n2 = min(n - n1, len(y2))
1669                 n1 = min(n - n2, len(y2))
1670                 if n1+n2 != n:
1671                     mesg = ("Point %g has %d derivatives, point %g"
1672                             " has %d derivatives, but order %d requested" % (
1673                                xi[i], len(y1), xi[i+1], len(y2), orders[i]))
1674                     raise ValueError(mesg)
1675 
1676                 if not (n1 <= len(y1) and n2 <= len(y2)):
1677                     raise ValueError("`order` input incompatible with"
1678                                      " length y1 or y2.")
1679 
1680             b = BPoly._construct_from_derivatives(xi[i], xi[i+1],
1681                                                   y1[:n1], y2[:n2])
1682             if len(b) < k:
1683                 b = BPoly._raise_degree(b, k - len(b))
1684             c.append(b)
1685 
1686         c = np.asarray(c)
1687         return cls(c.swapaxes(0, 1), xi, extrapolate)

1689     @staticmethod
1690     def _construct_from_derivatives(xa, xb, ya, yb):
1691         r"""Compute the coefficients of a polynomial in the Bernstein basis 
1692         given the values and derivatives at the edges. 
1693  
1694         Return the coefficients of a polynomial in the Bernstein basis 
1695         defined on ``[xa, xb]`` and having the values and derivatives at the 
1696         endpoints `xa` and `xb` as specified by `ya` and `yb`. 
1697         The polynomial constructed is of the minimal possible degree, i.e., 
1698         if the lengths of `ya` and `yb` are `na` and `nb`, the degree 
1699         of the polynomial is ``na + nb - 1``. 
1700  
1701         Parameters 
1702         ---------- 
1703         xa : float 
1704             Left-hand end point of the interval 
1705         xb : float 
1706             Right-hand end point of the interval 
1707         ya : array_like 
1708             Derivatives at `xa`. ``ya[0]`` is the value of the function, and 
1709             ``ya[i]`` for ``i > 0`` is the value of the ``i``\ th derivative. 
1710         yb : array_like 
1711             Derivatives at `xb`. 
1712  
1713         Returns 
1714         ------- 
1715         array 
1716             coefficient array of a polynomial having specified derivatives 
1717  
1718         Notes 
1719         ----- 
1720         This uses several facts from life of Bernstein basis functions. 
1721         First of all, 
1722  
1723             .. math:: b'_{a, n} = n (b_{a-1, n-1} - b_{a, n-1}) 
1724  
1725         If B(x) is a linear combination of the form 
1726  
1727             .. math:: B(x) = \sum_{a=0}^{n} c_a b_{a, n}, 
1728  
1729         then :math: B'(x) = n \sum_{a=0}^{n-1} (c_{a+1} - c_{a}) b_{a, n-1}. 
1730         Iterating the latter one, one finds for the q-th derivative 
1731  
1732             .. math:: B^{q}(x) = n!/(n-q)! \sum_{a=0}^{n-q} Q_a b_{a, n-q}, 
1733  
1734         with 
1735  
1736           .. math:: Q_a = \sum_{j=0}^{q} (-)^{j+q} comb(q, j) c_{j+a} 
1737  
1738         This way, only `a=0` contributes to :math: `B^{q}(x = xa)`, and 
1739         `c_q` are found one by one by iterating `q = 0, ..., na`. 
1740  
1741         At ``x = xb`` it's the same with ``a = n - q``. 
1742  
1743         """
1744         ya, yb = np.asarray(ya), np.asarray(yb)
1745         if ya.shape[1:] != yb.shape[1:]:
1746             raise ValueError(
1747                 f"Shapes of ya {ya.shape} and yb {yb.shape} are incompatible"
1748             )
1749 
1750         dta, dtb = ya.dtype, yb.dtype
1751         if (np.issubdtype(dta, np.complexfloating) or
1752                np.issubdtype(dtb, np.complexfloating)):
1753             dt = np.complex128
1754         else:
1755             dt = np.float64
1756 
1757         na, nb = len(ya), len(yb)
1758         n = na + nb
1759 
1760         c = np.empty((na+nb,) + ya.shape[1:], dtype=dt)
1761 
1762         # compute coefficients of a polynomial degree na+nb-1
1763         # walk left-to-right
1764         for q in range(0, na):
1765             c[q] = ya[q] / spec.poch(n - q, q) * (xb - xa)**q
1766             for j in range(0, q):
1767                 c[q] -= (-1)**(j+q) * comb(q, j) * c[j]
1768 
1769         # now walk right-to-left
1770         for q in range(0, nb):
1771             c[-q-1] = yb[q] / spec.poch(n - q, q) * (-1)**q * (xb - xa)**q
1772             for j in range(0, q):
1773                 c[-q-1] -= (-1)**(j+1) * comb(q, j+1) * c[-q+j]
1774 
1775         return c

1777     @staticmethod
1778     def _raise_degree(c, d):
1779         r"""Raise a degree of a polynomial in the Bernstein basis. 
1780  
1781         Given the coefficients of a polynomial degree `k`, return (the 
1782         coefficients of) the equivalent polynomial of degree `k+d`. 
1783  
1784         Parameters 
1785         ---------- 
1786         c : array_like 
1787             coefficient array, 1-D 
1788         d : integer 
1789  
1790         Returns 
1791         ------- 
1792         array 
1793             coefficient array, 1-D array of length `c.shape[0] + d` 
1794  
1795         Notes 
1796         ----- 
1797         This uses the fact that a Bernstein polynomial `b_{a, k}` can be 
1798         identically represented as a linear combination of polynomials of 
1799         a higher degree `k+d`: 
1800  
1801             .. math:: b_{a, k} = comb(k, a) \sum_{j=0}^{d} b_{a+j, k+d} \ 
1802                                  comb(d, j) / comb(k+d, a+j) 
1803  
1804         """
1805         if d == 0:
1806             return c
1807 
1808         k = c.shape[0] - 1
1809         out = np.zeros((c.shape[0] + d,) + c.shape[1:], dtype=c.dtype)
1810 
1811         for a in range(c.shape[0]):
1812             f = c[a] * comb(k, a)
1813             for j in range(d+1):
1814                 out[a+j] += f * comb(d, j) / comb(k+d, a+j)
1815         return out

1831         S = sum(c[k0-m0-1,...,kn-mn-1,i[0],...,i[n]] 
1832                 * (xp[0] - x[0][i[0]])**m0 
1833                 * ... 
1834                 * (xp[n] - x[n][i[n]])**mn 
1835                 for m0 in range(k[0]+1) 
1836                 ... 
1837                 for mn in range(k[n]+1)) 
1838  
1839     where ``k[j]`` is the degree of the polynomial in dimension j. This 
1840     representation is the piecewise multivariate power basis. 
1841  
1842     Parameters 
1843     ---------- 
1844     c : ndarray, shape (k0, ..., kn, m0, ..., mn, ...) 
1845         Polynomial coefficients, with polynomial order `kj` and 
1846         `mj+1` intervals for each dimension `j`. 
1847     x : ndim-tuple of ndarrays, shapes (mj+1,) 
1848         Polynomial breakpoints for each dimension. These must be 
1849         sorted in increasing order. 
1850     extrapolate : bool, optional 
1851         Whether to extrapolate to out-of-bounds points based on first 
1852         and last intervals, or to return NaNs. Default: True. 
1853  
1854     Attributes 
1855     ---------- 
1856     x : tuple of ndarrays 
1857         Breakpoints. 
1858     c : ndarray 
1859         Coefficients of the polynomials. 
1860  
1861     Methods 
1862     ------- 
1863     __call__ 
1864     derivative 
1865     antiderivative 
1866     integrate 
1867     integrate_1d 
1868     construct_fast 
1869  
1870     See also 
1871     -------- 
1872     PPoly : piecewise polynomials in 1D 
1873  
1874     Notes 
1875     ----- 
1876     High-order polynomials in the power basis can be numerically 
1877     unstable. 
1878  
1879     """

1881     def __init__(self, c, x, extrapolate=None):

1885             extrapolate = True

1935     def __call__(self, x, nu=None, extrapolate=None):
1936         """ 
1937         Evaluate the piecewise polynomial or its derivative 
1938  
1939         Parameters 
1940         ---------- 
1941         x : array-like 
1942             Points to evaluate the interpolant at. 
1943         nu : tuple, optional 
1944             Orders of derivatives to evaluate. Each must be non-negative. 
1945         extrapolate : bool, optional 
1946             Whether to extrapolate to out-of-bounds points based on first 
1947             and last intervals, or to return NaNs. 
1948  
1949         Returns 
1950         ------- 
1951         y : array-like 
1952             Interpolated values. Shape is determined by replacing 
1953             the interpolation axis in the original array with the shape of x. 
1954  
1955         Notes 
1956         ----- 
1957         Derivatives are evaluated piecewise for each polynomial 
1958         segment, even if the polynomial is not differentiable at the 
1959         breakpoints. The polynomial intervals are considered half-open, 
1960         ``[a, b)``, except for the last interval which is closed 
1961         ``[a, b]``. 
1962  
1963         """
1964         if extrapolate is None:
1965             extrapolate = self.extrapolate
1966         else:
1967             extrapolate = bool(extrapolate)
1968 
1969         ndim = len(self.x)
1970 
1971         x = _ndim_coords_from_arrays(x)
1972         x_shape = x.shape
1973         x = np.ascontiguousarray(x.reshape(-1, x.shape[-1]), dtype=np.float64)
1974 
1975         if nu is None:
1976             nu = np.zeros((ndim,), dtype=np.intc)
1977         else:
1978             nu = np.asarray(nu, dtype=np.intc)
1979             if nu.ndim != 1 or nu.shape[0] != ndim:
1980                 raise ValueError("invalid number of derivative orders nu")
1981 
1982         dim1 = prod(self.c.shape[:ndim])
1983         dim2 = prod(self.c.shape[ndim:2*ndim])
1984         dim3 = prod(self.c.shape[2*ndim:])
1985         ks = np.array(self.c.shape[:ndim], dtype=np.intc)
1986 
1987         out = np.empty((x.shape[0], dim3), dtype=self.c.dtype)
1988         self._ensure_c_contiguous()
1989 
1990         _ppoly.evaluate_nd(self.c.reshape(dim1, dim2, dim3),
1991                            self.x,
1992                            ks,
1993                            x,
1994                            nu,
1995                            bool(extrapolate),
1996                            out)
1997 
1998         return out.reshape(x_shape[:-1] + self.c.shape[2*ndim:])

2000     def _derivative_inplace(self, nu, axis):
2001         """ 
2002         Compute 1-D derivative along a selected dimension in-place 
2003         May result to non-contiguous c array. 
2004         """
2005         if nu < 0:
2006             return self._antiderivative_inplace(-nu, axis)
2007 
2008         ndim = len(self.x)
2009         axis = axis % ndim
2010 
2011         # reduce order
2012         if nu == 0:
2013             # noop
2014             return
2015         else:
2016             sl = [slice(None)]*ndim
2017             sl[axis] = slice(None, -nu, None)
2018             c2 = self.c[tuple(sl)]
2019 
2020         if c2.shape[axis] == 0:
2021             # derivative of order 0 is zero
2022             shp = list(c2.shape)
2023             shp[axis] = 1
2024             c2 = np.zeros(shp, dtype=c2.dtype)
2025 
2026         # multiply by the correct rising factorials
2027         factor = spec.poch(np.arange(c2.shape[axis], 0, -1), nu)
2028         sl = [None]*c2.ndim
2029         sl[axis] = slice(None)
2030         c2 *= factor[tuple(sl)]
2031 
2032         self.c = c2

2034     def _antiderivative_inplace(self, nu, axis):
2035         """ 
2036         Compute 1-D antiderivative along a selected dimension 
2037         May result to non-contiguous c array. 
2038         """
2039         if nu <= 0:
2040             return self._derivative_inplace(-nu, axis)
2041 
2042         ndim = len(self.x)
2043         axis = axis % ndim
2044 
2045         perm = list(range(ndim))
2046         perm[0], perm[axis] = perm[axis], perm[0]
2047         perm = perm + list(range(ndim, self.c.ndim))
2048 
2049         c = self.c.transpose(perm)
2050 
2051         c2 = np.zeros((c.shape[0] + nu,) + c.shape[1:],
2052                      dtype=c.dtype)
2053         c2[:-nu] = c
2054 
2055         # divide by the correct rising factorials
2056         factor = spec.poch(np.arange(c.shape[0], 0, -1), nu)
2057         c2[:-nu] /= factor[(slice(None),) + (None,)*(c.ndim-1)]
2058 
2059         # fix continuity of added degrees of freedom
2060         perm2 = list(range(c2.ndim))
2061         perm2[1], perm2[ndim+axis] = perm2[ndim+axis], perm2[1]
2062 
2063         c2 = c2.transpose(perm2)
2064         c2 = c2.copy()
2065         _ppoly.fix_continuity(c2.reshape(c2.shape[0], c2.shape[1], -1),
2066                               self.x[axis], nu-1)
2067 
2068         c2 = c2.transpose(perm2)
2069         c2 = c2.transpose(perm)
2070 
2071         # Done
2072         self.c = c2

2075         """ 
2076         Construct a new piecewise polynomial representing the derivative. 
2077  
2078         Parameters 
2079         ---------- 
2080         nu : ndim-tuple of int

2083

2085         -------

2171         """

2173             extrapolate = self.extrapolate
2174         else:
2175             extrapolate = bool(extrapolate)

2177         ndim = len(self.x)
2178         axis = int(axis) % ndim
