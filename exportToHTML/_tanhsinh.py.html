30   def tanhsinh(f, a, b, *, args=(), log=False, maxlevel=None, minlevel=2,
31                atol=None, rtol=None, preserve_shape=False, callback=None):
32       """Evaluate a convergent integral numerically using tanh-sinh quadrature. 
33    
34       In practice, tanh-sinh quadrature achieves quadratic convergence for 
35       many integrands: the number of accurate *digits* scales roughly linearly 
36       with the number of function evaluations [1]_. 
37    
38       Either or both of the limits of integration may be infinite, and 
39       singularities at the endpoints are acceptable. Divergent integrals and 
40       integrands with non-finite derivatives or singularities within an interval 
41       are out of scope, but the latter may be evaluated be calling `tanhsinh` on 
42       each sub-interval separately. 
43    
44       Parameters 
45       ---------- 
46       f : callable 
47           The function to be integrated. The signature must be:: 
48    
49               f(xi: ndarray, *argsi) -> ndarray 
50    
51           where each element of ``xi`` is a finite real number and ``argsi`` is a tuple, 
52           which may contain an arbitrary number of arrays that are broadcastable 
53           with ``xi``. `f` must be an elementwise function: see documentation of parameter 
54           `preserve_shape` for details. It must not mutate the array ``xi`` or the arrays 
55           in ``argsi``. 
56           If ``f`` returns a value with complex dtype when evaluated at 
57           either endpoint, subsequent arguments ``x`` will have complex dtype 
58           (but zero imaginary part). 
59       a, b : float array_like 
60           Real lower and upper limits of integration. Must be broadcastable with one 
61           another and with arrays in `args`. Elements may be infinite. 
62       args : tuple of array_like, optional 
63           Additional positional array arguments to be passed to `f`. Arrays 
64           must be broadcastable with one another and the arrays of `a` and `b`. 
65           If the callable for which the root is desired requires arguments that are 
66           not broadcastable with `x`, wrap that callable with `f` such that `f` 
67           accepts only `x` and broadcastable ``*args``. 
68       log : bool, default: False 
69           Setting to True indicates that `f` returns the log of the integrand 
70           and that `atol` and `rtol` are expressed as the logs of the absolute 
71           and relative errors. In this case, the result object will contain the 
72           log of the integral and error. This is useful for integrands for which 
73           numerical underflow or overflow would lead to inaccuracies. 
74           When ``log=True``, the integrand (the exponential of `f`) must be real, 
75           but it may be negative, in which case the log of the integrand is a 
76           complex number with an imaginary part that is an odd multiple of π. 
77       maxlevel : int, default: 10 
78           The maximum refinement level of the algorithm. 
79    
80           At the zeroth level, `f` is called once, performing 16 function 
81           evaluations. At each subsequent level, `f` is called once more, 
82           approximately doubling the number of function evaluations that have 
83           been performed. Accordingly, for many integrands, each successive level 
84           will double the number of accurate digits in the result (up to the 
85           limits of floating point precision). 
86    
87           The algorithm will terminate after completing level `maxlevel` or after 
88           another termination condition is satisfied, whichever comes first. 
89       minlevel : int, default: 2 
90           The level at which to begin iteration (default: 2). This does not 
91           change the total number of function evaluations or the abscissae at 
92           which the function is evaluated; it changes only the *number of times* 
93           `f` is called. If ``minlevel=k``, then the integrand is evaluated at 
94           all abscissae from levels ``0`` through ``k`` in a single call. 
95           Note that if `minlevel` exceeds `maxlevel`, the provided `minlevel` is 
96           ignored, and `minlevel` is set equal to `maxlevel`. 
97       atol, rtol : float, optional 
98           Absolute termination tolerance (default: 0) and relative termination 
99           tolerance (default: ``eps**0.75``, where ``eps`` is the precision of 
100          the result dtype), respectively.  Iteration will stop when 
101          ``res.error < atol`` or  ``res.error < res.integral * rtol``. The error 
102          estimate is as described in [1]_ Section 5 but with a lower bound of 
103          ``eps * res.integral``. While not theoretically rigorous or 
104          conservative, it is said to work well in practice. Must be non-negative 
105          and finite if `log` is False, and must be expressed as the log of a 
106          non-negative and finite number if `log` is True. 
107      preserve_shape : bool, default: False 
108          In the following, "arguments of `f`" refers to the array ``xi`` and 
109          any arrays within ``argsi``. Let ``shape`` be the broadcasted shape 
110          of `a`, `b`, and all elements of `args` (which is conceptually 
111          distinct from ``xi` and ``argsi`` passed into `f`). 
112   
113          - When ``preserve_shape=False`` (default), `f` must accept arguments 
114            of *any* broadcastable shapes. 
115   
116          - When ``preserve_shape=True``, `f` must accept arguments of shape 
117            ``shape`` *or* ``shape + (n,)``, where ``(n,)`` is the number of 
118            abscissae at which the function is being evaluated. 
119   
120          In either case, for each scalar element ``xi[j]`` within ``xi``, the array 
121          returned by `f` must include the scalar ``f(xi[j])`` at the same index. 
122          Consequently, the shape of the output is always the shape of the input 
123          ``xi``. 
124   
125          See Examples. 
126   
127      callback : callable, optional 
128          An optional user-supplied function to be called before the first 
129          iteration and after each iteration. 
130          Called as ``callback(res)``, where ``res`` is a ``_RichResult`` 
131          similar to that returned by `_differentiate` (but containing the 
132          current iterate's values of all variables). If `callback` raises a 
133          ``StopIteration``, the algorithm will terminate immediately and 
134          `tanhsinh` will return a result object. `callback` must not mutate 
135          `res` or its attributes. 
136   
137      Returns 
138      ------- 
139      res : _RichResult 
140          An object similar to an instance of `scipy.optimize.OptimizeResult` with the 
141          following attributes. (The descriptions are written as though the values will 
142          be scalars; however, if `f` returns an array, the outputs will be 
143          arrays of the same shape.) 
144   
145          success : bool array 
146              ``True`` when the algorithm terminated successfully (status ``0``). 
147              ``False`` otherwise. 
148          status : int array 
149              An integer representing the exit status of the algorithm. 
150   
151              ``0`` : The algorithm converged to the specified tolerances. 
152              ``-1`` : (unused) 
153              ``-2`` : The maximum number of iterations was reached. 
154              ``-3`` : A non-finite value was encountered. 
155              ``-4`` : Iteration was terminated by `callback`. 
156              ``1`` : The algorithm is proceeding normally (in `callback` only). 
157   
158          integral : float array 
159              An estimate of the integral. 
160          error : float array 
161              An estimate of the error. Only available if level two or higher 
162              has been completed; otherwise NaN. 
163          maxlevel : int array 
164              The maximum refinement level used. 
165          nfev : int array 
166              The number of points at which `f` was evaluated.

172      Notes 
173      ----- 
174      Implements the algorithm as described in [1]_ with minor adaptations for 
175      finite-precision arithmetic, including some described by [2]_ and [3]_. The 
176      tanh-sinh scheme was originally introduced in [4]_. 
177   
178      Due to floating-point error in the abscissae, the function may be evaluated 
179      at the endpoints of the interval during iterations, but the values returned by 
180      the function at the endpoints will be ignored. 
181   
182      References 
183      ---------- 
184      .. [1] Bailey, David H., Karthik Jeyabalan, and Xiaoye S. Li. "A comparison of 
185             three high-precision quadrature schemes." Experimental Mathematics 14.3 
186             (2005): 317-329. 
187      .. [2] Vanherck, Joren, Bart Sorée, and Wim Magnus. "Tanh-sinh quadrature for 
188             single and multiple integration using floating-point arithmetic." 
189             arXiv preprint arXiv:2007.15057 (2020). 
190      .. [3] van Engelen, Robert A.  "Improving the Double Exponential Quadrature 
191             Tanh-Sinh, Sinh-Sinh and Exp-Sinh Formulas." 
192             https://www.genivia.com/files/qthsh.pdf 
193      .. [4] Takahasi, Hidetosi, and Masatake Mori. "Double exponential formulas for 
194             numerical integration." Publications of the Research Institute for

197      Examples 
198      -------- 
199      Evaluate the Gaussian integral: 
200   
201      >>> import numpy as np 
202      >>> from scipy.integrate import tanhsinh 
203      >>> def f(x): 
204      ...     return np.exp(-x**2) 
205      >>> res = tanhsinh(f, -np.inf, np.inf) 
206      >>> res.integral  # true value is np.sqrt(np.pi), 1.7724538509055159 
207      1.7724538509055159 
208      >>> res.error  # actual error is 0 
209      4.0007963937534104e-16 
210   
211      The value of the Gaussian function (bell curve) is nearly zero for 
212      arguments sufficiently far from zero, so the value of the integral 
213      over a finite interval is nearly the same. 
214   
215      >>> tanhsinh(f, -20, 20).integral 
216      1.772453850905518 
217   
218      However, with unfavorable integration limits, the integration scheme 
219      may not be able to find the important region. 
220   
221      >>> tanhsinh(f, -np.inf, 1000).integral 
222      4.500490856616431 
223   
224      In such cases, or when there are singularities within the interval, 
225      break the integral into parts with endpoints at the important points. 
226   
227      >>> tanhsinh(f, -np.inf, 0).integral + tanhsinh(f, 0, 1000).integral 
228      1.772453850905404 
229   
230      For integration involving very large or very small magnitudes, use 
231      log-integration. (For illustrative purposes, the following example shows a 
232      case in which both regular and log-integration work, but for more extreme 
233      limits of integration, log-integration would avoid the underflow 
234      experienced when evaluating the integral normally.) 
235   
236      >>> res = tanhsinh(f, 20, 30, rtol=1e-10) 
237      >>> res.integral, res.error 
238      (4.7819613911309014e-176, 4.670364401645202e-187) 
239      >>> def log_f(x): 
240      ...     return -x**2 
241      >>> res = tanhsinh(log_f, 20, 30, log=True, rtol=np.log(1e-10)) 
242      >>> np.exp(res.integral), np.exp(res.error) 
243      (4.7819613911306924e-176, 4.670364401645093e-187) 
244   
245      The limits of integration and elements of `args` may be broadcastable 
246      arrays, and integration is performed elementwise. 
247   
248      >>> from scipy import stats 
249      >>> dist = stats.gausshyper(13.8, 3.12, 2.51, 5.18) 
250      >>> a, b = dist.support() 
251      >>> x = np.linspace(a, b, 100) 
252      >>> res = tanhsinh(dist.pdf, a, x) 
253      >>> ref = dist.cdf(x) 
254      >>> np.allclose(res.integral, ref) 
255      True 
256   
257      By default, `preserve_shape` is False, and therefore the callable 
258      `f` may be called with arrays of any broadcastable shapes. 
259      For example: 
260   
261      >>> shapes = [] 
262      >>> def f(x, c): 
263      ...    shape = np.broadcast_shapes(x.shape, c.shape) 
264      ...    shapes.append(shape) 
265      ...    return np.sin(c*x) 
266      >>> 
267      >>> c = [1, 10, 30, 100] 
268      >>> res = tanhsinh(f, 0, 1, args=(c,), minlevel=1) 
269      >>> shapes 
270      [(4,), (4, 34), (4, 32), (3, 64), (2, 128), (1, 256)] 
271   
272      To understand where these shapes are coming from - and to better 
273      understand how `tanhsinh` computes accurate results - note that 
274      higher values of ``c`` correspond with higher frequency sinusoids. 
275      The higher frequency sinusoids make the integrand more complicated, 
276      so more function evaluations are required to achieve the target 
277      accuracy: 
278   
279      >>> res.nfev 
280      array([ 67, 131, 259, 515], dtype=int32) 
281   
282      The initial ``shape``, ``(4,)``, corresponds with evaluating the 
283      integrand at a single abscissa and all four frequencies; this is used 
284      for input validation and to determine the size and dtype of the arrays 
285      that store results. The next shape corresponds with evaluating the 
286      integrand at an initial grid of abscissae and all four frequencies. 
287      Successive calls to the function double the total number of abscissae at 
288      which the function has been evaluated. However, in later function 
289      evaluations, the integrand is evaluated at fewer frequencies because 
290      the corresponding integral has already converged to the required 
291      tolerance. This saves function evaluations to improve performance, but 
292      it requires the function to accept arguments of any shape. 
293   
294      "Vector-valued" integrands, such as those written for use with 
295      `scipy.integrate.quad_vec`, are unlikely to satisfy this requirement. 
296      For example, consider 
297   
298      >>> def f(x): 
299      ...    return [x, np.sin(10*x), np.cos(30*x), x*np.sin(100*x)**2] 
300   
301      This integrand is not compatible with `tanhsinh` as written; for instance, 
302      the shape of the output will not be the same as the shape of ``x``. Such a 
303      function *could* be converted to a compatible form with the introduction of 
304      additional parameters, but this would be inconvenient. In such cases, 
305      a simpler solution would be to use `preserve_shape`. 
306   
307      >>> shapes = [] 
308      >>> def f(x): 
309      ...     shapes.append(x.shape) 
310      ...     x0, x1, x2, x3 = x 
311      ...     return [x0, np.sin(10*x1), np.cos(30*x2), x3*np.sin(100*x3)] 
312      >>> 
313      >>> a = np.zeros(4) 
314      >>> res = tanhsinh(f, a, 1, preserve_shape=True) 
315      >>> shapes 
316      [(4,), (4, 66), (4, 64), (4, 128), (4, 256)] 
317   
318      Here, the broadcasted shape of `a` and `b` is ``(4,)``. With 
319      ``preserve_shape=True``, the function may be called with argument 
320      ``x`` of shape ``(4,)`` or ``(4, n)``, and this is what we observe. 
321   
322      """
323      maxfun = None  # unused right now
324      (f, a, b, log, maxfun, maxlevel, minlevel,
325       atol, rtol, args, preserve_shape, callback, xp) = _tanhsinh_iv(
326          f, a, b, log, maxfun, maxlevel, minlevel, atol,
327          rtol, args, preserve_shape, callback)
328  
329      # Initialization
330      # `eim._initialize` does several important jobs, including
331      # ensuring that limits, each of the `args`, and the output of `f`
332      # broadcast correctly and are of consistent types. To save a function
333      # evaluation, I pass the midpoint of the integration interval. This comes
334      # at a cost of some gymnastics to ensure that the midpoint has the right
335      # shape and dtype. Did you know that 0d and >0d arrays follow different
336      # type promotion rules?
337      with np.errstate(over='ignore', invalid='ignore', divide='ignore'):
338          c = xp.reshape((xp_ravel(a) + xp_ravel(b))/2, a.shape)
339          inf_a, inf_b = xp.isinf(a), xp.isinf(b)
340          c[inf_a] = b[inf_a] - 1.  # takes care of infinite a
341          c[inf_b] = a[inf_b] + 1.  # takes care of infinite b
342          c[inf_a & inf_b] = 0.  # takes care of infinite a and b
343          temp = eim._initialize(f, (c,), args, complex_ok=True,
344                                 preserve_shape=preserve_shape, xp=xp)
345      f, xs, fs, args, shape, dtype, xp = temp
346      a = xp_ravel(xp.astype(xp.broadcast_to(a, shape), dtype))
347      b = xp_ravel(xp.astype(xp.broadcast_to(b, shape), dtype))
348  
349      # Transform improper integrals
350      a, b, a0, negative, abinf, ainf, binf = _transform_integrals(a, b, xp)
351  
352      # Define variables we'll need
353      nit, nfev = 0, 1  # one function evaluation performed above
354      zero = -xp.inf if log else 0
355      pi = xp.asarray(xp.pi, dtype=dtype)[()]
356      maxiter = maxlevel - minlevel + 1
357      eps = xp.finfo(dtype).eps
358      if rtol is None:
359          rtol = 0.75*math.log(eps) if log else eps**0.75
360  
361      Sn = xp_ravel(xp.full(shape, zero, dtype=dtype))  # latest integral estimate
362      Sn[xp.isnan(a) | xp.isnan(b) | xp.isnan(fs[0])] = xp.nan
363      Sk = xp.reshape(xp.empty_like(Sn), (-1, 1))[:, 0:0]  # all integral estimates
364      aerr = xp_ravel(xp.full(shape, xp.nan, dtype=dtype))  # absolute error
365      status = xp_ravel(xp.full(shape, eim._EINPROGRESS, dtype=xp.int32))
366      h0 = _get_base_step(dtype, xp)
367      h0 = xp_real(h0) # base step
368  
369      # For term `d4` of error estimate ([1] Section 5), we need to keep the
370      # most extreme abscissae and corresponding `fj`s, `wj`s in Euler-Maclaurin
371      # sum. Here, we initialize these variables.
372      xr0 = xp_ravel(xp.full(shape, -xp.inf, dtype=dtype))
373      fr0 = xp_ravel(xp.full(shape, xp.nan, dtype=dtype))
374      wr0 = xp_ravel(xp.zeros(shape, dtype=dtype))
375      xl0 = xp_ravel(xp.full(shape, xp.inf, dtype=dtype))
376      fl0 = xp_ravel(xp.full(shape, xp.nan, dtype=dtype))
377      wl0 = xp_ravel(xp.zeros(shape, dtype=dtype))
378      d4 = xp_ravel(xp.zeros(shape, dtype=dtype))
379  
380      work = _RichResult(
381          Sn=Sn, Sk=Sk, aerr=aerr, h=h0, log=log, dtype=dtype, pi=pi, eps=eps,
382          a=xp.reshape(a, (-1, 1)), b=xp.reshape(b, (-1, 1)),  # integration limits
383          n=minlevel, nit=nit, nfev=nfev, status=status,  # iter/eval counts
384          xr0=xr0, fr0=fr0, wr0=wr0, xl0=xl0, fl0=fl0, wl0=wl0, d4=d4,  # err est
385          ainf=ainf, binf=binf, abinf=abinf, a0=xp.reshape(a0, (-1, 1)),  # transforms
386          # Store the xjc/wj pair cache in an object so they can't get compressed
387          # Using RichResult to allow dot notation, but a dictionary would suffice
388          pair_cache=_RichResult(xjc=None, wj=None, indices=[0], h0=None))  # pair cache
389  
390      # Constant scalars don't need to be put in `work` unless they need to be
391      # passed outside `tanhsinh`. Examples: atol, rtol, h0, minlevel.
392  
393      # Correspondence between terms in the `work` object and the result
394      res_work_pairs = [('status', 'status'), ('integral', 'Sn'),
395                        ('error', 'aerr'), ('nit', 'nit'), ('nfev', 'nfev')]
396  
397      def pre_func_eval(work):
398          # Determine abscissae at which to evaluate `f`
399          work.h = h0 / 2**work.n
400          xjc, wj = _get_pairs(work.n, h0, dtype=work.dtype,
401                               inclusive=(work.n == minlevel), xp=xp, work=work)
402          work.xj, work.wj = _transform_to_limits(xjc, wj, work.a, work.b, xp)
403  
404          # Perform abscissae substitutions for infinite limits of integration
405          xj = xp_copy(work.xj)
406          # use xp_real here to avoid cupy/cupy#8434
407          xj[work.abinf] = xj[work.abinf] / (1 - xp_real(xj[work.abinf])**2)
408          xj[work.binf] = 1/xj[work.binf] - 1 + work.a0[work.binf]
409          xj[work.ainf] *= -1
410          return xj
411  
412      def post_func_eval(x, fj, work):
413          # Weight integrand as required by substitutions for infinite limits
414          if work.log:
415              fj[work.abinf] += (xp.log(1 + work.xj[work.abinf]**2)
416                                 - 2*xp.log(1 - work.xj[work.abinf]**2))
417              fj[work.binf] -= 2 * xp.log(work.xj[work.binf])
418          else:
419              fj[work.abinf] *= ((1 + work.xj[work.abinf]**2) /
420                                 (1 - work.xj[work.abinf]**2)**2)
421              fj[work.binf] *= work.xj[work.binf]**-2.
422  
423          # Estimate integral with Euler-Maclaurin Sum
424          fjwj, Sn = _euler_maclaurin_sum(fj, work, xp)
425          if work.Sk.shape[-1]:
426              Snm1 = work.Sk[:, -1]
427              Sn = (special.logsumexp(xp.stack([Snm1 - math.log(2), Sn]), axis=0) if log
428                    else Snm1 / 2 + Sn)
429  
430          work.fjwj = fjwj
431          work.Sn = Sn
432  
433      def check_termination(work):
434          """Terminate due to convergence or encountering non-finite values"""
435          stop = xp.zeros(work.Sn.shape, dtype=bool)
436  
437          # Terminate before first iteration if integration limits are equal
438          if work.nit == 0:
439              i = xp_ravel(work.a == work.b)  # ravel singleton dimension
440              zero = xp.asarray(-xp.inf if log else 0.)
441              zero = xp.full(work.Sn.shape, zero, dtype=Sn.dtype)
442              zero[xp.isnan(Sn)] = xp.nan
443              work.Sn[i] = zero[i]
444              work.aerr[i] = zero[i]
445              work.status[i] = eim._ECONVERGED
446              stop[i] = True
447          else:
448              # Terminate if convergence criterion is met
449              rerr, aerr = _estimate_error(work, xp)
450              i = (rerr < rtol) | (aerr < atol)
451              work.aerr =  xp.reshape(xp.astype(aerr, work.dtype), work.Sn.shape)
452              work.status[i] = eim._ECONVERGED
453              stop[i] = True
454  
455          # Terminate if integral estimate becomes invalid
456          if log:
457              Sn_real = xp_real(work.Sn)
458              Sn_pos_inf = xp.isinf(Sn_real) & (Sn_real > 0)
459              i = (Sn_pos_inf | xp.isnan(work.Sn)) & ~stop
460          else:
461              i = ~xp.isfinite(work.Sn) & ~stop
462          work.status[i] = eim._EVALUEERR
463          stop[i] = True
464  
465          return stop
466  
467      def post_termination_check(work):
468          work.n += 1
469          work.Sk = xp.concat((work.Sk, work.Sn[:, xp.newaxis]), axis=-1)
470          return
471  
472      def customize_result(res, shape):
473          # If the integration limits were such that b < a, we reversed them
474          # to perform the calculation, and the final result needs to be negated.
475          if log and xp.any(negative):
476              dtype = res['integral'].dtype
477              pi = xp.asarray(xp.pi, dtype=dtype)[()]
478              j = xp.asarray(1j, dtype=xp.complex64)[()]  # minimum complex type
479              res['integral'] = res['integral'] + negative*pi*j
480          else:
481              res['integral'][negative] *= -1
482  
483          # For this algorithm, it seems more appropriate to report the maximum
484          # level rather than the number of iterations in which it was performed.
485          res['maxlevel'] = minlevel + res['nit'] - 1
486          res['maxlevel'][res['nit'] == 0] = -1
487          del res['nit']
488          return shape
489  
490      # Suppress all warnings initially, since there are many places in the code
491      # for which this is expected behavior.
492      with np.errstate(over='ignore', invalid='ignore', divide='ignore'):
493          res = eim._loop(work, callback, shape, maxiter, f, args, dtype, pre_func_eval,
494                          post_func_eval, check_termination, post_termination_check,
495                          customize_result, res_work_pairs, xp, preserve_shape)
496      return res

499  def _get_base_step(dtype, xp):
500      # Compute the base step length for the provided dtype. Theoretically, the
501      # Euler-Maclaurin sum is infinite, but it gets cut off when either the
502      # weights underflow or the abscissae cannot be distinguished from the
503      # limits of integration. The latter happens to occur first for float32 and
504      # float64, and it occurs when `xjc` (the abscissa complement)
505      # in `_compute_pair` underflows. We can solve for the argument `tmax` at
506      # which it will underflow using [2] Eq. 13.
507      fmin = 4*xp.finfo(dtype).smallest_normal  # stay a little away from the limit
508      tmax = math.asinh(math.log(2/fmin - 1) / xp.pi)
509  
510      # Based on this, we can choose a base step size `h` for level 0.
511      # The number of function evaluations will be `2 + m*2^(k+1)`, where `k` is
512      # the level and `m` is an integer we get to choose. I choose
513      # m = _N_BASE_STEPS = `8` somewhat arbitrarily, but a rationale is that a
514      # power of 2 makes floating point arithmetic more predictable. It also
515      # results in a base step size close to `1`, which is what [1] uses (and I
516      # used here until I found [2] and these ideas settled).
517      h0 = tmax / _N_BASE_STEPS
518      return xp.asarray(h0, dtype=dtype)[()]

524  def _compute_pair(k, h0, xp):
525      # Compute the abscissa-weight pairs for each level k. See [1] page 9.
526  
527      # For now, we compute and store in 64-bit precision. If higher-precision
528      # data types become better supported, it would be good to compute these
529      # using the highest precision available. Or, once there is an Array API-
530      # compatible arbitrary precision array, we can compute at the required
531      # precision.
532  
533      # "....each level k of abscissa-weight pairs uses h = 2 **-k"
534      # We adapt to floating point arithmetic using ideas of [2].
535      h = h0 / 2**k
536      max = _N_BASE_STEPS * 2**k
537  
538      # For iterations after the first, "....the integrand function needs to be
539      # evaluated only at the odd-indexed abscissas at each level."
540      j = xp.arange(max+1) if k == 0 else xp.arange(1, max+1, 2)
541      jh = j * h
542  
543      # "In this case... the weights wj = u1/cosh(u2)^2, where..."
544      pi_2 = xp.pi / 2
545      u1 = pi_2*xp.cosh(jh)
546      u2 = pi_2*xp.sinh(jh)
547      # Denominators get big here. Overflow then underflow doesn't need warning.
548      # with np.errstate(under='ignore', over='ignore'):
549      wj = u1 / xp.cosh(u2)**2
550      # "We actually store 1-xj = 1/(...)."
551      xjc = 1 / (xp.exp(u2) * xp.cosh(u2))  # complement of xj = xp.tanh(u2)
552  
553      # When level k == 0, the zeroth xj corresponds with xj = 0. To simplify
554      # code, the function will be evaluated there twice; each gets half weight.
555      wj[0] = wj[0] / 2 if k == 0 else wj[0]
556  
557      return xjc, wj  # store at full precision

560  def _pair_cache(k, h0, xp, work):
561      # Cache the abscissa-weight pairs up to a specified level.
562      # Abscissae and weights of consecutive levels are concatenated.
563      # `index` records the indices that correspond with each level:
564      # `xjc[index[k]:index[k+1]` extracts the level `k` abscissae.
565      if not isinstance(h0, type(work.pair_cache.h0)) or h0 != work.pair_cache.h0:
566          work.pair_cache.xjc = xp.empty(0)
567          work.pair_cache.wj = xp.empty(0)
568          work.pair_cache.indices = [0]
569  
570      xjcs = [work.pair_cache.xjc]
571      wjs = [work.pair_cache.wj]
572  
573      for i in range(len(work.pair_cache.indices)-1, k + 1):
574          xjc, wj = _compute_pair(i, h0, xp)
575          xjcs.append(xjc)
576          wjs.append(wj)
577          work.pair_cache.indices.append(work.pair_cache.indices[-1] + xjc.shape[0])
578  
579      work.pair_cache.xjc = xp.concat(xjcs)
580      work.pair_cache.wj = xp.concat(wjs)
581      work.pair_cache.h0 = h0

584  def _get_pairs(k, h0, inclusive, dtype, xp, work):
585      # Retrieve the specified abscissa-weight pairs from the cache
586      # If `inclusive`, return all up to and including the specified level
587      if (len(work.pair_cache.indices) <= k+2
588          or not isinstance (h0, type(work.pair_cache.h0))
589          or h0 != work.pair_cache.h0):
590              _pair_cache(k, h0, xp, work)
591  
592      xjc = work.pair_cache.xjc
593      wj = work.pair_cache.wj
594      indices = work.pair_cache.indices
595  
596      start = 0 if inclusive else indices[k]
597      end = indices[k+1]
598  
599      return xp.astype(xjc[start:end], dtype), xp.astype(wj[start:end], dtype)

602  def _transform_to_limits(xjc, wj, a, b, xp):
603      # Transform integral according to user-specified limits. This is just
604      # math that follows from the fact that the standard limits are (-1, 1).
605      # Note: If we had stored xj instead of xjc, we would have
606      # xj = alpha * xj + beta, where beta = (a + b)/2
607      alpha = (b - a) / 2
608      xj = xp.concat((-alpha * xjc + b, alpha * xjc + a), axis=-1)
609      wj = wj*alpha  # arguments get broadcasted, so we can't use *=
610      wj = xp.concat((wj, wj), axis=-1)
611  
612      # Points at the boundaries can be generated due to finite precision
613      # arithmetic, but these function values aren't supposed to be included in
614      # the Euler-Maclaurin sum. Ideally we wouldn't evaluate the function at
615      # these points; however, we can't easily filter out points since this
616      # function is vectorized. Instead, zero the weights.
617      # Note: values may have complex dtype, but have zero imaginary part
618      xj_real, a_real, b_real = xp_real(xj), xp_real(a), xp_real(b)
619      invalid = (xj_real <= a_real) | (xj_real >= b_real)
620      wj[invalid] = 0
621      return xj, wj

624  def _euler_maclaurin_sum(fj, work, xp):
625      # Perform the Euler-Maclaurin Sum, [1] Section 4
626  
627      # The error estimate needs to know the magnitude of the last term
628      # omitted from the Euler-Maclaurin sum. This is a bit involved because
629      # it may have been computed at a previous level. I sure hope it's worth
630      # all the trouble.
631      xr0, fr0, wr0 = work.xr0, work.fr0, work.wr0
632      xl0, fl0, wl0 = work.xl0, work.fl0, work.wl0
633  
634      # It is much more convenient to work with the transposes of our work
635      # variables here.
636      xj, fj, wj = work.xj.T, fj.T, work.wj.T
637      n_x, n_active = xj.shape  # number of abscissae, number of active elements
638  
639      # We'll work with the left and right sides separately
640      xr, xl = xp_copy(xp.reshape(xj, (2, n_x // 2, n_active)))  # this gets modified
641      fr, fl = xp.reshape(fj, (2, n_x // 2, n_active))
642      wr, wl = xp.reshape(wj, (2, n_x // 2, n_active))
643  
644      invalid_r = ~xp.isfinite(fr) | (wr == 0)
645      invalid_l = ~xp.isfinite(fl) | (wl == 0)
646  
647      # integer index of the maximum abscissa at this level
648      xr[invalid_r] = -xp.inf
649      ir = xp.argmax(xp_real(xr), axis=0, keepdims=True)
650      # abscissa, function value, and weight at this index
651      ### Not Array API Compatible... yet ###
652      xr_max = xp_take_along_axis(xr, ir, axis=0)[0]
653      fr_max = xp_take_along_axis(fr, ir, axis=0)[0]
654      wr_max = xp_take_along_axis(wr, ir, axis=0)[0]
655      # boolean indices at which maximum abscissa at this level exceeds
656      # the incumbent maximum abscissa (from all previous levels)
657      # note: abscissa may have complex dtype, but will have zero imaginary part
658      j = xp_real(xr_max) > xp_real(xr0)
659      # Update record of the incumbent abscissa, function value, and weight
660      xr0[j] = xr_max[j]
661      fr0[j] = fr_max[j]
662      wr0[j] = wr_max[j]
663  
664      # integer index of the minimum abscissa at this level
665      xl[invalid_l] = xp.inf
666      il = xp.argmin(xp_real(xl), axis=0, keepdims=True)
667      # abscissa, function value, and weight at this index
668      xl_min = xp_take_along_axis(xl, il, axis=0)[0]
669      fl_min = xp_take_along_axis(fl, il, axis=0)[0]
670      wl_min = xp_take_along_axis(wl, il, axis=0)[0]
671      # boolean indices at which minimum abscissa at this level is less than
672      # the incumbent minimum abscissa (from all previous levels)
673      # note: abscissa may have complex dtype, but will have zero imaginary part
674      j = xp_real(xl_min) < xp_real(xl0)
675      # Update record of the incumbent abscissa, function value, and weight
676      xl0[j] = xl_min[j]
677      fl0[j] = fl_min[j]
678      wl0[j] = wl_min[j]
679      fj = fj.T
680  
681      # Compute the error estimate `d4` - the magnitude of the leftmost or
682      # rightmost term, whichever is greater.
683      flwl0 = fl0 + xp.log(wl0) if work.log else fl0 * wl0  # leftmost term
684      frwr0 = fr0 + xp.log(wr0) if work.log else fr0 * wr0  # rightmost term
685      magnitude = xp_real if work.log else xp.abs
686      work.d4 = xp.maximum(magnitude(flwl0), magnitude(frwr0))
687  
688      # There are two approaches to dealing with function values that are
689      # numerically infinite due to approaching a singularity - zero them, or
690      # replace them with the function value at the nearest non-infinite point.
691      # [3] pg. 22 suggests the latter, so let's do that given that we have the
692      # information.
693      fr0b = xp.broadcast_to(fr0[xp.newaxis, :], fr.shape)
694      fl0b = xp.broadcast_to(fl0[xp.newaxis, :], fl.shape)
695      fr[invalid_r] = fr0b[invalid_r]
696      fl[invalid_l] = fl0b[invalid_l]
697  
698      # When wj is zero, log emits a warning
699      # with np.errstate(divide='ignore'):
700      fjwj = fj + xp.log(work.wj) if work.log else fj * work.wj
701  
702      # update integral estimate
703      Sn = (special.logsumexp(fjwj + xp.log(work.h), axis=-1) if work.log
704            else xp.sum(fjwj, axis=-1) * work.h)
705  
706      work.xr0, work.fr0, work.wr0 = xr0, fr0, wr0
707      work.xl0, work.fl0, work.wl0 = xl0, fl0, wl0
708  
709      return fjwj, Sn

712  def _estimate_error(work, xp):
713      # Estimate the error according to [1] Section 5
714  
715      if work.n == 0 or work.nit == 0:
716          # The paper says to use "one" as the error before it can be calculated.
717          # NaN seems to be more appropriate.
718          nan = xp.full_like(work.Sn, xp.nan)
719          return nan, nan
720  
721      indices = work.pair_cache.indices
722  
723      n_active = work.Sn.shape[0]  # number of active elements
724      axis_kwargs = dict(axis=-1, keepdims=True)
725  
726      # With a jump start (starting at level higher than 0), we haven't
727      # explicitly calculated the integral estimate at lower levels. But we have
728      # all the function value-weight products, so we can compute the
729      # lower-level estimates.
730      if work.Sk.shape[-1] == 0:
731          h = 2 * work.h  # step size at this level
732          n_x = indices[work.n]  # number of abscissa up to this level
733          # The right and left fjwj terms from all levels are concatenated along
734          # the last axis. Get out only the terms up to this level.
735          fjwj_rl = xp.reshape(work.fjwj, (n_active, 2, -1))
736          fjwj = xp.reshape(fjwj_rl[:, :, :n_x], (n_active, 2*n_x))
737          # Compute the Euler-Maclaurin sum at this level
738          Snm1 = (special.logsumexp(fjwj, **axis_kwargs) + xp.log(h) if work.log
739                  else xp.sum(fjwj, **axis_kwargs) * h)
740          work.Sk = xp.concat((Snm1, work.Sk), axis=-1)
741  
742      if work.n == 1:
743          nan = xp.full_like(work.Sn, xp.nan)
744          return nan, nan
745  
746      # The paper says not to calculate the error for n<=2, but it's not clear
747      # about whether it starts at level 0 or level 1. We start at level 0, so
748      # why not compute the error beginning in level 2?
749      if work.Sk.shape[-1] < 2:
750          h = 4 * work.h  # step size at this level
751          n_x = indices[work.n-1]  # number of abscissa up to this level
752          # The right and left fjwj terms from all levels are concatenated along
753          # the last axis. Get out only the terms up to this level.
754          fjwj_rl = xp.reshape(work.fjwj, (work.Sn.shape[0], 2, -1))
755          fjwj = xp.reshape(fjwj_rl[..., :n_x], (n_active, 2*n_x))
756          # Compute the Euler-Maclaurin sum at this level
757          Snm2 = (special.logsumexp(fjwj, **axis_kwargs) + xp.log(h) if work.log
758                  else xp.sum(fjwj, **axis_kwargs) * h)
759          work.Sk = xp.concat((Snm2, work.Sk), axis=-1)
760  
761      Snm2 = work.Sk[..., -2]
762      Snm1 = work.Sk[..., -1]
763  
764      e1 = xp.asarray(work.eps)[()]
765  
766      if work.log:
767          log_e1 = xp.log(e1)
768          # Currently, only real integrals are supported in log-scale. All
769          # complex values have imaginary part in increments of pi*j, which just
770          # carries sign information of the original integral, so use of
771          # `xp.real` here is equivalent to absolute value in real scale.
772          d1 = xp_real(special.logsumexp(xp.stack([work.Sn, Snm1 + work.pi*1j]), axis=0))
773          d2 = xp_real(special.logsumexp(xp.stack([work.Sn, Snm2 + work.pi*1j]), axis=0))
774          d3 = log_e1 + xp.max(xp_real(work.fjwj), axis=-1)
775          d4 = work.d4
776          d5 = log_e1 + xp.real(work.Sn)
777          temp = xp.where(d1 > -xp.inf, d1 ** 2 / d2, -xp.inf)
778          ds = xp.stack([temp, 2 * d1, d3, d4, d5])
779          aerr = xp.max(ds, axis=0)
780          rerr = aerr - xp.real(work.Sn)
781      else:
782          # Note: explicit computation of log10 of each of these is unnecessary.
783          d1 = xp.abs(work.Sn - Snm1)
784          d2 = xp.abs(work.Sn - Snm2)
785          d3 = e1 * xp.max(xp.abs(work.fjwj), axis=-1)
786          d4 = work.d4
787          d5 = e1 * xp.abs(work.Sn)
788          temp = xp.where(d1 > 0, d1**(xp.log(d1)/xp.log(d2)), 0)
789          ds = xp.stack([temp, d1**2, d3, d4, d5])
790          aerr = xp.max(ds, axis=0)
791          rerr = aerr/xp.abs(work.Sn)
792  
793      return rerr, aerr

796  def _transform_integrals(a, b, xp):
797      # Transform integrals to a form with finite a <= b
798      # For b == a (even infinite), we ensure that the limits remain equal
799      # For b < a, we reverse the limits and will multiply the final result by -1
800      # For infinite limit on the right, we use the substitution x = 1/t - 1 + a
801      # For infinite limit on the left, we substitute x = -x and treat as above
802      # For infinite limits, we substitute x = t / (1-t**2)
803      ab_same = (a == b)
804      a[ab_same], b[ab_same] = 1, 1
805  
806      # `a, b` may have complex dtype but have zero imaginary part
807      negative = xp_real(b) < xp_real(a)
808      a[negative], b[negative] = b[negative], a[negative]
809  
810      abinf = xp.isinf(a) & xp.isinf(b)
811      a[abinf], b[abinf] = -1, 1
812  
813      ainf = xp.isinf(a)
814      a[ainf], b[ainf] = -b[ainf], -a[ainf]
815  
816      binf = xp.isinf(b)
817      a0 = xp_copy(a)
818      a[binf], b[binf] = 0, 1
819  
820      return a, b, a0, negative, abinf, ainf, binf

961  def nsum(f, a, b, *, step=1, args=(), log=False, maxterms=int(2**20), tolerances=None):
962      r"""Evaluate a convergent finite or infinite series. 
963   
964      For finite `a` and `b`, this evaluates:: 
965   
966          f(a + np.arange(n)*step).sum() 
967   
968      where ``n = int((b - a) / step) + 1``, where `f` is smooth, positive, and 
969      unimodal. The number of terms in the sum may be very large or infinite, 
970      in which case a partial sum is evaluated directly and the remainder is 
971      approximated using integration. 
972   
973      Parameters 
974      ---------- 
975      f : callable 
976          The function that evaluates terms to be summed. The signature must be:: 
977   
978              f(x: ndarray, *args) -> ndarray 
979   
980          where each element of ``x`` is a finite real and ``args`` is a tuple, 
981          which may contain an arbitrary number of arrays that are broadcastable 
982          with ``x``. 
983   
984          `f` must be an elementwise function: each element ``f(x)[i]`` 
985          must equal ``f(x[i])`` for all indices ``i``. It must not mutate the 
986          array ``x`` or the arrays in ``args``, and it must return NaN where 
987          the argument is NaN. 
988   
989          `f` must represent a smooth, positive, unimodal function of `x` defined at 
990          *all reals* between `a` and `b`. 
991      a, b : float array_like 
992          Real lower and upper limits of summed terms. Must be broadcastable. 
993          Each element of `a` must be less than the corresponding element in `b`.

1061     Notes 
1062     ----- 
1063     The method implemented for infinite summation is related to the integral 
1064     test for convergence of an infinite series: assuming `step` size 1 for 
1065     simplicity of exposition, the sum of a monotone decreasing function is bounded by 
1066  
1067     .. math:: 
1068  
1069         \int_u^\infty f(x) dx \leq \sum_{k=u}^\infty f(k) \leq \int_u^\infty f(x) dx + f(u) 
1070  
1071     Let :math:`a` represent  `a`, :math:`n` represent `maxterms`, :math:`\epsilon_a` 
1072     represent `atol`, and :math:`\epsilon_r` represent `rtol`. 
1073     The implementation first evaluates the integral :math:`S_l=\int_a^\infty f(x) dx` 
1074     as a lower bound of the infinite sum. Then, it seeks a value :math:`c > a` such 
1075     that :math:`f(c) < \epsilon_a + S_l \epsilon_r`, if it exists; otherwise, 
1076     let :math:`c = a + n`. Then the infinite sum is approximated as 
1077  
1078     .. math:: 
1079  
1080         \sum_{k=a}^{c-1} f(k) + \int_c^\infty f(x) dx + f(c)/2 
1081  
1082     and the reported error is :math:`f(c)/2` plus the error estimate of 
1083     numerical integration. Note that the integral approximations may require 
1084     evaluation of the function at points besides those that appear in the sum, 
1085     so `f` must be a continuous and monotonically decreasing function defined 
1086     for all reals within the integration interval. However, due to the nature 
1087     of the integral approximation, the shape of the function between points 
1088     that appear in the sum has little effect. If there is not a natural 
1089     extension of the function to all reals, consider using linear interpolation, 
1090     which is easy to evaluate and preserves monotonicity. 
1091  
1092     The approach described above is generalized for non-unit 
1093     `step` and finite `b` that is too large for direct evaluation of the sum, 
1094     i.e. ``b - a + 1 > maxterms``. It is further generalized to unimodal 
1095     functions by directly summing terms surrounding the maximum. 
1096     This strategy may fail: 
1097  
1098     - If the left limit is finite and the maximum is far from it. 
1099     - If the right limit is finite and the maximum is far from it. 
1100     - If both limits are finite and the maximum is far from the origin. 
1101  
1102     In these cases, accuracy may be poor, and `nsum` may return status code ``4``. 
1103  
1104     Although the callable `f` must be non-negative and unimodal, 
1105     `nsum` can be used to evaluate more general forms of series. For instance, to 
1106     evaluate an alternating series, pass a callable that returns the difference 
1107     between pairs of adjacent terms, and adjust `step` accordingly. See Examples. 
1108  
1109     References

1111     .. [1] Wikipedia. "Integral test for convergence." 
1112            https://en.wikipedia.org/wiki/Integral_test_for_convergence

1253 def _direct(f, a, b, step, args, constants, xp, inclusive=True):
1254     # Directly evaluate the sum.
1255 
1256     # When used in the context of distributions, `args` would contain the
1257     # distribution parameters. We have broadcasted for simplicity, but we could
1258     # reduce function evaluations when distribution parameters are the same but
1259     # sum limits differ. Roughly:
1260     # - compute the function at all points between min(a) and max(b),
1261     # - compute the cumulative sum,
1262     # - take the difference between elements of the cumulative sum
1263     #   corresponding with b and a.
1264     # This is left to future enhancement
1265 
1266     dtype, log, eps, zero, _, _, _ = constants
1267 
1268     # To allow computation in a single vectorized call, find the maximum number
1269     # of points (over all slices) at which the function needs to be evaluated.
1270     # Note: if `inclusive` is `True`, then we want `1` more term in the sum.
1271     # I didn't think it was great style to use `True` as `1` in Python, so I
1272     # explicitly converted it to an `int` before using it.
1273     inclusive_adjustment = int(inclusive)
1274     steps = xp.round((b - a) / step) + inclusive_adjustment
1275     # Equivalently, steps = xp.round((b - a) / step) + inclusive
1276     max_steps = int(xp.max(steps))
1277 
1278     # In each slice, the function will be evaluated at the same number of points,
1279     # but excessive points (those beyond the right sum limit `b`) are replaced
1280     # with NaN to (potentially) reduce the time of these unnecessary calculations.
1281     # Use a new last axis for these calculations for consistency with other
1282     # elementwise algorithms.
1283     a2, b2, step2 = a[:, xp.newaxis], b[:, xp.newaxis], step[:, xp.newaxis]
1284     args2 = [arg[:, xp.newaxis] for arg in args]
1285     ks = a2 + xp.arange(max_steps, dtype=dtype) * step2
1286     i_nan = ks >= (b2 + inclusive_adjustment*step2/2)
1287     ks[i_nan] = xp.nan
1288     fs = f(ks, *args2)
1289 
1290     # The function evaluated at NaN is NaN, and NaNs are zeroed in the sum.
1291     # In some cases it may be faster to loop over slices than to vectorize
1292     # like this. This is an optimization that can be added later.
1293     fs[i_nan] = zero
1294     nfev = max_steps - i_nan.sum(axis=-1)
1295     S = special.logsumexp(fs, axis=-1) if log else xp.sum(fs, axis=-1)
1296     # Rough, non-conservative error estimate. See gh-19667 for improvement ideas.
1297     E = xp_real(S) + math.log(eps) if log else eps * abs(S)
1298     return S, E, nfev

1301 def _integral_bound(f, a, b, step, args, constants, xp):
1302     # Estimate the sum with integral approximation
1303     dtype, log, _, _, rtol, atol, maxterms = constants
1304     log2 = xp.asarray(math.log(2), dtype=dtype)
1305 
1306     # Get a lower bound on the sum and compute effective absolute tolerance
1307     lb = tanhsinh(f, a, b, args=args, atol=atol, rtol=rtol, log=log)
1308     tol = xp.broadcast_to(xp.asarray(atol), lb.integral.shape)
1309     if log:
1310         tol = special.logsumexp(xp.stack((tol, rtol + lb.integral)), axis=0)
1311     else:
1312         tol = tol + rtol*lb.integral
1313     i_skip = lb.status < 0  # avoid unnecessary f_evals if integral is divergent
1314     tol[i_skip] = xp.nan
1315     status = lb.status
1316 
1317     # As in `_direct`, we'll need a temporary new axis for points
1318     # at which to evaluate the function. Append axis at the end for
1319     # consistency with other elementwise algorithms.
1320     a2 = a[..., xp.newaxis]
1321     step2 = step[..., xp.newaxis]
1322     args2 = [arg[..., xp.newaxis] for arg in args]
1323 
1324     # Find the location of a term that is less than the tolerance (if possible)
1325     log2maxterms = math.floor(math.log2(maxterms)) if maxterms else 0
1326     n_steps = xp.concat((2**xp.arange(0, log2maxterms), xp.asarray([maxterms])))
1327     n_steps = xp.astype(n_steps, dtype)
1328     nfev = len(n_steps) * 2
1329     ks = a2 + n_steps * step2
1330     fks = f(ks, *args2)
1331     fksp1 = f(ks + step2, *args2)  # check that the function is decreasing
1332     fk_insufficient = (fks > tol[:, xp.newaxis]) | (fksp1 > fks)
1333     n_fk_insufficient = xp.sum(fk_insufficient, axis=-1)
1334     nt = xp.minimum(n_fk_insufficient, xp.asarray(n_steps.shape[-1]-1))
1335     n_steps = n_steps[nt]
1336 
1337     # If `maxterms` is insufficient (i.e. either the magnitude of the last term of the
1338     # partial sum exceeds the tolerance or the function is not decreasing), finish the
1339     # calculation, but report nonzero status. (Improvement: separate the status codes
1340     # for these two cases.)
1341     i_fk_insufficient = (n_fk_insufficient == nfev//2)
1342 
1343     # Directly evaluate the sum up to this term
1344     k = a + n_steps * step
1345     left, left_error, left_nfev = _direct(f, a, k, step, args,
1346                                           constants, xp, inclusive=False)
1347     left_is_pos_inf = xp.isinf(left) & (left > 0)
1348     i_skip |= left_is_pos_inf  # if sum is infinite, no sense in continuing
1349     status[left_is_pos_inf] = -3
1350     k[i_skip] = xp.nan
1351 
1352     # Use integration to estimate the remaining sum
1353     # Possible optimization for future work: if there were no terms less than
1354     # the tolerance, there is no need to compute the integral to better accuracy.
1355     # Something like:
1356     # atol = xp.maximum(atol, xp.minimum(fk/2 - fb/2))
1357     # rtol = xp.maximum(rtol, xp.minimum((fk/2 - fb/2)/left))
1358     # where `fk`/`fb` are currently calculated below.
1359     right = tanhsinh(f, k, b, args=args, atol=atol, rtol=rtol, log=log)
1360 
1361     # Calculate the full estimate and error from the pieces
1362     fk = fks[xp.arange(len(fks)), nt]
1363 
1364     # fb = f(b, *args), but some functions return NaN at infinity.
1365     # instead of 0 like they must (for the sum to be convergent).
1366     fb = xp.full_like(fk, -xp.inf) if log else xp.zeros_like(fk)
1367     i = xp.isfinite(b)
1368     if xp.any(i):  # better not call `f` with empty arrays
1369         fb[i] = f(b[i], *[arg[i] for arg in args])
1370     nfev = nfev + xp.asarray(i, dtype=left_nfev.dtype)
1371 
1372     if log:
1373         log_step = xp.log(step)
1374         S_terms = (left, right.integral - log_step, fk - log2, fb - log2)
1375         S = special.logsumexp(xp.stack(S_terms), axis=0)
1376         E_terms = (left_error, right.error - log_step, fk-log2, fb-log2+xp.pi*1j)
1377         E = xp_real(special.logsumexp(xp.stack(E_terms), axis=0))
1378     else:
1379         S = left + right.integral/step + fk/2 + fb/2
1380         E = left_error + right.error/step + fk/2 - fb/2
1381     status[~i_skip] = right.status[~i_skip]
1382 
1383     status[(status == 0) & i_fk_insufficient] = -4
1384     return S, E, status, left_nfev + right.nfev + nfev + lb.nfev
