13   def _get_flat_args(
14       node: Node, node_to_additional_deps: dict[Node, OrderedSet[Node]]
15   ) -> list[Node]:
16       args = list[Any]()
17       map_arg((node.args, node.kwargs), args.append)
18       if node in node_to_additional_deps:
19           args.extend(node_to_additional_deps[node])
20       return args

23   def _get_flat_args_unique(
24       node: Node, node_to_additional_deps: dict[Node, OrderedSet[Node]]
25   ) -> OrderedSet[Node]:
26       args = OrderedSet[Node]()
27       map_arg((node.args, node.kwargs), args.add)
28       if node in node_to_additional_deps:
29           args.update(node_to_additional_deps[node])
30       return args

33   def _detect_cycles(
34       graph: Graph, node_to_additional_deps: dict[Node, OrderedSet[Node]]
35   ) -> str:
36       current_path: deque[Node] = deque()
37       current_path_set: set[Node] = set()
38       pending: deque[tuple[Node, Node]] = deque()
39   
40       def add_to_current_path(node: Node) -> None:
41           current_path.append(node)
42           current_path_set.add(node)
43   
44       def pop_current_path() -> None:
45           node = current_path.pop()
46           current_path_set.remove(node)
47   
48       def current_path_head() -> Node:
49           return current_path[-1]
50   
51       for origin in graph.find_nodes(op="output"):
52           current_path.clear()
53           current_path_set.clear()
54           add_to_current_path(origin)
55           for child in _get_flat_args_unique(origin, node_to_additional_deps):
56               pending.append((child, origin))
57   
58           while pending:
59               cur_node, parent = pending.pop()
60   
61               # handle backtracking
62               while current_path and current_path_head() != parent:
63                   pop_current_path()
64   
65               if not isinstance(cur_node, Node):
66                   continue
67   
68               if cur_node in current_path_set:
69                   current_path.append(cur_node)
70                   return f"cycle detected in path: {current_path}"
71   
72               add_to_current_path(cur_node)
73   
74               for child in _get_flat_args_unique(cur_node, node_to_additional_deps):
75                   pending.append((child, cur_node))
76   
77       return "no cycle detected"
