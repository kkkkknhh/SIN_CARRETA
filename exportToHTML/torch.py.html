4    """ 
5    This module implements variable tracking for torch functions and operations during Dynamo tracing. 
6     
7    It provides classes to handle different types of torch operations: 
8     
9    TorchInGraphFunctionVariable: Handles torch.* functions that should be captured in the FX graph. 
10   Provides special handling for constant folding, tensor methods, and torch function overrides. 
11   Manages complex cases like out= variants and parameter construction. 
12    
13   TorchCtxManagerClassVariable: Handles torch context managers like torch.no_grad(), autocast, etc. 
14   Provides implementations for entering/exiting these contexts during tracing. 
15    
16   DispatchKeySetVariable: Represents torch.DispatchKeySet for managing dispatch keys and 
17   device-specific operations during tracing. 
18    
19   The module includes special handling for: 
20   - Constant folding of pure functions 
21   - Tensor method calls 
22   - torch.nn.Parameter construction 
23   - __torch_function__ overrides 
24   - Context manager state tracking 
25   - Device and dtype management 
26    
27   This is a core part of Dynamo's tracing system, translating torch operations into 
28   traceable graph nodes while preserving correct semantics and handling edge cases. 
29   """

225  class BaseTorchVariable(VariableTracker):
226      """common base for all torch.* functions, classes, modules and other things"""
227  
228      @classmethod
229      def create_with_source(cls, value, source):
230          install_guard(source.make_guard(GuardBuilder.FUNCTION_MATCH))
231          return cls(value, source=source)
232  
233      def __init__(self, value, **kwargs) -> None:
234          super().__init__(**kwargs)
235          self.value = value
236  
237      def reconstruct(self, codegen: "PyCodegen"):
238          try:
239              name = f"{self.value.__module__}.{self.value.__name__}"
240          except Exception:
241              name = f"torch_obj_{id(self.value)}"
242          unique_var_name = "__" + re.sub(r"[^a-zA-Z0-9_]+", "_", name)
243          codegen.extend_output(
244              codegen.setup_globally_cached(unique_var_name, self.value)
245          )
246  
247      def as_proxy(self):
248          return self.value
249  
250      def as_python_constant(self):
251          return self.value
252  
253      def call_obj_hasattr(self, tx: "InstructionTranslator", name):
254          result = hasattr(self.value, name)
255          return variables.ConstantVariable.create(result)
256  
257      def can_constant_fold_through(self):
258          if self.value in constant_fold_functions:
259              return True
260          return getattr(self.value, "__module__", None) == "math"

263  class TorchCtxManagerClassVariable(BaseTorchVariable):
264      """Points to a context manager class in torch.* that dynamo has implementations"""
265  
266      def __repr__(self) -> str:
267          return f"TorchCtxManagerClassVariable({self.value})"
268  
269      @staticmethod
270      def is_matching_cls(value):
271          # Unwrap if it's a functools.lru_cache wrapper
272          value = unwrap_if_wrapper(value)
273          # We can't do isinstance(value, type) check because some ctx managers
274          # are implemented as a function decorated by contextlib.contextmanager,
275          # E.g., torch._functorch.vmap.vmap_increment_nesting.
276          return (
277              # Context manager type or function with @contextmanager is callable
278              callable(value)
279              and (
280                  hashable(value)  # accesses value.__hash__()
281                  and value in supported_ctx_manager_classes
282              )
283          )
284  
285      def call_function(
286          self,
287          tx: "InstructionTranslator",
288          args: Sequence[VariableTracker],
289          kwargs: "dict[str, VariableTracker]",
290      ) -> "VariableTracker":
291          from . import (
292              DisabledSavedTensorsHooksVariable,
293              DualLevelContextManager,
294              FSDPParamGroupUseTrainingStateVariable,
295              GradIncrementNestingCtxManagerVariable,
296              GradInplaceRequiresGradCtxManagerVariable,
297              GradModeVariable,
298              InferenceModeVariable,
299              JvpIncrementNestingCtxManagerVariable,
300              SDPAKernelVariable,
301              SetFwdGradEnabledContextManager,
302              StreamVariable,
303              VmapIncrementNestingCtxManagerVariable,
304          )
305  
306          if self.value is torch.no_grad:
307              if len(args) == 1 and isinstance(
308                  args[0], variables.functions.BaseUserFunctionVariable
309              ):
310                  ctx = GradModeVariable.create(tx, False)
311                  return ctx.call_function(tx, args, kwargs)
312              else:
313                  return GradModeVariable.create(tx, False)
314          elif self.value is torch.enable_grad:
315              if len(args) == 1 and isinstance(
316                  args[0], variables.functions.BaseUserFunctionVariable
317              ):
318                  ctx = GradModeVariable.create(tx, True)
319                  return ctx.call_function(tx, args, kwargs)
320              return GradModeVariable.create(tx, True)
321          elif self.value is torch.set_grad_enabled and len(args) == 1:
322              return GradModeVariable.create(
323                  tx, args[0].as_python_constant(), initialized=True
324              )
325          elif self.value is torch.inference_mode:
326              assert len(args) <= 1 and len(kwargs) == 0
327              inf_mode = args[0].as_python_constant() if len(args) == 1 else True
328              return InferenceModeVariable.create(tx, inf_mode)
329          elif inspect.isclass(self.value) and issubclass(self.value, torch.Stream):
330              from torch._dynamo.variables.builder import wrap_fx_proxy_cls
331  
332              return wrap_fx_proxy_cls(
333                  StreamVariable,
334                  tx,
335                  tx.output.create_proxy(
336                      "call_function",
337                      self.value,
338                      (),
339                      {},
340                  ),
341              )
342          elif self.value in (
343              torch.amp.autocast_mode.autocast,
344              torch.cuda.amp.autocast,
345              torch.cpu.amp.autocast,
346          ):
347              return AutocastModeVariable.create(self.value, args, kwargs)
348          elif self.value in (
349              # NOTE any class added here must align with the semantic
350              # requirements of `ProfilerContextVariable`.
351              torch.profiler.profile,
352              torch.profiler.record_function,
353              torch.autograd.profiler.profile,
354              torch.autograd.profiler.record_function,
355          ):
356              warning_once(log, "Profiler function %s will be ignored", self.value)
357              return ProfilerContextVariable()
358          elif (
359              self.value is torch._C.DisableTorchFunctionSubclass
360              or self.value is torch._C.DisableTorchFunction
361          ):
362              assert not (args or kwargs)
363              return TorchFunctionDisableVariable.create(
364                  tx, only_subclass=self.value is torch._C.DisableTorchFunctionSubclass
365              )
366          elif self.value is torch._functorch.vmap.vmap_increment_nesting:
367              assert len(args) == 2
368              return VmapIncrementNestingCtxManagerVariable.create(
369                  tx,
370                  args,
371              )
372          elif self.value is torch._functorch.eager_transforms.jvp_increment_nesting:
373              assert len(args) == 0
374              return JvpIncrementNestingCtxManagerVariable.create(tx)
375          elif self.value is torch.autograd.forward_ad._set_fwd_grad_enabled:
376              assert len(args) == 1
377              return SetFwdGradEnabledContextManager.create(
378                  tx,
379                  [guard_if_dyn(x) for x in args],
380              )
381          elif self.value is torch.autograd.forward_ad.dual_level:
382              assert len(args) == 0
383              return DualLevelContextManager.create(tx)
384          elif self.value is torch._functorch.eager_transforms.grad_increment_nesting:
385              assert len(args) == 0
386              return GradIncrementNestingCtxManagerVariable.create(tx)
387          elif (
388              self.value is torch._functorch.eager_transforms.enable_inplace_requires_grad
389          ):
390              assert len(args) == 1
391              return GradInplaceRequiresGradCtxManagerVariable.create(
392                  tx,
393                  [guard_if_dyn(x) for x in args],
394              )
395          elif self.value is torch.autograd.graph.disable_saved_tensors_hooks:
396              assert len(args) == 1
397              return DisabledSavedTensorsHooksVariable.create(
398                  tx, args[0].as_python_constant()
399              )
400          elif (
401              _fsdp_param_group is not None
402              and self.value is _fsdp_param_group.FSDPParamGroup.use_training_state
403          ):
404              assert len(args) == 2
405              return FSDPParamGroupUseTrainingStateVariable.create(
406                  tx, args[0], args[1].as_python_constant()
407              )
408          elif self.value is torch.nn.attention.sdpa_kernel:
409              assert len(args) == 1 or (len(kwargs) == 1 and "backends" in kwargs)
410              backends = args[0] if len(args) == 1 else kwargs["backends"]
411              set_priority = kwargs["set_priority"] if "set_priority" in kwargs else False
412              return SDPAKernelVariable.create(
413                  tx, backends.as_python_constant(), set_priority
414              )
415          elif self.value is torch.nn.attention._sdpa_kernel_variadic:
416              return SDPAKernelVariable.create(
417                  tx, [arg.as_python_constant() for arg in args]
418              )
419  
420          return super().call_function(tx, args, kwargs)

423  class TorchInGraphFunctionVariable(BaseTorchVariable):
424      """Points to a torch function/method that should be put in FX graph"""
425  
426      def __init__(self, value, nonstrict_traceable=None, **kwargs) -> None:
427          super().__init__(value, **kwargs)
428          from ..trace_rules import is_nonstrict_trace_callable
429  
430          if nonstrict_traceable is None:
431              nonstrict_traceable = is_nonstrict_trace_callable(value)
432          self.nonstrict_traceable = nonstrict_traceable
433  
434      def __repr__(self) -> str:
435          return f"TorchInGraphFunctionVariable({self.value}, nonstrict_traceable={self.nonstrict_traceable})"
436  
437      def get_function(self):
438          return self.value

440      @staticmethod
441      @functools.cache
442      def _get_handlers():
443          """Build a dict from function -> method to handle it so that we are O(1) 
444          in terms of the number of function with special handling."""
445          handlers = {}
446  
447          def register(*fns):
448              def _register(handler):
449                  for fn in fns:
450                      assert fn not in handlers, fn
451                      handlers[fn] = handler
452                  return handler
453  
454              assert callable(fns[0])
455              return _register
456  
457          from torch.backends.cuda import SDPAParams
458  
459          from . import (
460              ConstantVariable,
461              DeterministicAlgorithmsVariable,
462              GradModeVariable,
463              StreamContextVariable,
464              SymNodeVariable,
465              TensorVariable,
466              UserDefinedObjectVariable,
467          )
468          from .builder import wrap_fx_proxy, wrap_fx_proxy_cls

470          @register(*tracing_state_functions())
471          def handle_tracing_state_functions(
472              self, tx: "InstructionTranslator", *args, **kwargs
473          ):
474              assert not args and not kwargs
475              # See: https://github.com/pytorch/pytorch/issues/110765
476              if self.value in (
477                  torch._utils.is_compiling,
478                  torch._dynamo.external_utils.is_compiling,
479                  torch.compiler.is_compiling,
480                  torch.compiler.is_dynamo_compiling,
481                  torch.compiler.is_exporting,
482              ):
483                  tx.mark_inconsistent_side_effects()
484              return ConstantVariable.create(tracing_state_functions()[self.value])

486          @register(*dispatch_key_set_functions)
487          def handle_dispatch_key_set_functions(
488              self, tx: "InstructionTranslator", *args, **kwargs
489          ):
490              assert not kwargs
491              if self.value in (torch._C._dispatch_keys,):
492                  assert len(args) == 1
493                  assert isinstance(args[0], variables.TensorVariable)
494                  example_value = args[0].proxy.node.meta["example_value"]
495                  dks = self.value(example_value)
496                  # Remove Python and PythonTLSSnapshot from the dispatch key set,
497                  # as they originate from FakeTensor propagation.
498                  # This should only be done if the example_value is a FakeTensor.
499                  # However, if tensor subclasses are present,
500                  # it is reasonable for Python to remain in the dispatch key set.
501                  if isinstance(example_value, torch._subclasses.FakeTensor):
502                      dks = (
503                          dks
504                          - torch._C.DispatchKeySet(torch._C.DispatchKey.Python)
505                          - torch._C.DispatchKeySet(
506                              torch._C.DispatchKey.PythonTLSSnapshot
507                          )
508                      )
509                  return DispatchKeySetVariable.create(dks)
510              else:
511                  assert not args
512                  return DispatchKeySetVariable.create(self.value())

1114     def call_function(
1115         self,
1116         tx: "InstructionTranslator",
1117         args: Sequence[VariableTracker],
1118         kwargs: "dict[str, VariableTracker]",
1119     ) -> "VariableTracker":
1120         from . import ConstantVariable, SymNodeVariable, TensorVariable
1121         from .builder import wrap_fx_proxy
1122 
1123         if self.nonstrict_traceable:
1124             import torch._higher_order_ops.flat_apply as flat_apply
1125             from torch._higher_order_ops.flat_apply import (
1126                 func_to_graphable,
1127                 is_graphable_type,
1128             )
1129             from torch._subclasses.fake_tensor import fake_tensor_tls
1130             from torch.utils._pytree import tree_flatten
1131 
1132             from .base import AsPythonConstantNotImplementedError
1133 
1134             # 1. Convert `args, kwargs` into pytree-flattened proxy forms.
1135             #
1136             # Rather than reconstructing `args, kwargs` into python objects and
1137             # then tree_flatten them, we just let Dynamo symbolically interpret
1138             # `tree_flatten((args, kwargs))`. This saves us from having to
1139             # worry about the reconstruction logic, side effects, and guards.
1140             packed_input_vt = TupleVariable.build(
1141                 tx, (TupleVariable.build(tx, args), ConstDictVariable.build(tx, kwargs))
1142             )
1143             out_vt = variables.UserFunctionVariable(tree_flatten).call_function(
1144                 tx, [packed_input_vt], {}
1145             )
1146             assert isinstance(out_vt, TupleVariable) and len(out_vt.items) == 2
1147             flat_args_vts, input_spec_vt = out_vt.items
1148             assert isinstance(flat_args_vts, ListVariable)
1149 
1150             # Handle the case when the input contains a non-graphable type.
1151             for flat_arg_vt in flat_args_vts.items:
1152                 arg_type = flat_arg_vt.python_type()
1153                 if not is_graphable_type(arg_type):
1154                     type_name = flat_arg_vt.python_type().__qualname__
1155                     unimplemented(
1156                         f"""
1157 For `nonstrict_trace`-ed function, the only allowed input types are basic types (e.g., torch.Tensor, int, float) or pytree containers of those. Here you are calling the function with arguments that contain a value of type <{type_name}>, please use one of the following to register the type with pytree:
1158   * `torch.utils._pytree.register_constant`
1159   * `torch.utils._pytree.register_dataclass`
1160   * `torch.utils._pytree.register_pytree_node`
1161 """  # NOQA: B950
1162                     )
1163 
1164             # Since we checked with `is_graphable` above, `as_proxy` on the
1165             # flat_arg VT should always work.
1166             proxified_flat_args = [
1167                 flat_arg_vt.as_proxy() for flat_arg_vt in flat_args_vts.items
1168             ]
1169 
1170             # The downstream `flat_apply` call requires the input spec; however,
1171             # the spec not a graphable type, so we still have to reconstruct it
1172             # into a python object, and store it as a constant attribute on the
1173             # fx graph.
1174             try:
1175                 input_spec = input_spec_vt.as_python_constant()
1176             except AsPythonConstantNotImplementedError as e:
