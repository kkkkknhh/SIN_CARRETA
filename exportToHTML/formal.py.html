30   def rational_algorithm(f, x, k, order=4, full=False):
31       """ 
32       Rational algorithm for computing 
33       formula of coefficients of Formal Power Series 
34       of a function. 
35    
36       Explanation 
37       =========== 
38    
39       Applicable when f(x) or some derivative of f(x) 
40       is a rational function in x. 
41    
42       :func:`rational_algorithm` uses :func:`~.apart` function for partial fraction 
43       decomposition. :func:`~.apart` by default uses 'undetermined coefficients 
44       method'. By setting ``full=True``, 'Bronstein's algorithm' can be used 
45       instead. 
46    
47       Looks for derivative of a function up to 4'th order (by default). 
48       This can be overridden using order option. 
49    
50       Parameters 
51       ========== 
52    
53       x : Symbol 
54       order : int, optional 
55           Order of the derivative of ``f``, Default is 4. 
56       full : bool 
57    
58       Returns 
59       ======= 
60    
61       formula : Expr 
62       ind : Expr 
63           Independent terms. 
64       order : int 
65       full : bool 
66    
67       Examples 
68       ======== 
69    
70       >>> from sympy import log, atan 
71       >>> from sympy.series.formal import rational_algorithm as ra 
72       >>> from sympy.abc import x, k 
73    
74       >>> ra(1 / (1 - x), x, k) 
75       (1, 0, 0) 
76       >>> ra(log(1 + x), x, k) 
77       (-1/((-1)**k*k), 0, 1) 
78    
79       >>> ra(atan(x), x, k, full=True) 
80       ((-I/(2*(-I)**k) + I/(2*I**k))/k, 0, 1) 
81    
82       Notes 
83       ===== 
84    
85       By setting ``full=True``, range of admissible functions to be solved using 
86       ``rational_algorithm`` can be increased. This option should be used 
87       carefully as it can significantly slow down the computation as ``doit`` is 
88       performed on the :class:`~.RootSum` object returned by the :func:`~.apart` 
89       function. Use ``full=False`` whenever possible. 
90    
91       See Also 
92       ======== 
93    
94       sympy.polys.partfrac.apart 
95    
96       References 
97       ========== 
98    
99       .. [1] Formal Power Series - Dominik Gruntz, Wolfram Koepf 
100      .. [2] Power Series in Computer Algebra - Wolfram Koepf 
101   
102      """
103      from sympy.polys import RootSum, apart
104      from sympy.integrals import integrate
105  
106      diff = f
107      ds = []  # list of diff
108  
109      for i in range(order + 1):
110          if i:
111              diff = diff.diff(x)
112  
113          if diff.is_rational_function(x):
114              coeff, sep = S.Zero, S.Zero
115  
116              terms = apart(diff, x, full=full)
117              if terms.has(RootSum):
118                  terms = terms.doit()
119  
120              for t in Add.make_args(terms):
121                  num, den = t.as_numer_denom()
122                  if not den.has(x):
123                      sep += t
124                  else:
125                      if isinstance(den, Mul):
126                          # m*(n*x - a)**j -> (n*x - a)**j
127                          ind = den.as_independent(x)
128                          den = ind[1]
129                          num /= ind[0]
130  
131                      # (n*x - a)**j -> (x - b)
132                      den, j = den.as_base_exp()
133                      a, xterm = den.as_coeff_add(x)
134  
135                      # term -> m/x**n
136                      if not a:
137                          sep += t
138                          continue
139  
140                      xc = xterm[0].coeff(x)
141                      a /= -xc
142                      num /= xc**j
143  
144                      ak = ((-1)**j * num *
145                            binomial(j + k - 1, k).rewrite(factorial) /
146                            a**(j + k))
147                      coeff += ak
148  
149              # Hacky, better way?
150              if coeff.is_zero:
151                  return None
152              if (coeff.has(x) or coeff.has(zoo) or coeff.has(oo) or
153                      coeff.has(nan)):
154                  return None
155  
156              for j in range(i):
157                  coeff = (coeff / (k + j + 1))
158                  sep = integrate(sep, x)
159                  sep += (ds.pop() - sep).limit(x, 0)  # constant of integration
160              return (coeff.subs(k, k - i), sep, i)
161  
162          else:
163              ds.append(diff)
164  
165      return None

168  def rational_independent(terms, x):
169      """ 
170      Returns a list of all the rationally independent terms. 
171   
172      Examples 
173      ======== 
174   
175      >>> from sympy import sin, cos 
176      >>> from sympy.series.formal import rational_independent 
177      >>> from sympy.abc import x 
178   
179      >>> rational_independent([cos(x), sin(x)], x) 
180      [cos(x), sin(x)] 
181      >>> rational_independent([x**2, sin(x), x*sin(x), x**3], x) 
182      [x**3 + x**2, x*sin(x) + sin(x)] 
183      """
184      if not terms:
185          return []
186  
187      ind = terms[0:1]
188  
189      for t in terms[1:]:
190          n = t.as_independent(x)[1]
191          for i, term in enumerate(ind):
192              d = term.as_independent(x)[1]
193              q = (n / d).cancel()
194              if q.is_rational_function(x):
195                  ind[i] += t
196                  break
197          else:
198              ind.append(t)
199      return ind

202  def simpleDE(f, x, g, order=4):
203      r""" 
204      Generates simple DE. 
205   
206      Explanation 
207      =========== 
208   
209      DE is of the form 
210   
211      .. math:: 
212          f^k(x) + \sum\limits_{j=0}^{k-1} A_j f^j(x) = 0 
213   
214      where :math:`A_j` should be rational function in x. 
215   
216      Generates DE's upto order 4 (default). DE's can also have free parameters. 
217   
218      By increasing order, higher order DE's can be found. 
219   
220      Yields a tuple of (DE, order). 
221      """
222      from sympy.solvers.solveset import linsolve
223  
224      a = symbols('a:%d' % (order))
225  
226      def _makeDE(k):
227          eq = f.diff(x, k) + Add(*[a[i]*f.diff(x, i) for i in range(0, k)])
228          DE = g(x).diff(x, k) + Add(*[a[i]*g(x).diff(x, i) for i in range(0, k)])
229          return eq, DE
230  
231      found = False
232      for k in range(1, order + 1):
233          eq, DE = _makeDE(k)
234          eq = eq.expand()
235          terms = eq.as_ordered_terms()
236          ind = rational_independent(terms, x)
237          if found or len(ind) == k:
238              sol = dict(zip(a, (i for s in linsolve(ind, a[:k]) for i in s)))
239              if sol:
240                  found = True
241                  DE = DE.subs(sol)
242              DE = DE.as_numer_denom()[0]
243              DE = DE.factor().as_coeff_mul(Derivative)[1][0]
244              yield DE.collect(Derivative(g(x))), k

247  def exp_re(DE, r, k):
248      """Converts a DE with constant coefficients (explike) into a RE. 
249   
250      Explanation 
251      =========== 
252   
253      Performs the substitution: 
254   
255      .. math:: 
256          f^j(x) \\to r(k + j) 
257   
258      Normalises the terms so that lowest order of a term is always r(k). 
259   
260      Examples 
261      ======== 
262   
263      >>> from sympy import Function, Derivative 
264      >>> from sympy.series.formal import exp_re 
265      >>> from sympy.abc import x, k 
266      >>> f, r = Function('f'), Function('r') 
267   
268      >>> exp_re(-f(x) + Derivative(f(x)), r, k) 
269      -r(k) + r(k + 1) 
270      >>> exp_re(Derivative(f(x), x) + Derivative(f(x), (x, 2)), r, k) 
271      r(k) + r(k + 1) 
272   
273      See Also 
274      ======== 
275   
276      sympy.series.formal.hyper_re 
277      """
278      RE = S.Zero
279  
280      g = DE.atoms(Function).pop()
281  
282      mini = None
283      for t in Add.make_args(DE):
284          coeff, d = t.as_independent(g)
285          if isinstance(d, Derivative):
286              j = d.derivative_count
287          else:
288              j = 0
289          if mini is None or j < mini:
290              mini = j
291          RE += coeff * r(k + j)
292      if mini:
293          RE = RE.subs(k, k - mini)
294      return RE

297  def hyper_re(DE, r, k):
298      """ 
299      Converts a DE into a RE. 
300   
301      Explanation 
302      =========== 
303   
304      Performs the substitution: 
305   
306      .. math:: 
307          x^l f^j(x) \\to (k + 1 - l)_j . a_{k + j - l} 
308   
309      Normalises the terms so that lowest order of a term is always r(k). 
310   
311      Examples 
312      ======== 
313   
314      >>> from sympy import Function, Derivative 
315      >>> from sympy.series.formal import hyper_re 
316      >>> from sympy.abc import x, k 
317      >>> f, r = Function('f'), Function('r') 
318   
319      >>> hyper_re(-f(x) + Derivative(f(x)), r, k) 
320      (k + 1)*r(k + 1) - r(k) 
321      >>> hyper_re(-x*f(x) + Derivative(f(x), (x, 2)), r, k) 
322      (k + 2)*(k + 3)*r(k + 3) - r(k) 
323   
324      See Also 
325      ======== 
326   
327      sympy.series.formal.exp_re 
328      """
329      RE = S.Zero
330  
331      g = DE.atoms(Function).pop()
332      x = g.atoms(Symbol).pop()
333  
334      mini = None
335      for t in Add.make_args(DE.expand()):
336          coeff, d = t.as_independent(g)
337          c, v = coeff.as_independent(x)
338          l = v.as_coeff_exponent(x)[1]
339          if isinstance(d, Derivative):
340              j = d.derivative_count
341          else:
342              j = 0
343          RE += c * rf(k + 1 - l, j) * r(k + j - l)
344          if mini is None or j - l < mini:
345              mini = j - l
346  
347      RE = RE.subs(k, k - mini)
348  
349      m = Wild('m')
350      return RE.collect(r(k + m))

353  def _transformation_a(f, x, P, Q, k, m, shift):
354      f *= x**(-shift)
355      P = P.subs(k, k + shift)
356      Q = Q.subs(k, k + shift)
357      return f, P, Q, m

360  def _transformation_c(f, x, P, Q, k, m, scale):
361      f = f.subs(x, x**scale)
362      P = P.subs(k, k / scale)
363      Q = Q.subs(k, k / scale)
364      m *= scale
365      return f, P, Q, m

369      f = f.diff(x)
370      P = P.subs(k, k + 1) * (k + m + 1)
371      Q = Q.subs(k, k + 1) * (k + 1)

388  def _compute_formula(f, x, P, Q, k, m, k_max):
389      """Computes the formula for f."""
390      from sympy.polys import roots
391  
392      sol = []
393      for i in range(k_max + 1, k_max + m + 1):
394          if (i < 0) == True:
395              continue
396          r = f.diff(x, i).limit(x, 0) / factorial(i)
397          if r.is_zero:
398              continue
399  
400          kterm = m*k + i
401          res = r
402  
403          p = P.subs(k, kterm)
404          q = Q.subs(k, kterm)
405          c1 = p.subs(k, 1/k).leadterm(k)[0]
406          c2 = q.subs(k, 1/k).leadterm(k)[0]
407          res *= (-c1 / c2)**k
408  
409          res *= Mul(*[rf(-r, k)**mul for r, mul in roots(p, k).items()])
410          res /= Mul(*[rf(-r, k)**mul for r, mul in roots(q, k).items()])
411  
412          sol.append((res, kterm))
413  
414      return sol

417  def _rsolve_hypergeometric(f, x, P, Q, k, m):
418      """ 
419      Recursive wrapper to rsolve_hypergeometric. 
420   
421      Explanation 
422      =========== 
423   
424      Returns a Tuple of (formula, series independent terms, 
425      maximum power of x in independent terms) if successful 
426      otherwise ``None``. 
427   
428      See :func:`rsolve_hypergeometric` for details. 
429      """
430      from sympy.polys import lcm, roots
431      from sympy.integrals import integrate
432  
433      # transformation - c
434      proots, qroots = roots(P, k), roots(Q, k)
435      all_roots = dict(proots)
436      all_roots.update(qroots)
437      scale = lcm([r.as_numer_denom()[1] for r, t in all_roots.items()
438                   if r.is_rational])
439      f, P, Q, m = _transformation_c(f, x, P, Q, k, m, scale)
440  
441      # transformation - a
442      qroots = roots(Q, k)
443      if qroots:
444          k_min = Min(*qroots.keys())
445      else:
446          k_min = S.Zero
447      shift = k_min + m
448      f, P, Q, m = _transformation_a(f, x, P, Q, k, m, shift)
449  
450      l = (x*f).limit(x, 0)
451      if not isinstance(l, Limit) and l != 0:  # Ideally should only be l != 0
452          return None
453  
454      qroots = roots(Q, k)
455      if qroots:
456          k_max = Max(*qroots.keys())
457      else:
458          k_max = S.Zero
459  
460      ind, mp = S.Zero, -oo
461      for i in range(k_max + m + 1):
462          r = f.diff(x, i).limit(x, 0) / factorial(i)
463          if r.is_finite is False:
464              old_f = f
465              f, P, Q, m = _transformation_a(f, x, P, Q, k, m, i)
466              f, P, Q, m = _transformation_e(f, x, P, Q, k, m)
467              sol, ind, mp = _rsolve_hypergeometric(f, x, P, Q, k, m)
468              sol = _apply_integrate(sol, x, k)
469              sol = _apply_shift(sol, i)
470              ind = integrate(ind, x)
471              ind += (old_f - ind).limit(x, 0)  # constant of integration
472              mp += 1
473              return sol, ind, mp
474          elif r:
475              ind += r*x**(i + shift)
476              pow_x = Rational((i + shift), scale)
477              if pow_x > mp:
478                  mp = pow_x  # maximum power of x
479      ind = ind.subs(x, x**(1/scale))
480  
481      sol = _compute_formula(f, x, P, Q, k, m, k_max)
482      sol = _apply_shift(sol, shift)
483      sol = _apply_scale(sol, scale)
484  
485      return sol, ind, mp

488  def rsolve_hypergeometric(f, x, P, Q, k, m):
489      """ 
490      Solves RE of hypergeometric type. 
491   
492      Explanation 
493      =========== 
494   
495      Attempts to solve RE of the form 
496   
497      Q(k)*a(k + m) - P(k)*a(k) 
498   
499      Transformations that preserve Hypergeometric type: 
500   
501          a. x**n*f(x): b(k + m) = R(k - n)*b(k) 
502          b. f(A*x): b(k + m) = A**m*R(k)*b(k) 
503          c. f(x**n): b(k + n*m) = R(k/n)*b(k) 
504          d. f(x**(1/m)): b(k + 1) = R(k*m)*b(k) 
505          e. f'(x): b(k + m) = ((k + m + 1)/(k + 1))*R(k + 1)*b(k) 
506   
507      Some of these transformations have been used to solve the RE. 
508   
509      Returns 
510      ======= 
511   
512      formula : Expr 
513      ind : Expr 
514          Independent terms. 
515      order : int 
516   
517      Examples 
518      ======== 
519   
520      >>> from sympy import exp, ln, S 
521      >>> from sympy.series.formal import rsolve_hypergeometric as rh 
522      >>> from sympy.abc import x, k 
523   
524      >>> rh(exp(x), x, -S.One, (k + 1), k, 1) 
525      (Piecewise((1/factorial(k), Eq(Mod(k, 1), 0)), (0, True)), 1, 1) 
526   
527      >>> rh(ln(1 + x), x, k**2, k*(k + 1), k, 1) 
528      (Piecewise(((-1)**(k - 1)*factorial(k - 1)/RisingFactorial(2, k - 1), 
529       Eq(Mod(k, 1), 0)), (0, True)), x, 2) 
530   
531      References 
532      ========== 
533   
534      .. [1] Formal Power Series - Dominik Gruntz, Wolfram Koepf 
535      .. [2] Power Series in Computer Algebra - Wolfram Koepf 
536      """
537      result = _rsolve_hypergeometric(f, x, P, Q, k, m)
538  
539      if result is None:
540          return None
541  
542      sol_list, ind, mp = result
543  
544      sol_dict = defaultdict(lambda: S.Zero)
545      for res, cond in sol_list:
546          j, mk = cond.as_coeff_Add()
547          c = mk.coeff(k)
548  
549          if j.is_integer is False:
550              res *= x**frac(j)
551              j = floor(j)
552  
553          res = res.subs(k, (k - j) / c)
554          cond = Eq(k % c, j % c)
555          sol_dict[cond] += res  # Group together formula for same conditions
556  
557      sol = [(res, cond) for cond, res in sol_dict.items()]
558      sol.append((S.Zero, True))
559      sol = Piecewise(*sol)
560  
561      if mp is -oo:
562          s = S.Zero
563      elif mp.is_integer is False:
564          s = ceiling(mp)
565      else:
566          s = mp + 1
567  
568      #  save all the terms of
569      #  form 1/x**k in ind
570      if s < 0:
571          ind += sum(sequence(sol * x**k, (k, s, -1)))
572          s = S.Zero
573  
574      return (sol, ind, s)

578      """See docstring of :func:`rsolve_hypergeometric` for details."""

586              P, Q = Q, P
587              m = abs(m)
588          return rsolve_hypergeometric(f, x, P, Q, k, m)

591  def _solve_explike_DE(f, x, DE, g, k):
592      """Solves DE with constant coefficients."""
593      from sympy.solvers import rsolve
594  
595      for t in Add.make_args(DE):
596          coeff, d = t.as_independent(g)
597          if coeff.free_symbols:
598              return
599  
600      RE = exp_re(DE, g, k)
601  
602      init = {}
603      for i in range(len(Add.make_args(RE))):
604          if i:
605              f = f.diff(x)
606          init[g(k).subs(k, i)] = f.limit(x, 0)
607  
608      sol = rsolve(RE, g(k), init)
609  
610      if sol:
611          return (sol / factorial(k), S.Zero, S.Zero)

632  def _transform_explike_DE(DE, g, x, order, syms):
633      """Converts DE with free parameters into DE with constant coefficients."""
634      from sympy.solvers.solveset import linsolve
635  
636      eq = []
637      highest_coeff = DE.coeff(Derivative(g(x), x, order))
638      for i in range(order):
639          coeff = DE.coeff(Derivative(g(x), x, i))
640          coeff = (coeff / highest_coeff).expand().collect(x)
641          eq.extend(Add.make_args(coeff))
642      temp = []
643      for e in eq:
644          if e.has(x):
645              break
646          elif e.has(Symbol):
647              temp.append(e)
648      else:
649          eq = temp
650      if eq:
651          sol = dict(zip(syms, (i for s in linsolve(eq, list(syms)) for i in s)))
652          if sol:
653              DE = DE.subs(sol)
654              DE = DE.factor().as_coeff_mul(Derivative)[1][0]
655              DE = DE.collect(Derivative(g(x)))
656      return DE

659  def _transform_DE_RE(DE, g, k, order, syms):
660      """Converts DE with free parameters into RE of hypergeometric type."""
661      from sympy.solvers.solveset import linsolve
662  
663      RE = hyper_re(DE, g, k)
664  
665      eq = [RE.coeff(g(k + i)) for i in range(1, order)]
666      sol = dict(zip(syms, (i for s in linsolve(eq, list(syms)) for i in s)))
667      if sol:
668          m = Wild('m')
669          RE = RE.subs(sol)
670          RE = RE.factor().as_numer_denom()[0].collect(g(k + m))
671          RE = RE.as_coeff_mul(g)[1][0]
672          for i in range(order):  # smallest order should be g(k)
673              if RE.coeff(g(k + i)) and i:
674                  RE = RE.subs(k, k - i)
675                  break
676      return RE

679  def solve_de(f, x, DE, order, g, k):
680      """ 
681      Solves the DE. 
682   
683      Explanation 
684      =========== 
685   
686      Tries to solve DE by either converting into a RE containing two terms or 
687      converting into a DE having constant coefficients. 
688   
689      Returns 
690      ======= 
691   
692      formula : Expr 
693      ind : Expr 
694          Independent terms. 
695      order : int

697      Examples 
698      ======== 
699

701      >>> from sympy import exp, ln 
702      >>> from sympy.series.formal import solve_de 
703      >>> from sympy.abc import x, k 
704      >>> f = Function('f') 
705   
706      >>> solve_de(exp(x), x, D(f(x), x) - f(x), 1, f, k) 
707      (Piecewise((1/factorial(k), Eq(Mod(k, 1), 0)), (0, True)), 1, 1) 
708   
709      >>> solve_de(ln(1 + x), x, (x + 1)*D(f(x), x, 2) + D(f(x)), 2, f, k) 
710      (Piecewise(((-1)**(k - 1)*factorial(k - 1)/RisingFactorial(2, k - 1), 
711       Eq(Mod(k, 1), 0)), (0, True)), x, 2) 
712      """
713      sol = None
714      syms = DE.free_symbols.difference({g, x})
715  
716      if syms:
717          RE = _transform_DE_RE(DE, g, k, order, syms)
718      else:
719          RE = hyper_re(DE, g, k)
720      if not RE.free_symbols.difference({k}):
721          sol = _solve_hyper_RE(f, x, RE, g, k)
722  
723      if sol:
724          return sol
725  
726      if syms:
727          DE = _transform_explike_DE(DE, g, x, order, syms)
728      if not DE.free_symbols.difference({x}):
729          sol = _solve_explike_DE(f, x, DE, g, k)
730  
731      if sol:
732          return sol

736      """ 
737      Hypergeometric algorithm for computing Formal Power Series. 
738   
739      Explanation 
740      =========== 
741   
742      Steps: 
743          * Generates DE 
744          * Convert the DE into RE 
745          * Solves the RE 
746   
747      Examples 
748      ======== 
749   
750      >>> from sympy import exp, ln 
751      >>> from sympy.series.formal import hyper_algorithm 
752   
753      >>> from sympy.abc import x, k 
754   
755      >>> hyper_algorithm(exp(x), x, k) 
756      (Piecewise((1/factorial(k), Eq(Mod(k, 1), 0)), (0, True)), 1, 1) 
757   
758      >>> hyper_algorithm(ln(1 + x), x, k) 
759      (Piecewise(((-1)**(k - 1)*factorial(k - 1)/RisingFactorial(2, k - 1), 
760       Eq(Mod(k, 1), 0)), (0, True)), x, 2) 
761   
762      See Also 
763      ======== 
764   
765      sympy.series.formal.simpleDE 
766      sympy.series.formal.solve_de 
767      """
768      g = Function('g')
769  
770      des = []  # list of DE's
771      sol = None
772      for DE, i in simpleDE(f, x, g, order):
773          if DE is not None:
774              sol = solve_de(f, x, DE, i, g, k)
775          if sol:
776              return sol
777          if not DE.free_symbols.difference({x}):
778              des.append(DE)
779  
780      # If nothing works
781      # Try plain rsolve
782      for DE in des:
783          sol = _solve_simple(f, x, DE, g, k)
784          if sol:
785              return sol

788  def _compute_fps(f, x, x0, dir, hyper, order, rational, full):
789      """Recursive wrapper to compute fps. 
790   
791      See :func:`compute_fps` for details. 
792      """
793      if x0 in [S.Infinity, S.NegativeInfinity]:
794          dir = S.One if x0 is S.Infinity else -S.One
795          temp = f.subs(x, 1/x)
796          result = _compute_fps(temp, x, 0, dir, hyper, order, rational, full)
797          if result is None:
798              return None
799          return (result[0], result[1].subs(x, 1/x), result[2].subs(x, 1/x))
800      elif x0 or dir == -S.One:
801          if dir == -S.One:
802              rep = -x + x0
803              rep2 = -x
804              rep2b = x0
805          else:
806              rep = x + x0
807              rep2 = x
808              rep2b = -x0
809          temp = f.subs(x, rep)
810          result = _compute_fps(temp, x, 0, S.One, hyper, order, rational, full)
811          if result is None:
812              return None
813          return (result[0], result[1].subs(x, rep2 + rep2b),
814                  result[2].subs(x, rep2 + rep2b))
815  
816      if f.is_polynomial(x):
817          k = Dummy('k')
818          ak = sequence(Coeff(f, x, k), (k, 1, oo))
819          xk = sequence(x**k, (k, 0, oo))
820          ind = f.coeff(x, 0)
821          return ak, xk, ind
822  
823      #  Break instances of Add
824      #  this allows application of different
825      #  algorithms on different terms increasing the
826      #  range of admissible functions.
827      if isinstance(f, Add):
828          result = False
829          ak = sequence(S.Zero, (0, oo))
830          ind, xk = S.Zero, None
831          for t in Add.make_args(f):
832              res = _compute_fps(t, x, 0, S.One, hyper, order, rational, full)
833              if res:
834                  if not result:
835                      result = True
836                      xk = res[1]
837                  if res[0].start > ak.start:
838                      seq = ak
839                      s, f = ak.start, res[0].start
840                  else:
841                      seq = res[0]
842                      s, f = res[0].start, ak.start
843                  save = Add(*[z[0]*z[1] for z in zip(seq[0:(f - s)], xk[s:f])])
844                  ak += res[0]
845                  ind += res[2] + save
846              else:
847                  ind += t
848          if result:
849              return ak, xk, ind
850          return None
851  
852      # The symbolic term - symb, if present, is being separated from the function
853      # Otherwise symb is being set to S.One
854      syms = f.free_symbols.difference({x})
855      (f, symb) = expand(f).as_independent(*syms)
856  
857      result = None
858  
859      # from here on it's x0=0 and dir=1 handling
860      k = Dummy('k')
861      if rational:
862          result = rational_algorithm(f, x, k, order, full)
863  
864      if result is None and hyper:
865          result = hyper_algorithm(f, x, k, order)
866  
867      if result is None:
868          return None
869  
870      from sympy.simplify.powsimp import powsimp
871      if symb.is_zero:
872          symb = S.One
873      else:
874          symb = powsimp(symb)
875      ak = sequence(result[0], (k, result[2], oo))
876      xk_formula = powsimp(x**k * symb)
877      xk = sequence(xk_formula, (k, 0, oo))
878      ind = powsimp(result[1] * symb)
879  
880      return ak, xk, ind

883  def compute_fps(f, x, x0=0, dir=1, hyper=True, order=4, rational=True,
884                  full=False):
885      """ 
886      Computes the formula for Formal Power Series of a function. 
887   
888      Explanation 
889      =========== 
890   
891      Tries to compute the formula by applying the following techniques 
892      (in order): 
893   
894      * rational_algorithm 
895      * Hypergeometric algorithm 
896   
897      Parameters 
898      ========== 
899   
900      x : Symbol 
901      x0 : number, optional 
902          Point to perform series expansion about. Default is 0. 
903      dir : {1, -1, '+', '-'}, optional 
904          If dir is 1 or '+' the series is calculated from the right and 
905          for -1 or '-' the series is calculated from the left. For smooth 
906          functions this flag will not alter the results. Default is 1. 
907      hyper : {True, False}, optional 
908          Set hyper to False to skip the hypergeometric algorithm. 
909          By default it is set to False. 
910      order : int, optional 
911          Order of the derivative of ``f``, Default is 4. 
912      rational : {True, False}, optional 
913          Set rational to False to skip rational algorithm. By default it is set 
914          to True. 
915      full : {True, False}, optional 
916          Set full to True to increase the range of rational algorithm. 
917          See :func:`rational_algorithm` for details. By default it is set to 
918          False. 
919   
920      Returns 
921      ======= 
922   
923      ak : sequence 
924          Sequence of coefficients. 
925      xk : sequence 
926          Sequence of powers of x. 
927      ind : Expr 
928          Independent terms. 
929      mul : Pow 
930          Common terms. 
931   
932      See Also 
933      ======== 
934   
935      sympy.series.formal.rational_algorithm 
936      sympy.series.formal.hyper_algorithm 
937      """
938      f = sympify(f)
939      x = sympify(x)
940  
941      if not f.has(x):
942          return None
943  
944      x0 = sympify(x0)
945  
946      if dir == '+':
947          dir = S.One
948      elif dir == '-':
949          dir = -S.One
950      elif dir not in [S.One, -S.One]:
951          raise ValueError("Dir must be '+' or '-'")
952      else:
953          dir = sympify(dir)
954  
955      return _compute_fps(f, x, x0, dir, hyper, order, rational, full)

1262     def coeff_bell(self, n):
1263         r""" 
1264         self.coeff_bell(n) returns a sequence of Bell polynomials of the second kind. 
1265         Note that ``n`` should be a integer. 
1266  
1267         The second kind of Bell polynomials (are sometimes called "partial" Bell 
1268         polynomials or incomplete Bell polynomials) are defined as

1271             B_{n,k}(x_1, x_2,\dotsc x_{n-k+1}) = 
1272                 \sum_{j_1+j_2+j_2+\dotsb=k \atop j_1+2j_2+3j_2+\dotsb=n} 
1273                 \frac{n!}{j_1!j_2!\dotsb j_{n-k+1}!} 
1274                 \left(\frac{x_1}{1!} \right)^{j_1} 
1275                 \left(\frac{x_2}{2!} \right)^{j_2} \dotsb 
1276                 \left(\frac{x_{n-k+1}}{(n-k+1)!} \right) ^{j_{n-k+1}}. 
1277  
1278         * ``bell(n, k, (x1, x2, ...))`` gives Bell polynomials of the second kind, 
1279           `B_{n,k}(x_1, x_2, \dotsc, x_{n-k+1})`. 
1280  
1281         See Also 
1282         ======== 
1283  
1284         sympy.functions.combinatorial.numbers.bell 
1285  
1286         """

1291         return sequence(tuple(inner_coeffs), (k, 1, oo))

1380             \sum\limits_{k=0}^{n} (-1)^{k} x_0^{-k-1} B_{n,k}(x_1, x_2, \dotsc, x_{n-k+1})

1387             be truncated.

1391

1398         1 - x + x**2/2 - x**3/6 + x**4/24 - x**5/120 + O(x**6) 
1399  
1400         >>> f2.inverse(x).truncate(8) 
1401         1 + x**2/2 + 5*x**4/24 + 61*x**6/720 + O(x**8)

1404         ======== 
1405  
1406         sympy.functions.combinatorial.numbers.bell

1410         ========== 
1411  
1412         .. [1] Comtet, Louis: Advanced combinatorics; the art of finite and infinite expansions. Reidel, 1974.

1414         """
1415 
1416         if n is None:
1417             return iter(self)
1418 
1419         if self._eval_term(0).is_zero:
1420             raise ValueError("Constant coefficient should exist for an inverse of a formal"
1421                 " power series to exist.")
1422 
1423         return FormalPowerSeriesInverse(self)

1425     def __add__(self, other):
1426         other = sympify(other)
1427 
1428         if isinstance(other, FormalPowerSeries):
1429             if self.dir != other.dir:
1430                 raise ValueError("Both series should be calculated from the"
1431                                  " same direction.")
1432             elif self.x0 != other.x0:
1433                 raise ValueError("Both series should be calculated about the"
1434                                  " same point.")
1435 
1436             x, y = self.x, other.x
1437             f = self.function + other.function.subs(y, x)
1438 
1439             if self.x not in f.free_symbols:
1440                 return f
1441 
1442             ak = self.ak + other.ak
1443             if self.ak.start > other.ak.start:
1444                 seq = other.ak
1445                 s, e = other.ak.start, self.ak.start
1446             else:
1447                 seq = self.ak
1448                 s, e = self.ak.start, other.ak.start
1449             save = Add(*[z[0]*z[1] for z in zip(seq[0:(e - s)], self.xk[s:e])])
1450             ind = self.ind + other.ind + save
1451 
1452             return self.func(f, x, self.x0, self.dir, (ak, self.xk, ind))
1453 
1454         elif not other.has(self.x):
1455             f = self.function + other
1456             ind = self.ind + other
1457 
1458             return self.func(f, self.x, self.x0, self.dir,
1459                              (self.ak, self.xk, ind))
1460 
1461         return Add(self, other)
1462

1464         return self.__add__(other)
1465

1467         return self.func(-self.function, self.x, self.x0, self.dir,
1468                          (-self.ak, self.xk, -self.ind))

1471         return self.__add__(-other)
1472

1474         return (-self).__add__(other)

1477         other = sympify(other)
1478 
1479         if other.has(self.x):
1480             return Mul(self, other)
1481 
1482         f = self.function * other
1483         ak = self.ak.coeff_mul(other)
1484         ind = self.ind * other
1485 
1486         return self.func(f, self.x, self.x0, self.dir, (ak, self.xk, ind))

1544     """Represents the product of two formal power series of two functions. 
1545  
1546     Explanation 
1547     =========== 
1548  
1549     No computation is performed. Terms are calculated using a term by term logic, 
1550     instead of a point by point logic. 
1551  
1552     There are two differences between a :obj:`FormalPowerSeries` object and a 
1553     :obj:`FormalPowerSeriesProduct` object. The first argument contains the two 
1554     functions involved in the product. Also, the coefficient sequence contains 
1555     both the coefficient sequence of the formal power series of the involved functions. 
1556  
1557     See Also 
1558     ======== 
1559  
1560     sympy.series.formal.FormalPowerSeries 
1561     sympy.series.formal.FiniteFormalPowerSeries 
1562  
1563     """
1564 
1565     def __init__(self, *args):
1566         ffps, gfps = self.ffps, self.gfps
1567 
1568         k = ffps.ak.variables[0]
1569         self.coeff1 = sequence(ffps.ak.formula, (k, 0, oo))
1570 
1571         k = gfps.ak.variables[0]
1572         self.coeff2 = sequence(gfps.ak.formula, (k, 0, oo))
1573 
1574     @property
1575     def function(self):
1576         """Function of the product of two formal power series."""
1577         return self.f * self.g

1579     def _eval_terms(self, n):
1580         """ 
1581         Returns the first ``n`` terms of the product formal power series. 
1582         Term by term logic is implemented here. 
1583  
1584         Examples 
1585         ======== 
1586

1598  
1599         sympy.series.formal.FormalPowerSeries.product 
1600  
1601         """

1603 
1604         aks = convolution(coeff1[:n], coeff2[:n])
1605 
1606         terms = []
1607         for i in range(0, n):
1608             terms.append(aks[i] * self.ffps.xk.coeff(i))

1613 class FormalPowerSeriesCompose(FiniteFormalPowerSeries):
1614     """ 
1615     Represents the composed formal power series of two functions. 
1616  
1617     Explanation 
1618     =========== 
1619  
1620     No computation is performed. Terms are calculated using a term by term logic, 
1621     instead of a point by point logic. 
1622  
1623     There are two differences between a :obj:`FormalPowerSeries` object and a 
1624     :obj:`FormalPowerSeriesCompose` object. The first argument contains the outer 
1625     function and the inner function involved in the omposition. Also, the 
1626     coefficient sequence contains the generic sequence which is to be multiplied 
1627     by a custom ``bell_seq`` finite sequence. The finite terms will then be added up to 
1628     get the final terms. 
1629  
1630     See Also 
1631     ======== 
1632  
1633     sympy.series.formal.FormalPowerSeries 
1634     sympy.series.formal.FiniteFormalPowerSeries 
1635  
1636     """
1637 
1638     @property
1639     def function(self):
1640         """Function for the composed formal power series."""
1641         f, g, x = self.f, self.g, self.ffps.x
1642         return f.subs(x, g)
1643 
1644     def _eval_terms(self, n):
1645         """ 
1646         Returns the first `n` terms of the composed formal power series. 
1647         Term by term logic is implemented here. 
1648  
1649         Explanation 
1650         =========== 
1651  
1652         The coefficient sequence of the :obj:`FormalPowerSeriesCompose` object is the generic sequence. 
1653         It is multiplied by ``bell_seq`` to get a sequence, whose terms are added up to get 
1654         the final terms for the polynomial. 
1655  
1656         Examples 
1657         ======== 
1658  
1659         >>> from sympy import fps, sin, exp 
1660         >>> from sympy.abc import x 
1661         >>> f1 = fps(exp(x)) 
1662         >>> f2 = fps(sin(x)) 
1663         >>> fcomp = f1.compose(f2, x) 
1664  
1665         >>> fcomp._eval_terms(6) 
1666         -x**5/15 - x**4/8 + x**2/2 + x + 1 
1667  
1668         >>> fcomp._eval_terms(8) 
1669         x**7/90 - x**6/240 - x**5/15 - x**4/8 + x**2/2 + x + 1 
1670  
1671         See Also 
1672         ======== 
1673  
1674         sympy.series.formal.FormalPowerSeries.compose 
1675         sympy.series.formal.FormalPowerSeries.coeff_bell 
1676  
1677         """
1678 
1679         ffps, gfps = self.ffps, self.gfps
1680         terms = [ffps.zero_coeff()]
1681 
1682         for i in range(1, n):
1683             bell_seq = gfps.coeff_bell(i)
1684             seq = (ffps.bell_coeff_seq * bell_seq)
1685             terms.append(Add(*(seq[:i])) / ffps.fact_seq[i-1] * ffps.xk.coeff(i))
1686 
1687         return Add(*terms)

1690 class FormalPowerSeriesInverse(FiniteFormalPowerSeries):
1691     """ 
1692     Represents the Inverse of a formal power series. 
1693  
1694     Explanation 
1695     =========== 
1696  
1697     No computation is performed. Terms are calculated using a term by term logic, 
1698     instead of a point by point logic. 
1699  
1700     There is a single difference between a :obj:`FormalPowerSeries` object and a 
1701     :obj:`FormalPowerSeriesInverse` object. The coefficient sequence contains the 
1702     generic sequence which is to be multiplied by a custom ``bell_seq`` finite sequence. 
1703     The finite terms will then be added up to get the final terms. 
1704  
1705     See Also 
1706     ======== 
1707  
1708     sympy.series.formal.FormalPowerSeries 
1709     sympy.series.formal.FiniteFormalPowerSeries 
1710  
1711     """
1712     def __init__(self, *args):
1713         ffps = self.ffps
1714         k = ffps.xk.variables[0]
1715 
1716         inv = ffps.zero_coeff()
1717         inv_seq = sequence(inv ** (-(k + 1)), (k, 1, oo))
1718         self.aux_seq = ffps.sign_seq * ffps.fact_seq * inv_seq
1719 
1720     @property
1721     def function(self):
1722         """Function for the inverse of a formal power series."""
1723         f = self.f
1724         return 1 / f
1725 
1726     @property
1727     def g(self):
1728         raise ValueError("Only one function is considered while performing"
1729                         "inverse of a formal power series.")
1730 
1731     @property
1732     def gfps(self):
1733         raise ValueError("Only one function is considered while performing"
1734                         "inverse of a formal power series.")
1735 
1736     def _eval_terms(self, n):
1737         """ 
1738         Returns the first ``n`` terms of the composed formal power series. 
1739         Term by term logic is implemented here. 
1740  
1741         Explanation 
1742         =========== 
1743  
1744         The coefficient sequence of the `FormalPowerSeriesInverse` object is the generic sequence. 
1745         It is multiplied by ``bell_seq`` to get a sequence, whose terms are added up to get 
1746         the final terms for the polynomial. 
1747  
1748         Examples 
1749         ======== 
1750  
1751         >>> from sympy import fps, exp, cos 
1752         >>> from sympy.abc import x 
1753         >>> f1 = fps(exp(x)) 
1754         >>> f2 = fps(cos(x)) 
1755         >>> finv1, finv2 = f1.inverse(), f2.inverse() 
1756  
1757         >>> finv1._eval_terms(6) 
1758         -x**5/120 + x**4/24 - x**3/6 + x**2/2 - x + 1 
1759  
1760         >>> finv2._eval_terms(8) 
1761         61*x**6/720 + 5*x**4/24 + x**2/2 + 1 
1762  
1763         See Also 
1764         ======== 
1765  
1766         sympy.series.formal.FormalPowerSeries.inverse 
1767         sympy.series.formal.FormalPowerSeries.coeff_bell 
1768  
1769         """
1770         ffps = self.ffps
1771         terms = [ffps.zero_coeff()]
1772 
1773         for i in range(1, n):
1774             bell_seq = ffps.coeff_bell(i)
1775             seq = (self.aux_seq * bell_seq)
1776             terms.append(Add(*(seq[:i])) / ffps.fact_seq[i-1] * ffps.xk.coeff(i))
1777 
1778         return Add(*terms)

1781 def fps(f, x=None, x0=0, dir=1, hyper=True, order=4, rational=True, full=False):
1782     """ 
1783     Generates Formal Power Series of ``f``. 
1784  
1785     Explanation 
1786     =========== 
1787  
1788     Returns the formal series expansion of ``f`` around ``x = x0`` 
1789     with respect to ``x`` in the form of a ``FormalPowerSeries`` object. 
1790  
1791     Formal Power Series is represented using an explicit formula 
1792     computed using different algorithms. 
1793  
1794     See :func:`compute_fps` for the more details regarding the computation 
1795     of formula. 
1796  
1797     Parameters 
1798     ========== 
1799  
1800     x : Symbol, optional 
1801         If x is None and ``f`` is univariate, the univariate symbols will be 
1802         supplied, otherwise an error will be raised. 
1803     x0 : number, optional 
1804         Point to perform series expansion about. Default is 0. 
1805     dir : {1, -1, '+', '-'}, optional 
1806         If dir is 1 or '+' the series is calculated from the right and 
1807         for -1 or '-' the series is calculated from the left. For smooth 
1808         functions this flag will not alter the results. Default is 1. 
1809     hyper : {True, False}, optional 
1810         Set hyper to False to skip the hypergeometric algorithm. 
1811         By default it is set to False. 
1812     order : int, optional 
1813         Order of the derivative of ``f``, Default is 4. 
1814     rational : {True, False}, optional 
1815         Set rational to False to skip rational algorithm. By default it is set 
1816         to True. 
1817     full : {True, False}, optional 
1818         Set full to True to increase the range of rational algorithm. 
1819         See :func:`rational_algorithm` for details. By default it is set to 
1820         False. 
1821  
1822     Examples 
1823     ======== 
1824  
1825     >>> from sympy import fps, ln, atan, sin 
1826     >>> from sympy.abc import x, n 
1827  
1828     Rational Functions 
1829  
1830     >>> fps(ln(1 + x)).truncate() 
1831     x - x**2/2 + x**3/3 - x**4/4 + x**5/5 + O(x**6) 
1832  
1833     >>> fps(atan(x), full=True).truncate() 
1834     x - x**3/3 + x**5/5 + O(x**6) 
1835  
1836     Symbolic Functions 
1837  
1838     >>> fps(x**n*sin(x**2), x).truncate(8) 
1839     -x**(n + 6)/6 + x**(n + 2) + O(x**(n + 8)) 
1840  
1841     See Also 
1842     ======== 
1843  
1844     sympy.series.formal.FormalPowerSeries 
1845     sympy.series.formal.compute_fps 
1846     """
1847     f = sympify(f)
1848 
1849     if x is None:
1850         free = f.free_symbols
1851         if len(free) == 1:
1852             x = free.pop()
1853         elif not free:
1854             return f
1855         else:
1856             raise NotImplementedError("multivariate formal power series")
1857 
1858     result = compute_fps(f, x, x0, dir, hyper, order, rational, full)
1859 
1860     if result is None:
1861         return f
1862 
1863     return FormalPowerSeries(f, x, x0, dir, result)
