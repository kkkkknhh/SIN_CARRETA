17   class _csc_base(_cs_matrix):
18       _format = 'csc'

20       def transpose(self, axes=None, copy=False):
21           if axes is not None and axes != (1, 0):
22               raise ValueError("Sparse arrays/matrices do not support "
23                                 "an 'axes' parameter because swapping "
24                                 "dimensions is the only logical permutation.")
25   
26           M, N = self.shape
27   
28           return self._csr_container((self.data, self.indices,
29                                       self.indptr), (N, M), copy=copy)
30   
31       transpose.__doc__ = _spbase.transpose.__doc__

33       def __iter__(self):
34           yield from self.tocsr()

36       def tocsc(self, copy=False):
37           if copy:
38               return self.copy()
39           else:
40               return self

42       tocsc.__doc__ = _spbase.tocsc.__doc__

44       def tocsr(self, copy=False):
45           M,N = self.shape
46           idx_dtype = self._get_index_dtype((self.indptr, self.indices),
47                                       maxval=max(self.nnz, N))
48           indptr = np.empty(M + 1, dtype=idx_dtype)
49           indices = np.empty(self.nnz, dtype=idx_dtype)
50           data = np.empty(self.nnz, dtype=upcast(self.dtype))
51   
52           csc_tocsr(M, N,
53                     self.indptr.astype(idx_dtype),
54                     self.indices.astype(idx_dtype),
55                     self.data,
56                     indptr,
57                     indices,
58                     data)
59   
60           A = self._csr_container(
61               (data, indices, indptr),
62               shape=self.shape, copy=False
63           )
64           A.has_sorted_indices = True
65           return A
66   
67       tocsr.__doc__ = _spbase.tocsr.__doc__
68   
69       def nonzero(self):
70           # CSC can't use _cs_matrix's .nonzero method because it
71           # returns the indices sorted for self transposed.
72   
73           # Get row and col indices, from _cs_matrix.tocoo
74           major_dim, minor_dim = self._swap(self.shape)
75           minor_indices = self.indices
76           major_indices = np.empty(len(minor_indices), dtype=self.indices.dtype)
77           expandptr(major_dim, self.indptr, major_indices)
78           row, col = self._swap((major_indices, minor_indices))
79   
80           # Remove explicit zeros
81           nz_mask = self.data != 0
82           row = row[nz_mask]
83           col = col[nz_mask]
84   
85           # Sort them to be in C-style order
86           ind = np.argsort(row, kind='mergesort')
87           row = row[ind]
88           col = col[ind]
89   
90           return row, col
91   
92       nonzero.__doc__ = _cs_matrix.nonzero.__doc__

94       def _getrow(self, i):
95           """Returns a copy of row i of the matrix, as a (1 x n) 
96           CSR matrix (row vector). 
97           """
98           M, N = self.shape
99           i = int(i)
100          if i < 0:
101              i += M
102          if i < 0 or i >= M:
103              raise IndexError('index (%d) out of range' % i)
104          return self._get_submatrix(minor=i).tocsr()
105  
106      def _getcol(self, i):
107          """Returns a copy of column i of the matrix, as a (m x 1) 
108          CSC matrix (column vector). 
109          """
110          M, N = self.shape
111          i = int(i)
112          if i < 0:
113              i += N
114          if i < 0 or i >= N:
115              raise IndexError('index (%d) out of range' % i)
116          return self._get_submatrix(major=i, copy=True)
117  
118      def _get_intXarray(self, row, col):
119          return self._major_index_fancy(col)._get_submatrix(minor=row)
120  
121      def _get_intXslice(self, row, col):
122          if col.step in (1, None):
123              return self._get_submatrix(major=col, minor=row, copy=True)
124          return self._major_slice(col)._get_submatrix(minor=row)
125  
126      def _get_sliceXint(self, row, col):
127          if row.step in (1, None):
128              return self._get_submatrix(major=col, minor=row, copy=True)
129          return self._get_submatrix(major=col)._minor_slice(row)
130  
131      def _get_sliceXarray(self, row, col):
132          return self._major_index_fancy(col)._minor_slice(row)
133  
134      def _get_arrayXint(self, row, col):
135          res = self._get_submatrix(major=col)._minor_index_fancy(row)
136          if row.ndim > 1:
137              return res.reshape(row.shape)
138          return res
139  
140      def _get_arrayXslice(self, row, col):
141          return self._major_slice(col)._minor_index_fancy(row)

143      # these functions are used by the parent class (_cs_matrix)
144      # to remove redundancy between csc_array and csr_matrix
145      @staticmethod
146      def _swap(x):
147          """swap the members of x if this is a column-oriented matrix 
148          """
149          return x[1], x[0]

152  def isspmatrix_csc(x):
153      """Is `x` of csc_matrix type? 
154   
155      Parameters 
156      ---------- 
157      x 
158          object to check for being a csc matrix 
159   
160      Returns 
161      ------- 
162      bool 
163          True if `x` is a csc matrix, False otherwise 
164   
165      Examples 
166      -------- 
167      >>> from scipy.sparse import csc_array, csc_matrix, coo_matrix, isspmatrix_csc 
168      >>> isspmatrix_csc(csc_matrix([[5]])) 
169      True 
170      >>> isspmatrix_csc(csc_array([[5]])) 
171      False 
172      >>> isspmatrix_csc(coo_matrix([[5]])) 
173      False 
174      """
175      return isinstance(x, csc_matrix)

179  class csc_array(_csc_base, sparray):
180      """ 
181      Compressed Sparse Column array. 
182   
183      This can be instantiated in several ways: 
184          csc_array(D) 
185              where D is a 2-D ndarray 
186   
187          csc_array(S) 
188              with another sparse array or matrix S (equivalent to S.tocsc()) 
189   
190          csc_array((M, N), [dtype]) 
191              to construct an empty array with shape (M, N) 
192              dtype is optional, defaulting to dtype='d'. 
193   
194          csc_array((data, (row_ind, col_ind)), [shape=(M, N)]) 
195              where ``data``, ``row_ind`` and ``col_ind`` satisfy the 
196              relationship ``a[row_ind[k], col_ind[k]] = data[k]``. 
197   
198          csc_array((data, indices, indptr), [shape=(M, N)]) 
199              is the standard CSC representation where the row indices for 
200              column i are stored in ``indices[indptr[i]:indptr[i+1]]`` 
201              and their corresponding values are stored in 
202              ``data[indptr[i]:indptr[i+1]]``.  If the shape parameter is 
203              not supplied, the array dimensions are inferred from 
204              the index arrays. 
205   
206      Attributes 
207      ---------- 
208      dtype : dtype 
209          Data type of the array 
210      shape : 2-tuple 
211          Shape of the array 
212      ndim : int 
213          Number of dimensions (this is always 2) 
214      nnz 
215      size 
216      data 
217          CSC format data array of the array 
218      indices 
219          CSC format index array of the array 
220      indptr 
221          CSC format index pointer array of the array 
222      has_sorted_indices 
223      has_canonical_format 
224      T 
225   
226      Notes 
227      ----- 
228   
229      Sparse arrays can be used in arithmetic operations: they support 
230      addition, subtraction, multiplication, division, and matrix power. 
231   
232      Advantages of the CSC format 
233          - efficient arithmetic operations CSC + CSC, CSC * CSC, etc. 
234          - efficient column slicing 
235          - fast matrix vector products (CSR, BSR may be faster) 
236   
237      Disadvantages of the CSC format 
238        - slow row slicing operations (consider CSR) 
239        - changes to the sparsity structure are expensive (consider LIL or DOK) 
240   
241      Canonical format 
242        - Within each column, indices are sorted by row. 
243        - There are no duplicate entries. 
244   
245      Examples 
246      -------- 
247   
248      >>> import numpy as np 
249      >>> from scipy.sparse import csc_array 
250      >>> csc_array((3, 4), dtype=np.int8).toarray() 
251      array([[0, 0, 0, 0], 
252             [0, 0, 0, 0], 
253             [0, 0, 0, 0]], dtype=int8) 
254   
255      >>> row = np.array([0, 2, 2, 0, 1, 2]) 
256      >>> col = np.array([0, 0, 1, 2, 2, 2]) 
257      >>> data = np.array([1, 2, 3, 4, 5, 6]) 
258      >>> csc_array((data, (row, col)), shape=(3, 3)).toarray() 
259      array([[1, 0, 4], 
260             [0, 0, 5], 
261             [2, 3, 6]]) 
262   
263      >>> indptr = np.array([0, 2, 3, 6]) 
264      >>> indices = np.array([0, 2, 2, 0, 1, 2]) 
265      >>> data = np.array([1, 2, 3, 4, 5, 6]) 
266      >>> csc_array((data, indices, indptr), shape=(3, 3)).toarray() 
267      array([[1, 0, 4], 
268             [0, 0, 5], 
269             [2, 3, 6]]) 
270   
271      """
