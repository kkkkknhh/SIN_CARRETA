1    # Tests for a few of the "double-double" C++ functions defined in
2    # special/cephes/dd_real.h. Prior to gh-20390 which translated these
3    # functions from C to C++, there were test cases for _dd_expm1. It
4    # was determined that this function is not used anywhere internally
5    # in SciPy, so this function was not translated.
6    
7    
8    import pytest
9    from numpy.testing import assert_allclose
10   from scipy.special._test_internal import _dd_exp, _dd_log
11   
12   
13   # Each tuple in test_data contains:
14   #   (dd_func, xhi, xlo, expected_yhi, expected_ylo)
15   # The expected values were computed with mpmath, e.g.
16   #
17   #   import mpmath
18   #   mpmath.mp.dps = 100
19   #   xhi = 10.0
20   #   xlo = 0.0
21   #   x = mpmath.mpf(xhi) + mpmath.mpf(xlo)
22   #   y = mpmath.log(x)
23   #   expected_yhi = float(y)
24   #   expected_ylo = float(y - expected_yhi)
25   #
26   test_data = [
27       (_dd_exp, -0.3333333333333333, -1.850371707708594e-17,
28        0.7165313105737893, -2.0286948382455594e-17),
29       (_dd_exp, 0.0, 0.0, 1.0, 0.0),
30       (_dd_exp, 10.0, 0.0, 22026.465794806718, -1.3780134700517372e-12),
31       (_dd_log, 0.03125, 0.0, -3.4657359027997265, -4.930038229799327e-18),
32       (_dd_log, 10.0, 0.0, 2.302585092994046, -2.1707562233822494e-16),
33   ]
34   
35   
36   @pytest.mark.parametrize('dd_func, xhi, xlo, expected_yhi, expected_ylo',
37                            test_data)
38   def test_dd(dd_func, xhi, xlo, expected_yhi, expected_ylo):
39       yhi, ylo = dd_func(xhi, xlo)
40       assert yhi == expected_yhi, (f"high double ({yhi}) does not equal the "
41                                    f"expected value {expected_yhi}")
42       assert_allclose(ylo, expected_ylo, rtol=5e-15)
