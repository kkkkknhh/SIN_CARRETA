14           constexpr double besselpoly_EPS = 1.0e-17;

17       XSF_HOST_DEVICE inline double besselpoly(double a, double lambda, double nu) {
18   
19           int m, factor = 0;
20           double Sm, relerr, Sol;
21           double sum = 0.0;
22   
23           /* Special handling for a = 0.0 */
24           if (a == 0.0) {
25               if (nu == 0.0) {
26                   return 1.0 / (lambda + 1);
27               } else {
28                   return 0.0;
29               }
30           }
31           /* Special handling for negative and integer nu */
32           if ((nu < 0) && (std::floor(nu) == nu)) {
33               nu = -nu;
34               factor = static_cast<int>(nu) % 2;
35           }
36           Sm = std::exp(nu * std::log(a)) / (Gamma(nu + 1) * (lambda + nu + 1));
37           m = 0;
38           do {
39               sum += Sm;
40               Sol = Sm;
41               Sm *= -a * a * (lambda + nu + 1 + 2 * m) / ((nu + m + 1) * (m + 1) * (lambda + nu + 1 + 2 * m + 2));
42               m++;
43               relerr = std::abs((Sm - Sol) / Sm);
44           } while (relerr > detail::besselpoly_EPS && m < 1000);
45           if (!factor)
46               return sum;
47           else
48               return -sum;
49       }
