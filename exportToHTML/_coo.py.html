128      def reshape(self, *args, **kwargs):
129          shape = check_shape(args, self.shape, allow_nd=self._allow_nd)
130          order, copy = check_reshape_kwargs(kwargs)
131  
132          # Return early if reshape is not required
133          if shape == self.shape:
134              if copy:
135                  return self.copy()
136              else:
137                  return self
138  
139          # When reducing the number of dimensions, we need to be careful about
140          # index overflow. This is why we can't simply call
141          # `np.ravel_multi_index()` followed by `np.unravel_index()` here.
142          flat_coords = _ravel_coords(self.coords, self.shape, order=order)
143          if len(shape) == 2:
144              if order == 'C':
145                  new_coords = divmod(flat_coords, shape[1])
146              else:
147                  new_coords = divmod(flat_coords, shape[0])[::-1]
148          else:
149              new_coords = np.unravel_index(flat_coords, shape, order=order)
150  
151          idx_dtype = self._get_index_dtype(self.coords, maxval=max(shape))
152          new_coords = tuple(np.asarray(co, dtype=idx_dtype) for co in new_coords)
153  
154          # Handle copy here rather than passing on to the constructor so that no
155          # copy will be made of `new_coords` regardless.
156          if copy:
157              new_data = self.data.copy()
158          else:
159              new_data = self.data
160  
161          return self.__class__((new_data, new_coords), shape=shape, copy=False)

165      def _getnnz(self, axis=None):
166          if axis is None or (axis == 0 and self.ndim == 1):
167              nnz = len(self.data)
168              if any(len(idx) != nnz for idx in self.coords):
169                  raise ValueError('all index and data arrays must have the '
170                                   'same length')
171  
172              if self.data.ndim != 1 or any(idx.ndim != 1 for idx in self.coords):
173                  raise ValueError('coordinates and data arrays must be 1-D')
174  
175              return int(nnz)
176  
177          if axis < 0:
178              axis += self.ndim
179          if axis >= self.ndim:
180              raise ValueError('axis out of bounds')
181  
182          return np.bincount(downcast_intp_index(self.coords[1 - axis]),
183                             minlength=self.shape[1 - axis])

188          self.sum_duplicates()
189          if axis is None:
190              return np.count_nonzero(self.data)
191  
192          if axis < 0:
193              axis += self.ndim

195              raise ValueError('axis out of bounds')
196          mask = self.data != 0
197          coord = self.coords[1 - axis][mask]
198          return np.bincount(downcast_intp_index(coord), minlength=self.shape[1 - axis])

202      def _check(self):
203          """ Checks data structure for consistency """
204          if self.ndim != len(self.coords):
205              raise ValueError('mismatching number of index arrays for shape; '
206                               f'got {len(self.coords)}, expected {self.ndim}')
207  
208          # index arrays should have integer data types
209          for i, idx in enumerate(self.coords):
210              if idx.dtype.kind != 'i':
211                  warn(f'index array {i} has non-integer dtype ({idx.dtype.name})',
212                       stacklevel=3)
213  
214          idx_dtype = self._get_index_dtype(self.coords, maxval=max(self.shape))
215          self.coords = tuple(np.asarray(idx, dtype=idx_dtype)
216                               for idx in self.coords)
217          self.data = to_native(self.data)
218  
219          if self.nnz > 0:
220              for i, idx in enumerate(self.coords):
221                  if idx.max() >= self.shape[i]:
222                      raise ValueError(f'axis {i} index {idx.max()} exceeds '
223                                       f'matrix dimension {self.shape[i]}')
224                  if idx.min() < 0:
225                      raise ValueError(f'negative axis {i} index: {idx.min()}')

227      def transpose(self, axes=None, copy=False):
228          if axes is None:
229              axes = range(self.ndim)[::-1]
230          elif isinstance(self, sparray):
231              if not hasattr(axes, "__len__") or len(axes) != self.ndim:
232                  raise ValueError("axes don't match matrix dimensions")
233              if len(set(axes)) != self.ndim:
234                  raise ValueError("repeated axis in transpose")
235          elif axes != (1, 0):
236              raise ValueError("Sparse matrices do not support an 'axes' "
237                               "parameter because swapping dimensions is the "
238                               "only logical permutation.")
239  
240          permuted_shape = tuple(self._shape[i] for i in axes)
241          permuted_coords = tuple(self.coords[i] for i in axes)
242          return self.__class__((self.data, permuted_coords),
243                                shape=permuted_shape, copy=copy)

247      def resize(self, *shape) -> None:
248          shape = check_shape(shape, allow_nd=self._allow_nd)
249          if self.ndim > 2:
250              raise ValueError("only 1-D or 2-D input accepted")
251          if len(shape) > 2:
252              raise ValueError("shape argument must be 1-D or 2-D")
253          # Check for added dimensions.
254          if len(shape) > self.ndim:
255              flat_coords = _ravel_coords(self.coords, self.shape)
256              max_size = math.prod(shape)
257              self.coords = np.unravel_index(flat_coords[:max_size], shape)
258              self.data = self.data[:max_size]
259              self._shape = shape
260              return
261  
262          # Check for removed dimensions.
263          if len(shape) < self.ndim:
264              tmp_shape = (
265                  self._shape[:len(shape) - 1]  # Original shape without last axis
266                  + (-1,)  # Last axis is used to flatten the array
267                  + (1,) * (self.ndim - len(shape))  # Pad with ones
268              )
269              tmp = self.reshape(tmp_shape)
270              self.coords = tmp.coords[:len(shape)]
271              self._shape = tmp.shape[:len(shape)]
272  
273          # Handle truncation of existing dimensions.
274          is_truncating = any(old > new for old, new in zip(self.shape, shape))
275          if is_truncating:
276              mask = np.logical_and.reduce([
277                  idx < size for idx, size in zip(self.coords, shape)
278              ])
279              if not mask.all():
280                  self.coords = tuple(idx[mask] for idx in self.coords)
281                  self.data = self.data[mask]
282  
283          self._shape = shape

287      def toarray(self, order=None, out=None):
288          B = self._process_toarray_args(order, out)
289          fortran = int(B.flags.f_contiguous)
290          if not fortran and not B.flags.c_contiguous:
291              raise ValueError("Output array must be C or F contiguous")
292          # This handles both 0D and 1D cases correctly regardless of the
293          # original shape.
294          if self.ndim == 1:
295              coo_todense_nd(np.array([1]), self.nnz, self.ndim,
296                             self.coords[0], self.data, B.ravel('A'), fortran)
297          elif self.ndim == 2:
298              M, N = self.shape
299              coo_todense(M, N, self.nnz, self.row, self.col, self.data,
300                          B.ravel('A'), fortran)
301          else:
302              if fortran:
303                  strides = np.append(1, np.cumprod(self.shape[:-1]))
304              else:
305                  strides = np.append(np.cumprod(self.shape[1:][::-1])[::-1], 1)
306              coords = np.concatenate(self.coords)
307              coo_todense_nd(strides, self.nnz, self.ndim,
308                             coords, self.data, B.ravel('A'), fortran)
309          # Note: reshape() doesn't copy here, but does return a new array (view).
310          return B.reshape(self.shape)

314      def tocsc(self, copy=False):
315          """Convert this array/matrix to Compressed Sparse Column format 
316   
317          Duplicate entries will be summed together. 
318   
319          Examples 
320          -------- 
321          >>> from numpy import array 
322          >>> from scipy.sparse import coo_array 
323          >>> row  = array([0, 0, 1, 3, 1, 0, 0]) 
324          >>> col  = array([0, 2, 1, 3, 1, 0, 0]) 
325          >>> data = array([1, 1, 1, 1, 1, 1, 1]) 
326          >>> A = coo_array((data, (row, col)), shape=(4, 4)).tocsc() 
327          >>> A.toarray() 
328          array([[3, 0, 1, 0], 
329                 [0, 2, 0, 0], 
330                 [0, 0, 0, 0], 
331                 [0, 0, 0, 1]]) 
332   
333          """
334          if self.ndim != 2:
335              raise ValueError(f'Cannot convert. CSC format must be 2D. Got {self.ndim}D')
336          if self.nnz == 0:
337              return self._csc_container(self.shape, dtype=self.dtype)
338          else:
339              from ._csc import csc_array
340              indptr, indices, data, shape = self._coo_to_compressed(csc_array._swap)
341  
342              x = self._csc_container((data, indices, indptr), shape=shape)
343              if not self.has_canonical_format:
344                  x.sum_duplicates()
345              return x

347      def tocsr(self, copy=False):
348          """Convert this array/matrix to Compressed Sparse Row format 
349   
350          Duplicate entries will be summed together. 
351   
352          Examples 
353          -------- 
354          >>> from numpy import array 
355          >>> from scipy.sparse import coo_array 
356          >>> row  = array([0, 0, 1, 3, 1, 0, 0]) 
357          >>> col  = array([0, 2, 1, 3, 1, 0, 0]) 
358          >>> data = array([1, 1, 1, 1, 1, 1, 1]) 
359          >>> A = coo_array((data, (row, col)), shape=(4, 4)).tocsr() 
360          >>> A.toarray() 
361          array([[3, 0, 1, 0], 
362                 [0, 2, 0, 0], 
363                 [0, 0, 0, 0], 
364                 [0, 0, 0, 1]]) 
365   
366          """
367          if self.ndim > 2:
368              raise ValueError(f'Cannot convert. CSR must be 1D or 2D. Got {self.ndim}D')
369          if self.nnz == 0:
370              return self._csr_container(self.shape, dtype=self.dtype)
371          else:
372              from ._csr import csr_array
373              arrays = self._coo_to_compressed(csr_array._swap, copy=copy)
374              indptr, indices, data, shape = arrays
375  
376              x = self._csr_container((data, indices, indptr), shape=self.shape)
377              if not self.has_canonical_format:
378                  x.sum_duplicates()
379              return x

381      def _coo_to_compressed(self, swap, copy=False):
382          """convert (shape, coords, data) to (indptr, indices, data, shape)"""
383          M, N = swap(self._shape_as_2d)
384          # convert idx_dtype intc to int32 for pythran.
385          # tested in scipy/optimize/tests/test__numdiff.py::test_group_columns
386          idx_dtype = self._get_index_dtype(self.coords, maxval=max(self.nnz, N))
387  
388          if self.ndim == 1:
389              indices = self.coords[0].copy() if copy else self.coords[0]
390              nnz = len(indices)
391              indptr = np.array([0, nnz], dtype=idx_dtype)
392              data = self.data.copy() if copy else self.data
393              return indptr, indices, data, self.shape
394  
395          # ndim == 2
396          major, minor = swap(self.coords)
397          nnz = len(major)
398          major = major.astype(idx_dtype, copy=False)
399          minor = minor.astype(idx_dtype, copy=False)
400  
401          indptr = np.empty(M + 1, dtype=idx_dtype)
402          indices = np.empty_like(minor, dtype=idx_dtype)
403          data = np.empty_like(self.data, dtype=self.dtype)
404  
405          coo_tocsr(M, N, nnz, major, minor, self.data, indptr, indices, data)
406          return indptr, indices, data, self.shape

416      def todia(self, copy=False):
417          if self.ndim != 2:
418              raise ValueError(f'Cannot convert. DIA format must be 2D. Got {self.ndim}D')
419          self.sum_duplicates()
420          ks = self.col - self.row  # the diagonal for each nonzero
421          diags, diag_idx = np.unique(ks, return_inverse=True)
422  
423          if len(diags) > 100:
424              # probably undesired, should todia() have a maxdiags parameter?
425              warn(f"Constructing a DIA matrix with {len(diags)} diagonals "
426                   "is inefficient",
427                   SparseEfficiencyWarning, stacklevel=2)
428  
429          #initialize and fill in data array
430          if self.data.size == 0:
431              data = np.zeros((0, 0), dtype=self.dtype)
432          else:
433              data = np.zeros((len(diags), self.col.max()+1), dtype=self.dtype)
434              data[diag_idx, self.col] = self.data
435  
436          return self._dia_container((data, diags), shape=self.shape)

456      def diagonal(self, k=0):
457          if self.ndim != 2:
458              raise ValueError("diagonal requires two dimensions")
459          rows, cols = self.shape
460          if k <= -rows or k >= cols:
461              return np.empty(0, dtype=self.data.dtype)
462          diag = np.zeros(min(rows + min(k, 0), cols - max(k, 0)),
463                          dtype=self.dtype)
464          diag_mask = (self.row + k) == self.col
465  
466          if self.has_canonical_format:
467              row = self.row[diag_mask]
468              data = self.data[diag_mask]
469          else:
470              inds = tuple(idx[diag_mask] for idx in self.coords)
471              (row, _), data = self._sum_duplicates(inds, self.data[diag_mask])
472          diag[row + min(k, 0)] = data
473  
474          return diag

478      def _setdiag(self, values, k):
479          if self.ndim != 2:
480              raise ValueError("setting a diagonal requires two dimensions")
481          M, N = self.shape
482          if values.ndim and not len(values):
483              return
484          idx_dtype = self.row.dtype
485  
486          # Determine which triples to keep and where to put the new ones.
487          full_keep = self.col - self.row != k
488          if k < 0:
489              max_index = min(M+k, N)
490              if values.ndim:
491                  max_index = min(max_index, len(values))
492              keep = np.logical_or(full_keep, self.col >= max_index)
493              new_row = np.arange(-k, -k + max_index, dtype=idx_dtype)
494              new_col = np.arange(max_index, dtype=idx_dtype)
495          else:
496              max_index = min(M, N-k)
497              if values.ndim:
498                  max_index = min(max_index, len(values))
499              keep = np.logical_or(full_keep, self.row >= max_index)
500              new_row = np.arange(max_index, dtype=idx_dtype)
501              new_col = np.arange(k, k + max_index, dtype=idx_dtype)
502  
503          # Define the array of data consisting of the entries to be added.
504          if values.ndim:
505              new_data = values[:max_index]
506          else:
507              new_data = np.empty(max_index, dtype=self.dtype)
508              new_data[:] = values
509  
510          # Update the internal structure.
511          self.coords = (np.concatenate((self.row[keep], new_row)),
512                         np.concatenate((self.col[keep], new_col)))
513          self.data = np.concatenate((self.data[keep], new_data))
514          self.has_canonical_format = False

527      def sum_duplicates(self) -> None:
528          """Eliminate duplicate entries by adding them together 
529   
530          This is an *in place* operation 
531          """
532          if self.has_canonical_format:
533              return
534          summed = self._sum_duplicates(self.coords, self.data)
535          self.coords, self.data = summed
536          self.has_canonical_format = True

538      def _sum_duplicates(self, coords, data):
539          # Assumes coords not in canonical format.
540          if len(data) == 0:
541              return coords, data
542          # Sort coords w.r.t. rows, then cols. This corresponds to C-order,
543          # which we rely on for argmin/argmax to return the first index in the
544          # same way that numpy does (in the case of ties).
545          order = np.lexsort(coords[::-1])
546          coords = tuple(idx[order] for idx in coords)
547          data = data[order]
548          unique_mask = np.logical_or.reduce([
549              idx[1:] != idx[:-1] for idx in coords
550          ])
551          unique_mask = np.append(True, unique_mask)
552          coords = tuple(idx[unique_mask] for idx in coords)
553          unique_inds, = np.nonzero(unique_mask)
554          data = np.add.reduceat(data, downcast_intp_index(unique_inds), dtype=self.dtype)
555          return coords, data

557      def eliminate_zeros(self):
558          """Remove zero entries from the array/matrix 
559   
560          This is an *in place* operation 
561          """
562          mask = self.data != 0
563          self.data = self.data[mask]
564          self.coords = tuple(idx[mask] for idx in self.coords)

680      def _matmul_dispatch(self, other):
681          if isscalarlike(other):
682              return self.multiply(other)
683  
684          if not (issparse(other) or isdense(other)):
685              # If it's a list or whatever, treat it like an array
686              other_a = np.asanyarray(other)
687  
688              if other_a.ndim == 0 and other_a.dtype == np.object_:
689                  # Not interpretable as an array; return NotImplemented so that
690                  # other's __rmatmul__ can kick in if that's implemented.
691                  return NotImplemented
692  
693              try:
694                  other.shape
695              except AttributeError:
696                  other = other_a
697  
698          if self.ndim < 3 and other.ndim < 3:
699              return _spbase._matmul_dispatch(self, other)
700  
701          N = self.shape[-1]
702          err_prefix = "matmul: dimension mismatch with signature"
703          if other.__class__ is np.ndarray:
704              if other.shape == (N,):
705                  return self._matmul_vector(other)
706              if other.shape == (N, 1):
707                  result = self._matmul_vector(other.ravel())
708                  return result.reshape(*self.shape[:-1], 1)
709              if other.ndim == 1:
710                  msg = f"{err_prefix} (n,k={N}),(k={other.shape[0]},)->(n,)"
711                  raise ValueError(msg)
712              if other.shape[-2] == N:
713                  # check for batch dimensions compatibility
714                  batch_shape_A = self.shape[:-2]
715                  batch_shape_B = other.shape[:-2]
716                  if batch_shape_A != batch_shape_B:
717                      try:
718                          # This will raise an error if the shapes are not broadcastable
719                          np.broadcast_shapes(batch_shape_A, batch_shape_B)
720                      except ValueError:
721                          raise ValueError("Batch dimensions are not broadcastable")
722  
723                  return self._matmul_multivector(other)
724              else:
725                  raise ValueError(
726                      f"{err_prefix} (n,..,k={N}),(k={other.shape[-2]},..,m)->(n,..,m)"
727                  )
728  
729  
730          if isscalarlike(other):
731              # scalar value
732              return self._mul_scalar(other)
733  
734          if issparse(other):
735              self_is_1d = self.ndim == 1
736              other_is_1d = other.ndim == 1
737  
738              # reshape to 2-D if self or other is 1-D
739              if self_is_1d:
740                  self = self.reshape(self._shape_as_2d) # prepend 1 to shape
741  
742              if other_is_1d:
743                  other = other.reshape((other.shape[0], 1)) # append 1 to shape
744  
745              # Check if the inner dimensions match for matrix multiplication
746              if N != other.shape[-2]:
747                  raise ValueError(
748                      f"{err_prefix} (n,..,k={N}),(k={other.shape[-2]},..,m)->(n,..,m)"
749                  )
750  
751              # If A or B has more than 2 dimensions, check for
752              # batch dimensions compatibility
753              if self.ndim > 2 or other.ndim > 2:
754                  batch_shape_A = self.shape[:-2]
755                  batch_shape_B = other.shape[:-2]
756                  if batch_shape_A != batch_shape_B:
757                      try:
758                          # This will raise an error if the shapes are not broadcastable
759                          np.broadcast_shapes(batch_shape_A, batch_shape_B)
760                      except ValueError:
761                          raise ValueError("Batch dimensions are not broadcastable")
762  
763              result = self._matmul_sparse(other)
764  
765              # reshape back if a or b were originally 1-D
766              if self_is_1d:
767                  # if self was originally 1-D, reshape result accordingly
768                  result = result.reshape(tuple(result.shape[:-2]) +
769                                          tuple(result.shape[-1:]))
770              if other_is_1d:
771                  result = result.reshape(result.shape[:-1])
772              return result

775      def _matmul_multivector(self, other):
776          result_dtype = upcast_char(self.dtype.char, other.dtype.char)
777          if self.ndim >= 3 or other.ndim >= 3:
778              # if self has shape (N,), reshape to (1,N)
779              if self.ndim == 1:
780                  result = self.reshape(1, self.shape[0])._matmul_multivector(other)
781                  return result.reshape(tuple(other.shape[:-2]) + tuple(other.shape[-1:]))
782  
783              broadcast_shape = np.broadcast_shapes(self.shape[:-2], other.shape[:-2])
784              self_shape = broadcast_shape + self.shape[-2:]
785              other_shape = broadcast_shape + other.shape[-2:]
786  
787              self = self._broadcast_to(self_shape)
788              other = np.broadcast_to(other, other_shape)
789              result_shape = broadcast_shape + self.shape[-2:-1] + other.shape[-1:]
790              result = np.zeros(result_shape, dtype=result_dtype)
791              coo_matmat_dense_nd(self.nnz, len(self.shape), other.shape[-1],
792                                  np.array(other_shape), np.array(result_shape),
793                                  np.concatenate(self.coords),
794                                  self.data, other.ravel('C'), result)
795              return result
796  
797          if self.ndim == 2:
798              result_shape = (self.shape[0], other.shape[1])
799              col = self.col
800              row = self.row
801          elif self.ndim == 1:
802              result_shape = (other.shape[1],)
803              col = self.coords[0]
804              row = np.zeros_like(col)
805          result = np.zeros(result_shape, dtype=result_dtype)
806          coo_matmat_dense(self.nnz, other.shape[-1], row, col,
807                           self.data, other.ravel('C'), result)
808          return result.view(type=type(other))

811      def dot(self, other):
812          """Return the dot product of two arrays. 
813   
814          Strictly speaking a dot product involves two vectors. 
815          But in the sense that an array with ndim >= 1 is a collection 
816          of vectors, the function computes the collection of dot products 
817          between each vector in the first array with each vector in the 
818          second array. The axis upon which the sum of products is performed 
819          is the last axis of the first array and the second to last axis of 
820          the second array. If the second array is 1-D, the last axis is used. 
821   
822          Thus, if both arrays are 1-D, the inner product is returned. 
823          If both are 2-D, we have matrix multiplication. If `other` is 1-D, 
824          the sum product is taken along the last axis of each array. If 
825          `other` is N-D for N>=2, the sum product is over the last axis of 
826          the first array and the second-to-last axis of the second array. 
827   
828          Parameters 
829          ---------- 
830          other : array_like (dense or sparse) 
831              Second array 
832   
833          Returns 
834          ------- 
835          output : array (sparse or dense) 
836              The dot product of this array with `other`. 
837              It will be dense/sparse if `other` is dense/sparse. 
838   
839          Examples 
840          -------- 
841   
842          >>> import numpy as np 
843          >>> from scipy.sparse import coo_array 
844          >>> A = coo_array([[1, 2, 0], [0, 0, 3], [4, 0, 5]]) 
845          >>> v = np.array([1, 0, -1]) 
846          >>> A.dot(v) 
847          array([ 1, -3, -1], dtype=int64) 
848   
849          For 2-D arrays it is the matrix product: 
850   
851          >>> A = coo_array([[1, 0], [0, 1]]) 
852          >>> B = coo_array([[4, 1], [2, 2]]) 
853          >>> A.dot(B).toarray() 
854          array([[4, 1], 
855                 [2, 2]]) 
856   
857          For 3-D arrays the shape extends unused axes by other unused axes. 
858   
859          >>> A = coo_array(np.arange(3*4*5*6)).reshape((3,4,5,6)) 
860          >>> B = coo_array(np.arange(3*4*5*6)).reshape((5,4,6,3)) 
861          >>> A.dot(B).shape 
862          (3, 4, 5, 5, 4, 3) 
863          """
864          if not (issparse(other) or isdense(other) or isscalarlike(other)):
865              # If it's a list or whatever, treat it like an array
866              o_array = np.asanyarray(other)
867  
868              if o_array.ndim == 0 and o_array.dtype == np.object_:
869                  raise TypeError(f"dot argument not supported type: '{type(other)}'")
870              try:
871                  other.shape
872              except AttributeError:
873                  other = o_array
874  
875          if self.ndim < 3 and (np.isscalar(other) or other.ndim<3):
876              return _spbase.dot(self, other)
877          # Handle scalar multiplication
878          if np.isscalar(other):
879              return self * other
880          if isdense(other):
881              return self._dense_dot(other)
882          elif other.format != "coo":
883              raise TypeError("input must be a COO matrix/array")
884          elif self.ndim == 1 and other.ndim == 1:
885              # Handle inner product of vectors (1-D arrays)
886              if self.shape[0] != other.shape[0]:
887                  raise ValueError(f"shapes {self.shape} and {other.shape}"
888                                   " are not aligned for inner product")
889              return self @ other
890          elif self.ndim == 2 and other.ndim == 2:
891              # Handle matrix multiplication (2-D arrays)
892              if self.shape[1] != other.shape[0]:
893                  raise ValueError(f"shapes {self.shape} and {other.shape}"
894                                   " are not aligned for matmul")
895              return self @ other
896          else:
897              return self._sparse_dot(other)

900      def _sparse_dot(self, other):
901          self_is_1d = self.ndim == 1
902          other_is_1d = other.ndim == 1
903  
904          # reshape to 2-D if self or other is 1-D
905          if self_is_1d:
906              self = self.reshape(self._shape_as_2d)  # prepend 1 to shape
907          if other_is_1d:
908              other = other.reshape((other.shape[0], 1))  # append 1 to shape
909  
910          if self.shape[-1] != other.shape[-2]:
911                  raise ValueError(f"shapes {self.shape} and {other.shape}"
912                                   " are not aligned for n-D dot")
913  
914          # Prepare the tensors for dot operation
915          # Ravel non-reduced axes coordinates
916          self_raveled_coords = _ravel_non_reduced_axes(self.coords,
917                                                        self.shape, [self.ndim-1])
918          other_raveled_coords = _ravel_non_reduced_axes(other.coords,
919                                                         other.shape, [other.ndim-2])
920  
921          # Get the shape of the non-reduced axes
922          self_nonreduced_shape = self.shape[:-1]
923          other_nonreduced_shape = other.shape[:-2] + other.shape[-1:]
924  
925          # Create 2D coords arrays
926          ravel_coords_shape_self = (math.prod(self_nonreduced_shape), self.shape[-1])
927          ravel_coords_shape_other = (other.shape[-2], math.prod(other_nonreduced_shape))
928  
929          self_2d_coords = (self_raveled_coords, self.coords[-1])
930          other_2d_coords = (other.coords[-2], other_raveled_coords)
931  
932          self_2d = coo_array((self.data, self_2d_coords), ravel_coords_shape_self)
933          other_2d = coo_array((other.data, other_2d_coords), ravel_coords_shape_other)
934  
935          prod = (self_2d @ other_2d).tocoo() # routes via 2-D CSR
936  
937          # Combine the shapes of the non-reduced axes
938          combined_shape = self_nonreduced_shape + other_nonreduced_shape
939  
940          # Unravel the 2D coordinates to get multi-dimensional coordinates
941          shapes = (self_nonreduced_shape, other_nonreduced_shape)
942          prod_coords = []
943          for c, s in zip(prod.coords, shapes):
944              prod_coords.extend(np.unravel_index(c, s))
945  
946          prod_arr = coo_array((prod.data, prod_coords), combined_shape)
947  
948          # reshape back if a or b were originally 1-D
949          # TODO: Move this logic before computation of prod_coords for efficiency
950          if self_is_1d:
951              prod_arr = prod_arr.reshape(combined_shape[1:])
952          if other_is_1d:
953              prod_arr = prod_arr.reshape(combined_shape[:-1])
954  
955          return prod_arr

957      def _dense_dot(self, other):
958          self_is_1d = self.ndim == 1
959          other_is_1d = other.ndim == 1
960  
961          # reshape to 2-D if self or other is 1-D
962          if self_is_1d:
963              self = self.reshape(self._shape_as_2d)  # prepend 1 to shape
964          if other_is_1d:
965              other = other.reshape((other.shape[0], 1))  # append 1 to shape
966  
967          if self.shape[-1] != other.shape[-2]:
968                  raise ValueError(f"shapes {self.shape} and {other.shape}"
969                                   " are not aligned for n-D dot")
970  
971          new_shape_self = (
972              self.shape[:-1] + (1,) * (len(other.shape) - 1) + self.shape[-1:]
973          )
974          new_shape_other = (1,) * (len(self.shape) - 1) + other.shape
975  
976          result_shape = self.shape[:-1] + other.shape[:-2] + other.shape[-1:]
977          result = self.reshape(new_shape_self) @ other.reshape(new_shape_other)
978          prod_arr = result.reshape(result_shape)
979  
980          # reshape back if a or b were originally 1-D
981          if self_is_1d:
982              prod_arr = prod_arr.reshape(result_shape[1:])
983          if other_is_1d:
984              prod_arr = prod_arr.reshape(result_shape[:-1])
985  
986          return prod_arr

988      def tensordot(self, other, axes=2):
989          """Return the tensordot product with another array along the given axes. 
990   
991          The tensordot differs from dot and matmul in that any axis can be 
992          chosen for each of the first and second array and the sum of the 
993          products is computed just like for matrix multiplication, only not 
994          just for the rows of the first times the columns of the second. It 
995          takes the dot product of the collection of vectors along the specified 
996          axes.  Here we can even take the sum of the products along two or even 
997          more axes if desired. So, tensordot is a dot product computation 
998          applied to arrays of any dimension >= 1. It is like matmul but over 
999          arbitrary axes for each matrix. 
1000  
1001         Given two tensors, `a` and `b`, and the desired axes specified as a 
1002         2-tuple/list/array containing two sequences of axis numbers, 
1003         ``(a_axes, b_axes)``, sum the products of `a`'s and `b`'s elements 
1004         (components) over the axes specified by ``a_axes`` and ``b_axes``. 
1005         The `axes` input can be a single non-negative integer, ``N``; 
1006         if it is, then the last ``N`` dimensions of `a` and the first 
1007         ``N`` dimensions of `b` are summed over. 
1008  
1009         Parameters 
1010         ---------- 
1011         a, b : array_like 
1012             Tensors to "dot". 
1013  
1014         axes : int or (2,) array_like 
1015             * integer_like 
1016               If an int N, sum over the last N axes of `a` and the first N axes 
1017               of `b` in order. The sizes of the corresponding axes must match. 
1018             * (2,) array_like 
1019               A 2-tuple of sequences of axes to be summed over, the first applying 
1020               to `a`, the second to `b`. The sequences must be the same length. 
1021               The shape of the corresponding axes must match between `a` and `b`. 
1022  
1023         Returns 
1024         ------- 
1025         output : coo_array 
1026             The tensor dot product of this array with `other`. 
1027             It will be dense/sparse if `other` is dense/sparse. 
1028  
1029         See Also 
1030         -------- 
1031         dot 
1032  
1033         Examples 
1034         -------- 
1035         >>> import numpy as np 
1036         >>> import scipy.sparse 
1037         >>> A = scipy.sparse.coo_array([[[2, 3], [0, 0]], [[0, 1], [0, 5]]]) 
1038         >>> A.shape 
1039         (2, 2, 2) 
1040  
1041         Integer axes N are shorthand for (range(-N, 0), range(0, N)): 
1042  
1043         >>> A.tensordot(A, axes=1).toarray() 
1044         array([[[[ 4,  9], 
1045                  [ 0, 15]], 
1046         <BLANKLINE> 
1047                 [[ 0,  0], 
1048                  [ 0,  0]]], 
1049         <BLANKLINE> 
1050         <BLANKLINE> 
1051                [[[ 0,  1], 
1052                  [ 0,  5]], 
1053         <BLANKLINE> 
1054                 [[ 0,  5], 
1055                  [ 0, 25]]]]) 
1056         >>> A.tensordot(A, axes=2).toarray() 
1057         array([[ 4,  6], 
1058                [ 0, 25]]) 
1059         >>> A.tensordot(A, axes=3) 
1060         array(39) 
1061  
1062         Using tuple for axes: 
1063  
1064         >>> a = scipy.sparse.coo_array(np.arange(60).reshape(3,4,5)) 
1065         >>> b = np.arange(24).reshape(4,3,2) 
1066         >>> c = a.tensordot(b, axes=([1,0],[0,1])) 
1067         >>> c.shape 
1068         (5, 2) 
1069         >>> c 
1070         array([[4400, 4730], 
1071                [4532, 4874], 
1072                [4664, 5018], 
1073                [4796, 5162], 
1074                [4928, 5306]]) 
1075  
1076         """
1077         if not isdense(other) and not issparse(other):
1078             # If it's a list or whatever, treat it like an array
1079             other_array = np.asanyarray(other)
1080 
1081             if other_array.ndim == 0 and other_array.dtype == np.object_:
1082                 raise TypeError(f"tensordot arg not supported type: '{type(other)}'")
1083             try:
1084                 other.shape
1085             except AttributeError:
1086                 other = other_array
1087 
1088         axes_self, axes_other = _process_axes(self.ndim, other.ndim, axes)
1089 
1090         # Check for shape compatibility along specified axes
1091         if any(self.shape[ax] != other.shape[bx]
1092                for ax, bx in zip(axes_self, axes_other)):
1093             raise ValueError("sizes of the corresponding axes must match")
1094 
1095         if isdense(other):
1096             return self._dense_tensordot(other, axes_self, axes_other)
1097         else:
1098             return self._sparse_tensordot(other, axes_self, axes_other)

1101     def _sparse_tensordot(self, other, axes_self, axes_other):
1102         ndim_self = len(self.shape)
1103         ndim_other = len(other.shape)
1104 
1105         # Prepare the tensors for tensordot operation
1106         # Ravel non-reduced axes coordinates
1107         self_non_red_coords = _ravel_non_reduced_axes(self.coords, self.shape,
1108                                                       axes_self)
1109         self_reduced_coords = np.ravel_multi_index(
1110             [self.coords[ax] for ax in axes_self], [self.shape[ax] for ax in axes_self])
1111         other_non_red_coords = _ravel_non_reduced_axes(other.coords, other.shape,
1112                                                        axes_other)
1113         other_reduced_coords = np.ravel_multi_index(
1114             [other.coords[a] for a in axes_other], [other.shape[a] for a in axes_other]
1115         )

1117         self_nonreduced_shape = tuple(self.shape[ax] for ax in range(ndim_self)

1141         coords = []

1143             if s:

1147             return sum(prod.data)

1180     def _matmul_sparse(self, other):
1181         """ 
1182         Perform sparse-sparse matrix multiplication for two n-D COO arrays. 
1183         The method converts input n-D arrays to 2-D block array format, 
1184         uses csr_matmat to multiply them, and then converts the 
1185         result back to n-D COO array. 
1186  
1187         Parameters: 
1188         self (COO): The first n-D sparse array in COO format. 
1189         other (COO): The second n-D sparse array in COO format. 
1190  
1191         Returns: 
1192         prod (COO): The resulting n-D sparse array after multiplication. 
1193         """
1194         if self.ndim < 3 and other.ndim < 3:
1195             return _spbase._matmul_sparse(self, other)
1196 
1197         # Get the shapes of self and other
1198         self_shape = self.shape
1199         other_shape = other.shape
1200 
1201         # Determine the new shape to broadcast self and other
1202         broadcast_shape = np.broadcast_shapes(self_shape[:-2], other_shape[:-2])
1203         self_new_shape = tuple(broadcast_shape) + self_shape[-2:]
1204         other_new_shape = tuple(broadcast_shape) + other_shape[-2:]
1205 
1206         self_broadcasted = self._broadcast_to(self_new_shape)
1207         other_broadcasted = other._broadcast_to(other_new_shape)
1208 
1209         # Convert n-D COO arrays to 2-D block diagonal arrays
1210         self_block_diag = _block_diag(self_broadcasted)
1211         other_block_diag = _block_diag(other_broadcasted)
1212 
1213         # Use csr_matmat to perform sparse matrix multiplication
1214         prod_block_diag = (self_block_diag @ other_block_diag).tocoo()
1215 
1216         # Convert the 2-D block diagonal array back to n-D
1217         return _extract_block_diag(
1218             prod_block_diag,
1219             shape=(*broadcast_shape, self.shape[-2], other.shape[-1]),
1220         )

1223     def _broadcast_to(self, new_shape, copy=False):
1224         if self.shape == new_shape:
1225             return self.copy() if copy else self
1226 
1227         old_shape = self.shape
1228 
1229         # Check if the new shape is compatible for broadcasting
1230         if len(new_shape) < len(old_shape):
1231             raise ValueError("New shape must have at least as many dimensions"
1232                              " as the current shape")
1233 
1234         # Add leading ones to shape to ensure same length as `new_shape`
1235         shape = (1,) * (len(new_shape) - len(old_shape)) + tuple(old_shape)
1236 
1237         # Ensure the old shape can be broadcast to the new shape
1238         if any((o != 1 and o != n) for o, n in zip(shape, new_shape)):
1239             raise ValueError(f"current shape {old_shape} cannot be "
1240                              "broadcast to new shape {new_shape}")
1241 
1242         # Reshape the COO array to match the new dimensions
1243         self = self.reshape(shape)
1244 
1245         idx_dtype = get_index_dtype(self.coords, maxval=max(new_shape))
1246         coords = self.coords
1247         new_data = self.data
1248         new_coords = coords[-1:]  # Copy last coordinate to start
1249         cum_repeat = 1 # Cumulative repeat factor for broadcasting
1250 
1251         if shape[-1] != new_shape[-1]: # broadcasting the n-th (col) dimension
1252             repeat_count = new_shape[-1]
1253             cum_repeat *= repeat_count
1254             new_data = np.tile(new_data, repeat_count)
1255             new_dim = np.repeat(np.arange(0, repeat_count, dtype=idx_dtype), self.nnz)
1256             new_coords = (new_dim,)
1257 
1258         for i in range(-2, -(len(shape)+1), -1):
1259             if shape[i] != new_shape[i]:
1260                 repeat_count = new_shape[i] # number of times to repeat data, coords
1261                 cum_repeat *= repeat_count # update cumulative repeat factor
1262                 nnz = len(new_data) # Number of non-zero elements so far
1263 
1264                 # Tile data and coordinates to match the new repeat count
1265                 new_data = np.tile(new_data, repeat_count)
1266                 new_coords = tuple(np.tile(new_coords[i+1:], repeat_count))
1267 
1268                 # Create new dimensions and stack them
1269                 new_dim = np.repeat(np.arange(0, repeat_count, dtype=idx_dtype), nnz)
1270                 new_coords = (new_dim,) + new_coords
1271             else:
1272                 # If no broadcasting needed, tile the coordinates
1273                 new_dim = np.tile(coords[i], cum_repeat)
1274                 new_coords = (new_dim,) + new_coords
1275 
1276         return coo_array((new_data, new_coords), new_shape)

1279 def _block_diag(self):
1280     """ 
1281     Converts an N-D COO array into a 2-D COO array in block diagonal form. 
1282  
1283     Parameters: 
1284     self (coo_array): An N-Dimensional COO sparse array. 
1285  
1286     Returns: 
1287     coo_array: A 2-Dimensional COO sparse array in block diagonal form. 
1288     """
1289     if self.ndim<2:
1290         raise ValueError("array must have atleast dim=2")
1291     num_blocks = math.prod(self.shape[:-2])
1292     n_col = self.shape[-1]
1293     n_row = self.shape[-2]
1294     res_arr = self.reshape((num_blocks, n_row, n_col))
1295     new_coords = (
1296         res_arr.coords[1] + res_arr.coords[0] * res_arr.shape[1],
1297         res_arr.coords[2] + res_arr.coords[0] * res_arr.shape[2],
1298     )
1299 
1300     new_shape = (num_blocks * n_row, num_blocks * n_col)
1301     return coo_array((self.data, tuple(new_coords)), shape=new_shape)

1304 def _extract_block_diag(self, shape):
1305     n_row, n_col = shape[-2], shape[-1]
1306 
1307     # Extract data and coordinates from the block diagonal COO array
1308     data = self.data
1309     row, col = self.row, self.col
1310 
1311     # Initialize new coordinates array
1312     new_coords = np.empty((len(shape), self.nnz), dtype=int)
1313 
1314     # Calculate within-block indices
1315     new_coords[-2] = row % n_row
1316     new_coords[-1] = col % n_col
1317 
1318     # Calculate coordinates for higher dimensions
1319     temp_block_idx = row // n_row
1320     for i in range(len(shape) - 3, -1, -1):
1321         size = shape[i]
1322         new_coords[i] = temp_block_idx % size
1323         temp_block_idx = temp_block_idx // size
1324 
1325     # Create the new COO array with the original n-D shape
1326     return coo_array((data, tuple(new_coords)), shape=shape)

1370 def _ravel_coords(coords, shape, order='C'):
1371     """Like np.ravel_multi_index, but avoids some overflow issues."""
1372     if len(coords) == 1:
1373         return coords[0]
1374     # Handle overflow as in https://github.com/scipy/scipy/pull/9132
1375     if len(coords) == 2:
1376         nrows, ncols = shape
1377         row, col = coords
1378         if order == 'C':
1379             maxval = (ncols * max(0, nrows - 1) + max(0, ncols - 1))
1380             idx_dtype = get_index_dtype(maxval=maxval)
1381             return np.multiply(ncols, row, dtype=idx_dtype) + col
1382         elif order == 'F':
1383             maxval = (nrows * max(0, ncols - 1) + max(0, nrows - 1))
1384             idx_dtype = get_index_dtype(maxval=maxval)
1385             return np.multiply(nrows, col, dtype=idx_dtype) + row
1386         else:
1387             raise ValueError("'order' must be 'C' or 'F'")
1388     return np.ravel_multi_index(coords, shape, order=order)
