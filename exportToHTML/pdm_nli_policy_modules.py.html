26   class SpanishNLIDetector:
27       """ 
28       NLI-based contradiction detector for Spanish text. 
29       Uses multilingual models fine-tuned for Spanish. 
30       """
31   
32       def __init__(self, light_mode: bool = False):
33           """ 
34           Initialize NLI detector. 
35    
36           Args: 
37               light_mode: Use lightweight model if True 
38           """
39           self.light_mode = light_mode
40   
41           # Select model based on mode
42           if light_mode:
43               # Lightweight multilingual model
44               model_name = "MoritzLaurer/mDeBERTa-v3-base-xnli-multilingual-nli-2mil7"
45           else:
46               # Full-size Spanish-optimized model
47               model_name = "Recognai/bert-base-spanish-wwm-cased-xnli"
48   
49           logger.info(f"Loading NLI model: {model_name}")
50   
51           # Load model and tokenizer
52           self.tokenizer = AutoTokenizer.from_pretrained(model_name)
53           self.model = AutoModelForSequenceClassification.from_pretrained(
54               model_name)
55   
56           # Move to GPU if available
57           self.device = torch.device(
58               "cuda" if torch.cuda.is_available() else "cpu")
59           self.model = self.model.to(self.device)
60           self.model.eval()
61   
62           # Label mapping
63           self.label_map = {0: "entailment", 1: "neutral", 2: "contradiction"}
64   
65           logger.info(f"NLI model loaded on {self.device}")
66   
67       def check_contradiction(
68           self, premise: str, hypothesis: str, return_all_scores: bool = False
69       ) -> Dict[str, any]:
70           """ 
71           Check if premise and hypothesis contradict each other. 
72    
73           Args: 
74               premise: First statement 
75               hypothesis: Second statement 
76               return_all_scores: Return scores for all labels 
77    
78           Returns: 
79               Dictionary with label, score, and optionally all scores 
80           """
81           # Tokenize inputs
82           inputs = self.tokenizer(
83               premise,
84               hypothesis,
85               truncation=True,
86               padding=True,
87               max_length=512,
88               return_tensors="pt",
89           ).to(self.device)
90   
91           # Get predictions
92           with torch.no_grad():
93               outputs = self.model(**inputs)
94               logits = outputs.logits
95               probs = torch.softmax(logits, dim=-1).cpu().numpy()[0]
96   
97           # Get predicted label
98           predicted_label_idx = np.argmax(probs)
99           predicted_label = self.label_map[predicted_label_idx]
100          confidence = float(probs[predicted_label_idx])
101  
102          result = {
103              "label": predicted_label,
104              "score": confidence,
105              "is_contradiction": predicted_label == "contradiction",
106          }
107  
108          if return_all_scores:
109              result["all_scores"] = {
110                  label: float(probs[idx]) for idx, label in self.label_map.items()
111              }
112  
113          return result
114  
115      def batch_check(
116          self, pairs: List[Tuple[str, str]], batch_size: int = 8
117      ) -> List[Dict[str, any]]:
118          """ 
119          Check multiple premise-hypothesis pairs in batches. 
120   
121          Args: 
122              pairs: List of (premise, hypothesis) tuples 
123              batch_size: Batch size for processing 
124   
125          Returns: 
126              List of results for each pair 
127          """
128          results = []
129  
130          for i in range(0, len(pairs), batch_size):
131              batch_pairs = pairs[i: i + batch_size]
132              premises, hypotheses = zip(*batch_pairs)
133  
134              # Tokenize batch
135              inputs = self.tokenizer(
136                  list(premises),
137                  list(hypotheses),
138                  truncation=True,
139                  padding=True,
140                  max_length=512,
141                  return_tensors="pt",
142              ).to(self.device)
143  
144              # Get batch predictions
145              with torch.no_grad():
146                  outputs = self.model(**inputs)
147                  logits = outputs.logits
148                  probs = torch.softmax(logits, dim=-1).cpu().numpy()
149  
150              # Process each result
151              for j, prob_dist in enumerate(probs):
152                  predicted_idx = np.argmax(prob_dist)
153                  predicted_label = self.label_map[predicted_idx]
154  
155                  results.append(
156                      {
157                          "premise": premises[j][:100],
158                          "hypothesis": hypotheses[j][:100],
159                          "label": predicted_label,
160                          "score": float(prob_dist[predicted_idx]),
161                          "is_contradiction": predicted_label == "contradiction",
162                          "contradiction_score": float(
163                              prob_dist[2]
164                          ),  # Index 2 is contradiction
165                      }
166                  )
167  
168          return results

170      def find_contradictory_pairs(
171          self, statements: List[str], threshold: float = 0.7
172      ) -> List[Dict[str, any]]:
173          """ 
174          Find all contradictory pairs in a list of statements. 
175   
176          Args: 
177              statements: List of statements to compare 
178              threshold: Minimum contradiction score 
179   
180          Returns: 
181              List of contradictory pairs 
182          """
183          contradictions = []
184  
185          # Generate all pairs
186          pairs = []
187          indices = []
188          for i in range(len(statements)):
189              for j in range(i + 1, len(statements)):
190                  pairs.append((statements[i], statements[j]))
191                  indices.append((i, j))
192  
193          if not pairs:
194              return contradictions
195  
196          # Check all pairs
197          results = self.batch_check(pairs)
198  
199          # Filter contradictions
200          for (i, j), result in zip(indices, results):
201              if result["contradiction_score"] >= threshold:
202                  contradictions.append(
203                      {
204                          "statement1": statements[i],
205                          "statement2": statements[j],
206                          "indices": (i, j),
207                          "contradiction_score": result["contradiction_score"],
208                          "confidence": result["score"],
209                      }
210                  )
211  
212          return contradictions

460      def validate_segment(
461          self, text: str, sectors: List[str], level: str = "municipal"
462      ) -> List[Dict[str, any]]:
463          """ 
464          Validate competences in a text segment. 
465   
466          Args: 
467              text: Text to validate 
468              sectors: Relevant sectors 
469              level: Government level (municipal/departmental/national) 
470   
471          Returns: 
472              List of competence validation results 
473          """
474          issues = []
475  
476          # Check for overreach patterns
477          for sector in sectors:
478              if sector in self.overreach_patterns:
479                  pattern = self.overreach_patterns[sector]
480  
481                  for match in pattern.finditer(text):
482                      # Extract context
483                      start = max(0, match.start() - 50)
484                      end = min(len(text), match.end() + 50)
485                      context = text[start:end]
486  
487                      # Check if it's mitigated by valid coordination verb
488                      if self.valid_patterns.search(context):
489                          continue
490  
491                      issue = {
492                          "type": "competence_overreach",
493                          "sector": sector,
494                          "level": level,
495                          "text": match.group(),
496                          "position": {"start": match.start(), "end": match.end()},
497                          "context": context,
498                          "required_level": self._get_required_level(
499                              match.group(), sector
500                          ),
501                          "legal_basis": self._get_legal_basis(sector),
502                          "explanation": self._explain_overreach(
503                              match.group(), sector, level
504                          ),
505                          "suggested_fix": self._suggest_fix(match.group(), sector),
506                      }
507  
508                      issues.append(issue)
509  
510          # Check for missing competences
511          issues.extend(self._check_missing_competences(text, sectors, level))
512  
513          return issues

515      def _get_required_level(self, action: str, sector: str) -> str:
516          """Determine required government level for an action."""
517          action_lower = action.lower()
518  
519          # Check each level's competences
520          for level, competences in [
521              ("municipal", self.matrix.municipal),
522              ("departmental", self.matrix.departmental),
523              ("national", self.matrix.national),
524          ]:
525              if sector in competences:
526                  for verb in competences[sector]:
527                      if verb in action_lower:
528                          return level
529  
530          # Check for specific keywords
531          if any(
532              word in action_lower
533              for word in ["hospital", "ESE nivel 2", "ESE nivel 3", "red hospitalaria"]
534          ):
535              return "departmental"
536  
537          if any(
538              word in action_lower
539              for word in [
540                  "docentes",
541                  "profesores",
542                  "n√≥mina educativa",

667  class RiskScorer:
668      """ 
669      Risk scoring with conformal prediction for PDM analysis. 
670      Provides calibrated uncertainty estimates. 
671      """
672  
673      def __init__(self, alpha: float = 0.1):
674          """ 
675          Initialize risk scorer. 
676   
677          Args: 
678              alpha: Significance level for conformal prediction (default 0.1 = 90% coverage) 
679          """
680          self.alpha = alpha
681          self.calibration_data = []
682          self.risk_model = None
683          self.conformal_predictor = None
684  
685      def calculate_risk(
686          self,
687          contradictions: List[Any],
688          competence_issues: List[Any],
689          agenda_gaps: List[Any],
690      ) -> Dict[str, Any]:
691          """ 
692          Calculate overall risk with uncertainty quantification. 
693   
694          Args: 
695              contradictions: List of detected contradictions 
696              competence_issues: List of competence validation issues 
697              agenda_gaps: List of agenda alignment gaps 
698   
699          Returns: 
700              Risk analysis with scores and confidence intervals 
701          """
702          # Calculate component scores
703          contradiction_score = self._score_contradictions(contradictions)
704          competence_score = self._score_competences(competence_issues)
705          agenda_score = self._score_agenda(agenda_gaps)
706  
707          # Combine scores
708          weights = {"contradiction": 0.4, "competence": 0.35, "agenda": 0.25}
709  
710          overall_risk = (
711              weights["contradiction"] * contradiction_score
712              + weights["competence"] * competence_score
713              + weights["agenda"] * agenda_score
714          )
715  
716          # Get confidence intervals using conformal prediction
717          confidence_intervals = self._get_confidence_intervals(
718              contradiction_score, competence_score, agenda_score
719          )
720  
721          # Determine risk level
722          risk_level = self._score_to_level(overall_risk)
723  
724          # Calculate empirical coverage if we have calibration data
725          empirical_coverage = (
726              self._calculate_coverage() if self.calibration_data else 0.9
727          )
728  
729          return {
730              "overall_risk": float(overall_risk),
731              "risk_level": (
732                  risk_level.value if hasattr(
733                      risk_level, "value") else risk_level
734              ),
735              "component_scores": {
736                  "contradiction": float(contradiction_score),
737                  "competence": float(competence_score),
738                  "agenda": float(agenda_score),
739              },
740              "confidence_intervals": confidence_intervals,
741              "empirical_coverage": empirical_coverage,
742              "calibration_samples": len(self.calibration_data),
743          }

745      def _score_contradictions(self, contradictions: List[Any]) -> float:
746          """Score contradiction severity."""
747          if not contradictions:
748              return 0.0
749  
750          scores = []
751          for c in contradictions:
752              # Base score from confidence
753              base_score = getattr(c, "confidence", 0.5)
754  
755              # Adjust for type
756              type_multiplier = 1.0
757              if hasattr(c, "type"):
758                  type_multipliers = {
759                      "semantic": 1.2,
760                      "quantitative": 1.1,
761                      "budgetary": 1.3,
762                      "temporal": 0.9,
763                  }
764                  type_str = str(c.type).lower()
765                  type_multiplier = type_multipliers.get(type_str, 1.0)
766  
767              # Adjust for NLI score if available
768              nli_boost = 0.0
769              if hasattr(c, "nli_score") and c.nli_score is not None:
770                  nli_boost = c.nli_score * 0.2
771  
772              score = min(1.0, base_score * type_multiplier + nli_boost)
773              scores.append(score)
774  
775          # Aggregate with emphasis on high scores
776          if scores:
777              # Use a combination of mean and max
778              mean_score = np.mean(scores)
779              max_score = np.max(scores)
780              # Normalize by expected count
781              count_factor = min(1.0, len(scores) / 10)
782  
783              return float(0.6 * mean_score + 0.3 * max_score + 0.1 * count_factor)
784  
785          return 0.0

787      def _score_competences(self, issues: List[Any]) -> float:
788          """Score competence issues severity."""
789          if not issues:
790              return 0.0
791  
792          scores = []
793          for issue in issues:
794              # Base severity
795              if isinstance(issue, dict):
796                  issue_type = issue.get("type", "")
797                  if "overreach" in issue_type:
798                      base_score = 0.8
799                  elif "missing" in issue_type:
800                      base_score = 0.6
801                  else:
802                      base_score = 0.5
803              else:
804                  base_score = getattr(issue, "confidence", 0.5)
805  
806              scores.append(base_score)
807  
808          if scores:
809              # Weight by count and max severity
810              mean_score = np.mean(scores)
811              max_score = np.max(scores)
812              count_penalty = min(0.3, len(scores) * 0.05)
813  
814              return float(min(1.0, 0.5 * mean_score + 0.2 * max_score + count_penalty))
815  
816          return 0.0

818      def _score_agenda(self, gaps: List[Any]) -> float:
819          """Score agenda alignment gaps."""
820          if not gaps:
821              return 0.0
822  
823          scores = []
824          for gap in gaps:
825              if isinstance(gap, dict):
826                  gap_type = gap.get("type", "")
827                  severity = gap.get("severity", "medium")
828  
829                  # Type-based scoring
830                  type_scores = {
831                      "missing_backward_alignment": 0.6,
832                      "missing_forward_element": 0.8,
833                      "broken_chain": 0.9,
834                  }
835                  base_score = type_scores.get(gap_type, 0.5)
836  
837                  # Severity adjustment
838                  severity_multipliers = {"low": 0.7, "medium": 1.0, "high": 1.3}
839                  multiplier = severity_multipliers.get(severity, 1.0)
840  
841                  scores.append(base_score * multiplier)
842              else:
843                  scores.append(0.5)
844  
845          if scores:
846              return float(min(1.0, np.mean(scores) + len(scores) * 0.02))
847  
848          return 0.0

850      def _get_confidence_intervals(
851          self, contradiction_score: float, competence_score: float, agenda_score: float
852      ) -> Dict[str, List[float]]:
853          """ 
854          Calculate confidence intervals using conformal prediction. 
855   
856          Returns: 
857              Dictionary with confidence intervals for each component 
858          """
859          # If we have enough calibration data, use conformal prediction
860          if len(self.calibration_data) >= 30:
861              intervals = self._conformal_intervals(
862                  contradiction_score, competence_score, agenda_score
863              )
864          else:
865              # Use bootstrap-based intervals as fallback
866              intervals = self._bootstrap_intervals(
867                  contradiction_score, competence_score, agenda_score
868              )
869  
870          return intervals

872      def _conformal_intervals(
873          self, contra_score: float, comp_score: float, agenda_score: float
874      ) -> Dict[str, List[float]]:
875          """Calculate conformal prediction intervals."""
876          # Prepare features
877          X = np.array([[contra_score, comp_score, agenda_score]])
878  
879          if self.conformal_predictor is None:
880              # Initialize and train conformal predictor
881              self._train_conformal_predictor()
882  
883          if self.conformal_predictor is not None:
884              # Get prediction intervals
885              y_pred, y_pis = self.conformal_predictor.predict(
886                  X, ensemble=True, alpha=self.alpha
887              )
888  
889              # Extract intervals
890              lower = float(y_pis[0, 0, 0])
891              upper = float(y_pis[0, 1, 0])
892  
893              # Component-wise intervals (simplified)
894              return {
895                  "overall": [lower, upper],
896                  "contradiction": [
897                      max(0, contra_score - 0.1),
898                      min(1, contra_score + 0.1),
899                  ],
900                  "competence": [max(0, comp_score - 0.1), min(1, comp_score + 0.1)],
901                  "agenda": [max(0, agenda_score - 0.1), min(1, agenda_score + 0.1)],
902              }
903          else:
904              # Fallback to bootstrap
905              return self._bootstrap_intervals(contra_score, comp_score, agenda_score)
906  
907      def _bootstrap_intervals(
908          self, contra_score: float, comp_score: float, agenda_score: float
909      ) -> Dict[str, List[float]]:
910          """Calculate bootstrap confidence intervals."""
911          # Simple bootstrap simulation
912          n_bootstrap = 1000
913          samples = []
914  
915          for _ in range(n_bootstrap):
916              # Add noise to simulate uncertainty
917              noise = np.random.normal(0, 0.05, 3)
918              sample_contra = np.clip(contra_score + noise[0], 0, 1)
919              sample_comp = np.clip(comp_score + noise[1], 0, 1)
920              sample_agenda = np.clip(agenda_score + noise[2], 0, 1)
921  
922              # Calculate overall with weights
923              sample_overall = (
924                  0.4 * sample_contra + 0.35 * sample_comp + 0.25 * sample_agenda
925              )
926              samples.append(sample_overall)
927  
928          # Calculate percentiles
929          lower = np.percentile(samples, (self.alpha / 2) * 100)
930          upper = np.percentile(samples, (1 - self.alpha / 2) * 100)
931  
932          return {
933              "overall": [float(lower), float(upper)],
934              "contradiction": [max(0, contra_score - 0.1), min(1, contra_score + 0.1)],
935              "competence": [max(0, comp_score - 0.1), min(1, comp_score + 0.1)],
936              "agenda": [max(0, agenda_score - 0.1), min(1, agenda_score + 0.1)],
937          }

939      def _train_conformal_predictor(self):
940          """Train conformal predictor on calibration data."""
941          if len(self.calibration_data) < 30:
942              logger.warning(
943                  "Insufficient calibration data for conformal prediction")
944              return
945  
946          try:
947              # Prepare training data
948              X = np.array(
949                  [[d["contra"], d["comp"], d["agenda"]]
950                      for d in self.calibration_data]
951              )
952              y = np.array([d["overall"] for d in self.calibration_data])
953  
954              # Split data
955              X_train, X_cal, y_train, y_cal = train_test_split(
956                  X, y, test_size=0.3, random_state=42
957              )
958  
959              # Base regressor
960              base_regressor = RandomForestRegressor(
961                  n_estimators=50, max_depth=5, random_state=42
962              )
963  
964              # Conformal predictor
965              self.conformal_predictor = MapieRegressor(
966                  estimator=base_regressor, method="plus", cv="prefit"
967              )
968  
969              # Fit on training data
970              base_regressor.fit(X_train, y_train)
971  
972              # Calibrate on calibration set
973              self.conformal_predictor.fit(X_cal, y_cal)
974  
975              logger.info("Conformal predictor trained successfully")
976  
977          except Exception as e:
978              logger.error(f"Failed to train conformal predictor: {e}")
979              self.conformal_predictor = None

981      def _calculate_coverage(self) -> float:
982          """Calculate empirical coverage from calibration data."""
983          if not self.calibration_data or len(self.calibration_data) < 10:
984              return 0.9  # Default assumption
985  
986          # Simple coverage calculation
987          in_interval_count = 0
988          for data in self.calibration_data[-100:]:  # Last 100 samples
989              if "interval" in data and "true_value" in data:
990                  lower, upper = data["interval"]
991                  if lower <= data["true_value"] <= upper:
992                      in_interval_count += 1
993  
994          if len(self.calibration_data) > 0:
995              return in_interval_count / min(100, len(self.calibration_data))
996          return 0.9

998      def _score_to_level(self, score: float) -> str:
999          """Convert numerical score to risk level."""
1000         if score >= 0.8:
1001             return "CRITICAL"
1002         elif score >= 0.6:
1003             return "HIGH"
1004         elif score >= 0.4:
1005             return "MEDIUM_HIGH"
1006         elif score >= 0.2:
1007             return "MEDIUM"
1008         else:
1009             return "LOW"
