3    import numpy as np
4    from numpy.testing import assert_, assert_equal
5    from scipy.linalg import clarkson_woodruff_transform

11   class TestClarksonWoodruffTransform:
12       """ 
13       Testing the Clarkson Woodruff Transform 
14       """
15       # set seed for generating test matrices
16       rng = np.random.default_rng(1179103485)
17   
18       # Test matrix parameters
19       n_rows = 2000
20       n_cols = 100
21       density = 0.1
22   
23       # Sketch matrix dimensions
24       n_sketch_rows = 200
25   
26       # Seeds to test with
27       seeds = [1755490010, 934377150, 1391612830, 1752708722, 2008891431,
28                1302443994, 1521083269, 1501189312, 1126232505, 1533465685]
29   
30       A_dense = rng.random((n_rows, n_cols))
31       A_csc = rand(
32           n_rows, n_cols, density=density, format='csc', random_state=rng,
33       )
34       A_csr = rand(
35           n_rows, n_cols, density=density, format='csr', random_state=rng,
36       )
37       A_coo = rand(
38           n_rows, n_cols, density=density, format='coo', random_state=rng,
39       )
40   
41       # Collect the test matrices
42       test_matrices = [
43           A_dense, A_csc, A_csr, A_coo,
44       ]
45   
46       # Test vector with norm ~1
47       x = rng.random((n_rows, 1)) / np.sqrt(n_rows)
48   
49       def test_sketch_dimensions(self):
50           for A in self.test_matrices:
51               for seed in self.seeds:
52                   # seed to ensure backwards compatibility post SPEC7
53                   sketch = clarkson_woodruff_transform(
54                       A, self.n_sketch_rows, seed=seed
55                   )
56                   assert_(sketch.shape == (self.n_sketch_rows, self.n_cols))
57   
58       def test_seed_returns_identical_transform_matrix(self):
59           for seed in self.seeds:
60               S1 = cwt_matrix(
61                   self.n_sketch_rows, self.n_rows, rng=seed
62               ).toarray()
63               S2 = cwt_matrix(
64                   self.n_sketch_rows, self.n_rows, rng=seed
65               ).toarray()
66               assert_equal(S1, S2)
67   
68       def test_seed_returns_identically(self):
69           for A in self.test_matrices:
70               for seed in self.seeds:
71                   sketch1 = clarkson_woodruff_transform(
72                       A, self.n_sketch_rows, rng=seed
73                   )
74                   sketch2 = clarkson_woodruff_transform(
75                       A, self.n_sketch_rows, rng=seed
76                   )
77                   if issparse(sketch1):
78                       sketch1 = sketch1.toarray()
79                   if issparse(sketch2):
80                       sketch2 = sketch2.toarray()
81                   assert_equal(sketch1, sketch2)
82   
83       def test_sketch_preserves_frobenius_norm(self):
84           # Given the probabilistic nature of the sketches
85           # we run the test multiple times and check that
86           # we pass all/almost all the tries.
87           n_errors = 0
88           for A in self.test_matrices:
89               if issparse(A):
90                   true_norm = norm(A)
91               else:
92                   true_norm = np.linalg.norm(A)
93               for seed in self.seeds:
94                   sketch = clarkson_woodruff_transform(
95                       A, self.n_sketch_rows, rng=seed,
96                   )
97                   if issparse(sketch):
98                       sketch_norm = norm(sketch)
99                   else:
100                      sketch_norm = np.linalg.norm(sketch)
101  
102                  if np.abs(true_norm - sketch_norm) > 0.1 * true_norm:
103                      n_errors += 1
104          assert_(n_errors == 0)
105  
106      def test_sketch_preserves_vector_norm(self):
107          n_errors = 0
108          n_sketch_rows = int(np.ceil(2. / (0.01 * 0.5**2)))
109          true_norm = np.linalg.norm(self.x)
110          for seed in self.seeds:
111              sketch = clarkson_woodruff_transform(
112                  self.x, n_sketch_rows, rng=seed,
113              )
114              sketch_norm = np.linalg.norm(sketch)
115  
116              if np.abs(true_norm - sketch_norm) > 0.5 * true_norm:
117                  n_errors += 1
118          assert_(n_errors == 0)
