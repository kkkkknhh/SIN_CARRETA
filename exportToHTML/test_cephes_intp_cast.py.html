1    import pytest
2    import numpy as np
3    from scipy.special._ufuncs import (
4        _smirnovc, _smirnovci, _smirnovp,
5        _struve_asymp_large_z, _struve_bessel_series, _struve_power_series,
6        bdtr, bdtrc, bdtri, expn, kn, nbdtr, nbdtrc, nbdtri, pdtri,
7        smirnov, smirnovi, yn
8    )
9    
10   
11   #
12   # For each ufunc here, verify that the default integer type, np.intp,
13   # can be safely cast to the integer type found in the input type signatures.
14   # For this particular set of functions, the code expects to find just one
15   # integer type among the input signatures.
16   #
17   @pytest.mark.parametrize(
18       'ufunc',
19       [_smirnovc, _smirnovci, _smirnovp,
20        _struve_asymp_large_z, _struve_bessel_series, _struve_power_series,
21        bdtr, bdtrc, bdtri, expn, kn, nbdtr, nbdtrc, nbdtri, pdtri,
22        smirnov, smirnovi, yn],
23   )
24   def test_intp_safe_cast(ufunc):
25       int_chars = {'i', 'l', 'q'}
26       int_input = [set(sig.split('->')[0]) & int_chars for sig in ufunc.types]
27       int_char = ''.join(s.pop() if s else '' for s in int_input)
28       assert len(int_char) == 1, "More integer types in the signatures than expected"
29       assert np.can_cast(np.intp, np.dtype(int_char))
