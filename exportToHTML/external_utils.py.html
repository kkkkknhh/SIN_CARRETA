3    """ 
4    This module contains utility functions that are explicitly allowed to be called during 
5    TorchDynamo compilation. These functions are carefully vetted to ensure they work 
6    correctly within the TorchDynamo tracing and compilation process. 
7     
8    Key functionality groups: 
9     
10   - Compilation State: 
11     Functions for checking compilation state (is_compiling) 
12    
13   - Function Wrapping: 
14     Utilities for wrapping functions (wrap_inline, wrap_numpy) to work with 
15     TorchDynamo compilation 
16    
17   - Autograd Hooks: 
18     Functions and classes for handling autograd hooks and backward passes 
19     (call_hook, FakeBackwardCFunction, etc.) 
20    
21   - Tensor Operations: 
22     Utility functions for tensor operations and transformations 
23   """

58           # NOTE: With `@torch.compile(backend="eager")`, torch._dynamo.is_compiling() will get traced
59           # and return true. torch.compiler.is_compiling() is skipped and will return false.

63   def wrap_inline(fn: Callable[_P, _R]) -> Callable[_P, _R]:
64       """ 
65       Create an extra frame around fn that is not in skipfiles. 
66       """
67   
68       @functools.wraps(fn)
69       def inner(*args: _P.args, **kwargs: _P.kwargs) -> _R:
70           return fn(*args, **kwargs)
71   
72       return inner

75   def call_hook(
76       hook: Callable[..., Optional[torch.Tensor]], *args: Any, **kwargs: Any
77   ) -> torch.Tensor:
78       """ 
79       Used by compiled autograd to handle hook returning None. 
80       """
81       result = hook(*args)
82       if result is None:
83           return args[0]
84       elif kwargs.get("hook_type") == "post_acc_grad_hook":
85           raise RuntimeError("Tensor post accumulate grad hooks should return None.")
86       return result

89   def wrap_numpy(f: Callable[_P, _R]) -> Callable[_P, _R]:
90       r"""Decorator that turns a function from ``np.ndarray``s to ``np.ndarray``s into a function 
91       from ``torch.Tensor``s to ``torch.Tensor``s. 
92       """
93       if not np:
94           return f
95   
96       @functools.wraps(f)
97       def wrap(*args: _P.args, **kwargs: _P.kwargs) -> pytree.PyTree:
98           args, kwargs = pytree.tree_map_only(
99               torch.Tensor, lambda x: x.numpy(), (args, kwargs)
100          )
101          out = f(*args, **kwargs)
102          return pytree.tree_map_only(np.ndarray, lambda x: torch.as_tensor(x), out)
103  
104      return wrap

107  class FakeBackwardCFunction:
108      def __init__(
109          self,
110          real: torch.autograd.function.BackwardCFunction,
111          saved_tensors: list[torch.Tensor],
112      ) -> None:
113          self.real = real
114          self.saved_tensors = saved_tensors
115  
116      def __getattr__(self, name: str) -> Any:
117          if name == "saved_variables":
118              warnings.warn(
119                  "'saved_variables' is deprecated; use 'saved_tensors'",
120                  DeprecationWarning,
121              )
122              return self.saved_tensors
123  
124          return getattr(self.real, name)

127  def call_backward(
128      backward_c_function: torch.autograd.function.BackwardCFunction,
129      saved_tensors: list[torch.Tensor],
130      *args: Any,
131  ) -> Union[torch.Tensor, tuple[torch.Tensor, ...]]:
132      fake = FakeBackwardCFunction(backward_c_function, saved_tensors)
133      grads = fake._forward_cls.backward(fake, *args)  # type: ignore[attr-defined]
134  
135      if not isinstance(grads, tuple):
136          grads = (grads,)
137  
138      return grads

141  def normalize_as_list(x: Any) -> list[Any]:
142      if isinstance(x, tuple):
143          return list(x)
144      elif isinstance(x, list):
145          return x
146      return [x]

153  class FakeCompiledAutogradEngine:
154      @staticmethod
155      def queue_callback(
156          final_callbacks: list[Callable[[], None]], cb: Callable[[], None]
157      ) -> None:
158          final_callbacks.append(cb)
159  
160      @staticmethod
161      def exec_final_callbacks(final_callbacks: list[Callable[[], None]]) -> None:
162          i = 0
163          while i < len(final_callbacks):
164              cb = final_callbacks[i]
165              cb()
166              i += 1
167          final_callbacks.clear()
168  
169      @staticmethod
170      def _exec_final_callbacks_stub() -> None:
171          pass

174  def call_hook_from_backward_state(
175      *args: Any, bw_state: Any, hook_name: str, **kwargs: Any
176  ) -> Any:
177      return getattr(bw_state, hook_name)(*args, **kwargs)

180  def call_module_hooks_from_backward_state(
181      _: Any, result: Any, *args: Any, bw_state: Any, hooks_name: str, module_name: str
182  ) -> Any:
183      module = getattr(bw_state, module_name)
184      hooks = getattr(bw_state, hooks_name)
185      for hook in hooks:
186          new_result = hook(module, result, *args)
187          if new_result is not None:
188              result = new_result
189      return result

193  def get_nonrecursive_disable_wrapper(fn: Callable[_P, _R]) -> Callable[_P, _R]:
194      # wrap function to get the right error message
195      # this function is in external_utils so that convert_frame doesn't skip it.
196      @functools.wraps(fn)
197      def nonrecursive_disable_wrapper(*args: _P.args, **kwargs: _P.kwargs) -> _R:
198          return fn(*args, **kwargs)
199  
200      return nonrecursive_disable_wrapper

203  def _dynamo_config_patch_proxy_dunder_call(
204      self: Any, func: Callable[_P, _R]
205  ) -> Callable[_P, _R]:
206      @functools.wraps(func)
207      def inner(*args: _P.args, **kwargs: _P.kwargs) -> _R:
208          with self:
209              return func(*args, **kwargs)
210  
211      return inner

214  # Use only on ints marked dynamic via torch.empty(0, integer)
215  # Currently only way to mark ints as dynamic: https://github.com/pytorch/pytorch/issues/129623
216  def unwrap_maybe_dynamic_int(x: Union[torch.Tensor, int]) -> int:
217      if isinstance(x, torch.Tensor):
218          # x.size() is expected to be [0, dynamic_int]
219          return x.size(1)
220      return x

223  def call_accumulate_grad(
224      variable: torch.Tensor, grad: torch.Tensor, has_post_hooks: bool
225  ) -> None:
226      updated_grad = torch._dynamo.compiled_autograd.ops.AccumulateGrad(  # type: ignore[attr-defined]
227          [grad], variable, variable.grad, has_post_hooks
228      )
229      variable.grad = updated_grad[0]
