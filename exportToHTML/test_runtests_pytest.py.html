33   class TestUpdateArgsWithPaths:
34   
35       @staticmethod
36       def test_no_paths():
37           """If no paths are passed, only `sympy` and `doc/src` are appended. 
38    
39           `sympy` and `doc/src` are the `testpaths` stated in `pytest.ini`. They 
40           need to be manually added as if any path-related arguments are passed 
41           to `pytest.main` then the settings in `pytest.ini` may be ignored. 
42    
43           """
44           paths = []
45           args = update_args_with_paths(paths=paths, keywords=None, args=[])
46           expected = [
47               str(pathlib.Path(sympy_dir(), 'sympy')),
48               str(pathlib.Path(sympy_dir(), 'doc/src')),
49           ]
50           assert args == expected
