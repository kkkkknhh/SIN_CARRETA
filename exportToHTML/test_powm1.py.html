14   powm1_test_cases = [
15       (1.25, 0.75, 0.18217701125396976, 1e-15),

17       (25.0, 5e-11, 1.6094379125636148e-10, 1e-15),

19       (0.9999999999990905, 20, -1.81898940353014e-11, 1e-15),

21   ]

24   @pytest.mark.parametrize('x, y, expected, rtol', powm1_test_cases)
25   def test_powm1(x, y, expected, rtol):
