111  class TorchTensor(ir.Tensor):
112      def __init__(self, tensor: torch.Tensor, name: str | None = None):
113          # Pass the tensor as the raw data to ir.Tensor's constructor
114          if tensor.dtype == torch.float4_e2m1fn_x2:
115              # Change the shape to the unpacked shape
116              shape = ir.Shape(_type_casting.get_float4_shape(tensor), frozen=True)
117          else:
118              # The base class will set the shape to the tensor's shape
119              shape = None
120          super().__init__(
121              tensor,
122              dtype=torch_dtype_to_onnx_dtype(tensor.dtype),
123              shape=shape,
124              name=name,
125          )
126  
127      def numpy(self) -> npt.NDArray:
128          self.raw: torch.Tensor
129  
130          # Handle dtypes that are not natively supported by NumPy:
131          # We pick an uint dtype that has the same size as the original dtype,
132          # view the tensor as that dtype so that it is convertible to NumPy,
133          # and then view it back to the proper dtype (using ml_dtypes obtained by
134          # calling dtype.numpy()).
135          if self.dtype == ir.DataType.BFLOAT16:
136              return (
137                  self.raw.view(torch.uint16).numpy(force=True).view(self.dtype.numpy())
138              )
139          if self.dtype in {
140              ir.DataType.FLOAT8E4M3FN,
141              ir.DataType.FLOAT8E4M3FNUZ,
142              ir.DataType.FLOAT8E5M2,
143              ir.DataType.FLOAT8E5M2FNUZ,
144          }:
145              return self.raw.view(torch.uint8).numpy(force=True).view(self.dtype.numpy())
146          if self.dtype == ir.DataType.FLOAT4E2M1:
147              return _type_casting.unpack_float4x2_as_uint8(self.raw).view(
148                  self.dtype.numpy()
149              )
150  
151          return self.raw.numpy(force=True)
152  
153      def __array__(self, dtype: Any = None, copy: bool | None = None) -> npt.NDArray:
154          del copy  # Unused, but needed for the signature
155          if dtype is None:
156              return self.numpy()
157          return self.numpy().__array__(dtype)
158  
159      def tobytes(self) -> bytes:
160          # Implement tobytes to support native PyTorch types so we can use types like bloat16
161          # Reading from memory directly is also more efficient because
162          # it avoids copying to a NumPy array
163          import torch._subclasses.fake_tensor
164  
165          with torch._subclasses.fake_tensor.unset_fake_temporarily():
166              # Disable any fake mode so calling detach() etc. will return a real tensor
167              tensor = self.raw.detach().cpu().contiguous()
168  
169          if isinstance(tensor, torch._subclasses.fake_tensor.FakeTensor):
170              raise TypeError(
171                  f"Cannot take content out from the FakeTensor ('{self.name}'). Please replace the tensor "
172                  "with a tensor backed by real data using ONNXProgram.apply_weights() "
173                  "or save the model without initializers by setting include_initializers=False."
174              )
175  
176          return bytes(
177              (ctypes.c_ubyte * tensor.element_size() * tensor.numel()).from_address(
178                  tensor.data_ptr()
179              )
180          )

205      meta_vals: Sequence[torch.Tensor],
206      complex_to_float: bool = True,
207  ) -> None:
208      if not isinstance(meta_vals, Sequence):
209          logger.warning(
210              "Expected meta_vals to be a sequence, but got %s. There may be an internal error.",
211              meta_vals,
212          )

218  def _set_shape_type(
219      value: ir.Value,
220      meta_val: torch.Tensor
221      | torch.SymBool
222      | torch.SymInt
223      | torch.SymFloat
224      | tuple[torch.Tensor],
225      complex_to_float: bool,
226  ) -> None:
227      if isinstance(meta_val, tuple):
228          logger.warning("Setting shape and type of tensors is not supported yet")
229      if isinstance(meta_val, torch.Tensor):
230          dims = []
231          shape: tuple[int, ...]
232          if meta_val.dtype == torch.float4_e2m1fn_x2:
233              # Change the shape to the unpacked shape
234              shape = _type_casting.get_float4_shape(meta_val)
235          else:
236              shape = meta_val.shape
237          for dim in shape:
238              if isinstance(dim, int):
239                  dims.append(dim)
240              else:
241                  dims.append(str(dim.node))
242  
243          # If the dtype is set already (e.g. by the onnx_symbolic ops),
244          # we don't need to set it again.
245          #
246          # When a user specifies complex in onnx_symbolic, we consider that to
247          # be the intention even though non of the ONNX ops deals with complex values.
248          # In this case, we don't change the dtype or the shape of the tensor.
249          if value.dtype is None:
250              value.dtype = torch_dtype_to_onnx_dtype(meta_val.dtype)
251              if complex_to_float:
252                  if meta_val.dtype == torch.complex64:
253                      value.dtype = ir.DataType.FLOAT
254                      # Add 2 as the last dimension if the tensor is complex to hold the real/imag parts
255                      dims.append(2)
256                  elif meta_val.dtype == torch.complex128:
257                      value.dtype = ir.DataType.DOUBLE
258                      # Add 2 as the last dimension if the tensor is complex to hold the real/imag parts
259                      dims.append(2)
260  
261          value.shape = ir.Shape(dims)
262      elif isinstance(meta_val, (int, torch.SymInt)):
263          # aten::sym_size output is a int, not a tensor, which stands
264          # for the size of one dim. We treat it as a scalar.
265          value.dtype = ir.DataType.INT64
266          value.shape = ir.Shape([])
267      elif isinstance(meta_val, (bool, torch.SymBool)):
268          value.dtype = ir.DataType.BOOL
269          value.shape = ir.Shape([])
270      elif isinstance(meta_val, (float, torch.SymFloat)):
271          value.dtype = ir.DataType.FLOAT
272          value.shape = ir.Shape([])
273      else:
274          pass

278      if isinstance(cls, str):
279          return cls
280      module = cls.__module__
281      if module is None or module == str.__class__.__module__:
282          return cls.__name__
283      return module + "." + cls.__name__

286  def _get_node_namespace(node: torch.fx.Node) -> tuple[str, list[str], list[str]]:
287      """Get the namespace and scope of the node. 
288   
289      Example:: 
290   
291          { 
292              'L__self__': ('', <class 'torchvision.models.resnet.ResNet'>), 
293              'L__self___avgpool': ('avgpool', <class 'torch.nn.modules.pooling.AdaptiveAvgPool2d'>) 
294          } 
295   
296      Will yield 
297   
298      namespace: ": torchvision.models.resnet.ResNet/avgpool: torch.nn.modules.pooling.AdaptiveAvgPool2d/node_name: node_target" 
299      class_hierarchy: ["torchvision.models.resnet.ResNet", "torch.nn.modules.pooling.AdaptiveAvgPool2d", <node_target>] 
300      name_scopes: ["", "avgpool", <node_name>] 
301   
302      Args: 
303          node: The node to get the namespace and scope of. 
304   
305      Returns: 
306          (namespace, class_hierarchy, name_scope) 
307      """
308      nn_module_stack = node.meta.get("nn_module_stack")
309      logger.debug("%s", nn_module_stack)
310      if nn_module_stack is None:
311          logger.warning(
312              "nn_module_stack not found for node '%s'. Skip adding metadata...",
313              node.name,
314          )
315          return f"{node.name}: {node.target}", [str(node.target)], [node.name]
316      namespaces = []
317      class_hierarchy = []
318      name_scopes = []
319      for name, nn_module in nn_module_stack.values():
320          name_scopes.append(name)
321          nn_module_name = _get_qualified_module_name(nn_module)
322          class_hierarchy.append(nn_module_name)
323          namespaces.append(f"{name}: {_get_qualified_module_name(nn_module)}")
324      namespaces.append(f"{node.name}: {node.target}")
325      class_hierarchy.append(str(node.target))
326      name_scopes.append(node.name)
327  
328      return "/".join(namespaces), class_hierarchy, name_scopes

343  def _handle_getitem_node(
344      node: torch.fx.Node, node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]]
345  ) -> ir.Value:
346      """Handle a getitem node. 
347   
348      Add the input value it is getting to the mapping, then return the value. 
349   
350      There are two cases for this node: 
351      1. The output is a Sequence (traced), we can simply get the value from the sequence 
352      2. The output is produced by a SplitToSequence node, we need to get the value from the sequence value 
353      This function only handles the first case 
354      """
355      assert len(node.all_input_nodes) == 1
356      source = node.all_input_nodes[0]
357      source_outputs = node_name_to_values[source.name]
358      assert isinstance(source_outputs, Sequence), (
359          f"Expected {source.name} to output sequence, got {node_name_to_values[source.name]}"
360      )
361      index = typing.cast(int, node.args[1])
362      value = source_outputs[index]
363      # Save the getitem value to the values mapping to in case
364      # it is one of the graph outputs
365      node_name_to_values[node.name] = value
366      # Rename the name of value with the getitem name.
367      value.name = node.name
368      return value

371  def _handle_call_function_node(
372      graph_like: ir.Graph | ir.Function,
373      node: torch.fx.Node,
374      node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]],
375  ) -> None:
376      """Handle a call_function node. 
377   
378      Args: 
379          graph: The ONNX graph at construction. 
380          node: The FX node to translate. 
381          node_name_to_values: A mapping of FX node names to their produced ir.Value. 
382      """
383      if node.target == operator.getitem:
384          _handle_getitem_node(node, node_name_to_values)
385      # Add op to the graph
386      op = str(node.target)
387      fx_inputs, attributes, input_names, output_names = _get_inputs_and_attributes(node)
388      inputs: list[ir.Value | None] = []
389      for i, input_ in enumerate(fx_inputs):
390          if input_ is None:
391              inputs.append(None)
392          elif hasattr(input_, "name"):
393              if isinstance(input_, torch.fx.Node) and input_.target == operator.getitem:
394                  actual_input = _handle_getitem_node(input_, node_name_to_values)
395                  inputs.append(actual_input)
396              else:
397                  value = node_name_to_values[input_.name]
398                  assert not isinstance(value, Sequence)
399                  inputs.append(value)
400          else:
401              attributes[f"arg_{i}"] = input_
402  
403      outputs = [ir.Value(name=name) for name in output_names]
404      if len(outputs) > 1:
405          _set_shape_types(outputs, node.meta["val"], complex_to_float=False)
406          node_name_to_values[node.name] = outputs
407      else:
408          _set_shape_type(outputs[0], node.meta["val"], complex_to_float=False)
409          node_name_to_values[node.name] = outputs[0]
410      ir_node = ir.Node(
411          "pkg.torch.ops",
412          op,
413          inputs,
414          attributes=ir_convenience.convert_attributes(attributes),
415          outputs=outputs,
416          name=node.name,
417      )
418      ir_node.meta["node"] = node
419      ir_node.metadata_props["pkg.torch.onnx.input_names"] = repr(input_names)
420      # Record the nn.Module stack for the node
421      _set_node_metadata(node, ir_node)
422  
423      graph_like.append(ir_node)

426  def _convert_fx_arg_to_onnx_arg(
427      arg,
428      node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]],
429      node_name_to_local_functions: dict[str, ir.Function],
430  ) -> Any:
431      """Convert an FX argument to an ONNX compatible argument. 
432   
433      This function 
434      - Converts a torch dtype to an integer 
435      - Converts a torch device/memory_format/layout to a string 
436      - Converts a torch.fx.Node to an ir.Value 
437      - Converts a sequence of torch.fx.Node to a sequence of ir.Value 
438      - Converts a get_attr node to an ir.Function 
439      """
440      if arg is None:
441          # None arguments are not modified because when the arg is an ONNX input
442          # we need to preserve the None value; when the arg is an ONNX attribute,
443          # we want to drop the value.
444          # The actual dropping of a None attribute value is done by OpRecorder
445          return None
446      if hasattr(arg, "name"):
447          if isinstance(arg, torch.fx.Node) and arg.target == operator.getitem:
448              source = arg.all_input_nodes[0]
449              source_outputs = node_name_to_values[source.name]
450              if isinstance(source_outputs, Sequence):
451                  # If the node is getting an input from another node, get the actual value the node is retrieving
452                  return _handle_getitem_node(arg, node_name_to_values)
453              else:
454                  # `source_outputs` is a sequence(tensor()) value and we need to
455                  # use SequenceAt to get the value. This is handled by torchlib
456                  pass
457          if isinstance(arg, torch.fx.Node) and arg.op == "get_attr":
458              return node_name_to_local_functions[arg.name]
459          # If the input is a node, get the value from the mapping
460          return node_name_to_values[arg.name]
461      if isinstance(arg, (list, tuple)):
462          return [
463              _convert_fx_arg_to_onnx_arg(
464                  elem, node_name_to_values, node_name_to_local_functions
465              )
466              for elem in arg
467          ]
468      if isinstance(arg, (torch.device, torch.memory_format, torch.layout)):
469          return str(arg)
470      if isinstance(arg, torch.dtype):
471          return torch_dtype_to_onnx_dtype(arg)
472      # Maybe a Python value
473      return arg

480  def _is_onnx_op(op: Any) -> bool:
481      """Whether the op overload is an ONNX custom op implemented with PyTorch."""
482      if not isinstance(op, torch._ops.OpOverload):
483          return False
484      return op.name().startswith("onnx::")

487  def _parse_onnx_op(op: torch._ops.OpOverload) -> tuple[str, int]:
488      """Parse the ONNX custom op overload name to get the op type and opset version."""
489      name = op.name()[len("onnx::") :]
490      name, _, opset = name.partition(".opset")
491      return name, int(opset)

494  def _handle_call_function_node_with_lowering(
495      model: ir.Model,
496      node: torch.fx.Node,
497      node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]],
498      *,
499      graph_like: ir.Graph | ir.Function,
500      constant_farm: dict[Any, ir.Value],
501      registry: _registration.ONNXRegistry,
502      opset: onnxscript.values.Opset,
503      node_name_to_local_functions: dict[str, ir.Function],
504  ) -> None:
505      """Translate a call_function node to an ONNX node. 
506   
507      Args: 
508          model: The ONNX model at construction. 
509          node: The FX node to translate. 
510          node_name_to_values: A mapping of FX node names to their produced ONNX ``Value``. 
511          graph_like: The current ONNX graph at construction. 
512              Must add nodes to this graph because it can be a subgraph that is currently being constructed. 
513          constant_farm: A mapping of constant values to existing ONNX ``Value``s. 
514          registry: The registry of all aten to ONNX decomposition functions. 
515          opset: The ONNX Script opset object for constructing ONNX nodes. 
516          node_name_to_local_functions: A mapping of subgraph names to the corresponding ONNX functions. 
517      """
518      if node.target == operator.getitem:
519          source = node.all_input_nodes[0]
520          source_outputs = node_name_to_values[source.name]
521          if isinstance(source_outputs, Sequence):
522              _handle_getitem_node(node, node_name_to_values)
523              return
524          else:
525              # `source_outputs` is a sequence(tensor()) value and we need to
526              # use SequenceAt to get the value. This is handled by torchlib
527              pass
528  
529      # Map FX inputs to ONNX inputs and fill optional inputs.
530      # torch_args and torch_kwargs are for op-level validation
531      fx_args = node.args
532      fx_kwargs = node.kwargs
533  
534      # Replace the input FX nodes with ONNX values
535      onnx_args = [
536          _convert_fx_arg_to_onnx_arg(
537              input_, node_name_to_values, node_name_to_local_functions
538          )
539          for input_ in fx_args
540      ]
541  
542      onnx_kwargs = {}
543      for key, value in fx_kwargs.items():
544          onnx_kwargs[key] = _convert_fx_arg_to_onnx_arg(
545              value, node_name_to_values, node_name_to_local_functions
546          )
547          if key == "dtype" and onnx_kwargs[key] is None:
548              # Set dtype to -1 if it is None
549              # TODO(justinchuby): Maybe keep it as None?
550              onnx_kwargs[key] = -1
551  
552      if _is_onnx_op(node.target):
553          # Handle torch.ops.onnx.* ops. These ops can be directly added to the graph
554          op_type, opset_version = _parse_onnx_op(node.target)  # type: ignore[arg-type]
555          # If final inputs are None, strip them from the node inputs
556          for input_ in reversed(onnx_args):
557              if input_ is not None:
558                  break
559              onnx_args.pop()
560          onnx_node = ir.Node(
561              "",
562              op_type,
563              onnx_args,
564              ir.convenience.convert_attributes(onnx_kwargs),
565              name=node.name,
566              num_outputs=len(node.target._schema.returns),  # type: ignore[union-attr]
567              version=opset_version,
568          )
569          # Store the single node in a list to be consistent with the rest of the code for further processing
570          onnx_nodes = [onnx_node]
571          if len(onnx_node.outputs) == 1:
572              outputs = onnx_node.outputs[0]
573          else:
574              outputs = onnx_node.outputs  # type: ignore[assignment]
575      else:
576          # Find the matching ONNX overload for the node
577          # TODO: Log the message here to expose false positives
578          onnx_function, message = _dispatching.dispatch(node, registry)
579  
580          if onnx_function is None:
581              raise _errors.DispatchError(
582                  f"No ONNX function found for {node.target!r}. Failure message: {message}"
583              )
584  
585          with onnxscript.evaluator.default_as(
586              tracer := _building.OpRecorder(opset, constant_farm)
587          ):
588              global current_tracer
589              current_tracer = tracer
590              try:
591                  outputs = onnx_function(*onnx_args, **onnx_kwargs)
592              except Exception as e:
593                  raise _errors.GraphConstructionError(
594                      f"Error when calling function '{onnx_function}' with args '{onnx_args}' and kwargs '{onnx_kwargs}'"
595                  ) from e
596              finally:
597                  current_tracer = None
598  
599          # Add the defined functions to the model
600          for identifier, onnxscript_function in tracer.functions.items():
601              if identifier in model.functions:
602                  continue
603              if isinstance(onnxscript_function, ir.Function):
604                  ir_function = onnxscript_function
605              else:
606                  # TODO: Get IR function directly when onnxscript is updated
607                  proto = onnxscript_function.to_function_proto()
608                  ir_function = ir.serde.deserialize_function(proto)
609              model.functions[identifier] = ir_function
610              # Opset imports are added to the model in the final add_opset_imports pass
611  
612          onnx_nodes = tracer.nodes
613          del tracer  # tracer is no longer needed
614  
615      # NOTE: Instead of using the output names from node.target._schema,
616      # we always use the index if there are more than one outputs so the
617      # names can be programmatically reconstructed. This is useful for
618      # comparing values from the ONNX graph with those from the FX graph.
619      #
620      # When there are multiple outputs, the output names will be
621      # node_name__0, node_name__1, etc.
622      if isinstance(outputs, Sequence):
623          _set_shape_types(outputs, node.meta["val"], complex_to_float=True)
624          node_name_to_values[node.name] = outputs
625          for i, output in enumerate(outputs):
626              output.name = f"{node.name}__{i}"
627              # Set the name of the producing node using the value name for correspondence
628              producer = output.producer()
629              if producer is not None:
630                  producer.name = f"node_{output.name}"
631      else:
632          _set_shape_type(outputs, node.meta["val"], complex_to_float=True)
633          node_name_to_values[node.name] = outputs
634          outputs.name = node.name
635          producer = outputs.producer()
636          if producer is not None:
637              producer.name = f"node_{outputs.name}"
638  
639      for ir_node in onnx_nodes:
640          ir_node.meta["node"] = node
641          # Record the nn.Module stack for the node
642          _set_node_metadata(node, ir_node)
643  
644      # Add the traced nodes to the current graph
645      # Must add nodes to this graph, not model.graph, because it can be a subgraph that is currently being constructed
646      graph_like.extend(onnx_nodes)

669  def _handle_get_attr_node(
670      node: torch.fx.Node,
671      *,
672      owned_graphs: Mapping[str, ir.Function],
673      node_name_to_local_functions: dict[str, ir.Function],
674  ) -> None:
675      """Handle a get_attr node by assigning the corresponding ONNX function to the node name. 
676   
677      An example ExportedProgram that has uses get_attr nodes is: 
678   
679          ExportedProgram: 
680          class GraphModule(torch.nn.Module): 
681              def forward(self, arg0_1: "f32[5]"): 
682                  true_graph_0 = self.true_graph_0  # get_attr 
683                  false_graph_0 = self.false_graph_0  # get_attr 
684                  conditional = torch.ops.higher_order.cond(False, true_graph_0, false_graph_0, [arg0_1]);  true_graph_0 = false_graph_0 = arg0_1 = None 
685                  getitem: "f32[5]" = conditional[0];  conditional = None 
686                  return (getitem,) 
687   
688              class <lambda>(torch.nn.Module): 
689                  def forward(self, arg0_1: "f32[5]"): 
690                      cos: "f32[5]" = torch.ops.aten.cos.default(arg0_1);  arg0_1 = None 
691                      return (cos,) 
692   
693              class <lambda>(torch.nn.Module): 
694                  def forward(self, arg0_1: "f32[5]"): 
695                      sin: "f32[5]" = torch.ops.aten.sin.default(arg0_1);  arg0_1 = None 
696                      return (sin,) 
697   
698      Args: 
699          node: The FX node to translate. 
700          owned_graphs: A mapping of subgraph names to the corresponding ONNX functions. 
701          node_name_to_local_functions: A mapping of local function names to their corresponding ONNX functions. 
702      """
703      if not isinstance(node.target, str):
704          logger.warning(
705              "Expected node.target for the node %s to be a string, but got '%s'. There may be an internal error.",
706              node,
707              type(node.target),
708          )
709          return
710      function = owned_graphs[node.target]
711      node_name_to_local_functions[node.name] = function

714  def _handle_output_node(
715      node: torch.fx.Node,
716      node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]],
717      graph_like: ir.Graph | ir.Function,
718  ) -> None:
719      """Handle an output node by adding the output to the graph's outputs. 
720   
721      Args: 
722          node: The FX node to translate. 
723          node_name_to_values: A mapping of FX node names to their produced ONNX ``Value``. 
724          graph_like: The ONNX graph at construction. 
725      """
726      # node.args[0] can be a tuple with more than one elements. This happens when,
727      # for example, a subgraph has multiple outputs. We flatten them all as ONNX graph outputs
728      for output in node.args[0]:  # type: ignore[index,union-attr]
729          output_value_name = output.name  # type: ignore[union-attr]
730          assert isinstance(output_value_name, str), (
731              f"Bug: Expected {output_value_name!r} to be a string"
732          )
733          values = node_name_to_values[output_value_name]
734          if isinstance(values, Sequence):
735              graph_like.outputs.extend(values)
736              return
737          graph_like.outputs.append(values)

740  def _translate_fx_graph(
741      fx_graph: torch.fx.Graph,
742      model: ir.Model,
743      *,
744      graph_like: ir.Graph | ir.Function,
745      owned_graphs: Mapping[str, ir.Function],
746      lower: Literal["at_conversion", "none"],
747      registry: _registration.ONNXRegistry,
748  ) -> dict[str, ir.Value | Sequence[ir.Value]]:
749      """Translate a submodule to an ONNX function. 
750   
751      Any functions used by the traced functions will be added to the model. 
752   
753      Args: 
754          fx_graph: The FX graph module to translate. 
755          model: The ONNX model at construction. 
756          current_scope: The current name scope of the submodule, excluding the current module name. 
757              E.g. "true_graph_0.false_graph_0". 
758          graph_name: The name of the submodule. E.g. "true_graph_0". 
759          graph: The ONNX graph at construction. 
760          owned_graphs: The subgraphs owned by the current graph. 
761          lower: The lowering strategy to use. 
762          registry: The registry of all aten to ONNX decomposition functions. 
763   
764      Returns: 
765          A mapping of FX node names to their produced ONNX ``Value``. 
766      """
767      node_name_to_values: dict[str, ir.Value | Sequence[ir.Value]] = {}
768      # The reason we need node_name_to_local_functions in addition to owned_graphs
769      # is because the get_attr nodes may assign a different name than the GraphModule name
770      # to the subgraph. This is not typical but is valid Python.
771      node_name_to_local_functions: dict[str, ir.Function] = {}
772      constant_farm: dict[Any, ir.Value] = {}
773      opset = _get_onnxscript_opset(registry.opset_version)
774  
775      for node in fx_graph.nodes:
776          logger.debug(
777              "%s", (node.name, node.args, node.target, node.op, node.type, node.kwargs)
778          )
779          try:
780              if node.op == "placeholder":
781                  _handle_placeholder_node(
782                      node,
783                      node_name_to_values,
784                      graph_like=graph_like,
785                      lower=lower,
786                      opset=opset,
787                  )
788              elif node.op == "call_function":
789                  if lower == "at_conversion":
790                      _handle_call_function_node_with_lowering(
791                          model,
792                          node,
793                          node_name_to_values,
794                          graph_like=graph_like,
795                          constant_farm=constant_farm,
796                          registry=registry,
797                          opset=opset,
798                          node_name_to_local_functions=node_name_to_local_functions,
799                      )
800                  else:
801                      # No lowering
802                      _handle_call_function_node(graph_like, node, node_name_to_values)
803              elif node.op == "get_attr":
804                  _handle_get_attr_node(
805                      node,
806                      owned_graphs=owned_graphs,
807                      node_name_to_local_functions=node_name_to_local_functions,
808                  )
809              elif node.op == "output":
810                  _handle_output_node(
811                      node,
812                      node_name_to_values,
813                      graph_like=graph_like,
814                  )
815          except Exception as e:
816              raise _errors.ConversionError(
817                  f"Error when translating node {node.format_node()}. See the stack trace for more information."
818              ) from e
819      return node_name_to_values

822  def _get_inputs_and_attributes(
823      node: torch.fx.Node,
824  ) -> tuple[list[torch.fx.Node | None], dict[str, Any], list[str], list[str]]:
825      """Find and Fill in the not provided kwargs with default values. 
826   
827      Returns: 
828          (inputs, attributes, input_names, output_names) 
829      """
830      if inspect.isbuiltin(node.target) or isinstance(node.target, str):
831          inputs = list(node.args)
832          return inputs, {}, [], [node.name]  # type: ignore[return-value]
833  
834      # The target should be an ATen operator now
835      assert hasattr(node.target, "_schema"), (
836          f"The target should be an ATen operator now, but node target {node.target} has no schema"
837      )
838      node_schema: torch.FunctionSchema = node.target._schema
839  
840      # This function assumes the order of arguments in FX op is the
841      # same as the order of arguments in TorchScript op.
842      inputs: list[Any] = []  # type: ignore[no-redef]
843      input_names: list[str] = []
844      attributes: dict[str, Any] = {}
845  
846      if inspect.isbuiltin(node.target):
847          inputs = list(node.args)
848      else:
849          for arg, schema_arg in zip(node.args, node_schema.arguments):
850              if arg is None or isinstance(arg, torch.fx.Node):
851                  inputs.append(arg)
852                  input_names.append(schema_arg.name)
853              elif isinstance(arg, Sequence) and all(
854                  elem is None or isinstance(elem, torch.fx.Node) for elem in arg
855              ):
856                  inputs.extend(arg)
857                  input_names.extend([schema_arg.name] * len(arg))
858              elif isinstance(arg, torch.device):
859                  attributes[schema_arg.name] = str(arg)
860              elif isinstance(arg, torch.dtype):
861                  attributes[schema_arg.name] = torch_dtype_to_onnx_dtype(arg)
862              else:
863                  attributes[schema_arg.name] = arg
864          for schema_arg in node_schema.arguments:
865              if schema_arg.name not in node.kwargs:
866                  continue
867              kwarg = node.kwargs[schema_arg.name]
868              if schema_arg.name in {
869                  "layout",
870                  "device",
871                  "requires_grad",
872                  "memory_format",
873                  "implicit",
874              } or isinstance(kwarg, torch.device):
875                  attr = str(kwarg)
876              elif isinstance(kwarg, torch.dtype):
877                  attr = torch_dtype_to_onnx_dtype(kwarg)  # type: ignore[assignment]
878              else:
879                  attr = kwarg  # type: ignore[assignment]
880  
881              attributes[schema_arg.name] = attr
882  
883      output_names = [f"{node.name}_{output.name}" for output in node_schema.returns]
884  
885      return inputs, attributes, input_names, output_names  # type: ignore[return-value]

985  def _get_scope_name(scoped_name: str) -> tuple[str, str]:
986      """Get the scope and name of a node. 
987   
988      Examples:: 
989          >>> _get_scope_name('') 
990          ('', '') 
991          >>> _get_scope_name('true_graph') 
992          ('', 'true_graph') 
993          >>> _get_scope_name('true_graph.false_graph') 
994          ('true_graph', 'false_graph') 
995          >>> _get_scope_name('true_graph.false_graph.some_graph') 
996          ('true_graph.false_graph', 'some_graph') 
997   
998      Args: 
999          scoped_name: The scoped name of the node. 
1000  
1001     Returns: 
1002         (scope, name) 
1003     """

1005         scope, name = scoped_name.rsplit(".", 1)
1006     else:
1007         scope, name = "", scoped_name
1008     return scope, name

1011 def _exported_program_to_onnx_program(
1012     exported_program: torch.export.ExportedProgram,
1013     *,
1014     registry: _registration.ONNXRegistry,
1015     lower: Literal["at_conversion", "none"] = "at_conversion",
1016 ) -> _onnx_program.ONNXProgram:
1017     """Convert an exported program to an ONNX Program. 
1018  
1019     The exported_program field in the returned ONNXProgram is one that is after 
1020     decompositions have been applied. 
1021  
1022     Reference: 
1023         - ExportedProgram spec: https://pytorch.org/docs/stable/export.ir_spec.html 
1024  
1025     Args: 
1026         exported_program: The exported program to convert. The exported program 
1027             should be the one that is after decompositions have been applied. 
1028         lower: Whether to lower the graph to core ONNX operators. 
1029             at_conversion: Lower whe translating the FX graph to ONNX IR. 
1030             none: Do not lower the graph. 
1031         registry: The registry of all ONNX Script decomposition. 
1032     """
1033     model = ir.Model(
1034         graph=ir.Graph(
1035             [],
1036             [],
1037             nodes=[],
1038             # Opset imports are added to the model in the final add_opset_imports pass
1039             name="main_graph",
1040             metadata_props={
1041                 "pkg.torch.export.ExportedProgram.graph_signature": str(
1042                     exported_program.graph_signature
1043                 ),
1044                 "pkg.torch.export.ExportedProgram.range_constraints": str(
1045                     exported_program.range_constraints
1046                 ),
1047             },
1048         ),
1049         ir_version=_constants.ONNX_IR_VERSION,
1050         producer_name="pytorch",
1051         producer_version=torch.__version__,
1052     )
1053 
1054     # A dictionary storing the translated subgraphs as ONNX functions made available to outer graphs
1055     # {<subgraph_scope>: {<subgraph_name>: <IR function>}}
1056     scoped_subgraphs: dict[str, dict[str, ir.Function]] = {}
1057     values = None
1058 
1059     # 1. Translate all nodes in all subgraphs and the main graph
1060     # Create a dictionary of values for the main graph for step 2-3 to add inputs and outputs
1061     module: torch.fx.GraphModule
1062     # Reverse the order of the modules so that the innermost module is processed first
1063     # and made available to the outer module
1064     for name, module in reversed(
1065         tuple(exported_program.graph_module.named_modules(remove_duplicate=False))
1066     ):
1067         # Obtain the graphs (previously built) owned by the current module
1068         owned_graphs = scoped_subgraphs.setdefault(name, {})
1069         fx_graph = module.graph
1070 
1071         graph_like: ir.Graph | ir.Function
1072         if name == "":
1073             # Root graph
1074             graph_like = model.graph
1075         else:
1076             function_name = name.replace(".", "__")
1077             # Inputs and outputs will be created within _translate_fx_graph
1078             func = ir.Function(
1079                 domain=_constants.LOCAL_FUNCTION_DOMAIN,
1080                 name=function_name,
1081                 graph=ir.Graph((), (), nodes=()),
1082                 attributes=(),
1083             )
1084             # Make this function available to the outer graph
1085             scope, subgraph_name = _get_scope_name(name)
1086             scoped_subgraphs.setdefault(scope, {})[subgraph_name] = func
1087             model.functions[func.identifier()] = func
1088             graph_like = func
1089 
1090         values = _translate_fx_graph(
1091             fx_graph,
1092             model,
1093             graph_like=graph_like,
1094             owned_graphs=owned_graphs,
1095             lower=lower,
1096             registry=registry,
1097         )
1098 
1099     assert name == "", "The last module processed should be the root module"
1100     assert values is not None
1101 
1102     # Clear the input/output of the main graph and add them back in step 2-3
1103     # using the more accurate graph signature
1104     model.graph.inputs.clear()
1105     model.graph.outputs.clear()
1106 
1107     # 2. Add user inputs and all parameters/buffers to the graph.
1108     # Since the node names and the tensor names are different, we need to rename
1109     # the nodes to match the tensor names later. For now we will just use the node names.
1110     user_inputs = [
1111         spec
1112         for spec in exported_program.graph_signature.input_specs
1113         if spec.kind == graph_signature.InputKind.USER_INPUT
1114     ]
1115     non_user_inputs = [
1116         spec
1117         for spec in exported_program.graph_signature.input_specs
1118         if spec.kind != graph_signature.InputKind.USER_INPUT
1119     ]
1120 
1121     for spec in itertools.chain(user_inputs, non_user_inputs):
1122         # Put the user inputs first and then the parameters/buffers
1123         if isinstance(spec.arg, graph_signature.ConstantArgument):
1124             logger.debug("Skipping constant argument %s", spec.arg)
1125             continue
1126         value_name = spec.arg.name
1127         input_kind = spec.kind
1128         persistent = spec.persistent
1129         value = values[value_name]
1130 
1131         assert not isinstance(value, Sequence), (
1132             f"Input '{value_name}' should not be a sequence. This is unexpected."
1133         )
1134 
1135         value.metadata_props["pkg.torch.export.graph_signature.InputSpec.kind"] = (
1136             input_kind.name
1137         )
1138         value.metadata_props[
1139             "pkg.torch.export.graph_signature.InputSpec.persistent"
1140         ] = str(persistent)
1141 
1142         if input_kind == graph_signature.InputKind.USER_INPUT:
1143             # Add only user inputs to the graph
1144             # Subsequent passes can decide if they want to add initializers as inputs
1145             model.graph.inputs.append(value)
1146         else:
1147             model.graph.initializers[value_name] = value
1148 
1149     # 3. Add user outputs to the graph and assign metadata to all outputs
1150     user_outputs = [
1151         spec
1152         for spec in exported_program.graph_signature.output_specs
1153         if spec.kind == graph_signature.OutputKind.USER_OUTPUT
1154     ]
1155     non_user_outputs = [
1156         spec
1157         for spec in exported_program.graph_signature.output_specs

1161         if isinstance(spec.arg, graph_signature.ConstantArgument):
1162             logger.warning("Skipping constant argument %s", spec.arg)
1163             continue

1165         output_kind = spec.kind
1166         value = values[value_name]
1167 
1168         if not isinstance(value, (ir.Value, Sequence)):
1169             raise TypeError(
1170                 f"Output '{value_name}' should be an ir.Value. Actual type is '{type(value)}': {value!r}. "
1171                 "This may be due to an incorrect implementation of the ONNX function that produced this output."
1172             )
1173 
1174         # The output value may be a sequence, meaning the operator has multiple outputs
1175         _values = (value,) if not isinstance(value, Sequence) else value
1176

1178             logger.warning(
1179                 "Model output '%s' has multiple values: %s (output spec: %s). Please make sure this is expected.",
1180                 value_name,
1181                 _values,
1182                 spec,
1183             )

1185         for value in _values:
1186             value.metadata_props["pkg.torch.export.graph_signature.OutputSpec.kind"] = (
1187                 output_kind.name
1188             )
1189             if output_kind == graph_signature.OutputKind.USER_OUTPUT:
1190                 model.graph.outputs.append(value)
1191 
1192     # 4. Rename the initializers to match the tensor names
1193     for name, param_name in itertools.chain(
1194         exported_program.graph_signature.inputs_to_parameters.items(),
1195         exported_program.graph_signature.inputs_to_buffers.items(),
1196         exported_program.graph_signature.inputs_to_lifted_tensor_constants.items(),
1197     ):
1198         initializer = model.graph.initializers.pop(name)
1199         initializer.name = param_name
1200         # Record the original name so users can search the metadata and correspond
1201         # with the FX graph
1202         initializer.metadata_props["pkg.torch.onnx.original_node_name"] = name
1203         model.graph.initializers[param_name] = initializer
1204 
1205     # 5. Add initializers to the graph
1206     # ExportedProgram stores parameters and buffers in state_dict,
1207     # but non_persistent_buffers and lifted_tensor_constants are not there
1208     # so we need to get them from the name_* apis.
1209     for name, torch_tensor in itertools.chain(
1210         exported_program.named_parameters(),
1211         exported_program.named_buffers(),
1212         exported_program.constants.items(),
1213     ):
1214         initializer = model.graph.initializers.get(name)  # type: ignore[assignment]
1215         if initializer is None:
1216             logger.warning("Tensor '%s' is not one of the initializers", name)
1217             continue
1218         if not isinstance(torch_tensor, torch.Tensor):
1219             raise NotImplementedError(
1220                 f"Tensor '{name}' should be a torch.Tensor. Actual type is '{type(torch_tensor)}': {torch_tensor!r}. "
1221                 "This is unexpected and not yet supported."
1222             )
1223         ir_tensor = TorchTensor(torch_tensor, name=name)
1224         initializer.const_value = ir_tensor
1225         _set_shape_type(
1226             initializer,
1227             torch_tensor,
1228             complex_to_float=lower != "none",
1229         )
1230 
1231     # TODO: Decide if we should keep mutated buffers as inputs/outputs
1232 
1233     # TODO(justinchuby): Remove the hack
1234     _ir_passes.add_torchlib_common_imports(model)
1235 
1236     # Collect and add opset imports to the model
1237     _ir_passes.add_opset_imports(model)
1238 
1239     return _onnx_program.ONNXProgram(model, exported_program)
