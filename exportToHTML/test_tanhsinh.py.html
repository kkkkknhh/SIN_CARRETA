28   def _vectorize(xp):
29       # xp-compatible version of np.vectorize
30       # assumes arguments are all arrays of the same shape
31       def decorator(f):
32           def wrapped(*arg_arrays):
33               shape = arg_arrays[0].shape
34               arg_arrays = [xp_ravel(arg_array) for arg_array in arg_arrays]
35               res = []
36               for i in range(math.prod(shape)):
37                   arg_scalars = [arg_array[i] for arg_array in arg_arrays]
38                   res.append(f(*arg_scalars))
39               return res
40   
41           return wrapped
42   
43       return decorator

519              return -norm_logpdf(x)*norm_pdf(x)

877      def test_integral(self, maxterms, xp):
878          # test precise behavior of integral approximation
879          f = self.f1
880  
881          def logf(x):
882              return -2*xp.log(x)
883  
884          def F(x):
885              return -1 / x
886  
887          a = xp.asarray([1, 5], dtype=xp.float64)[:, xp.newaxis]
888          b = xp.asarray([20, 100, xp.inf], dtype=xp.float64)[:, xp.newaxis, xp.newaxis]
889          step = xp.asarray([0.5, 1, 2], dtype=xp.float64).reshape((-1, 1, 1, 1))
890          nsteps = xp.floor((b - a)/step)
891          b_original = b
892          b = a + nsteps*step
893  
894          k = a + maxterms*step
895          # partial sum
896          direct = xp.sum(f(a + xp.arange(maxterms)*step), axis=-1, keepdims=True)
897          integral = (F(b) - F(k))/step  # integral approximation of remainder
898          low = direct + integral + f(b)  # theoretical lower bound
899          high = direct + integral + f(k)  # theoretical upper bound
900          ref_sum = (low + high)/2  # nsum uses average of the two
901          ref_err = (high - low)/2  # error (assuming perfect quadrature)
902  
903          # correct reference values where number of terms < maxterms

905          a, b, step = xp_test.broadcast_arrays(a, b, step)
906          for i in np.ndindex(a.shape):
907              ai, bi, stepi = float(a[i]), float(b[i]), float(step[i])
908              if (bi - ai)/stepi + 1 <= maxterms:
909                  direct = xp.sum(f(xp.arange(ai, bi+stepi, stepi, dtype=xp.float64)))
910                  ref_sum[i] = direct
911                  ref_err[i] = direct * xp.finfo(direct.dtype).eps
912  
913          rtol = 1e-12
914          res = nsum(f, a, b_original, step=step, maxterms=maxterms,
915                     tolerances=dict(rtol=rtol))
916          xp_assert_close(res.sum, ref_sum, rtol=10*rtol)
917          xp_assert_close(res.error, ref_err, rtol=100*rtol)
