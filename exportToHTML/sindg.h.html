5    /*                                                     sindg.c 
6     * 
7     *     Circular sine of angle in degrees 
8     * 
9     * 
10    * 
11    * SYNOPSIS: 
12    * 
13    * double x, y, sindg(); 
14    * 
15    * y = sindg( x ); 
16    * 
17    * 
18    * 
19    * DESCRIPTION: 
20    * 
21    * Range reduction is into intervals of 45 degrees. 
22    * 
23    * Two polynomial approximating functions are employed. 
24    * Between 0 and pi/4 the sine is approximated by 
25    *      x  +  x**3 P(x**2). 
26    * Between pi/4 and pi/2 the cosine is represented as 
27    *      1  -  x**2 P(x**2). 
28    * 
29    * 
30    * 
31    * ACCURACY: 
32    * 
33    *                      Relative error: 
34    * arithmetic   domain      # trials      peak         rms 
35    *    IEEE      +-1000       30000      2.3e-16      5.6e-17 
36    * 
37    * ERROR MESSAGES: 
38    * 
39    *   message           condition        value returned 
40    * sindg total loss   x > 1.0e14 (IEEE)     0.0 
41    * 
42    */

93           constexpr double sincof[] = {1.58962301572218447952E-10, -2.50507477628503540135E-8,
94                                        2.75573136213856773549E-6,  -1.98412698295895384658E-4,

99                                        4.99999999999999999798E-1};

101          constexpr double sindg_lossth = 1.0e14;

105      XSF_HOST_DEVICE inline double sindg(double x) {
106          double y, z, zz;
107          int j, sign;
108  
109          /* make argument positive but save the sign */
110          sign = 1;
111          if (x < 0) {
112              x = -x;
113              sign = -1;
114          }
115  
116          if (x > detail::sindg_lossth) {
117              set_error("sindg", SF_ERROR_NO_RESULT, NULL);
118              return (0.0);
119          }
120  
121          y = std::floor(x / 45.0); /* integer part of x/M_PI_4 */
122  
123          /* strip high bits of integer part to prevent integer overflow */
124          z = std::ldexp(y, -4);
125          z = std::floor(z);        /* integer part of y/8 */
126          z = y - std::ldexp(z, 4); /* y - 16 * (y/16) */
127  
128          j = z; /* convert to integer for tests on the phase angle */
129          /* map zeros to origin */
130          if (j & 1) {
131              j += 1;
132              y += 1.0;
133          }
134          j = j & 07; /* octant modulo 360 degrees */
135          /* reflect in x axis */
136          if (j > 3) {
137              sign = -sign;
138              j -= 4;
139          }
140  
141          z = x - y * 45.0;   /* x mod 45 degrees */
142          z *= detail::PI180; /* multiply by pi/180 to convert to radians */
143          zz = z * z;
144  
145          if ((j == 1) || (j == 2)) {
146              y = 1.0 - zz * polevl(zz, detail::coscof, 6);
147          } else {
148              y = z + z * (zz * polevl(zz, detail::sincof, 5));
149          }
150  
151          if (sign < 0)
152              y = -y;
153  
154          return (y);
155      }

157      XSF_HOST_DEVICE inline double cosdg(double x) {
158          double y, z, zz;
159          int j, sign;
160  
161          /* make argument positive */
162          sign = 1;
163          if (x < 0)
164              x = -x;
165  
166          if (x > detail::sindg_lossth) {
167              set_error("cosdg", SF_ERROR_NO_RESULT, NULL);
168              return (0.0);
169          }
170  
171          y = std::floor(x / 45.0);
172          z = std::ldexp(y, -4);
173          z = std::floor(z);        /* integer part of y/8 */
174          z = y - std::ldexp(z, 4); /* y - 16 * (y/16) */
175  
176          /* integer and fractional part modulo one octant */
177          j = z;
178          if (j & 1) { /* map zeros to origin */
179              j += 1;
180              y += 1.0;
181          }
182          j = j & 07;
183          if (j > 3) {
184              j -= 4;
185              sign = -sign;
186          }
187  
188          if (j > 1)
189              sign = -sign;
190  
191          z = x - y * 45.0;   /* x mod 45 degrees */
192          z *= detail::PI180; /* multiply by pi/180 to convert to radians */
193  
194          zz = z * z;
195  
196          if ((j == 1) || (j == 2)) {
197              y = z + z * (zz * polevl(zz, detail::sincof, 5));
198          } else {
199              y = 1.0 - zz * polevl(zz, detail::coscof, 6);
200          }
201  
202          if (sign < 0)
203              y = -y;
204  
205          return (y);
206      }

208      /* Degrees, minutes, seconds to radians: */
209  
210      /* 1 arc second, in radians = 4.848136811095359935899141023579479759563533023727e-6 */
211  
212      namespace detail {
213          constexpr double sindg_P64800 = 4.848136811095359935899141023579479759563533023727e-6;
214      }
215  
216      XSF_HOST_DEVICE inline double radian(double d, double m, double s) {
217          return (((d * 60.0 + m) * 60.0 + s) * detail::sindg_P64800);
218      }
