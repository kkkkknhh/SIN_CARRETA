45   class Beam:
46       """
47       A Beam is a structural element that is capable of withstanding load
48       primarily by resisting against bending. Beams are characterized by
49       their cross sectional profile(Second moment of area), their length
50       and their material.
51   
52       .. note::
53          A consistent sign convention must be used while solving a beam
54          bending problem; the results will
55          automatically follow the chosen sign convention. However, the
56          chosen sign convention must respect the rule that, on the positive
57          side of beam's axis (in respect to current section), a loading force
58          giving positive shear yields a negative moment, as below (the
59          curved arrow shows the positive moment and rotation):
60   
61       .. image:: allowed-sign-conventions.png
62   
63       Examples
64       ========
65       There is a beam of length 4 meters. A constant distributed load of 6 N/m
66       is applied from half of the beam till the end. There are two simple supports
67       below the beam, one at the starting point and another at the ending point
68       of the beam. The deflection of the beam at the end is restricted.
69   
70       Using the sign convention of downwards forces being positive.
71   
72       >>> from sympy.physics.continuum_mechanics.beam import Beam
73       >>> from sympy import symbols, Piecewise
74       >>> E, I = symbols('E, I')
75       >>> R1, R2 = symbols('R1, R2')
76       >>> b = Beam(4, E, I)
77       >>> b.apply_load(R1, 0, -1)
78       >>> b.apply_load(6, 2, 0)
79       >>> b.apply_load(R2, 4, -1)
80       >>> b.bc_deflection = [(0, 0), (4, 0)]
81       >>> b.boundary_conditions
82       {'bending_moment': [], 'deflection': [(0, 0), (4, 0)], 'shear_force': [], 'slope': []}
83       >>> b.load
84       R1*SingularityFunction(x, 0, -1) + R2*SingularityFunction(x, 4, -1) + 6*SingularityFunction(x, 2, 0)
85       >>> b.solve_for_reaction_loads(R1, R2)
86       >>> b.load
87       -3*SingularityFunction(x, 0, -1) + 6*SingularityFunction(x, 2, 0) - 9*SingularityFunction(x, 4, -1)
88       >>> b.shear_force()
89       3*SingularityFunction(x, 0, 0) - 6*SingularityFunction(x, 2, 1) + 9*SingularityFunction(x, 4, 0)
90       >>> b.bending_moment()
91       3*SingularityFunction(x, 0, 1) - 3*SingularityFunction(x, 2, 2) + 9*SingularityFunction(x, 4, 1)
92       >>> b.slope()
93       (-3*SingularityFunction(x, 0, 2)/2 + SingularityFunction(x, 2, 3) - 9*SingularityFunction(x, 4, 2)/2 + 7)/(E*I)
94       >>> b.deflection()
95       (7*x - SingularityFunction(x, 0, 3)/2 + SingularityFunction(x, 2, 4)/4 - 3*SingularityFunction(x, 4, 3)/2)/(E*I)
96       >>> b.deflection().rewrite(Piecewise)
97       (7*x - Piecewise((x**3, x >= 0), (0, True))/2
98            - 3*Piecewise(((x - 4)**3, x >= 4), (0, True))/2
99            + Piecewise(((x - 2)**4, x >= 2), (0, True))/4)/(E*I)
100  
101      Calculate the support reactions for a fully symbolic beam of length L.
102      There are two simple supports below the beam, one at the starting point
103      and another at the ending point of the beam. The deflection of the beam
104      at the end is restricted. The beam is loaded with:
105  
106      * a downward point load P1 applied at L/4
107      * an upward point load P2 applied at L/8
108      * a counterclockwise moment M1 applied at L/2
109      * a clockwise moment M2 applied at 3*L/4
110      * a distributed constant load q1, applied downward, starting from L/2
111        up to 3*L/4
112      * a distributed constant load q2, applied upward, starting from 3*L/4
113        up to L
114  
115      No assumptions are needed for symbolic loads. However, defining a positive
116      length will help the algorithm to compute the solution.
117  
118      >>> E, I = symbols('E, I')
119      >>> L = symbols("L", positive=True)
120      >>> P1, P2, M1, M2, q1, q2 = symbols("P1, P2, M1, M2, q1, q2")
121      >>> R1, R2 = symbols('R1, R2')
122      >>> b = Beam(L, E, I)
123      >>> b.apply_load(R1, 0, -1)
124      >>> b.apply_load(R2, L, -1)
125      >>> b.apply_load(P1, L/4, -1)
126      >>> b.apply_load(-P2, L/8, -1)
127      >>> b.apply_load(M1, L/2, -2)
128      >>> b.apply_load(-M2, 3*L/4, -2)
129      >>> b.apply_load(q1, L/2, 0, 3*L/4)
130      >>> b.apply_load(-q2, 3*L/4, 0, L)
131      >>> b.bc_deflection = [(0, 0), (L, 0)]
132      >>> b.solve_for_reaction_loads(R1, R2)
133      >>> print(b.reaction_loads[R1])
134      (-3*L**2*q1 + L**2*q2 - 24*L*P1 + 28*L*P2 - 32*M1 + 32*M2)/(32*L)
135      >>> print(b.reaction_loads[R2])
136      (-5*L**2*q1 + 7*L**2*q2 - 8*L*P1 + 4*L*P2 + 32*M1 - 32*M2)/(32*L)
137      """

139      def __init__(self, length, elastic_modulus, second_moment, area=Symbol('A'), variable=Symbol('x'), base_char='C', ild_variable=Symbol('a')):
140          """Initializes the class.
141  
142          Parameters
143          ==========
144  
145          length : Sympifyable
146              A Symbol or value representing the Beam's length.
147  
148          elastic_modulus : Sympifyable
149              A SymPy expression representing the Beam's Modulus of Elasticity.
150              It is a measure of the stiffness of the Beam material. It can
151              also be a continuous function of position along the beam.
152  
153          second_moment : Sympifyable or Geometry object
154              Describes the cross-section of the beam via a SymPy expression
155              representing the Beam's second moment of area. It is a geometrical
156              property of an area which reflects how its points are distributed
157              with respect to its neutral axis. It can also be a continuous
158              function of position along the beam. Alternatively ``second_moment``
159              can be a shape object such as a ``Polygon`` from the geometry module
160              representing the shape of the cross-section of the beam. In such cases,
161              it is assumed that the x-axis of the shape object is aligned with the
162              bending axis of the beam. The second moment of area will be computed
163              from the shape object internally.
164  
165          area : Symbol/float
166              Represents the cross-section area of beam
167  
168          variable : Symbol, optional
169              A Symbol object that will be used as the variable along the beam
170              while representing the load, shear, moment, slope and deflection
171              curve. By default, it is set to ``Symbol('x')``.
172  
173          base_char : String, optional
174              A String that will be used as base character to generate sequential
175              symbols for integration constants in cases where boundary conditions
176              are not sufficient to solve them.
177  
178          ild_variable : Symbol, optional
179              A Symbol object that will be used as the variable specifying the
180              location of the moving load in ILD calculations. By default, it
181              is set to ``Symbol('a')``.
182          """
183          self.length = length
184          self.elastic_modulus = elastic_modulus
185          if isinstance(second_moment, GeometryEntity):
186              self.cross_section = second_moment
187          else:
188              self.cross_section = None
189              self.second_moment = second_moment
190          self.variable = variable
191          self.ild_variable = ild_variable
192          self._base_char = base_char
193          self._boundary_conditions = {'deflection': [], 'slope': [], 'bending_moment': [], 'shear_force': []}
194          self._load = 0
195          self.area = area
196          self._applied_supports = []
197          self._applied_rotation_hinges = []
198          self._applied_sliding_hinges = []
199          self._rotation_hinge_symbols = []
200          self._sliding_hinge_symbols = []
201          self._support_as_loads = []
202          self._applied_loads = []
203          self._reaction_loads = {}
204          self._ild_reactions = {}
205          self._ild_shear = 0
206          self._ild_moment = 0
207          # _original_load is a copy of _load equations with unsubstituted reaction
208          # forces. It is used for calculating reaction forces in case of I.L.D.
209          self._original_load = 0
210          self._joined_beam = False

419      def join(self, beam, via="fixed"):
420          """
421          This method joins two beams to make a new composite beam system.
422          Passed Beam class instance is attached to the right end of calling
423          object. This method can be used to form beams having Discontinuous
424          values of Elastic modulus or Second moment.
425  
426          Parameters
427          ==========
428          beam : Beam class object
429              The Beam object which would be connected to the right of calling
430              object.
431          via : String
432              States the way two Beam object would get connected
433              - For axially fixed Beams, via="fixed"
434              - For Beams connected via rotation hinge, via="hinge"
435  
436          Examples
437          ========
438          There is a cantilever beam of length 4 meters. For first 2 meters
439          its moment of inertia is `1.5*I` and `I` for the other end.
440          A pointload of magnitude 4 N is applied from the top at its free end.
441  
442          >>> from sympy.physics.continuum_mechanics.beam import Beam
443          >>> from sympy import symbols
444          >>> E, I = symbols('E, I')
445          >>> R1, R2 = symbols('R1, R2')
446          >>> b1 = Beam(2, E, 1.5*I)
447          >>> b2 = Beam(2, E, I)
448          >>> b = b1.join(b2, "fixed")
449          >>> b.apply_load(20, 4, -1)
450          >>> b.apply_load(R1, 0, -1)
451          >>> b.apply_load(R2, 0, -2)
452          >>> b.bc_slope = [(0, 0)]
453          >>> b.bc_deflection = [(0, 0)]
454          >>> b.solve_for_reaction_loads(R1, R2)
455          >>> b.load
456          80*SingularityFunction(x, 0, -2) - 20*SingularityFunction(x, 0, -1) + 20*SingularityFunction(x, 4, -1)
457          >>> b.slope()
458          (-((-80*SingularityFunction(x, 0, 1) + 10*SingularityFunction(x, 0, 2) - 10*SingularityFunction(x, 4, 2))/I + 120/I)/E + 80.0/(E*I))*SingularityFunction(x, 2, 0)
459          - 0.666666666666667*(-80*SingularityFunction(x, 0, 1) + 10*SingularityFunction(x, 0, 2) - 10*SingularityFunction(x, 4, 2))*SingularityFunction(x, 0, 0)/(E*I)
460          + 0.666666666666667*(-80*SingularityFunction(x, 0, 1) + 10*SingularityFunction(x, 0, 2) - 10*SingularityFunction(x, 4, 2))*SingularityFunction(x, 2, 0)/(E*I)
461          """
462          x = self.variable
463          E = self.elastic_modulus
464          new_length = self.length + beam.length
465          if self.elastic_modulus != beam.elastic_modulus:
466              raise NotImplementedError('Joining beams with different Elastic modulus is not implemented.')
467  
468          if self.second_moment != beam.second_moment:
469              new_second_moment = Piecewise((self.second_moment, x<=self.length),
470                                      (beam.second_moment, x<=new_length))
471          else:
472              new_second_moment = self.second_moment
473  
474          if via == "fixed":
475              new_beam = Beam(new_length, E, new_second_moment, x)
476              new_beam._joined_beam = True
477              return new_beam
478  
479          if via == "hinge":
480              new_beam = Beam(new_length, E, new_second_moment, x)
481              new_beam._joined_beam = True
482              new_beam.apply_rotation_hinge(self.length)
483              return new_beam

485      def apply_support(self, loc, type="fixed"):
486          """
487          This method applies support to a particular beam object and returns
488          the symbol of the unknown reaction load(s).
489  
490          Parameters
491          ==========
492          loc : Sympifyable
493              Location of point at which support is applied.
494          type : String
495              Determines type of Beam support applied. To apply support structure
496              with
497              - zero degree of freedom, type = "fixed"
498              - one degree of freedom, type = "pin"
499              - two degrees of freedom, type = "roller"
500  
501          Returns
502          =======
503          Symbol or tuple of Symbol
504              The unknown reaction load as a symbol.
505              - Symbol(reaction_force) if type = "pin" or "roller"
506              - Symbol(reaction_force), Symbol(reaction_moment) if type = "fixed"
507  
508          Examples
509          ========
510          There is a beam of length 20 meters. A moment of magnitude 100 Nm is
511          applied in the clockwise direction at the end of the beam. A pointload
512          of magnitude 8 N is applied from the top of the beam at a distance of 10 meters.
513          There is one fixed support at the start of the beam and a roller at the end.
514  
515          Using the sign convention of upward forces and clockwise moment
516          being positive.
517  
518          >>> from sympy.physics.continuum_mechanics.beam import Beam
519          >>> from sympy import symbols
520          >>> E, I = symbols('E, I')
521          >>> b = Beam(20, E, I)
522          >>> p0, m0 = b.apply_support(0, 'fixed')
523          >>> p1 = b.apply_support(20, 'roller')
524          >>> b.apply_load(-8, 10, -1)
525          >>> b.apply_load(100, 20, -2)
526          >>> b.solve_for_reaction_loads(p0, m0, p1)
527          >>> b.reaction_loads
528          {M_0: 20, R_0: -2, R_20: 10}
529          >>> b.reaction_loads[p0]
530          -2
531          >>> b.load
532          20*SingularityFunction(x, 0, -2) - 2*SingularityFunction(x, 0, -1)
533          - 8*SingularityFunction(x, 10, -1) + 100*SingularityFunction(x, 20, -2)
534          + 10*SingularityFunction(x, 20, -1)
535          """
536          loc = sympify(loc)
537  
538          self._applied_supports.append((loc, type))
539          if type in ("pin", "roller"):
540              reaction_load = Symbol('R_'+str(loc))
541              self.apply_load(reaction_load, loc, -1)
542              self.bc_deflection.append((loc, 0))
543          else:
544              reaction_load = Symbol('R_'+str(loc))
545              reaction_moment = Symbol('M_'+str(loc))
546              self.apply_load(reaction_load, loc, -1)
547              self.apply_load(reaction_moment, loc, -2)
548              self.bc_deflection.append((loc, 0))
549              self.bc_slope.append((loc, 0))
550              self._support_as_loads.append((reaction_moment, loc, -2, None))
551  
552          self._support_as_loads.append((reaction_load, loc, -1, None))
553  
554          if type in ("pin", "roller"):
555              return reaction_load
556          else:
557              return reaction_load, reaction_moment

559      def _get_I(self, loc):
560          """
561          Helper function that returns the Second moment (I) at a location in the beam.
562          """
563          I = self.second_moment
564          if not isinstance(I, Piecewise):
565              return I
566          else:
567              for i in range(len(I.args)):
568                  if loc <= I.args[i][1].args[1]:
569                      return I.args[i][0]

571      def apply_rotation_hinge(self, loc):
572          """
573          This method applies a rotation hinge at a single location on the beam.
574  
575          Parameters
576          ----------
577          loc : Sympifyable
578              Location of point at which hinge is applied.
579  
580          Returns
581          =======
582          Symbol
583              The unknown rotation jump multiplied by the elastic modulus and second moment as a symbol.
584  
585          Examples
586          ========
587          There is a beam of length 15 meters. Pin supports are placed at distances
588          of 0 and 10 meters. There is a fixed support at the end. There are two rotation hinges
589          in the structure, one at 5 meters and one at 10 meters. A pointload of magnitude
590          10 kN is applied on the hinge at 5 meters. A distributed load of 5 kN works on
591          the structure from 10 meters to the end.
592  
593          Using the sign convention of upward forces and clockwise moment
594          being positive.
595  
596          >>> from sympy.physics.continuum_mechanics.beam import Beam
597          >>> from sympy import Symbol
598          >>> E = Symbol('E')
599          >>> I = Symbol('I')
600          >>> b = Beam(15, E, I)
601          >>> r0 = b.apply_support(0, type='pin')
602          >>> r10 = b.apply_support(10, type='pin')

686      def apply_load(self, value, start, order, end=None):
687          """
688          This method adds up the loads given to a particular beam object.
689  
690          Parameters
691          ==========
692          value : Sympifyable
693              The value inserted should have the units [Force/(Distance**(n+1)]
694              where n is the order of applied load.
695              Units for applied loads:
696  
697                 - For moments, unit = kN*m
698                 - For point loads, unit = kN
699                 - For constant distributed load, unit = kN/m
700                 - For ramp loads, unit = kN/m/m
701                 - For parabolic ramp loads, unit = kN/m/m/m
702                 - ... so on.
703  
704          start : Sympifyable
705              The starting point of the applied load. For point moments and
706              point forces this is the location of application.
707          order : Integer
708              The order of the applied load.
709  
710                 - For moments, order = -2
711                 - For point loads, order =-1
712                 - For constant distributed load, order = 0
713                 - For ramp loads, order = 1
714                 - For parabolic ramp loads, order = 2
715                 - ... so on.
716  
717          end : Sympifyable, optional
718              An optional argument that can be used if the load has an end point
719              within the length of the beam.
720  
721          Examples
722          ========
723          There is a beam of length 4 meters. A moment of magnitude 3 Nm is
724          applied in the clockwise direction at the starting point of the beam.
725          A point load of magnitude 4 N is applied from the top of the beam at
726          2 meters from the starting point and a parabolic ramp load of magnitude
727          2 N/m is applied below the beam starting from 2 meters to 3 meters
728          away from the starting point of the beam.
729  
730          >>> from sympy.physics.continuum_mechanics.beam import Beam
731          >>> from sympy import symbols
732          >>> E, I = symbols('E, I')
733          >>> b = Beam(4, E, I)
734          >>> b.apply_load(-3, 0, -2)
735          >>> b.apply_load(4, 2, -1)
736          >>> b.apply_load(-2, 2, 2, end=3)
737          >>> b.load
738          -3*SingularityFunction(x, 0, -2) + 4*SingularityFunction(x, 2, -1) - 2*SingularityFunction(x, 2, 2) + 2*SingularityFunction(x, 3, 0) + 4*SingularityFunction(x, 3, 1) + 2*SingularityFunction(x, 3, 2)
739  
740          """
741          x = self.variable
742          value = sympify(value)
743          start = sympify(start)
744          order = sympify(order)
745  
746          self._applied_loads.append((value, start, order, end))
747          self._load += value*SingularityFunction(x, start, order)
748          self._original_load += value*SingularityFunction(x, start, order)
749  
750          if end:
751              # load has an end point within the length of the beam.
752              self._handle_end(x, value, start, order, end, type="apply")

754      def remove_load(self, value, start, order, end=None):
755          """
756          This method removes a particular load present on the beam object.
757          Returns a ValueError if the load passed as an argument is not
758          present on the beam.
759  
760          Parameters
761          ==========
762          value : Sympifyable
763              The magnitude of an applied load.
764          start : Sympifyable
765              The starting point of the applied load. For point moments and
766              point forces this is the location of application.
767          order : Integer
768              The order of the applied load.
769              - For moments, order= -2
770              - For point loads, order=-1
771              - For constant distributed load, order=0
772              - For ramp loads, order=1
773              - For parabolic ramp loads, order=2
774              - ... so on.
775          end : Sympifyable, optional
776              An optional argument that can be used if the load has an end point
777              within the length of the beam.
778  
779          Examples
780          ========
781          There is a beam of length 4 meters. A moment of magnitude 3 Nm is
782          applied in the clockwise direction at the starting point of the beam.
783          A pointload of magnitude 4 N is applied from the top of the beam at
784          2 meters from the starting point and a parabolic ramp load of magnitude
785          2 N/m is applied below the beam starting from 2 meters to 3 meters
786          away from the starting point of the beam.
787  
788          >>> from sympy.physics.continuum_mechanics.beam import Beam
789          >>> from sympy import symbols
790          >>> E, I = symbols('E, I')
791          >>> b = Beam(4, E, I)
792          >>> b.apply_load(-3, 0, -2)
793          >>> b.apply_load(4, 2, -1)
794          >>> b.apply_load(-2, 2, 2, end=3)
795          >>> b.load
796          -3*SingularityFunction(x, 0, -2) + 4*SingularityFunction(x, 2, -1) - 2*SingularityFunction(x, 2, 2) + 2*SingularityFunction(x, 3, 0) + 4*SingularityFunction(x, 3, 1) + 2*SingularityFunction(x, 3, 2)
797          >>> b.remove_load(-2, 2, 2, end = 3)
798          >>> b.load
799          -3*SingularityFunction(x, 0, -2) + 4*SingularityFunction(x, 2, -1)
800          """
801          x = self.variable
802          value = sympify(value)
803          start = sympify(start)
804          order = sympify(order)
805  
806          if (value, start, order, end) in self._applied_loads:
807              self._load -= value*SingularityFunction(x, start, order)
808              self._original_load -= value*SingularityFunction(x, start, order)
809              self._applied_loads.remove((value, start, order, end))
810          else:
811              msg = "No such load distribution exists on the beam object."
812              raise ValueError(msg)
813  
814          if end:
815              # load has an end point within the length of the beam.
816              self._handle_end(x, value, start, order, end, type="remove")

818      def _handle_end(self, x, value, start, order, end, type):
819          """
820          This functions handles the optional `end` value in the
821          `apply_load` and `remove_load` functions. When the value
822          of end is not NULL, this function will be executed.
823          """
824          if order.is_negative:
825              msg = ("If 'end' is provided the 'order' of the load cannot "
826                      "be negative, i.e. 'end' is only valid for distributed "
827                      "loads.")
828              raise ValueError(msg)
829          # NOTE : A Taylor series can be used to define the summation of
830          # singularity functions that subtract from the load past the end
831          # point such that it evaluates to zero past 'end'.
832          f = value*x**order
833  
834          if type == "apply":
835              # iterating for "apply_load" method
836              for i in range(0, order + 1):
837                  self._load -= (f.diff(x, i).subs(x, end - start) *
838                                  SingularityFunction(x, end, i)/factorial(i))
839                  self._original_load -= (f.diff(x, i).subs(x, end - start) *
840                                  SingularityFunction(x, end, i)/factorial(i))
841          elif type == "remove":
842              # iterating for "remove_load" method
843              for i in range(0, order + 1):
844                  self._load += (f.diff(x, i).subs(x, end - start) *
845                                  SingularityFunction(x, end, i)/factorial(i))
846                  self._original_load += (f.diff(x, i).subs(x, end - start) *
847                                  SingularityFunction(x, end, i)/factorial(i))

905      def solve_for_reaction_loads(self, *reactions):
906          """
907          Solves for the reaction forces.
908  
909          Examples
910          ========
911          There is a beam of length 30 meters. A moment of magnitude 120 Nm is
912          applied in the clockwise direction at the end of the beam. A pointload
913          of magnitude 8 N is applied from the top of the beam at the starting
914          point. There are two simple supports below the beam. One at the end
915          and another one at a distance of 10 meters from the start. The
916          deflection is restricted at both the supports.
917  
918          Using the sign convention of upward forces and clockwise moment
919          being positive.
920  
921          >>> from sympy.physics.continuum_mechanics.beam import Beam
922          >>> from sympy import symbols
923          >>> E, I = symbols('E, I')
924          >>> R1, R2 = symbols('R1, R2')
925          >>> b = Beam(30, E, I)
926          >>> b.apply_load(-8, 0, -1)
927          >>> b.apply_load(R1, 10, -1)  # Reaction force at x = 10
928          >>> b.apply_load(R2, 30, -1)  # Reaction force at x = 30
929          >>> b.apply_load(120, 30, -2)
930          >>> b.bc_deflection = [(10, 0), (30, 0)]
931          >>> b.load
932          R1*SingularityFunction(x, 10, -1) + R2*SingularityFunction(x, 30, -1)
933              - 8*SingularityFunction(x, 0, -1) + 120*SingularityFunction(x, 30, -2)
934          >>> b.solve_for_reaction_loads(R1, R2)
935          >>> b.reaction_loads
936          {R1: 6, R2: 2}
937          >>> b.load
938          -8*SingularityFunction(x, 0, -1) + 6*SingularityFunction(x, 10, -1)
939              + 120*SingularityFunction(x, 30, -2) + 2*SingularityFunction(x, 30, -1)
940          """
941  
942          x = self.variable
943          l = self.length
944          C3 = Symbol('C3')
945          C4 = Symbol('C4')
946          rotation_jumps = tuple(self._rotation_hinge_symbols)
947          deflection_jumps = tuple(self._sliding_hinge_symbols)
948  
949          shear_curve = limit(self.shear_force(), x, l)
950          moment_curve = limit(self.bending_moment(), x, l)
951  
952          shear_force_eqs = []
953          bending_moment_eqs = []
954          slope_eqs = []
955          deflection_eqs = []
956  
957          for position, value in self._boundary_conditions['shear_force']:
958              eqs = self.shear_force().subs(x, position) - value
959              new_eqs = sum(arg for arg in eqs.args if not any(num.is_infinite for num in arg.args))
960              shear_force_eqs.append(new_eqs)
961  
962          for position, value in self._boundary_conditions['bending_moment']:
963              eqs = self.bending_moment().subs(x, position) - value
964              new_eqs = sum(arg for arg in eqs.args if not any(num.is_infinite for num in arg.args))
965              bending_moment_eqs.append(new_eqs)
966  
967          slope_curve = integrate(self.bending_moment(), x) + C3
968          for position, value in self._boundary_conditions['slope']:
969              eqs = slope_curve.subs(x, position) - value
970              slope_eqs.append(eqs)
971  
972          deflection_curve = integrate(slope_curve, x) + C4
973          for position, value in self._boundary_conditions['deflection']:
974              eqs = deflection_curve.subs(x, position) - value
975              deflection_eqs.append(eqs)
976  
977          solution = list((linsolve([shear_curve, moment_curve] + shear_force_eqs + bending_moment_eqs + slope_eqs
978                              + deflection_eqs, (C3, C4) + reactions + rotation_jumps + deflection_jumps).args)[0])
979          reaction_index = 2+len(reactions)
980          rotation_index = reaction_index + len(rotation_jumps)
981          reaction_solution = solution[2:reaction_index]
982          rotation_solution = solution[reaction_index:rotation_index]
983          deflection_solution = solution[rotation_index:]
984  
985          self._reaction_loads = dict(zip(reactions, reaction_solution))
986          self._rotation_jumps = dict(zip(rotation_jumps, rotation_solution))
987          self._deflection_jumps = dict(zip(deflection_jumps, deflection_solution))
988          self._load = self._load.subs(self._reaction_loads)
989          self._load = self._load.subs(self._rotation_jumps)
990          self._load = self._load.subs(self._deflection_jumps)

1026     def max_shear_force(self):
1027         """Returns maximum Shear force and its coordinate
1028         in the Beam object."""
1029         shear_curve = self.shear_force()
1030         x = self.variable
1031 
1032         terms = shear_curve.args
1033         singularity = []        # Points at which shear function changes
1034         for term in terms:
1035             if isinstance(term, Mul):
1036                 term = term.args[-1]    # SingularityFunction in the term
1037             singularity.append(term.args[1])
1038         singularity = list(set(singularity))
1039         singularity.sort()
1040 
1041         intervals = []    # List of Intervals with discrete value of shear force
1042         shear_values = []   # List of values of shear force in each interval
1043         for i, s in enumerate(singularity):
1044             if s == 0:
1045                 continue
1046             try:
1047                 shear_slope = Piecewise((float("nan"), x<=singularity[i-1]),(self._load.rewrite(Piecewise), x<s), (float("nan"), True))
1048                 points = solve(shear_slope, x)
1049                 val = []
1050                 for point in points:
1051                     val.append(abs(shear_curve.subs(x, point)))
1052                 points.extend([singularity[i-1], s])
1053                 val += [abs(limit(shear_curve, x, singularity[i-1], '+')), abs(limit(shear_curve, x, s, '-'))]
1054                 max_shear = max(val)
1055                 shear_values.append(max_shear)
1056                 intervals.append(points[val.index(max_shear)])
1057             # If shear force in a particular Interval has zero or constant
1058             # slope, then above block gives NotImplementedError as
1059             # solve can't represent Interval solutions.
1060             except NotImplementedError:
1061                 initial_shear = limit(shear_curve, x, singularity[i-1], '+')
1062                 final_shear = limit(shear_curve, x, s, '-')
1063                 # If shear_curve has a constant slope(it is a line).
1064                 if shear_curve.subs(x, (singularity[i-1] + s)/2) == (initial_shear + final_shear)/2 and initial_shear != final_shear:
1065                     shear_values.extend([initial_shear, final_shear])
1066                     intervals.extend([singularity[i-1], s])
1067                 else:    # shear_curve has same value in whole Interval
1068                     shear_values.append(final_shear)
1069                     intervals.append(Interval(singularity[i-1], s))
1070 
1071         shear_values = list(map(abs, shear_values))
1072         maximum_shear = max(shear_values)
1073         point = intervals[shear_values.index(maximum_shear)]
1074         return (point, maximum_shear)

1132                     (float("nan"), x <= singularity[i - 1]),
1133                     (self.shear_force().rewrite(Piecewise), x < s),
1134                     (float("nan"), True))
1135                 points = solve(moment_slope, x)
1136                 val = []
1137                 for point in points:
1138                     val.append(abs(bending_curve.subs(x, point)))
1139                 points.extend([singularity[i-1], s])
1140                 val += [abs(limit(bending_curve, x, singularity[i-1], '+')), abs(limit(bending_curve, x, s, '-'))]
1141                 max_moment = max(val)
1142                 moment_values.append(max_moment)
1143                 intervals.append(points[val.index(max_moment)])
1144 
1145             # If bending moment in a particular Interval has zero or constant
1146             # slope, then above block gives NotImplementedError as solve
1147             # can't represent Interval solutions.
1148             except NotImplementedError:
1149                 initial_moment = limit(bending_curve, x, singularity[i-1], '+')
1150                 final_moment = limit(bending_curve, x, s, '-')
1151                 # If bending_curve has a constant slope(it is a line).
1152                 if bending_curve.subs(x, (singularity[i-1] + s)/2) == (initial_moment + final_moment)/2 and initial_moment != final_moment:
1153                     moment_values.extend([initial_moment, final_moment])
1154                     intervals.extend([singularity[i-1], s])
1155                 else:    # bending_curve has same value in whole Interval
1156                     moment_values.append(final_moment)
1157                     intervals.append(Interval(singularity[i-1], s))
1158 
1159         moment_values = list(map(abs, moment_values))
1160         maximum_moment = max(moment_values)
1161         point = intervals[moment_values.index(maximum_moment)]
1162         return (point, maximum_moment)

1214     def slope(self):
1215         """
1216         Returns a Singularity Function expression which represents
1217         the slope the elastic curve of the Beam object.
1218 
1219         Examples
1220         ========
1221         There is a beam of length 30 meters. A moment of magnitude 120 Nm is
1222         applied in the clockwise direction at the end of the beam. A pointload
1223         of magnitude 8 N is applied from the top of the beam at the starting
1224         point. There are two simple supports below the beam. One at the end
1225         and another one at a distance of 10 meters from the start. The
1226         deflection is restricted at both the supports.
1227 
1228         Using the sign convention of upward forces and clockwise moment
1229         being positive.
1230 
1231         >>> from sympy.physics.continuum_mechanics.beam import Beam
1232         >>> from sympy import symbols
1233         >>> E, I = symbols('E, I')
1234         >>> R1, R2 = symbols('R1, R2')
1235         >>> b = Beam(30, E, I)
1236         >>> b.apply_load(-8, 0, -1)
1237         >>> b.apply_load(R1, 10, -1)
1238         >>> b.apply_load(R2, 30, -1)
1239         >>> b.apply_load(120, 30, -2)
1240         >>> b.bc_deflection = [(10, 0), (30, 0)]
1241         >>> b.solve_for_reaction_loads(R1, R2)
1242         >>> b.slope()
1243         (-4*SingularityFunction(x, 0, 2) + 3*SingularityFunction(x, 10, 2)
1244             + 120*SingularityFunction(x, 30, 1) + SingularityFunction(x, 30, 2) + 4000/3)/(E*I)
1245         """
1246         x = self.variable
1247         E = self.elastic_modulus
1248         I = self.second_moment
1249 
1250         if not self._boundary_conditions['slope']:
1251             return diff(self.deflection(), x)
1252         if isinstance(I, Piecewise) and self._joined_beam:
1253             args = I.args
1254             slope = 0
1255             prev_slope = 0
1256             prev_end = 0
1257             for i in range(len(args)):
1258                 if i != 0:
1259                     prev_end = args[i-1][1].args[1]
1260                 slope_value = -S.One/E*integrate(self.bending_moment()/args[i][0], (x, prev_end, x))
1261                 if i != len(args) - 1:
1262                     slope += (prev_slope + slope_value)*SingularityFunction(x, prev_end, 0) - \
1263                         (prev_slope + slope_value)*SingularityFunction(x, args[i][1].args[1], 0)
1264                 else:
1265                     slope += (prev_slope + slope_value)*SingularityFunction(x, prev_end, 0)
1266                 prev_slope = slope_value.subs(x, args[i][1].args[1])
1267             return slope
1268 
1269         C3 = Symbol('C3')
1270         slope_curve = -integrate(S.One/(E*I)*self.bending_moment(), x) + C3
1271 
1272         bc_eqs = []
1273         for position, value in self._boundary_conditions['slope']:
1274             eqs = slope_curve.subs(x, position) - value
1275             bc_eqs.append(eqs)
1276         constants = list(linsolve(bc_eqs, C3))
1277         slope_curve = slope_curve.subs({C3: constants[0][0]})
1278         return slope_curve

1280     def deflection(self):
1281         """
1282         Returns a Singularity Function expression which represents
1283         the elastic curve or deflection of the Beam object.
1284 
1285         Examples
1286         ========
1287         There is a beam of length 30 meters. A moment of magnitude 120 Nm is
1288         applied in the clockwise direction at the end of the beam. A pointload
1289         of magnitude 8 N is applied from the top of the beam at the starting
1290         point. There are two simple supports below the beam. One at the end
1291         and another one at a distance of 10 meters from the start. The

1310             + 60*SingularityFunction(x, 30, 2) + SingularityFunction(x, 30, 3)/3 - 12000)/(E*I)

1313         E = self.elastic_modulus
1314         I = self.second_moment
1315         if not self._boundary_conditions['deflection'] and not self._boundary_conditions['slope']:
1316             if isinstance(I, Piecewise) and self._joined_beam:
1317                 args = I.args
1318                 prev_slope = 0
1319                 prev_def = 0
1320                 prev_end = 0
1321                 deflection = 0
1322                 for i in range(len(args)):
1323                     if i != 0:
1324                         prev_end = args[i-1][1].args[1]
1325                     slope_value = -S.One/E*integrate(self.bending_moment()/args[i][0], (x, prev_end, x))
1326                     recent_segment_slope = prev_slope + slope_value
1327                     deflection_value = integrate(recent_segment_slope, (x, prev_end, x))
1328                     if i != len(args) - 1:
1329                         deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0) \
1330                             - (prev_def + deflection_value)*SingularityFunction(x, args[i][1].args[1], 0)
1331                     else:
1332                         deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0)

1334                     prev_def = deflection_value.subs(x, args[i][1].args[1])

1336             base_char = self._base_char
1337             constants = symbols(base_char + '3:5')
1338             return S.One/(E*I)*integrate(-integrate(self.bending_moment(), x), x) + constants[0]*x + constants[1]
1339         elif not self._boundary_conditions['deflection']:
1340             base_char = self._base_char
1341             constant = symbols(base_char + '4')
1342             return integrate(self.slope(), x) + constant
1343         elif not self._boundary_conditions['slope'] and self._boundary_conditions['deflection']:
1344             if isinstance(I, Piecewise) and self._joined_beam:
1345                 args = I.args
1346                 prev_slope = 0
1347                 prev_def = 0
1348                 prev_end = 0
1349                 deflection = 0
1350                 for i in range(len(args)):
1351                     if i != 0:
1352                         prev_end = args[i-1][1].args[1]
1353                     slope_value = -S.One/E*integrate(self.bending_moment()/args[i][0], (x, prev_end, x))
1354                     recent_segment_slope = prev_slope + slope_value
1355                     deflection_value = integrate(recent_segment_slope, (x, prev_end, x))
1356                     if i != len(args) - 1:
1357                         deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0) \
1358                             - (prev_def + deflection_value)*SingularityFunction(x, args[i][1].args[1], 0)
1359                     else:
1360                         deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0)
1361                     prev_slope = slope_value.subs(x, args[i][1].args[1])
1362                     prev_def = deflection_value.subs(x, args[i][1].args[1])
1363                 return deflection
1364             base_char = self._base_char
1365             C3, C4 = symbols(base_char + '3:5')    # Integration constants
1366             slope_curve = -integrate(self.bending_moment(), x) + C3
1367             deflection_curve = integrate(slope_curve, x) + C4
1368             bc_eqs = []
1369             for position, value in self._boundary_conditions['deflection']:
1370                 eqs = deflection_curve.subs(x, position) - value
1371                 bc_eqs.append(eqs)
1372             constants = list(linsolve(bc_eqs, (C3, C4)))
1373             deflection_curve = deflection_curve.subs({C3: constants[0][0], C4: constants[0][1]})
1374             return S.One/(E*I)*deflection_curve
1375 
1376         if isinstance(I, Piecewise) and self._joined_beam:
1377             args = I.args
1378             prev_slope = 0
1379             prev_def = 0
1380             prev_end = 0
1381             deflection = 0
1382             for i in range(len(args)):
1383                 if i != 0:
1384                     prev_end = args[i-1][1].args[1]
1385                 slope_value = S.One/E*integrate(self.bending_moment()/args[i][0], (x, prev_end, x))
1386                 recent_segment_slope = prev_slope + slope_value
1387                 deflection_value = integrate(recent_segment_slope, (x, prev_end, x))
1388                 if i != len(args) - 1:
1389                     deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0) \
1390                         - (prev_def + deflection_value)*SingularityFunction(x, args[i][1].args[1], 0)
1391                 else:
1392                     deflection += (prev_def + deflection_value)*SingularityFunction(x, prev_end, 0)
1393                 prev_slope = slope_value.subs(x, args[i][1].args[1])
1394                 prev_def = deflection_value.subs(x, args[i][1].args[1])
1395             return deflection
1396 
1397         C4 = Symbol('C4')
1398         deflection_curve = integrate(self.slope(), x) + C4
1399 
1400         bc_eqs = []
1401         for position, value in self._boundary_conditions['deflection']:
1402             eqs = deflection_curve.subs(x, position) - value
1403             bc_eqs.append(eqs)
1404 
1405         constants = list(linsolve(bc_eqs, C4))
1406         deflection_curve = deflection_curve.subs({C4: constants[0][0]})
1407         return deflection_curve

1819     def solve_for_ild_reactions(self, value, *reactions):
1820         """
1821 
1822         Determines the Influence Line Diagram equations for reaction
1823         forces under the effect of a moving load.
1824 
1825         Parameters
1826         ==========
1827         value : Integer
1828             Magnitude of moving load
1829         reactions :
1830             The reaction forces applied on the beam.
1831 
1832         Warning
1833         =======
1834         This method creates equations that can give incorrect results when
1835         substituting a = 0 or a = l, with l the length of the beam.
1836 
1837         Examples
1838         ========
1839 
1840         There is a beam of length 10 meters. There are two simple supports
1841         below the beam, one at the starting point and another at the ending
1842         point of the beam. Calculate the I.L.D. equations for reaction forces
1843         under the effect of a moving load of magnitude 1kN.
1844 
1845         Using the sign convention of downwards forces being positive.
1846 
1847         .. plot::
1848             :context: close-figs
1849             :format: doctest
1850             :include-source: True
1851 
1852             >>> from sympy import symbols
1853             >>> from sympy.physics.continuum_mechanics.beam import Beam
1854             >>> E, I = symbols('E, I')
1855             >>> R_0, R_10 = symbols('R_0, R_10')
1856             >>> b = Beam(10, E, I)
1857             >>> p0 = b.apply_support(0, 'pin')
1858             >>> p10 = b.apply_support(10, 'roller')
1859             >>> b.solve_for_ild_reactions(1,R_0,R_10)
1860             >>> b.ild_reactions
1861             {R_0: -SingularityFunction(a, 0, 0) + SingularityFunction(a, 0, 1)/10 - SingularityFunction(a, 10, 1)/10,
1862             R_10: -SingularityFunction(a, 0, 1)/10 + SingularityFunction(a, 10, 0) + SingularityFunction(a, 10, 1)/10}
1863 
1864         """
1865         shear_force, bending_moment = self._solve_for_ild_equations(value)
1866         x = self.variable
1867         l = self.length
1868         a = self.ild_variable
1869 
1870         rotation_jumps = tuple(self._rotation_hinge_symbols)
1871         deflection_jumps = tuple(self._sliding_hinge_symbols)
1872 
1873         C3 = Symbol('C3')
1874         C4 = Symbol('C4')
1875 
1876         shear_curve = limit(shear_force, x, l) - value*(SingularityFunction(a, 0, 0) - SingularityFunction(a, l, 0))
1877         moment_curve = (limit(bending_moment, x, l) - value * (l * SingularityFunction(a, 0, 0)
1878                                                                - SingularityFunction(a, 0, 1)
1879                                                                + SingularityFunction(a, l, 1)))
1880 
1881         shear_force_eqs = []
1882         bending_moment_eqs = []
1883         slope_eqs = []
1884         deflection_eqs = []
1885 
1886         for position, val in self._boundary_conditions['shear_force']:
1887             eqs = self.shear_force().subs(x, position) - val
1888             eqs_without_inf = sum(arg for arg in eqs.args if not any(num.is_infinite for num in arg.args))
1889             shear_sinc = value * (SingularityFunction(- a, - position, 0) - SingularityFunction(-a, 0, 0))
1890             eqs_with_shear_sinc = eqs_without_inf - shear_sinc
1891             shear_force_eqs.append(eqs_with_shear_sinc)
1892 
1893         for position, val in self._boundary_conditions['bending_moment']:
1894             eqs = self.bending_moment().subs(x, position) - val
1895             eqs_without_inf = sum(arg for arg in eqs.args if not any(num.is_infinite for num in arg.args))
1896             moment_sinc = value * (position * SingularityFunction(a, 0, 0)
1897                                    - SingularityFunction(a, 0, 1) + SingularityFunction(a, position, 1))
1898             eqs_with_moment_sinc = eqs_without_inf - moment_sinc
1899             bending_moment_eqs.append(eqs_with_moment_sinc)
1900 
1901         slope_curve = integrate(bending_moment, x) + C3
1902         for position, val in self._boundary_conditions['slope']:
1903             eqs = slope_curve.subs(x, position) - val + value * (SingularityFunction(-a, 0, 1) + position * SingularityFunction(-a, 0, 0))**2 / 2
1904             slope_eqs.append(eqs)
1905 
1906         deflection_curve = integrate(slope_curve, x) + C4
1907         for position, val in self._boundary_conditions['deflection']:
1908             eqs = deflection_curve.subs(x, position) - val + value * (SingularityFunction(-a, 0, 1) + position * SingularityFunction(-a, 0, 0)) ** 3 / 6
1909             deflection_eqs.append(eqs)
1910 
1911         solution = list((linsolve([shear_curve, moment_curve] + shear_force_eqs + bending_moment_eqs + slope_eqs
1912                                   + deflection_eqs, (C3, C4) + reactions + rotation_jumps + deflection_jumps).args)[0])
1913 
1914         reaction_index = 2 + len(reactions)
1915         rotation_index = reaction_index + len(rotation_jumps)
1916         reaction_solution = solution[2:reaction_index]
1917         rotation_solution = solution[reaction_index:rotation_index]
1918         deflection_solution = solution[rotation_index:]
1919 
1920         self._ild_reactions = dict(zip(reactions, reaction_solution))
1921         self._ild_rotations_jumps = dict(zip(rotation_jumps, rotation_solution))
1922         self._ild_deflection_jumps = dict(zip(deflection_jumps, deflection_solution))

2875     def solve_for_reaction_loads(self, *reaction):
2876         """
2877         Solves for the reaction forces.
2878 
2879         Examples
2880         ========
2881         There is a beam of length 30 meters. It it supported by rollers at
2882         of its end. A constant distributed load of magnitude 8 N is applied
2883         from start till its end along y-axis. Another linear load having
2884         slope equal to 9 is applied along z-axis.
2885 
2886         >>> from sympy.physics.continuum_mechanics.beam import Beam3D
2887         >>> from sympy import symbols
2888         >>> l, E, G, I, A, x = symbols('l, E, G, I, A, x')
2889         >>> b = Beam3D(30, E, G, I, A, x)
2890         >>> b.apply_load(8, start=0, order=0, dir="y")
2891         >>> b.apply_load(9*x, start=0, order=0, dir="z")
2892         >>> b.bc_deflection = [(0, [0, 0, 0]), (30, [0, 0, 0])]
2893         >>> R1, R2, R3, R4 = symbols('R1, R2, R3, R4')
2894         >>> b.apply_load(R1, start=0, order=-1, dir="y")
2895         >>> b.apply_load(R2, start=30, order=-1, dir="y")
2896         >>> b.apply_load(R3, start=0, order=-1, dir="z")
2897         >>> b.apply_load(R4, start=30, order=-1, dir="z")
2898         >>> b.solve_for_reaction_loads(R1, R2, R3, R4)
2899         >>> b.reaction_loads
2900         {R1: -120, R2: -120, R3: -1350, R4: -2700}
2901         """
2902         x = self.variable
2903         l = self.length
2904         q = self._load_Singularity
2905         shear_curves = [integrate(load, x) for load in q]
2906         moment_curves = [integrate(shear, x) for shear in shear_curves]
2907         for i in range(3):
2908             react = [r for r in reaction if (shear_curves[i].has(r) or moment_curves[i].has(r))]
2909             if len(react) == 0:
2910                 continue
2911             shear_curve = limit(shear_curves[i], x, l)
2912             moment_curve = limit(moment_curves[i], x, l)
2913             sol = list((linsolve([shear_curve, moment_curve], react).args)[0])
2914             sol_dict = dict(zip(react, sol))
2915             reaction_loads = self._reaction_loads
2916             # Check if any of the evaluated reaction exists in another direction
2917             # and if it exists then it should have same value.
2918             for key in sol_dict:
2919                 if key in reaction_loads and sol_dict[key] != reaction_loads[key]:
2920                     raise ValueError("Ambiguous solution for %s in different directions." % key)
2921             self._reaction_loads.update(sol_dict)

3005     def solve_slope_deflection(self):
3006         x = self.variable
3007         l = self.length
3008         E = self.elastic_modulus
3009         G = self.shear_modulus
3010         I = self.second_moment
3011         if isinstance(I, list):
3012             I_y, I_z = I[0], I[1]
3013         else:
3014             I_y = I_z = I
3015         A = self._area
3016         load = self._load_vector
3017         moment = self._moment_load_vector
3018         defl = Function('defl')
3019         theta = Function('theta')
3020 
3021         # Finding deflection along x-axis(and corresponding slope value by differentiating it)
3022         # Equation used: Derivative(E*A*Derivative(def_x(x), x), x) + load_x = 0
3023         eq = Derivative(E*A*Derivative(defl(x), x), x) + load[0]
3024         def_x = dsolve(Eq(eq, 0), defl(x)).args[1]
3025         # Solving constants originated from dsolve
3026         C1 = Symbol('C1')
3027         C2 = Symbol('C2')
3028         constants = list((linsolve([def_x.subs(x, 0), def_x.subs(x, l)], C1, C2).args)[0])
3029         def_x = def_x.subs({C1:constants[0], C2:constants[1]})
3030         slope_x = def_x.diff(x)
3031         self._deflection[0] = def_x
3032         self._slope[0] = slope_x
3033 
3034         # Finding deflection along y-axis and slope across z-axis. System of equation involved:
3035         # 1: Derivative(E*I_z*Derivative(theta_z(x), x), x) + G*A*(Derivative(defl_y(x), x) - theta_z(x)) + moment_z = 0
3036         # 2: Derivative(G*A*(Derivative(defl_y(x), x) - theta_z(x)), x) + load_y = 0
3037         C_i = Symbol('C_i')
3038         # Substitute value of `G*A*(Derivative(defl_y(x), x) - theta_z(x))` from (2) in (1)
3039         eq1 = Derivative(E*I_z*Derivative(theta(x), x), x) + (integrate(-load[1], x) + C_i) + moment[2]
3040         slope_z = dsolve(Eq(eq1, 0)).args[1]
3041 
3042         # Solve for constants originated from using dsolve on eq1
3043         constants = list((linsolve([slope_z.subs(x, 0), slope_z.subs(x, l)], C1, C2).args)[0])
3044         slope_z = slope_z.subs({C1:constants[0], C2:constants[1]})
3045 
3046         # Put value of slope obtained back in (2) to solve for `C_i` and find deflection across y-axis
3047         eq2 = G*A*(Derivative(defl(x), x)) + load[1]*x - C_i - G*A*slope_z
3048         def_y = dsolve(Eq(eq2, 0), defl(x)).args[1]
3049         # Solve for constants originated from using dsolve on eq2
3050         constants = list((linsolve([def_y.subs(x, 0), def_y.subs(x, l)], C1, C_i).args)[0])
3051         self._deflection[1] = def_y.subs({C1:constants[0], C_i:constants[1]})
3052         self._slope[2] = slope_z.subs(C_i, constants[1])
3053 
3054         # Finding deflection along z-axis and slope across y-axis. System of equation involved:
3055         # 1: Derivative(E*I_y*Derivative(theta_y(x), x), x) - G*A*(Derivative(defl_z(x), x) + theta_y(x)) + moment_y = 0
3056         # 2: Derivative(G*A*(Derivative(defl_z(x), x) + theta_y(x)), x) + load_z = 0
3057 
3058         # Substitute value of `G*A*(Derivative(defl_y(x), x) + theta_z(x))` from (2) in (1)
3059         eq1 = Derivative(E*I_y*Derivative(theta(x), x), x) + (integrate(load[2], x) - C_i) + moment[1]
3060         slope_y = dsolve(Eq(eq1, 0)).args[1]
3061         # Solve for constants originated from using dsolve on eq1
3062         constants = list((linsolve([slope_y.subs(x, 0), slope_y.subs(x, l)], C1, C2).args)[0])
3063         slope_y = slope_y.subs({C1:constants[0], C2:constants[1]})
3064 
3065         # Put value of slope obtained back in (2) to solve for `C_i` and find deflection across z-axis
3066         eq2 = G*A*(Derivative(defl(x), x)) + load[2]*x - C_i + G*A*slope_y
3067         def_z = dsolve(Eq(eq2,0)).args[1]
3068         # Solve for constants originated from using dsolve on eq2
3069         constants = list((linsolve([def_z.subs(x, 0), def_z.subs(x, l)], C1, C_i).args)[0])
3070         self._deflection[2] = def_z.subs({C1:constants[0], C_i:constants[1]})
3071         self._slope[1] = slope_y.subs(C_i, constants[1])

3829         """
