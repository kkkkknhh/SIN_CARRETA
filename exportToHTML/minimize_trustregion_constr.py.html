26   class HessianLinearOperator:
27       """Build LinearOperator from hessp"""
28       def __init__(self, hessp, n):
29           self.hessp = hessp
30           self.n = n
31   
32       def __call__(self, x, *args):
33           def matvec(p):
34               return self.hessp(x, p, *args)
35   
36           return LinearOperator((self.n, self.n), matvec=matvec)

39   class LagrangianHessian:
40       """The Hessian of the Lagrangian as LinearOperator. 
41    
42       The Lagrangian is computed as the objective function plus all the 
43       constraints multiplied with some numbers (Lagrange multipliers). 
44       """
45       def __init__(self, n, objective_hess, constraints_hess):
46           self.n = n
47           self.objective_hess = objective_hess
48           self.constraints_hess = constraints_hess
49   
50       def __call__(self, x, v_eq, v_ineq=None):
51           if v_ineq is None:
52               v_ineq = np.empty(0)
53           H_objective = self.objective_hess(x)
54           H_constraints = self.constraints_hess(x, v_eq, v_ineq)
55   
56           def matvec(p):
57               return H_objective.dot(p) + H_constraints.dot(p)
58   
59           return LinearOperator((self.n, self.n), matvec)

62   def update_state_sqp(state, x, last_iteration_failed, objective, prepared_constraints,
63                        start_time, tr_radius, constr_penalty, cg_info):
64       state.nit += 1
65       state.nfev = objective.nfev
66       state.njev = objective.ngev
67       state.nhev = objective.nhev
68       state.constr_nfev = [c.fun.nfev if isinstance(c.fun, VectorFunction) else 0
69                            for c in prepared_constraints]
70       state.constr_njev = [c.fun.njev if isinstance(c.fun, VectorFunction) else 0
71                            for c in prepared_constraints]
72       state.constr_nhev = [c.fun.nhev if isinstance(c.fun, VectorFunction) else 0
73                            for c in prepared_constraints]
74   
75       if not last_iteration_failed:
76           state.x = x
77           state.fun = objective.f
78           state.grad = objective.g
79           state.v = [c.fun.v for c in prepared_constraints]
80           state.constr = [c.fun.f for c in prepared_constraints]
81           state.jac = [c.fun.J for c in prepared_constraints]
82           # Compute Lagrangian Gradient
83           state.lagrangian_grad = np.copy(state.grad)
84           for c in prepared_constraints:
85               state.lagrangian_grad += c.fun.J.T.dot(c.fun.v)
86           state.optimality = np.linalg.norm(state.lagrangian_grad, np.inf)
87           # Compute maximum constraint violation
88           state.constr_violation = 0
89           for i in range(len(prepared_constraints)):
90               lb, ub = prepared_constraints[i].bounds
91               c = state.constr[i]
92               state.constr_violation = np.max([state.constr_violation,
93                                                np.max(lb - c),
94                                                np.max(c - ub)])
95   
96       state.execution_time = time.time() - start_time
97       state.tr_radius = tr_radius
98       state.constr_penalty = constr_penalty
99       state.cg_niter += cg_info["niter"]
100      state.cg_stop_cond = cg_info["stop_cond"]
101  
102      return state

105  def update_state_ip(state, x, last_iteration_failed, objective,
106                      prepared_constraints, start_time,
107                      tr_radius, constr_penalty, cg_info,
108                      barrier_parameter, barrier_tolerance):
109      state = update_state_sqp(state, x, last_iteration_failed, objective,

111                               constr_penalty, cg_info)
112      state.barrier_parameter = barrier_parameter
113      state.barrier_tolerance = barrier_tolerance
114      return state

117  def _minimize_trustregion_constr(fun, x0, args, grad,
118                                   hess, hessp, bounds, constraints,
119                                   xtol=1e-8, gtol=1e-8,
120                                   barrier_tol=1e-8,
121                                   sparse_jacobian=None,
122                                   callback=None, maxiter=1000,
123                                   verbose=0, finite_diff_rel_step=None,
124                                   initial_constr_penalty=1.0, initial_tr_radius=1.0,
125                                   initial_barrier_parameter=0.1,
126                                   initial_barrier_tolerance=0.1,
127                                   factorization_method=None,
128                                   disp=False):
129      """Minimize a scalar function subject to constraints. 
130   
131      Parameters 
132      ---------- 
133      gtol : float, optional 
134          Tolerance for termination by the norm of the Lagrangian gradient. 
135          The algorithm will terminate when both the infinity norm (i.e., max 
136          abs value) of the Lagrangian gradient and the constraint violation 
137          are smaller than ``gtol``. Default is 1e-8. 
138      xtol : float, optional 
139          Tolerance for termination by the change of the independent variable. 
140          The algorithm will terminate when ``tr_radius < xtol``, where 
141          ``tr_radius`` is the radius of the trust region used in the algorithm. 
142          Default is 1e-8. 
143      barrier_tol : float, optional 
144          Threshold on the barrier parameter for the algorithm termination. 
145          When inequality constraints are present, the algorithm will terminate 
146          only when the barrier parameter is less than `barrier_tol`. 
147          Default is 1e-8. 
148      sparse_jacobian : {bool, None}, optional 
149          Determines how to represent Jacobians of the constraints. If bool, 
150          then Jacobians of all the constraints will be converted to the 
151          corresponding format. If None (default), then Jacobians won't be 
152          converted, but the algorithm can proceed only if they all have the 
153          same format. 
154      initial_tr_radius: float, optional 
155          Initial trust radius. The trust radius gives the maximum distance 
156          between solution points in consecutive iterations. It reflects the 
157          trust the algorithm puts in the local approximation of the optimization 
158          problem. For an accurate local approximation the trust-region should be 
159          large and for an  approximation valid only close to the current point it 
160          should be a small one. The trust radius is automatically updated throughout 
161          the optimization process, with ``initial_tr_radius`` being its initial value. 
162          Default is 1 (recommended in [1]_, p. 19). 
163      initial_constr_penalty : float, optional 
164          Initial constraints penalty parameter. The penalty parameter is used for 
165          balancing the requirements of decreasing the objective function 
166          and satisfying the constraints. It is used for defining the merit function: 
167          ``merit_function(x) = fun(x) + constr_penalty * constr_norm_l2(x)``, 
168          where ``constr_norm_l2(x)`` is the l2 norm of a vector containing all 
169          the constraints. The merit function is used for accepting or rejecting 
170          trial points and ``constr_penalty`` weights the two conflicting goals 
171          of reducing objective function and constraints. The penalty is automatically 
172          updated throughout the optimization  process, with 
173          ``initial_constr_penalty`` being its  initial value. Default is 1 
174          (recommended in [1]_, p 19). 
175      initial_barrier_parameter, initial_barrier_tolerance: float, optional 
176          Initial barrier parameter and initial tolerance for the barrier subproblem. 
177          Both are used only when inequality constraints are present. For dealing with 
178          optimization problems ``min_x f(x)`` subject to inequality constraints 
179          ``c(x) <= 0`` the algorithm introduces slack variables, solving the problem 
180          ``min_(x,s) f(x) + barrier_parameter*sum(ln(s))`` subject to the equality 
181          constraints  ``c(x) + s = 0`` instead of the original problem. This subproblem 
182          is solved for decreasing values of ``barrier_parameter`` and with decreasing 
183          tolerances for the termination, starting with ``initial_barrier_parameter`` 
184          for the barrier parameter and ``initial_barrier_tolerance`` for the 
185          barrier tolerance. Default is 0.1 for both values (recommended in [1]_ p. 19). 
186          Also note that ``barrier_parameter`` and ``barrier_tolerance`` are updated 
187          with the same prefactor. 
188      factorization_method : string or None, optional 
189          Method to factorize the Jacobian of the constraints. Use None (default) 
190          for the auto selection or one of: 
191   
192          - 'NormalEquation' (requires scikit-sparse) 
193          - 'AugmentedSystem' 
194          - 'QRFactorization' 
195          - 'SVDFactorization' 
196   
197          The methods 'NormalEquation' and 'AugmentedSystem' can be used only 
198          with sparse constraints. The projections required by the algorithm 
199          will be computed using, respectively, the normal equation  and the 
200          augmented system approaches explained in [1]_. 'NormalEquation' 
201          computes the Cholesky factorization of ``A A.T`` and 'AugmentedSystem' 
202          performs the LU factorization of an augmented system. They usually 
203          provide similar results. 'AugmentedSystem' is used by default for 
204          sparse matrices. 
205   
206          The methods 'QRFactorization' and 'SVDFactorization' can be used 
207          only with dense constraints. They compute the required projections 
208          using, respectively, QR and SVD factorizations. The 'SVDFactorization' 
209          method can cope with Jacobian matrices with deficient row rank and will 
210          be used whenever other factorization methods fail (which may imply the 
211          conversion of sparse matrices to a dense format when required). 
212          By default, 'QRFactorization' is used for dense matrices. 
213      finite_diff_rel_step : None or array_like, optional 
214          Relative step size for the finite difference approximation. 
215      maxiter : int, optional 
216          Maximum number of algorithm iterations. Default is 1000. 
217      verbose : {0, 1, 2, 3}, optional 
218          Level of algorithm's verbosity: 
219   
220          * 0 (default) : work silently. 
221          * 1 : display a termination report. 
222          * 2 : display progress during iterations. 
223          * 3 : display progress during iterations (more complete report). 
224   
225      disp : bool, optional 
226          If True (default), then `verbose` will be set to 1 if it was 0. 
227   
228      Returns 
229      ------- 
230      `OptimizeResult` with the fields documented below. Note the following: 
231   
232          1. All values corresponding to the constraints are ordered as they 
233             were passed to the solver. And values corresponding to `bounds` 
234             constraints are put *after* other constraints. 
235          2. All numbers of function, Jacobian or Hessian evaluations correspond 
236             to numbers of actual Python function calls. It means, for example, 
237             that if a Jacobian is estimated by finite differences, then the 
238             number of Jacobian evaluations will be zero and the number of 
239             function evaluations will be incremented by all calls during the 
240             finite difference estimation. 
241   
242      x : ndarray, shape (n,) 
243          Solution found. 
244      optimality : float 
245          Infinity norm of the Lagrangian gradient at the solution. 
246      constr_violation : float 
247          Maximum constraint violation at the solution. 
248      fun : float 
249          Objective function at the solution. 
250      grad : ndarray, shape (n,) 
251          Gradient of the objective function at the solution. 
252      lagrangian_grad : ndarray, shape (n,) 
253          Gradient of the Lagrangian function at the solution. 
254      nit : int 
255          Total number of iterations. 
256      nfev : integer 
257          Number of the objective function evaluations. 
258      njev : integer 
259          Number of the objective function gradient evaluations. 
260      nhev : integer 
261          Number of the objective function Hessian evaluations. 
262      cg_niter : int 
263          Total number of the conjugate gradient method iterations. 
264      method : {'equality_constrained_sqp', 'tr_interior_point'} 
265          Optimization method used. 
266      constr : list of ndarray 
267          List of constraint values at the solution. 
268      jac : list of {ndarray, sparse matrix} 
269          List of the Jacobian matrices of the constraints at the solution. 
270      v : list of ndarray 
271          List of the Lagrange multipliers for the constraints at the solution. 
272          For an inequality constraint a positive multiplier means that the upper 
273          bound is active, a negative multiplier means that the lower bound is 
274          active and if a multiplier is zero it means the constraint is not 
275          active. 
276      constr_nfev : list of int 
277          Number of constraint evaluations for each of the constraints. 
278      constr_njev : list of int 
279          Number of Jacobian matrix evaluations for each of the constraints. 
280      constr_nhev : list of int 
281          Number of Hessian evaluations for each of the constraints. 
282      tr_radius : float 
283          Radius of the trust region at the last iteration. 
284      constr_penalty : float 
285          Penalty parameter at the last iteration, see `initial_constr_penalty`. 
286      barrier_tolerance : float 
287          Tolerance for the barrier subproblem at the last iteration. 
288          Only for problems with inequality constraints. 
289      barrier_parameter : float 
290          Barrier parameter at the last iteration. Only for problems 
291          with inequality constraints. 
292      execution_time : float 
293          Total execution time. 
294      message : str 
295          Termination message. 
296      status : {0, 1, 2, 3, 4} 
297          Termination status: 
298   
299          * 0 : The maximum number of function evaluations is exceeded. 
300          * 1 : `gtol` termination condition is satisfied. 
301          * 2 : `xtol` termination condition is satisfied. 
302          * 3 : `callback` function requested termination. 
303          * 4 : Constraint violation exceeds 'gtol'. 
304   
305          .. versionchanged:: 1.15.0 
306              If the constraint violation exceeds `gtol`, then ``result.success`` 
307              will now be False. 
308   
309      cg_stop_cond : int 
310          Reason for CG subproblem termination at the last iteration: 
311   
312          * 0 : CG subproblem not evaluated. 
313          * 1 : Iteration limit was reached. 
314          * 2 : Reached the trust-region boundary. 
315          * 3 : Negative curvature detected. 
316          * 4 : Tolerance was satisfied. 
317   
318      References 
319      ---------- 
320      .. [1] Conn, A. R., Gould, N. I., & Toint, P. L. 
321             Trust region methods. 2000. Siam. pp. 19. 
322      """
323      x0 = np.atleast_1d(x0).astype(float)
324      n_vars = np.size(x0)
325      if hess is None:
326          if callable(hessp):
327              hess = HessianLinearOperator(hessp, n_vars)
328          else:
329              hess = BFGS()
330      if disp and verbose == 0:
331          verbose = 1
332  
333      if bounds is not None:
334          modified_lb = np.nextafter(bounds.lb, -np.inf, where=bounds.lb > -np.inf)
335          modified_ub = np.nextafter(bounds.ub, np.inf, where=bounds.ub < np.inf)
336          modified_lb = np.where(np.isfinite(bounds.lb), modified_lb, bounds.lb)
337          modified_ub = np.where(np.isfinite(bounds.ub), modified_ub, bounds.ub)
338          bounds = Bounds(modified_lb, modified_ub, keep_feasible=bounds.keep_feasible)
339          finite_diff_bounds = strict_bounds(bounds.lb, bounds.ub,
340                                             bounds.keep_feasible, n_vars)
341      else:
342          finite_diff_bounds = (-np.inf, np.inf)
343  
344      # Define Objective Function
345      objective = ScalarFunction(fun, x0, args, grad, hess,
346                                 finite_diff_rel_step, finite_diff_bounds)
347  
348      # Put constraints in list format when needed.
349      if isinstance(constraints, (NonlinearConstraint | LinearConstraint)):
350          constraints = [constraints]
351  
352      # Prepare constraints.
353      prepared_constraints = [
354          PreparedConstraint(c, x0, sparse_jacobian, finite_diff_bounds)
355          for c in constraints]
356  
357      # Check that all constraints are either sparse or dense.
358      n_sparse = sum(c.fun.sparse_jacobian for c in prepared_constraints)
359      if 0 < n_sparse < len(prepared_constraints):
360          raise ValueError("All constraints must have the same kind of the "
361                           "Jacobian --- either all sparse or all dense. "
362                           "You can set the sparsity globally by setting "
363                           "`sparse_jacobian` to either True of False.")
364      if prepared_constraints:
365          sparse_jacobian = n_sparse > 0
366  
367      if bounds is not None:
368          if sparse_jacobian is None:
369              sparse_jacobian = True
370          prepared_constraints.append(PreparedConstraint(bounds, x0,
371                                                         sparse_jacobian))
372  
373      # Concatenate initial constraints to the canonical form.
374      c_eq0, c_ineq0, J_eq0, J_ineq0 = initial_constraints_as_canonical(
375          n_vars, prepared_constraints, sparse_jacobian)
376  
377      # Prepare all canonical constraints and concatenate it into one.
378      canonical_all = [CanonicalConstraint.from_PreparedConstraint(c)
379                       for c in prepared_constraints]
380  
381      if len(canonical_all) == 0:
382          canonical = CanonicalConstraint.empty(n_vars)
383      elif len(canonical_all) == 1:
384          canonical = canonical_all[0]
385      else:
386          canonical = CanonicalConstraint.concatenate(canonical_all,
387                                                      sparse_jacobian)
388  
389      # Generate the Hessian of the Lagrangian.
390      lagrangian_hess = LagrangianHessian(n_vars, objective.hess, canonical.hess)
391  
392      # Choose appropriate method
393      if canonical.n_ineq == 0:
394          method = 'equality_constrained_sqp'
395      else:
396          method = 'tr_interior_point'
397  
398      # Construct OptimizeResult
399      state = OptimizeResult(
400          nit=0, nfev=0, njev=0, nhev=0,
401          cg_niter=0, cg_stop_cond=0,
402          fun=objective.f, grad=objective.g,
403          lagrangian_grad=np.copy(objective.g),
404          constr=[c.fun.f for c in prepared_constraints],
405          jac=[c.fun.J for c in prepared_constraints],
406          constr_nfev=[0 for c in prepared_constraints],
407          constr_njev=[0 for c in prepared_constraints],
408          constr_nhev=[0 for c in prepared_constraints],
409          v=[c.fun.v for c in prepared_constraints],
410          method=method)
411  
412      # Start counting
413      start_time = time.time()
414  
415      # Define stop criteria
416      if method == 'equality_constrained_sqp':
417          def stop_criteria(state, x, last_iteration_failed,
418                            optimality, constr_violation,
419                            tr_radius, constr_penalty, cg_info):
420              state = update_state_sqp(state, x, last_iteration_failed,
421                                       objective, prepared_constraints,
422                                       start_time, tr_radius, constr_penalty,
423                                       cg_info)
424              if verbose == 2:
425                  BasicReport.print_iteration(state.nit,
426                                              state.nfev,
427                                              state.cg_niter,
428                                              state.fun,
429                                              state.tr_radius,
430                                              state.optimality,
431                                              state.constr_violation)
432              elif verbose > 2:
433                  SQPReport.print_iteration(state.nit,
434                                            state.nfev,
435                                            state.cg_niter,
436                                            state.fun,
437                                            state.tr_radius,
438                                            state.optimality,
439                                            state.constr_violation,
440                                            state.constr_penalty,
441                                            state.cg_stop_cond)
442              state.status = None
443              state.niter = state.nit  # Alias for callback (backward-compatibility)
444              if callback is not None:
445                  callback_stop = False
446                  try:
447                      callback_stop = callback(state)
448                  except StopIteration:
449                      callback_stop = True
450                  if callback_stop:
451                      state.status = 3
452                      return True
453              if state.optimality < gtol and state.constr_violation < gtol:
454                  state.status = 1
455              elif state.tr_radius < xtol:
456                  state.status = 2
457              elif state.nit >= maxiter:
458                  state.status = 0
459              return state.status in (0, 1, 2, 3)
460      elif method == 'tr_interior_point':
461          def stop_criteria(state, x, last_iteration_failed, tr_radius,
462                            constr_penalty, cg_info, barrier_parameter,
463                            barrier_tolerance):
464              state = update_state_ip(state, x, last_iteration_failed,
465                                      objective, prepared_constraints,
466                                      start_time, tr_radius, constr_penalty,
467                                      cg_info, barrier_parameter, barrier_tolerance)
468              if verbose == 2:
469                  BasicReport.print_iteration(state.nit,
470                                              state.nfev,
471                                              state.cg_niter,
472                                              state.fun,
473                                              state.tr_radius,
474                                              state.optimality,
475                                              state.constr_violation)
476              elif verbose > 2:
477                  IPReport.print_iteration(state.nit,
478                                           state.nfev,
479                                           state.cg_niter,
480                                           state.fun,
481                                           state.tr_radius,
482                                           state.optimality,
483                                           state.constr_violation,
484                                           state.constr_penalty,
485                                           state.barrier_parameter,
486                                           state.cg_stop_cond)
487              state.status = None
488              state.niter = state.nit  # Alias for callback (backward compatibility)
489              if callback is not None:
490                  callback_stop = False
491                  try:
492                      callback_stop = callback(state)
493                  except StopIteration:
494                      callback_stop = True
495                  if callback_stop:
496                      state.status = 3
497                      return True
498              if state.optimality < gtol and state.constr_violation < gtol:
499                  state.status = 1
500              elif (state.tr_radius < xtol
501                    and state.barrier_parameter < barrier_tol):
502                  state.status = 2
503              elif state.nit >= maxiter:
504                  state.status = 0
505              return state.status in (0, 1, 2, 3)
506  
507      if verbose == 2:
508          BasicReport.print_header()
509      elif verbose > 2:
510          if method == 'equality_constrained_sqp':
511              SQPReport.print_header()
512          elif method == 'tr_interior_point':
513              IPReport.print_header()
514  
515      # Call inferior function to do the optimization
516      if method == 'equality_constrained_sqp':
517          def fun_and_constr(x):
518              f = objective.fun(x)
519              c_eq, _ = canonical.fun(x)
520              return f, c_eq
521  
522          def grad_and_jac(x):
523              g = objective.grad(x)
524              J_eq, _ = canonical.jac(x)
525              return g, J_eq
526  
527          _, result = equality_constrained_sqp(
528              fun_and_constr, grad_and_jac, lagrangian_hess,
529              x0, objective.f, objective.g,
530              c_eq0, J_eq0,
531              stop_criteria, state,
532              initial_constr_penalty, initial_tr_radius,
533              factorization_method)
534  
535      elif method == 'tr_interior_point':
536          _, result = tr_interior_point(
537              objective.fun, objective.grad, lagrangian_hess,
538              n_vars, canonical.n_ineq, canonical.n_eq,
539              canonical.fun, canonical.jac,
540              x0, objective.f, objective.g,
541              c_ineq0, J_ineq0, c_eq0, J_eq0,
542              stop_criteria,
543              canonical.keep_feasible,
544              xtol, state, initial_barrier_parameter,
545              initial_barrier_tolerance,
546              initial_constr_penalty, initial_tr_radius,
547              factorization_method, finite_diff_bounds)
548  
549      # Status 4 occurs when minimize is successful but constraints are not satisfied.
550      if result.status in (1, 2) and state.constr_violation > gtol:
551          result.status = 4
552  
553      # Status 3 occurs when the callback function requests termination,
554      # this is assumed to not be a success.
555      result.success = True if result.status in (1, 2) else False
556      result.message = TERMINATION_MESSAGES[result.status]
557  
558      # Alias (for backward compatibility with 1.1.0)
559      result.niter = result.nit
560  
561      if verbose == 2:
562          BasicReport.print_footer()
563      elif verbose > 2:
564          if method == 'equality_constrained_sqp':
565              SQPReport.print_footer()
566          elif method == 'tr_interior_point':
567              IPReport.print_footer()
568      if verbose >= 1:
569          print(result.message)
570          print(f"Number of iterations: {result.nit}, "
571                f"function evaluations: {result.nfev}, "
572                f"CG iterations: {result.cg_niter}, "
573                f"optimality: {result.optimality:.2e}, "
574                f"constraint violation: {result.constr_violation:.2e}, "
575                f"execution time: {result.execution_time:4.2} s.")
576      return result
