3    """ 
4    This module implements TorchDynamo's backend registry system for managing compiler backends. 
5     
6    The registry provides a centralized way to register, discover and manage different compiler 
7    backends that can be used with torch.compile(). It handles: 
8     
9    - Backend registration and discovery through decorators and entry points 
10   - Lazy loading of backend implementations 
11   - Lookup and validation of backend names 
12   - Categorization of backends using tags (debug, experimental, etc.) 
13    
14   Key components: 
15   - CompilerFn: Type for backend compiler functions that transform FX graphs 
16   - _BACKENDS: Registry mapping backend names to entry points 
17   - _COMPILER_FNS: Registry mapping backend names to loaded compiler functions 
18    
19   Example usage: 
20       @register_backend 
21       def my_compiler(fx_graph, example_inputs): 
22           # Transform FX graph into optimized implementation 
23           return compiled_fn 
24    
25       # Use registered backend 
26       torch.compile(model, backend="my_compiler") 
27    
28   The registry also supports discovering backends through setuptools entry points 
29   in the "torch_dynamo_backends" group. Example: 
30   ``` 
31   setup.py 
32   --- 
33   from setuptools import setup 
34    
35   setup( 
36       name='my_torch_backend', 
37       version='0.1', 
38       packages=['my_torch_backend'], 
39       entry_points={ 
40           'torch_dynamo_backends': [ 
41               # name = path to entry point of backend implementation 
42               'my_compiler = my_torch_backend.compiler:my_compiler_function', 
43           ], 
44       }, 
45   ) 
46   ``` 
47   ``` 
48   my_torch_backend/compiler.py 
49   --- 
50   def my_compiler_function(fx_graph, example_inputs): 
51       # Transform FX graph into optimized implementation 
52       return compiled_fn 
53   ``` 
54   Using `my_compiler` backend: 
55   ``` 
56   import torch 
57    
58   model = ...  # Your PyTorch model 
59   optimized_model = torch.compile(model, backend="my_compiler") 
60   ``` 
61   """
62   
63   import functools
64   import logging
65   import sys
66   from collections.abc import Sequence
67   from importlib.metadata import EntryPoint
68   from typing import Callable, Optional, Protocol
69   
70   import torch
71   from torch import fx
72   
73   
74   log = logging.getLogger(__name__)
75   
76   
77   class CompiledFn(Protocol):
78       def __call__(self, *args: torch.Tensor) -> tuple[torch.Tensor, ...]: ...
79   
80   
81   CompilerFn = Callable[[fx.GraphModule, list[torch.Tensor]], CompiledFn]
82   
83   _BACKENDS: dict[str, Optional[EntryPoint]] = {}
84   _COMPILER_FNS: dict[str, CompilerFn] = {}
85   
86   
87   def register_backend(
88       compiler_fn: Optional[CompilerFn] = None,
89       name: Optional[str] = None,
90       tags: Sequence[str] = (),
91   ):
92       """ 
93       Decorator to add a given compiler to the registry to allow calling 
94       `torch.compile` with string shorthand.  Note: for projects not 
95       imported by default, it might be easier to pass a function directly 
96       as a backend and not use a string. 
97    
98       Args: 
99           compiler_fn: Callable taking a FX graph and fake tensor inputs 
100          name: Optional name, defaults to `compiler_fn.__name__` 
101          tags: Optional set of string tags to categorize backend with 
102      """
103      if compiler_fn is None:
104          # @register_backend(name="") syntax
105          return functools.partial(register_backend, name=name, tags=tags)
106      assert callable(compiler_fn)
107      name = name or compiler_fn.__name__
108      assert name not in _COMPILER_FNS, f"duplicate name: {name}"
109      if compiler_fn not in _BACKENDS:
110          _BACKENDS[name] = None
111      _COMPILER_FNS[name] = compiler_fn
112      compiler_fn._tags = tuple(tags)
113      return compiler_fn
114  
115  
116  register_debug_backend = functools.partial(register_backend, tags=("debug",))
117  register_experimental_backend = functools.partial(
118      register_backend, tags=("experimental",)
119  )
120  
121  
122  def lookup_backend(compiler_fn):
123      """Expand backend strings to functions"""
124      if isinstance(compiler_fn, str):
125          if compiler_fn not in _BACKENDS:
126              _lazy_import()
127          if compiler_fn not in _BACKENDS:
128              from ..exc import InvalidBackend
129  
130              raise InvalidBackend(name=compiler_fn)
131  
132          if compiler_fn not in _COMPILER_FNS:
133              entry_point = _BACKENDS[compiler_fn]
134              register_backend(compiler_fn=entry_point.load(), name=compiler_fn)
135          compiler_fn = _COMPILER_FNS[compiler_fn]
136      return compiler_fn
137  
138  
139  def list_backends(exclude_tags=("debug", "experimental")) -> list[str]:
140      """ 
141      Return valid strings that can be passed to: 
142   
143          torch.compile(..., backend="name") 
144      """
145      _lazy_import()
146      exclude_tags = set(exclude_tags or ())
147  
148      backends = [
149          name
150          for name in _BACKENDS.keys()
151          if name not in _COMPILER_FNS
152          or not exclude_tags.intersection(_COMPILER_FNS[name]._tags)
153      ]
154      return sorted(backends)

157  @functools.cache
158  def _lazy_import():
159      from .. import backends
160      from ..utils import import_submodule
161  
162      import_submodule(backends)
163  
164      from ..repro.after_dynamo import dynamo_minifier_backend
165  
166      assert dynamo_minifier_backend is not None
167  
168      _discover_entrypoint_backends()

171  @functools.cache
172  def _discover_entrypoint_backends():
173      # importing here so it will pick up the mocked version in test_backends.py
174      from importlib.metadata import entry_points
175  
176      group_name = "torch_dynamo_backends"
177      if sys.version_info < (3, 10):
178          eps = entry_points()
179          eps = eps[group_name] if group_name in eps else []
180          eps = {ep.name: ep for ep in eps}
181      else:
182          eps = entry_points(group=group_name)
183          eps = {name: eps[name] for name in eps.names}
184      for backend_name in eps:
185          _BACKENDS[backend_name] = eps[backend_name]
