156      eq = eqn(npos, dneg, pow)

620  def test_power_dispatcher():
621  
622      class NewBase(Expr):
623          pass
624      class NewPow(NewBase, Pow):
625          pass
626      a, b = Symbol('a'), NewBase()
627  
628      @power.register(Expr, NewBase)
629      @power.register(NewBase, Expr)
630      @power.register(NewBase, NewBase)
631      def _(a, b):
632          return NewPow(a, b)
633  
634      # Pow called as fallback
635      assert power(2, 3) == 8*S.One
636      assert power(a, 2) == Pow(a, 2)
637      assert power(a, a) == Pow(a, a)

639      # NewPow called by dispatch
640      assert power(a, b) == NewPow(a, b)
641      assert power(b, a) == NewPow(b, a)
642      assert power(b, b) == NewPow(b, b)
