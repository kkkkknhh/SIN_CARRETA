34   def test_GaussianUnitaryEnsemble():
35       H = RandomMatrixSymbol('H', 3, 3)
36       G = GUE('U', 3)
37       assert density(G)(H) == sqrt(2)*exp(-3*Trace(H**2)/2)/(4*pi**Rational(9, 2))
38       i, j = (Dummy('i', integer=True, positive=True),
39               Dummy('j', integer=True, positive=True))
40       l = IndexedBase('l')
41       assert joint_eigen_distribution(G).dummy_eq(
42               Lambda((l[1], l[2], l[3]),
43               27*sqrt(6)*exp(-3*(l[1]**2)/2 - 3*(l[2]**2)/2 - 3*(l[3]**2)/2)*
44               Product(Abs(l[i] - l[j])**2, (j, i + 1, 3), (i, 1, 2))/(16*pi**Rational(3, 2))))
45       s = Dummy('s')
46       assert level_spacing_distribution(G).dummy_eq(Lambda(s, 32*s**2*exp(-4*s**2/pi)/pi**2))

49   def test_GaussianOrthogonalEnsemble():
50       H = RandomMatrixSymbol('H', 3, 3)
51       _H = MatrixSymbol('_H', 3, 3)
52       G = GOE('O', 3)
53       assert density(G)(H) == exp(-3*Trace(H**2)/4)/Integral(exp(-3*Trace(_H**2)/4), _H)
54       i, j = (Dummy('i', integer=True, positive=True),
55               Dummy('j', integer=True, positive=True))
56       l = IndexedBase('l')
57       assert joint_eigen_distribution(G).dummy_eq(
58               Lambda((l[1], l[2], l[3]),
59               9*sqrt(2)*exp(-3*l[1]**2/2 - 3*l[2]**2/2 - 3*l[3]**2/2)*
60               Product(Abs(l[i] - l[j]), (j, i + 1, 3), (i, 1, 2))/(32*pi)))
61       s = Dummy('s')
62       assert level_spacing_distribution(G).dummy_eq(Lambda(s, s*pi*exp(-s**2*pi/4)/2))

112  def test_JointEigenDistribution():
113      A = Matrix([[Normal('A00', 0, 1), Normal('A01', 1, 1)],
114                  [Beta('A10', 1, 1), Beta('A11', 1, 1)]])
115      assert JointEigenDistribution(A) == \
116      JointDistributionHandmade(-sqrt(A[0, 0]**2 - 2*A[0, 0]*A[1, 1] + 4*A[0, 1]*A[1, 0] + A[1, 1]**2)/2 +
117      A[0, 0]/2 + A[1, 1]/2, sqrt(A[0, 0]**2 - 2*A[0, 0]*A[1, 1] + 4*A[0, 1]*A[1, 0] + A[1, 1]**2)/2 + A[0, 0]/2 + A[1, 1]/2)
118      raises(ValueError, lambda: JointEigenDistribution(Matrix([[1, 0], [2, 1]])))

120  def test_issue_19841():
121      G1 = GUE('U', 2)
122      G2 = G1.xreplace({2: 2})
123      assert G1.args == G2.args
124  
125      X = MatrixSymbol('X', 2, 2)
126      G = GSE('U', 2)
127      h_pspace = RandomMatrixPSpace('P', model=density(G))
128      H = RandomMatrixSymbol('H', 2, 2, pspace=h_pspace)
129      H2 = RandomMatrixSymbol('H', 2, 2, pspace=None)
130      assert H.doit() == H
131  
132      assert (2*H).xreplace({H: X}) == 2*X
133      assert (2*H).xreplace({H2: X}) == 2*H
